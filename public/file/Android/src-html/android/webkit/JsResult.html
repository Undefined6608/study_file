017    package android.webkit;
018    
019    
020    public class JsResult {
021        // This prevents a user from interacting with the result before WebCore is
022        // ready to handle it.
023        private boolean mReady;
024        // Tells us if the user tried to confirm or cancel the result before WebCore
025        // is ready.
026        private boolean mTriedToNotifyBeforeReady;
027        // This is a basic result of a confirm or prompt dialog.
028        protected boolean mResult;
029        // This is the caller of the prompt and is the object that is waiting.
030        protected final CallbackProxy mProxy;
031        // This is the default value of the result.
032        private final boolean mDefaultValue;
033    
034        /**
035         * Handle the result if the user cancelled the dialog.
036         */
037        public final void cancel() {
038            mResult = false;
039            wakeUp();
040        }
041    
042        /**
043         * Handle a confirmation response from the user.
044         */
045        public final void confirm() {
046            mResult = true;
047            wakeUp();
048        }
049    
050        /*package*/ JsResult(CallbackProxy proxy, boolean defaultVal) {
051            mProxy = proxy;
052            mDefaultValue = defaultVal;
053        }
054    
055        /*package*/ final boolean getResult() {
056            return mResult;
057        }
058    
059        /*package*/ final void setReady() {
060            mReady = true;
061            if (mTriedToNotifyBeforeReady) {
062                wakeUp();
063            }
064        }
065    
066        /*package*/ void handleDefault() {
067            setReady();
068            mResult = mDefaultValue;
069            wakeUp();
070        }
071    
072        /* Wake up the WebCore thread. */
073        protected final void wakeUp() {
074            if (mReady) {
075                synchronized (mProxy) {
076                    mProxy.notify();
077                }
078            } else {
079                mTriedToNotifyBeforeReady = true;
080            }
081        }
082    }
