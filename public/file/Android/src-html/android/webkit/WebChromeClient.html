017    package android.webkit;
018    
019    import android.graphics.Bitmap;
020    import android.os.Message;
021    
022    public class WebChromeClient {
023    
024        /**
025         * Tell the host application the current progress of loading a page.
026         * @param view The WebView that initiated the callback.
027         * @param newProgress Current page loading progress, represented by
028         *                    an integer between 0 and 100.
029         */
030        public void onProgressChanged(WebView view, int newProgress) {}
031    
032        /**
033         * Notify the host application of a change in the document title.
034         * @param view The WebView that initiated the callback.
035         * @param title A String containing the new title of the document.
036         */
037        public void onReceivedTitle(WebView view, String title) {}
038    
039        /**
040         * Notify the host application of a new favicon for the current page.
041         * @param view The WebView that initiated the callback.
042         * @param icon A Bitmap containing the favicon for the current page.
043         */
044        public void onReceivedIcon(WebView view, Bitmap icon) {}
045    
046        /**
047         * Request the host application to create a new Webview. The host
048         * application should handle placement of the new WebView in the view
049         * system. The default behavior returns null.
050         * @param view The WebView that initiated the callback.
051         * @param dialog True if the new window is meant to be a small dialog
052         *               window.
053         * @param userGesture True if the request was initiated by a user gesture
054         *                    such as clicking a link.
055         * @param resultMsg The message to send when done creating a new WebView.
056         *                  Set the new WebView through resultMsg.obj which is 
057         *                  WebView.WebViewTransport() and then call
058         *                  resultMsg.sendToTarget();
059         * @return Similar to javscript dialogs, this method should return true if
060         *         the client is going to handle creating a new WebView. Note that
061         *         the WebView will halt processing if this method returns true so
062         *         make sure to call resultMsg.sendToTarget(). It is undefined
063         *         behavior to call resultMsg.sendToTarget() after returning false
064         *         from this method.
065         */
066        public boolean onCreateWindow(WebView view, boolean dialog,
067                boolean userGesture, Message resultMsg) {
068            return false;
069        }
070    
071        /**
072         * Request display and focus for this WebView. This may happen due to
073         * another WebView opening a link in this WebView and requesting that this
074         * WebView be displayed.
075         * @param view The WebView that needs to be focused.
076         */
077        public void onRequestFocus(WebView view) {}
078    
079        /**
080         * Notify the host application to close the given WebView and remove it
081         * from the view system if necessary. At this point, WebCore has stopped
082         * any loading in this window and has removed any cross-scripting ability
083         * in javascript.
084         * @param window The WebView that needs to be closed.
085         */
086        public void onCloseWindow(WebView window) {}
087    
088        /**
089         * Tell the client to display a javascript alert dialog.  If the client
090         * returns true, WebView will assume that the client will handle the
091         * dialog.  If the client returns false, it will continue execution.
092         * @param view The WebView that initiated the callback.
093         * @param url The url of the page requesting the dialog.
094         * @param message Message to be displayed in the window.
095         * @param result A JsResult to confirm that the user hit enter.
096         * @return boolean Whether the client will handle the alert dialog.
097         */
098        public boolean onJsAlert(WebView view, String url, String message,
099                JsResult result) {
100            return false;
101        }
102    
103        /**
104         * Tell the client to display a confirm dialog to the user. If the client
105         * returns true, WebView will assume that the client will handle the
106         * confirm dialog and call the appropriate JsResult method. If the
107         * client returns false, a default value of false will be returned to
108         * javascript. The default behavior is to return false.
109         * @param view The WebView that initiated the callback.
110         * @param url The url of the page requesting the dialog.
111         * @param message Message to be displayed in the window.
112         * @param result A JsResult used to send the user's response to
113         *               javascript.
114         * @return boolean Whether the client will handle the confirm dialog.
115         */
116        public boolean onJsConfirm(WebView view, String url, String message,
117                JsResult result) {
118            return false;
119        }
120    
121        /**
122         * Tell the client to display a prompt dialog to the user. If the client
123         * returns true, WebView will assume that the client will handle the
124         * prompt dialog and call the appropriate JsPromptResult method. If the
125         * client returns false, a default value of false will be returned to to
126         * javascript. The default behavior is to return false.
127         * @param view The WebView that initiated the callback.
128         * @param url The url of the page requesting the dialog.
129         * @param message Message to be displayed in the window.
130         * @param defaultValue The default value displayed in the prompt dialog.
131         * @param result A JsPromptResult used to send the user's reponse to
132         *               javascript.
133         * @return boolean Whether the client will handle the prompt dialog.
134         */
135        public boolean onJsPrompt(WebView view, String url, String message,
136                String defaultValue, JsPromptResult result) {
137            return false;
138        }
139    
140        /**
141         * Tell the client to display a dialog to confirm navigation away from the
142         * current page. This is the result of the onbeforeunload javascript event.
143         * If the client returns true, WebView will assume that the client will
144         * handle the confirm dialog and call the appropriate JsResult method. If
145         * the client returns false, a default value of true will be returned to
146         * javascript to accept navigation away from the current page. The default
147         * behavior is to return false. Setting the JsResult to true will navigate
148         * away from the current page, false will cancel the navigation.
149         * @param view The WebView that initiated the callback.
150         * @param url The url of the page requesting the dialog.
151         * @param message Message to be displayed in the window.
152         * @param result A JsResult used to send the user's response to
153         *               javascript.
154         * @return boolean Whether the client will handle the confirm dialog.
155         */
156        public boolean onJsBeforeUnload(WebView view, String url, String message,
157                JsResult result) {
158            return false;
159        }
160    }
