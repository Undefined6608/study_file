017    package android.webkit;
018    
019    import android.webkit.CacheManager.CacheResult;
020    import android.webkit.PluginData;
021    import android.webkit.UrlInterceptHandler;
022    
023    import java.util.Iterator;
024    import java.util.LinkedList;
025    import java.util.Map;
026    
027    public final class UrlInterceptRegistry {
028    
029        private final static String LOGTAG = "intercept";
030    
031        private static boolean mDisabled = false;
032    
033        private static LinkedList mHandlerList;
034    
035        private static synchronized LinkedList getHandlers() {
036            if(mHandlerList == null)
037                mHandlerList = new LinkedList<UrlInterceptHandler>();
038            return mHandlerList;
039        }
040    
041        /**
042         * set the flag to control whether url intercept is enabled or disabled
043         * 
044         * @param disabled true to disable the cache
045         */
046        public static synchronized void setUrlInterceptDisabled(boolean disabled) {
047            mDisabled = disabled;
048        }
049    
050        /**
051         * get the state of the url intercept, enabled or disabled
052         * 
053         * @return return if it is disabled
054         */
055        public static synchronized boolean urlInterceptDisabled() {
056            return mDisabled;
057        }
058    
059        /**
060         * Register a new UrlInterceptHandler. This handler will be called
061         * before any that were previously registered.
062         *
063         * @param handler The new UrlInterceptHandler object
064         * @return true if the handler was not previously registered.
065         */
066        public static synchronized boolean registerHandler(
067                UrlInterceptHandler handler) {
068            if (!getHandlers().contains(handler)) {
069                getHandlers().addFirst(handler);
070                return true;
071            } else {
072                return false;
073            }
074        }
075    
076        /**
077         * Unregister a previously registered UrlInterceptHandler.
078         *
079         * @param handler A previously registered UrlInterceptHandler.
080         * @return true if the handler was found and removed from the list.
081         */
082        public static synchronized boolean unregisterHandler(
083                UrlInterceptHandler handler) {
084            return getHandlers().remove(handler);
085        }
086    
087        /**
088         * Given an url, returns the CacheResult of the first
089         * UrlInterceptHandler interested, or null if none are.
090         *
091         * @return A CacheResult containing surrogate content.
092         * @Deprecated Use PluginData getPluginData( String url,
093         * Map<String, String> headers) instead.
094         */
095        @Deprecated
096        public static synchronized CacheResult getSurrogate(
097                String url, Map<String, String> headers) {
098            if (urlInterceptDisabled()) {
099                return null;
100            }
101            Iterator iter = getHandlers().listIterator();
102            while (iter.hasNext()) {
103                UrlInterceptHandler handler = (UrlInterceptHandler) iter.next();
104                CacheResult result = handler.service(url, headers);
105                if (result != null) {
106                    return result;
107                }
108            }
109            return null;
110        }
111    
112        /**
113         * Given an url, returns the PluginData of the first
114         * UrlInterceptHandler interested, or null if none are or if
115         * intercepts are disabled.
116         *
117         * @return A PluginData instance containing surrogate content.
118         */
119        public static synchronized PluginData getPluginData(
120                String url, Map<String, String> headers) {
121            if (urlInterceptDisabled()) {
122                return null;
123            }
124            Iterator iter = getHandlers().listIterator();
125            while (iter.hasNext()) {
126                UrlInterceptHandler handler = (UrlInterceptHandler) iter.next();
127                PluginData data = handler.getPluginData(url, headers);
128                if (data != null) {
129                    return data;
130                }
131            }
132            return null;
133        }
134    }
