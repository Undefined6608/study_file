017    package android.webkit;
018    
019    import android.graphics.Bitmap;
020    
021    /**
022     * A convenience class for accessing fields in an entry in the back/forward list
023     * of a WebView. Each WebHistoryItem is a snapshot of the requested history
024     * item. Each history item may be updated during the load of a page.
025     * @see WebBackForwardList
026     */
027    public class WebHistoryItem implements Cloneable {
028        // Global identifier count.
029        private static int sNextId = 0;
030        // Unique identifier.
031        private final int mId;
032        // The title of this item's document.
033        private String mTitle;
034        // The base url of this item.
035        private String mUrl;
036        // The original requested url of this item.
037        private String mOriginalUrl;
038        // The favicon for this item.
039        private Bitmap mFavicon;
040        // The pre-flattened data used for saving the state.
041        private byte[] mFlattenedData;
042    
043        /**
044         * Basic constructor that assigns a unique id to the item. Called by JNI
045         * only.
046         */
047        private WebHistoryItem() {
048            synchronized (WebHistoryItem.class) {
049                mId = sNextId++;
050            }
051        }
052    
053        /**
054         * Construct a new WebHistoryItem with initial flattened data.
055         * @param data The pre-flattened data coming from restoreState.
056         */
057        /*package*/ WebHistoryItem(byte[] data) {
058            mUrl = null; // This will be updated natively
059            mFlattenedData = data;
060            synchronized (WebHistoryItem.class) {
061                mId = sNextId++;
062            }
063        }
064    
065        /**
066         * Construct a clone of a WebHistoryItem from the given item.
067         * @param item The history item to clone.
068         */
069        private WebHistoryItem(WebHistoryItem item) {
070            mUrl = item.mUrl;
071            mTitle = item.mTitle;
072            mFlattenedData = item.mFlattenedData;
073            mFavicon = item.mFavicon;
074            mId = item.mId;
075    }
076    
077        /**
078         * Return an identifier for this history item. If an item is a copy of
079         * another item, the identifiers will be the same even if they are not the
080         * same object.
081         * @return The id for this item.
082         */
083        public int getId() {
084            return mId;
085        }
086    
087        /**
088         * Return the url of this history item. The url is the base url of this
089         * history item. See getTargetUrl() for the url that is the actual target of
090         * this history item.
091         * @return The base url of this history item.
092         * Note: The VM ensures 32-bit atomic read/write operations so we don't have
093         * to synchronize this method.
094         */
095        public String getUrl() {
096            return mUrl;
097        }
098    
099        /**
100         * Return the original url of this history item. This was the requested
101         * url, the final url may be different as there might have been 
102         * redirects while loading the site.
103         * @return The original url of this history item.
104         */
105        public String getOriginalUrl() {
106            return mOriginalUrl;
107        }
108        
109        /**
110         * Return the document title of this history item.
111         * @return The document title of this history item.
112         * Note: The VM ensures 32-bit atomic read/write operations so we don't have
113         * to synchronize this method.
114         */
115        public String getTitle() {
116            return mTitle;
117        }
118    
119        /**
120         * Return the favicon of this history item or null if no favicon was found.
121         * @return A Bitmap containing the favicon for this history item or null.
122         * Note: The VM ensures 32-bit atomic read/write operations so we don't have
123         * to synchronize this method.
124         */
125        public Bitmap getFavicon() {
126            return mFavicon;
127        }
128    
129        /**
130         * Set the favicon.
131         * @param icon A Bitmap containing the favicon for this history item.
132         * Note: The VM ensures 32-bit atomic read/write operations so we don't have
133         * to synchronize this method.
134         */
135        /*package*/ void setFavicon(Bitmap icon) {
136            mFavicon = icon;
137        }
138    
139        /**
140         * Get the pre-flattened data.
141         * Note: The VM ensures 32-bit atomic read/write operations so we don't have
142         * to synchronize this method.
143         */
144        /*package*/ byte[] getFlattenedData() {
145            return mFlattenedData;
146        }
147    
148        /**
149         * Inflate this item.
150         * Note: The VM ensures 32-bit atomic read/write operations so we don't have
151         * to synchronize this method.
152         */
153        /*package*/ void inflate(int nativeFrame) {
154            inflate(nativeFrame, mFlattenedData);
155        }
156    
157        /**
158         * Clone the history item for use by clients of WebView.
159         */
160        protected synchronized WebHistoryItem clone() {
161            return new WebHistoryItem(this);
162        }
163    
164        /* Natively inflate this item, this method is called in the WebCore thread.
165         */
166        private native void inflate(int nativeFrame, byte[] data);
167    
168        /* Called by jni when the item is updated */
169        private void update(String url, String originalUrl, String title, 
170                Bitmap favicon, byte[] data) {
171            mUrl = url;
172            mOriginalUrl = originalUrl;
173            mTitle = title;
174            mFavicon = favicon;
175            mFlattenedData = data;
176        }
177    }
