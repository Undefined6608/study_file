037    /**
038     * The class CacheManager provides the persistent cache of content that is
039     * received over the network. The component handles parsing of HTTP headers and
040     * utilizes the relevant cache headers to determine if the content should be
041     * stored and if so, how long it is valid for. Network requests are provided to
042     * this component and if they can not be resolved by the cache, the HTTP headers
043     * are attached, as appropriate, to the request for revalidation of content. The
044     * class also manages the cache size.
045     */
046    public final class CacheManager {
047

051        static final String HEADER_KEY_IFNONEMATCH = "if-none-match";
052

054        private static final String NO_CACHE = "no-cache";
055        private static final String PRIVATE  = "private";
056        private static final String MAX_AGE = "max-age";

069        // timer base. We share the same timer with less frequent update.

151        /**
152         * initialize the CacheManager. WebView should handle this for each process.
153         * 
154         * @param context The application context.
155         */
156        static void init(Context context) {
157            mDataBase = WebViewDatabase.getInstance(context);
158            mBaseDir = new File(context.getCacheDir(), "webviewCache");
159            if (createCacheDirectory() && mClearCacheOnInit) {
160                removeAllCacheFiles();
161                mClearCacheOnInit = false;
162            }
163        }
164        
165        /**
166         * Create the cache directory if it does not already exist.
167         * 
168         * @return true if the cache directory didn't exist and was created.
169         */
170        static private boolean createCacheDirectory() {
171            if (!mBaseDir.exists()) {
172                if(!mBaseDir.mkdirs()) {
173                    Log.w(LOGTAG, "Unable to create webviewCache directory");
174                    return false;
175                }
176                FileUtils.setPermissions(
177                        mBaseDir.toString(),
178                        FileUtils.S_IRWXU|FileUtils.S_IRWXG|FileUtils.S_IXOTH,
179                        -1, -1);
180                // If we did create the directory, we need to flush 
181                // the cache database. The directory could be recreated
182                // because the system flushed all the data/cache directories
183                // to free up disk space.
184                WebViewCore.endCacheTransaction();
185                mDataBase.clearCache();
186                WebViewCore.startCacheTransaction();
187                return true;
188            }
189            return false;
190        }

227        // only called from WebCore thread
228        // make sure to call enableTransaction/disableTransaction in pair
229        static boolean enableTransaction() {
230            if (++mRefCount == 1) {
231                mDataBase.startCacheTransaction();
232                return true;
233            }
234            return false;
235        }
236    
237        // only called from WebCore thread
238        // make sure to call enableTransaction/disableTransaction in pair
239        static boolean disableTransaction() {
240            if (mRefCount == 0) {
241                Log.e(LOGTAG, "disableTransaction is out of sync");
242            }
243            if (--mRefCount == 0) {
244                mDataBase.endCacheTransaction();
245                return true;
246            }
247            return false;
248        }
249    
250        // only called from WebCore thread
251        // make sure to call startCacheTransaction/endCacheTransaction in pair
252        public static boolean startCacheTransaction() {
253            return mDataBase.startCacheTransaction();
254        }
255    
256        // only called from WebCore thread
257        // make sure to call startCacheTransaction/endCacheTransaction in pair
258        public static boolean endCacheTransaction() {
259            boolean ret = mDataBase.endCacheTransaction();
260            if (++mTrimCacheCount >= TRIM_CACHE_INTERVAL) {
261                mTrimCacheCount = 0;
262                trimCacheIfNeeded();
263            }
264            return ret;
265        }

267        /**
268         * Given a url, returns the CacheResult if exists. Otherwise returns null.
269         * If headers are provided and a cache needs validation,
270         * HEADER_KEY_IFNONEMATCH or HEADER_KEY_IFMODIFIEDSINCE will be set in the
271         * cached headers.
272         * 
273         * @return the CacheResult for a given url
274         */
275        // only called from WebCore thread
276        public static CacheResult getCacheFile(String url,
277                Map<String, String> headers) {
278            if (mDisabled) {
279                return null;
280            }
281    
282            CacheResult result = mDataBase.getCache(url);
283            if (result != null) {
284                if (result.contentLength == 0) {
285                    if (!checkCacheRedirect(result.httpStatusCode)) {
286                        // this should not happen. If it does, remove it.
287                        mDataBase.removeCache(url);
288                        return null;
289                    }
290                } else {
291                    File src = new File(mBaseDir, result.localPath);
292                    try {
293                        // open here so that even the file is deleted, the content
294                        // is still readable by the caller until close() is called
295                        result.inStream = new FileInputStream(src);
296                    } catch (FileNotFoundException e) {
297                        // the files in the cache directory can be removed by the
298                        // system. If it is gone, clean up the database
299                        mDataBase.removeCache(url);
300                        return null;
301                    }
302                }
303            } else {
304                return null;
305            }
306    
307            // null headers request coming from CACHE_MODE_CACHE_ONLY
308            // which implies that it needs cache even it is expired.
309            // negative expires means time in the far future.
310            if (headers != null && result.expires >= 0
311                    && result.expires <= System.currentTimeMillis()) {
312                if (result.lastModified == null && result.etag == null) {
313                    return null;
314                }
315                // return HEADER_KEY_IFNONEMATCH or HEADER_KEY_IFMODIFIEDSINCE
316                // for requesting validation
317                if (result.etag != null) {
318                    headers.put(HEADER_KEY_IFNONEMATCH, result.etag);
319                }
320                if (result.lastModified != null) {
321                    headers.put(HEADER_KEY_IFMODIFIEDSINCE, result.lastModified);
322                }
323            }
324    
325            if (Config.LOGV) {
326                Log.v(LOGTAG, "getCacheFile for url " + url);
327            }
328    
329            return result;
330        }

332        /**
333         * Given a url and its full headers, returns CacheResult if a local cache
334         * can be stored. Otherwise returns null. The mimetype is passed in so that
335         * the function can use the mimetype that will be passed to WebCore which
336         * could be different from the mimetype defined in the headers.
337         * forceCache is for out-of-package callers to force creation of a
338         * CacheResult, and is used to supply surrogate responses for URL
339         * interception.
340         * @return CacheResult for a given url
341         * @hide - hide createCacheFile since it has a parameter of type headers, which is
342         * in a hidden package.
343         */
344        // can be called from any thread
345        public static CacheResult createCacheFile(String url, int statusCode,
346                Headers headers, String mimeType, boolean forceCache) {
347            if (!forceCache && mDisabled) {
348                return null;
349            }
350    
351            // according to the rfc 2616, the 303 response MUST NOT be cached.
352            if (statusCode == 303) {
353                return null;
354            }
355    
356            // like the other browsers, do not cache redirects containing a cookie
357            // header.
358            if (checkCacheRedirect(statusCode) && !headers.getSetCookie().isEmpty()) {
359                return null;
360            }
361    
362            CacheResult ret = parseHeaders(statusCode, headers, mimeType);
363            if (ret != null) {
364                setupFiles(url, ret);
365                try {
366                    ret.outStream = new FileOutputStream(ret.outFile);
367                } catch (FileNotFoundException e) {
368                    // This can happen with the system did a purge and our
369                    // subdirectory has gone, so lets try to create it again
370                    if (createCacheDirectory()) {
371                        try {
372                            ret.outStream = new FileOutputStream(ret.outFile);
373                        } catch  (FileNotFoundException e2) {
374                            // We failed to create the file again, so there
375                            // is something else wrong. Return null.
376                            return null;
377                        }
378                    } else {
379                        // Failed to create cache directory
380                        return null;
381                    }
382                }
383                ret.mimeType = mimeType;
384            }
385    
386            return ret;
387        }

389        /**
390         * Save the info of a cache file for a given url to the CacheMap so that it
391         * can be reused later
392         */
393        // only called from WebCore thread
394        public static void saveCacheFile(String url, CacheResult cacheRet) {
395            try {
396                cacheRet.outStream.close();
397            } catch (IOException e) {
398                return;
399            }
400    
401            if (!cacheRet.outFile.exists()) {
402                // the file in the cache directory can be removed by the system
403                return;
404            }
405    
406            cacheRet.contentLength = cacheRet.outFile.length();
407            if (checkCacheRedirect(cacheRet.httpStatusCode)) {
408                // location is in database, no need to keep the file
409                cacheRet.contentLength = 0;
410                cacheRet.localPath = new String();
411                cacheRet.outFile.delete();
412            } else if (cacheRet.contentLength == 0) {
413                cacheRet.outFile.delete();
414                return;
415            }
416    
417            mDataBase.addCache(url, cacheRet);
418    
419            if (Config.LOGV) {
420                Log.v(LOGTAG, "saveCacheFile for url " + url);
421            }
422        }

424        /**
425         * remove all cache files
426         * 
427         * @return true if it succeeds
428         */
429        // only called from WebCore thread
430        static boolean removeAllCacheFiles() {
431            // Note, this is called before init() when the database is
432            // created or upgraded.
433            if (mBaseDir == null) {
434                // Init() has not been called yet, so just flag that
435                // we need to clear the cache when init() is called.
436                mClearCacheOnInit = true;
437                return true;
438            }
439            // delete cache in a separate thread to not block UI.
440            final Runnable clearCache = new Runnable() {
441                public void run() {
442                    // delete all cache files
443                    try {
444                        String[] files = mBaseDir.list();
445                        // if mBaseDir doesn't exist, files can be null.
446                        if (files != null) {
447                            for (int i = 0; i < files.length; i++) {
448                                new File(mBaseDir, files[i]).delete();
449                            }
450                        }
451                    } catch (SecurityException e) {
452                        // Ignore SecurityExceptions.
453                    }
454                    // delete database
455                    mDataBase.clearCache();
456                }
457            };
458            new Thread(clearCache).start();
459            return true;
460        }

470        // only called from WebCore thread
471        static void trimCacheIfNeeded() {
472            if (mDataBase.getCacheTotalSize() > CACHE_THRESHOLD) {
473                ArrayList<String> pathList = mDataBase.trimCache(CACHE_TRIM_AMOUNT);
474                int size = pathList.size();
475                for (int i = 0; i < size; i++) {
476                    new File(mBaseDir, pathList.get(i)).delete();
477                }
478            }
479        }

481        private static boolean checkCacheRedirect(int statusCode) {
482            if (statusCode == 301 || statusCode == 302 || statusCode == 307) {
483                // as 303 can't be cached, we do not return true
484                return true;
485            } else {
486                return false;
487            }
488        }

490        @SuppressWarnings("deprecation")
491        private static void setupFiles(String url, CacheResult cacheRet) {
492            if (true) {
493                // Note: SHA1 is much stronger hash. But the cost of setupFiles() is
494                // 3.2% cpu time for a fresh load of nytimes.com. While a simple
495                // String.hashCode() is only 0.6%. If adding the collision resolving
496                // to String.hashCode(), it makes the cpu time to be 1.6% for a 
497                // fresh load, but 5.3% for the worst case where all the files 
498                // already exist in the file system, but database is gone. So it
499                // needs to resolve collision for every file at least once.
500                int hashCode = url.hashCode();
501                StringBuffer ret = new StringBuffer(8);
502                appendAsHex(hashCode, ret);
503                String path = ret.toString();
504                File file = new File(mBaseDir, path);
505                if (true) {
506                    boolean checkOldPath = true;
507                    // Check hash collision. If the hash file doesn't exist, just
508                    // continue. There is a chance that the old cache file is not
509                    // same as the hash file. As mDataBase.getCache() is more 
510                    // expansive than "leak" a file until clear cache, don't bother.
511                    // If the hash file exists, make sure that it is same as the 
512                    // cache file. If it is not, resolve the collision.
513                    while (file.exists()) {
514                        if (checkOldPath) {
515                            // as this is called from http thread through 
516                            // createCacheFile, we need endCacheTransaction before 
517                            // database access.
518                            WebViewCore.endCacheTransaction();
519                            CacheResult oldResult = mDataBase.getCache(url);
520                            WebViewCore.startCacheTransaction();
521                            if (oldResult != null && oldResult.contentLength > 0) {
522                                if (path.equals(oldResult.localPath)) {
523                                    path = oldResult.localPath;
524                                } else {
525                                    path = oldResult.localPath;
526                                    file = new File(mBaseDir, path);
527                                }
528                                break;
529                            }
530                            checkOldPath = false;
531                        }
532                        ret = new StringBuffer(8);
533                        appendAsHex(++hashCode, ret);
534                        path = ret.toString();
535                        file = new File(mBaseDir, path);
536                    }
537                }
538                cacheRet.localPath = path;
539                cacheRet.outFile = file;
540            } else {
541                // get hash in byte[]
542                Digest digest = new SHA1Digest();
543                int digestLen = digest.getDigestSize();
544                byte[] hash = new byte[digestLen];
545                int urlLen = url.length();
546                byte[] data = new byte[urlLen];
547                url.getBytes(0, urlLen, data, 0);
548                digest.update(data, 0, urlLen);
549                digest.doFinal(hash, 0);
550                // convert byte[] to hex String
551                StringBuffer result = new StringBuffer(2 * digestLen);
552                for (int i = 0; i < digestLen; i = i + 4) {
553                    int h = (0x00ff & hash[i]) << 24 | (0x00ff & hash[i + 1]) << 16
554                            | (0x00ff & hash[i + 2]) << 8 | (0x00ff & hash[i + 3]);
555                    appendAsHex(h, result);
556                }
557                cacheRet.localPath = result.toString();
558                cacheRet.outFile = new File(mBaseDir, cacheRet.localPath);
559            }
560        }

562        private static void appendAsHex(int i, StringBuffer ret) {
563            String hex = Integer.toHexString(i);
564            switch (hex.length()) {
565                case 1:
566                    ret.append("0000000");
567                    break;
568                case 2:
569                    ret.append("000000");
570                    break;
571                case 3:
572                    ret.append("00000");
573                    break;
574                case 4:
575                    ret.append("0000");
576                    break;
577                case 5:
578                    ret.append("000");
579                    break;
580                case 6:
581                    ret.append("00");
582                    break;
583                case 7:
584                    ret.append("0");
585                    break;
586            }
587            ret.append(hex);
588        }

590        private static CacheResult parseHeaders(int statusCode, Headers headers,
591                String mimeType) {
592            // TODO: if authenticated or secure, return null
593            CacheResult ret = new CacheResult();
594            ret.httpStatusCode = statusCode;
595    
596            String location = headers.getLocation();
597            if (location != null) ret.location = location;
598    
599            ret.expires = -1;
600            String expires = headers.getExpires();
601            if (expires != null) {
602                try {
603                    ret.expires = HttpDateTime.parse(expires);
604                } catch (IllegalArgumentException ex) {
605                    // Take care of the special "-1" and "0" cases
606                    if ("-1".equals(expires) || "0".equals(expires)) {
607                        // make it expired, but can be used for history navigation
608                        ret.expires = 0;
609                    } else {
610                        Log.e(LOGTAG, "illegal expires: " + expires);
611                    }
612                }
613            }
614    
615            String lastModified = headers.getLastModified();
616            if (lastModified != null) ret.lastModified = lastModified;
617    
618            String etag = headers.getEtag();
619            if (etag != null) ret.etag = etag;
620    
621            String cacheControl = headers.getCacheControl();
622            if (cacheControl != null) {
623                String[] controls = cacheControl.toLowerCase().split("[ ,;]");
624                for (int i = 0; i < controls.length; i++) {
625                    if (NO_STORE.equals(controls[i])) {
626                        return null;
627                    }
628                    // According to the spec, 'no-cache' means that the content
629                    // must be re-validated on every load. It does not mean that
630                    // the content can not be cached. set to expire 0 means it
631                    // can only be used in CACHE_MODE_CACHE_ONLY case
632                    if (NO_CACHE.equals(controls[i]) || PRIVATE.equals(controls[i])) {
633                        ret.expires = 0;
634                    } else if (controls[i].startsWith(MAX_AGE)) {
635                        int separator = controls[i].indexOf('=');
636                        if (separator < 0) {
637                            separator = controls[i].indexOf(':');
638                        }
639                        if (separator > 0) {
640                            String s = controls[i].substring(separator + 1);
641                            try {
642                                long sec = Long.parseLong(s);
643                                if (sec >= 0) {
644                                    ret.expires = System.currentTimeMillis() + 1000
645                                            * sec;
646                                }
647                            } catch (NumberFormatException ex) {
648                                if ("1d".equals(s)) {
649                                    // Take care of the special "1d" case
650                                    ret.expires = System.currentTimeMillis() + 86400000; // 24*60*60*1000
651                                } else {
652                                    Log.e(LOGTAG, "exception in parseHeaders for "
653                                            + "max-age:"
654                                            + controls[i].substring(separator + 1));
655                                    ret.expires = 0;
656                                }
657                            }
658                        }
659                    }
660                }
661            }
662    
663            // According to RFC 2616 section 14.32:
664            // HTTP/1.1 caches SHOULD treat "Pragma: no-cache" as if the
665            // client had sent "Cache-Control: no-cache"
666            if (NO_CACHE.equals(headers.getPragma())) {
667                ret.expires = 0;
668            }
669    
670            // According to RFC 2616 section 13.2.4, if an expiration has not been
671            // explicitly defined a heuristic to set an expiration may be used.
672            if (ret.expires == -1) {
673                if (ret.httpStatusCode == 301) {
674                    // If it is a permanent redirect, and it did not have an
675                    // explicit cache directive, then it never expires
676                    ret.expires = Long.MAX_VALUE;
677                } else if (ret.httpStatusCode == 302 || ret.httpStatusCode == 307) {
678                    // If it is temporary redirect, expires
679                    ret.expires = 0;
680                } else if (ret.lastModified == null) {
681                    // When we have no last-modified, then expire the content with
682                    // in 24hrs as, according to the RFC, longer time requires a
683                    // warning 113 to be added to the response.
684    
685                    // Only add the default expiration for non-html markup. Some
686                    // sites like news.google.com have no cache directives.
687                    if (!mimeType.startsWith("text/html")) {
688                        ret.expires = System.currentTimeMillis() + 86400000; // 24*60*60*1000
689                    } else {
690                        // Setting a expires as zero will cache the result for
691                        // forward/back nav.
692                        ret.expires = 0;
693                    }
694                } else {
695                    // If we have a last-modified value, we could use it to set the
696                    // expiration. Suggestion from RFC is 10% of time since
697                    // last-modified. As we are on mobile, loads are expensive,
698                    // increasing this to 20%.
699    
700                    // 24 * 60 * 60 * 1000
701                    long lastmod = System.currentTimeMillis() + 86400000;
702                    try {
703                        lastmod = HttpDateTime.parse(ret.lastModified);
704                    } catch (IllegalArgumentException ex) {
705                        Log.e(LOGTAG, "illegal lastModified: " + ret.lastModified);
706                    }
707                    long difference = System.currentTimeMillis() - lastmod;
708                    if (difference > 0) {
709                        ret.expires = System.currentTimeMillis() + difference / 5;
710                    } else {
711                        // last modified is in the future, expire the content
712                        // on the last modified
713                        ret.expires = lastmod;
714                    }
715                }
716            }
717    
718            return ret;
719        }
