017    package android.webkit;
018    
019    import android.content.Context;
020    import android.content.res.Resources;
021    
022    import java.util.Calendar;
023    import java.util.Date;
024    
025    /**
026     * Sorts dates into the following groups:
027     *   Today
028     *   Yesterday
029     *   five days ago
030     *   one month ago
031     *   older than a month ago
032     */
033    
034    public class DateSorter {
035    
036        private static final String LOGTAG = "webkit";
037    
038        /** must be >= 3 */
039        public static final int DAY_COUNT = 5;
040    
041        private long [] mBins = new long[DAY_COUNT];
042        private String [] mLabels = new String[DAY_COUNT];
043        
044        private static final int NUM_DAYS_AGO = 5;
045    
046        Date mDate = new Date();
047        Calendar mCal = Calendar.getInstance();
048    
049        /**
050         * @param context Application context
051         */
052        public DateSorter(Context context) {
053            Resources resources = context.getResources();
054    
055            Calendar c = Calendar.getInstance();
056            beginningOfDay(c);
057            
058            // Create the bins
059            mBins[0] = c.getTimeInMillis(); // Today
060            c.roll(Calendar.DAY_OF_YEAR, -1);
061            mBins[1] = c.getTimeInMillis();  // Yesterday
062            c.roll(Calendar.DAY_OF_YEAR, -(NUM_DAYS_AGO - 1));
063            mBins[2] = c.getTimeInMillis();  // Five days ago
064            c.roll(Calendar.DAY_OF_YEAR, NUM_DAYS_AGO); // move back to today
065            c.roll(Calendar.MONTH, -1);
066            mBins[3] = c.getTimeInMillis();  // One month ago
067            c.roll(Calendar.MONTH, -1);
068            mBins[4] = c.getTimeInMillis();  // Over one month ago
069    
070            // build labels
071            mLabels[0] = context.getText(com.android.internal.R.string.today).toString();
072            mLabels[1] = context.getText(com.android.internal.R.string.yesterday).toString();
073    
074            int resId = com.android.internal.R.plurals.num_days_ago;
075            String format = resources.getQuantityString(resId, NUM_DAYS_AGO);
076            mLabels[2] = String.format(format, NUM_DAYS_AGO);
077    
078            mLabels[3] = context.getText(com.android.internal.R.string.oneMonthDurationPast).toString();
079            mLabels[4] = context.getText(com.android.internal.R.string.beforeOneMonthDurationPast)
080                    .toString();
081        }
082    
083        /**
084         * @param time time since the Epoch in milliseconds, such as that
085         * returned by Calendar.getTimeInMillis()
086         * @return an index from 0 to (DAY_COUNT - 1) that identifies which
087         * date bin this date belongs to
088         */
089        public int getIndex(long time) {
090            // Lame linear search
091            for (int i = 0; i < DAY_COUNT; i++) {
092                if (time > mBins[i]) return i;
093            }
094            return DAY_COUNT - 1;
095        }
096    
097        /**
098         * @param index date bin index as returned by getIndex()
099         * @return string label suitable for display to user
100         */
101        public String getLabel(int index) {
102            return mLabels[index];
103        }
104    
105    
106        /**
107         * @param index date bin index as returned by getIndex()
108         * @return date boundary at given index
109         */
110        public long getBoundary(int index) {
111            return mBins[index];
112        }
113    
114        /**
115         * Calcuate 12:00am by zeroing out hour, minute, second, millisecond
116         */
117        private Calendar beginningOfDay(Calendar c) {
118            c.set(Calendar.HOUR_OF_DAY, 0);
119            c.set(Calendar.MINUTE, 0);
120            c.set(Calendar.SECOND, 0);
121            c.set(Calendar.MILLISECOND, 0);
122            return c;
123        }
124    }
