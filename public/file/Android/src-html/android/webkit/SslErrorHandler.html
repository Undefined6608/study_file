017    package android.webkit;
018    
019    import junit.framework.Assert;
020    
021    import android.net.http.SslError;
022    import android.os.Bundle;
023    import android.os.Handler;
024    import android.os.Message;
025    import android.util.Config;
026    import android.util.Log;
027    
028    import java.util.LinkedList;
029    import java.util.ListIterator;
030    
031    /**
032     * SslErrorHandler: class responsible for handling SSL errors. This class is
033     * passed as a parameter to BrowserCallback.displaySslErrorDialog and is meant
034     * to receive the user's response.
035     */
036    public class SslErrorHandler extends Handler {
037        /* One problem here is that there may potentially be multiple SSL errors
038         * coming from mutiple loaders. Therefore, we keep a queue of loaders
039         * that have SSL-related problems and process errors one by one in the
040         * order they were received.
041         */
042    
043        private static final String LOGTAG = "network";
044    
045        /**
046         * Network.
047         */
048        private Network mNetwork;
049    
050        /**
051         * Queue of loaders that experience SSL-related problems.
052         */
053        private LinkedList<LoadListener> mLoaderQueue;
054    
055        /**
056         * SSL error preference table.
057         */
058        private Bundle mSslPrefTable;
059    
060        // Message id for handling the response
061        private final int HANDLE_RESPONSE = 100;
062    
063        @Override
064        public void handleMessage(Message msg) {
065            switch (msg.what) {
066                case HANDLE_RESPONSE:
067                    handleSslErrorResponse(msg.arg1 == 1);
068                    fastProcessQueuedSslErrors();
069                    break;
070            }
071        }
072    
073        /**
074         * Creates a new error handler with an empty loader queue.
075         */
076        /* package */ SslErrorHandler(Network network) {
077            mNetwork = network;
078    
079            mLoaderQueue = new LinkedList<LoadListener>();
080            mSslPrefTable = new Bundle();
081        }
082    
083        /**
084         * Saves this handler's state into a map.
085         * @return True iff succeeds.
086         */
087        /* package */ boolean saveState(Bundle outState) {
088            boolean success = (outState != null);
089            if (success) {
090                // TODO?
091                outState.putBundle("ssl-error-handler", mSslPrefTable);
092            }
093    
094            return success;
095        }
096    
097        /**
098         * Restores this handler's state from a map.
099         * @return True iff succeeds.
100         */
101        /* package */ boolean restoreState(Bundle inState) {
102            boolean success = (inState != null);
103            if (success) {
104                success = inState.containsKey("ssl-error-handler");
105                if (success) {
106                    mSslPrefTable = inState.getBundle("ssl-error-handler");
107                }
108            }
109    
110            return success;
111        }
112    
113        /**
114         * Clears SSL error preference table.
115         */
116        /* package */ synchronized void clear() {
117            mSslPrefTable.clear();
118        }
119    
120        /**
121         * Handles SSL error(s) on the way up to the user.
122         */
123        /* package */ synchronized void handleSslErrorRequest(LoadListener loader) {
124            if (Config.LOGV) {
125                Log.v(LOGTAG, "SslErrorHandler.handleSslErrorRequest(): " +
126                      "url=" + loader.url());
127            }
128    
129            if (!loader.cancelled()) {
130                mLoaderQueue.offer(loader);
131                if (loader == mLoaderQueue.peek()) {
132                    fastProcessQueuedSslErrors();
133                }
134            }
135        }
136    
137        /**
138         * Processes queued SSL-error confirmation requests in
139         * a tight loop while there is no need to ask the user.
140         */
141        /* package */void fastProcessQueuedSslErrors() {
142            while (processNextLoader());
143        }
144    
145        /**
146         * Processes the next loader in the queue.
147         * @return True iff should proceed to processing the
148         * following loader in the queue
149         */
150        private synchronized boolean processNextLoader() {
151            LoadListener loader = mLoaderQueue.peek();
152            if (loader != null) {
153                // if this loader has been cancelled
154                if (loader.cancelled()) {
155                    // go to the following loader in the queue
156                    return true;
157                }
158    
159                SslError error = loader.sslError();
160    
161                if (Config.DEBUG) {
162                    Assert.assertNotNull(error);
163                }
164    
165                int primary = error.getPrimaryError();
166                String host = loader.host();
167    
168                if (Config.DEBUG) {
169                    Assert.assertTrue(host != null && primary != 0);
170                }
171    
172                if (mSslPrefTable.containsKey(host)) {
173                    if (primary <= mSslPrefTable.getInt(host)) {
174                        handleSslErrorResponse(true);
175                        return true;
176                    }
177                }
178    
179                // if we do not have information on record, ask
180                // the user (display a dialog)
181                CallbackProxy proxy = loader.getFrame().getCallbackProxy();
182                proxy.onReceivedSslError(this, error);
183            }
184    
185            // the queue must be empty, stop
186            return false;
187        }
188    
189        /**
190         * Proceed with the SSL certificate.
191         */
192        public void proceed() {
193            sendMessage(obtainMessage(HANDLE_RESPONSE, 1, 0));
194        }
195    
196        /**
197         * Cancel this request and all pending requests for the WebView that had
198         * the error.
199         */
200        public void cancel() {
201            sendMessage(obtainMessage(HANDLE_RESPONSE, 0, 0));
202        }
203    
204        /**
205         * Handles SSL error(s) on the way down from the user.
206         */
207        /* package */ synchronized void handleSslErrorResponse(boolean proceed) {
208            LoadListener loader = mLoaderQueue.poll();
209            if (Config.DEBUG) {
210                Assert.assertNotNull(loader);
211            }
212    
213            if (Config.LOGV) {
214                Log.v(LOGTAG, "SslErrorHandler.handleSslErrorResponse():"
215                      + " proceed: " + proceed
216                      + " url:" + loader.url());
217            }
218    
219            if (!loader.cancelled()) {
220                if (proceed) {
221                    // update the user's SSL error preference table
222                    int primary = loader.sslError().getPrimaryError();
223                    String host = loader.host();
224    
225                    if (Config.DEBUG) {
226                        Assert.assertTrue(host != null && primary != 0);
227                    }
228                    boolean hasKey = mSslPrefTable.containsKey(host);
229                    if (!hasKey ||
230                        primary > mSslPrefTable.getInt(host)) {
231                        mSslPrefTable.putInt(host, new Integer(primary));
232                    }
233                }
234                loader.handleSslErrorResponse(proceed);
235            }
236        }
237    }
