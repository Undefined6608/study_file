639                    }
640                });

643        private void updateZoomButtonsEnabled() {
644            boolean canZoomIn = mActualScale < mMaxZoomScale;
645            boolean canZoomOut = mActualScale > mMinZoomScale;
646            if (!canZoomIn && !canZoomOut) {
647                // Hide the zoom in and out buttons, as well as the fit to page
648                // button, if the page cannot zoom
649                mZoomButtonsController.getZoomControls().setVisibility(View.GONE);
650                mZoomFitPageButton.setVisibility(View.GONE);
651            } else {
652                // Bring back the hidden zoom controls.
653                mZoomButtonsController.getZoomControls()
654                        .setVisibility(View.VISIBLE);
655                mZoomFitPageButton.setVisibility(View.VISIBLE);
656                // Set each one individually, as a page may be able to zoom in
657                // or out.

918        /**
919         * Save the state of this WebView used in 
920         * {@link android.app.Activity#onSaveInstanceState}. Please note that this
921         * method no longer stores the display data for this WebView. The previous
922         * behavior could potentially leak files if {@link #restoreState} was never
923         * called. See {@link #savePicture} and {@link #restorePicture} for saving
924         * and restoring the display data.
925         * @param outState The Bundle to store the WebView state.
926         * @return The same copy of the back/forward list used to save the state. If
927         *         saveState fails, the returned list will be null.
928         * @see #savePicture
929         * @see #restorePicture
930         */
931        public WebBackForwardList saveState(Bundle outState) {
932            if (outState == null) {
933                return null;
934            }
935            // We grab a copy of the back/forward list because a client of WebView
936            // may have invalidated the history list by calling clearHistory.
937            WebBackForwardList list = copyBackForwardList();
938            final int currentIndex = list.getCurrentIndex();
939            final int size = list.getSize();
940            // We should fail saving the state if the list is empty or the index is
941            // not in a valid range.
942            if (currentIndex < 0 || currentIndex >= size || size == 0) {
943                return null;
944            }
945            outState.putInt("index", currentIndex);
946            // FIXME: This should just be a byte[][] instead of ArrayList but
947            // Parcel.java does not have the code to handle multi-dimensional
948            // arrays.
949            ArrayList<byte[]> history = new ArrayList<byte[]>(size);
950            for (int i = 0; i < size; i++) {
951                WebHistoryItem item = list.getItemAtIndex(i);
952                byte[] data = item.getFlattenedData();
953                if (data == null) {
954                    // It would be very odd to not have any data for a given history
955                    // item. And we will fail to rebuild the history list without
956                    // flattened data.
957                    return null;
958                }
959                history.add(data);
960            }
961            outState.putSerializable("history", history);
962            if (mCertificate != null) {
963                outState.putBundle("certificate",
964                                   SslCertificate.saveState(mCertificate));
965            }
966            return list;
967        }

1048        /**
1049         * Restore the state of this WebView from the given map used in
1050         * {@link android.app.Activity#onRestoreInstanceState}. This method should 
1051         * be called to restore the state of the WebView before using the object. If 
1052         * it is called after the WebView has had a chance to build state (load 
1053         * pages, create a back/forward list, etc.) there may be undesirable 
1054         * side-effects. Please note that this method no longer restores the
1055         * display data for this WebView. See {@link #savePicture} and {@link
1056         * #restorePicture} for saving and restoring the display data.
1057         * @param inState The incoming Bundle of state.
1058         * @return The restored back/forward list or null if restoreState failed.
1059         * @see #savePicture
1060         * @see #restorePicture
1061         */
1062        public WebBackForwardList restoreState(Bundle inState) {
1063            WebBackForwardList returnList = null;
1064            if (inState == null) {
1065                return returnList;
1066            }
1067            if (inState.containsKey("index") && inState.containsKey("history")) {
1068                mCertificate = SslCertificate.restoreState(
1069                    inState.getBundle("certificate"));
1070    
1071                final WebBackForwardList list = mCallbackProxy.getBackForwardList();
1072                final int index = inState.getInt("index");
1073                // We can't use a clone of the list because we need to modify the
1074                // shared copy, so synchronize instead to prevent concurrent
1075                // modifications.
1076                synchronized (list) {
1077                    final List<byte[]> history =
1078                            (List<byte[]>) inState.getSerializable("history");
1079                    final int size = history.size();
1080                    // Check the index bounds so we don't crash in native code while
1081                    // restoring the history index.
1082                    if (index < 0 || index >= size) {
1083                        return null;
1084                    }
1085                    for (int i = 0; i < size; i++) {
1086                        byte[] data = history.remove(0);
1087                        if (data == null) {
1088                            // If we somehow have null data, we cannot reconstruct
1089                            // the item and thus our history list cannot be rebuilt.
1090                            return null;
1091                        }
1092                        WebHistoryItem item = new WebHistoryItem(data);
1093                        list.addHistoryItem(item);
1094                    }
1095                    // Grab the most recent copy to return to the caller.
1096                    returnList = copyBackForwardList();
1097                    // Update the copy to have the correct index.
1098                    returnList.setCurrentIndex(index);
1099                }
1100                // Remove all pending messages because we are restoring previous
1101                // state.
1102                mWebViewCore.removeMessages();
1103                // Send a restore state message.
1104                mWebViewCore.sendMessage(EventHub.RESTORE_STATE, index);
1105            }
1106            return returnList;
1107        }

1418        /**
1419         * Return a HitTestResult based on the current focus node. If a HTML::a tag
1420         * is found and the anchor has a non-javascript url, the HitTestResult type
1421         * is set to SRC_ANCHOR_TYPE and the url is set in the "extra" field. If the
1422         * anchor does not have a url or if it is a javascript url, the type will
1423         * be UNKNOWN_TYPE and the url has to be retrieved through
1424         * {@link #requestFocusNodeHref} asynchronously. If a HTML::img tag is
1425         * found, the HitTestResult type is set to IMAGE_TYPE and the url is set in
1426         * the "extra" field. A type of
1427         * SRC_IMAGE_ANCHOR_TYPE indicates an anchor with a url that has an image as
1428         * a child node. If a phone number is found, the HitTestResult type is set
1429         * to PHONE_TYPE and the phone number is set in the "extra" field of
1430         * HitTestResult. If a map address is found, the HitTestResult type is set
1431         * to GEO_TYPE and the address is set in the "extra" field of HitTestResult.
1432         * If an email address is found, the HitTestResult type is set to EMAIL_TYPE
1433         * and the email is set in the "extra" field of HitTestResult. Otherwise,
1434         * HitTestResult type is set to UNKNOWN_TYPE.
1435         */
1436        public HitTestResult getHitTestResult() {
1437            if (mNativeClass == 0) {
1438                return null;
1439            }
1440    
1441            HitTestResult result = new HitTestResult();
1442    
1443            if (nativeUpdateFocusNode()) {
1444                FocusNode node = mFocusNode;
1445                if (node.mIsTextField || node.mIsTextArea) {
1446                    result.setType(HitTestResult.EDIT_TEXT_TYPE);
1447                } else if (node.mText != null) {
1448                    String text = node.mText;
1449                    if (text.startsWith(SCHEME_TEL)) {
1450                        result.setType(HitTestResult.PHONE_TYPE);
1451                        result.setExtra(text.substring(SCHEME_TEL.length()));
1452                    } else if (text.startsWith(SCHEME_MAILTO)) {
1453                        result.setType(HitTestResult.EMAIL_TYPE);
1454                        result.setExtra(text.substring(SCHEME_MAILTO.length()));
1455                    } else if (text.startsWith(SCHEME_GEO)) {
1456                        result.setType(HitTestResult.GEO_TYPE);
1457                        result.setExtra(URLDecoder.decode(text
1458                                .substring(SCHEME_GEO.length())));
1459                    } else if (node.mIsAnchor) {
1460                        result.setType(HitTestResult.SRC_ANCHOR_TYPE);
1461                        result.setExtra(text);
1462                    }
1463                }
1464            }
1465            int type = result.getType();
1466            if (type == HitTestResult.UNKNOWN_TYPE
1467                    || type == HitTestResult.SRC_ANCHOR_TYPE) {
1468                // Now check to see if it is an image.
1469                int contentX = viewToContent((int) mLastTouchX + mScrollX);
1470                int contentY = viewToContent((int) mLastTouchY + mScrollY);
1471                String text = nativeImageURI(contentX, contentY);
1472                if (text != null) {
1473                    result.setType(type == HitTestResult.UNKNOWN_TYPE ? 
1474                            HitTestResult.IMAGE_TYPE : 
1475                            HitTestResult.SRC_IMAGE_ANCHOR_TYPE);
1476                    result.setExtra(text);
1477                }
1478            }
1479            return result;
1480        }

1610        private void setNewZoomScale(float scale, boolean force) {
1611            if (scale < mMinZoomScale) {
1612                scale = mMinZoomScale;
1613            } else if (scale > mMaxZoomScale) {
1614                scale = mMaxZoomScale;
1615            }
1616            if (scale != mActualScale || force) {
1617                if (mDrawHistory) {
1618                    // If history Picture is drawn, don't update scroll. They will
1619                    // be updated when we get out of that mode.
1620                    if (scale != mActualScale && !mPreviewZoomOnly) {
1621                        mCallbackProxy.onScaleChanged(mActualScale, scale);
1622                    }
1623                    mActualScale = scale;
1624                    mInvActualScale = 1 / scale;
1625                    if (!mPreviewZoomOnly) {
1626                        sendViewSizeZoom();
1627                    }
1628                } else {
1629                    // update our scroll so we don't appear to jump
1630                    // i.e. keep the center of the doc in the center of the view
1631    
1632                    int oldX = mScrollX;
1633                    int oldY = mScrollY;
1634                    float ratio = scale * mInvActualScale;   // old inverse
1635                    float sx = ratio * oldX + (ratio - 1) * mZoomCenterX;
1636                    float sy = ratio * oldY + (ratio - 1) * mZoomCenterY;
1637    
1638                    // now update our new scale and inverse
1639                    if (scale != mActualScale && !mPreviewZoomOnly) {
1640                        mCallbackProxy.onScaleChanged(mActualScale, scale);
1641                    }
1642                    mActualScale = scale;
1643                    mInvActualScale = 1 / scale;
1644    
1645                    // as we don't have animation for scaling, don't do animation 
1646                    // for scrolling, as it causes weird intermediate state
1647                    //        pinScrollTo(Math.round(sx), Math.round(sy));
1648                    mScrollX = pinLocX(Math.round(sx));
1649                    mScrollY = pinLocY(Math.round(sy));
1650    
1651                    if (!mPreviewZoomOnly) {
1652                        sendViewSizeZoom();
1653                        sendOurVisibleRect();
1654                    }
1655                }
1656            }
1657        }

1703        /**
1704         * Compute unzoomed width and height, and if they differ from the last
1705         * values we sent, send them to webkit (to be used has new viewport)
1706         *
1707         * @return true if new values were sent
1708         */
1709        private boolean sendViewSizeZoom() {
1710            int newWidth = Math.round(getViewWidth() * mInvActualScale);
1711            int newHeight = Math.round(getViewHeight() * mInvActualScale);
1712            /*
1713             * Because the native side may have already done a layout before the
1714             * View system was able to measure us, we have to send a height of 0 to
1715             * remove excess whitespace when we grow our width. This will trigger a
1716             * layout and a change in content size. This content size change will
1717             * mean that contentSizeChanged will either call this method directly or
1718             * indirectly from onSizeChanged.
1719             */
1720            if (newWidth > mLastWidthSent && mWrapContent) {
1721                newHeight = 0;
1722            }
1723            // Avoid sending another message if the dimensions have not changed.
1724            if (newWidth != mLastWidthSent || newHeight != mLastHeightSent) {
1725                mWebViewCore.sendMessage(EventHub.VIEW_SIZE_CHANGED,
1726                        newWidth, newHeight, new Float(mActualScale));
1727                mLastWidthSent = newWidth;
1728                mLastHeightSent = newHeight;
1729                return true;
1730            }
1731            return false;
1732        }

1997        // returns true if the scroll was changed
1998        private boolean pinScrollTo(int x, int y, boolean animate, int animationDuration) {
1999            x = pinLocX(x);
2000            y = pinLocY(y);
2001            int dx = x - mScrollX;
2002            int dy = y - mScrollY;
2003    
2004            if ((dx | dy) == 0) {
2005                return false;
2006            }
2007    
2008            if (true && animate) {
2009                //        Log.d(LOGTAG, "startScroll: " + dx + " " + dy);
2010    
2011                mScroller.startScroll(mScrollX, mScrollY, dx, dy,
2012                        animationDuration > 0 ? animationDuration : computeDuration(dx, dy));
2013                invalidate();
2014            } else {
2015                mScroller.abortAnimation(); // just in case
2016                scrollTo(x, y);
2017            }
2018            return true;
2019        }

2424        // Returns frame surrounding magified portion of screen while 
2425        // scroll-zoom is enabled. The frame is also used to center the
2426        // zoom-in zoom-out points at the start and end of the animation.
2427        private Rect scrollZoomFrame(int width, int height, float halfScale) {
2428            Rect scrollFrame = new Rect();
2429            scrollFrame.set(mZoomScrollX, mZoomScrollY, 
2430                    mZoomScrollX + width, mZoomScrollY + height);
2431            if (mContentWidth * mZoomScrollLimit < width) {
2432                float scale = zoomFrameScaleX(width, halfScale, 1.0f);
2433                float offsetX = (width * scale - width) * 0.5f;
2434                scrollFrame.left -= offsetX;
2435                scrollFrame.right += offsetX;
2436            }
2437            if (mContentHeight * mZoomScrollLimit < height) {
2438                float scale = zoomFrameScaleY(height, halfScale, 1.0f);
2439                float offsetY = (height * scale - height) * 0.5f;
2440                scrollFrame.top -= offsetY;
2441                scrollFrame.bottom += offsetY;
2442            }
2443            return scrollFrame;
2444        }

2446        private float zoomFrameScaleX(int width, float halfScale, float noScale) {
2447            // mContentWidth > width > mContentWidth * mZoomScrollLimit
2448            if (mContentWidth <= width) {
2449                return halfScale;
2450            }
2451            float part = (width - mContentWidth * mZoomScrollLimit)  
2452                    / (width * (1 - mZoomScrollLimit));
2453            return halfScale * part + noScale * (1.0f - part);
2454        }

2458                return halfScale;
2459            }
2460            float part = (height - mContentHeight * mZoomScrollLimit)  
2461                    / (height * (1 - mZoomScrollLimit));
2462            return halfScale * part + noScale * (1.0f - part);
2463        }

2469        private void scrollZoomDraw(Canvas canvas) {
2470            float invScale = mZoomScrollInvLimit; 
2471            int elapsed = 0;
2472            if (mTouchMode != SCROLL_ZOOM_OUT) {
2473                elapsed = (int) Math.min(System.currentTimeMillis() 
2474                    - mZoomScrollStart, SCROLL_ZOOM_DURATION);
2475                float transitionScale = (mZoomScrollInvLimit - mInvActualScale) 
2476                        * elapsed / SCROLL_ZOOM_DURATION;
2477                if (mTouchMode == SCROLL_ZOOM_ANIMATION_OUT) {
2478                    invScale = mInvActualScale + transitionScale;
2479                } else { /* if (mTouchMode == SCROLL_ZOOM_ANIMATION_IN) */
2480                    invScale = mZoomScrollInvLimit - transitionScale;
2481                }
2482            }
2483            float scale = scrollZoomGridScale(invScale);
2484            invScale = 1.0f / scale;
2485            int width = getViewWidth();
2486            int height = getViewHeight();
2487            float halfScale = scrollZoomMagScale(invScale);
2488            Rect scrollFrame = scrollZoomFrame(width, height, halfScale);
2489            if (elapsed == SCROLL_ZOOM_DURATION) {
2490                if (mTouchMode == SCROLL_ZOOM_ANIMATION_IN) {
2491                    setHorizontalScrollBarEnabled(true);
2492                    setVerticalScrollBarEnabled(true);
2493                    updateTextEntry();
2494                    scrollTo((int) (scrollFrame.centerX() * mActualScale) 
2495                            - (width >> 1), (int) (scrollFrame.centerY() 
2496                            * mActualScale) - (height >> 1));
2497                    mTouchMode = TOUCH_DONE_MODE;
2498                } else {
2499                    mTouchMode = SCROLL_ZOOM_OUT;
2500                }
2501            }
2502            float newX = scrollZoomX(scale);
2503            float newY = scrollZoomY(scale);
2504            if (LOGV_ENABLED) {
2505                Log.v(LOGTAG, "scrollZoomDraw scale=" + scale + " + (" + newX
2506                        + ", " + newY + ") mZoomScroll=(" + mZoomScrollX + ", "
2507                        + mZoomScrollY + ")" + " invScale=" + invScale + " scale=" 
2508                        + scale);
2509            }
2510            canvas.translate(newX, newY);
2511            canvas.scale(scale, scale);
2512            boolean animating = mTouchMode != SCROLL_ZOOM_OUT;
2513            if (mDrawHistory) {
2514                int sc = canvas.save(Canvas.CLIP_SAVE_FLAG);
2515                Rect clip = new Rect(0, 0, mHistoryPicture.getWidth(),
2516                        mHistoryPicture.getHeight());
2517                canvas.clipRect(clip, Region.Op.DIFFERENCE);
2518                canvas.drawColor(mBackgroundColor);
2519                canvas.restoreToCount(sc);
2520                canvas.drawPicture(mHistoryPicture);
2521            } else {
2522                mWebViewCore.drawContentPicture(canvas, mBackgroundColor,
2523                        animating, true);
2524            }
2525            if (mTouchMode == TOUCH_DONE_MODE) {
2526                return;
2527            }
2528            Paint paint = new Paint(Paint.ANTI_ALIAS_FLAG);
2529            paint.setStyle(Paint.Style.STROKE);
2530            paint.setStrokeWidth(30.0f);
2531            paint.setARGB(0x50, 0, 0, 0);
2532            int maxX = mContentWidth - width;
2533            int maxY = mContentHeight - height;
2534            if (true) { // experiment: draw hint to place finger off magnify area
2535                drawMagnifyFrame(canvas, scrollFrame, paint);
2536            } else {
2537                canvas.drawRect(scrollFrame, paint);
2538            }
2539            int sc = canvas.save();
2540            canvas.clipRect(scrollFrame);
2541            float halfX = (float) mZoomScrollX / maxX;
2542            if (mContentWidth * mZoomScrollLimit < width) {
2543                halfX = zoomFrameScaleX(width, 0.5f, halfX);
2544            }
2545            float halfY = (float) mZoomScrollY / maxY;
2546            if (mContentHeight * mZoomScrollLimit < height) {
2547                halfY = zoomFrameScaleY(height, 0.5f, halfY);
2548            }
2549            canvas.scale(halfScale, halfScale, mZoomScrollX + width * halfX
2550                    , mZoomScrollY + height * halfY);

2660        private void moveZoomScrollWindow(float x, float y) {
2661            if (Math.abs(x - mLastZoomScrollRawX) < 1.5f 
2662                    && Math.abs(y - mLastZoomScrollRawY) < 1.5f) {
2663                return;
2664            }
2665            mLastZoomScrollRawX = x;
2666            mLastZoomScrollRawY = y;
2667            int oldX = mZoomScrollX;
2668            int oldY = mZoomScrollY;
2669            int width = getViewWidth();
2670            int height = getViewHeight();
2671            int maxZoomX = mContentWidth - width;
2672            if (maxZoomX > 0) {
2673                int maxScreenX = width - (int) Math.ceil(width 
2674                        * mZoomScrollLimit) - SCROLL_ZOOM_FINGER_BUFFER;
2675                if (LOGV_ENABLED) {
2676                    Log.v(LOGTAG, "moveZoomScrollWindow-X" 
2677                            + " maxScreenX=" + maxScreenX + " width=" + width
2678                            + " mZoomScrollLimit=" + mZoomScrollLimit + " x=" + x); 
2679                }
2680                x += maxScreenX * mLastScrollX / maxZoomX - mLastTouchX;
2681                x *= Math.max(maxZoomX / maxScreenX, mZoomScrollInvLimit);
2682                mZoomScrollX = Math.max(0, Math.min(maxZoomX, (int) x));
2683            }

2685            if (maxZoomY > 0) {

2687                        * mZoomScrollLimit) - SCROLL_ZOOM_FINGER_BUFFER;
2688                if (LOGV_ENABLED) {
2689                    Log.v(LOGTAG, "moveZoomScrollWindow-Y" 
2690                            + " maxScreenY=" + maxScreenY + " height=" + height
2691                            + " mZoomScrollLimit=" + mZoomScrollLimit + " y=" + y); 
2692                }
2693                y += maxScreenY * mLastScrollY / maxZoomY - mLastTouchY;
2694                y *= Math.max(maxZoomY / maxScreenY, mZoomScrollInvLimit);
2695                mZoomScrollY = Math.max(0, Math.min(maxZoomY, (int) y));
2696            }
2697            if (oldX != mZoomScrollX || oldY != mZoomScrollY) {
2698                invalidate();
2699            }

2701                Log.v(LOGTAG, "moveZoomScrollWindow" 
2702                        + " scrollTo=(" + mZoomScrollX + ", " + mZoomScrollY + ")" 
2703                        + " mLastTouch=(" + mLastTouchX + ", " + mLastTouchY + ")" 
2704                        + " maxZoom=(" + maxZoomX + ", " + maxZoomY + ")" 
2705                        + " last=("+mLastScrollX+", "+mLastScrollY+")" 
2706                        + " x=" + x + " y=" + y);

2864        private void updateTextEntry() {

2866                mTextEntry = new TextDialog(mContext, WebView.this);
2867                // Initialize our generation number.
2868                mTextGeneration = 0;
2869            }
2870            // If we do not have focus, do nothing until we gain focus.
2871            if (!hasFocus() && !mTextEntry.hasFocus()
2872                    || (mTouchMode >= FIRST_SCROLL_ZOOM 
2873                    && mTouchMode <= LAST_SCROLL_ZOOM)) {
2874                mNeedsUpdateTextEntry = true;
2875                return;
2876            }
2877            boolean alreadyThere = inEditingMode();
2878            if (0 == mNativeClass || !nativeUpdateFocusNode()) {
2879                if (alreadyThere) {
2880                    mTextEntry.remove();
2881                }
2882                return;
2883            }
2884            FocusNode node = mFocusNode;
2885            if (!node.mIsTextField && !node.mIsTextArea) {
2886                if (alreadyThere) {
2887                    mTextEntry.remove();
2888                }
2889                return;
2890            }
2891            mTextEntry.setTextSize(contentToView(node.mTextSize));
2892            Rect visibleRect = sendOurVisibleRect();
2893            // Note that sendOurVisibleRect calls viewToContent, so the coordinates
2894            // should be in content coordinates.
2895            if (!Rect.intersects(node.mBounds, visibleRect)) {
2896                // Node is not on screen, so do not bother.
2897                return;
2898            }
2899            int x = node.mBounds.left;
2900            int y = node.mBounds.top;
2901            int width = node.mBounds.width();
2902            int height = node.mBounds.height();
2903            if (alreadyThere && mTextEntry.isSameTextField(node.mNodePointer)) {
2904                // It is possible that we have the same textfield, but it has moved,
2905                // i.e. In the case of opening/closing the screen.
2906                // In that case, we need to set the dimensions, but not the other
2907                // aspects.

2909                // calling setTextEntryRect.
2910                Spannable spannable = (Spannable) mTextEntry.getText();

2915                // there has been more UI text input, ignore it.  We will receive
2916                // another update when that text is recognized.
2917                if (node.mText != null && !node.mText.equals(spannable.toString())
2918                        && node.mRootTextGeneration == mTextGeneration) {
2919                    mTextEntry.setTextAndKeepSelection(node.mText);
2920                } else {
2921                    Selection.setSelection(spannable, start, end);
2922                }
2923            } else {
2924                String text = node.mText;
2925                setTextEntryRect(x, y, width, height);
2926                mTextEntry.setGravity(node.mIsRtlText ? Gravity.RIGHT : 
2927                        Gravity.NO_GRAVITY);
2928                // this needs to be called before update adapter thread starts to
2929                // ensure the mTextEntry has the same node pointer
2930                mTextEntry.setNodePointer(node.mNodePointer);
2931                int maxLength = -1;
2932                if (node.mIsTextField) {
2933                    maxLength = node.mMaxLength;
2934                    if (mWebViewCore.getSettings().getSaveFormData()
2935                            && node.mName != null) {
2936                        HashMap data = new HashMap();
2937                        data.put("text", node.mText);
2938                        Message update = mPrivateHandler.obtainMessage(
2939                                UPDATE_TEXT_ENTRY_ADAPTER, node.mNodePointer, 0,
2940                                data);
2941                        UpdateTextEntryAdapter updater = new UpdateTextEntryAdapter(
2942                                node.mName, getUrl(), update);

2944                        t.start();
2945                    }
2946                }
2947                mTextEntry.setMaxLength(maxLength);
2948                AutoCompleteAdapter adapter = null;
2949                mTextEntry.setAdapterCustom(adapter);
2950                mTextEntry.setSingleLine(node.mIsTextField);
2951                mTextEntry.setInPassword(node.mIsPassword);
2952                if (null == text) {
2953                    mTextEntry.setText("", 0, 0);
2954                } else {

2956                    // entering a textfield/textarea always set the selection to the
2957                    // whole field.  This was desirable for the case where the user
2958                    // intends to scroll past the field using the trackball.

2966                        int length = text.length();

3734                                // but if we tap, zoom in where we tap
3735                                if (eventTime - mLastTouchTime < TAP_TIMEOUT) {
3736                                    zoomScrollTap(x, y);
3737                                }
3738                                // start zooming in back to the original view
3739                                setZoomScrollIn();
3740                                mTouchMode = SCROLL_ZOOM_ANIMATION_IN;
3741                                invalidate();
3742                            }
3743                            break;
3744                        case TOUCH_DRAG_MODE:
3745                            // if the user waits a while w/o moving before the
3746                            // up, we don't want to do a fling
3747                            if (eventTime - mLastTouchTime <= MIN_FLING_TIME) {
3748                                mVelocityTracker.addMovement(ev);
3749                                doFling();
3750                                break;
3751                            }
3752                            WebViewCore.resumeUpdate(mWebViewCore);
3753                            break;
3754                        case TOUCH_DRAG_START_MODE:
3755                        case TOUCH_DONE_MODE:
3756                            // do nothing
3757                            break;
3758                    }
3759                    // we also use mVelocityTracker == null to tell us that we are
3760                    // not "moving around", so we can take the slower/prettier
3761                    // mode in the drawing code
3762                    if (mVelocityTracker != null) {
3763                        mVelocityTracker.recycle();
3764                        mVelocityTracker = null;
3765                    }
3766                    break;
3767                }
3768                case MotionEvent.ACTION_CANCEL: {
3769                    // we also use mVelocityTracker == null to tell us that we are
3770                    // not "moving around", so we can take the slower/prettier
3771                    // mode in the drawing code
3772                    if (mVelocityTracker != null) {
3773                        mVelocityTracker.recycle();
3774                        mVelocityTracker = null;
3775                    }
3776                    if (mTouchMode == SCROLL_ZOOM_OUT ||
3777                            mTouchMode == SCROLL_ZOOM_ANIMATION_IN) {
3778                        scrollTo(mZoomScrollX, mZoomScrollY);
3779                    } else if (mTouchMode == TOUCH_DRAG_MODE) {
3780                        WebViewCore.resumeUpdate(mWebViewCore);
3781                    }
3782                    mPrivateHandler.removeMessages(SWITCH_TO_SHORTPRESS);
3783                    mPrivateHandler.removeMessages(SWITCH_TO_LONGPRESS);
3784                    mTouchMode = TOUCH_DONE_MODE;
3785                    int contentX = viewToContent((int) mLastTouchX + mScrollX);
3786                    int contentY = viewToContent((int) mLastTouchY + mScrollY);
3787                    if (inEditingMode()) {
3788                        mTextEntry.updateCachedTextfield();
3789                    }
3790                    nativeClearFocus(contentX, contentY);
3791                    break;
3792                }
3793            }
3794            return true;
3795        }

3997        private void doTrackball(long time) {
3998            int elapsed = (int) (mTrackballLastTime - mTrackballFirstTime);
3999            if (elapsed == 0) {
4000                elapsed = TRACKBALL_TIMEOUT;
4001            }
4002            float xRate = mTrackballRemainsX * 1000 / elapsed; 
4003            float yRate = mTrackballRemainsY * 1000 / elapsed;
4004            if (mShiftIsPressed) {
4005                moveSelection(xRate, yRate);
4006                mTrackballRemainsX = mTrackballRemainsY = 0;
4007                return;
4008            }
4009            float ax = Math.abs(xRate);
4010            float ay = Math.abs(yRate);
4011            float maxA = Math.max(ax, ay);
4012            if (LOGV_ENABLED) {
4013                Log.v(LOGTAG, "doTrackball elapsed=" + elapsed
4014                        + " xRate=" + xRate
4015                        + " yRate=" + yRate
4016                        + " mTrackballRemainsX=" + mTrackballRemainsX
4017                        + " mTrackballRemainsY=" + mTrackballRemainsY);
4018            }
4019            int width = mContentWidth - getViewWidth();
4020            int height = mContentHeight - getViewHeight();
4021            if (width < 0) width = 0;
4022            if (height < 0) height = 0;
4023            if (mTouchMode == SCROLL_ZOOM_OUT) {
4024                int oldX = mZoomScrollX;
4025                int oldY = mZoomScrollY;
4026                int maxWH = Math.max(width, height);
4027                mZoomScrollX += scaleTrackballX(xRate, maxWH);
4028                mZoomScrollY += scaleTrackballY(yRate, maxWH);
4029                if (LOGV_ENABLED) {
4030                    Log.v(LOGTAG, "doTrackball SCROLL_ZOOM_OUT"
4031                            + " mZoomScrollX=" + mZoomScrollX 
4032                            + " mZoomScrollY=" + mZoomScrollY);
4033                }
4034                mZoomScrollX = Math.min(width, Math.max(0, mZoomScrollX));
4035                mZoomScrollY = Math.min(height, Math.max(0, mZoomScrollY));
4036                if (oldX != mZoomScrollX || oldY != mZoomScrollY) {
4037                    invalidate();
4038                }
4039                mTrackballRemainsX = mTrackballRemainsY = 0;
4040                return;
4041            }
4042            ax = Math.abs(mTrackballRemainsX * TRACKBALL_MULTIPLIER);
4043            ay = Math.abs(mTrackballRemainsY * TRACKBALL_MULTIPLIER);
4044            maxA = Math.max(ax, ay);
4045            int count = Math.max(0, (int) maxA);
4046            int oldScrollX = mScrollX;
4047            int oldScrollY = mScrollY;
4048            if (count > 0) {
4049                int selectKeyCode = ax < ay ? mTrackballRemainsY < 0 ? 
4050                        KeyEvent.KEYCODE_DPAD_UP : KeyEvent.KEYCODE_DPAD_DOWN : 
4051                        mTrackballRemainsX < 0 ? KeyEvent.KEYCODE_DPAD_LEFT :
4052                        KeyEvent.KEYCODE_DPAD_RIGHT;
4053                count = Math.min(count, TRACKBALL_MOVE_COUNT);
4054                if (LOGV_ENABLED) {
4055                    Log.v(LOGTAG, "doTrackball keyCode=" + selectKeyCode 
4056                            + " count=" + count
4057                            + " mTrackballRemainsX=" + mTrackballRemainsX
4058                            + " mTrackballRemainsY=" + mTrackballRemainsY);
4059                }
4060                if (navHandledKey(selectKeyCode, count, false, time)) {
4061                    playSoundEffect(keyCodeToSoundsEffect(selectKeyCode));
4062                }
4063                mTrackballRemainsX = mTrackballRemainsY = 0;
4064            }
4065            if (count >= TRACKBALL_SCROLL_COUNT) {
4066                int xMove = scaleTrackballX(xRate, width);
4067                int yMove = scaleTrackballY(yRate, height);
4068                if (LOGV_ENABLED) {
4069                    Log.v(LOGTAG, "doTrackball pinScrollBy"
4070                            + " count=" + count
4071                            + " xMove=" + xMove + " yMove=" + yMove
4072                            + " mScrollX-oldScrollX=" + (mScrollX-oldScrollX) 
4073                            + " mScrollY-oldScrollY=" + (mScrollY-oldScrollY) 
4074                            );
4075                }
4076                if (Math.abs(mScrollX - oldScrollX) > Math.abs(xMove)) {
4077                    xMove = 0;
4078                }
4079                if (Math.abs(mScrollY - oldScrollY) > Math.abs(yMove)) {
4080                    yMove = 0;
4081                }
4082                if (xMove != 0 || yMove != 0) {
4083                    pinScrollBy(xMove, yMove, true, 0);
4084                }
4085                mUserScroll = true;
4086            } 
4087            mWebViewCore.sendMessage(EventHub.UNBLOCK_FOCUS);        
4088        }

5046        // return true if the key was handled
5047        private boolean navHandledKey(int keyCode, int count, boolean noScroll
5048                , long time) {
5049            if (mNativeClass == 0) {
5050                return false;
5051            }
5052            mLastFocusTime = time;
5053            mLastFocusBounds = nativeGetFocusRingBounds();
5054            boolean keyHandled = nativeMoveFocus(keyCode, count, noScroll) == false;
5055            if (LOGV_ENABLED) {
5056                Log.v(LOGTAG, "navHandledKey mLastFocusBounds=" + mLastFocusBounds
5057                        + " mLastFocusTime=" + mLastFocusTime
5058                        + " handled=" + keyHandled);
5059            }
5060            if (keyHandled == false || mHeightCanMeasure == false) {
5061                return keyHandled;
5062            }
5063            Rect contentFocus = nativeGetFocusRingBounds();
5064            if (contentFocus.isEmpty()) return keyHandled;
5065            Rect viewFocus = contentToView(contentFocus);
5066            Rect visRect = new Rect();
5067            calcOurVisibleRect(visRect);
5068            Rect outset = new Rect(visRect);
5069            int maxXScroll = visRect.width() / 2;
5070            int maxYScroll = visRect.height() / 2;
5071            outset.inset(-maxXScroll, -maxYScroll);
5072            if (Rect.intersects(outset, viewFocus) == false) {
5073                return keyHandled;
5074            }
5075            // FIXME: Necessary because ScrollView/ListView do not scroll left/right
5076            int maxH = Math.min(viewFocus.right - visRect.right, maxXScroll);
5077            if (maxH > 0) {
5078                pinScrollBy(maxH, 0, true, 0);
5079            } else {
5080                maxH = Math.max(viewFocus.left - visRect.left, -maxXScroll);
5081                if (maxH < 0) {
5082                    pinScrollBy(maxH, 0, true, 0);
5083                }
5084            }
5085            if (mLastFocusBounds.isEmpty()) return keyHandled;
5086            if (mLastFocusBounds.equals(contentFocus)) return keyHandled;
5087            if (LOGV_ENABLED) {
5088                Log.v(LOGTAG, "navHandledKey contentFocus=" + contentFocus);
5089            }
5090            requestRectangleOnScreen(viewFocus);
5091            mUserScroll = true;
5092            return keyHandled;
5093        }
