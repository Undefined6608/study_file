022    /**
023     * Two-way map that maps MIME-types to file extensions and vice versa.
024     */

026    
027        /**
028         * Singleton MIME-type map instance:
029         */
030        private static MimeTypeMap sMimeTypeMap;
031    
032        /**
033         * MIME-type to file extension mapping:
034         */
035        private HashMap<String, String> mMimeTypeToExtensionMap;
036    
037        /**
038         * File extension to MIME type mapping:
039         */
040        private HashMap<String, String> mExtensionToMimeTypeMap;
041    
042    
043        /**
044         * Creates a new MIME-type map.
045         */
046        private MimeTypeMap() {
047            mMimeTypeToExtensionMap = new HashMap<String, String>();
048            mExtensionToMimeTypeMap = new HashMap<String, String>();
049        }
050    
051        /**
052         * Returns the file extension or an empty string iff there is no
053         * extension.
054         */
055        public static String getFileExtensionFromUrl(String url) {
056            if (url != null && url.length() > 0) {
057                int query = url.lastIndexOf('?');
058                if (query > 0) {
059                    url = url.substring(0, query);
060                }
061                int filenamePos = url.lastIndexOf('/');
062                String filename =
063                    0 <= filenamePos ? url.substring(filenamePos + 1) : url;
064    
065                // if the filename contains special characters, we don't
066                // consider it valid for our matching purposes:
067                if (filename.length() > 0 &&
068                    Pattern.matches("[a-zA-Z_0-9\\.\\-\\(\\)]+", filename)) {
069                    int dotPos = filename.lastIndexOf('.');
070                    if (0 <= dotPos) {
071                        return filename.substring(dotPos + 1);
072                    }
073                }
074            }
075    
076            return "";
077        }
078    
079        /**
080         * Load an entry into the map. This does not check if the item already
081         * exists, it trusts the caller!
082         */
083        private void loadEntry(String mimeType, String extension, 
084                boolean textType) {
085            //
086            // if we have an existing x --> y mapping, we do not want to
087            // override it with another mapping x --> ?
088            // this is mostly because of the way the mime-type map below
089            // is constructed (if a mime type maps to several extensions
090            // the first extension is considered the most popular and is
091            // added first; we do not want to overwrite it later).
092            //
093            if (!mMimeTypeToExtensionMap.containsKey(mimeType)) {
094                mMimeTypeToExtensionMap.put(mimeType, extension);
095            }
096    
097            //
098            // here, we don't want to map extensions to text MIME types;
099            // otherwise, we will start replacing generic text/plain and
100            // text/html with text MIME types that our platform does not
101            // understand.
102            //
103            if (!textType) {
104                mExtensionToMimeTypeMap.put(extension, mimeType);
105            }
106        }
107    
108        /**
109         * @return True iff there is a mimeType entry in the map.
110         */
111        public boolean hasMimeType(String mimeType) {
112            if (mimeType != null && mimeType.length() > 0) {
113                return mMimeTypeToExtensionMap.containsKey(mimeType);
114            }
115    
116            return false;
117        }
118    
119        /**
120         * @return The extension for the MIME type or null iff there is none.
121         */
122        public String getMimeTypeFromExtension(String extension) {
123            if (extension != null && extension.length() > 0) {
124                return mExtensionToMimeTypeMap.get(extension);
125            }
126    
127            return null;
128        }
129    
130        /**
131         * @return True iff there is an extension entry in the map.
132         */
133        public boolean hasExtension(String extension) {
134            if (extension != null && extension.length() > 0) {
135                return mExtensionToMimeTypeMap.containsKey(extension);
136            }
137    
138            return false;
139        }
140    
141        /**
142         * @return The MIME type for the extension or null iff there is none.
143         */
144        public String getExtensionFromMimeType(String mimeType) {
145            if (mimeType != null && mimeType.length() > 0) {
146                return mMimeTypeToExtensionMap.get(mimeType);
147            }
148    
149            return null;
150        }
