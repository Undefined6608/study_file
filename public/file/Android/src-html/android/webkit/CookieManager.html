109        /**
110         * Package level class to be accessed by cookie sync manager
111         */
112        static class Cookie {
113            static final byte MODE_NEW = 0;
114    
115            static final byte MODE_NORMAL = 1;
116    
117            static final byte MODE_DELETED = 2;
118    
119            static final byte MODE_REPLACED = 3;
120    
121            String domain;
122    
123            String path;
124    
125            String name;
126    
127            String value;
128    
129            long expires;
130    
131            long lastAcessTime;
132    
133            long lastUpdateTime;
134    
135            boolean secure;
136    
137            byte mode;
138    
139            Cookie() {
140            }
141    
142            Cookie(String defaultDomain, String defaultPath) {
143                domain = defaultDomain;
144                path = defaultPath;
145                expires = -1;
146            }
147    
148            boolean exactMatch(Cookie in) {
149                return domain.equals(in.domain) && path.equals(in.path) &&
150                        name.equals(in.name);
151            }
152    
153            boolean domainMatch(String urlHost) {
154                if (domain.startsWith(".")) {
155                    if (urlHost.endsWith(domain.substring(1))) {
156                        int len = domain.length();
157                        int urlLen = urlHost.length();
158                        if (urlLen > len - 1) {
159                            // make sure bar.com doesn't match .ar.com
160                            return urlHost.charAt(urlLen - len) == PERIOD;
161                        }
162                        return true;
163                    }
164                    return false;
165                } else {
166                    // exact match if domain is not leading w/ dot
167                    return urlHost.equals(domain);
168                }
169            }
170    
171            boolean pathMatch(String urlPath) {
172                if (urlPath.startsWith(path)) {
173                    int len = path.length();
174                    if (len == 0) {
175                        Log.w(LOGTAG, "Empty cookie path");
176                        return false;
177                    }
178                    int urlLen = urlPath.length();
179                    if (path.charAt(len-1) != PATH_DELIM && urlLen > len) {
180                        // make sure /wee doesn't match /we
181                        return urlPath.charAt(len) == PATH_DELIM;
182                    }
183                    return true;
184                }
185                return false;
186            }
187    
188            public String toString() {
189                return "domain: " + domain + "; path: " + path + "; name: " + name
190                        + "; value: " + value;
191            }
192        }

235         * not have expiration which implies it is session cookie.
236         * @param url The url which cookie is set for
237         * @param value The value for set-cookie: in http response header
238         */
239        public void setCookie(String url, String value) {
240            WebAddress uri;
241            try {
242                uri = new WebAddress(url);

244                Log.e(LOGTAG, "Bad address: " + url);
245                return;
246            }
247            setCookie(uri, value);

250        /**
251         * Set cookie for a given uri. The old cookie with same host/path/name will
252         * be removed. The new cookie will be added if it is not expired or it does
253         * not have expiration which implies it is session cookie.
254         * @param uri The uri which cookie is set for
255         * @param value The value for set-cookie: in http response header
256         * @hide - hide this because it takes in a parameter of type WebAddress,
257         * a system private class.
258         */
259        public synchronized void setCookie(WebAddress uri, String value) {
260            if (value != null && value.length() > MAX_COOKIE_LENGTH) {
261                return;
262            }
263            if (!mAcceptCookie || uri == null) {
264                return;
265            }
266            if (Config.LOGV) {
267                Log.v(LOGTAG, "setCookie: uri: " + uri + " value: " + value);
268            }
269    
270            String[] hostAndPath = getHostAndPath(uri);
271            if (hostAndPath == null) {
272                return;
273            }
274            
275            // For default path, when setting a cookie, the spec says:
276            //Path:   Defaults to the path of the request URL that generated the
277            // Set-Cookie response, up to, but not including, the
278            // right-most /.
279            if (hostAndPath[1].length() > 1) {
280                int index = hostAndPath[1].lastIndexOf(PATH_DELIM);
281                hostAndPath[1] = hostAndPath[1].substring(0, 
282                        index > 0 ? index : index + 1);
283            }
284    
285            ArrayList<Cookie> cookies = null;
286            try {
287                cookies = parseCookie(hostAndPath[0], hostAndPath[1], value);
288            } catch (RuntimeException ex) {
289                Log.e(LOGTAG, "parse cookie failed for: " + value);
290            }
291    
292            if (cookies == null || cookies.size() == 0) {
293                return;
294            }
295    
296            String baseDomain = getBaseDomain(hostAndPath[0]);
297            ArrayList<Cookie> cookieList = mCookieMap.get(baseDomain);
298            if (cookieList == null) {
299                cookieList = CookieSyncManager.getInstance()
300                        .getCookiesForDomain(baseDomain);
301                mCookieMap.put(baseDomain, cookieList);
302            }
303    
304            long now = System.currentTimeMillis();
305            int size = cookies.size();
306            for (int i = 0; i < size; i++) {
307                Cookie cookie = cookies.get(i);
308    
309                boolean done = false;
310                Iterator<Cookie> iter = cookieList.iterator();
311                while (iter.hasNext()) {
312                    Cookie cookieEntry = iter.next();
313                    if (cookie.exactMatch(cookieEntry)) {
314                        // expires == -1 means no expires defined. Otherwise
315                        // negative means far future
316                        if (cookie.expires < 0 || cookie.expires > now) {
317                            // secure cookies can't be overwritten by non-HTTPS url
318                            if (!cookieEntry.secure || HTTPS.equals(uri.mScheme)) {
319                                cookieEntry.value = cookie.value;
320                                cookieEntry.expires = cookie.expires;
321                                cookieEntry.secure = cookie.secure;
322                                cookieEntry.lastAcessTime = now;
323                                cookieEntry.lastUpdateTime = now;
324                                cookieEntry.mode = Cookie.MODE_REPLACED;
325                            }
326                        } else {
327                            cookieEntry.lastUpdateTime = now;
328                            cookieEntry.mode = Cookie.MODE_DELETED;
329                        }
330                        done = true;
331                        break;
332                    }
333                }
334    
335                // expires == -1 means no expires defined. Otherwise negative means
336                // far future
337                if (!done && (cookie.expires < 0 || cookie.expires > now)) {
338                    cookie.lastAcessTime = now;
339                    cookie.lastUpdateTime = now;
340                    cookie.mode = Cookie.MODE_NEW;
341                    if (cookieList.size() > MAX_COOKIE_COUNT_PER_BASE_DOMAIN) {
342                        Cookie toDelete = new Cookie();
343                        toDelete.lastAcessTime = now;
344                        Iterator<Cookie> iter2 = cookieList.iterator();
345                        while (iter2.hasNext()) {
346                            Cookie cookieEntry2 = iter2.next();
347                            if ((cookieEntry2.lastAcessTime < toDelete.lastAcessTime)
348                                    && cookieEntry2.mode != Cookie.MODE_DELETED) {
349                                toDelete = cookieEntry2;
350                            }
351                        }
352                        toDelete.mode = Cookie.MODE_DELETED;
353                    }
354                    cookieList.add(cookie);
355                }
356            }
357        }

360         * Get cookie(s) for a given url so that it can be set to "cookie:" in http
361         * request header.
362         * @param url The url needs cookie
363         * @return The cookies in the format of NAME=VALUE [; NAME=VALUE]
364         */
365        public String getCookie(String url) {
366            WebAddress uri;
367            try {
368                uri = new WebAddress(url);
369            } catch (ParseException ex) {
370                Log.e(LOGTAG, "Bad address: " + url);
371                return null;
372            }
373            return getCookie(uri);

376        /**
377         * Get cookie(s) for a given uri so that it can be set to "cookie:" in http
378         * request header.
379         * @param uri The uri needs cookie
380         * @return The cookies in the format of NAME=VALUE [; NAME=VALUE]
381         * @hide - hide this because it has a parameter of type WebAddress, which
382         * is a system private class.
383         */
384        public synchronized String getCookie(WebAddress uri) {
385            if (!mAcceptCookie || uri == null) {
386                return null;
387            }
388       
389            String[] hostAndPath = getHostAndPath(uri);
390            if (hostAndPath == null) {
391                return null;
392            }
393    
394            String baseDomain = getBaseDomain(hostAndPath[0]);
395            ArrayList<Cookie> cookieList = mCookieMap.get(baseDomain);
396            if (cookieList == null) {
397                cookieList = CookieSyncManager.getInstance()
398                        .getCookiesForDomain(baseDomain);
399                mCookieMap.put(baseDomain, cookieList);
400            }
401    
402            long now = System.currentTimeMillis();
403            boolean secure = HTTPS.equals(uri.mScheme);
404            Iterator<Cookie> iter = cookieList.iterator();
405            StringBuilder ret = new StringBuilder(256);
406    
407            while (iter.hasNext()) {
408                Cookie cookie = iter.next();
409                if (cookie.domainMatch(hostAndPath[0]) &&
410                        cookie.pathMatch(hostAndPath[1])
411                        // expires == -1 means no expires defined. Otherwise
412                        // negative means far future
413                        && (cookie.expires < 0 || cookie.expires > now)
414                        && (!cookie.secure || secure)
415                        && cookie.mode != Cookie.MODE_DELETED) {
416                    cookie.lastAcessTime = now;
417    
418                    if (ret.length() > 0) {
419                        ret.append(SEMICOLON);
420                        // according to RC2109, SEMICOLON is office separator,
421                        // but when log in yahoo.com, it needs WHITE_SPACE too.
422                        ret.append(WHITE_SPACE);
423                    }
424    
425                    ret.append(cookie.name);
426                    ret.append(EQUAL);
427                    ret.append(cookie.value);
428                }
429            }
430            if (ret.length() > 0) {
431                if (Config.LOGV) {
432                    Log.v(LOGTAG, "getCookie: uri: " + uri + " value: " + ret);
433                }
434                return ret.toString();
435            } else {
436                if (Config.LOGV) {
437                    Log.v(LOGTAG, "getCookie: uri: " + uri
438                            + " But can't find cookie.");
439                }
440                return null;
441            }
442        }

444        /**
445         * Remove all session cookies, which are cookies without expiration date
446         */
447        public void removeSessionCookie() {
448            final Runnable clearCache = new Runnable() {
449                public void run() {
450                    synchronized(CookieManager.this) {
451                        Collection<ArrayList<Cookie>> cookieList = mCookieMap.values();
452                        Iterator<ArrayList<Cookie>> listIter = cookieList.iterator();
453                        while (listIter.hasNext()) {
454                            ArrayList<Cookie> list = listIter.next();
455                            Iterator<Cookie> iter = list.iterator();
456                            while (iter.hasNext()) {
457                                Cookie cookie = iter.next();
458                                if (cookie.expires == -1) {
459                                    iter.remove();
460                                }
461                            }
462                        }
463                        CookieSyncManager.getInstance().clearSessionCookies();
464                    }
465                }
466            };
467            new Thread(clearCache).start();
468        }

501                        Collection<ArrayList<Cookie>> cookieList = mCookieMap.values();

503                        while (listIter.hasNext()) {

505                            Iterator<Cookie> iter = list.iterator();
506                            while (iter.hasNext()) {
507                                Cookie cookie = iter.next();
508                                // expires == -1 means no expires defined. Otherwise

513                            }
514                        }
515                        CookieSyncManager.getInstance().clearExpiredCookies(now);
516                    }
517                }
518            };
519            new Thread(clearCache).start();
520        }

552        synchronized void deleteACookie(Cookie cookie) {

554                String baseDomain = getBaseDomain(cookie.domain);
555                ArrayList<Cookie> cookieList = mCookieMap.get(baseDomain);
556                if (cookieList != null) {

558                    if (cookieList.isEmpty()) {
559                        mCookieMap.remove(baseDomain);
560                    }
561                }
562            }

575        /**
576         * Package level api, called from CookieSyncManager
577         *
578         * Delete the least recent used domains if the total cookie count in RAM
579         * exceeds the limit
580         * @return A list of cookies which are removed from RAM
581         */
582        synchronized ArrayList<Cookie> deleteLRUDomain() {
583            int count = 0;
584            int byteCount = 0;
585            int mapSize = mCookieMap.size();
586    
587            if (mapSize < MAX_RAM_DOMAIN_COUNT) {
588                Collection<ArrayList<Cookie>> cookieLists = mCookieMap.values();
589                Iterator<ArrayList<Cookie>> listIter = cookieLists.iterator();
590                while (listIter.hasNext() && count < MAX_RAM_COOKIES_COUNT) {
591                    ArrayList<Cookie> list = listIter.next();
592                    if (Config.DEBUG) {
593                        Iterator<Cookie> iter = list.iterator();
594                        while (iter.hasNext() && count < MAX_RAM_COOKIES_COUNT) {
595                            Cookie cookie = iter.next();
596                            // 14 is 3 * sizeof(long) + sizeof(boolean)
597                            // + sizeof(byte)
598                            byteCount += cookie.domain.length()
599                                    + cookie.path.length()
600                                    + cookie.name.length()
601                                    + cookie.value.length() + 14;
602                            count++;
603                        }
604                    } else {
605                        count += list.size();
606                    }
607                }
608            }
609    
610            ArrayList<Cookie> retlist = new ArrayList<Cookie>();
611            if (mapSize >= MAX_RAM_DOMAIN_COUNT || count >= MAX_RAM_COOKIES_COUNT) {
612                if (Config.DEBUG) {
613                    Log.v(LOGTAG, count + " cookies used " + byteCount
614                            + " bytes with " + mapSize + " domains");
615                }
616                Object[] domains = mCookieMap.keySet().toArray();
617                int toGo = mapSize / 10 + 1;
618                while (toGo-- > 0){
619                    String domain = domains[toGo].toString();
620                    if (Config.LOGV) {
621                        Log.v(LOGTAG, "delete domain: " + domain
622                                + " from RAM cache");
623                    }
624                    retlist.addAll(mCookieMap.get(domain));
625                    mCookieMap.remove(domain);
626                }
627            }
628            return retlist;
629        }

631        /**
632         * Extract the host and path out of a uri
633         * @param uri The given WebAddress
634         * @return The host and path in the format of String[], String[0] is host
635         *          which has at least two periods, String[1] is path which always
636         *          ended with "/"
637         */
638        private String[] getHostAndPath(WebAddress uri) {
639            if (uri.mHost != null && uri.mPath != null) {
640                String[] ret = new String[2];
641                ret[0] = uri.mHost;
642                ret[1] = uri.mPath;
643    
644                int index = ret[0].indexOf(PERIOD);
645                if (index == -1) {
646                    if (uri.mScheme.equalsIgnoreCase("file")) {
647                        // There is a potential bug where a local file path matches
648                        // another file in the local web server directory. Still
649                        // "localhost" is the best pseudo domain name.
650                        ret[0] = "localhost";
651                    } else if (!ret[0].equals("localhost")) {
652                        return null;
653                    }
654                } else if (index == ret[0].lastIndexOf(PERIOD)) {
655                    // cookie host must have at least two periods
656                    ret[0] = PERIOD + ret[0];
657                }
658    
659                if (ret[1].charAt(0) != PATH_DELIM) {
660                    return null;
661                }
662    
663                /*
664                 * find cookie path, e.g. for http://www.google.com, the path is "/"
665                 * for http://www.google.com/lab/, the path is "/lab"
666                 * for http://www.google.com/lab/foo, the path is "/lab/foo"
667                 * for http://www.google.com/lab?hl=en, the path is "/lab"
668                 * for http://www.google.com/lab.asp?hl=en, the path is "/lab.asp"
669                 * Note: the path from URI has at least one "/"
670                 * See:
671                 * http://www.unix.com.ua/rfc/rfc2109.html
672                 */
673                index = ret[1].indexOf(QUESTION_MARK);
674                if (index != -1) {
675                    ret[1] = ret[1].substring(0, index);
676                }
677                return ret;
678            } else
679                return null;
680        }

682        /**
683         * Get the base domain for a give host. E.g. mail.google.com will return
684         * google.com
685         * @param host The give host
686         * @return the base domain
687         */
688        private String getBaseDomain(String host) {
689            int startIndex = 0;
690            int nextIndex = host.indexOf(PERIOD);
691            int lastIndex = host.lastIndexOf(PERIOD);
692            while (nextIndex < lastIndex) {
693                startIndex = nextIndex + 1;
694                nextIndex = host.indexOf(PERIOD, startIndex);
695            }
696            if (startIndex > 0) {
697                return host.substring(startIndex);
698            } else {
699                return host;
700            }
701        }

703        /**
704         * parseCookie() parses the cookieString which is a comma-separated list of
705         * one or more cookies in the format of "NAME=VALUE; expires=DATE;
706         * path=PATH; domain=DOMAIN_NAME; secure httponly" to a list of Cookies.
707         * Here is a sample: IGDND=1, IGPC=ET=UB8TSNwtDmQ:AF=0; expires=Sun,
708         * 17-Jan-2038 19:14:07 GMT; path=/ig; domain=.google.com, =,
709         * PREF=ID=408909b1b304593d:TM=1156459854:LM=1156459854:S=V-vCAU6Sh-gobCfO;
710         * expires=Sun, 17-Jan-2038 19:14:07 GMT; path=/; domain=.google.com which
711         * contains 3 cookies IGDND, IGPC, PREF and an empty cookie
712         * @param host The default host
713         * @param path The default path
714         * @param cookieString The string coming from "Set-Cookie:"
715         * @return A list of Cookies
716         */
717        private ArrayList<Cookie> parseCookie(String host, String path,
718                String cookieString) {
719            ArrayList<Cookie> ret = new ArrayList<Cookie>();
720    
721            int index = 0;
722            int length = cookieString.length();
723            while (true) {
724                Cookie cookie = null;
725    
726                // done
727                if (index < 0 || index >= length) {
728                    break;
729                }
730    
731                // skip white space
732                if (cookieString.charAt(index) == WHITE_SPACE) {
733                    index++;
734                    continue;
735                }
736    
737                /*
738                 * get NAME=VALUE; pair. detecting the end of a pair is tricky, it
739                 * can be the end of a string, like "foo=bluh", it can be semicolon
740                 * like "foo=bluh;path=/"; or it can be enclosed by \", like
741                 * "foo=\"bluh bluh\";path=/"
742                 *
743                 * Note: in the case of "foo=bluh, bar=bluh;path=/", we interpret
744                 * it as one cookie instead of two cookies.
745                 */
746                int semicolonIndex = cookieString.indexOf(SEMICOLON, index);
747                int equalIndex = cookieString.indexOf(EQUAL, index);
748                if (equalIndex == -1) {
749                    // bad format, force return
750                    break;
751                }
752                if (semicolonIndex > -1 && semicolonIndex < equalIndex) {
753                    // empty cookie, like "; path=/", return
754                    break;
755                }
756                cookie = new Cookie(host, path);
757                cookie.name = cookieString.substring(index, equalIndex);
758                if (cookieString.charAt(equalIndex + 1) == QUOTATION) {
759                    index = cookieString.indexOf(QUOTATION, equalIndex + 2);
760                    if (index == -1) {
761                        // bad format, force return
762                        break;
763                    }
764                }
765                semicolonIndex = cookieString.indexOf(SEMICOLON, index);
766                if (semicolonIndex == -1) {
767                    semicolonIndex = length;
768                }
769                if (semicolonIndex - equalIndex > MAX_COOKIE_LENGTH) {
770                    // cookie is too big, trim it
771                    cookie.value = cookieString.substring(equalIndex + 1,
772                            equalIndex + MAX_COOKIE_LENGTH);
773                } else if (equalIndex + 1 == semicolonIndex
774                        || semicolonIndex < equalIndex) {
775                    // these are unusual case like foo=; and foo; path=/
776                    cookie.value = "";
777                } else {
778                    cookie.value = cookieString.substring(equalIndex + 1,
779                            semicolonIndex);
780                }
781                // get attributes
782                index = semicolonIndex;
783                while (true) {
784                    // done
785                    if (index < 0 || index >= length) {
786                        break;
787                    }
788    
789                    // skip white space and semicolon
790                    if (cookieString.charAt(index) == WHITE_SPACE
791                            || cookieString.charAt(index) == SEMICOLON) {
792                        index++;
793                        continue;
794                    }
795    
796                    // comma means next cookie
797                    if (cookieString.charAt(index) == COMMA) {
798                        index++;
799                        break;
800                    }
801    
802                    // "secure" is a known attribute doesn't use "=";
803                    // while sites like live.com uses "secure="
804                    if (length - index > SECURE_LENGTH
805                            && cookieString.substring(index, index + SECURE_LENGTH).
806                            equalsIgnoreCase(SECURE)) {
807                        index += SECURE_LENGTH;
808                        cookie.secure = true;
809                        if (cookieString.charAt(index) == EQUAL) index++;
810                        continue;
811                    }
812    
813                    // "httponly" is a known attribute doesn't use "=";
814                    // while sites like live.com uses "httponly="
815                    if (length - index > HTTP_ONLY_LENGTH
816                            && cookieString.substring(index,
817                                index + HTTP_ONLY_LENGTH).
818                            equalsIgnoreCase(HTTP_ONLY)) {
819                        index += HTTP_ONLY_LENGTH;
820                        if (cookieString.charAt(index) == EQUAL) index++;
821                        // FIXME: currently only parse the attribute
822                        continue;
823                    }
824                    equalIndex = cookieString.indexOf(EQUAL, index);
825                    if (equalIndex > 0) {
826                        String name = cookieString.substring(index, equalIndex)
827                                .toLowerCase();
828                        if (name.equals(EXPIRES)) {
829                            int comaIndex = cookieString.indexOf(COMMA, equalIndex);
830    
831                            // skip ',' in (Wdy, DD-Mon-YYYY HH:MM:SS GMT) or
832                            // (Weekday, DD-Mon-YY HH:MM:SS GMT) if it applies.
833                            // "Wednesday" is the longest Weekday which has length 9
834                            if ((comaIndex != -1) &&
835                                    (comaIndex - equalIndex <= 10)) {
836                                index = comaIndex + 1;
837                            }
838                        }
839                        semicolonIndex = cookieString.indexOf(SEMICOLON, index);
840                        int commaIndex = cookieString.indexOf(COMMA, index);
841                        if (semicolonIndex == -1 && commaIndex == -1) {
842                            index = length;
843                        } else if (semicolonIndex == -1) {
844                            index = commaIndex;
845                        } else if (commaIndex == -1) {
846                            index = semicolonIndex;
847                        } else {
848                            index = Math.min(semicolonIndex, commaIndex);
849                        }
850                        String value =
851                                cookieString.substring(equalIndex + 1, index);
852                        
853                        // Strip quotes if they exist
854                        if (value.length() > 2 && value.charAt(0) == QUOTATION) {
855                            int endQuote = value.indexOf(QUOTATION, 1);
856                            if (endQuote > 0) {
857                                value = value.substring(1, endQuote);
858                            }
859                        }
860                        if (name.equals(EXPIRES)) {
861                            try {
862                                cookie.expires = HttpDateTime.parse(value);
863                            } catch (IllegalArgumentException ex) {
864                                Log.e(LOGTAG,
865                                        "illegal format for expires: " + value);
866                            }
867                        } else if (name.equals(MAX_AGE)) {
868                            try {
869                                cookie.expires = System.currentTimeMillis() + 1000
870                                        * Long.parseLong(value);
871                            } catch (NumberFormatException ex) {
872                                Log.e(LOGTAG,
873                                        "illegal format for max-age: " + value);
874                            }
875                        } else if (name.equals(PATH)) {
876                            // only allow non-empty path value
877                            if (value.length() > 0) {
878                                cookie.path = value;
879                            }
880                        } else if (name.equals(DOMAIN)) {
881                            int lastPeriod = value.lastIndexOf(PERIOD);
882                            if (lastPeriod == 0) {
883                                // disallow cookies set for TLDs like [.com]
884                                cookie.domain = null;
885                                continue;
886                            }
887                            try {
888                                Integer.parseInt(value.substring(lastPeriod + 1));
889                                // no wildcard for ip address match
890                                if (!value.equals(host)) {
891                                    // no cross-site cookie
892                                    cookie.domain = null;
893                                }
894                                continue;
895                            } catch (NumberFormatException ex) {
896                                // ignore the exception, value is a host name
897                            }
898                            value = value.toLowerCase();
899                            if (value.charAt(0) != PERIOD) {
900                                // pre-pended dot to make it as a domain cookie
901                                value = PERIOD + value;
902                                lastPeriod++;
903                            }
904                            if (host.endsWith(value.substring(1))) {
905                                int len = value.length();
906                                int hostLen = host.length();
907                                if (hostLen > (len - 1)
908                                        && host.charAt(hostLen - len) != PERIOD) {
909                                    // make sure the bar.com doesn't match .ar.com
910                                    cookie.domain = null;
911                                    continue;
912                                }
913                                // disallow cookies set on ccTLDs like [.co.uk]
914                                if ((len == lastPeriod + 3)
915                                        && (len >= 6 && len <= 8)) {
916                                    String s = value.substring(1, lastPeriod);
917                                    if (Arrays.binarySearch(BAD_COUNTRY_2LDS, s) >= 0) {
918                                        cookie.domain = null;
919                                        continue;
920                                    }
921                                }
922                                cookie.domain = value;
923                            } else {
924                                // no cross-site or more specific sub-domain cookie
925                                cookie.domain = null;
926                            }
927                        }
928                    } else {
929                        // bad format, force return
930                        index = length;
931                    }
932                }
933                if (cookie != null && cookie.domain != null) {
934                    ret.add(cookie);
935                }
936            }
937            return ret;
938        }
939    }
