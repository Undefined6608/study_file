017    package android.webkit;
018    
019    import android.graphics.Bitmap;
020    import android.net.http.SslError;
021    import android.os.Message;
022    import android.view.KeyEvent;
023    
024    public class WebViewClient {
025    
026        /**
027         * Give the host application a chance to take over the control when a new
028         * url is about to be loaded in the current WebView. If WebViewClient is not
029         * provided, by default WebView will ask Activity Manager to choose the
030         * proper handler for the url. If WebViewClient is provided, return true
031         * means the host application handles the url, while return false means the
032         * current WebView handles the url.
033         * 
034         * @param view The WebView that is initiating the callback.
035         * @param url The url to be loaded.
036         * @return True if the host application wants to leave the current WebView
037         *         and handle the url itself, otherwise return false.
038         */
039        public boolean shouldOverrideUrlLoading(WebView view, String url) {
040            return false;
041        }
042    
043        /**
044         * Notify the host application that a page has started loading. This method
045         * is called once for each main frame load so a page with iframes or
046         * framesets will call onPageStarted one time for the main frame. This also
047         * means that onPageStarted will not be called when the contents of an
048         * embedded frame changes, i.e. clicking a link whose target is an iframe.
049         * 
050         * @param view The WebView that is initiating the callback.
051         * @param url The url to be loaded.
052         * @param favicon The favicon for this page if it already exists in the
053         *            database.
054         */
055        public void onPageStarted(WebView view, String url, Bitmap favicon) {
056        }
057    
058        /**
059         * Notify the host application that a page has finished loading. This method
060         * is called only for main frame. When onPageFinished() is called, the
061         * rendering picture may not be updated yet. To get the notification for the
062         * new Picture, use {@link WebView.PictureListener#onNewPicture}.
063         * 
064         * @param view The WebView that is initiating the callback.
065         * @param url The url of the page.
066         */
067        public void onPageFinished(WebView view, String url) {
068        }
069    
070        /**
071         * Notify the host application that the WebView will load the resource
072         * specified by the given url.
073         * 
074         * @param view The WebView that is initiating the callback.
075         * @param url The url of the resource the WebView will load.
076         */
077        public void onLoadResource(WebView view, String url) {
078        }
079    
080        /**
081         * Notify the host application that there have been an excessive number of
082         * HTTP redirects. As the host application if it would like to continue
083         * trying to load the resource. The default behavior is to send the cancel
084         * message.
085         * 
086         * @param view The WebView that is initiating the callback.
087         * @param cancelMsg The message to send if the host wants to cancel
088         * @param continueMsg The message to send if the host wants to continue
089         */
090        public void onTooManyRedirects(WebView view, Message cancelMsg,
091                Message continueMsg) {
092            cancelMsg.sendToTarget();
093        }
094    
095        /**
096         * Report an error to an activity. These errors come up from WebCore, and
097         * are network errors.
098         * 
099         * @param view The WebView that is initiating the callback.
100         * @param errorCode The HTTP error code.
101         * @param description A String description.
102         * @param failingUrl The url that failed.
103         */
104        public void onReceivedError(WebView view, int errorCode,
105                String description, String failingUrl) {
106        }
107    
108        /**
109         * As the host application if the browser should resend data as the
110         * requested page was a result of a POST. The default is to not resend the
111         * data.
112         * 
113         * @param view The WebView that is initiating the callback.
114         * @param dontResend The message to send if the browser should not resend
115         * @param resend The message to send if the browser should resend data
116         */
117        public void onFormResubmission(WebView view, Message dontResend,
118                Message resend) {
119            dontResend.sendToTarget();
120        }
121    
122        /**
123         * Notify the host application to update its visited links database.
124         * 
125         * @param view The WebView that is initiating the callback.
126         * @param url The url being visited.
127         * @param isReload True if this url is being reloaded.
128         */
129        public void doUpdateVisitedHistory(WebView view, String url,
130                boolean isReload) {
131        }
132    
133        /**
134         * Notify the host application to handle a ssl certificate error request
135         * (display the error to the user and ask whether to proceed or not). The
136         * host application has to call either handler.cancel() or handler.proceed()
137         * as the connection is suspended and waiting for the response. The default
138         * behavior is to cancel the load.
139         * 
140         * @param view The WebView that is initiating the callback.
141         * @param handler An SslErrorHandler object that will handle the user's
142         *            response.
143         * @param error The SSL error object.
144         * @hide - hide this because it contains a parameter of type SslError,
145         * which is located in a hidden package.
146         */
147        public void onReceivedSslError(WebView view, SslErrorHandler handler,
148                SslError error) {
149            handler.cancel();
150        }
151    
152        /**
153         * Notify the host application to handle an authentication request. The
154         * default behavior is to cancel the request.
155         * 
156         * @param view The WebView that is initiating the callback.
157         * @param handler The HttpAuthHandler that will handle the user's response.
158         * @param host The host requiring authentication.
159         * @param realm A description to help store user credentials for future
160         *            visits.
161         */
162        public void onReceivedHttpAuthRequest(WebView view,
163                HttpAuthHandler handler, String host, String realm) {
164            handler.cancel();
165        }
166    
167        /**
168         * Give the host application a chance to handle the key event synchronously.
169         * e.g. menu shortcut key events need to be filtered this way. If return
170         * true, WebView will not handle the key event. If return false, WebView
171         * will always handle the key event, so none of the super in the view chain
172         * will see the key event. The default behavior returns false.
173         * 
174         * @param view The WebView that is initiating the callback.
175         * @param event The key event.
176         * @return True if the host application wants to handle the key event
177         *         itself, otherwise return false
178         */
179        public boolean shouldOverrideKeyEvent(WebView view, KeyEvent event) {
180            return false;
181        }
182    
183        /**
184         * Notify the host application that a key was not handled by the WebView.
185         * Except system keys, WebView always consumes the keys in the normal flow
186         * or if shouldOverrideKeyEvent returns true. This is called asynchronously
187         * from where the key is dispatched. It gives the host application an chance
188         * to handle the unhandled key events.
189         * 
190         * @param view The WebView that is initiating the callback.
191         * @param event The key event.
192         */
193        public void onUnhandledKeyEvent(WebView view, KeyEvent event) {
194        }
195    
196        /**
197         * Notify the host application that the scale applied to the WebView has
198         * changed.
199         * 
200         * @param view he WebView that is initiating the callback.
201         * @param oldScale The old scale factor
202         * @param newScale The new scale factor
203         */
204        public void onScaleChanged(WebView view, float oldScale, float newScale) {
205        }
206    }
