017    package android.webkit;
018    
019    import android.util.Config;
020    import java.io.Serializable;
021    import java.util.ArrayList;
022    
023    /**
024     * This class contains the back/forward list for a WebView.
025     * WebView.copyBackForwardList() will return a copy of this class used to
026     * inspect the entries in the list.
027     */
028    public class WebBackForwardList implements Cloneable, Serializable {
029        // Current position in the list.
030        private int mCurrentIndex;
031        // ArrayList of WebHistoryItems for maintaining our copy.
032        private ArrayList<WebHistoryItem> mArray;
033        // Flag to indicate that the list is invalid
034        private boolean mClearPending;
035    
036        /**
037         * Construct a back/forward list used by clients of WebView.
038         */
039        /*package*/ WebBackForwardList() {
040            mCurrentIndex = -1;
041            mArray = new ArrayList<WebHistoryItem>();
042        }
043    
044        /**
045         * Return the current history item. This method returns null if the list is
046         * empty.
047         * @return The current history item.
048         */
049        public synchronized WebHistoryItem getCurrentItem() {
050            return getItemAtIndex(mCurrentIndex);
051        }
052    
053        /**
054         * Get the index of the current history item. This index can be used to
055         * directly index into the array list.
056         * @return The current index from 0...n or -1 if the list is empty.
057         */
058        public synchronized int getCurrentIndex() {
059            return mCurrentIndex;
060        }
061    
062        /**
063         * Get the history item at the given index. The index range is from 0...n
064         * where 0 is the first item and n is the last item.
065         * @param index The index to retrieve.
066         */
067        public synchronized WebHistoryItem getItemAtIndex(int index) {
068            if (index < 0 || index >= getSize()) {
069                return null;
070            }
071            return mArray.get(index);
072        }
073    
074        /**
075         * Get the total size of the back/forward list.
076         * @return The size of the list.
077         */
078        public synchronized int getSize() {
079            return mArray.size();
080        }
081    
082        /**
083         * Mark the back/forward list as having a pending clear. This is used on the
084         * UI side to mark the list as being invalid during the clearHistory method.
085         */
086        /*package*/ synchronized void setClearPending() {
087            mClearPending = true;
088        }
089    
090        /**
091         * Return the status of the clear flag. This is used on the UI side to
092         * determine if the list is valid for checking things like canGoBack.
093         */
094        /*package*/ synchronized boolean getClearPending() {
095            return mClearPending;
096        }
097    
098        /**
099         * Add a new history item to the list. This will remove all items after the
100         * current item and append the new item to the end of the list. Called from
101         * the WebCore thread only. Synchronized because the UI thread may be
102         * reading the array or the current index.
103         * @param item A new history item.
104         */
105        /*package*/ synchronized void addHistoryItem(WebHistoryItem item) {
106            // Update the current position because we are going to add the new item
107            // in that slot.
108            ++mCurrentIndex;
109            // If the current position is not at the end, remove all history items
110            // after the current item.
111            final int size = mArray.size();
112            final int newPos = mCurrentIndex;
113            if (newPos != size) {
114                for (int i = size - 1; i >= newPos; i--) {
115                    final WebHistoryItem h = mArray.remove(i);
116                }
117            }
118            // Add the item to the list.
119            mArray.add(item);
120        }
121    
122        /**
123         * Clear the back/forward list. Called from the WebCore thread.
124         */
125        /*package*/ synchronized void close(int nativeFrame) {
126            // Clear the array first because nativeClose will call addHistoryItem
127            // with the current item.
128            mArray.clear();
129            mCurrentIndex = -1;
130            nativeClose(nativeFrame);
131            // Reset the clear flag
132            mClearPending = false;
133        }
134    
135        /* Remove the item at the given index. Called by JNI only. */
136        private synchronized void removeHistoryItem(int index) {
137            // XXX: This is a special case. Since the callback is only triggered
138            // when removing the first item, we can assert that the index is 0.
139            // This lets us change the current index without having to query the
140            // native BackForwardList.
141            if (Config.DEBUG && (index != 0)) {
142                throw new AssertionError();
143            }
144            final WebHistoryItem h = mArray.remove(index);
145            // XXX: If we ever add another callback for removing history items at
146            // any index, this will no longer be valid.
147            mCurrentIndex--;
148        }
149    
150        /**
151         * Clone the entire object to be used in the UI thread by clients of
152         * WebView. This creates a copy that should never be modified by any of the
153         * webkit package classes.
154         */
155        protected synchronized WebBackForwardList clone() {
156            WebBackForwardList l = new WebBackForwardList();
157            if (mClearPending) {
158                // If a clear is pending, return a copy with only the current item.
159                l.addHistoryItem(getCurrentItem());
160                return l;
161            }
162            l.mCurrentIndex = mCurrentIndex;
163            int size = getSize();
164            l.mArray = new ArrayList<WebHistoryItem>(size);
165            for (int i = 0; i < size; i++) {
166                // Add a copy of each WebHistoryItem
167                l.mArray.add(mArray.get(i).clone());
168            }
169            return l;
170        }
171    
172        /**
173         * Set the new history index.
174         * @param newIndex The new history index.
175         */
176        /*package*/ synchronized void setCurrentIndex(int newIndex) {
177            mCurrentIndex = newIndex;
178        }
179    
180        /**
181         * Restore the history index.
182         */
183        /*package*/ static native synchronized void restoreIndex(int nativeFrame,
184                int index);
185    
186        /* Close the native list. */
187        private static native void nativeClose(int nativeFrame);
188    }
