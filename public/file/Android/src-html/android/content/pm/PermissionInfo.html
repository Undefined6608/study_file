017    package android.content.pm;
018    
019    import android.os.Parcel;
020    import android.os.Parcelable;
021    import android.text.TextUtils;
022    
023    /**
024     * Information you can retrieve about a particular security permission
025     * known to the system.  This corresponds to information collected from the
026     * AndroidManifest.xml's &lt;permission&gt; tags.
027     */
028    public class PermissionInfo extends PackageItemInfo implements Parcelable {
029        /**
030         * A normal application value for {@link #protectionLevel}, corresponding
031         * to the <code>normal</code> value of
032         * {@link android.R.attr#protectionLevel}.
033         */
034        public static final int PROTECTION_NORMAL = 0;
035    
036        /**
037         * Dangerous value for {@link #protectionLevel}, corresponding
038         * to the <code>dangerous</code> value of
039         * {@link android.R.attr#protectionLevel}.
040         */
041        public static final int PROTECTION_DANGEROUS = 1;
042    
043        /**
044         * System-level value for {@link #protectionLevel}, corresponding
045         * to the <code>signature</code> value of
046         * {@link android.R.attr#protectionLevel}.
047         */
048        public static final int PROTECTION_SIGNATURE = 2;
049    
050        /**
051         * System-level value for {@link #protectionLevel}, corresponding
052         * to the <code>signatureOrSystem</code> value of
053         * {@link android.R.attr#protectionLevel}.
054         */
055        public static final int PROTECTION_SIGNATURE_OR_SYSTEM = 3;
056    
057        /**
058         * The group this permission is a part of, as per
059         * {@link android.R.attr#permissionGroup}.
060         */
061        public String group;
062        
063        /**
064         * A string resource identifier (in the package's resources) of this
065         * permission's description.  From the "description" attribute or,
066         * if not set, 0.
067         */
068        public int descriptionRes;
069    
070        /**
071         * The description string provided in the AndroidManifest file, if any.  You
072         * probably don't want to use this, since it will be null if the description
073         * is in a resource.  You probably want
074         * {@link PermissionInfo#loadDescription} instead.
075         */
076        public CharSequence nonLocalizedDescription;
077    
078        /**
079         * The level of access this permission is protecting, as per
080         * {@link android.R.attr#protectionLevel}.  Values may be
081         * {@link #PROTECTION_NORMAL}, {@link #PROTECTION_DANGEROUS}, or
082         * {@link #PROTECTION_SIGNATURE}.
083         */
084        public int protectionLevel;
085    
086        public PermissionInfo() {
087        }
088    
089        public PermissionInfo(PermissionInfo orig) {
090            super(orig);
091            group = orig.group;
092            descriptionRes = orig.descriptionRes;
093            protectionLevel = orig.protectionLevel;
094            nonLocalizedDescription = orig.nonLocalizedDescription;
095        }
096    
097        /**
098         * Retrieve the textual description of this permission.  This
099         * will call back on the given PackageManager to load the description from
100         * the application.
101         *
102         * @param pm A PackageManager from which the label can be loaded; usually
103         * the PackageManager from which you originally retrieved this item.
104         *
105         * @return Returns a CharSequence containing the permission's description.
106         * If there is no description, null is returned.
107         */
108        public CharSequence loadDescription(PackageManager pm) {
109            if (nonLocalizedDescription != null) {
110                return nonLocalizedDescription;
111            }
112            if (descriptionRes != 0) {
113                CharSequence label = pm.getText(packageName, descriptionRes, null);
114                if (label != null) {
115                    return label;
116                }
117            }
118            return null;
119        }
120    
121        public String toString() {
122            return "PermissionInfo{"
123                + Integer.toHexString(System.identityHashCode(this))
124                + " " + name + "}";
125        }
126    
127        public int describeContents() {
128            return 0;
129        }
130    
131        public void writeToParcel(Parcel dest, int parcelableFlags) {
132            super.writeToParcel(dest, parcelableFlags);
133            dest.writeString(group);
134            dest.writeInt(descriptionRes);
135            dest.writeInt(protectionLevel);
136            TextUtils.writeToParcel(nonLocalizedDescription, dest, parcelableFlags);
137        }
138    
139        public static final Creator<PermissionInfo> CREATOR =
140            new Creator<PermissionInfo>() {
141            public PermissionInfo createFromParcel(Parcel source) {
142                return new PermissionInfo(source);
143            }
144            public PermissionInfo[] newArray(int size) {
145                return new PermissionInfo[size];
146            }
147        };
148    
149        private PermissionInfo(Parcel source) {
150            super(source);
151            group = source.readString();
152            descriptionRes = source.readInt();
153            protectionLevel = source.readInt();
154            nonLocalizedDescription = TextUtils.CHAR_SEQUENCE_CREATOR.createFromParcel(source);
155        }
156    }
