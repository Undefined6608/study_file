001    package android.content.pm;
002    
003    import android.graphics.drawable.Drawable;
004    import android.os.Parcel;
005    import android.util.Printer;
006    
007    /**
008     * Base class containing information common to all application components
009     * ({@link ActivityInfo}, {@link ServiceInfo}).  This class is not intended
010     * to be used by itself; it is simply here to share common definitions
011     * between all application components.  As such, it does not itself
012     * implement Parcelable, but does provide convenience methods to assist
013     * in the implementation of Parcelable in subclasses.
014     */
015    public class ComponentInfo extends PackageItemInfo {
016        /**
017         * Global information about the application/package this component is a
018         * part of.
019         */
020        public ApplicationInfo applicationInfo;
021        
022        /**
023         * The name of the process this component should run in.
024         * From the "android:process" attribute or, if not set, the same
025         * as <var>applicationInfo.processName</var>.
026         */
027        public String processName;
028    
029        /**
030         * Indicates whether or not this component may be instantiated.  Note that this value can be
031         * overriden by the one in its parent {@link ApplicationInfo}.
032         */
033        public boolean enabled = true;
034    
035        /**
036         * Set to true if this component is available for use by other applications.
037         * Comes from {@link android.R.attr#exported android:exported} of the
038         * &lt;activity&gt;, &lt;receiver&gt;, &lt;service&gt;, or
039         * &lt;provider&gt; tag.
040         */
041        public boolean exported = false;
042        
043        public ComponentInfo() {
044        }
045    
046        public ComponentInfo(ComponentInfo orig) {
047            super(orig);
048            applicationInfo = orig.applicationInfo;
049            processName = orig.processName;
050            enabled = orig.enabled;
051            exported = orig.exported;
052        }
053    
054        @Override public CharSequence loadLabel(PackageManager pm) {
055            if (nonLocalizedLabel != null) {
056                return nonLocalizedLabel;
057            }
058            ApplicationInfo ai = applicationInfo;
059            CharSequence label;
060            if (labelRes != 0) {
061                label = pm.getText(packageName, labelRes, ai);
062                if (label != null) {
063                    return label;
064                }
065            }
066            if (ai.nonLocalizedLabel != null) {
067                return ai.nonLocalizedLabel;
068            }
069            if (ai.labelRes != 0) {
070                label = pm.getText(packageName, ai.labelRes, ai);
071                if (label != null) {
072                    return label;
073                }
074            }
075            return name;
076        }
077        
078        @Override public Drawable loadIcon(PackageManager pm) {
079            ApplicationInfo ai = applicationInfo;
080            Drawable dr;
081            if (icon != 0) {
082                dr = pm.getDrawable(packageName, icon, ai);
083                if (dr != null) {
084                    return dr;
085                }
086            }
087            if (ai.icon != 0) {
088                dr = pm.getDrawable(packageName, ai.icon, ai);
089                if (dr != null) {
090                    return dr;
091                }
092            }
093            return pm.getDefaultActivityIcon();
094        }
095        
096        /**
097         * Return the icon resource identifier to use for this component.  If
098         * the component defines an icon, that is used; else, the application
099         * icon is used.
100         * 
101         * @return The icon associated with this component.
102         */
103        public final int getIconResource() {
104            return icon != 0 ? icon : applicationInfo.icon;
105        }
106        
107        protected void dumpFront(Printer pw, String prefix) {
108            super.dumpFront(pw, prefix);
109            pw.println(prefix + "enabled=" + enabled + " exported=" + exported
110                    + " processName=" + processName);
111        }
112        
113        protected void dumpBack(Printer pw, String prefix) {
114            if (applicationInfo != null) {
115                pw.println(prefix + "ApplicationInfo:");
116                applicationInfo.dump(pw, prefix + "  ");
117            } else {
118                pw.println(prefix + "ApplicationInfo: null");
119            }
120            super.dumpBack(pw, prefix);
121        }
122        
123        public void writeToParcel(Parcel dest, int parcelableFlags) {
124            super.writeToParcel(dest, parcelableFlags);
125            applicationInfo.writeToParcel(dest, parcelableFlags);
126            dest.writeString(processName);
127            dest.writeInt(enabled ? 1 : 0);
128            dest.writeInt(exported ? 1 : 0);
129        }
130    
131        protected ComponentInfo(Parcel source) {
132            super(source);
133            applicationInfo = ApplicationInfo.CREATOR.createFromParcel(source);
134            processName = source.readString();
135            enabled = (source.readInt() != 0);
136            exported = (source.readInt() != 0);
137        }
138    }
