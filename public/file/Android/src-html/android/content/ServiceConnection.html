017    package android.content;
018    
019    import android.os.IBinder;
020    
021    /**
022     * Interface for monitoring the state of an application service.  See
023     * {@link android.app.Service} and
024     * {@link Context#bindService Context.bindService()} for more information.
025     * <p>Like many callbacks from the system, the methods on this class are called
026     * from the main thread of your process.
027     */
028    public interface ServiceConnection {
029        /**
030         * Called when a connection to the Service has been established, with
031         * the {@link android.os.IBinder} of the communication channel to the
032         * Service.
033         *
034         * @param name The concrete component name of the service that has
035         * been connected.
036         *
037         * @param service The IBinder of the Service's communication channel,
038         * which you can now make calls on.
039         */
040        public void onServiceConnected(ComponentName name, IBinder service);
041    
042        /**
043         * Called when a connection to the Service has been lost.  This typically
044         * happens when the process hosting the service has crashed or been killed.
045         * This does <em>not</em> remove the ServiceConnection itself -- this
046         * binding to the service will remain active, and you will receive a call
047         * to {@link #onServiceConnected} when the Service is next running.
048         *
049         * @param name The concrete component name of the service whose
050         * connection has been lost.
051         */
052        public void onServiceDisconnected(ComponentName name);
053    }
