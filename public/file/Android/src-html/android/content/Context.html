040    /**
041     * Interface to global information about an application environment.  This is
042     * an abstract class whose implementation is provided by
043     * the Android system.  It
044     * allows access to application-specific resources and classes, as well as
045     * up-calls for application-level operations such as launching activities,
046     * broadcasting and receiving intents, etc.
047     */
048    public abstract class Context {
049        /**
050         * File creation mode: the default mode, where the created file can only
051         * be accessed by the calling application (or all applications sharing the
052         * same user ID).
053         * @see #MODE_WORLD_READABLE
054         * @see #MODE_WORLD_WRITEABLE
055         */
056        public static final int MODE_PRIVATE = 0x0000;
057        /**
058         * File creation mode: allow all other applications to have read access
059         * to the created file.
060         * @see #MODE_PRIVATE
061         * @see #MODE_WORLD_WRITEABLE
062         */
063        public static final int MODE_WORLD_READABLE = 0x0001;
064        /**
065         * File creation mode: allow all other applications to have write access
066         * to the created file.
067         * @see #MODE_PRIVATE
068         * @see #MODE_WORLD_READABLE
069         */
070        public static final int MODE_WORLD_WRITEABLE = 0x0002;
071        /**
072         * File creation mode: for use with {@link #openFileOutput}, if the file
073         * already exists then write data to the end of the existing file
074         * instead of erasing it.
075         * @see #openFileOutput
076         */
077        public static final int MODE_APPEND = 0x8000;

079        /**
080         * Flag for {@link #bindService}: automatically create the service as long
081         * as the binding exists.  Note that while this will create the service,
082         * its {@link android.app.Service#onStart} method will still only be called due to an
083         * explicit call to {@link #startService}.  Even without that, though,
084         * this still provides you with access to the service object while the
085         * service is created.
086         *
087         * <p>Specifying this flag also tells the system to treat the service
088         * as being as important as your own process -- that is, when deciding
089         * which process should be killed to free memory, the service will only
090         * be considered a candidate as long as the processes of any such bindings
091         * is also a candidate to be killed.  This is to avoid situations where
092         * the service is being continually created and killed due to low memory.
093         */
094        public static final int BIND_AUTO_CREATE = 0x0001;

100         * incorrect unbind call is made.  Note that doing this requires retaining
101         * information about the binding that was made for the lifetime of the app,
102         * resulting in a leak -- this should only be used for debugging.

671         * asynchronous; it will return before
672         * resultReceiver.onReceive() is called.
673         *
674         * <p>See {@link BroadcastReceiver} for more information on Intent broadcasts.
675         *
676         * @param intent The Intent to broadcast; all receivers matching this
677         *               Intent will receive the broadcast.
