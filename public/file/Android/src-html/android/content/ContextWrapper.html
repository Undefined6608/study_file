038    /**
039     * Proxying implementation of Context that simply delegates all of its calls to
040     * another Context.  Can be subclassed to modify behavior without changing
041     * the original Context.
042     */
043    public class ContextWrapper extends Context {
044        Context mBase;
045    
046        public ContextWrapper(Context base) {
047            mBase = base;
048        }
049        
050        /**
051         * Set the base context for this ContextWrapper.  All calls will then be
052         * delegated to the base context.  Throws
053         * IllegalStateException if a base context has already been set.
054         * 
055         * @param base The new base context for this wrapper.
056         */
057        protected void attachBaseContext(Context base) {
058            if (mBase != null) {
059                throw new IllegalStateException("Base context already set");
060            }
061            mBase = base;
062        }
063    
064        /**
065         * @return the base context as set by the constructor or setBaseContext
066         */
067        public Context getBaseContext() {
068            return mBase;
069        }
