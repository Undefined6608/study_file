028    /**
029     * This class is used to store a set of values that the {@link ContentResolver}
030     * can process.
031     */
032    public final class ContentValues implements Parcelable {
033        public static final String TAG = "ContentValues";
034    
035        /** Holds the actual values */
036        private HashMap<String, Object> mValues;
037    
038        /**
039         * Creates an empty set of values using the default initial size
040         */
041        public ContentValues() {
042            // Choosing a default size of 8 based on analysis of typical 
043            // consumption by applications.
044            mValues = new HashMap<String, Object>(8);
045        }
046    
047        /**
048         * Creates an empty set of values using the given initial size
049         * 
050         * @param size the initial size of the set of values
051         */
052        public ContentValues(int size) {
053            mValues = new HashMap<String, Object>(size, 1.0f);
054        }
055    
056        /**
057         * Creates a set of values copied from the given set
058         * 
059         * @param from the values to copy
060         */
061        public ContentValues(ContentValues from) {
062            mValues = new HashMap<String, Object>(from.mValues);
063        }
064    
065        /**
066         * Creates a set of values copied from the given HashMap. This is used
067         * by the Parcel unmarshalling code.
068         * 
069         * @param from the values to start with
070         * {@hide}
071         */
072        private ContentValues(HashMap<String, Object> values) {
073            mValues = values;
074        }
075    
076        @Override
077        public boolean equals(Object object) {
078            if (!(object instanceof ContentValues)) {
079                return false;
080            }
081            return mValues.equals(((ContentValues) object).mValues);
082        }
083    
084        @Override
085        public int hashCode() {
086            return mValues.hashCode();
087        }

096            mValues.put(key, value);
097        }

106        }

243        /**
244         * Gets a value and converts it to a String.
245         * 
246         * @param key the value to get
247         * @return the String for the value
248         */
249        public String getAsString(String key) {
250            Object value = mValues.get(key);
251            return value != null ? mValues.get(key).toString() : null;
252        }
253    
254        /**
255         * Gets a value and converts it to a Long.
256         * 
257         * @param key the value to get
258         * @return the Long value, or null if the value is missing or cannot be converted
259         */
260        public Long getAsLong(String key) {
261            Object value = mValues.get(key);
262            try {
263                return value != null ? ((Number) value).longValue() : null;
264            } catch (ClassCastException e) {
265                if (value instanceof CharSequence) {
266                    try {
267                        return Long.valueOf(value.toString());
268                    } catch (NumberFormatException e2) {
269                        Log.e(TAG, "Cannot parse Long value for " + value + " at key " + key);
270                        return null;
271                    }
272                } else {
273                    Log.e(TAG, "Cannot cast value for " + key + " to a Long");
274                    return null;
275                }
276            }
277        }

424        /**
425         * Gets a value that is a byte array. Note that this method will not convert
426         * any other types to byte arrays.
427         * 
428         * @param key the value to get
429         * @return the byte[] value, or null is the value is missing or not a byte[]
430         */
431        public byte[] getAsByteArray(String key) {
432            Object value = mValues.get(key);
433            if (value instanceof byte[]) {
434                return (byte[]) value;
435            } else {
436                return null;
437            }
438        }

449        public static final Parcelable.Creator<ContentValues> CREATOR =

453                // TODO - what ClassLoader should be passed to readHashMap?
454                HashMap<String, Object> values = in.readHashMap(null);

456            }

491        @Override
492        public String toString() {
493            StringBuilder sb = new StringBuilder();
494            for (String name : mValues.keySet()) {
495                String value = getAsString(name);
496                if (sb.length() > 0) sb.append(" ");
497                sb.append(name + "=" + value);
498            }
499            return sb.toString();
500        }
