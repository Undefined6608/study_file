017    package android.content;
018    
019    import android.database.ContentObserver;
020    import android.database.Cursor;
021    import android.os.Handler;
022    
023    import java.util.HashMap;
024    import java.util.Map;
025    import java.util.Observable;
026    
027    /**
028     * Caches the contents of a cursor into a Map of String->ContentValues and optionally
029     * keeps the cache fresh by registering for updates on the content backing the cursor. The column of
030     * the database that is to be used as the key of the map is user-configurable, and the
031     * ContentValues contains all columns other than the one that is designated the key.
032     * <p>
033     * The cursor data is accessed by row key and column name via getValue().
034     */
035    public class ContentQueryMap extends Observable {
036        private Cursor mCursor;
037        private String[] mColumnNames;
038        private int mKeyColumn;
039    
040        private Handler mHandlerForUpdateNotifications = null;
041        private boolean mKeepUpdated = false;
042    
043        private Map<String, ContentValues> mValues = null;
044    
045        private ContentObserver mContentObserver;
046    
047        /** Set when a cursor change notification is received and is cleared on a call to requery(). */
048        private boolean mDirty = false;
049    
050        /**
051         * Creates a ContentQueryMap that caches the content backing the cursor
052         *
053         * @param cursor the cursor whose contents should be cached
054         * @param columnNameOfKey the column that is to be used as the key of the values map
055         * @param keepUpdated true if the cursor's ContentProvider should be monitored for changes and 
056         * the map updated when changes do occur
057         * @param handlerForUpdateNotifications the Handler that should be used to receive
058         *  notifications of changes (if requested). Normally you pass null here, but if
059         *  you know that the thread that is creating this isn't a thread that can receive
060         *  messages then you can create your own handler and use that here.
061         */
062        public ContentQueryMap(Cursor cursor, String columnNameOfKey, boolean keepUpdated,
063                Handler handlerForUpdateNotifications) {
064            mCursor = cursor;
065            mColumnNames = mCursor.getColumnNames();
066            mKeyColumn = mCursor.getColumnIndexOrThrow(columnNameOfKey);
067            mHandlerForUpdateNotifications = handlerForUpdateNotifications;
068            setKeepUpdated(keepUpdated);
069    
070            // If we aren't keeping the cache updated with the current state of the cursor's 
071            // ContentProvider then read it once into the cache. Otherwise the cache will be filled 
072            // automatically.
073            if (!keepUpdated) {
074                readCursorIntoCache();
075            }
076        }
077    
078        /**
079         * Change whether or not the ContentQueryMap will register with the cursor's ContentProvider 
080         * for change notifications. If you use a ContentQueryMap in an activity you should call this
081         * with false in onPause(), which means you need to call it with true in onResume()
082         * if want it to be kept updated.
083         * @param keepUpdated if true the ContentQueryMap should be registered with the cursor's
084         * ContentProvider, false otherwise
085         */
086        public void setKeepUpdated(boolean keepUpdated) {
087            if (keepUpdated == mKeepUpdated) return;
088            mKeepUpdated = keepUpdated;
089    
090            if (!mKeepUpdated) {
091                mCursor.unregisterContentObserver(mContentObserver);
092                mContentObserver = null;
093            } else {
094                if (mHandlerForUpdateNotifications == null) {
095                    mHandlerForUpdateNotifications = new Handler();
096                }
097                if (mContentObserver == null) {
098                    mContentObserver = new ContentObserver(mHandlerForUpdateNotifications) {
099                        @Override
100                        public void onChange(boolean selfChange) {
101                            // If anyone is listening, we need to do this now to broadcast
102                            // to the observers.  Otherwise, we'll just set mDirty and
103                            // let it query lazily when they ask for the values.
104                            if (countObservers() != 0) {
105                                requery();
106                            } else {
107                                mDirty = true;
108                            }
109                        }
110                    };
111                }
112                mCursor.registerContentObserver(mContentObserver);
113                // mark dirty, since it is possible the cursor's backing data had changed before we 
114                // registered for changes
115                mDirty = true;
116            }
117        }
118    
119        /**
120         * Access the ContentValues for the row specified by rowName
121         * @param rowName which row to read
122         * @return the ContentValues for the row, or null if the row wasn't present in the cursor
123         */
124        public synchronized ContentValues getValues(String rowName) {
125            if (mDirty) requery();
126            return mValues.get(rowName);
127        }
128    
129        /** Requeries the cursor and reads the contents into the cache */
130        public void requery() {
131            mDirty = false;
132            mCursor.requery();
133            readCursorIntoCache();
134            setChanged();
135            notifyObservers();
136        }
137    
138        private synchronized void readCursorIntoCache() {
139            // Make a new map so old values returned by getRows() are undisturbed.
140            int capacity = mValues != null ? mValues.size() : 0;
141            mValues = new HashMap<String, ContentValues>(capacity);
142            while (mCursor.moveToNext()) {
143                ContentValues values = new ContentValues();
144                for (int i = 0; i < mColumnNames.length; i++) {
145                    if (i != mKeyColumn) {
146                        values.put(mColumnNames[i], mCursor.getString(i));
147                    }
148                }
149                mValues.put(mCursor.getString(mKeyColumn), values);
150            }
151        }
152    
153        public synchronized Map<String, ContentValues> getRows() {
154            if (mDirty) requery();
155            return mValues;
156        }
157    
158        public synchronized void close() {
159            if (mContentObserver != null) {
160                mCursor.unregisterContentObserver(mContentObserver);
161                mContentObserver = null;
162            }
163            mCursor.close();
164            mCursor = null;
165        }
166    
167        @Override
168        protected void finalize() throws Throwable {
169            if (mCursor != null) close();
170            super.finalize();
171        }
172    }
