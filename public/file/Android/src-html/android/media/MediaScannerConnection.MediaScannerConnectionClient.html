017    package android.media;
018    
019    import android.content.ComponentName;
020    import android.content.Context;
021    import android.content.Intent;
022    import android.content.ServiceConnection;
023    import android.media.IMediaScannerListener;
024    import android.media.IMediaScannerService;
025    import android.net.Uri;
026    import android.os.IBinder;
027    import android.os.RemoteException;
028    import android.util.Config;
029    import android.util.Log;
030    
031    
032    /**
033     * MediaScannerConnection provides a way for applications to pass a 
034     * newly created or downloaded media file to the media scanner service.
035     * The media scanner service will read metadata from the file and add 
036     * the file to the media content provider.
037     * The MediaScannerConnectionClient provides an interface for the 
038     * media scanner service to return the Uri for a newly scanned file
039     * to the client of the MediaScannerConnection class.
040     */
041    public class MediaScannerConnection implements ServiceConnection {
042    
043        private static final String TAG = "MediaScannerConnection";
044    
045        private Context mContext;
046        private MediaScannerConnectionClient mClient;
047        private IMediaScannerService mService;
048        private boolean mConnected; // true if connect() has been called since last disconnect()
049        
050        private IMediaScannerListener.Stub mListener = new IMediaScannerListener.Stub() {
051            public void scanCompleted(String path, Uri uri) {
052                MediaScannerConnectionClient client = mClient;
053                if (client != null) {
054                    client.onScanCompleted(path, uri);
055                }
056            }
057        };
058    
059        /**
060         * An interface for notifying clients of MediaScannerConnection
061         * when a connection to the MediaScanner service has been established
062         * and when the scanning of a file has completed.
063         */
064        public interface MediaScannerConnectionClient {
065            /**
066             * Called to notify the client when a connection to the 
067             * MediaScanner service has been established.
068             */    
069            public void onMediaScannerConnected();
070    
071            /**
072             * Called to notify the client when the media scanner has finished
073             * scanning a file.
074             * @param path the path to the file that has been scanned.
075             * @param uri the Uri for the file if the scanning operation succeeded 
076             * and the file was added to the media database, or null if scanning failed. 
077             */    
078            public void onScanCompleted(String path, Uri uri);
079        }
080    
081        /**
082         * Constructs a new MediaScannerConnection object.
083         * @param context the Context object, required for establishing a connection to
084         * the media scanner service.
085         * @param client an optional object implementing the MediaScannerConnectionClient
086         * interface, for receiving notifications from the media scanner.
087         */
088        public MediaScannerConnection(Context context, MediaScannerConnectionClient client) {
089            mContext = context;
090            mClient = client;
091        }
092    
093        /**
094         * Initiates a connection to the media scanner service.
095         * {@link MediaScannerConnectionClient#onMediaScannerConnected()}
096         * will be called when the connection is established.
097         */
098        public void connect() {
099            synchronized (this) {
100                if (!mConnected) {
101                    Intent intent = new Intent(IMediaScannerService.class.getName());
102                    mContext.bindService(intent, this, Context.BIND_AUTO_CREATE);
103                    mConnected = true;
104                }
105            }
106        }
107    
108        /**
109         * Releases the connection to the media scanner service.
110         */
111        public void disconnect() {
112            synchronized (this) {
113                if (mConnected) {
114                    if (Config.LOGV) {
115                        Log.v(TAG, "Disconnecting from Media Scanner");
116                    }
117                    try {
118                        mContext.unbindService(this);
119                    } catch (IllegalArgumentException ex) {
120                        if (Config.LOGV) {
121                            Log.v(TAG, "disconnect failed: " + ex);
122                        }
123                    }
124                    mConnected = false;
125                }
126            }
127        }
128        
129        /**
130         * Returns whether we are connected to the media scanner service
131         * @return true if we are connected, false otherwise
132         */
133        public synchronized boolean isConnected() {
134            return (mService != null && mConnected);
135        }
136    
137        /**
138         * Requests the media scanner to scan a file.
139         * @param path the path to the file to be scanned.
140         * @param mimeType  an optional mimeType for the file.
141         * If mimeType is null, then the mimeType will be inferred from the file extension.
142         * Success or failure of the scanning operation cannot be determined until 
143         * {@link MediaScannerConnectionClient#onScanCompleted(String, Uri)} is called.
144         */
145         public void scanFile(String path, String mimeType) {
146            synchronized (this) {
147                if (mService == null || !mConnected) {
148                    throw new IllegalStateException("not connected to MediaScannerService");
149                }
150                try {
151                    if (Config.LOGV) {
152                        Log.v(TAG, "Scanning file " + path);
153                    }
154                    mService.requestScanFile(path, mimeType, mListener);
155                } catch (RemoteException e) {
156                    if (Config.LOGD) {
157                        Log.d(TAG, "Failed to scan file " + path);
158                    }
159                }
160            }
161        }
162    
163        /**
164         * Part of the ServiceConnection interface.  Do not call.
165         */
166        public void onServiceConnected(ComponentName className, IBinder service) {
167            if (Config.LOGV) {
168                Log.v(TAG, "Connected to Media Scanner");
169            }
170            synchronized (this) {
171                mService = IMediaScannerService.Stub.asInterface(service);
172                if (mService != null && mClient != null) {
173                    mClient.onMediaScannerConnected();
174                }
175            }
176        }
177    
178        /**
179         * Part of the ServiceConnection interface.  Do not call.
180         */
181        public void onServiceDisconnected(ComponentName className) {
182            if (Config.LOGV) {
183                Log.v(TAG, "Disconnected from Media Scanner");
184            }
185            synchronized (this) {
186                mService = null;
187            }
188        }
189    }
