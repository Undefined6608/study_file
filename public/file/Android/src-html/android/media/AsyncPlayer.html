017    package android.media;
018    
019    import android.content.Context;
020    import android.net.Uri;
021    import android.os.PowerManager;
022    import android.util.Log;
023    
024    import java.io.IOException;
025    import java.lang.IllegalStateException;
026    
027    /**
028     * Plays a series of audio URIs, but does all the hard work on another thread
029     * so that any slowness with preparing or loading doesn't block the calling thread.
030     */
031    public class AsyncPlayer {
032        private static final int PLAY = 1;
033        private static final int STOP = 2;
034    
035        private static final class Command {
036            Command next;
037            int code;
038            Context context;
039            Uri uri;
040            boolean looping;
041            int stream;
042    
043            public String toString() {
044                return "{ code=" + code + " looping=" + looping + " stream=" + stream
045                        + " uri=" + uri + " }";
046            }
047        }
048    
049        private final class Thread extends java.lang.Thread {
050            Thread() {
051                super("AsyncPlayer-" + mTag);
052            }
053    
054            public void run() {
055                while (true) {
056                    Command cmd = null;
057    
058                    synchronized (mLock) {
059                        if (mHead != null) {
060                            cmd = mHead;
061                            mHead = cmd.next;
062                            if (mTail == cmd) {
063                                mTail = null;
064                            }
065                        }
066                    }
067    
068                    switch (cmd.code) {
069                    case PLAY:
070                        try {
071                            // Preparing can be slow, so if there is something else
072                            // is playing, let it continue until we're done, so there
073                            // is less of a glitch.
074                            MediaPlayer player = new MediaPlayer();
075                            player.setAudioStreamType(cmd.stream);
076                            player.setDataSource(cmd.context, cmd.uri);
077                            player.setLooping(cmd.looping);
078                            player.prepare();
079                            player.start();
080                            if (mPlayer != null) {
081                                mPlayer.release();
082                            }
083                            mPlayer = player;
084                        }
085                        catch (IOException e) {
086                            Log.w(mTag, "error loading sound for " + cmd.uri, e);
087                        } catch (IllegalStateException e) {
088                            Log.w(mTag, "IllegalStateException (content provider died?) " + cmd.uri, e);
089                        }
090                        break;
091                    case STOP:
092                        if (mPlayer != null) {
093                            mPlayer.stop();
094                            mPlayer.release();
095                            mPlayer = null;
096                        } else {
097                            Log.w(mTag, "STOP command without a player");
098                        }
099                        break;
100                    }
101    
102                    synchronized (mLock) {
103                        if (mHead == null) {
104                            // nothing left to do, quit
105                            // doing this check after we're done prevents the case where they
106                            // added it during the operation from spawning two threads and
107                            // trying to do them in parallel.
108                            mThread = null;
109                            releaseWakeLock();
110                            return;
111                        }
112                    }
113                }
114            }
115        }
116    
117        private String mTag;
118        private Command mHead;
119        private Command mTail;
120        private Thread mThread;
121        private MediaPlayer mPlayer;
122        private Object mLock = new Object();
123        private PowerManager.WakeLock mWakeLock;
124    
125        // The current state according to the caller.  Reality lags behind
126        // because of the asynchronous nature of this class.
127        private int mState = STOP;
128    
129        /**
130         * Construct an AsyncPlayer object.
131         *
132         * @param tag a string to use for debugging
133         */
134        public AsyncPlayer(String tag) {
135            if (tag != null) {
136                mTag = tag;
137            } else {
138                mTag = "AsyncPlayer";
139            }
140        }
141    
142        /**
143         * Start playing the sound.  It will actually start playing at some
144         * point in the future.  There are no guarantees about latency here.
145         * Calling this before another audio file is done playing will stop
146         * that one and start the new one.
147         *
148         * @param context Your application's context.
149         * @param uri The URI to play.  (see {@link MediaPlayer#setDataSource(Context, Uri)})
150         * @param looping Whether the audio should loop forever.  
151         *          (see {@link MediaPlayer#setLooping(boolean)})
152         * @param stream the AudioStream to use.
153         *          (see {@link MediaPlayer#setAudioStreamType(int)})
154         */
155        public void play(Context context, Uri uri, boolean looping, int stream) {
156            Command cmd = new Command();
157            cmd.code = PLAY;
158            cmd.context = context;
159            cmd.uri = uri;
160            cmd.looping = looping;
161            cmd.stream = stream;
162            synchronized (mLock) {
163                enqueueLocked(cmd);
164                mState = PLAY;
165            }
166        }
167        
168        /**
169         * Stop a previously played sound.  It can't be played again or unpaused
170         * at this point.  Calling this multiple times has no ill effects.
171         */
172        public void stop() {
173            synchronized (mLock) {
174                // This check allows stop to be called multiple times without starting
175                // a thread that ends up doing nothing.
176                if (mState != STOP) {
177                    Command cmd = new Command();
178                    cmd.code = STOP;
179                    enqueueLocked(cmd);
180                    mState = STOP;
181                }
182            }
183        }
184    
185        private void enqueueLocked(Command cmd) {
186            if (mTail == null) {
187                mHead = cmd;
188            } else {
189                mTail.next = cmd;
190            }
191            mTail = cmd;
192            if (mThread == null) {
193                acquireWakeLock();
194                mThread = new Thread();
195                mThread.start();
196            }
197        }
198    
199        /**
200         * We want to hold a wake lock while we do the prepare and play.  The stop probably is
201         * optional, but it won't hurt to have it too.  The problem is that if you start a sound
202         * while you're holding a wake lock (e.g. an alarm starting a notification), you want the
203         * sound to play, but if the CPU turns off before mThread gets to work, it won't.  The
204         * simplest way to deal with this is to make it so there is a wake lock held while the
205         * thread is starting or running.  You're going to need the WAKE_LOCK permission if you're
206         * going to call this.
207         *
208         * This must be called before the first time play is called.
209         *
210         * @hide
211         */
212        public void setUsesWakeLock(Context context) {
213            if (mWakeLock != null || mThread != null) {
214                // if either of these has happened, we've already played something.
215                // and our releases will be out of sync.
216                throw new RuntimeException("assertion failed mWakeLock=" + mWakeLock
217                        + " mThread=" + mThread);
218            }
219            PowerManager pm = (PowerManager)context.getSystemService(Context.POWER_SERVICE);
220            mWakeLock = pm.newWakeLock(PowerManager.PARTIAL_WAKE_LOCK, mTag);
221        }
222    
223        private void acquireWakeLock() {
224            if (mWakeLock != null) {
225                mWakeLock.acquire();
226            }
227        }
228    
229        private void releaseWakeLock() {
230            if (mWakeLock != null) {
231                mWakeLock.release();
232            }
233        }
234    }
