017    package android.view;
018    
019    import android.content.Context;
020    import android.content.ContextWrapper;
021    import android.content.res.Resources;
022    
023    /**
024     * A ContextWrapper that allows you to modify the theme from what is in the 
025     * wrapped context. 
026     */
027    public class ContextThemeWrapper extends ContextWrapper {
028        private Context mBase;
029        private int mThemeResource;
030        private Resources.Theme mTheme;
031        private LayoutInflater mInflater;
032    
033        public ContextThemeWrapper() {
034            super(null);
035        }
036        
037        public ContextThemeWrapper(Context base, int themeres) {
038            super(base);
039            mBase = base;
040            mThemeResource = themeres;
041        }
042    
043        @Override protected void attachBaseContext(Context newBase) {
044            super.attachBaseContext(newBase);
045            mBase = newBase;
046        }
047        
048        @Override public void setTheme(int resid) {
049            mThemeResource = resid;
050            initializeTheme();
051        }
052        
053        @Override public Resources.Theme getTheme() {
054            if (mTheme != null) {
055                return mTheme;
056            }
057    
058            if (mThemeResource == 0) {
059                mThemeResource = com.android.internal.R.style.Theme;
060            }
061            initializeTheme();
062    
063            return mTheme;
064        }
065    
066        @Override public Object getSystemService(String name) {
067            if (LAYOUT_INFLATER_SERVICE.equals(name)) {
068                if (mInflater == null) {
069                    mInflater = LayoutInflater.from(mBase).cloneInContext(this);
070                }
071                return mInflater;
072            }
073            return mBase.getSystemService(name);
074        }
075        
076        /**
077         * Called by {@link #setTheme} and {@link #getTheme} to apply a theme
078         * resource to the current Theme object.  Can override to change the
079         * default (simple) behavior.  This method will not be called in multiple
080         * threads simultaneously.
081         *
082         * @param theme The Theme object being modified.
083         * @param resid The theme style resource being applied to <var>theme</var>.
084         * @param first Set to true if this is the first time a style is being
085         *              applied to <var>theme</var>.
086         */
087        protected void onApplyThemeResource(Resources.Theme theme, int resid, boolean first) {
088            theme.applyStyle(resid, true);
089        }
090    
091        private void initializeTheme() {
092            final boolean first = mTheme == null;
093            if (first) {
094                mTheme = getResources().newTheme();
095                Resources.Theme theme = mBase.getTheme();
096                if (theme != null) {
097                    mTheme.setTo(theme);
098                }
099            }
100            onApplyThemeResource(mTheme, mThemeResource, first);
101        }
102    }
