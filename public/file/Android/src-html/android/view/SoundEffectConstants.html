017    package android.view;
018    
019    /**
020     * Constants to be used to play sound effects via {@link View#playSoundEffect(int)} 
021     */
022    public class SoundEffectConstants {
023    
024        private SoundEffectConstants() {}
025    
026        public static final int CLICK = 0;
027    
028        public static final int NAVIGATION_LEFT = 1;
029        public static final int NAVIGATION_UP = 2;
030        public static final int NAVIGATION_RIGHT = 3;
031        public static final int NAVIGATION_DOWN = 4;
032    
033        /**
034         * Get the sonification constant for the focus directions.
035         * @param direction One of {@link View#FOCUS_UP}, {@link View#FOCUS_DOWN},
036         *     {@link View#FOCUS_LEFT}, {@link View#FOCUS_RIGHT}, {@link View#FOCUS_FORWARD}
037         *     or {@link View#FOCUS_BACKWARD}
038    
039         * @return The appropriate sonification constant.
040         */
041        public static int getContantForFocusDirection(int direction) {
042            switch (direction) {
043                case View.FOCUS_RIGHT:
044                    return SoundEffectConstants.NAVIGATION_RIGHT;
045                case View.FOCUS_FORWARD:
046                case View.FOCUS_DOWN:
047                    return SoundEffectConstants.NAVIGATION_DOWN;
048                case View.FOCUS_LEFT:
049                    return SoundEffectConstants.NAVIGATION_LEFT;
050                case View.FOCUS_BACKWARD:
051                case View.FOCUS_UP:
052                    return SoundEffectConstants.NAVIGATION_UP;
053            }
054            throw new IllegalArgumentException("direction must be one of "
055                    + "{FOCUS_UP, FOCUS_DOWN, FOCUS_LEFT, FOCUS_RIGHT, FOCUS_FORWARD, FOCUS_BACKWARD}.");
056        }
057    }
