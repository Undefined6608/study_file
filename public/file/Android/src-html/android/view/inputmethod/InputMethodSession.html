017    package android.view.inputmethod;
018    
019    import android.graphics.Rect;
020    import android.os.Bundle;
021    import android.view.KeyEvent;
022    import android.view.MotionEvent;
023    
024    /**
025     * The InputMethodSession interface provides the per-client functionality
026     * of {@link InputMethod} that is safe to expose to applications.
027     * 
028     * <p>Applications will not normally use this interface themselves, instead
029     * relying on the standard interaction provided by
030     * {@link android.widget.TextView} and {@link android.widget.EditText}.
031     */
032    public interface InputMethodSession {
033        
034        public interface EventCallback {
035            void finishedEvent(int seq, boolean handled);
036        }
037        
038        /**
039         * This method is called when the application would like to stop
040         * receiving text input.
041         */
042        public void finishInput();
043    
044        /**
045         * This method is called when the selection or cursor in the current
046         * target input field has changed.
047         * 
048         * @param oldSelStart The previous text offset of the cursor selection
049         * start position.
050         * @param oldSelEnd The previous text offset of the cursor selection
051         * end position.
052         * @param newSelStart The new text offset of the cursor selection
053         * start position.
054         * @param newSelEnd The new text offset of the cursor selection
055         * end position.
056         * @param candidatesStart The text offset of the current candidate
057         * text start position.
058         * @param candidatesEnd The text offset of the current candidate
059         * text end position.
060         */
061        public void updateSelection(int oldSelStart, int oldSelEnd,
062                int newSelStart, int newSelEnd,
063                int candidatesStart, int candidatesEnd);
064    
065        /**
066         * This method is called when cursor location of the target input field
067         * has changed within its window.  This is not normally called, but will
068         * only be reported if requested by the input method.
069         * 
070         * @param newCursor The rectangle of the cursor currently being shown in
071         * the input field's window coordinates.
072         */
073        public void updateCursor(Rect newCursor);
074        
075        /**
076         * Called by a text editor that performs auto completion, to tell the
077         * input method about the completions it has available.  This can be used
078         * by the input method to display them to the user to select the text to
079         * be inserted.
080         * 
081         * @param completions Array of text completions that are available, starting with
082         * the best.  If this array is null, any existing completions will be
083         * removed.
084         */
085        public void displayCompletions(CompletionInfo[] completions);
086        
087        /**
088         * Called by a text editor to report its new extracted text when its
089         * contents change.  This will only be called if the input method
090         * calls {@link InputConnection#getExtractedText(ExtractedTextRequest, int)
091         * InputConnection.getExtractedText()} with the option to report updates.
092         * 
093         * @param token The input method supplied token for identifying its request.
094         * @param text The new extracted text.
095         */
096        public void updateExtractedText(int token, ExtractedText text);
097        
098        /**
099         * This method is called when a key is pressed.  When done with the event,
100         * the implementation must call back on <var>callback</var> with its
101         * result.
102         * 
103         * <p>
104         * If the input method wants to handle this event, return true, otherwise
105         * return false and the caller (i.e. the application) will handle the event.
106         * 
107         * @param event The key event.
108         * 
109         * @return Whether the input method wants to handle this event.
110         * 
111         * @see #dispatchKeyUp
112         * @see android.view.KeyEvent
113         */
114        public void dispatchKeyEvent(int seq, KeyEvent event, EventCallback callback);
115    
116        /**
117         * This method is called when there is a track ball event.
118         * 
119         * <p>
120         * If the input method wants to handle this event, return true, otherwise
121         * return false and the caller (i.e. the application) will handle the event.
122         * 
123         * @param event The motion event.
124         * 
125         * @return Whether the input method wants to handle this event.
126         * 
127         * @see android.view.MotionEvent
128         */
129        public void dispatchTrackballEvent(int seq, MotionEvent event, EventCallback callback);
130    
131        /**
132         * Process a private command sent from the application to the input method.
133         * This can be used to provide domain-specific features that are
134         * only known between certain input methods and their clients.
135         * 
136         * @param action Name of the command to be performed.  This <em>must</em>
137         * be a scoped name, i.e. prefixed with a package name you own, so that
138         * different developers will not create conflicting commands.
139         * @param data Any data to include with the command.
140         */
141        public void appPrivateCommand(String action, Bundle data);
142        
143        /**
144         * Toggle the soft input window. 
145         * Applications can toggle the state of the soft input window.
146         * @param showFlags Provides additional operating flags.  May be
147         * 0 or have the {@link InputMethodManager#SHOW_IMPLICIT},
148         * {@link InputMethodManager#SHOW_FORCED} bit set.
149         * @param hideFlags Provides additional operating flags.  May be
150         * 0 or have the {@link  InputMethodManager#HIDE_IMPLICIT_ONLY},
151         * {@link  InputMethodManager#HIDE_NOT_ALWAYS} bit set.
152         */
153        public void toggleSoftInput(int showFlags, int hideFlags);
154    }
