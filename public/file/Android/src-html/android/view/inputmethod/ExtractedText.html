001    package android.view.inputmethod;
002    
003    import android.os.Parcel;
004    import android.os.Parcelable;
005    import android.text.TextUtils;
006    
007    /**
008     * Information about text that has been extracted for use by an input method.
009     */
010    public class ExtractedText implements Parcelable {
011        /**
012         * The text that has been extracted.
013         */
014        public CharSequence text;
015    
016        /**
017         * The offset in the overall text at which the extracted text starts.
018         */
019        public int startOffset;
020        
021        /**
022         * If the content is a report of a partial text change, this is the
023         * offset where the change starts and it runs until
024         * {@link #partialEndOffset}.  If the content is the full text, this
025         * field is -1.
026         */
027        public int partialStartOffset;
028        
029        /**
030         * If the content is a report of a partial text change, this is the offset
031         * where the change ends.  Note that the actual text may be larger or
032         * smaller than the difference between this and {@link #partialEndOffset},
033         * meaning a reduction or increase, respectively, in the total text.
034         */
035        public int partialEndOffset;
036        
037        /**
038         * The offset where the selection currently starts within the extracted
039         * text.  The real selection start position is at
040         * <var>startOffset</var>+<var>selectionStart</var>.
041         */
042        public int selectionStart;
043        
044        /**
045         * The offset where the selection currently ends within the extracted
046         * text.  The real selection end position is at
047         * <var>startOffset</var>+<var>selectionEnd</var>.
048         */
049        public int selectionEnd;
050        
051        /**
052         * Bit for {@link #flags}: set if the text being edited can only be on
053         * a single line.
054         */
055        public static final int FLAG_SINGLE_LINE = 0x0001;
056        
057        /**
058         * Bit for {@link #flags}: set if the editor is currently in selection mode.
059         */
060        public static final int FLAG_SELECTING = 0x0002;
061        
062        /**
063         * Additional bit flags of information about the edited text.
064         */
065        public int flags;
066        
067        /**
068         * Used to package this object into a {@link Parcel}.
069         * 
070         * @param dest The {@link Parcel} to be written.
071         * @param flags The flags used for parceling.
072         */
073        public void writeToParcel(Parcel dest, int flags) {
074            TextUtils.writeToParcel(text, dest, flags);
075            dest.writeInt(startOffset);
076            dest.writeInt(partialStartOffset);
077            dest.writeInt(partialEndOffset);
078            dest.writeInt(selectionStart);
079            dest.writeInt(selectionEnd);
080            dest.writeInt(this.flags);
081        }
082    
083        /**
084         * Used to make this class parcelable.
085         */
086        public static final Parcelable.Creator<ExtractedText> CREATOR = new Parcelable.Creator<ExtractedText>() {
087            public ExtractedText createFromParcel(Parcel source) {
088                ExtractedText res = new ExtractedText();
089                res.text = TextUtils.CHAR_SEQUENCE_CREATOR.createFromParcel(source);
090                res.startOffset = source.readInt();
091                res.partialStartOffset = source.readInt();
092                res.partialEndOffset = source.readInt();
093                res.selectionStart = source.readInt();
094                res.selectionEnd = source.readInt();
095                res.flags = source.readInt();
096                return res;
097            }
098    
099            public ExtractedText[] newArray(int size) {
100                return new ExtractedText[size];
101            }
102        };
103    
104        public int describeContents() {
105            return 0;
106        }
107    }
