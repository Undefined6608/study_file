017    package android.view.inputmethod;
018    
019    import android.os.Bundle;
020    import android.view.KeyCharacterMap;
021    import android.view.KeyEvent;
022    
023    /**
024     * The InputConnection interface is the communication channel from an
025     * {@link InputMethod} back to the application that is receiving its input. It
026     * is used to perform such things as reading text around the cursor,
027     * committing text to the text box, and sending raw key events to the application.
028     * 
029     * <p>Implementations of this interface should generally be done by
030     * subclassing {@link BaseInputConnection}.
031     */
032    public interface InputConnection {
033        /**
034         * Flag for use with {@link #getTextAfterCursor} and
035         * {@link #getTextBeforeCursor} to have style information returned along
036         * with the text.  If not set, you will receive only the raw text.  If
037         * set, you may receive a complex CharSequence of both text and style
038         * spans.
039         */
040        static final int GET_TEXT_WITH_STYLES = 0x0001;
041        
042        /**
043         * Flag for use with {@link #getExtractedText} to indicate you would
044         * like to receive updates when the extracted text changes.
045         */
046        public static final int GET_EXTRACTED_TEXT_MONITOR = 0x0001;
047        
048        /**
049         * Get <var>n</var> characters of text before the current cursor position.
050         * 
051         * <p>This method may fail either if the input connection has become invalid
052         * (such as its process crashing) or the client is taking too long to
053         * respond with the text (it is given a couple seconds to return).
054         * In either case, a null is returned.
055         * 
056         * @param n The expected length of the text.
057         * @param flags Supplies additional options controlling how the text is
058         * returned.  May be either 0 or {@link #GET_TEXT_WITH_STYLES}.
059         * 
060         * @return Returns the text before the cursor position; the length of the
061         * returned text might be less than <var>n</var>.
062         */
063        public CharSequence getTextBeforeCursor(int n, int flags);
064    
065        /**
066         * Get <var>n</var> characters of text after the current cursor position.
067         * 
068         * <p>This method may fail either if the input connection has become invalid
069         * (such as its process crashing) or the client is taking too long to
070         * respond with the text (it is given a couple seconds to return).
071         * In either case, a null is returned.
072         * 
073         * @param n The expected length of the text.
074         * @param flags Supplies additional options controlling how the text is
075         * returned.  May be either 0 or {@link #GET_TEXT_WITH_STYLES}.
076         * 
077         * @return Returns the text after the cursor position; the length of the
078         * returned text might be less than <var>n</var>.
079         */
080        public CharSequence getTextAfterCursor(int n, int flags);
081    
082        /**
083         * Retrieve the current capitalization mode in effect at the current
084         * cursor position in the text.  See
085         * {@link android.text.TextUtils#getCapsMode TextUtils.getCapsMode} for
086         * more information.
087         * 
088         * <p>This method may fail either if the input connection has become invalid
089         * (such as its process crashing) or the client is taking too long to
090         * respond with the text (it is given a couple seconds to return).
091         * In either case, a 0 is returned.
092         * 
093         * @param reqModes The desired modes to retrieve, as defined by
094         * {@link android.text.TextUtils#getCapsMode TextUtils.getCapsMode}.  These
095         * constants are defined so that you can simply pass the current
096         * {@link EditorInfo#inputType TextBoxAttribute.contentType} value
097         * directly in to here.
098         * 
099         * @return Returns the caps mode flags that are in effect.
100         */
101        public int getCursorCapsMode(int reqModes);
102        
103        /**
104         * Retrieve the current text in the input connection's editor, and monitor
105         * for any changes to it.  This function returns with the current text,
106         * and optionally the input connection can send updates to the
107         * input method when its text changes.
108         * 
109         * <p>This method may fail either if the input connection has become invalid
110         * (such as its process crashing) or the client is taking too long to
111         * respond with the text (it is given a couple seconds to return).
112         * In either case, a null is returned.
113         * 
114         * @param request Description of how the text should be returned.
115         * @param flags Additional options to control the client, either 0 or
116         * {@link #GET_EXTRACTED_TEXT_MONITOR}.
117         * 
118         * @return Returns an ExtractedText object describing the state of the
119         * text view and containing the extracted text itself.
120         */
121        public ExtractedText getExtractedText(ExtractedTextRequest request,
122                int flags);
123    
124        /**
125         * Delete <var>leftLength</var> characters of text before the current cursor
126         * position, and delete <var>rightLength</var> characters of text after the
127         * current cursor position, excluding composing text.
128         * 
129         * @param leftLength The number of characters to be deleted before the
130         *        current cursor position.
131         * @param rightLength The number of characters to be deleted after the
132         *        current cursor position.
133         *        
134         * @return Returns true on success, false if the input connection is no longer
135         * valid.
136         */
137        public boolean deleteSurroundingText(int leftLength, int rightLength);
138    
139        /**
140         * Set composing text around the current cursor position with the given text,
141         * and set the new cursor position.  Any composing text set previously will
142         * be removed automatically.
143         * 
144         * @param text The composing text with styles if necessary. If no style
145         *        object attached to the text, the default style for composing text
146         *        is used. See {#link android.text.Spanned} for how to attach style
147         *        object to the text. {#link android.text.SpannableString} and
148         *        {#link android.text.SpannableStringBuilder} are two
149         *        implementations of the interface {#link android.text.Spanned}.
150         * @param newCursorPosition The new cursor position around the text.  If
151         *        > 0, this is relative to the end of the text - 1; if <= 0, this
152         *        is relative to the start of the text.  So a value of 1 will
153         *        always advance you to the position after the full text being
154         *        inserted.  Note that this means you can't position the cursor
155         *        within the text, because the editor can make modifications to
156         *        the text you are providing so it is not possible to correctly
157         *        specify locations there.
158         * 
159         * @return Returns true on success, false if the input connection is no longer
160         * valid.
161         */
162        public boolean setComposingText(CharSequence text, int newCursorPosition);
163    
164        /**
165         * Have the text editor finish whatever composing text is currently
166         * active.  This simply leaves the text as-is, removing any special
167         * composing styling or other state that was around it.  The cursor
168         * position remains unchanged.
169         */
170        public boolean finishComposingText();
171        
172        /**
173         * Commit text to the text box and set the new cursor position.
174         * Any composing text set previously will be removed
175         * automatically.
176         * 
177         * @param text The committed text.
178         * @param newCursorPosition The new cursor position around the text.  If
179         *        > 0, this is relative to the end of the text - 1; if <= 0, this
180         *        is relative to the start of the text.  So a value of 1 will
181         *        always advance you to the position after the full text being
182         *        inserted.  Note that this means you can't position the cursor
183         *        within the text, because the editor can make modifications to
184         *        the text you are providing so it is not possible to correctly
185         *        specify locations there.
186         * 
187         *        
188         * @return Returns true on success, false if the input connection is no longer
189         * valid.
190         */
191        public boolean commitText(CharSequence text, int newCursorPosition);
192    
193        /**
194         * Commit a completion the user has selected from the possible ones
195         * previously reported to {@link InputMethodSession#displayCompletions
196         * InputMethodSession.displayCompletions()}.  This will result in the
197         * same behavior as if the user had selected the completion from the
198         * actual UI.
199         * 
200         * @param text The committed completion.
201         *        
202         * @return Returns true on success, false if the input connection is no longer
203         * valid.
204         */
205        public boolean commitCompletion(CompletionInfo text);
206    
207        /**
208         * Set the selection of the text editor.  To set the cursor position,
209         * start and end should have the same value.
210         * @return Returns true on success, false if the input connection is no longer
211         * valid.
212         */
213        public boolean setSelection(int start, int end);
214        
215        /**
216         * Have the editor perform an action it has said it can do.
217         * 
218         * @param editorAction This must be one of the action constants for
219         * {@link EditorInfo#imeOptions EditorInfo.editorType}, such as
220         * {@link EditorInfo#IME_ACTION_GO EditorInfo.EDITOR_ACTION_GO}.
221         * 
222         * @return Returns true on success, false if the input connection is no longer
223         * valid.
224         */
225        public boolean performEditorAction(int editorAction);
226        
227        /**
228         * Perform a context menu action on the field.  The given id may be one of:
229         * {@link android.R.id#selectAll},
230         * {@link android.R.id#startSelectingText}, {@link android.R.id#stopSelectingText},
231         * {@link android.R.id#cut}, {@link android.R.id#copy},
232         * {@link android.R.id#paste}, {@link android.R.id#copyUrl},
233         * or {@link android.R.id#switchInputMethod}
234         */
235        public boolean performContextMenuAction(int id);
236        
237        /**
238         * Tell the editor that you are starting a batch of editor operations.
239         * The editor will try to avoid sending you updates about its state
240         * until {@link #endBatchEdit} is called.
241         */
242        public boolean beginBatchEdit();
243        
244        /**
245         * Tell the editor that you are done with a batch edit previously
246         * initiated with {@link #endBatchEdit}.
247         */
248        public boolean endBatchEdit();
249        
250        /**
251         * Send a key event to the process that is currently attached through
252         * this input connection.  The event will be dispatched like a normal
253         * key event, to the currently focused; this generally is the view that
254         * is providing this InputConnection, but due to the asynchronous nature
255         * of this protocol that can not be guaranteed and the focus may have
256         * changed by the time the event is received.
257         * 
258         * <p>
259         * This method can be used to send key events to the application. For
260         * example, an on-screen keyboard may use this method to simulate a hardware
261         * keyboard. There are three types of standard keyboards, numeric (12-key),
262         * predictive (20-key) and ALPHA (QWERTY). You can specify the keyboard type
263         * by specify the device id of the key event.
264         * 
265         * <p>
266         * You will usually want to set the flag
267         * {@link KeyEvent#FLAG_SOFT_KEYBOARD KeyEvent.FLAG_SOFT_KEYBOARD} on all
268         * key event objects you give to this API; the flag will not be set
269         * for you.
270         * 
271         * @param event The key event.
272         *        
273         * @return Returns true on success, false if the input connection is no longer
274         * valid.
275         * 
276         * @see KeyEvent
277         * @see KeyCharacterMap#NUMERIC
278         * @see KeyCharacterMap#PREDICTIVE
279         * @see KeyCharacterMap#ALPHA
280         */
281        public boolean sendKeyEvent(KeyEvent event);
282    
283        /**
284         * Clear the given meta key pressed states in the given input connection.
285         * 
286         * @param states The states to be cleared, may be one or more bits as
287         * per {@link KeyEvent#getMetaState() KeyEvent.getMetaState()}.
288         * 
289         * @return Returns true on success, false if the input connection is no longer
290         * valid.
291         */
292        public boolean clearMetaKeyStates(int states);
293        
294        /**
295         * Called by the IME to tell the client when it switches between fullscreen
296         * and normal modes.  This will normally be called for you by the standard
297         * implementation of {@link android.inputmethodservice.InputMethodService}.
298         */
299        public boolean reportFullscreenMode(boolean enabled);
300        
301        /**
302         * API to send private commands from an input method to its connected
303         * editor.  This can be used to provide domain-specific features that are
304         * only known between certain input methods and their clients.  Note that
305         * because the InputConnection protocol is asynchronous, you have no way
306         * to get a result back or know if the client understood the command; you
307         * can use the information in {@link EditorInfo} to determine if
308         * a client supports a particular command.
309         * 
310         * @param action Name of the command to be performed.  This <em>must</em>
311         * be a scoped name, i.e. prefixed with a package name you own, so that
312         * different developers will not create conflicting commands.
313         * @param data Any data to include with the command.
314         * @return Returns true if the command was sent (whether or not the
315         * associated editor understood it), false if the input connection is no longer
316         * valid.
317         */
318        public boolean performPrivateCommand(String action, Bundle data);
319    }
