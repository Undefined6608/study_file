017    package android.view.inputmethod;
018    
019    import android.os.Bundle;
020    import android.view.KeyEvent;
021    
022    /**
023     * <p>Wrapper class for proxying calls to another InputConnection.  Subclass
024     * and have fun!
025     */
026    public class InputConnectionWrapper implements InputConnection {
027        private InputConnection mTarget;
028        final boolean mMutable;
029        
030        public InputConnectionWrapper(InputConnection target, boolean mutable) {
031            mMutable = mutable;
032            mTarget = target;
033        }
034    
035        /**
036         * Change the target of the input connection.
037         */
038        public void setTarget(InputConnection target) {
039            if (mTarget != null && !mMutable) {
040                throw new SecurityException("not mutable");
041            }
042            mTarget = target;
043        }
044        
045        public CharSequence getTextBeforeCursor(int n, int flags) {
046            return mTarget.getTextBeforeCursor(n, flags);
047        }
048        
049        public CharSequence getTextAfterCursor(int n, int flags) {
050            return mTarget.getTextAfterCursor(n, flags);
051        }
052    
053        public int getCursorCapsMode(int reqModes) {
054            return mTarget.getCursorCapsMode(reqModes);
055        }
056        
057        public ExtractedText getExtractedText(ExtractedTextRequest request,
058                int flags) {
059            return mTarget.getExtractedText(request, flags);
060        }
061    
062        public boolean deleteSurroundingText(int leftLength, int rightLength) {
063            return mTarget.deleteSurroundingText(leftLength, rightLength);
064        }
065    
066        public boolean setComposingText(CharSequence text, int newCursorPosition) {
067            return mTarget.setComposingText(text, newCursorPosition);
068        }
069    
070        public boolean finishComposingText() {
071            return mTarget.finishComposingText();
072        }
073        
074        public boolean commitText(CharSequence text, int newCursorPosition) {
075            return mTarget.commitText(text, newCursorPosition);
076        }
077    
078        public boolean commitCompletion(CompletionInfo text) {
079            return mTarget.commitCompletion(text);
080        }
081    
082        public boolean setSelection(int start, int end) {
083            return mTarget.setSelection(start, end);
084        }
085        
086        public boolean performEditorAction(int editorAction) {
087            return mTarget.performEditorAction(editorAction);
088        }
089        
090        public boolean performContextMenuAction(int id) {
091            return mTarget.performContextMenuAction(id);
092        }
093        
094        public boolean beginBatchEdit() {
095            return mTarget.beginBatchEdit();
096        }
097        
098        public boolean endBatchEdit() {
099            return mTarget.endBatchEdit();
100        }
101        
102        public boolean sendKeyEvent(KeyEvent event) {
103            return mTarget.sendKeyEvent(event);
104        }
105    
106        public boolean clearMetaKeyStates(int states) {
107            return mTarget.clearMetaKeyStates(states);
108        }
109        
110        public boolean reportFullscreenMode(boolean enabled) {
111            return mTarget.reportFullscreenMode(enabled);
112        }
113        
114        public boolean performPrivateCommand(String action, Bundle data) {
115            return mTarget.performPrivateCommand(action, data);
116        }
117    }
