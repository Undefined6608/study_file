017    package android.view;
018    
019    import android.content.Context;
020    import android.hardware.Sensor;
021    import android.hardware.SensorEvent;
022    import android.hardware.SensorEventListener;
023    import android.hardware.SensorManager;
024    import android.util.Config;
025    import android.util.Log;
026    
027    /**
028     * Helper class for receiving notifications from the SensorManager when
029     * the orientation of the device has changed.
030     */
031    public abstract class OrientationEventListener {
032        private static final String TAG = "OrientationEventListener";
033        private static final boolean DEBUG = false;
034        private static final boolean localLOGV = DEBUG ? Config.LOGD : Config.LOGV;
035        private int mOrientation = ORIENTATION_UNKNOWN;
036        private SensorManager mSensorManager;
037        private boolean mEnabled = false;
038        private int mRate;
039        private Sensor mSensor;
040        private SensorEventListener mSensorEventListener;
041        private OrientationListener mOldListener;
042        
043        /**
044         * Returned from onOrientationChanged when the device orientation cannot be determined
045         * (typically when the device is in a close to flat position).
046         *
047         *  @see #onOrientationChanged
048         */
049        public static final int ORIENTATION_UNKNOWN = -1;
050    
051        /**
052         * Creates a new OrientationEventListener.
053         * 
054         * @param context for the OrientationEventListener.
055         */
056        public OrientationEventListener(Context context) {
057            this(context, SensorManager.SENSOR_DELAY_NORMAL);
058        }
059        
060        /**
061         * Creates a new OrientationEventListener.
062         * 
063         * @param context for the OrientationEventListener.
064         * @param rate at which sensor events are processed (see also
065         * {@link android.hardware.SensorManager SensorManager}). Use the default
066         * value of {@link android.hardware.SensorManager#SENSOR_DELAY_NORMAL 
067         * SENSOR_DELAY_NORMAL} for simple screen orientation change detection.
068         */
069        public OrientationEventListener(Context context, int rate) {
070            mSensorManager = (SensorManager)context.getSystemService(Context.SENSOR_SERVICE);
071            mRate = rate;
072            mSensor = mSensorManager.getDefaultSensor(Sensor.TYPE_ACCELEROMETER);
073            if (mSensor != null) {
074                // Create listener only if sensors do exist
075                mSensorEventListener = new SensorEventListenerImpl();
076            }
077        }
078        
079        void registerListener(OrientationListener lis) {
080            mOldListener = lis;
081        }
082    
083        /**
084         * Enables the OrientationEventListener so it will monitor the sensor and call
085         * {@link #onOrientationChanged} when the device orientation changes.
086         */
087        public void enable() {
088            if (mSensor == null) {
089                Log.w(TAG, "Cannot detect sensors. Not enabled");
090                return;
091            }
092            if (mEnabled == false) {
093                if (localLOGV) Log.d(TAG, "OrientationEventListener enabled");
094                mSensorManager.registerListener(mSensorEventListener, mSensor, mRate);
095                mEnabled = true;
096            }
097        }
098    
099        /**
100         * Disables the OrientationEventListener.
101         */
102        public void disable() {
103            if (mSensor == null) {
104                Log.w(TAG, "Cannot detect sensors. Invalid disable");
105                return;
106            }
107            if (mEnabled == true) {
108                if (localLOGV) Log.d(TAG, "OrientationEventListener disabled");
109                mSensorManager.unregisterListener(mSensorEventListener);
110                mEnabled = false;
111            }
112        }
113    
114        class SensorEventListenerImpl implements SensorEventListener {
115            private static final int _DATA_X = 0;
116            private static final int _DATA_Y = 1;
117            private static final int _DATA_Z = 2;
118            
119            public void onSensorChanged(SensorEvent event) {
120                float[] values = event.values;
121                int orientation = ORIENTATION_UNKNOWN;
122                float X = -values[_DATA_X];
123                float Y = -values[_DATA_Y];
124                float Z = -values[_DATA_Z];        
125                float magnitude = X*X + Y*Y;
126                // Don't trust the angle if the magnitude is small compared to the y value
127                if (magnitude * 4 >= Z*Z) {
128                    float OneEightyOverPi = 57.29577957855f;
129                    float angle = (float)Math.atan2(-Y, X) * OneEightyOverPi;
130                    orientation = 90 - (int)Math.round(angle);
131                    // normalize to 0 - 359 range
132                    while (orientation >= 360) {
133                        orientation -= 360;
134                    } 
135                    while (orientation < 0) {
136                        orientation += 360;
137                    }
138                }
139                if (mOldListener != null) {
140                    mOldListener.onSensorChanged(Sensor.TYPE_ACCELEROMETER, event.values);
141                }
142                if (orientation != mOrientation) {
143                    mOrientation = orientation;
144                    onOrientationChanged(orientation);
145                }
146            }
147    
148            public void onAccuracyChanged(Sensor sensor, int accuracy) {
149    
150            }
151        }
152        
153        /*
154         * Returns true if sensor is enabled and false otherwise
155         */
156        public boolean canDetectOrientation() {
157            return mSensor != null;
158        }
159    
160        /**
161         * Called when the orientation of the device has changed.
162         * orientation parameter is in degrees, ranging from 0 to 359.
163         * orientation is 0 degrees when the device is oriented in its natural position,
164         * 90 degrees when its left side is at the top, 180 degrees when it is upside down, 
165         * and 270 degrees when its right side is to the top.
166         * {@link #ORIENTATION_UNKNOWN} is returned when the device is close to flat
167         * and the orientation cannot be determined.
168         *
169         * @param orientation The new orientation of the device.
170         *
171         *  @see #ORIENTATION_UNKNOWN
172         */
173        abstract public void onOrientationChanged(int orientation);
174    }
