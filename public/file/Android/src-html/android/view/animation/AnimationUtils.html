035    public class AnimationUtils {
036        /**
037         * Returns the current animation time in milliseconds. This time should be used when invoking
038         * {@link Animation#setStartTime(long)}. Refer to {@link android.os.SystemClock} for more
039         * information about the different available clocks. The clock used by this method is
040         * <em>not</em> the "wall" clock (it is not {@link System#currentTimeMillis}).
041         *
042         * @return the current animation time in milliseconds
043         *
044         * @see android.os.SystemClock
045         */
046        public static long currentAnimationTimeMillis() {
047            return SystemClock.uptimeMillis();
048        }
049    
050        /**
051         * Loads an {@link Animation} object from a resource
052         * 
053         * @param context Application context used to access resources
054         * @param id The resource id of the animation to load
055         * @return The animation object reference by the specified id
056         * @throws NotFoundException when the animation cannot be loaded
057         */
058        public static Animation loadAnimation(Context context, int id)
059                throws NotFoundException {
060    
061            XmlResourceParser parser = null;
062            try {
063                parser = context.getResources().getAnimation(id);
064                return createAnimationFromXml(context, parser);
065            } catch (XmlPullParserException ex) {
066                NotFoundException rnf = new NotFoundException(
067                        "Can't load animation resource ID #0x"
068                                + Integer.toHexString(id));
069                rnf.initCause(ex);
070                throw rnf;
071            } catch (IOException ex) {
072                NotFoundException rnf = new NotFoundException(
073                        "Can't load animation resource ID #0x"
074                                + Integer.toHexString(id));
075                rnf.initCause(ex);
076                throw rnf;
077            } finally {
078                if (parser != null) parser.close();
079            }
080        }
081    
082        private static Animation createAnimationFromXml(Context c, XmlPullParser parser)
083                throws XmlPullParserException, IOException {
084            return createAnimationFromXml(c, parser, null, Xml.asAttributeSet(parser));
085        }
086        
087        private static Animation createAnimationFromXml(Context c, XmlPullParser parser, AnimationSet parent, AttributeSet attrs)
088        throws XmlPullParserException, IOException {
089            
090            Animation anim = null;
091     
092            // Make sure we are on a start tag.
093            int type = parser.getEventType();
094            int depth = parser.getDepth();
095    
096            while (((type=parser.next()) != XmlPullParser.END_TAG || parser.getDepth() > depth)
097                   && type != XmlPullParser.END_DOCUMENT) {
098    
099                if (type != XmlPullParser.START_TAG) {
100                    continue;
101                }
102    
103                String  name = parser.getName();
104        
105                if (name.equals("set")) {
106                    anim = new AnimationSet(c, attrs);
107                    createAnimationFromXml(c, parser, (AnimationSet)anim, attrs);
108                } else if (name.equals("alpha")) {
109                    anim = new AlphaAnimation(c, attrs);
110                } else if (name.equals("scale")) {
111                    anim = new ScaleAnimation(c, attrs);
112                }  else if (name.equals("rotate")) {
113                    anim = new RotateAnimation(c, attrs);
114                }  else if (name.equals("translate")) {
115                    anim = new TranslateAnimation(c, attrs);
116                } else {
117                    throw new RuntimeException("Unknown animation name: " + parser.getName());
118                }
119    
120                if (parent != null) {
121                    parent.addAnimation(anim);
122                }
123            }
124        
125            return anim;
126    
127        }

153        private static LayoutAnimationController createLayoutAnimationFromXml(
154                Context c, XmlPullParser parser)
155                throws XmlPullParserException, IOException {
156            return createLayoutAnimationFromXml(c, parser,
157                    Xml.asAttributeSet(parser));
158        }
159    
160        private static LayoutAnimationController createLayoutAnimationFromXml(
161                Context c, XmlPullParser parser, AttributeSet attrs)
162                throws XmlPullParserException, IOException {
163    
164            LayoutAnimationController controller = null;
165    
166            int type;
167            int depth = parser.getDepth();
168    
169            while (((type = parser.next()) != XmlPullParser.END_TAG
170                    || parser.getDepth() > depth)
171                    && type != XmlPullParser.END_DOCUMENT) {
172    
173                if (type != XmlPullParser.START_TAG) {
174                    continue;
175                }
176    
177                String name = parser.getName();
178    
179                if ("layoutAnimation".equals(name)) {
180                    controller = new LayoutAnimationController(c, attrs);
181                } else if ("gridLayoutAnimation".equals(name)) {
182                    controller = new GridLayoutAnimationController(c, attrs);
183                } else {
184                    throw new RuntimeException("Unknown layout animation name: " +
185                            name);
186                }
187            }
188    
189            return controller;
190        }

192        /**
193         * Make an animation for objects becoming visible. Uses a slide and fade
194         * effect.
195         * 
196         * @param c Context for loading resources
197         * @param fromLeft is the object to be animated coming from the left
198         * @return The new animation
199         */
200        public static Animation makeInAnimation(Context c, boolean fromLeft)
201        {
202            
203            Animation a;
204            if (fromLeft) {
205                a = AnimationUtils.loadAnimation(c, com.android.internal.R.anim.slide_in_left);
206            } else {
207                a = AnimationUtils.loadAnimation(c, com.android.internal.R.anim.slide_in_right);
208            }
209    
210            a.setInterpolator(new DecelerateInterpolator());
211            a.setStartTime(currentAnimationTimeMillis());
212            return a;
213        }

265                throws NotFoundException {

267            XmlResourceParser parser = null;
268            try {
269                parser = context.getResources().getAnimation(id);
270                return createInterpolatorFromXml(context, parser);
271            } catch (XmlPullParserException ex) {

273                        "Can't load animation resource ID #0x"
274                                + Integer.toHexString(id));
275                rnf.initCause(ex);
276                throw rnf;
277            } catch (IOException ex) {
278                NotFoundException rnf = new NotFoundException(
279                        "Can't load animation resource ID #0x"
280                                + Integer.toHexString(id));
281                rnf.initCause(ex);
282                throw rnf;
283            } finally {
284                if (parser != null) parser.close();
285            }
286    
287        }

289        private static Interpolator createInterpolatorFromXml(Context c, XmlPullParser parser)
290        throws XmlPullParserException, IOException {
291            
292            Interpolator interpolator = null;
293     
294            // Make sure we are on a start tag.
295            int type = parser.getEventType();
296            int depth = parser.getDepth();
297    
298            while (((type=parser.next()) != XmlPullParser.END_TAG || parser.getDepth() > depth)
299                   && type != XmlPullParser.END_DOCUMENT) {
300    
301                if (type != XmlPullParser.START_TAG) {
302                    continue;
303                }
304    
305                AttributeSet attrs = Xml.asAttributeSet(parser);
306                
307                String  name = parser.getName();
308        
309                
310                if (name.equals("linearInterpolator")) {
311                    interpolator = new LinearInterpolator(c, attrs);
312                } else if (name.equals("accelerateInterpolator")) {
313                    interpolator = new AccelerateInterpolator(c, attrs);
314                } else if (name.equals("decelerateInterpolator")) {
315                    interpolator = new DecelerateInterpolator(c, attrs);
316                }  else if (name.equals("accelerateDecelerateInterpolator")) {
317                    interpolator = new AccelerateDecelerateInterpolator(c, attrs);
318                }  else if (name.equals("cycleInterpolator")) {
319                    interpolator = new CycleInterpolator(c, attrs);
320                } else {
321                    throw new RuntimeException("Unknown interpolator name: " + parser.getName());
322                }
323    
324            }
325        
326            return interpolator;
327    
328        }
