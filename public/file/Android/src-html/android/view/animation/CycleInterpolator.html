017    package android.view.animation;
018    
019    import android.content.Context;
020    import android.content.res.TypedArray;
021    import android.util.AttributeSet;
022    
023    /**
024     * Repeats the animation for a specified number of cycles. The
025     * rate of change follows a sinusoidal pattern.
026     *
027     */
028    public class CycleInterpolator implements Interpolator {
029        public CycleInterpolator(float cycles) {
030            mCycles = cycles;
031        }
032        
033        public CycleInterpolator(Context context, AttributeSet attrs) {
034            TypedArray a =
035                context.obtainStyledAttributes(attrs, com.android.internal.R.styleable.CycleInterpolator);
036            
037            mCycles = a.getFloat(com.android.internal.R.styleable.CycleInterpolator_cycles, 1.0f);
038            
039            a.recycle();
040        }
041        
042        public float getInterpolation(float input) {
043            return (float)(Math.sin(2 * mCycles * Math.PI * input));
044        }
045        
046        private float mCycles;
047    }
