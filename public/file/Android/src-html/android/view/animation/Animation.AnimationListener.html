025    /**
026     * Abstraction for an Animation that can be applied to Views, Surfaces, or
027     * other objects. See the {@link android.view.animation animation package

030    public abstract class Animation implements Cloneable {

032         * Repeat the animation indefinitely.
033         */
034        public static final int INFINITE = -1;
035    
036        /**
037         * When the animation reaches the end and the repeat count is INFINTE_REPEAT
038         * or a positive value, the animation restarts from the beginning.
039         */
040        public static final int RESTART = 1;
041    
042        /**
043         * When the animation reaches the end and the repeat count is INFINTE_REPEAT
044         * or a positive value, the animation plays backward (and then forward again).
045         */
046        public static final int REVERSE = 2;

048        /**
049         * Can be used as the start time to indicate the start time should be the current

056         * The specified dimension is an absolute number of pixels.

058        public static final int ABSOLUTE = 0;

062         * height or width of the object being animated.
063         */
064        public static final int RELATIVE_TO_SELF = 1;
065

067         * The specified dimension holds a float and should be multiplied by the
068         * height or width of the parent of the object being animated.

194        /**
195         * Creates a new animation whose parameters come from the specified context and
196         * attributes set.
197         *
198         * @param context the application environment
199         * @param attrs the set of attributes holding the animation parameters
200         */
201        public Animation(Context context, AttributeSet attrs) {
202            TypedArray a = context.obtainStyledAttributes(attrs, com.android.internal.R.styleable.Animation);
203    
204            setDuration((long) a.getInt(com.android.internal.R.styleable.Animation_duration, 0));
205            setStartOffset((long) a.getInt(com.android.internal.R.styleable.Animation_startOffset, 0));
206            
207            setFillEnabled(a.getBoolean(com.android.internal.R.styleable.Animation_fillEnabled, mFillEnabled));
208            setFillBefore(a.getBoolean(com.android.internal.R.styleable.Animation_fillBefore, mFillBefore));
209            setFillAfter(a.getBoolean(com.android.internal.R.styleable.Animation_fillAfter, mFillAfter));
210    
211            final int resID = a.getResourceId(com.android.internal.R.styleable.Animation_interpolator, 0);
212            if (resID > 0) {
213                setInterpolator(context, resID);
214            }
215    
216            setRepeatCount(a.getInt(com.android.internal.R.styleable.Animation_repeatCount, mRepeatCount));
217            setRepeatMode(a.getInt(com.android.internal.R.styleable.Animation_repeatMode, RESTART));
218    
219            setZAdjustment(a.getInt(com.android.internal.R.styleable.Animation_zAdjustment, ZORDER_NORMAL));
220            
221            ensureInterpolator();
222    
223            a.recycle();
224        }

323         *
324         * @attr ref android.R.styleable#Animation_duration
325         */

331        }

333        /**
334         * Ensure that the duration that this animation will run is not longer
335         * than <var>durationMillis</var>.  In addition to adjusting the duration
336         * itself, this ensures that the repeat count also will not make it run
337         * longer than the given time.
338         * 
339         * @param durationMillis The maximum duration the animation is allowed
340         * to run.
341         */
342        public void restrictDuration(long durationMillis) {
343            // If we start after the duration, then we just won't run.
344            if (mStartOffset > durationMillis) {
345                mStartOffset = durationMillis;
346                mDuration = 0;
347                mRepeatCount = 0;
348                return;
349            }
350            
351            long dur = mDuration + mStartOffset;
352            if (dur > durationMillis) {
353                mDuration = durationMillis-mStartOffset;
354                dur = durationMillis;
355            }
356            // If the duration is 0 or less, then we won't run.
357            if (mDuration <= 0) {
358                mDuration = 0;
359                mRepeatCount = 0;
360                return;
361            }
362            // Reduce the number of repeats to keep below the maximum duration.
363            // The comparison between mRepeatCount and duration is to catch
364            // overflows after multiplying them.
365            if (mRepeatCount < 0 || mRepeatCount > durationMillis
366                    || (dur*mRepeatCount) > durationMillis) {
367                // Figure out how many times to do the animation.  Subtract 1 since
368                // repeat count is the number of times to repeat so 0 runs once.
369                mRepeatCount = (int)(durationMillis/dur) - 1;
370                if (mRepeatCount < 0) {
371                    mRepeatCount = 0;
372                }
373            }
374        }

658        /**
659         * Compute a hint at how long the entire animation may last, in milliseconds.
660         * Animations can be written to cause themselves to run for a different
661         * duration than what is computed here, but generally this should be

668        /**
669         * Gets the transformation to apply at a specified point in time. Implementations of this
670         * method should always replace the specified Transformation or document they are doing
671         * otherwise.
672         *
673         * @param currentTime Where we are in the animation. This is wall clock time.
674         * @param outTransformation A tranformation object that is provided by the
675         *        caller and will be filled in by the animation.
676         * @return True if the animation is still running
677         */
678        public boolean getTransformation(long currentTime, Transformation outTransformation) {
679            if (mStartTime == -1) {
680                mStartTime = currentTime;
681            }
682    
683            final long startOffset = getStartOffset();
684            final long duration = mDuration;
685            float normalizedTime;
686            if (duration != 0) {
687                normalizedTime = ((float) (currentTime - (mStartTime + startOffset))) /
688                        (float) duration;
689            } else {
690                // time is a step-change with a zero duration
691                normalizedTime = currentTime < mStartTime ? 0.0f : 1.0f;
692            }
693    
694            final boolean expired = normalizedTime >= 1.0f;
695            mMore = !expired;
696    
697            if (!mFillEnabled) normalizedTime = Math.max(Math.min(normalizedTime, 1.0f), 0.0f);
698    
699            if ((normalizedTime >= 0.0f || mFillBefore) && (normalizedTime <= 1.0f || mFillAfter)) {
700                if (!mStarted) {
701                    if (mListener != null) {
702                        mListener.onAnimationStart(this);
703                    }
704                    mStarted = true;
705                }
706    
707                if (mFillEnabled) normalizedTime = Math.max(Math.min(normalizedTime, 1.0f), 0.0f);
708    
709                if (mCycleFlip) {
710                    normalizedTime = 1.0f - normalizedTime;
711                }
712    
713                final float interpolatedTime = mInterpolator.getInterpolation(normalizedTime);
714                applyTransformation(interpolatedTime, outTransformation);
715            }
716    
717            if (expired) {
718                if (mRepeatCount == mRepeated) {
719                    if (!mEnded) {
720                        if (mListener != null) {
721                            mListener.onAnimationEnd(this);
722                        }
723                        mEnded = true;
724                    }
725                } else {
726                    if (mRepeatCount > 0) {
727                        mRepeated++;
728                    }
729    
730                    if (mRepeatMode == REVERSE) {
731                        mCycleFlip = !mCycleFlip;
732                    }
733    
734                    mStartTime = -1;
735                    mMore = true;
736    
737                    if (mListener != null) {
738                        mListener.onAnimationRepeat(this);
739                    }
740                }
741            }
742    
743            if (!mMore && mOneMoreTime) {
744                mOneMoreTime = false;

783        /**
784         * Convert the information in the description of a size to an actual
785         * dimension
786         *
787         * @param type One of Animation.ABSOLUTE, Animation.RELATIVE_TO_SELF, or
788         *             Animation.RELATIVE_TO_PARENT.
789         * @param value The dimension associated with the type parameter
790         * @param size The size of the object being animated
791         * @param parentSize The size of the parent of the object being animated
792         * @return The dimension to use for the animation
793         */
794        protected float resolveSize(int type, float value, int size, int parentSize) {
795            switch (type) {
796                case ABSOLUTE:
797                    return value;
798                case RELATIVE_TO_SELF:
799                    return size * value;
800                case RELATIVE_TO_PARENT:
801                    return parentSize * value;
802                default:
803                    return value;
804            }
805        }

808         * @param left
809         * @param top
810         * @param right
811         * @param bottom
812         * @param invalidate
813         * @param transformation
814         * 
815         * @hide
816         */
817        public void getInvalidateRegion(int left, int top, int right, int bottom,
818                RectF invalidate, Transformation transformation) {
819    
820            final RectF tempRegion = mRegion;
821            final RectF previousRegion = mPreviousRegion;
822    
823            invalidate.set(left, top, right, bottom);
824            transformation.getMatrix().mapRect(invalidate);
825            // Enlarge the invalidate region to account for rounding errors
826            invalidate.inset(-1.0f, -1.0f);
827            tempRegion.set(invalidate);
828            invalidate.union(previousRegion);
829    
830            previousRegion.set(tempRegion);
831    
832            final Transformation tempTransformation = mTransformation;
833            final Transformation previousTransformation = mPreviousTransformation;
834    
835            tempTransformation.set(transformation);
836            transformation.set(previousTransformation);
837            previousTransformation.set(tempTransformation);
838        }

859        /**
860         * Utility class to parse a string description of a size.
861         */
862        protected static class Description {
863            /**
864             * One of Animation.ABSOLUTE, Animation.RELATIVE_TO_SELF, or
865             * Animation.RELATIVE_TO_PARENT.
866             */
867            public int type;
868    
869            /**
870             * The absolute or relative dimension for this Description.
871             */
872            public float value;
873    
874            /**
875             * Size descriptions can appear inthree forms:
876             * <ol>
877             * <li>An absolute size. This is represented by a number.</li>
878             * <li>A size relative to the size of the object being animated. This
879             * is represented by a number followed by "%".</li> *
880             * <li>A size relative to the size of the parent of object being
881             * animated. This is represented by a number followed by "%p".</li>
882             * </ol>
883             * @param value The typed value to parse
884             * @return The parsed version of the description
885             */
886            static Description parseValue(TypedValue value) {
887                Description d = new Description();
888                if (value == null) {
889                    d.type = ABSOLUTE;
890                    d.value = 0;
891                } else {
892                    if (value.type == TypedValue.TYPE_FRACTION) {
893                        d.type = (value.data & TypedValue.COMPLEX_UNIT_MASK) ==
894                                TypedValue.COMPLEX_UNIT_FRACTION_PARENT ?
895                                        RELATIVE_TO_PARENT : RELATIVE_TO_SELF;
896                        d.value = TypedValue.complexToFloat(value.data);
897                        return d;
898                    } else if (value.type == TypedValue.TYPE_FLOAT) {
899                        d.type = ABSOLUTE;
900                        d.value = value.getFloat();
901                        return d;
902                    } else if (value.type >= TypedValue.TYPE_FIRST_INT &&
903                            value.type <= TypedValue.TYPE_LAST_INT) {
904                        d.type = ABSOLUTE;
905                        d.value = value.data;
906                        return d;
907                    }
908                }
909    
910                d.type = ABSOLUTE;
911                d.value = 0.0f;
912    
913                return d;
914            }
915        }
