017    package android.view;
018    
019    import android.content.Context;
020    import android.hardware.SensorListener;
021    
022    /**
023     * Helper class for receiving notifications from the SensorManager when
024     * the orientation of the device has changed.
025     *  @deprecated use {@link android.view.OrientationEventListener} instead.
026     *  This class internally uses the OrientationEventListener.
027     */
028    @Deprecated
029    public abstract class OrientationListener implements SensorListener {
030        private OrientationEventListener mOrientationEventLis;
031        
032        /**
033         * Returned from onOrientationChanged when the device orientation cannot be determined
034         * (typically when the device is in a close to flat position).
035         *
036         *  @see #onOrientationChanged
037         */
038        public static final int ORIENTATION_UNKNOWN = OrientationEventListener.ORIENTATION_UNKNOWN;
039    
040        /**
041         * Creates a new OrientationListener.
042         * 
043         * @param context for the OrientationListener.
044         */
045        public OrientationListener(Context context) {
046            mOrientationEventLis = new OrientationEventListenerInternal(context);
047        }
048    
049        /**
050         * Creates a new OrientationListener.
051         * 
052         * @param context for the OrientationListener.
053         * @param rate at which sensor events are processed (see also
054         * {@link android.hardware.SensorManager SensorManager}). Use the default
055         * value of {@link android.hardware.SensorManager#SENSOR_DELAY_NORMAL 
056         * SENSOR_DELAY_NORMAL} for simple screen orientation change detection.
057         */
058        public OrientationListener(Context context, int rate) {
059            mOrientationEventLis = new OrientationEventListenerInternal(context, rate);
060        }
061        
062        class OrientationEventListenerInternal extends OrientationEventListener {
063            OrientationEventListenerInternal(Context context) {
064                super(context);
065            }
066            
067            OrientationEventListenerInternal(Context context, int rate) {
068                super(context, rate);
069                // register so that onSensorChanged gets invoked
070                registerListener(OrientationListener.this);
071            }
072                    
073            public void onOrientationChanged(int orientation) {
074                OrientationListener.this.onOrientationChanged(orientation);
075            }
076        }
077        
078        /**
079         * Enables the OrientationListener so it will monitor the sensor and call
080         * {@link #onOrientationChanged} when the device orientation changes.
081         */
082        public void enable() {
083            mOrientationEventLis.enable();
084        }
085    
086        /**
087         * Disables the OrientationListener.
088         */
089        public void disable() {
090            mOrientationEventLis.disable();
091        }
092        
093        public void onAccuracyChanged(int sensor, int accuracy) {
094        }
095        
096        public void onSensorChanged(int sensor, float[] values) {
097            // just ignore the call here onOrientationChanged is invoked anyway
098        }
099    
100    
101        /**
102         * Look at {@link android.view.OrientationEventListener#onOrientationChanged}
103         * for method description and usage
104         * @param orientation The new orientation of the device.
105         *
106         *  @see #ORIENTATION_UNKNOWN
107         */
108        abstract public void onOrientationChanged(int orientation);
109        
110    }
