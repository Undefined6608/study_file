017    package android.view;
018    
019    import android.graphics.Rect;
020    import android.view.MotionEvent;
021    import android.view.View;
022    import android.view.ViewConfiguration;
023    
024    /**
025     * Helper class to handle situations where you want a view to have a larger touch area than its
026     * actual view bounds. The view whose touch area is changed is called the delegate view. This
027     * class should be used by an ancestor of the delegate. To use a TouchDelegate, first create an
028     * instance that specifies the bounds that should be mapped to the delegate and the delegate
029     * view itself.
030     * <p>
031     * The ancestor should then forward all of its touch events received in its
032     * {@link android.view.View#onTouchEvent(MotionEvent)} to {@link #onTouchEvent(MotionEvent)}.
033     * </p>
034     */
035    public class TouchDelegate {
036        
037        /**
038         * View that should receive forwarded touch events 
039         */
040        private View mDelegateView;
041        
042        /**
043         * Bounds in local coordinates of the containing view that should be mapped to the delegate
044         * view. This rect is used for initial hit testing.
045         */
046        private Rect mBounds;
047        
048        /**
049         * mBounds inflated to include some slop. This rect is to track whether the motion events
050         * should be considered to be be within the delegate view.
051         */
052        private Rect mSlopBounds;
053        
054        /**
055         * True if the delegate had been targeted on a down event (intersected mBounds).
056         */
057        private boolean mDelegateTargeted;
058    
059        /**
060         * The touchable region of the View extends above its actual extent.
061         */
062        public static final int ABOVE = 1;
063    
064        /**
065         * The touchable region of the View extends below its actual extent.
066         */
067        public static final int BELOW = 2;
068    
069        /**
070         * The touchable region of the View extends to the left of its
071         * actual extent.
072         */
073        public static final int TO_LEFT = 4;
074    
075        /**
076         * The touchable region of the View extends to the right of its
077         * actual extent.
078         */
079        public static final int TO_RIGHT = 8;
080    
081        private int mSlop;
082    
083        /**
084         * Constructor
085         * 
086         * @param bounds Bounds in local coordinates of the containing view that should be mapped to
087         *        the delegate view
088         * @param delegateView The view that should receive motion events
089         */
090        public TouchDelegate(Rect bounds, View delegateView) {
091            mBounds = bounds;
092    
093            mSlop = ViewConfiguration.get(delegateView.getContext()).getScaledTouchSlop();
094            mSlopBounds = new Rect(bounds);
095            mSlopBounds.inset(-mSlop, -mSlop);
096            mDelegateView = delegateView;
097        }
098    
099        /**
100         * Will forward touch events to the delegate view if the event is within the bounds
101         * specified in the constructor.
102         * 
103         * @param event The touch event to forward
104         * @return True if the event was forwarded to the delegate, false otherwise.
105         */
106        public boolean onTouchEvent(MotionEvent event) {
107            int x = (int)event.getX();
108            int y = (int)event.getY();
109            boolean sendToDelegate = false;
110            boolean hit = true;
111            boolean handled = false;
112    
113            switch (event.getAction()) {
114            case MotionEvent.ACTION_DOWN:
115                Rect bounds = mBounds;
116    
117                if (bounds.contains(x, y)) {
118                    mDelegateTargeted = true;
119                    sendToDelegate = true;
120                }
121                break;
122            case MotionEvent.ACTION_UP:
123            case MotionEvent.ACTION_MOVE:
124                sendToDelegate = mDelegateTargeted;
125                if (sendToDelegate) {
126                    Rect slopBounds = mSlopBounds;
127                    if (!slopBounds.contains(x, y)) {
128                        hit = false;
129                    }
130                }
131                break;
132            case MotionEvent.ACTION_CANCEL:
133                sendToDelegate = mDelegateTargeted;
134                mDelegateTargeted = false;
135                break;
136            }
137            if (sendToDelegate) {
138                final View delegateView = mDelegateView;
139                
140                if (hit) {
141                    // Offset event coordinates to be inside the target view
142                    event.setLocation(delegateView.getWidth() / 2, delegateView.getHeight() / 2);
143                } else {
144                    // Offset event coordinates to be outside the target view (in case it does
145                    // something like tracking pressed state)
146                    int slop = mSlop;
147                    event.setLocation(-(slop * 2), -(slop * 2));
148                }
149                handled = delegateView.dispatchTouchEvent(event);
150            }
151            return handled;
152        }
153    }
