017    package android.database;
018    
019    import java.util.ArrayList;
020    
021    /**
022     * Provides methods for (un)registering arbitrary observers in an ArrayList.
023     */
024    public abstract class Observable<T> {
025        /**
026         * The list of observers.  An observer can be in the list at most
027         * once and will never be null.
028         */
029        protected final ArrayList<T> mObservers = new ArrayList<T>();
030    
031        /**
032         * Adds an observer to the list. The observer cannot be null and it must not already
033         * be registered.
034         * @param observer the observer to register
035         * @throws IllegalArgumentException the observer is null
036         * @throws IllegalStateException the observer is already registered
037         */
038        public void registerObserver(T observer) {
039            if (observer == null) {
040                throw new IllegalArgumentException("The observer is null.");
041            }
042            synchronized(mObservers) {
043                if (mObservers.contains(observer)) {
044                    throw new IllegalStateException("Observer " + observer + " is already registered.");
045                }
046                mObservers.add(observer);
047            }
048        }
049    
050        /**
051         * Removes a previously registered observer. The observer must not be null and it
052         * must already have been registered.
053         * @param observer the observer to unregister
054         * @throws IllegalArgumentException the observer is null
055         * @throws IllegalStateException the observer is not yet registered
056         */
057        public void unregisterObserver(T observer) {
058            if (observer == null) {
059                throw new IllegalArgumentException("The observer is null.");
060            }
061            synchronized(mObservers) {
062                int index = mObservers.indexOf(observer);
063                if (index == -1) {
064                    throw new IllegalStateException("Observer " + observer + " was not registered.");
065                }
066                mObservers.remove(index);
067            }
068        }
069        
070        /**
071         * Remove all registered observer
072         */
073        public void unregisterAll() {
074            synchronized(mObservers) {
075                mObservers.clear();
076            }        
077        }
078    }
