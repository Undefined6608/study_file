017    package android.database;
018    
019    import android.os.Handler;
020    
021    /**
022     * Receives call backs for changes to content. Must be implemented by objects which are added
023     * to a {@link ContentObservable}.
024     */
025    public abstract class ContentObserver {
026    
027        private Transport mTransport;
028    
029        // Protects mTransport
030        private Object lock = new Object();
031    
032        /* package */ Handler mHandler;
033    
034        private final class NotificationRunnable implements Runnable {
035    
036            private boolean mSelf;
037    
038            public NotificationRunnable(boolean self) {
039                mSelf = self;
040            }
041    
042            public void run() {
043                ContentObserver.this.onChange(mSelf);
044            }
045        }
046    
047        private static final class Transport extends IContentObserver.Stub {
048            ContentObserver mContentObserver;
049    
050            public Transport(ContentObserver contentObserver) {
051                mContentObserver = contentObserver;
052            }
053    
054            public boolean deliverSelfNotifications() {
055                ContentObserver contentObserver = mContentObserver;
056                if (contentObserver != null) {
057                    return contentObserver.deliverSelfNotifications();
058                }
059                return false;
060            }
061    
062            public void onChange(boolean selfChange) {
063                ContentObserver contentObserver = mContentObserver;
064                if (contentObserver != null) {
065                    contentObserver.dispatchChange(selfChange);
066                }
067            }
068    
069            public void releaseContentObserver() {
070                mContentObserver = null;
071            }
072        }
073    
074        /**
075         * onChange() will happen on the provider Handler.
076         *
077         * @param handler The handler to run {@link #onChange} on.
078         */
079        public ContentObserver(Handler handler) {
080            mHandler = handler;
081        }
082    
083        /**
084         * Gets access to the binder transport object. Not for public consumption.
085         *
086         * {@hide}
087         */
088        public IContentObserver getContentObserver() {
089            synchronized(lock) {
090                if (mTransport == null) {
091                    mTransport = new Transport(this);
092                }
093                return mTransport;
094            }
095        }
096    
097        /**
098         * Gets access to the binder transport object, and unlinks the transport object
099         * from the ContentObserver. Not for public consumption.
100         *
101         * {@hide}
102         */
103        public IContentObserver releaseContentObserver() {
104            synchronized(lock) {
105                Transport oldTransport = mTransport;
106                if (oldTransport != null) {
107                    oldTransport.releaseContentObserver();
108                    mTransport = null;
109                }
110                return oldTransport;
111            }
112        }
113    
114        /**
115         * Returns true if this observer is interested in notifications for changes
116         * made through the cursor the observer is registered with.
117         */
118        public boolean deliverSelfNotifications() {
119            return false;
120        }
121    
122        /**
123         * This method is called when a change occurs to the cursor that
124         * is being observed.
125         *  
126         * @param selfChange true if the update was caused by a call to <code>commit</code> on the
127         *  cursor that is being observed.
128         */
129        public void onChange(boolean selfChange) {}
130    
131        public final void dispatchChange(boolean selfChange) {
132            if (mHandler == null) {
133                onChange(selfChange);
134            } else {
135                mHandler.post(new NotificationRunnable(selfChange));
136            }
137        }
138    }
