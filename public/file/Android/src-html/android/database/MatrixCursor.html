017    package android.database;
018    
019    import java.util.ArrayList;
020    
021    /**
022     * A mutable cursor implementation backed by an array of {@code Object}s. Use
023     * {@link #newRow()} to add rows. Automatically expands internal capacity
024     * as needed.
025     */
026    public class MatrixCursor extends AbstractCursor {
027    
028        private final String[] columnNames;
029        private Object[] data;
030        private int rowCount = 0;
031        private final int columnCount;
032    
033        /**
034         * Constructs a new cursor with the given initial capacity.
035         *
036         * @param columnNames names of the columns, the ordering of which
037         *  determines column ordering elsewhere in this cursor
038         * @param initialCapacity in rows
039         */
040        public MatrixCursor(String[] columnNames, int initialCapacity) {
041            this.columnNames = columnNames;
042            this.columnCount = columnNames.length;
043    
044            if (initialCapacity < 1) {
045                initialCapacity = 1;
046            }
047    
048            this.data = new Object[columnCount * initialCapacity];
049        }
050    
051        /**
052         * Constructs a new cursor.
053         *
054         * @param columnNames names of the columns, the ordering of which
055         *  determines column ordering elsewhere in this cursor
056         */
057        public MatrixCursor(String[] columnNames) {
058            this(columnNames, 16);
059        }
060    
061        /**
062         * Gets value at the given column for the current row.
063         */
064        private Object get(int column) {
065            if (column < 0 || column >= columnCount) {
066                throw new CursorIndexOutOfBoundsException("Requested column: "
067                        + column + ", # of columns: " +  columnCount);
068            }
069            if (mPos < 0) {
070                throw new CursorIndexOutOfBoundsException("Before first row.");
071            }
072            if (mPos >= rowCount) {
073                throw new CursorIndexOutOfBoundsException("After last row.");
074            }
075            return data[mPos * columnCount + column];
076        }
077    
078        /**
079         * Adds a new row to the end and returns a builder for that row. Not safe
080         * for concurrent use.
081         *
082         * @return builder which can be used to set the column values for the new
083         *  row
084         */
085        public RowBuilder newRow() {
086            rowCount++;
087            int endIndex = rowCount * columnCount;
088            ensureCapacity(endIndex);
089            int start = endIndex - columnCount;
090            return new RowBuilder(start, endIndex);
091        }
092    
093        /**
094         * Adds a new row to the end with the given column values. Not safe
095         * for concurrent use.
096         *
097         * @throws IllegalArgumentException if {@code columnValues.length !=
098         *  columnNames.length}
099         * @param columnValues in the same order as the the column names specified
100         *  at cursor construction time
101         */
102        public void addRow(Object[] columnValues) {
103            if (columnValues.length != columnCount) {
104                throw new IllegalArgumentException("columnNames.length = "
105                        + columnCount + ", columnValues.length = "
106                        + columnValues.length);
107            }
108    
109            int start = rowCount++ * columnCount;
110            ensureCapacity(start + columnCount);
111            System.arraycopy(columnValues, 0, data, start, columnCount);
112        }
113    
114        /**
115         * Adds a new row to the end with the given column values. Not safe
116         * for concurrent use.
117         *
118         * @throws IllegalArgumentException if {@code columnValues.size() !=
119         *  columnNames.length}
120         * @param columnValues in the same order as the the column names specified
121         *  at cursor construction time
122         */
123        public void addRow(Iterable<?> columnValues) {
124            int start = rowCount * columnCount;
125            int end = start + columnCount;
126            ensureCapacity(end);
127    
128            if (columnValues instanceof ArrayList<?>) {
129                addRow((ArrayList<?>) columnValues, start);
130                return;
131            }
132    
133            int current = start;
134            Object[] localData = data;
135            for (Object columnValue : columnValues) {
136                if (current == end) {
137                    // TODO: null out row?
138                    throw new IllegalArgumentException(
139                            "columnValues.size() > columnNames.length");
140                }
141                localData[current++] = columnValue;
142            }
143    
144            if (current != end) {
145                // TODO: null out row?
146                throw new IllegalArgumentException(
147                        "columnValues.size() < columnNames.length");
148            }
149    
150            // Increase row count here in case we encounter an exception.
151            rowCount++;
152        }
153    
154        /** Optimization for {@link ArrayList}. */
155        private void addRow(ArrayList<?> columnValues, int start) {
156            int size = columnValues.size();
157            if (size != columnCount) {
158                throw new IllegalArgumentException("columnNames.length = "
159                        + columnCount + ", columnValues.size() = " + size);
160            }
161    
162            rowCount++;
163            Object[] localData = data;
164            for (int i = 0; i < size; i++) {
165                localData[start + i] = columnValues.get(i);
166            }
167        }
168    
169        /** Ensures that this cursor has enough capacity. */
170        private void ensureCapacity(int size) {
171            if (size > data.length) {
172                Object[] oldData = this.data;
173                int newSize = data.length * 2;
174                if (newSize < size) {
175                    newSize = size;
176                }
177                this.data = new Object[newSize];
178                System.arraycopy(oldData, 0, this.data, 0, oldData.length);
179            }
180        }
181    
182        /**
183         * Builds a row, starting from the left-most column and adding one column
184         * value at a time. Follows the same ordering as the column names specified
185         * at cursor construction time.
186         */
187        public class RowBuilder {
188    
189            private int index;
190            private final int endIndex;
191    
192            RowBuilder(int index, int endIndex) {
193                this.index = index;
194                this.endIndex = endIndex;
195            }
196    
197            /**
198             * Sets the next column value in this row.
199             *
200             * @throws CursorIndexOutOfBoundsException if you try to add too many
201             *  values
202             * @return this builder to support chaining
203             */
204            public RowBuilder add(Object columnValue) {
205                if (index == endIndex) {
206                    throw new CursorIndexOutOfBoundsException(
207                            "No more columns left.");
208                }
209    
210                data[index++] = columnValue;
211                return this;
212            }
213        }
214    
215        // AbstractCursor implementation.
216    
217        public int getCount() {
218            return rowCount;
219        }
220    
221        public String[] getColumnNames() {
222            return columnNames;
223        }
224    
225        public String getString(int column) {
226            return String.valueOf(get(column));
227        }
228    
229        public short getShort(int column) {
230            Object value = get(column);
231            return (value instanceof String)
232                    ? Short.valueOf((String) value)
233                    : ((Number) value).shortValue();
234        }
235    
236        public int getInt(int column) {
237            Object value = get(column);
238            return (value instanceof String)
239                    ? Integer.valueOf((String) value)
240                    : ((Number) value).intValue();
241        }
242    
243        public long getLong(int column) {
244            Object value = get(column);
245            return (value instanceof String)
246                    ? Long.valueOf((String) value)
247                    : ((Number) value).longValue();
248        }
249    
250        public float getFloat(int column) {
251            Object value = get(column);
252            return (value instanceof String)
253                    ? Float.valueOf((String) value)
254                    : ((Number) value).floatValue();
255        }
256    
257        public double getDouble(int column) {
258            Object value = get(column);
259            return (value instanceof String)
260                    ? Double.valueOf((String) value)
261                    : ((Number) value).doubleValue();
262        }
263    
264        public boolean isNull(int column) {
265            return get(column) == null;
266        }
267    }
