017    package android.os;
018    
019    import android.util.Log;
020    
021    /**
022     * Schedule a countdown until a time in the future, with
023     * regular notifications on intervals along the way.
024     *
025     * Example of showing a 30 second countdown in a text field:
026     *
027     * <pre class="prettyprint">
028     * new CountdownTimer(30000, 1000) {
029     *
030     *     public void onTick(long millisUntilFinished) {
031     *         mTextField.setText("seconds remaining: " + millisUntilFinished / 1000);
032     *     }
033     *
034     *     public void onFinish() {
035     *         mTextField.setText("done!");
036     *     }
037     *  }.start();
038     * </pre>
039     *
040     * The calls to {@link #onTick(long)} are synchronized to this object so that
041     * one call to {@link #onTick(long)} won't ever occur before the previous
042     * callback is complete.  This is only relevant when the implementation of
043     * {@link #onTick(long)} takes an amount of time to execute that is significant
044     * compared to the countdown interval.
045     */
046    public abstract class CountDownTimer {
047    
048        /**
049         * Millis since epoch when alarm should stop.
050         */
051        private final long mMillisInFuture;
052    
053        /**
054         * The interval in millis that the user receives callbacks
055         */
056        private final long mCountdownInterval;
057    
058        private long mStopTimeInFuture;
059    
060        /**
061         * @param millisInFuture The number of millis in the future from the call
062         *   to {@link #start()} until the countdown is done and {@link #onFinish()}
063         *   is called.
064         * @param countDownInterval The interval along the way to receive
065         *   {@link #onTick(long)} callbacks.
066         */
067        public CountDownTimer(long millisInFuture, long countDownInterval) {
068            mMillisInFuture = millisInFuture;
069            mCountdownInterval = countDownInterval;
070        }
071    
072        /**
073         * Cancel the countdown.
074         */
075        public final void cancel() {
076            mHandler.removeMessages(MSG);
077        }
078    
079        /**
080         * Start the countdown.
081         */
082        public synchronized final CountDownTimer start() {
083            if (mMillisInFuture <= 0) {
084                onFinish();
085                return this;
086            }
087            mStopTimeInFuture = SystemClock.elapsedRealtime() + mMillisInFuture;
088            mHandler.sendMessage(mHandler.obtainMessage(MSG));
089            return this;
090        }
091    
092    
093        /**
094         * Callback fired on regular interval.
095         * @param millisUntilFinished The amount of time until finished.
096         */
097        public abstract void onTick(long millisUntilFinished);
098    
099        /**
100         * Callback fired when the time is up.
101         */
102        public abstract void onFinish();
103    
104    
105        private static final int MSG = 1;
106    
107    
108        // handles counting down
109        private Handler mHandler = new Handler() {
110    
111            @Override
112            public void handleMessage(Message msg) {
113    
114                synchronized (CountDownTimer.this) {
115                    final long millisLeft = mStopTimeInFuture - SystemClock.elapsedRealtime();
116    
117                    if (millisLeft <= 0) {
118                        onFinish();
119                    } else if (millisLeft < mCountdownInterval) {
120                        // no tick, just delay until done
121                        sendMessageDelayed(obtainMessage(MSG), millisLeft);
122                    } else {
123                        long lastTickStart = SystemClock.elapsedRealtime();
124                        onTick(millisLeft);
125    
126                        // take into account user's onTick taking time to execute
127                        long delay = lastTickStart + mCountdownInterval - SystemClock.elapsedRealtime();
128    
129                        // special case: user's onTick took more than interval to
130                        // complete, skip to next interval
131                        while (delay < 0) delay += mCountdownInterval;
132    
133                        sendMessageDelayed(obtainMessage(MSG), delay);
134                    }
135                }
136            }
137        };
138    }
