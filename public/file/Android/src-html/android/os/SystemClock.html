017    package android.os;
018    
019    
020    /**
021     * Core timekeeping facilities.
022     *
023     * <p> Three different clocks are available, and they should not be confused:
024     *
025     * <ul>
026     *     <li> <p> {@link System#currentTimeMillis System.currentTimeMillis()}
027     *     is the standard "wall" clock (time and date) expressing milliseconds
028     *     since the epoch.  The wall clock can be set by the user or the phone
029     *     network (see {@link #setCurrentTimeMillis}), so the time may jump
030     *     backwards or forwards unpredictably.  This clock should only be used
031     *     when correspondence with real-world dates and times is important, such
032     *     as in a calendar or alarm clock application.  Interval or elapsed
033     *     time measurements should use a different clock.
034     *
035     *     <li> <p> {@link #uptimeMillis} is counted in milliseconds since the
036     *     system was booted.  This clock stops when the system enters deep
037     *     sleep (CPU off, display dark, device waiting for external input),
038     *     but is not affected by clock scaling, idle, or other power saving
039     *     mechanisms.  This is the basis for most interval timing
040     *     such as {@link Thread#sleep(long) Thread.sleep(millls)},
041     *     {@link Object#wait(long) Object.wait(millis)}, and
042     *     {@link System#nanoTime System.nanoTime()}.  This clock is guaranteed
043     *     to be monotonic, and is the recommended basis for the general purpose
044     *     interval timing of user interface events, performance measurements,
045     *     and anything else that does not need to measure elapsed time during
046     *     device sleep.  Most methods that accept a timestamp value expect the
047     *     {@link #uptimeMillis} clock.
048     *
049     *     <li> <p> {@link #elapsedRealtime} is counted in milliseconds since the
050     *     system was booted, including deep sleep.  This clock should be used
051     *     when measuring time intervals that may span periods of system sleep.
052     * </ul>
053     *
054     * There are several mechanisms for controlling the timing of events:
055     *
056     * <ul>
057     *     <li> <p> Standard functions like {@link Thread#sleep(long)
058     *     Thread.sleep(millis)} and {@link Object#wait(long) Object.wait(millis)}
059     *     are always available.  These functions use the {@link #uptimeMillis}
060     *     clock; if the device enters sleep, the remainder of the time will be
061     *     postponed until the device wakes up.  These synchronous functions may
062     *     be interrupted with {@link Thread#interrupt Thread.interrupt()}, and
063     *     you must handle {@link InterruptedException}.
064     *
065     *     <li> <p> {@link #sleep SystemClock.sleep(millis)} is a utility function
066     *     very similar to {@link Thread#sleep(long) Thread.sleep(millis)}, but it
067     *     ignores {@link InterruptedException}.  Use this function for delays if
068     *     you do not use {@link Thread#interrupt Thread.interrupt()}, as it will
069     *     preserve the interrupted state of the thread.
070     *
071     *     <li> <p> The {@link android.os.Handler} class can schedule asynchronous
072     *     callbacks at an absolute or relative time.  Handler objects also use the
073     *     {@link #uptimeMillis} clock, and require an {@link android.os.Looper
074     *     event loop} (normally present in any GUI application).
075     *
076     *     <li> <p> The {@link android.app.AlarmManager} can trigger one-time or
077     *     recurring events which occur even when the device is in deep sleep
078     *     or your application is not running.  Events may be scheduled with your
079     *     choice of {@link java.lang.System#currentTimeMillis} (RTC) or
080     *     {@link #elapsedRealtime} (ELAPSED_REALTIME), and cause an
081     *     {@link android.content.Intent} broadcast when they occur.
082     * </ul>
083     */
084    public final class SystemClock {
085        /**
086         * This class is uninstantiable.
087         */
088        private SystemClock() {
089            // This space intentionally left blank.
090        }
091    
092        /**
093         * Waits a given number of milliseconds (of uptimeMillis) before returning.
094         * Similar to {@link java.lang.Thread#sleep(long)}, but does not throw
095         * {@link InterruptedException}; {@link Thread#interrupt()} events are
096         * deferred until the next interruptible operation.  Does not return until
097         * at least the specified number of milliseconds has elapsed.
098         *
099         * @param ms to sleep before returning, in milliseconds of uptime.
100         */
101        public static void sleep(long ms)
102        {
103            long start = uptimeMillis();
104            long duration = ms;
105            boolean interrupted = false;
106            do {
107                try {
108                    Thread.sleep(duration);
109                }
110                catch (InterruptedException e) {
111                    interrupted = true;
112                }
113                duration = start + ms - uptimeMillis();
114            } while (duration > 0);
115            
116            if (interrupted) {
117                // Important: we don't want to quietly eat an interrupt() event,
118                // so we make sure to re-interrupt the thread so that the next
119                // call to Thread.sleep() or Object.wait() will be interrupted.
120                Thread.currentThread().interrupt();
121            }
122        }
123        
124        /**
125         * Sets the current wall time, in milliseconds.  Requires the calling
126         * process to have appropriate permissions.
127         *
128         * @return if the clock was successfully set to the specified time.
129         */
130        native public static boolean setCurrentTimeMillis(long millis);
131    
132        /**
133         * Returns milliseconds since boot, not counting time spent in deep sleep.
134         * <b>Note:</b> This value may get reset occasionally (before it would
135         * otherwise wrap around).
136         *
137         * @return milliseconds of non-sleep uptime since boot.
138         */
139        native public static long uptimeMillis();
140    
141        /**
142         * Returns milliseconds since boot, including time spent in sleep.
143         *
144         * @return elapsed milliseconds since boot.
145         */
146        native public static long elapsedRealtime();
147        
148        /**
149         * Returns milliseconds running in the current thread.
150         * 
151         * @return elapsed milliseconds in the thread
152         */
153        public static native long currentThreadTimeMillis();
154    }
