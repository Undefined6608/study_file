017    package android.os;
018    
019    import java.util.WeakHashMap;
020    import java.util.Set;
021    import android.util.Log;
022    
023    /**
024     * Helper class that helps you use IBinder objects as reference counted
025     * tokens.  IBinders make good tokens because we find out when they are
026     * removed
027     *
028     */
029    public abstract class TokenWatcher
030    {
031        /**
032         * Construct the TokenWatcher
033         *
034         * @param h A handler to call {@link #acquired} and {@link #released}
035         * on.  If you don't care, just call it like this, although your thread
036         * will have to be a Looper thread.
037         * <code>new TokenWatcher(new Handler())</code>
038         * @param tag A debugging tag for this TokenWatcher
039         */
040        public TokenWatcher(Handler h, String tag)
041        {
042            mHandler = h;
043            mTag = tag != null ? tag : "TokenWatcher";
044        }
045    
046        /**
047         * Called when the number of active tokens goes from 0 to 1.
048         */
049        public abstract void acquired();
050    
051        /**
052         * Called when the number of active tokens goes from 1 to 0.
053         */
054        public abstract void released();
055    
056        /**
057         * Record that this token has been acquired.  When acquire is called, and
058         * the current count is 0, the acquired method is called on the given
059         * handler.
060         * 
061         * @param token An IBinder object.  If this token has already been acquired,
062         *              no action is taken.
063         * @param tag   A string used by the {@link #dump} method for debugging,
064         *              to see who has references.
065         */
066        public void acquire(IBinder token, String tag)
067        {
068            synchronized (mTokens) {
069                // explicitly checked to avoid bogus sendNotification calls because
070                // of the WeakHashMap and the GC
071                int oldSize = mTokens.size();
072    
073                Death d = new Death(token, tag);
074                try {
075                    token.linkToDeath(d, 0);
076                } catch (RemoteException e) {
077                    return;
078                }
079                mTokens.put(token, d);
080    
081                if (oldSize == 0 && !mAcquired) {
082                    sendNotificationLocked(true);
083                    mAcquired = true;
084                }
085            }
086        }
087    
088        public void cleanup(IBinder token, boolean unlink)
089        {
090            synchronized (mTokens) {
091                Death d = mTokens.remove(token);
092                if (unlink && d != null) {
093                    d.token.unlinkToDeath(d, 0);
094                    d.token = null;
095                }
096    
097                if (mTokens.size() == 0 && mAcquired) {
098                    sendNotificationLocked(false);
099                    mAcquired = false;
100                }
101            }
102        }
103    
104        public void release(IBinder token)
105        {
106            cleanup(token, true);
107        }
108    
109        public boolean isAcquired()
110        {
111            synchronized (mTokens) {
112                return mAcquired;
113            }
114        }
115    
116        public void dump()
117        {
118            synchronized (mTokens) {
119                Set<IBinder> keys = mTokens.keySet();
120                Log.i(mTag, "Token count: " + mTokens.size());
121                int i = 0;
122                for (IBinder b: keys) {
123                    Log.i(mTag, "[" + i + "] " + mTokens.get(b).tag + " - " + b);
124                    i++;
125                }
126            }
127        }
128    
129        private Runnable mNotificationTask = new Runnable() {
130            public void run()
131            {
132                int value;
133                synchronized (mTokens) {
134                    value = mNotificationQueue;
135                    mNotificationQueue = -1;
136                }
137                if (value == 1) {
138                    acquired();
139                }
140                else if (value == 0) {
141                    released();
142                }
143            }
144        };
145    
146        private void sendNotificationLocked(boolean on)
147        {
148            int value = on ? 1 : 0;
149            if (mNotificationQueue == -1) {
150                // empty
151                mNotificationQueue = value;
152                mHandler.post(mNotificationTask);
153            }
154            else if (mNotificationQueue != value) {
155                // it's a pair, so cancel it
156                mNotificationQueue = -1;
157                mHandler.removeCallbacks(mNotificationTask);
158            }
159            // else, same so do nothing -- maybe we should warn?
160        }
161    
162        private class Death implements IBinder.DeathRecipient
163        {
164            IBinder token;
165            String tag;
166    
167            Death(IBinder token, String tag)
168            {
169                this.token = token;
170                this.tag = tag;
171            }
172    
173            public void binderDied()
174            {
175                cleanup(token, false);
176            }
177    
178            protected void finalize() throws Throwable
179            {
180                try {
181                    if (token != null) {
182                        Log.w(mTag, "cleaning up leaked reference: " + tag);
183                        release(token);
184                    }
185                }
186                finally {
187                    super.finalize();
188                }
189            }
190        }
191    
192        private WeakHashMap<IBinder,Death> mTokens = new WeakHashMap<IBinder,Death>();
193        private Handler mHandler;
194        private String mTag;
195        private int mNotificationQueue = -1;
196        private volatile boolean mAcquired = false;
197    }
