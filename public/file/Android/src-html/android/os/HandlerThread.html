017    package android.os;
018    
019    /**
020     * Handy class for starting a new thread that has a looper. The looper can then be 
021     * used to create handler classes. Note that start() must still be called.
022     */
023    public class HandlerThread extends Thread {
024        private int mPriority;
025        private int mTid = -1;
026        private Looper mLooper;
027    
028        public HandlerThread(String name) {
029            super(name);
030            mPriority = Process.THREAD_PRIORITY_DEFAULT;
031        }
032        
033        /**
034         * Constructs a HandlerThread.
035         * @param name
036         * @param priority The priority to run the thread at. The value supplied must be from 
037         * {@link android.os.Process} and not from java.lang.Thread.
038         */
039        public HandlerThread(String name, int priority) {
040            super(name);
041            mPriority = priority;
042        }
043        
044        /**
045         * Call back method that can be explicitly over ridden if needed to execute some
046         * setup before Looper loops.
047         */
048        protected void onLooperPrepared() {
049        }
050    
051        public void run() {
052            mTid = Process.myTid();
053            Looper.prepare();
054            synchronized (this) {
055                mLooper = Looper.myLooper();
056                Process.setThreadPriority(mPriority);
057                notifyAll();
058            }
059            onLooperPrepared();
060            Looper.loop();
061            mTid = -1;
062        }
063        
064        /**
065         * This method returns the Looper associated with this thread. If this thread not been started
066         * or for any reason is isAlive() returns false, this method will return null. If this thread 
067         * has been started, this method will blocked until the looper has been initialized.  
068         * @return The looper.
069         */
070        public Looper getLooper() {
071            if (!isAlive()) {
072                return null;
073            }
074            
075            // If the thread has been started, wait until the looper has been created.
076            synchronized (this) {
077                while (isAlive() && mLooper == null) {
078                    try {
079                        wait();
080                    } catch (InterruptedException e) {
081                    }
082                }
083            }
084            return mLooper;
085        }
086        
087        /**
088         * Returns the identifier of this thread. See Process.myTid().
089         */
090        public int getThreadId() {
091            return mTid;
092        }
093    }
