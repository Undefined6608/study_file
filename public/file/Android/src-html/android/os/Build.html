017    package android.os;
018    
019    /**
020     * Information about the current build, extracted from system properties.
021     */
022    public class Build {
023        /** Value used for when a build property is unknown. */
024        private static final String UNKNOWN = "unknown";
025    
026        /** Either a changelist number, or a label like "M4-rc20". */
027        public static final String ID = getString("ro.build.id");
028    
029        /** A build ID string meant for displaying to the user */
030        public static final String DISPLAY = getString("ro.build.display.id");
031    
032        /** The name of the overall product. */
033        public static final String PRODUCT = getString("ro.product.name");
034    
035        /** The name of the industrial design. */
036        public static final String DEVICE = getString("ro.product.device");
037    
038        /** The name of the underlying board, like "goldfish". */
039        public static final String BOARD = getString("ro.product.board");
040    
041        /** The brand (e.g., carrier) the software is customized for, if any. */
042        public static final String BRAND = getString("ro.product.brand");
043    
044        /** The end-user-visible name for the end product. */
045        public static final String MODEL = getString("ro.product.model");
046    
047        /** Various version strings. */
048        public static class VERSION {
049            /**
050             * The internal value used by the underlying source control to
051             * represent this build.  E.g., a perforce changelist number
052             * or a git hash.
053             */
054            public static final String INCREMENTAL = getString("ro.build.version.incremental");
055    
056            /**
057             * The user-visible version string.  E.g., "1.0" or "3.4b5".
058             */
059            public static final String RELEASE = getString("ro.build.version.release");
060    
061            /**
062             * The user-visible SDK version of the framework. It is an integer starting at 1.
063             */
064            public static final String SDK = getString("ro.build.version.sdk");
065        }
066    
067        /** The type of build, like "user" or "eng". */
068        public static final String TYPE = getString("ro.build.type");
069    
070        /** Comma-separated tags describing the build, like "unsigned,debug". */
071        public static final String TAGS = getString("ro.build.tags");
072    
073        /** A string that uniquely identifies this build.  Do not attempt to parse this value. */
074        public static final String FINGERPRINT = getString("ro.build.fingerprint");
075    
076        // The following properties only make sense for internal engineering builds.
077        public static final long TIME = getLong("ro.build.date.utc") * 1000;
078        public static final String USER = getString("ro.build.user");
079        public static final String HOST = getString("ro.build.host");
080    
081        private static String getString(String property) {
082            return SystemProperties.get(property, UNKNOWN);
083        }
084    
085        private static long getLong(String property) {
086            try {
087                return Long.parseLong(SystemProperties.get(property));
088            } catch (NumberFormatException e) {
089                return -1;
090            }
091        }
092    }
