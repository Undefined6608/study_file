104        /**
105         * Constructs a Bundle containing a copy of the mappings from the given
106         * Bundle.
107         *
108         * @param b a Bundle to be copied.
109         */
110        public Bundle(Bundle b) {
111            if (b.mParcelledData != null) {
112                mParcelledData = Parcel.obtain();
113                mParcelledData.appendFrom(b.mParcelledData, 0, b.mParcelledData.dataSize());
114                mParcelledData.setDataPosition(0);
115            } else {
116                mParcelledData = null;
117            }
118    
119            if (b.mMap != null) {
120                mMap = new HashMap<String, Object>(b.mMap);
121            } else {
122                mMap = null;
123            }
124    
125            mHasFds = b.mHasFds;
126            mFdsKnown = b.mFdsKnown;
127            mClassLoader = b.mClassLoader;
128        }

149        /**
150         * If the underlying data are stored as a Parcel, unparcel them
151         * using the currently assigned class loader.
152         */
153        /* package */ synchronized void unparcel() {
154            if (mParcelledData == null) {
155                return;
156            }
157    
158            mParcelledData.setDataPosition(0);
159            Bundle b = mParcelledData.readBundleUnpacked(mClassLoader);
160            mMap = b.mMap;
161    
162            mHasFds = mParcelledData.hasFileDescriptors();
163            mFdsKnown = true;
164            
165            mParcelledData.recycle();
166            mParcelledData = null;
167        }

255        /**
256         * Reports whether the bundle contains any parcelled file descriptors.
257         */
258        public boolean hasFileDescriptors() {
259            if (!mFdsKnown) {
260                boolean fdFound = false;    // keep going until we find one or run out of data
261                
262                if (mParcelledData != null) {
263                    if (mParcelledData.hasFileDescriptors()) {
264                        fdFound = true;
265                    }
266                } else {
267                    // It's been unparcelled, so we need to walk the map
268                    Iterator<Map.Entry<String, Object>> iter = mMap.entrySet().iterator();
269                    while (!fdFound && iter.hasNext()) {
270                        Object obj = iter.next().getValue();
271                        if (obj instanceof Parcelable) {
272                            if ((((Parcelable)obj).describeContents()
273                                    & Parcelable.CONTENTS_FILE_DESCRIPTOR) != 0) {
274                                fdFound = true;
275                                break;
276                            }
277                        } else if (obj instanceof Parcelable[]) {
278                            Parcelable[] array = (Parcelable[]) obj;
279                            for (int n = array.length - 1; n >= 0; n--) {
280                                if ((array[n].describeContents()
281                                        & Parcelable.CONTENTS_FILE_DESCRIPTOR) != 0) {
282                                    fdFound = true;
283                                    break;
284                                }
285                            }
286                        } else if (obj instanceof SparseArray) {
287                            SparseArray<? extends Parcelable> array =
288                                    (SparseArray<? extends Parcelable>) obj;
289                            for (int n = array.size() - 1; n >= 0; n--) {
290                                if ((array.get(n).describeContents()
291                                        & Parcelable.CONTENTS_FILE_DESCRIPTOR) != 0) {
292                                    fdFound = true;
293                                    break;
294                                }
295                            }
296                        } else if (obj instanceof ArrayList) {
297                            ArrayList array = (ArrayList) obj;
298                            // an ArrayList here might contain either Strings or
299                            // Parcelables; only look inside for Parcelables
300                            if ((array.size() > 0)
301                                    && (array.get(0) instanceof Parcelable)) {
302                                for (int n = array.size() - 1; n >= 0; n--) {
303                                    Parcelable p = (Parcelable) array.get(n);
304                                    if (p != null && ((p.describeContents()
305                                            & Parcelable.CONTENTS_FILE_DESCRIPTOR) != 0)) {
306                                        fdFound = true;
307                                        break;
308                                    }
309                                }
310                            }
311                        }
312                    }
313                }
314    
315                mHasFds = fdFound;
316                mFdsKnown = true;
317            }
318            return mHasFds;
319        }

682        // Log a message if the value was non-null but not of the expected type
683        private void typeWarning(String key, Object value, String className,
684            Object defaultValue, ClassCastException e) {
685            StringBuilder sb = new StringBuilder();
686            sb.append("Key ");
687            sb.append(key);
688            sb.append(" expected ");
689            sb.append(className);
690            sb.append(" but value was a ");
691            sb.append(value.getClass().getName());
692            sb.append(".  The default value ");
693            sb.append(defaultValue);
694            sb.append(" was returned.");
695            Log.w(LOG_TAG, sb.toString());
696            Log.w(LOG_TAG, "Attempt to cast generated internal exception:", e);
697        }
698    
699        private void typeWarning(String key, Object value, String className,
700            ClassCastException e) {
701            typeWarning(key, value, className, "<null>", e);
702        }

704        /**
705         * Returns the value associated with the given key, or defaultValue if
706         * no mapping of the desired type exists for the given key.
707         *
708         * @param key a String
709         * @return a boolean value
710         */
711        public boolean getBoolean(String key, boolean defaultValue) {
712            unparcel();
713            Object o = mMap.get(key);
714            if (o == null) {
715                return defaultValue;
716            }
717            try {
718                return (Boolean) o;
719            } catch (ClassCastException e) {
720                typeWarning(key, o, "Boolean", defaultValue, e);
721                return defaultValue;
722            }
723        }
