017    package android.os;
018    
019    import com.android.internal.os.IResultReceiver;
020    
021    /**
022     * Generic interface for receiving a callback result from someone.  Use this
023     * by creating a subclass and implement {@link #onReceiveResult}, which you can
024     * then pass to others and send through IPC, and receive results they
025     * supply with {@link #send}.
026     */
027    public class ResultReceiver implements Parcelable {
028        final boolean mLocal;
029        final Handler mHandler;
030        
031        IResultReceiver mReceiver;
032        
033        class MyRunnable implements Runnable {
034            final int mResultCode;
035            final Bundle mResultData;
036            
037            MyRunnable(int resultCode, Bundle resultData) {
038                mResultCode = resultCode;
039                mResultData = resultData;
040            }
041            
042            public void run() {
043                onReceiveResult(mResultCode, mResultData);
044            }
045        }
046        
047        class MyResultReceiver extends IResultReceiver.Stub {
048            public void send(int resultCode, Bundle resultData) {
049                if (mHandler != null) {
050                    mHandler.post(new MyRunnable(resultCode, resultData));
051                } else {
052                    onReceiveResult(resultCode, resultData);
053                }
054            }
055        }
056        
057        /**
058         * Create a new ResultReceive to receive results.  Your
059         * {@link #onReceiveResult} method will be called from the thread running
060         * <var>handler</var> if given, or from an arbitrary thread if null.
061         */
062        public ResultReceiver(Handler handler) {
063            mLocal = true;
064            mHandler = handler;
065        }
066        
067        /**
068         * Deliver a result to this receiver.  Will call {@link #onReceiveResult},
069         * always asynchronously if the receiver has supplied a Handler in which
070         * to dispatch the result.
071         * @param resultCode Arbitrary result code to deliver, as defined by you.
072         * @param resultData Any additional data provided by you.
073         */
074        public void send(int resultCode, Bundle resultData) {
075            if (mLocal) {
076                if (mHandler != null) {
077                    mHandler.post(new MyRunnable(resultCode, resultData));
078                } else {
079                    onReceiveResult(resultCode, resultData);
080                }
081                return;
082            }
083            
084            if (mReceiver != null) {
085                try {
086                    mReceiver.send(resultCode, resultData);
087                } catch (RemoteException e) {
088                }
089            }
090        }
091        
092        /**
093         * Override to receive results delivered to this object.
094         * 
095         * @param resultCode Arbitrary result code delivered by the sender, as
096         * defined by the sender.
097         * @param resultData Any additional data provided by the sender.
098         */
099        protected void onReceiveResult(int resultCode, Bundle resultData) {
100        }
101        
102        public int describeContents() {
103            return 0;
104        }
105    
106        public void writeToParcel(Parcel out, int flags) {
107            synchronized (this) {
108                if (mReceiver == null) {
109                    mReceiver = new MyResultReceiver();
110                }
111                out.writeStrongBinder(mReceiver.asBinder());
112            }
113        }
114    
115        ResultReceiver(Parcel in) {
116            mLocal = false;
117            mHandler = null;
118            mReceiver = IResultReceiver.Stub.asInterface(in.readStrongBinder());
119        }
120        
121        public static final Parcelable.Creator<ResultReceiver> CREATOR
122                = new Parcelable.Creator<ResultReceiver>() {
123            public ResultReceiver createFromParcel(Parcel in) {
124                return new ResultReceiver(in);
125            }
126            public ResultReceiver[] newArray(int size) {
127                return new ResultReceiver[size];
128            }
129        };
130    }
