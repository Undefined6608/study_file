017    package android.text;
018    
019    /**
020     * This is the interface for text to which markup objects can be
021     * attached and detached.  Not all Spannable classes have mutable text;
022     * see {@link Editable} for that.
023     */
024    public interface Spannable
025    extends Spanned
026    {
027        /**
028         * Attach the specified markup object to the range <code>start&hellip;end</code>
029         * of the text, or move the object to that range if it was already
030         * attached elsewhere.  See {@link Spanned} for an explanation of
031         * what the flags mean.  The object can be one that has meaning only
032         * within your application, or it can be one that the text system will
033         * use to affect text display or behavior.  Some noteworthy ones are
034         * the subclasses of {@link android.text.style.CharacterStyle} and
035         * {@link android.text.style.ParagraphStyle}, and
036         * {@link android.text.TextWatcher} and
037         * {@link android.text.SpanWatcher}.
038         */
039        public void setSpan(Object what, int start, int end, int flags);
040    
041        /**
042         * Remove the specified object from the range of text to which it
043         * was attached, if any.  It is OK to remove an object that was never
044         * attached in the first place.
045         */
046        public void removeSpan(Object what);
047    
048        /**
049         * Factory used by TextView to create new Spannables.  You can subclass
050         * it to provide something other than SpannableString.
051         */
052        public static class Factory {
053            private static Spannable.Factory sInstance = new Spannable.Factory();
054    
055            /**
056             * Returns the standard Spannable Factory.
057             */ 
058            public static Spannable.Factory getInstance() {
059                return sInstance;
060            }
061    
062            /**
063             * Returns a new SpannableString from the specified CharSequence.
064             * You can override this to provide a different kind of Spannable.
065             */
066            public Spannable newSpannable(CharSequence source) {
067                return new SpannableString(source);
068            }
069        }
070    }
