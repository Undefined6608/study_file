024    /**
025     * A BoringLayout is a very simple Layout implementation for text that
026     * fits on a single line and is all left-to-right characters.
027     * You will probably never want to make one of these yourself;
028     * if you do, be sure to call {@link #isBoring} first to make sure
029     * the text meets the criteria.
030     * <p>This class is used by widgets to control text layout. You should not need
031     * to use this class directly unless you are implementing your own widget
032     * or custom display object, in which case
033     * you are encouraged to use a Layout instead of calling
034     * {@link android.graphics.Canvas#drawText(java.lang.CharSequence, int, int, float, float, android.graphics.Paint)
035     *  Canvas.drawText()} directly.</p>
036     */
037    public class BoringLayout extends Layout implements TextUtils.EllipsizeCallback {
038        public static BoringLayout make(CharSequence source,
039                            TextPaint paint, int outerwidth,
040                            Alignment align,
041                            float spacingmult, float spacingadd,
042                            BoringLayout.Metrics metrics, boolean includepad) {
043            return new BoringLayout(source, paint, outerwidth, align,
044                                    spacingmult, spacingadd, metrics,
045                                    includepad);
046        }

049                            TextPaint paint, int outerwidth,
050                            Alignment align,
051                            float spacingmult, float spacingadd,
052                            BoringLayout.Metrics metrics, boolean includepad,
053                            TextUtils.TruncateAt ellipsize, int ellipsizedWidth) {
054            return new BoringLayout(source, paint, outerwidth, align,
055                                    spacingmult, spacingadd, metrics,
056                                    includepad, ellipsize, ellipsizedWidth);
057        }

059        /**
060         * Returns a BoringLayout for the specified text, potentially reusing
061         * this one if it is already suitable.  The caller must make sure that
062         * no one is still using this Layout.
063         */
064        public BoringLayout replaceOrMake(CharSequence source, TextPaint paint,
065                                          int outerwidth, Alignment align,
066                                          float spacingmult, float spacingadd,
067                                          BoringLayout.Metrics metrics,
068                                          boolean includepad) {
069            replaceWith(source, paint, outerwidth, align, spacingmult,
070                        spacingadd);
071    
072            mEllipsizedWidth = outerwidth;
073            mEllipsizedStart = 0;
074            mEllipsizedCount = 0;
075    
076            init(source, paint, outerwidth, align, spacingmult, spacingadd,
077                 metrics, includepad, true);
078            return this;
079        }

081        /**
082         * Returns a BoringLayout for the specified text, potentially reusing
083         * this one if it is already suitable.  The caller must make sure that
084         * no one is still using this Layout.
085         */
086        public BoringLayout replaceOrMake(CharSequence source, TextPaint paint,
087                                          int outerwidth, Alignment align,
088                                          float spacingmult, float spacingadd,
089                                          BoringLayout.Metrics metrics,
090                                          boolean includepad,
091                                          TextUtils.TruncateAt ellipsize,
092                                          int ellipsizedWidth) {
093            boolean trust;
094    
095            if (ellipsize == null || ellipsize == TextUtils.TruncateAt.MARQUEE) {
096                replaceWith(source, paint, outerwidth, align, spacingmult,
097                            spacingadd);
098    
099                mEllipsizedWidth = outerwidth;
100                mEllipsizedStart = 0;
101                mEllipsizedCount = 0;
102                trust = true;
103            } else {
104                replaceWith(TextUtils.ellipsize(source, paint, ellipsizedWidth,
105                                               ellipsize, true, this),
106                            paint, outerwidth, align, spacingmult,
107                            spacingadd);
108    
109                mEllipsizedWidth = ellipsizedWidth;
110                trust = false;
111            }
112    
113            init(getText(), paint, outerwidth, align, spacingmult, spacingadd,
114                 metrics, includepad, trust);
115            return this;
116        }

118        public BoringLayout(CharSequence source,
119                            TextPaint paint, int outerwidth,
120                            Alignment align,
121                            float spacingmult, float spacingadd,
122                            BoringLayout.Metrics metrics, boolean includepad) {
123            super(source, paint, outerwidth, align, spacingmult, spacingadd);
124    
125            mEllipsizedWidth = outerwidth;
126            mEllipsizedStart = 0;
127            mEllipsizedCount = 0;
128    
129            init(source, paint, outerwidth, align, spacingmult, spacingadd,
130                 metrics, includepad, true);
131        }
132    
133        public BoringLayout(CharSequence source,
134                            TextPaint paint, int outerwidth,
135                            Alignment align,
136                            float spacingmult, float spacingadd,
137                            BoringLayout.Metrics metrics, boolean includepad,
138                            TextUtils.TruncateAt ellipsize, int ellipsizedWidth) {
139            /*
140             * It is silly to have to call super() and then replaceWith(),
141             * but we can't use "this" for the callback until the call to
142             * super() finishes.
143             */
144            super(source, paint, outerwidth, align, spacingmult, spacingadd);
145    
146            boolean trust;
147    
148            if (ellipsize == null || ellipsize == TextUtils.TruncateAt.MARQUEE) {
149                mEllipsizedWidth = outerwidth;
150                mEllipsizedStart = 0;
151                mEllipsizedCount = 0;
152                trust = true;
153            } else {
154                replaceWith(TextUtils.ellipsize(source, paint, ellipsizedWidth,
155                                               ellipsize, true, this),
156                            paint, outerwidth, align, spacingmult,
157                            spacingadd);
158    
159    
160                mEllipsizedWidth = ellipsizedWidth;
161                trust = false;
162            }
163    
164            init(getText(), paint, outerwidth, align, spacingmult, spacingadd,
165                 metrics, includepad, trust);
166        }
167    
168        /* package */ void init(CharSequence source,
169                                TextPaint paint, int outerwidth,
170                                Alignment align,
171                                float spacingmult, float spacingadd,
172                                BoringLayout.Metrics metrics, boolean includepad,
173                                boolean trustWidth) {
174            int spacing;
175    
176            if (source instanceof String && align == Layout.Alignment.ALIGN_NORMAL) {
177                mDirect = source.toString();
178            } else {
179                mDirect = null;
180            }
181    
182            mPaint = paint;
183    
184            if (includepad) {
185                spacing = metrics.bottom - metrics.top;
186            } else {
187                spacing = metrics.descent - metrics.ascent;
188            }
189    
190            if (spacingmult != 1 || spacingadd != 0) {
191                spacing = (int)(spacing * spacingmult + spacingadd + 0.5f);
192            }
193    
194            mBottom = spacing;
195    
196            if (includepad) {
197                mDesc = spacing + metrics.top;
198            } else {
199                mDesc = spacing + metrics.ascent;
200            }
201    
202            if (trustWidth) {
203                mMax = metrics.width;
204            } else {
205                /*
206                 * If we have ellipsized, we have to actually calculate the
207                 * width because the width that was passed in was for the
208                 * full text, not the ellipsized form.
209                 */
210                synchronized (sTemp) {
211                    mMax = (int) (FloatMath.ceil(Styled.measureText(paint, sTemp,
212                                                    source, 0, source.length(),
213                                                    null)));
214                }
215            }
216    
217            if (includepad) {
218                mTopPadding = metrics.top - metrics.ascent;
219                mBottomPadding = metrics.bottom - metrics.descent;
220            }
221        }

223        /**
224         * Returns null if not boring; the width, ascent, and descent if boring.
225         */
226        public static Metrics isBoring(CharSequence text,
227                                       TextPaint paint) {
228            return isBoring(text, paint, null);
229        }
230    
231        /**
232         * Returns null if not boring; the width, ascent, and descent in the
233         * provided Metrics object (or a new one if the provided one was null)
234         * if boring.
235         */
236        public static Metrics isBoring(CharSequence text, TextPaint paint,
237                                       Metrics metrics) {
238            char[] temp = TextUtils.obtain(500);
239            int len = text.length();
240            boolean boring = true;
241    
242            outer:
243            for (int i = 0; i < len; i += 500) {
244                int j = i + 500;
245    
246                if (j > len)
247                    j = len;
248    
249                TextUtils.getChars(text, i, j, temp, 0);
250    
251                int n = j - i;
252    
253                for (int a = 0; a < n; a++) {
254                    char c = temp[a];
255    
256                    if (c == '\n' || c == '\t' || c >= FIRST_RIGHT_TO_LEFT) {
257                        boring = false;
258                        break outer;
259                    }
260                }
261            }
262    
263            TextUtils.recycle(temp);
264    
265            if (boring) {
266                Metrics fm = metrics;
267                if (fm == null) {
268                    fm = new Metrics();
269                }
270        
271                int wid;
272    
273                synchronized (sTemp) {
274                    wid = (int) (FloatMath.ceil(Styled.measureText(paint, sTemp,
275                                                    text, 0, text.length(), fm)));
276                }
277                fm.width = wid;
278                return fm;
279            } else {
280                return null;
281            }
282        }

284        @Override public int getHeight() {
285            return mBottom;
286        }
287    
288        @Override public int getLineCount() {
289            return 1;
290        }
291    
292        @Override public int getLineTop(int line) {
293            if (line == 0)
294                return 0;
295            else
296                return mBottom;
297        }
298    
299        @Override public int getLineDescent(int line) {
300            return mDesc;
301        }
302    
303        @Override public int getLineStart(int line) {
304            if (line == 0)
305                return 0;
306            else
307                return getText().length();
308        }
309    
310        @Override public int getParagraphDirection(int line) {
311            return DIR_LEFT_TO_RIGHT;
312        }
313    
314        @Override public boolean getLineContainsTab(int line) {
315            return false;
316        }

319            return mMax;
320        }

323            return Layout.DIRS_ALL_LEFT_TO_RIGHT;

349        // Override draw so it will be faster.
350        @Override
351        public void draw(Canvas c, Path highlight, Paint highlightpaint,
352                         int cursorOffset) {
353            if (mDirect != null && highlight == null) {
354                c.drawText(mDirect, 0, mBottom - mDesc, mPaint);
355            } else {
356                super.draw(c, highlight, highlightpaint, cursorOffset);
357            }
358        }

360        /**
361         * Callback for the ellipsizer to report what region it ellipsized.
362         */
363        public void ellipsized(int start, int end) {
364            mEllipsizedStart = start;
365            mEllipsizedCount = end - start;
366        }

368        private static final char FIRST_RIGHT_TO_LEFT = '\u0590';
369    
370        private String mDirect;
371        private Paint mPaint;
372    
373        /* package */ int mBottom, mDesc;   // for Direct
374        private int mTopPadding, mBottomPadding;
375        private float mMax;
376        private int mEllipsizedWidth, mEllipsizedStart, mEllipsizedCount;
377    
378        private static final TextPaint sTemp =
379                                    new TextPaint();
380    
381        public static class Metrics extends Paint.FontMetricsInt {
382            public int width;
383            
384            @Override public String toString() {
385                return super.toString() + " width=" + width;
386            }
