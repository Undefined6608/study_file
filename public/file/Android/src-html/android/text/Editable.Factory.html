017    package android.text;
018    
019    /**
020     * This is the interface for text whose content and markup
021     * can be changed (as opposed
022     * to immutable text like Strings).  If you make a {@link DynamicLayout}
023     * of an Editable, the layout will be reflowed as the text is changed.
024     */
025    public interface Editable
026    extends CharSequence, GetChars, Spannable, Appendable
027    {
028        /**
029         * Replaces the specified range (<code>st&hellip;en</code>) of text in this
030         * Editable with a copy of the slice <code>start&hellip;end</code> from
031         * <code>source</code>.  The destination slice may be empty, in which case
032         * the operation is an insertion, or the source slice may be empty,
033         * in which case the operation is a deletion.
034         * <p>
035         * Before the change is committed, each filter that was set with
036         * {@link #setFilters} is given the opportunity to modify the
037         * <code>source</code> text.
038         * <p>
039         * If <code>source</code>
040         * is Spanned, the spans from it are preserved into the Editable.
041         * Existing spans within the Editable that entirely cover the replaced
042         * range are retained, but any that were strictly within the range
043         * that was replaced are removed.  As a special case, the cursor
044         * position is preserved even when the entire range where it is
045         * located is replaced.
046         * @return  a reference to this object.
047         */
048        public Editable replace(int st, int en, CharSequence source, int start, int end);
049    
050        /**
051         * Convenience for replace(st, en, text, 0, text.length())
052         * @see #replace(int, int, CharSequence, int, int)
053         */
054        public Editable replace(int st, int en, CharSequence text);
055    
056        /**
057         * Convenience for replace(where, where, text, start, end)
058         * @see #replace(int, int, CharSequence, int, int)
059         */
060        public Editable insert(int where, CharSequence text, int start, int end);
061    
062        /**
063         * Convenience for replace(where, where, text, 0, text.length());
064         * @see #replace(int, int, CharSequence, int, int)
065         */
066        public Editable insert(int where, CharSequence text);
067    
068        /**
069         * Convenience for replace(st, en, "", 0, 0)
070         * @see #replace(int, int, CharSequence, int, int)
071         */
072        public Editable delete(int st, int en);
073    
074        /**
075         * Convenience for replace(length(), length(), text, 0, text.length())
076         * @see #replace(int, int, CharSequence, int, int)
077         */
078        public Editable append(CharSequence text);
079    
080        /**
081         * Convenience for replace(length(), length(), text, start, end)
082         * @see #replace(int, int, CharSequence, int, int)
083         */
084        public Editable append(CharSequence text, int start, int end);
085    
086        /**
087         * Convenience for append(String.valueOf(text)).
088         * @see #replace(int, int, CharSequence, int, int)
089         */
090        public Editable append(char text);
091    
092        /**
093         * Convenience for replace(0, length(), "", 0, 0)
094         * @see #replace(int, int, CharSequence, int, int)
095         * Note that this clears the text, not the spans;
096         * use {@link #clearSpans} if you need that.
097         */
098        public void clear();
099    
100        /**
101         * Removes all spans from the Editable, as if by calling
102         * {@link #removeSpan} on each of them.
103         */
104        public void clearSpans();
105    
106        /**
107         * Sets the series of filters that will be called in succession
108         * whenever the text of this Editable is changed, each of which has
109         * the opportunity to limit or transform the text that is being inserted.
110         */
111        public void setFilters(InputFilter[] filters);
112    
113        /**
114         * Returns the array of input filters that are currently applied
115         * to changes to this Editable.
116         */
117        public InputFilter[] getFilters();
118    
119        /**
120         * Factory used by TextView to create new Editables.  You can subclass
121         * it to provide something other than SpannableStringBuilder.
122         */
123        public static class Factory {
124            private static Editable.Factory sInstance = new Editable.Factory();
125    
126            /**
127             * Returns the standard Editable Factory.
128             */
129            public static Editable.Factory getInstance() {
130                return sInstance;
131            }
132    
133            /**
134             * Returns a new SpannedStringBuilder from the specified
135             * CharSequence.  You can override this to provide
136             * a different kind of Spanned.
137             */
138            public Editable newEditable(CharSequence source) {
139                return new SpannableStringBuilder(source);
140            }
141        }
142    }
