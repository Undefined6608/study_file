034    /**
035        Utility class for producing strings with formatted date/time.
036    
037        <p>
038        This class takes as inputs a format string and a representation of a date/time.
039        The format string controls how the output is generated.
040        </p>
041        <p>
042        Formatting characters may be repeated in order to get more detailed representations
043        of that field.  For instance, the format character &apos;M&apos; is used to
044        represent the month.  Depending on how many times that character is repeated
045        you get a different representation.
046        </p>
047        <p>
048        For the month of September:<br/>
049        M -&gt; 9<br/>
050        MM -&gt; 09<br/>
051        MMM -&gt; Sep<br/>
052        MMMM -&gt; September
053        </p>
054        <p>
055        The effects of the duplication vary depending on the nature of the field.
056        See the notes on the individual field formatters for details.  For purely numeric
057        fields such as <code>HOUR</code> adding more copies of the designator will
058        zero-pad the value to that number of characters.
059        </p>
060        <p>
061        For 7 minutes past the hour:<br/>
062        m -&gt; 7<br/>
063        mm -&gt; 07<br/>
064        mmm -&gt; 007<br/>
065        mmmm -&gt; 0007
066        </p>
067        <p>
068        Examples for April 6, 1970 at 3:23am:<br/>
069        &quot;MM/dd/yy h:mmaa&quot; -&gt; &quot;04/06/70 3:23am&quot<br/>
070        &quot;MMM dd, yyyy h:mmaa&quot; -&gt; &quot;Apr 6, 1970 3:23am&quot<br/>
071        &quot;MMMM dd, yyyy h:mmaa&quot; -&gt; &quot;April 6, 1970 3:23am&quot<br/>
072        &quot;E, MMMM dd, yyyy h:mmaa&quot; -&gt; &quot;Mon, April 6, 1970 3:23am&<br/>
073        &quot;EEEE, MMMM dd, yyyy h:mmaa&quot; -&gt; &quot;Monday, April 6, 1970 3:23am&quot;<br/>
074        &quot;&apos;Noteworthy day: &apos;M/d/yy&quot; -&gt; &quot;Noteworthy day: 4/6/70&quot;
075     */
076

193        private static final Object sLocaleLock = new Object();
194        private static Locale sIs24HourLocale;
195        private static boolean sIs24Hour;

198        /**
199         * Returns true if user preference is set to 24-hour format.
200         * @param context the context to use for the content resolver
201         * @return true if 24 hour time format is selected, false otherwise.
202         */
203        public static boolean is24HourFormat(Context context) {
204            String value = Settings.System.getString(context.getContentResolver(),
205                    Settings.System.TIME_12_24);
206    
207            if (value == null) {
208                Locale locale = context.getResources().getConfiguration().locale;
209    
210                synchronized (sLocaleLock) {
211                    if (sIs24HourLocale != null && sIs24HourLocale.equals(locale)) {
212                        return sIs24Hour;
213                    }
214                }
215    
216                java.text.DateFormat natural =
217                    java.text.DateFormat.getTimeInstance(
218                        java.text.DateFormat.LONG, locale);
219    
220                if (natural instanceof SimpleDateFormat) {
221                    SimpleDateFormat sdf = (SimpleDateFormat) natural;
222                    String pattern = sdf.toPattern();
223    
224                    if (pattern.indexOf('H') >= 0) {
225                        value = "24";
226                    } else {
227                        value = "12";
228                    }
229                } else {
230                    value = "12";
231                }
232    
233                synchronized (sLocaleLock) {
234                    sIs24HourLocale = locale;
235                    sIs24Hour = !value.equals("12");
236                }
237            }
238    
239            boolean b24 =  !(value == null || value.equals("12"));
240            return b24;
241        }

243        /**
244         * Returns a {@link java.text.DateFormat} object that can format the time according
245         * to the current user preference. 
246         * @param context the application context
247         * @return the {@link java.text.DateFormat} object that properly formats the time.
248         */
249        public static final java.text.DateFormat getTimeFormat(Context context) {
250            boolean b24 = is24HourFormat(context);
251            int res;
252    
253            if (b24) {
254                res = R.string.twenty_four_hour_time_format;
255            } else {
256                res = R.string.twelve_hour_time_format;
257            }
258    
259            return new java.text.SimpleDateFormat(context.getString(res));
260        }

274         * Returns a {@link java.text.DateFormat} object that can format the date
275         * in long form (such as December 31, 1999) based on user preference.
276         * @param context the application context
277         * @return the {@link java.text.DateFormat} object that formats the date in long form.
278         */
279        public static final java.text.DateFormat getLongDateFormat(Context context) {
280            String value = getDateFormatString(context);
281            if (value.indexOf('M') < value.indexOf('d')) {
282                value = context.getString(R.string.full_date_month_first);
283            } else {
284                value = context.getString(R.string.full_date_day_first);
285            }
286            return new java.text.SimpleDateFormat(value);
287        }

305        /**
306         * Gets the current date format stored as a char array. The array will contain
307         * 3 elements ({@link #DATE}, {@link #MONTH}, and {@link #YEAR}) in the order    
308         * preferred by the user.
309         */    
310        public static final char[] getDateFormatOrder(Context context) {
311            char[] order = new char[] {DATE, MONTH, YEAR};
312            String value = getDateFormatString(context);
313            int index = 0;
314            boolean foundDate = false;
315            boolean foundMonth = false;
316            boolean foundYear = false;
317    
318            for (char c : value.toCharArray()) {
319                if (!foundDate && (c == DATE)) {
320                    foundDate = true;
321                    order[index] = DATE;
322                    index++;
323                }
324    
325                if (!foundMonth && (c == MONTH)) {
326                    foundMonth = true;
327                    order[index] = MONTH;
328                    index++;
329                }
330    
331                if (!foundYear && (c == YEAR)) {
332                    foundYear = true;
333                    order[index] = YEAR;
334                    index++;
335                }
336            }
337            return order;
338        }

340        private static String getDateFormatString(Context context) {
341            String value = Settings.System.getString(context.getContentResolver(),
342                    Settings.System.DATE_FORMAT);
343            if (value == null || value.length() < 6) {
344                /*
345                 * No need to localize -- this is an emergency fallback in case
346                 * the setting is missing, but it should always be set.
347                 */
348                value = "MM-dd-yyyy";
349            }
350            return value;
351        }

376            return format(inFormat, c);

379        /**
380         * Given a format string and a {@link java.util.Calendar} object, returns a CharSequence 
381         * containing the requested date.
382         * @param inFormat the format string, as described in {@link android.text.format.DateFormat}
383         * @param inDate the date to format
384         * @return a {@link CharSequence} containing the requested text
385         */
386        public static final CharSequence format(CharSequence inFormat, Calendar inDate) {
387            SpannableStringBuilder      s = new SpannableStringBuilder(inFormat);
388            int             c;
389            int             count;
390    
391            int len = inFormat.length();
392    
393            for (int i = 0; i < len; i += count) {
394                int temp;
395    
396                count = 1;
397                c = s.charAt(i);
398    
399                if (c == QUOTE) {
400                    count = appendQuotedText(s, i, len);
401                    len = s.length();
402                    continue;
403                }
404    
405                while ((i + count < len) && (s.charAt(i + count) == c)) {
406                    count++;
407                }
408    
409                String replacement;
410    
411                switch (c) {
412                    case AM_PM:
413                        replacement = DateUtils.getAMPMString(inDate.get(Calendar.AM_PM));
414                        break;
415                                            
416                    case CAPITAL_AM_PM:
417                        //FIXME: this is the same as AM_PM? no capital?
418                        replacement = DateUtils.getAMPMString(inDate.get(Calendar.AM_PM));
419                        break;
420                    
421                    case DATE:
422                        replacement = zeroPad(inDate.get(Calendar.DATE), count);
423                        break;
424                        
425                    case DAY:
426                        temp = inDate.get(Calendar.DAY_OF_WEEK);
427                        replacement = DateUtils.getDayOfWeekString(temp,
428                                                                   count < 4 ? 
429                                                                   DateUtils.LENGTH_MEDIUM : 
430                                                                   DateUtils.LENGTH_LONG);
431                        break;
432                        
433                    case HOUR:
434                        temp = inDate.get(Calendar.HOUR);
435    
436                        if (0 == temp)
437                            temp = 12;
438                        
439                        replacement = zeroPad(temp, count);
440                        break;
441                        
442                    case HOUR_OF_DAY:
443                        replacement = zeroPad(inDate.get(Calendar.HOUR_OF_DAY), count);
444                        break;
445                        
446                    case MINUTE:
447                        replacement = zeroPad(inDate.get(Calendar.MINUTE), count);
448                        break;
449                        
450                    case MONTH:
451                        replacement = getMonthString(inDate, count);
452                        break;
453                        
454                    case SECONDS:
455                        replacement = zeroPad(inDate.get(Calendar.SECOND), count);
456                        break;
457                        
458                    case TIME_ZONE:
459                        replacement = getTimeZoneString(inDate, count);
460                        break;
461                        
462                    case YEAR:
463                        replacement = getYearString(inDate, count);
464                        break;
465    
466                    default:
467                        replacement = null;
468                        break;
469                }
470                
471                if (replacement != null) {
472                    s.replace(i, i + count, replacement);
473                    count = replacement.length(); // CARE: count is used in the for loop above
474                    len = s.length();
475                }
476            }
477            
478            if (inFormat instanceof Spanned)
479                return new SpannedString(s);
480            else
481                return s.toString();
482        }

484        private static final String getMonthString(Calendar inDate, int count) {
485            int month = inDate.get(Calendar.MONTH);
486            
487            if (count >= 4)
488                return DateUtils.getMonthString(month, DateUtils.LENGTH_LONG);
489            else if (count == 3)
490                return DateUtils.getMonthString(month, DateUtils.LENGTH_MEDIUM);
491            else {
492                // Calendar.JANUARY == 0, so add 1 to month.
493                return zeroPad(month+1, count);
494            }
495        }

497        private static final String getTimeZoneString(Calendar inDate, int count) {
498            TimeZone tz = inDate.getTimeZone();
499            
500            if (count < 2) { // FIXME: shouldn't this be <= 2 ?
501                return formatZoneOffset(inDate.get(Calendar.DST_OFFSET) +
502                                        inDate.get(Calendar.ZONE_OFFSET),
503                                        count);
504            } else {
505                boolean dst = inDate.get(Calendar.DST_OFFSET) != 0;
506                return tz.getDisplayName(dst, TimeZone.SHORT);
507            }
508        }
509    
510        private static final String formatZoneOffset(int offset, int count) {
511            offset /= 1000; // milliseconds to seconds
512            StringBuilder tb = new StringBuilder();
513    
514            if (offset < 0) {
515                tb.insert(0, "-");
516                offset = -offset;
517            } else {
518                tb.insert(0, "+");
519            }
520    
521            int hours = offset / 3600;
522            int minutes = (offset % 3600) / 60;
523    
524            tb.append(zeroPad(hours, 2));
525            tb.append(zeroPad(minutes, 2));
526            return tb.toString();
527        }
528        
529        private static final String getYearString(Calendar inDate, int count) {
530            int year = inDate.get(Calendar.YEAR);
531            return (count <= 2) ? zeroPad(year % 100, 2) : String.valueOf(year);
532        }

534        private static final int appendQuotedText(SpannableStringBuilder s, int i, int len) {
535            if (i + 1 < len && s.charAt(i + 1) == QUOTE) {
536                s.delete(i, i + 1);
537                return 1;
538            }
539    
540            int count = 0;
541    
542            // delete leading quote
543            s.delete(i, i + 1);
544            len--;
545    
546            while (i < len) {
547                char c = s.charAt(i);
548    
549                if (c == QUOTE) {
550                    //  QUOTEQUOTE -> QUOTE
551                    if (i + 1 < len && s.charAt(i + 1) == QUOTE) {
552    
553                        s.delete(i, i + 1);
554                        len--;
555                        count++;
556                        i++;
557                    } else {
558                        //  Closing QUOTE ends quoted text copying
559                        s.delete(i, i + 1);
560                        break;
561                    }
562                } else {
563                    i++;
564                    count++;
565                }
566            }
567    
568            return count;
569        }

571        private static final String zeroPad(int inValue, int inMinDigits) {
572            String val = String.valueOf(inValue);
573    
574            if (val.length() < inMinDigits) {
575                char[] buf = new char[inMinDigits];
576    
577                for (int i = 0; i < inMinDigits; i++)
578                    buf[i] = '0';
579    
580                val.getChars(0, val.length(), buf, inMinDigits - val.length());
581                val = new String(buf);
582            }
583            return val;
584        }
