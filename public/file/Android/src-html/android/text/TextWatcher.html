017    package android.text;
018    
019    /**
020     * When an object of a type is attached to an Editable, its methods will
021     * be called when the text is changed.
022     */
023    public interface TextWatcher extends NoCopySpan {
024        /**
025         * This method is called to notify you that, within <code>s</code>,
026         * the <code>count</code> characters beginning at <code>start</code>
027         * are about to be replaced by new text with length <code>after</code>.
028         * It is an error to attempt to make changes to <code>s</code> from
029         * this callback.
030         */
031        public void beforeTextChanged(CharSequence s, int start,
032                                      int count, int after);
033        /**
034         * This method is called to notify you that, within <code>s</code>,
035         * the <code>count</code> characters beginning at <code>start</code>
036         * have just replaced old text that had length <code>before</code>.
037         * It is an error to attempt to make changes to <code>s</code> from
038         * this callback.
039         */
040        public void onTextChanged(CharSequence s, int start, int before, int count);
041    
042        /**
043         * This method is called to notify you that, somewhere within
044         * <code>s</code>, the text has been changed.
045         * It is legitimate to make further changes to <code>s</code> from
046         * this callback, but be careful not to get yourself into an infinite
047         * loop, because any changes you make will cause this method to be
048         * called again recursively.
049         * (You are not told where the change took place because other
050         * afterTextChanged() methods may already have made other changes
051         * and invalidated the offsets.  But if you need to know here,
052         * you can use {@link Spannable#setSpan} in {@link #onTextChanged}
053         * to mark your place and then look up from here where the span
054         * ended up.
055         */
056        public void afterTextChanged(Editable s);
057    }
