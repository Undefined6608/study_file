017    package android.text.style;
018    
019    import android.graphics.Paint;
020    import android.text.TextPaint;
021    
022    /**
023     * The classes that affect character-level text formatting in a way that
024     * changes the width or height of characters extend this class.
025     */
026    public abstract class MetricAffectingSpan
027    extends CharacterStyle
028    implements UpdateLayout {
029    
030            public abstract void updateMeasureState(TextPaint p);
031    
032        /**
033         * Returns "this" for most MetricAffectingSpans, but for 
034         * MetricAffectingSpans that were generated by {@link #wrap},
035         * returns the underlying MetricAffectingSpan.
036         */
037        @Override
038        public MetricAffectingSpan getUnderlying() {
039            return this;
040        }
041    
042        /**
043         * A Passthrough MetricAffectingSpan is one that
044         * passes {@link #updateDrawState} and {@link #updateMeasureState}
045         * calls through to the specified MetricAffectingSpan 
046         * while still being a distinct object,
047         * and is therefore able to be attached to the same Spannable
048         * to which the specified MetricAffectingSpan is already attached.
049         */
050        /* package */ static class Passthrough extends MetricAffectingSpan {
051            private MetricAffectingSpan mStyle;
052            
053            /**
054             * Creates a new Passthrough of the specfied MetricAffectingSpan.
055             */
056            public Passthrough(MetricAffectingSpan cs) {
057                mStyle = cs;
058            }
059    
060            /**
061             * Passes updateDrawState through to the underlying MetricAffectingSpan.
062             */
063            @Override
064            public void updateDrawState(TextPaint tp) {
065                mStyle.updateDrawState(tp);
066            }
067    
068            /**
069             * Passes updateMeasureState through to the underlying MetricAffectingSpan.
070             */
071            @Override
072            public void updateMeasureState(TextPaint tp) {
073                mStyle.updateMeasureState(tp);
074            }
075        
076            /**
077             * Returns the MetricAffectingSpan underlying this one, or the one
078             * underlying it if it too is a Passthrough.
079             */
080            @Override
081            public MetricAffectingSpan getUnderlying() {
082                return mStyle.getUnderlying();
083            }
084        }
085    }
