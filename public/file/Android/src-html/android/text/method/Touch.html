017    package android.text.method;
018    
019    import android.text.Layout;
020    import android.text.NoCopySpan;
021    import android.text.Layout.Alignment;
022    import android.text.Spannable;
023    import android.view.MotionEvent;
024    import android.view.ViewConfiguration;
025    import android.widget.TextView;
026    
027    public class Touch {
028        private Touch() { }
029    
030        /**
031         * Scrolls the specified widget to the specified coordinates, except
032         * constrains the X scrolling position to the horizontal regions of
033         * the text that will be visible after scrolling to the specified
034         * Y position.
035         */
036        public static void scrollTo(TextView widget, Layout layout, int x, int y) {
037            int padding = widget.getTotalPaddingTop() +
038                          widget.getTotalPaddingBottom();
039            int top = layout.getLineForVertical(y);
040            int bottom = layout.getLineForVertical(y + widget.getHeight() -
041                                                   padding);
042    
043            int left = Integer.MAX_VALUE;
044            int right = 0;
045            Alignment a = null;
046    
047            for (int i = top; i <= bottom; i++) {
048                left = (int) Math.min(left, layout.getLineLeft(i));
049                right = (int) Math.max(right, layout.getLineRight(i));
050    
051                if (a == null) {
052                    a = layout.getParagraphAlignment(i);
053                }
054            }
055    
056            padding = widget.getTotalPaddingLeft() + widget.getTotalPaddingRight();
057            int width = widget.getWidth();
058            int diff = 0;
059    
060            if (right - left < width - padding) {
061                if (a == Alignment.ALIGN_CENTER) {
062                    diff = (width - padding - (right - left)) / 2;
063                } else if (a == Alignment.ALIGN_OPPOSITE) {
064                    diff = width - padding - (right - left);
065                }
066            }
067    
068            x = Math.min(x, right - (width - padding) - diff);
069            x = Math.max(x, left - diff);
070    
071            widget.scrollTo(x, y);
072        }
073    
074        /**
075         * Handles touch events for dragging.  You may want to do other actions
076         * like moving the cursor on touch as well.
077         */
078        public static boolean onTouchEvent(TextView widget, Spannable buffer,
079                                           MotionEvent event) {
080            DragState[] ds;
081    
082            switch (event.getAction()) {
083            case MotionEvent.ACTION_DOWN:
084                buffer.setSpan(new DragState(event.getX(), event.getY(),
085                                widget.getScrollX(), widget.getScrollY()),
086                        0, 0, Spannable.SPAN_MARK_MARK);
087                return true;
088    
089            case MotionEvent.ACTION_UP:
090                ds = buffer.getSpans(0, buffer.length(), DragState.class);
091    
092                for (int i = 0; i < ds.length; i++) {
093                    buffer.removeSpan(ds[i]);
094                }
095    
096                if (ds.length > 0 && ds[0].mUsed) {
097                    return true;
098                } else {
099                    return false;
100                }
101    
102            case MotionEvent.ACTION_MOVE:
103                ds = buffer.getSpans(0, buffer.length(), DragState.class);
104    
105                if (ds.length > 0) {
106                    if (ds[0].mFarEnough == false) {
107                        int slop = ViewConfiguration.get(widget.getContext()).getScaledTouchSlop();
108    
109                        if (Math.abs(event.getX() - ds[0].mX) >= slop ||
110                            Math.abs(event.getY() - ds[0].mY) >= slop) {
111                            ds[0].mFarEnough = true;
112                        }
113                    }
114    
115                    if (ds[0].mFarEnough) {
116                        ds[0].mUsed = true;
117    
118                        float dx = ds[0].mX - event.getX();
119                        float dy = ds[0].mY - event.getY();
120    
121                        ds[0].mX = event.getX();
122                        ds[0].mY = event.getY();
123    
124                        int nx = widget.getScrollX() + (int) dx;
125                        int ny = widget.getScrollY() + (int) dy;
126    
127                        int padding = widget.getTotalPaddingTop() +
128                                      widget.getTotalPaddingBottom();
129                        Layout layout = widget.getLayout();
130    
131                        ny = Math.min(ny, layout.getHeight() - (widget.getHeight() -
132                                                                padding));
133                        ny = Math.max(ny, 0);
134            
135                        scrollTo(widget, layout, nx, ny);
136                        widget.cancelLongPress();
137                        return true;
138                    }
139                }
140            }
141    
142            return false;
143        }
144    
145        public static int getInitialScrollX(TextView widget, Spannable buffer) {
146            DragState[] ds = buffer.getSpans(0, buffer.length(), DragState.class);
147            return ds.length > 0 ? ds[0].mScrollX : -1;
148        }
149        
150        public static int getInitialScrollY(TextView widget, Spannable buffer) {
151            DragState[] ds = buffer.getSpans(0, buffer.length(), DragState.class);
152            return ds.length > 0 ? ds[0].mScrollY : -1;
153        }
154        
155        private static class DragState implements NoCopySpan {
156            public float mX;
157            public float mY;
158            public int mScrollX;
159            public int mScrollY;
160            public boolean mFarEnough;
161            public boolean mUsed;
162    
163            public DragState(float x, float y, int scrollX, int scrollY) {
164                mX = x;
165                mY = y;
166                mScrollX = scrollX;
167                mScrollY = scrollY;
168            }
169        }
