017    package android.text.method;
018    
019    import android.graphics.Rect;
020    import android.view.View;
021    import android.widget.TextView;
022    
023    /**
024     * TextView uses TransformationMethods to do things like replacing the
025     * characters of passwords with dots, or keeping the newline characters
026     * from causing line breaks in single-line text fields.
027     */
028    public interface TransformationMethod
029    {
030        /**
031         * Returns a CharSequence that is a transformation of the source text --
032         * for example, replacing each character with a dot in a password field.
033         * Beware that the returned text must be exactly the same length as
034         * the source text, and that if the source text is Editable, the returned
035         * text must mirror it dynamically instead of doing a one-time copy.
036         */
037        public CharSequence getTransformation(CharSequence source, View view);
038    
039        /**
040         * This method is called when the TextView that uses this
041         * TransformationMethod gains or loses focus.
042         */
043        public void onFocusChanged(View view, CharSequence sourceText,
044                                   boolean focused, int direction,
045                                   Rect previouslyFocusedRect);
046    }
