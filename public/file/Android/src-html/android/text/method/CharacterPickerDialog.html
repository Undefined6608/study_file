017    package android.text.method;
018    
019    import com.android.internal.R;
020    
021    import android.app.Dialog;
022    import android.content.Context;
023    import android.os.Bundle;
024    import android.text.*;
025    import android.view.LayoutInflater;
026    import android.view.View.OnClickListener;
027    import android.view.View;
028    import android.view.ViewGroup.LayoutParams;
029    import android.view.ViewGroup;
030    import android.view.WindowManager;
031    import android.widget.AdapterView.OnItemClickListener;
032    import android.widget.AdapterView;
033    import android.widget.BaseAdapter;
034    import android.widget.Button;
035    import android.widget.GridView;
036    import android.widget.TextView;
037    
038    /**
039     * Dialog for choosing accented characters related to a base character.
040     */
041    public class CharacterPickerDialog extends Dialog
042            implements OnItemClickListener, OnClickListener {
043        private View mView;
044        private Editable mText;
045        private String mOptions;
046        private boolean mInsert;
047        private LayoutInflater mInflater;
048    
049        /**
050         * Creates a new CharacterPickerDialog that presents the specified
051         * <code>options</code> for insertion or replacement (depending on
052         * the sense of <code>insert</code>) into <code>text</code>.
053         */
054        public CharacterPickerDialog(Context context, View view,
055                                     Editable text, String options,
056                                     boolean insert) {
057            super(context);
058    
059            mView = view;
060            mText = text;
061            mOptions = options;
062            mInsert = insert;
063            mInflater = LayoutInflater.from(context);
064        }
065    
066        @Override
067        protected void onCreate(Bundle savedInstanceState) {
068            super.onCreate(savedInstanceState);
069    
070            WindowManager.LayoutParams params = getWindow().getAttributes();
071            params.token = mView.getApplicationWindowToken();
072            params.type = params.TYPE_APPLICATION_ATTACHED_DIALOG;
073    
074            setTitle(R.string.select_character);
075            setContentView(R.layout.character_picker);
076    
077            GridView grid = (GridView) findViewById(R.id.characterPicker);
078            grid.setAdapter(new OptionsAdapter(getContext()));
079            grid.setOnItemClickListener(this);
080    
081            findViewById(R.id.cancel).setOnClickListener(this);
082        }
083    
084        /**
085         * Handles clicks on the character buttons.
086         */
087        public void onItemClick(AdapterView parent, View view, int position, long id) {
088            int selEnd = Selection.getSelectionEnd(mText);
089            String result = String.valueOf(mOptions.charAt(position));
090    
091            if (mInsert || selEnd == 0) {
092                mText.insert(selEnd, result);
093            } else {
094                mText.replace(selEnd - 1, selEnd, result);
095            }
096    
097            dismiss();
098        }
099    
100        /**
101         * Handles clicks on the Cancel button.
102         */
103        public void onClick(View v) {
104            dismiss();
105        }
106    
107        private class OptionsAdapter extends BaseAdapter {
108            private Context mContext;
109    
110            public OptionsAdapter(Context context) {
111                super();
112                mContext = context;
113            }
114    
115            public View getView(int position, View convertView, ViewGroup parent) {
116                Button b = (Button)
117                    mInflater.inflate(R.layout.character_picker_button, null);
118                b.setText(String.valueOf(mOptions.charAt(position)));
119                return b;
120            }
121    
122            public final int getCount() {
123                return mOptions.length();
124            }
125    
126            public final Object getItem(int position) {
127                return String.valueOf(mOptions.charAt(position));
128            }
129    
130            public final long getItemId(int position) {
131                return position;
132            }
133        }
134    }
