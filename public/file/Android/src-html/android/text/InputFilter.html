017    package android.text;
018    
019    /**
020     * InputFilters can be attached to {@link Editable}s to constrain the
021     * changes that can be made to them.
022     */
023    public interface InputFilter
024    {
025        /**
026         * This method is called when the buffer is going to replace the
027         * range <code>dstart &hellip; dend</code> of <code>dest</code>
028         * with the new text from the range <code>start &hellip; end</code>
029         * of <code>source</code>.  Return the CharSequence that you would
030         * like to have placed there instead, including an empty string
031         * if appropriate, or <code>null</code> to accept the original
032         * replacement.  Be careful to not to reject 0-length replacements,
033         * as this is what happens when you delete text.  Also beware that
034         * you should not attempt to make any changes to <code>dest</code>
035         * from this method; you may only examine it for context.
036         * 
037         * Note: If <var>source</var> is an instance of {@link Spanned} or
038         * {@link Spannable}, the span objects in the <var>source</var> should be 
039         * copied into the filtered result (i.e. the non-null return value). 
040         * {@link TextUtils#copySpansFrom} can be used for convenience.
041         */
042        public CharSequence filter(CharSequence source, int start, int end,
043                                   Spanned dest, int dstart, int dend);
044    
045        /**
046         * This filter will capitalize all the lower case letters that are added
047         * through edits.
048         */
049        public static class AllCaps implements InputFilter {
050            public CharSequence filter(CharSequence source, int start, int end,
051                                       Spanned dest, int dstart, int dend) {
052                for (int i = start; i < end; i++) {
053                    if (Character.isLowerCase(source.charAt(i))) {
054                        char[] v = new char[end - start];
055                        TextUtils.getChars(source, start, end, v, 0);
056                        String s = new String(v).toUpperCase();
057    
058                        if (source instanceof Spanned) {
059                            SpannableString sp = new SpannableString(s);
060                            TextUtils.copySpansFrom((Spanned) source,
061                                                    start, end, null, sp, 0);
062                            return sp;
063                        } else {
064                            return s;
065                        }
066                    }
067                }
068    
069                return null; // keep original
070            }
071        }
072    
073        /**
074         * This filter will constrain edits not to make the length of the text
075         * greater than the specified length.
076         */
077        public static class LengthFilter implements InputFilter {
078            public LengthFilter(int max) {
079                mMax = max;
080            }
081    
082            public CharSequence filter(CharSequence source, int start, int end,
083                                       Spanned dest, int dstart, int dend) {
084                int keep = mMax - (dest.length() - (dend - dstart));
085    
086                if (keep <= 0) {
087                    return "";
088                } else if (keep >= end - start) {
089                    return null; // keep original
090                } else {
091                    return source.subSequence(start, start + keep);
092                }
093            }
094    
095            private int mMax;
096        }
097    }
