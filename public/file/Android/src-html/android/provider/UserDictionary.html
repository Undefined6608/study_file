017    package android.provider;
018    
019    import java.util.Locale;
020    
021    import android.content.ContentResolver;
022    import android.content.ContentValues;
023    import android.content.Context;
024    import android.net.Uri;
025    import android.text.TextUtils;
026    
027    /**
028     * A provider of user defined words for input methods to use for predictive text input.
029     * Applications and input methods may add words into the dictionary. Words can have associated
030     * frequency information and locale information.
031     */
032    public class UserDictionary {
033    
034        /** Authority string for this provider. */
035        public static final String AUTHORITY = "user_dictionary";
036    
037        /**
038         * The content:// style URL for this provider
039         */
040        public static final Uri CONTENT_URI =
041            Uri.parse("content://" + AUTHORITY);
042    
043        /**
044         * Contains the user defined words.
045         */
046        public static class Words implements BaseColumns {
047            /**
048             * The content:// style URL for this table
049             */
050            public static final Uri CONTENT_URI =
051                    Uri.parse("content://" + AUTHORITY + "/words");
052    
053            /**
054             * The MIME type of {@link #CONTENT_URI} providing a directory of words.
055             */
056            public static final String CONTENT_TYPE = "vnd.android.cursor.dir/vnd.google.userword";
057    
058            /**
059             * The MIME type of a {@link #CONTENT_URI} sub-directory of a single word.
060             */
061            public static final String CONTENT_ITEM_TYPE = "vnd.android.cursor.item/vnd.google.userword";
062    
063            public static final String _ID = BaseColumns._ID;
064    
065            /**
066             * The word column.
067             * <p>TYPE: TEXT</p>
068             */
069            public static final String WORD = "word";
070    
071            /**
072             * The frequency column. A value between 1 and 255. Higher values imply higher frequency.
073             * <p>TYPE: INTEGER</p>
074             */
075            public static final String FREQUENCY = "frequency";
076    
077            /**
078             * The locale that this word belongs to. Null if it pertains to all
079             * locales. Locale is as defined by the string returned by Locale.toString().
080             * <p>TYPE: TEXT</p>
081             */
082            public static final String LOCALE = "locale";
083    
084            /**
085             * The uid of the application that inserted the word.
086             * <p>TYPE: INTEGER</p>
087             */
088            public static final String APP_ID = "appid";
089    
090            /** The locale type to specify that the word is common to all locales. */
091            public static final int LOCALE_TYPE_ALL = 0;
092            
093            /** The locale type to specify that the word is for the current locale. */
094            public static final int LOCALE_TYPE_CURRENT = 1;
095            
096            /**
097             * Sort by descending order of frequency.
098             */
099            public static final String DEFAULT_SORT_ORDER = FREQUENCY + " DESC";
100    
101            /** Adds a word to the dictionary, with the given frequency and the specified
102             *  specified locale type.
103             *  @param context the current application context
104             *  @param word the word to add to the dictionary. This should not be null or
105             *  empty.
106             *  @param localeType the locale type for this word. It should be one of
107             *  {@link #LOCALE_TYPE_ALL} or {@link #LOCALE_TYPE_CURRENT}.
108             */
109            public static void addWord(Context context, String word, 
110                    int frequency, int localeType) {
111                final ContentResolver resolver = context.getContentResolver();
112    
113                if (TextUtils.isEmpty(word) || localeType < 0 || localeType > 1) {
114                    return;
115                }
116                
117                if (frequency < 0) frequency = 0;
118                if (frequency > 255) frequency = 255;
119    
120                String locale = null;
121    
122                // TODO: Verify if this is the best way to get the current locale
123                if (localeType == LOCALE_TYPE_CURRENT) {
124                    locale = Locale.getDefault().toString();
125                }
126                ContentValues values = new ContentValues(4);
127    
128                values.put(WORD, word);
129                values.put(FREQUENCY, frequency);
130                values.put(LOCALE, locale);
131                values.put(APP_ID, 0); // TODO: Get App UID
132    
133                Uri result = resolver.insert(CONTENT_URI, values);
134                // It's ok if the insert doesn't succeed because the word
135                // already exists.
136            }
137        }
138    }
