017    package android.widget;
018    
019    import android.database.DataSetObserver;
020    import android.view.KeyEvent;
021    import android.view.View;
022    import android.view.ViewGroup;
023    
024    /**
025     * An adapter that links a {@link ExpandableListView} with the underlying
026     * data. The implementation of this interface will provide access
027     * to the data of the children (categorized by groups), and also instantiate
028     * {@link View}s for children and groups.
029     */
030    public interface ExpandableListAdapter {
031        /**
032         * @see Adapter#registerDataSetObserver(DataSetObserver)
033         */
034        void registerDataSetObserver(DataSetObserver observer);
035    
036        /**
037         * @see Adapter#unregisterDataSetObserver(DataSetObserver)
038         */
039        void unregisterDataSetObserver(DataSetObserver observer);
040    
041        /**
042         * Gets the number of groups.
043         * 
044         * @return the number of groups
045         */
046        int getGroupCount();
047    
048        /**
049         * Gets the number of children in a specified group.
050         * 
051         * @param groupPosition the position of the group for which the children
052         *            count should be returned
053         * @return the children count in the specified group
054         */
055        int getChildrenCount(int groupPosition);
056    
057        /**
058         * Gets the data associated with the given group.
059         * 
060         * @param groupPosition the position of the group
061         * @return the data child for the specified group
062         */
063        Object getGroup(int groupPosition);
064        
065        /**
066         * Gets the data associated with the given child within the given group.
067         * 
068         * @param groupPosition the position of the group that the child resides in
069         * @param childPosition the position of the child with respect to other
070         *            children in the group
071         * @return the data of the child
072         */
073        Object getChild(int groupPosition, int childPosition);
074    
075        /**
076         * Gets the ID for the group at the given position. This group ID must be
077         * unique across groups. The combined ID (see
078         * {@link #getCombinedGroupId(long)}) must be unique across ALL items
079         * (groups and all children).
080         * 
081         * @param groupPosition the position of the group for which the ID is wanted
082         * @return the ID associated with the group
083         */
084        long getGroupId(int groupPosition);
085    
086        /**
087         * Gets the ID for the given child within the given group. This ID must be
088         * unique across all children within the group. The combined ID (see
089         * {@link #getCombinedChildId(long, long)}) must be unique across ALL items
090         * (groups and all children).
091         * 
092         * @param groupPosition the position of the group that contains the child
093         * @param childPosition the position of the child within the group for which
094         *            the ID is wanted
095         * @return the ID associated with the child
096         */
097        long getChildId(int groupPosition, int childPosition);
098    
099        /**
100         * Indicates whether the child and group IDs are stable across changes to the
101         * underlying data.
102         * 
103         * @return whether or not the same ID always refers to the same object
104         * @see Adapter#hasStableIds()
105         */
106        boolean hasStableIds();
107    
108        /**
109         * Gets a View that displays the given group. This View is only for the
110         * group--the Views for the group's children will be fetched using
111         * getChildrenView.
112         * 
113         * @param groupPosition the position of the group for which the View is
114         *            returned
115         * @param isExpanded whether the group is expanded or collapsed
116         * @param convertView the old view to reuse, if possible. You should check
117         *            that this view is non-null and of an appropriate type before
118         *            using. If it is not possible to convert this view to display
119         *            the correct data, this method can create a new view. It is not
120         *            guaranteed that the convertView will have been previously
121         *            created by
122         *            {@link #getGroupView(int, boolean, View, ViewGroup)}.
123         * @param parent the parent that this view will eventually be attached to
124         * @return the View corresponding to the group at the specified position
125         */
126        View getGroupView(int groupPosition, boolean isExpanded, View convertView, ViewGroup parent);
127    
128        /**
129         * Gets a View that displays the data for the given child within the given
130         * group.
131         * 
132         * @param groupPosition the position of the group that contains the child
133         * @param childPosition the position of the child (for which the View is
134         *            returned) within the group
135         * @param isLastChild Whether the child is the last child within the group
136         * @param convertView the old view to reuse, if possible. You should check
137         *            that this view is non-null and of an appropriate type before
138         *            using. If it is not possible to convert this view to display
139         *            the correct data, this method can create a new view. It is not
140         *            guaranteed that the convertView will have been previously
141         *            created by
142         *            {@link #getChildView(int, int, boolean, View, ViewGroup)}.
143         * @param parent the parent that this view will eventually be attached to
144         * @return the View corresponding to the child at the specified position
145         */
146        View getChildView(int groupPosition, int childPosition, boolean isLastChild,
147                View convertView, ViewGroup parent);
148    
149        /**
150         * Whether the child at the specified position is selectable.
151         * 
152         * @param groupPosition the position of the group that contains the child
153         * @param childPosition the position of the child within the group
154         * @return whether the child is selectable.
155         */
156        boolean isChildSelectable(int groupPosition, int childPosition);
157    
158        /**
159         * @see ListAdapter#areAllItemsEnabled()
160         */
161        boolean areAllItemsEnabled();
162        
163        /**
164         * @see ListAdapter#isEmpty()
165         */
166        boolean isEmpty();
167    
168        /**
169         * Called when a group is expanded.
170         * 
171         * @param groupPosition The group being expanded.
172         */
173        void onGroupExpanded(int groupPosition);
174        
175        /**
176         * Called when a group is collapsed.
177         * 
178         * @param groupPosition The group being collapsed.
179         */
180        void onGroupCollapsed(int groupPosition);
181        
182        /**
183         * Gets an ID for a child that is unique across any item (either group or
184         * child) that is in this list. Expandable lists require each item (group or
185         * child) to have a unique ID among all children and groups in the list.
186         * This method is responsible for returning that unique ID given a child's
187         * ID and its group's ID. Furthermore, if {@link #hasStableIds()} is true, the
188         * returned ID must be stable as well.
189         * 
190         * @param groupId The ID of the group that contains this child.
191         * @param childId The ID of the child.
192         * @return The unique (and possibly stable) ID of the child across all
193         *         groups and children in this list.
194         */
195        long getCombinedChildId(long groupId, long childId);
196    
197        /**
198         * Gets an ID for a group that is unique across any item (either group or
199         * child) that is in this list. Expandable lists require each item (group or
200         * child) to have a unique ID among all children and groups in the list.
201         * This method is responsible for returning that unique ID given a group's
202         * ID. Furthermore, if {@link #hasStableIds()} is true, the returned ID must be
203         * stable as well.
204         * 
205         * @param groupId The ID of the group
206         * @return The unique (and possibly stable) ID of the group across all
207         *         groups and children in this list.
208         */
209        long getCombinedGroupId(long groupId);
210    }
