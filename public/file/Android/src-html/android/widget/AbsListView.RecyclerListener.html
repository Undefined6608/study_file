438         * Interface definition for a callback to be invoked when the list or grid
439         * has been scrolled.

441        public interface OnScrollListener {
442    
443            /**
444             * The view is not scrolling. Note navigating the list using the trackball counts as
445             * being in the idle state since these transitions are not animated.
446             */
447            public static int SCROLL_STATE_IDLE = 0;
448    
449            /**
450             * The user is scrolling using touch, and their finger is still on the screen
451             */
452            public static int SCROLL_STATE_TOUCH_SCROLL = 1;
453    
454            /**
455             * The user had previously been scrolling using touch and had performed a fling. The
456             * animation is now coasting to a stop
457             */
458            public static int SCROLL_STATE_FLING = 2;
459    
460            /**
461             * Callback method to be invoked while the list view or grid view is being scrolled. If the
462             * view is being scrolled, this method will be called before the next frame of the scroll is
463             * rendered. In particular, it will be called before any calls to
464             * {@link Adapter#getView(int, View, ViewGroup)}.
465             *
466             * @param view The view whose scroll state is being reported
467             *
468             * @param scrollState The current scroll state. One of {@link #SCROLL_STATE_IDLE},
469             * {@link #SCROLL_STATE_TOUCH_SCROLL} or {@link #SCROLL_STATE_IDLE}.
470             */
471            public void onScrollStateChanged(AbsListView view, int scrollState);
472    
473            /**
474             * Callback method to be invoked when the list or grid has been scrolled. This will be
475             * called after the scroll has completed
476             * @param view The view whose scroll state is being reported
477             * @param firstVisibleItem the index of the first visible cell (ignore if
478             *        visibleItemCount == 0)
479             * @param visibleItemCount the number of visible cells
480             * @param totalItemCount the number of items in the list adaptor
481             */
482            public void onScroll(AbsListView view, int firstVisibleItem, int visibleItemCount,
483                    int totalItemCount);
484        }

540        /**
541         * Enables fast scrolling by letting the user quickly scroll through lists by 
542         * dragging the fast scroll thumb. The adapter attached to the list may want 
543         * to implement {@link SectionIndexer} if it wishes to display alphabet preview and
544         * jump between sections of the list. 
545         * @see SectionIndexer
546         * @see #isFastScrollEnabled()
547         * @param enabled whether or not to enable fast scrolling

554                }
555            } else {
556                if (mFastScroller != null) {
557                    mFastScroller.stop();
558                    mFastScroller = null;
559                }
560            }
561        }

699        @Override
700        public void getFocusedRect(Rect r) {
701            View view = getSelectedView();
702            if (view != null) {
703                // the focused rectangle of the selected view offset into the
704                // coordinate space of this view.
705                view.getFocusedRect(r);
706                offsetDescendantRectToMyCoords(view, r);
707            } else {
708                // otherwise, just the norm
709                super.getFocusedRect(r);
710            }
711        }

825        @Override
826        public Parcelable onSaveInstanceState() {
827            /*
828             * This doesn't really make sense as the place to dismiss the
829             * popup, but there don't seem to be any other useful hooks
830             * that happen early enough to keep from getting complaints
831             * about having leaked the window.
832             */
833            dismissPopup();
834    
835            Parcelable superState = super.onSaveInstanceState();
836    
837            SavedState ss = new SavedState(superState);
838    
839            boolean haveChildren = getChildCount() > 0;
840            long selectedId = getSelectedItemId();
841            ss.selectedId = selectedId;
842            ss.height = getHeight();
843    
844            if (selectedId >= 0) {
845                // Remember the selection
846                ss.viewTop = mSelectedTop;
847                ss.position = getSelectedItemPosition();
848                ss.firstId = INVALID_POSITION;
849            } else {
850                if (haveChildren) {
851                    // Remember the position of the first child
852                    View v = getChildAt(0);
853                    ss.viewTop = v.getTop();
854                    ss.position = mFirstPosition;
855                    ss.firstId = mAdapter.getItemId(mFirstPosition);
856                } else {
857                    ss.viewTop = 0;
858                    ss.firstId = INVALID_POSITION;
859                    ss.position = 0;
860                }
861            }
862    
863            ss.filter = null;
864            if (mFiltered) {
865                final EditText textFilter = mTextFilter;
866                if (textFilter != null) {
867                    Editable filterText = textFilter.getText();
868                    if (filterText != null) {
869                        ss.filter = filterText.toString();
870                    }
871                }
872            }
873    
874            return ss;
875        }

1000        protected int computeVerticalScrollExtent() {
1001            final int count = getChildCount();
1002            if (count > 0) {
1003                if (mSmoothScrollbarEnabled) {
1004                    int extent = count * 100;
1005    
1006                    View view = getChildAt(0);
1007                    final int top = view.getTop();
1008                    int height = view.getHeight();
1009                    if (height > 0) {
1010                        extent += (top * 100) / height;
1011                    }
1012    
1013                    view = getChildAt(count - 1);
1014                    final int bottom = view.getBottom();
1015                    height = view.getHeight();
1016                    if (height > 0) {
1017                        extent -= ((bottom - getHeight()) * 100) / height;
1018                    }
1019    
1020                    return extent;
1021                } else {
1022                    return 1;
1023                }
1024            }
1025            return 0;
1026        }

1028        @Override
1029        protected int computeVerticalScrollOffset() {
1030            final int firstPosition = mFirstPosition;
1031            final int childCount = getChildCount();
1032            if (firstPosition >= 0 && childCount > 0) {
1033                if (mSmoothScrollbarEnabled) {
1034                    final View view = getChildAt(0);
1035                    final int top = view.getTop();
1036                    int height = view.getHeight();
1037                    if (height > 0) {
1038                        return Math.max(firstPosition * 100 - (top * 100) / height, 0);
1039                    }
1040                } else {
1041                    int index;
1042                    final int count = mItemCount;
1043                    if (firstPosition == 0) {
1044                        index = 0;
1045                    } else if (firstPosition + childCount == count) {
1046                        index = count;
1047                    } else {
1048                        index = firstPosition + childCount / 2;
1049                    }
1050                    return (int) (firstPosition + childCount * (index / (float) count));
1051                }
1052            }
1053            return 0;
1054        }

1239        View obtainView(int position) {
1240            View scrapView;
1241

1243    
1244            View child;
1245            if (scrapView != null) {

1252

1259                    mRecycler.addScrapView(scrapView);
1260                    if (mCacheColorHint != 0) {

1268            } else {

1300        @Override
1301        protected void dispatchDraw(Canvas canvas) {
1302            int saveCount = 0;
1303            final boolean clipToPadding = (mGroupFlags & CLIP_TO_PADDING_MASK) == CLIP_TO_PADDING_MASK;
1304            if (clipToPadding) {
1305                saveCount = canvas.save();
1306                final int scrollX = mScrollX;
1307                final int scrollY = mScrollY;
1308                canvas.clipRect(scrollX + mPaddingLeft, scrollY + mPaddingTop,
1309                        scrollX + mRight - mLeft - mPaddingRight,
1310                        scrollY + mBottom - mTop - mPaddingBottom);
1311                mGroupFlags &= ~CLIP_TO_PADDING_MASK;
1312            }
1313    
1314            final boolean drawSelectorOnTop = mDrawSelectorOnTop;
1315            if (!drawSelectorOnTop) {
1316                drawSelector(canvas);
1317            }
1318    
1319            super.dispatchDraw(canvas);
1320    
1321            if (drawSelectorOnTop) {
1322                drawSelector(canvas);
1323            }
1324    
1325            if (clipToPadding) {
1326                canvas.restoreToCount(saveCount);
1327                mGroupFlags |= CLIP_TO_PADDING_MASK;
1328            }
1329        }

1427         * Sets the selector state to "pressed" and posts a CheckForKeyLongPress to see if
1428         * this is a long press.
1429         */
1430        void keyPressed() {
1431            Drawable selector = mSelector;
1432            Rect selectorRect = mSelectorRect;
1433            if (selector != null && (isFocused() || touchModeDrawsInPressedState())
1434                    && selectorRect != null && !selectorRect.isEmpty()) {
1435    
1436                final View v = getChildAt(mSelectedPosition - mFirstPosition);
1437    
1438                if (v != null) {
1439                    if (v.hasFocusable()) return;
1440                    v.setPressed(true);
1441                }

1443    
1444                final boolean longClickable = isLongClickable();
1445                Drawable d = selector.getCurrent();
1446                if (d != null && d instanceof TransitionDrawable) {
1447                    if (longClickable) {
1448                        ((TransitionDrawable) d).startTransition(ViewConfiguration
1449                                .getLongPressTimeout());
1450                    } else {
1451                        ((TransitionDrawable) d).resetTransition();
1452                    }
1453                }
1454                if (longClickable && !mDataChanged) {
1455                    if (mPendingCheckForKeyLongPress == null) {
1456                        mPendingCheckForKeyLongPress = new CheckForKeyLongPress();
1457                    }
1458                    mPendingCheckForKeyLongPress.rememberWindowAttachCount();
1459                    postDelayed(mPendingCheckForKeyLongPress, ViewConfiguration.getLongPressTimeout());
1460                }
1461            }

1488            final int enabledState = ENABLED_STATE_SET[0];
1489    
1490            // If we don't have any extra space, it will return one of the static state arrays,
1491            // and clearing the enabled state on those arrays is a bad thing!  If we specify
1492            // we need extra space, it will create+copy into a new array that safely mutable.
1493            int[] state = super.onCreateDrawableState(extraSpace + 1);
1494            int enabledPos = -1;
1495            for (int i = state.length - 1; i >= 0; i--) {
1496                if (state[i] == enabledState) {
1497                    enabledPos = i;
1498                    break;
1499                }
1500            }
1501    
1502            // Remove the enabled state
1503            if (enabledPos >= 0) {
1504                System.arraycopy(state, enabledPos + 1, state, enabledPos,
1505                        state.length - enabledPos - 1);
1506            }
1507            
1508            return state;
1509        }

1832        private boolean startScrollIfNeeded(int deltaY) {
1833            // Check if we have moved far enough that it looks more like a
1834            // scroll than a tap
1835            final int distance = Math.abs(deltaY);
1836            if (distance > mTouchSlop) {
1837                createScrollingCache();
1838                mTouchMode = TOUCH_MODE_SCROLL;
1839                mMotionCorrection = deltaY;
1840                final Handler handler = getHandler();
1841                // Handler should not be null unless the AbsListView is not attached to a
1842                // window, which would make it very hard to scroll it... but the monkeys
1843                // say it's possible.
1844                if (handler != null) {
1845                    handler.removeCallbacks(mPendingCheckForLongPress);
1846                }
1847                setPressed(false);
1848                View motionView = getChildAt(mMotionPosition - mFirstPosition);
1849                if (motionView != null) {
1850                    motionView.setPressed(false);
1851                }
1852                reportScrollStateChange(OnScrollListener.SCROLL_STATE_TOUCH_SCROLL);
1853                // Time to start stealing events! Once we've stolen them, don't let anyone
1854                // steal from us
1855                requestDisallowInterceptTouchEvent(true);
1856                return true;
1857            }
1858    
1859            return false;
1860        }

1878        @Override
1879        public boolean onTouchEvent(MotionEvent ev) {
1880            
1881            if (mFastScroller != null) {
1882                boolean intercepted = mFastScroller.onTouchEvent(ev);
1883                if (intercepted) {
1884                    return true;
1885                }            
1886            }
1887            final int action = ev.getAction();
1888            final int x = (int) ev.getX();
1889            final int y = (int) ev.getY();
1890    
1891            View v;
1892            int deltaY;
1893    
1894            if (mVelocityTracker == null) {
1895                mVelocityTracker = VelocityTracker.obtain();
1896            }
1897            mVelocityTracker.addMovement(ev);
1898    
1899            switch (action) {
1900            case MotionEvent.ACTION_DOWN: {
1901                int motionPosition = pointToPosition(x, y);
1902                if (!mDataChanged) {
1903                    if ((mTouchMode != TOUCH_MODE_FLING) && (motionPosition >= 0)
1904                            && (getAdapter().isEnabled(motionPosition))) {
1905                        // User clicked on an actual view (and was not stopping a fling). It might be a
1906                        // click or a scroll. Assume it is a click until proven otherwise
1907                        mTouchMode = TOUCH_MODE_DOWN;
1908                        // FIXME Debounce
1909                        if (mPendingCheckForTap == null) {
1910                            mPendingCheckForTap = new CheckForTap();
1911                        }
1912                        postDelayed(mPendingCheckForTap, ViewConfiguration.getTapTimeout());
1913                    } else {
1914                        if (ev.getEdgeFlags() != 0 && motionPosition < 0) {
1915                            // If we couldn't find a view to click on, but the down event was touching
1916                            // the edge, we will bail out and try again. This allows the edge correcting
1917                            // code in ViewRoot to try to find a nearby view to select
1918                            return false;
1919                        }
1920                        // User clicked on whitespace, or stopped a fling. It is a scroll.
1921                        createScrollingCache();
1922                        mTouchMode = TOUCH_MODE_SCROLL;
1923                        motionPosition = findMotionRow(y);
1924                        reportScrollStateChange(OnScrollListener.SCROLL_STATE_TOUCH_SCROLL);
1925                    }
1926                }
1927    
1928                if (motionPosition >= 0) {
1929                    // Remember where the motion event started
1930                    v = getChildAt(motionPosition - mFirstPosition);
1931                    mMotionViewOriginalTop = v.getTop();
1932                    mMotionX = x;
1933                    mMotionY = y;
1934                    mMotionPosition = motionPosition;
1935                }
1936                mLastY = Integer.MIN_VALUE;
1937                break;
1938            }
1939    
1940            case MotionEvent.ACTION_MOVE: {
1941                deltaY = y - mMotionY;
1942                switch (mTouchMode) {
1943                case TOUCH_MODE_DOWN:
1944                case TOUCH_MODE_TAP:
1945                case TOUCH_MODE_DONE_WAITING:
1946                    // Check if we have moved far enough that it looks more like a
1947                    // scroll than a tap
1948                    startScrollIfNeeded(deltaY);
1949                    break;
1950                case TOUCH_MODE_SCROLL:
1951                    if (PROFILE_SCROLLING) {
1952                        if (!mScrollProfilingStarted) {
1953                            Debug.startMethodTracing("AbsListViewScroll");
1954                            mScrollProfilingStarted = true;
1955                        }
1956                    }
1957    
1958                    if (y != mLastY) {
1959                        deltaY -= mMotionCorrection;
1960                        int incrementalDeltaY = mLastY != Integer.MIN_VALUE ? y - mLastY : deltaY;
1961                        trackMotionScroll(deltaY, incrementalDeltaY);
1962    
1963                        // Check to see if we have bumped into the scroll limit
1964                        View motionView = this.getChildAt(mMotionPosition - mFirstPosition);
1965                        if (motionView != null) {
1966                            // Check if the top of the motion view is where it is
1967                            // supposed to be
1968                            if (motionView.getTop() != mMotionViewNewTop) {
1969                                // We did not scroll the full amount. Treat this essentially like the
1970                                // start of a new touch scroll
1971                                final int motionPosition = findMotionRow(y);
1972    
1973                                mMotionCorrection = 0;
1974                                motionView = getChildAt(motionPosition - mFirstPosition);
1975                                mMotionViewOriginalTop = motionView.getTop();
1976                                mMotionY = y;
1977                                mMotionPosition = motionPosition;
1978                            }
1979                        }
1980                        mLastY = y;
1981                    }
1982                    break;
1983                }
1984    
1985                break;
1986            }
1987    
1988            case MotionEvent.ACTION_UP: {
1989                switch (mTouchMode) {
1990                case TOUCH_MODE_DOWN:
1991                case TOUCH_MODE_TAP:
1992                case TOUCH_MODE_DONE_WAITING:
1993                    final int motionPosition = mMotionPosition;
1994                    final View child = getChildAt(motionPosition - mFirstPosition);
1995                    if (child != null && !child.hasFocusable()) {
1996                        if (mTouchMode != TOUCH_MODE_DOWN) {
1997                            child.setPressed(false);
1998                        }
1999    
2000                        if (mPerformClick == null) {
2001                            mPerformClick = new PerformClick();
2002                        }
2003    
2004                        final AbsListView.PerformClick performClick = mPerformClick;
2005                        performClick.mChild = child;
2006                        performClick.mClickMotionPosition = motionPosition;
2007                        performClick.rememberWindowAttachCount();
2008    
2009                        mResurrectToPosition = motionPosition;
2010    
2011                        if (mTouchMode == TOUCH_MODE_DOWN || mTouchMode == TOUCH_MODE_TAP) {
2012                            final Handler handler = getHandler();
2013                            if (handler != null) {
2014                                handler.removeCallbacks(mTouchMode == TOUCH_MODE_DOWN ?
2015                                        mPendingCheckForTap : mPendingCheckForLongPress);
2016                            }
2017                            mLayoutMode = LAYOUT_NORMAL;
2018                            mTouchMode = TOUCH_MODE_TAP;
2019                            if (!mDataChanged) {
2020                                setSelectedPositionInt(mMotionPosition);
2021                                layoutChildren();
2022                                child.setPressed(true);
2023                                positionSelector(child);
2024                                setPressed(true);
2025                                if (mSelector != null) {
2026                                    Drawable d = mSelector.getCurrent();
2027                                    if (d != null && d instanceof TransitionDrawable) {
2028                                        ((TransitionDrawable)d).resetTransition();
2029                                    }
2030                                }
2031                                postDelayed(new Runnable() {
2032                                    public void run() {
2033                                        child.setPressed(false);
2034                                        setPressed(false);
2035                                        if (!mDataChanged) {
2036                                            post(performClick);
2037                                        }
2038                                        mTouchMode = TOUCH_MODE_REST;
2039                                    }
2040                                }, ViewConfiguration.getPressedStateDuration());
2041                            }
2042                            return true;
2043                        } else {
2044                            if (!mDataChanged) {
2045                                post(performClick);
2046                            }
2047                        }
2048                    }
2049                    mTouchMode = TOUCH_MODE_REST;
2050                    break;
2051                case TOUCH_MODE_SCROLL:
2052                    final VelocityTracker velocityTracker = mVelocityTracker;
2053                    velocityTracker.computeCurrentVelocity(1000);
2054                    int initialVelocity = (int)velocityTracker.getYVelocity();
2055    
2056                    if ((Math.abs(initialVelocity) >
2057                            ViewConfiguration.get(mContext).getScaledMinimumFlingVelocity()) &&
2058                            (getChildCount() > 0)) {
2059                        if (mFlingRunnable == null) {
2060                            mFlingRunnable = new FlingRunnable();
2061                        }
2062                        reportScrollStateChange(OnScrollListener.SCROLL_STATE_FLING);
2063                        mFlingRunnable.start(-initialVelocity);
2064                    } else {
2065                        mTouchMode = TOUCH_MODE_REST;
2066                        reportScrollStateChange(OnScrollListener.SCROLL_STATE_IDLE);
2067                    }
2068                }
2069    
2070                setPressed(false);
2071                
2072                // Need to redraw since we probably aren't drawing the selector anymore
2073                invalidate();
2074                
2075                final Handler handler = getHandler();
2076                if (handler != null) {
2077                    handler.removeCallbacks(mPendingCheckForLongPress);
2078                }
2079    
2080                if (mVelocityTracker != null) {
2081                    mVelocityTracker.recycle();
2082                    mVelocityTracker = null;
2083                }
2084    
2085                if (PROFILE_SCROLLING) {
2086                    if (mScrollProfilingStarted) {
2087                        Debug.stopMethodTracing();
2088                        mScrollProfilingStarted = false;
2089                    }
2090                }
2091                break;
2092            }
2093    
2094            case MotionEvent.ACTION_CANCEL: {
2095                mTouchMode = TOUCH_MODE_REST;
2096                setPressed(false);
2097                View motionView = this.getChildAt(mMotionPosition - mFirstPosition);
2098                if (motionView != null) {
2099                    motionView.setPressed(false);
2100                }
2101                clearScrollingCache();
2102    
2103                final Handler handler = getHandler();
2104                if (handler != null) {
2105                    handler.removeCallbacks(mPendingCheckForLongPress);
2106                }
2107    
2108                if (mVelocityTracker != null) {
2109                    mVelocityTracker.recycle();
2110                    mVelocityTracker = null;
2111                }
2112            }
2113    
2114            }
2115    
2116            return true;
2117        }

2825            int dX, dY; // dest x, y
2826            switch (direction) {
