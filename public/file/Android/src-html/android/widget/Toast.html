034    /**
035     * A toast is a view containing a quick little message for the user.  The toast class
036     * helps you create and show those.
037     * {@more}
038     *
039     * <p>
040     * When the view is shown to the user, appears as a floating view over the
041     * application.  It will never receive focus.  The user will probably be in the
042     * middle of typing something else.  The idea is to be as unobtrusive as
043     * possible, while still showing the user the information you want them to see.
044     * Two examples are the volume control, and the brief message saying that your
045     * settings have been saved.
046     * <p>
047     * The easiest way to use this class is to call one of the static methods that constructs
048     * everything you need and returns a new Toast object.
049     */ 
050    public class Toast {
051        static final String TAG = "Toast";
052        static final boolean localLOGV = false;
053    
054        /**
055         * Show the view or text notification for a short period of time.  This time
056         * could be user-definable.  This is the default.
057         * @see #setDuration
058         */
059        public static final int LENGTH_SHORT = 0;
060    
061        /**
062         * Show the view or text notification for a long period of time.  This time
063         * could be user-definable.
064         * @see #setDuration
065         */
066        public static final int LENGTH_LONG = 1;
067    
068        final Context mContext;
069        final TN mTN;
070        int mDuration;
071        int mGravity = Gravity.CENTER_HORIZONTAL | Gravity.BOTTOM;
072        int mX, mY;
073        float mHorizontalMargin;
074        float mVerticalMargin;
075        View mView;
076        View mNextView;
077    
078        /**
079         * Construct an empty Toast object.  You must call {@link #setView} before you
080         * can call {@link #show}.
081         *
082         * @param context  The context to use.  Usually your {@link android.app.Application}
083         *                 or {@link android.app.Activity} object.
084         */
085        public Toast(Context context) {
086            mContext = context;
087            mTN = new TN(context);
088            mY = context.getResources().getDimensionPixelSize(
089                    com.android.internal.R.dimen.toast_y_offset);
090        }
091        
092        /**
093         * Show the view for the specified duration.
094         */
095        public void show() {
096            if (mNextView == null) {
097                throw new RuntimeException("setView must have been called");
098            }
099    
100            INotificationManager service = getService();
101    
102            String pkg = mContext.getPackageName();
103    
104            TN tn = mTN;
105    
106            try {
107                service.enqueueToast(pkg, tn, mDuration);
108            } catch (RemoteException e) {
109                // Empty
110            }
111        }
112    
113        /**
114         * Close the view if it's showing, or don't show it if it isn't showing yet.
115         * You do not normally have to call this.  Normally view will disappear on its own
116         * after the appropriate duration.
117         */
118        public void cancel() {
119            mTN.hide();
120            // TODO this still needs to cancel the inflight notification if any
121        }
122        
123        /**
124         * Set the view to show.
125         * @see #getView
126         */
127        public void setView(View view) {
128            mNextView = view;
129        }
130    
131        /**
132         * Return the view.
133         * @see #setView
134         */
135        public View getView() {
136            return mNextView;
137        }
138    
139        /**
140         * Set how long to show the view for.
141         * @see #LENGTH_SHORT
142         * @see #LENGTH_LONG
143         */
144        public void setDuration(int duration) {
145            mDuration = duration;
146        }
147    
148        /**
149         * Return the duration.
150         * @see #setDuration
151         */
152        public int getDuration() {
153            return mDuration;
154        }
155        
156        /**
157         * Set the margins of the view.
158         *
159         * @param horizontalMargin The horizontal margin, in percentage of the
160         *        container width, between the container's edges and the
161         *        notification
162         * @param verticalMargin The vertical margin, in percentage of the
163         *        container height, between the container's edges and the
164         *        notification
165         */
166        public void setMargin(float horizontalMargin, float verticalMargin) {
167            mHorizontalMargin = horizontalMargin;
168            mVerticalMargin = verticalMargin;
169        }
170    
171        /**
172         * Return the horizontal margin.
173         */
174        public float getHorizontalMargin() {
175            return mHorizontalMargin;
176        }
177    
178        /**
179         * Return the vertical margin.
180         */
181        public float getVerticalMargin() {
182            return mVerticalMargin;
183        }
184    
185        /**
186         * Set the location at which the notification should appear on the screen.
187         * @see android.view.Gravity
188         * @see #getGravity
189         */
190        public void setGravity(int gravity, int xOffset, int yOffset) {
191            mGravity = gravity;
192            mX = xOffset;
193            mY = yOffset;
194        }
195    
196         /**
197         * Get the location at which the notification should appear on the screen.
198         * @see android.view.Gravity
199         * @see #getGravity
200         */
201        public int getGravity() {
202            return mGravity;
203        }
204    
205        /**
206         * Return the X offset in pixels to apply to the gravity's location.
207         */
208        public int getXOffset() {
209            return mX;
210        }
211        
212        /**
213         * Return the Y offset in pixels to apply to the gravity's location.
214         */
215        public int getYOffset() {
216            return mY;
217        }
218        
219        /**
220         * Make a standard toast that just contains a text view.
221         *
222         * @param context  The context to use.  Usually your {@link android.app.Application}
223         *                 or {@link android.app.Activity} object.
224         * @param text     The text to show.  Can be formatted text.
225         * @param duration How long to display the message.  Either {@link #LENGTH_SHORT} or
226         *                 {@link #LENGTH_LONG}
227         *
228         */
229        public static Toast makeText(Context context, CharSequence text, int duration) {
230            Toast result = new Toast(context);
231    
232            LayoutInflater inflate = (LayoutInflater)context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
233            View v = inflate.inflate(com.android.internal.R.layout.transient_notification, null);
234            TextView tv = (TextView)v.findViewById(com.android.internal.R.id.message);
235            tv.setText(text);
236            
237            result.mNextView = v;
238            result.mDuration = duration;
239    
240            return result;
241        }
242    
243        /**
244         * Make a standard toast that just contains a text view with the text from a resource.
245         *
246         * @param context  The context to use.  Usually your {@link android.app.Application}
247         *                 or {@link android.app.Activity} object.
248         * @param resId    The resource id of the string resource to use.  Can be formatted text.
249         * @param duration How long to display the message.  Either {@link #LENGTH_SHORT} or
250         *                 {@link #LENGTH_LONG}
251         *
252         * @throws Resources.NotFoundException if the resource can't be found.
253         */
254        public static Toast makeText(Context context, int resId, int duration)
255                                    throws Resources.NotFoundException {
256            return makeText(context, context.getResources().getText(resId), duration);
257        }
258    
259        /**
260         * Update the text in a Toast that was previously created using one of the makeText() methods.
261         * @param resId The new text for the Toast.
262         */
263        public void setText(int resId) {
264            setText(mContext.getText(resId));
265        }
266        
267        /**
268         * Update the text in a Toast that was previously created using one of the makeText() methods.
269         * @param s The new text for the Toast.
270         */
271        public void setText(CharSequence s) {
272            if (mNextView == null) {
273                throw new RuntimeException("This Toast was not created with Toast.makeText()");
274            }
275            TextView tv = (TextView) mNextView.findViewById(com.android.internal.R.id.message);
276            if (tv == null) {
277                throw new RuntimeException("This Toast was not created with Toast.makeText()");
278            }
279            tv.setText(s);
280        }

283        // All the gunk below is the interaction with the Notification Service, which handles
284        // the proper ordering of these system-wide.
285        // =======================================================================================
286    
287        private static INotificationManager sService;
288    
289        static private INotificationManager getService()
290        {
291            if (sService != null) {
292                return sService;
293            }
294            sService = INotificationManager.Stub.asInterface(ServiceManager.getService("notification"));
295            return sService;
296        }
297    
298        private class TN extends ITransientNotification.Stub
299        {
300            TN(Context context)
301            {
302                // XXX This should be changed to use a Dialog, with a Theme.Toast
303                // defined that sets up the layout params appropriately.
304                mParams.height = WindowManager.LayoutParams.WRAP_CONTENT;
305                mParams.width = WindowManager.LayoutParams.WRAP_CONTENT;
306                mParams.flags = WindowManager.LayoutParams.FLAG_NOT_FOCUSABLE
307                        | WindowManager.LayoutParams.FLAG_NOT_TOUCHABLE
308                        | WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON;
309                mParams.format = PixelFormat.TRANSLUCENT;
310                mParams.windowAnimations = com.android.internal.R.style.Animation_Toast;
311                mParams.type = WindowManager.LayoutParams.TYPE_TOAST;
312                mParams.setTitle("Toast");
313            }
314    
315            /**
316             * schedule handleShow into the right thread
317             */
318            public void show()
319            {
320                if (localLOGV) Log.v(TAG, "SHOW: " + this);
321                mHandler.post(mShow);
322            }
323    
324            /**
325             * schedule handleHide into the right thread
326             */
327            public void hide()
328            {
329                if (localLOGV) Log.v(TAG, "HIDE: " + this);
330                mHandler.post(mHide);
331            }
332    
333            public void handleShow()
334            {
335                if (localLOGV) Log.v(TAG, "HANDLE SHOW: " + this + " mView=" + mView
336                        + " mNextView=" + mNextView);
337                if (mView != mNextView) {
338                    // remove the old view if necessary
339                    handleHide();
340                    mView = mNextView;
341                    mWM = WindowManagerImpl.getDefault();
342                    final int gravity = mGravity;
343                    mParams.gravity = gravity;
344                    if ((gravity & Gravity.HORIZONTAL_GRAVITY_MASK) == Gravity.FILL_HORIZONTAL) {
345                        mParams.horizontalWeight = 1.0f;
346                    }
347                    if ((gravity & Gravity.VERTICAL_GRAVITY_MASK) == Gravity.FILL_VERTICAL) {
348                        mParams.verticalWeight = 1.0f;
349                    }
350                    mParams.x = mX;
351                    mParams.y = mY;
352                    mParams.verticalMargin = mVerticalMargin;
353                    mParams.horizontalMargin = mHorizontalMargin;
354                    if (mView.getParent() != null) {
355                        if (localLOGV) Log.v(
356                                TAG, "REMOVE! " + mView + " in " + this);
357                        mWM.removeView(mView);
358                    }
359                    if (localLOGV) Log.v(TAG, "ADD! " + mView + " in " + this);
360                    mWM.addView(mView, mParams);
361                }
362            }
363    
364            public void handleHide()
365            {
366                if (localLOGV) Log.v(TAG, "HANDLE HIDE: " + this + " mView=" + mView);
367                if (mView != null) {
368                    // note: checking parent() just to make sure the view has
369                    // been added...  i have seen cases where we get here when
370                    // the view isn't yet added, so let's try not to crash.
371                    if (mView.getParent() != null) {
372                        if (localLOGV) Log.v(
373                                TAG, "REMOVE! " + mView + " in " + this);
374                        mWM.removeView(mView);
375                    }
376                    mView = null;
377                }
378            }
379    
380            Runnable mShow = new Runnable() {
381                public void run() {
382                    handleShow();
383                }
384            };
385    
386            Runnable mHide = new Runnable() {
387                public void run() {
388                    handleHide();
389                }
390            };
391    
392            private final WindowManager.LayoutParams mParams = new WindowManager.LayoutParams();
393            
394            WindowManagerImpl mWM;
395        }
396    
397        final Handler mHandler = new Handler();
398    }
