017    package android.widget;
018    
019    import android.content.Context;
020    import android.database.Cursor;
021    import android.view.View;
022    import android.view.ViewGroup;
023    import android.view.LayoutInflater;
024    
025    
026    /**
027     * An easy adapter that creates views defined in an XML file. You can specify
028     * the XML file that defines the appearance of the views.
029     */
030    public abstract class ResourceCursorAdapter extends CursorAdapter {
031        private int mLayout;
032    
033        private int mDropDownLayout;
034        
035        private LayoutInflater mInflater;
036        
037        /**
038         * Constructor.
039         * 
040         * @param context The context where the ListView associated with this
041         *            SimpleListItemFactory is running
042         * @param layout resource identifier of a layout file that defines the views
043         *            for this list item.  Unless you override them later, this will
044         *            define both the item views and the drop down views.
045         */
046        public ResourceCursorAdapter(Context context, int layout, Cursor c) {
047            super(context, c);
048            mLayout = mDropDownLayout = layout;
049            mInflater = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
050        }
051        
052        /**
053         * Constructor.
054         * 
055         * @param context The context where the ListView associated with this
056         *            SimpleListItemFactory is running
057         * @param layout resource identifier of a layout file that defines the views
058         *            for this list item.  Unless you override them later, this will
059         *            define both the item views and the drop down views.
060         * @param c The cursor from which to get the data.
061         * @param autoRequery If true the adapter will call requery() on the
062         *                    cursor whenever it changes so the most recent
063         *                    data is always displayed.
064         */
065        public ResourceCursorAdapter(Context context, int layout, Cursor c, boolean autoRequery) {
066            super(context, c, autoRequery);
067            mLayout = mDropDownLayout = layout;
068            mInflater = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
069        }
070    
071        /**
072         * Inflates view(s) from the specified XML file.
073         * 
074         * @see android.widget.CursorAdapter#newView(android.content.Context,
075         *      android.database.Cursor, ViewGroup)
076         */
077        @Override
078        public View newView(Context context, Cursor cursor, ViewGroup parent) {
079            return mInflater.inflate(mLayout, parent, false);
080        }
081    
082        @Override
083        public View newDropDownView(Context context, Cursor cursor, ViewGroup parent) {
084            return mInflater.inflate(mDropDownLayout, parent, false);
085        }
086    
087        /**
088         * <p>Sets the layout resource of the item views.</p>
089         *
090         * @param layout the layout resources used to create item views
091         */
092        public void setViewResource(int layout) {
093            mLayout = layout;
094        }
095        
096        /**
097         * <p>Sets the layout resource of the drop down views.</p>
098         *
099         * @param dropDownLayout the layout resources used to create drop down views
100         */
101        public void setDropDownViewResource(int dropDownLayout) {
102            mDropDownLayout = dropDownLayout;
103        }
104    }
