017    package android.widget;
018    
019    import android.annotation.Widget;
020    import android.content.Context;
021    import android.content.res.TypedArray;
022    import android.os.Parcel;
023    import android.os.Parcelable;
024    import android.text.format.DateFormat;
025    import android.util.AttributeSet;
026    import android.util.SparseArray;
027    import android.view.LayoutInflater;
028    
029    import com.android.internal.R;
030    import com.android.internal.widget.NumberPicker;
031    import com.android.internal.widget.NumberPicker.OnChangedListener;
032    
033    import java.text.DateFormatSymbols;
034    import java.util.Calendar;
035    
036    /**
037     * A view for selecting a month / year / day based on a calendar like layout.
038     *
039     * For a dialog using this view, see {@link android.app.DatePickerDialog}.
040     */
041    @Widget
042    public class DatePicker extends FrameLayout {
043    
044        private static final int DEFAULT_START_YEAR = 1900;
045        private static final int DEFAULT_END_YEAR = 2100;
046        
047        /* UI Components */
048        private final NumberPicker mDayPicker;
049        private final NumberPicker mMonthPicker;
050        private final NumberPicker mYearPicker;
051    
052        /**
053         * How we notify users the date has changed.
054         */
055        private OnDateChangedListener mOnDateChangedListener;
056        
057        private int mDay;
058        private int mMonth;
059        private int mYear;
060    
061        /**
062         * The callback used to indicate the user changes the date.
063         */
064        public interface OnDateChangedListener {
065    
066            /**
067             * @param view The view associated with this listener.
068             * @param year The year that was set.
069             * @param monthOfYear The month that was set (0-11) for compatibility
070             *  with {@link java.util.Calendar}.
071             * @param dayOfMonth The day of the month that was set.
072             */
073            void onDateChanged(DatePicker view, int year, int monthOfYear, int dayOfMonth);
074        }
075    
076        public DatePicker(Context context) {
077            this(context, null);
078        }
079        
080        public DatePicker(Context context, AttributeSet attrs) {
081            this(context, attrs, 0);
082        }
083    
084        public DatePicker(Context context, AttributeSet attrs, int defStyle) {
085            super(context, attrs, defStyle);
086    
087            LayoutInflater inflater = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
088            inflater.inflate(R.layout.date_picker, this, true);
089    
090            mDayPicker = (NumberPicker) findViewById(R.id.day);
091            mDayPicker.setFormatter(NumberPicker.TWO_DIGIT_FORMATTER);
092            mDayPicker.setSpeed(100);
093            mDayPicker.setOnChangeListener(new OnChangedListener() {
094                public void onChanged(NumberPicker picker, int oldVal, int newVal) {
095                    mDay = newVal;
096                    if (mOnDateChangedListener != null) {
097                        mOnDateChangedListener.onDateChanged(DatePicker.this, mYear, mMonth, mDay);
098                    }
099                }
100            });
101            mMonthPicker = (NumberPicker) findViewById(R.id.month);
102            mMonthPicker.setFormatter(NumberPicker.TWO_DIGIT_FORMATTER);
103            DateFormatSymbols dfs = new DateFormatSymbols();
104            mMonthPicker.setRange(1, 12, dfs.getShortMonths());
105            mMonthPicker.setSpeed(200);
106            mMonthPicker.setOnChangeListener(new OnChangedListener() {
107                public void onChanged(NumberPicker picker, int oldVal, int newVal) {
108                    
109                    /* We display the month 1-12 but store it 0-11 so always
110                     * subtract by one to ensure our internal state is always 0-11
111                     */
112                    mMonth = newVal - 1;
113                    if (mOnDateChangedListener != null) {
114                        mOnDateChangedListener.onDateChanged(DatePicker.this, mYear, mMonth, mDay);
115                    }
116                    updateDaySpinner();
117                }
118            });
119            mYearPicker = (NumberPicker) findViewById(R.id.year);
120            mYearPicker.setSpeed(100);
121            mYearPicker.setOnChangeListener(new OnChangedListener() {
122                public void onChanged(NumberPicker picker, int oldVal, int newVal) {
123                    mYear = newVal;
124                    if (mOnDateChangedListener != null) {
125                        mOnDateChangedListener.onDateChanged(DatePicker.this, mYear, mMonth, mDay);
126                    }
127                }
128            });
129            
130            // attributes
131            TypedArray a = context.obtainStyledAttributes(attrs, R.styleable.DatePicker);
132    
133            int mStartYear = a.getInt(R.styleable.DatePicker_startYear, DEFAULT_START_YEAR);
134            int mEndYear = a.getInt(R.styleable.DatePicker_endYear, DEFAULT_END_YEAR);
135            mYearPicker.setRange(mStartYear, mEndYear);
136            
137            a.recycle();
138            
139            // initialize to current date
140            Calendar cal = Calendar.getInstance();
141            init(cal.get(Calendar.YEAR), cal.get(Calendar.MONTH), cal.get(Calendar.DAY_OF_MONTH), null);
142            
143            // re-order the number pickers to match the current date format
144            reorderPickers();
145            
146            if (!isEnabled()) {
147                setEnabled(false);
148            }
149        }
150        
151        @Override
152        public void setEnabled(boolean enabled) {
153            super.setEnabled(enabled);
154            mDayPicker.setEnabled(enabled);
155            mMonthPicker.setEnabled(enabled);
156            mYearPicker.setEnabled(enabled);
157        }
158    
159        private void reorderPickers() {
160            char[] order = DateFormat.getDateFormatOrder(mContext);
161            
162            /* Default order is month, date, year so if that's the order then
163             * do nothing.
164             */
165            if ((order[0] == DateFormat.MONTH) && (order[1] == DateFormat.DATE)) {
166                return;
167            }
168            
169            /* Remove the 3 pickers from their parent and then add them back in the
170             * required order.
171             */
172            LinearLayout parent = (LinearLayout) findViewById(R.id.parent);
173            parent.removeAllViews();
174            for (char c : order) {
175                if (c == DateFormat.DATE) {
176                    parent.addView(mDayPicker);
177                } else if (c == DateFormat.MONTH) {
178                    parent.addView(mMonthPicker);
179                } else {
180                    parent.addView (mYearPicker);
181                }
182            }
183        }
184    
185        public void updateDate(int year, int monthOfYear, int dayOfMonth) {
186            mYear = year;
187            mMonth = monthOfYear;
188            mDay = dayOfMonth;
189            updateSpinners();
190        }
191    
192        private static class SavedState extends BaseSavedState {
193    
194            private final int mYear;
195            private final int mMonth;
196            private final int mDay;
197    
198            /**
199             * Constructor called from {@link DatePicker#onSaveInstanceState()}
200             */
201            private SavedState(Parcelable superState, int year, int month, int day) {
202                super(superState);
203                mYear = year;
204                mMonth = month;
205                mDay = day;
206            }
207            
208            /**
209             * Constructor called from {@link #CREATOR}
210             */
211            private SavedState(Parcel in) {
212                super(in);
213                mYear = in.readInt();
214                mMonth = in.readInt();
215                mDay = in.readInt();
216            }
217    
218            public int getYear() {
219                return mYear;
220            }
221    
222            public int getMonth() {
223                return mMonth;
224            }
225    
226            public int getDay() {
227                return mDay;
228            }
229    
230            @Override
231            public void writeToParcel(Parcel dest, int flags) {
232                super.writeToParcel(dest, flags);
233                dest.writeInt(mYear);
234                dest.writeInt(mMonth);
235                dest.writeInt(mDay);
236            }
237    
238            public static final Parcelable.Creator<SavedState> CREATOR =
239                    new Creator<SavedState>() {
240    
241                        public SavedState createFromParcel(Parcel in) {
242                            return new SavedState(in);
243                        }
244    
245                        public SavedState[] newArray(int size) {
246                            return new SavedState[size];
247                        }
248                    };
249        }
250    
251    
252        /**
253         * Override so we are in complete control of save / restore for this widget.
254         */
255        @Override
256        protected void dispatchRestoreInstanceState(SparseArray<Parcelable> container) {
257            dispatchThawSelfOnly(container);
258        }
259    
260        @Override
261        protected Parcelable onSaveInstanceState() {
262            Parcelable superState = super.onSaveInstanceState();
263            
264            return new SavedState(superState, mYear, mMonth, mDay);
265        }
266    
267        @Override
268        protected void onRestoreInstanceState(Parcelable state) {
269            SavedState ss = (SavedState) state;
270            super.onRestoreInstanceState(ss.getSuperState());
271            mYear = ss.getYear();
272            mMonth = ss.getMonth();
273            mDay = ss.getDay();
274        }
275    
276        /**
277         * Initialize the state.
278         * @param year The initial year.
279         * @param monthOfYear The initial month.
280         * @param dayOfMonth The initial day of the month.
281         * @param onDateChangedListener How user is notified date is changed by user, can be null.
282         */
283        public void init(int year, int monthOfYear, int dayOfMonth,
284                OnDateChangedListener onDateChangedListener) {
285            mYear = year;
286            mMonth = monthOfYear;
287            mDay = dayOfMonth;
288            mOnDateChangedListener = onDateChangedListener;
289            updateSpinners();
290        }
291    
292        private void updateSpinners() {
293            updateDaySpinner();
294            mYearPicker.setCurrent(mYear);
295            
296            /* The month display uses 1-12 but our internal state stores it
297             * 0-11 so add one when setting the display.
298             */
299            mMonthPicker.setCurrent(mMonth + 1);
300        }
301    
302        private void updateDaySpinner() {
303            Calendar cal = Calendar.getInstance();
304            cal.set(mYear, mMonth, mDay);
305            int max = cal.getActualMaximum(Calendar.DAY_OF_MONTH);
306            mDayPicker.setRange(1, max);
307            mDayPicker.setCurrent(mDay);
308        }
309    
310        public int getYear() {
311            return mYear;
312        }
313    
314        public int getMonth() {
315            return mMonth;
316        }
317    
318        public int getDayOfMonth() {
319            return mDay;
320        }
321    }
