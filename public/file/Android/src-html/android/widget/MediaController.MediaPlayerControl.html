041    /**
042     * A view containing controls for a MediaPlayer. Typically contains the
043     * buttons like "Play/Pause", "Rewind", "Fast Forward" and a progress
044     * slider. It takes care of synchronizing the controls with the state
045     * of the MediaPlayer.
046     * <p>
047     * The way to use this class is to instantiate it programatically.
048     * The MediaController will create a default set of controls
049     * and put them in a window floating above your application. Specifically,
050     * the controls will float above the view specified with setAnchorView().
051     * The window will disappear if left idle for three seconds and reappear
052     * when the user touches the anchor view.
053     * <p>
054     * Functions like show() and hide() have no effect when MediaController
055     * is created in an xml layout.
056     * 
057     * MediaController will hide and
058     * show the buttons according to these rules:
059     * <ul>
060     * <li> The "previous" and "next" buttons are hidden until setPrevNextListeners()
061     *   has been called
062     * <li> The "previous" and "next" buttons are visible but disabled if
063     *   setPrevNextListeners() was called with null listeners
064     * <li> The "rewind" and "fastforward" buttons are shown unless requested
065     *   otherwise by using the MediaController(Context, boolean) constructor
066     *   with the boolean set to false
067     * </ul>
068     */
069    public class MediaController extends FrameLayout {

125        private void initFloatingWindow() {
126            mWindowManager = (WindowManager)mContext.getSystemService("window");
127            mWindow = PolicyManager.makeNewWindow(mContext);
128            mWindow.setWindowManager(mWindowManager, null, null);
129            mWindow.requestFeature(Window.FEATURE_NO_TITLE);
130            mDecor = mWindow.getDecorView();
131            mDecor.setOnTouchListener(mTouchListener);
132            mWindow.setContentView(this);
133            mWindow.setBackgroundDrawableResource(android.R.color.transparent);
134            
135            // While the media controller is up, the volume control keys should
136            // affect the media stream type
137            mWindow.setVolumeControlStream(AudioManager.STREAM_MUSIC);
138    
139            setFocusable(true);
140            setFocusableInTouchMode(true);
141            setDescendantFocusability(ViewGroup.FOCUS_AFTER_DESCENDANTS);
142            requestFocus();
143        }
144    
145        private OnTouchListener mTouchListener = new OnTouchListener() {
146            public boolean onTouch(View v, MotionEvent event) {
147                if (event.getAction() == MotionEvent.ACTION_DOWN) {
148                    if (mShowing) {
149                        hide();
150                    }
151                }
152                return false;
153            }
154        };

161        /**
162         * Set the view that acts as the anchor for the control view.
163         * This can for example be a VideoView, or your Activity's main view.
164         * @param view The view to which to anchor the controller when it is visible.
165         */
166        public void setAnchorView(View view) {
167            mAnchor = view;
168    
169            FrameLayout.LayoutParams frameParams = new FrameLayout.LayoutParams(
170                    ViewGroup.LayoutParams.FILL_PARENT,
171                    ViewGroup.LayoutParams.FILL_PARENT
172            );
173    
174            removeAllViews();
175            View v = makeControllerView();
176            addView(v, frameParams);
177        }

179        /**
180         * Create the view that holds the widgets that control playback.
181         * Derived classes can override this to create their own.
182         * @return The controller view.
183         * @hide This doesn't work as advertised
184         */
185        protected View makeControllerView() {
186            LayoutInflater inflate = (LayoutInflater) mContext.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
187            mRoot = inflate.inflate(com.android.internal.R.layout.media_controller, null);
188    
189            initControllerView(mRoot);
190    
191            return mRoot;
192        }

194        private void initControllerView(View v) {
195            mPauseButton = (ImageButton) v.findViewById(com.android.internal.R.id.pause);
196            if (mPauseButton != null) {
197                mPauseButton.requestFocus();
198                mPauseButton.setOnClickListener(mPauseListener);
199            }
200    
201            mFfwdButton = (ImageButton) v.findViewById(com.android.internal.R.id.ffwd);
202            if (mFfwdButton != null) {
203                mFfwdButton.setOnClickListener(mFfwdListener);
204                if (!mFromXml) {
205                    mFfwdButton.setVisibility(mUseFastForward ? View.VISIBLE : View.GONE);
206                }
207            }
208    
209            mRewButton = (ImageButton) v.findViewById(com.android.internal.R.id.rew);
210            if (mRewButton != null) {
211                mRewButton.setOnClickListener(mRewListener);
212                if (!mFromXml) {
213                    mRewButton.setVisibility(mUseFastForward ? View.VISIBLE : View.GONE);
214                }
215            }
216    
217            // By default these are hidden. They will be enabled when setPrevNextListeners() is called 
218            mNextButton = (ImageButton) v.findViewById(com.android.internal.R.id.next);
219            if (mNextButton != null && !mFromXml && !mListenersSet) {
220                mNextButton.setVisibility(View.GONE);
221            }
222            mPrevButton = (ImageButton) v.findViewById(com.android.internal.R.id.prev);
223            if (mPrevButton != null && !mFromXml && !mListenersSet) {
224                mPrevButton.setVisibility(View.GONE);
225            }
226    
227            mProgress = (ProgressBar) v.findViewById(com.android.internal.R.id.mediacontroller_progress);
228            if (mProgress != null) {
229                if (mProgress instanceof SeekBar) {
230                    SeekBar seeker = (SeekBar) mProgress;
231                    seeker.setOnSeekBarChangeListener(mSeekListener);
232                }
233                mProgress.setMax(1000);
234            }
235    
236            mEndTime = (TextView) v.findViewById(com.android.internal.R.id.time);
237            mCurrentTime = (TextView) v.findViewById(com.android.internal.R.id.time_current);
238            mFormatBuilder = new StringBuilder();
239            mFormatter = new Formatter(mFormatBuilder, Locale.getDefault());
240    
241            installPrevNextListeners();
242        }

244        /**
245         * Show the controller on screen. It will go away
246         * automatically after 3 seconds of inactivity.
247         */
248        public void show() {
249            show(sDefaultTimeout);
250        }
251    
252        /**
253         * Show the controller on screen. It will go away
254         * automatically after 'timeout' milliseconds of inactivity.
255         * @param timeout The timeout in milliseconds. Use 0 to show
256         * the controller until hide() is called.
257         */
258        public void show(int timeout) {
259    
260            if (!mShowing && mAnchor != null) {
261                setProgress();
262    
263                int [] anchorpos = new int[2];
264                mAnchor.getLocationOnScreen(anchorpos);
265    
266                WindowManager.LayoutParams p = new WindowManager.LayoutParams();
267                p.gravity = Gravity.TOP;
268                p.width = mAnchor.getWidth();
269                p.height = LayoutParams.WRAP_CONTENT;
270                p.x = 0;
271                p.y = anchorpos[1] + mAnchor.getHeight() - p.height;
272                p.format = PixelFormat.TRANSLUCENT;
273                p.type = WindowManager.LayoutParams.TYPE_APPLICATION_PANEL;
274                p.flags |= WindowManager.LayoutParams.FLAG_ALT_FOCUSABLE_IM;
275                p.token = null;
276                p.windowAnimations = 0; // android.R.style.DropDownAnimationDown;
277                mWindowManager.addView(mDecor, p);
278                mShowing = true;
279            }
280            updatePausePlay();
281            
282            // cause the progress bar to be updated even if mShowing
283            // was already true.  This happens, for example, if we're
284            // paused with the progress bar showing the user hits play.
285            mHandler.sendEmptyMessage(SHOW_PROGRESS);
286    
287            Message msg = mHandler.obtainMessage(FADE_OUT);
288            if (timeout != 0) {
289                mHandler.removeMessages(FADE_OUT);
290                mHandler.sendMessageDelayed(msg, timeout);
291            }
292        }
293        
294        public boolean isShowing() {
295            return mShowing;
296        }
297    
298        /**
299         * Remove the controller from the screen.
300         */
301        public void hide() {
302            if (mAnchor == null)
303                return;
304    
305            if (mShowing) {
306                try {
307                    mHandler.removeMessages(SHOW_PROGRESS);
308                    mWindowManager.removeView(mDecor);
309                } catch (IllegalArgumentException ex) {
310                    Log.w("MediaController", "already removed");
311                }
312                mShowing = false;
313            }
314        }
315    
316        private Handler mHandler = new Handler() {
317            @Override
318            public void handleMessage(Message msg) {
319                int pos;
320                switch (msg.what) {
321                    case FADE_OUT:
322                        hide();
323                        break;
324                    case SHOW_PROGRESS:
325                        pos = setProgress();
326                        if (!mDragging && mShowing && mPlayer.isPlaying()) {
327                            msg = obtainMessage(SHOW_PROGRESS);
328                            sendMessageDelayed(msg, 1000 - (pos % 1000));
329                        }
330                        break;
331                }
332            }
333        };

335        private String stringForTime(int timeMs) {
336            int totalSeconds = timeMs / 1000;
337    
338            int seconds = totalSeconds % 60;
339            int minutes = (totalSeconds / 60) % 60;
340            int hours   = totalSeconds / 3600;
341    
342            mFormatBuilder.setLength(0);
343            if (hours > 0) {
344                return mFormatter.format("%d:%02d:%02d", hours, minutes, seconds).toString();
345            } else {
346                return mFormatter.format("%02d:%02d", minutes, seconds).toString();
347            }
348        }
349    
350        private int setProgress() {
351            if (mPlayer == null || mDragging) {
352                return 0;
353            }
354            int position = mPlayer.getCurrentPosition();
355            int duration = mPlayer.getDuration();
356            if (mProgress != null) {
357                if (duration > 0) {
358                    // use long to avoid overflow
359                    long pos = 1000L * position / duration;
360                    mProgress.setProgress( (int) pos);
361                }
362                int percent = mPlayer.getBufferPercentage();
363                mProgress.setSecondaryProgress(percent * 10);
364            }
365    
366            if (mEndTime != null)
367                mEndTime.setText(stringForTime(duration));
368            if (mCurrentTime != null)
369                mCurrentTime.setText(stringForTime(position));
370    
371            return position;
372        }

386        @Override
387        public boolean dispatchKeyEvent(KeyEvent event) {
388            int keyCode = event.getKeyCode();
389            if (event.getRepeatCount() == 0 && event.isDown() && (
390                    keyCode ==  KeyEvent.KEYCODE_HEADSETHOOK ||
391                    keyCode ==  KeyEvent.KEYCODE_MEDIA_PLAY_PAUSE ||
392                    keyCode ==  KeyEvent.KEYCODE_SPACE)) {
393                doPauseResume();
394                show(sDefaultTimeout);
395                return true;
396            } else if (keyCode ==  KeyEvent.KEYCODE_MEDIA_STOP) {
397                if (mPlayer.isPlaying()) {
398                    mPlayer.pause();
399                    updatePausePlay();
400                }
401                return true;
402            } else if (keyCode == KeyEvent.KEYCODE_VOLUME_DOWN ||
403                    keyCode == KeyEvent.KEYCODE_VOLUME_UP) {
404                // don't show the controls for volume adjustment
405                return super.dispatchKeyEvent(event);
406            } else if (keyCode == KeyEvent.KEYCODE_BACK || keyCode == KeyEvent.KEYCODE_MENU) {
407                hide();
408    
409                return true;
410            } else {
411                show(sDefaultTimeout);
412            }
413            return super.dispatchKeyEvent(event);
414        }

416        private View.OnClickListener mPauseListener = new View.OnClickListener() {
417            public void onClick(View v) {
418                doPauseResume();
419                show(sDefaultTimeout);
420            }
421        };
422    
423        private void updatePausePlay() {
424            if (mRoot == null)
425                return;
426    
427            ImageButton button = (ImageButton) mRoot.findViewById(com.android.internal.R.id.pause);
428            if (button == null)
429                return;
430    
431            if (mPlayer.isPlaying()) {
432                button.setImageResource(com.android.internal.R.drawable.ic_media_pause);
433            } else {
434                button.setImageResource(com.android.internal.R.drawable.ic_media_play);
435            }
436        }
437    
438        private void doPauseResume() {
439            if (mPlayer.isPlaying()) {
440                mPlayer.pause();
441            } else {
442                mPlayer.start();
443            }
444            updatePausePlay();
445        }
446    
447        private OnSeekBarChangeListener mSeekListener = new OnSeekBarChangeListener() {
448            long duration;
449            public void onStartTrackingTouch(SeekBar bar) {
450                show(3600000);
451                duration = mPlayer.getDuration();
452            }
453            public void onProgressChanged(SeekBar bar, int progress, boolean fromtouch) {
454                if (fromtouch) {
455                    mDragging = true;
456                    duration = mPlayer.getDuration();
457                    long newposition = (duration * progress) / 1000L;
458                    mPlayer.seekTo( (int) newposition);
459                    if (mCurrentTime != null)
460                        mCurrentTime.setText(stringForTime( (int) newposition));
461                }
462            }
463            public void onStopTrackingTouch(SeekBar bar) {
464                mDragging = false;
465                setProgress();
466                updatePausePlay();
467                show(sDefaultTimeout);
468            }
469        };

471        @Override
472        public void setEnabled(boolean enabled) {
473            if (mPauseButton != null) {
474                mPauseButton.setEnabled(enabled);
475            }
476            if (mFfwdButton != null) {
477                mFfwdButton.setEnabled(enabled);
478            }
479            if (mRewButton != null) {
480                mRewButton.setEnabled(enabled);
481            }
482            if (mNextButton != null) {
483                mNextButton.setEnabled(enabled && mNextListener != null);
484            }
485            if (mPrevButton != null) {
486                mPrevButton.setEnabled(enabled && mPrevListener != null);
487            }
488            if (mProgress != null) {
489                mProgress.setEnabled(enabled);
490            }
491    
492            super.setEnabled(enabled);
493        }
494    
495        private View.OnClickListener mRewListener = new View.OnClickListener() {
496            public void onClick(View v) {
497                int pos = mPlayer.getCurrentPosition();
498                pos -= 5000; // milliseconds
499                mPlayer.seekTo(pos);
500                setProgress();
501    
502                show(sDefaultTimeout);
503            }
504        };
505    
506        private View.OnClickListener mFfwdListener = new View.OnClickListener() {
507            public void onClick(View v) {
508                int pos = mPlayer.getCurrentPosition();
509                pos += 15000; // milliseconds
510                mPlayer.seekTo(pos);
511                setProgress();
512    
513                show(sDefaultTimeout);
514            }
515        };
516    
517        private void installPrevNextListeners() {
518            if (mNextButton != null) {
519                mNextButton.setOnClickListener(mNextListener);
520                mNextButton.setEnabled(mNextListener != null);
521            }
522    
523            if (mPrevButton != null) {
524                mPrevButton.setOnClickListener(mPrevListener);
525                mPrevButton.setEnabled(mPrevListener != null);
526            }
527        }
528    
529        public void setPrevNextListeners(View.OnClickListener next, View.OnClickListener prev) {
530            mNextListener = next;
531            mPrevListener = prev;
532            mListenersSet = true;
533    
534            if (mRoot != null) {
535                installPrevNextListeners();
536                
537                if (mNextButton != null && !mFromXml) {
538                    mNextButton.setVisibility(View.VISIBLE);
539                }
540                if (mPrevButton != null && !mFromXml) {
541                    mPrevButton.setVisibility(View.VISIBLE);
542                }
543            }
544        }

546        public interface MediaPlayerControl {
547            void    start();
548            void    pause();
549            int     getDuration();
550            int     getCurrentPosition();
551            void    seekTo(int pos);
552            boolean isPlaying();
553            int     getBufferPercentage();
554        };
