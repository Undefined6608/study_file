121        /**
122         * Equivalent to calling
123         * {@link android.view.View#setOnClickListener(android.view.View.OnClickListener)}
124         * to launch the provided {@link PendingIntent}.
125         */
126        private class SetOnClickPendingIntent extends Action {
127            public SetOnClickPendingIntent(int id, PendingIntent pendingIntent) {
128                this.viewId = id;
129                this.pendingIntent = pendingIntent;
130            }
131            
132            public SetOnClickPendingIntent(Parcel parcel) {
133                viewId = parcel.readInt();
134                pendingIntent = PendingIntent.readPendingIntentOrNullFromParcel(parcel);
135            }
136            
137            public void writeToParcel(Parcel dest, int flags) {
138                dest.writeInt(TAG);
139                dest.writeInt(viewId);
140                pendingIntent.writeToParcel(dest, 0 /* no flags */);
141            }
142            
143            @Override
144            public void apply(View root) {
145                final View target = root.findViewById(viewId);
146                if (target != null && pendingIntent != null) {
147                    OnClickListener listener = new OnClickListener() {
148                        public void onClick(View v) {
149                            try {
150                                // TODO: Unregister this handler if PendingIntent.FLAG_ONE_SHOT?
151                                pendingIntent.send();
152                            } catch (CanceledException e) {
153                                throw new ActionException(e.toString());
154                            }
155                        }
156                    };
157                    target.setOnClickListener(listener);
158                }
159            }
160            
161            int viewId;
162            PendingIntent pendingIntent;
163    
164            public final static int TAG = 1;
165        }

167        /**
168         * Equivalent to calling a combination of {@link Drawable#setAlpha(int)},
169         * {@link Drawable#setColorFilter(int, android.graphics.PorterDuff.Mode)},
170         * and/or {@link Drawable#setLevel(int)} on the {@link Drawable} of a given view.
171         * <p>
172         * These operations will be performed on the {@link Drawable} returned by the
173         * target {@link View#getBackground()} by default.  If targetBackground is false,
174         * we assume the target is an {@link ImageView} and try applying the operations
175         * to {@link ImageView#getDrawable()}.
176         * <p>
177         * You can omit specific calls by marking their values with null or -1.
178         */
179        private class SetDrawableParameters extends Action {
180            public SetDrawableParameters(int id, boolean targetBackground, int alpha,
181                    int colorFilter, PorterDuff.Mode mode, int level) {
182                this.viewId = id;
183                this.targetBackground = targetBackground;
184                this.alpha = alpha;
185                this.colorFilter = colorFilter;
186                this.filterMode = mode;
187                this.level = level;
188            }
189            
190            public SetDrawableParameters(Parcel parcel) {
191                viewId = parcel.readInt();
192                targetBackground = parcel.readInt() != 0;
193                alpha = parcel.readInt();
194                colorFilter = parcel.readInt();
195                boolean hasMode = parcel.readInt() != 0;
196                if (hasMode) {
197                    filterMode = PorterDuff.Mode.valueOf(parcel.readString());
198                } else {
199                    filterMode = null;
200                }
201                level = parcel.readInt();
202            }
203            
204            public void writeToParcel(Parcel dest, int flags) {
205                dest.writeInt(TAG);
206                dest.writeInt(viewId);
207                dest.writeInt(targetBackground ? 1 : 0);
208                dest.writeInt(alpha);
209                dest.writeInt(colorFilter);
210                if (filterMode != null) {
211                    dest.writeInt(1);
212                    dest.writeString(filterMode.toString());
213                } else {
214                    dest.writeInt(0);
215                }
216                dest.writeInt(level);
217            }
218            
219            @Override
220            public void apply(View root) {
221                final View target = root.findViewById(viewId);
222                if (target == null) {
223                    return;
224                }
225                
226                // Pick the correct drawable to modify for this view
227                Drawable targetDrawable = null;
228                if (targetBackground) {
229                    targetDrawable = target.getBackground();
230                } else if (target instanceof ImageView) {
231                    ImageView imageView = (ImageView) target;
232                    targetDrawable = imageView.getDrawable();
233                }
234                
235                // Perform modifications only if values are set correctly
236                if (alpha != -1) {
237                    targetDrawable.setAlpha(alpha);
238                }
239                if (colorFilter != -1 && filterMode != null) {
240                    targetDrawable.setColorFilter(colorFilter, filterMode);
241                }
242                if (level != -1) {
243                    targetDrawable.setLevel(level);
244                }
245            }
246            
247            int viewId;
248            boolean targetBackground;
249            int alpha;
250            int colorFilter;
251            PorterDuff.Mode filterMode;
252            int level;
253    
254            public final static int TAG = 3;
255        }

257        /**
258         * Base class for the reflection actions.
259         */
260        private class ReflectionAction extends Action {
261            static final int TAG = 2;
262    
263            static final int BOOLEAN = 1;
264            static final int BYTE = 2;
265            static final int SHORT = 3;
266            static final int INT = 4;
267            static final int LONG = 5;
268            static final int FLOAT = 6;
269            static final int DOUBLE = 7;
270            static final int CHAR = 8;
271            static final int STRING = 9;
272            static final int CHAR_SEQUENCE = 10;
273            static final int URI = 11;
274            static final int BITMAP = 12;
275    
276            int viewId;
277            String methodName;
278            int type;
279            Object value;
280    
281            ReflectionAction(int viewId, String methodName, int type, Object value) {
282                this.viewId = viewId;
283                this.methodName = methodName;
284                this.type = type;
285                this.value = value;
286            }
287    
288            ReflectionAction(Parcel in) {
289                this.viewId = in.readInt();
290                this.methodName = in.readString();
291                this.type = in.readInt();
292                if (false) {
293                    Log.d("RemoteViews", "read viewId=0x" + Integer.toHexString(this.viewId)
294                            + " methodName=" + this.methodName + " type=" + this.type);
295                }
296                switch (this.type) {
297                    case BOOLEAN:
298                        this.value = in.readInt() != 0;
299                        break;
300                    case BYTE:
301                        this.value = in.readByte();
302                        break;
303                    case SHORT:
304                        this.value = (short)in.readInt();
305                        break;
306                    case INT:
307                        this.value = in.readInt();
308                        break;
309                    case LONG:
310                        this.value = in.readLong();
311                        break;
312                    case FLOAT:
313                        this.value = in.readFloat();
314                        break;
315                    case DOUBLE:
316                        this.value = in.readDouble();
317                        break;
318                    case CHAR:
319                        this.value = (char)in.readInt();
320                        break;
321                    case STRING:
322                        this.value = in.readString();
323                        break;
324                    case CHAR_SEQUENCE:
325                        this.value = TextUtils.CHAR_SEQUENCE_CREATOR.createFromParcel(in);
326                        break;
327                    case URI:
328                        this.value = Uri.CREATOR.createFromParcel(in);
329                        break;
330                    case BITMAP:
331                        this.value = Bitmap.CREATOR.createFromParcel(in);
332                        break;
333                    default:
334                        break;
335                }
336            }
337    
338            public void writeToParcel(Parcel out, int flags) {
339                out.writeInt(TAG);
340                out.writeInt(this.viewId);
341                out.writeString(this.methodName);
342                out.writeInt(this.type);
343                if (false) {
344                    Log.d("RemoteViews", "write viewId=0x" + Integer.toHexString(this.viewId)
345                            + " methodName=" + this.methodName + " type=" + this.type);
346                }
347                switch (this.type) {
348                    case BOOLEAN:
349                        out.writeInt(((Boolean)this.value).booleanValue() ? 1 : 0);
350                        break;
351                    case BYTE:
352                        out.writeByte(((Byte)this.value).byteValue());
353                        break;
354                    case SHORT:
355                        out.writeInt(((Short)this.value).shortValue());
356                        break;
357                    case INT:
358                        out.writeInt(((Integer)this.value).intValue());
359                        break;
360                    case LONG:
361                        out.writeLong(((Long)this.value).longValue());
362                        break;
363                    case FLOAT:
364                        out.writeFloat(((Float)this.value).floatValue());
365                        break;
366                    case DOUBLE:
367                        out.writeDouble(((Double)this.value).doubleValue());
368                        break;
369                    case CHAR:
370                        out.writeInt((int)((Character)this.value).charValue());
371                        break;
372                    case STRING:
373                        out.writeString((String)this.value);
374                        break;
375                    case CHAR_SEQUENCE:
376                        TextUtils.writeToParcel((CharSequence)this.value, out, flags);   
377                        break;
378                    case URI:
379                        ((Uri)this.value).writeToParcel(out, flags);
380                        break;
381                    case BITMAP:
382                        ((Bitmap)this.value).writeToParcel(out, flags);
383                        break;
384                    default:
385                        break;
386                }
387            }

389            private Class getParameterType() {
390                switch (this.type) {
391                    case BOOLEAN:
392                        return boolean.class;
393                    case BYTE:
394                        return byte.class;
395                    case SHORT:
396                        return short.class;
397                    case INT:
398                        return int.class;
399                    case LONG:
400                        return long.class;
401                    case FLOAT:
402                        return float.class;
403                    case DOUBLE:
404                        return double.class;
405                    case CHAR:
406                        return char.class;
407                    case STRING:
408                        return String.class;
409                    case CHAR_SEQUENCE:
410                        return CharSequence.class;
411                    case URI:
412                        return Uri.class;
413                    case BITMAP:
414                        return Bitmap.class;
415                    default:
416                        return null;
417                }
418            }
419    
420            @Override
421            public void apply(View root) {
422                final View view = root.findViewById(viewId);
423                if (view == null) {
424                    throw new ActionException("can't find view: 0x" + Integer.toHexString(viewId));
425                }
426    
427                Class param = getParameterType();
428                if (param == null) {
429                    throw new ActionException("bad type: " + this.type);
430                }
431    
432                Class klass = view.getClass();
433                Method method = null;
434                try {
435                    method = klass.getMethod(this.methodName, getParameterType());
436                }
437                catch (NoSuchMethodException ex) {
438                    throw new ActionException("view: " + klass.getName() + " doesn't have method: "
439                            + this.methodName + "(" + param.getName() + ")");
440                }
441    
442                if (!method.isAnnotationPresent(RemotableViewMethod.class)) {
443                    throw new ActionException("view: " + klass.getName()
444                            + " can't use method with RemoteViews: "
445                            + this.methodName + "(" + param.getName() + ")");
446                }
447    
448                try {
449                    if (false) {
450                        Log.d("RemoteViews", "view: " + klass.getName() + " calling method: "
451                            + this.methodName + "(" + param.getName() + ") with "
452                            + (this.value == null ? "null" : this.value.getClass().getName()));
453                    }
454                    method.invoke(view, this.value);
455                }
456                catch (Exception ex) {
457                    throw new ActionException(ex);
458                }
459            }
460        }

473        }

475        /**
476         * Reads a RemoteViews object from a parcel.
477         * 
478         * @param parcel
479         */
480        public RemoteViews(Parcel parcel) {
481            mPackage = parcel.readString();
482            mLayoutId = parcel.readInt();
483            int count = parcel.readInt();
484            if (count > 0) {
485                mActions = new ArrayList<Action>(count);
486                for (int i=0; i<count; i++) {
487                    int tag = parcel.readInt();
488                    switch (tag) {
489                    case SetOnClickPendingIntent.TAG:
490                        mActions.add(new SetOnClickPendingIntent(parcel));
491                        break;
492                    case SetDrawableParameters.TAG:
493                        mActions.add(new SetDrawableParameters(parcel));
494                        break;
495                    case ReflectionAction.TAG:
496                        mActions.add(new ReflectionAction(parcel));
497                        break;
498                    default:
499                        throw new ActionException("Tag " + tag + " not found");
500                    }
501                }
502            }
503        }

506            return mPackage;
507        }

509        public int getLayoutId() {
510            return mLayoutId;
511        }

513        /**
514         * Add an action to be executed on the remote side when apply is called.
515         * 
516         * @param a The action to add
517         */
518        private void addAction(Action a) {
519            if (mActions == null) {
520                mActions = new ArrayList<Action>();
521            }
522            mActions.add(a);
523        }

807        /**
808         * Inflates the view hierarchy represented by this object and applies
809         * all of the actions.
810         * 
811         * <p><strong>Caller beware: this may throw</strong>
812         * 
813         * @param context Default context to use
814         * @param parent Parent that the resulting view hierarchy will be attached to. This method
815         * does <strong>not</strong> attach the hierarchy. The caller should do so when appropriate.
816         * @return The inflated view hierarchy
817         */
818        public View apply(Context context, ViewGroup parent) {
819            View result = null;
820    
821            Context c = prepareContext(context);
822    
823            Resources r = c.getResources();
824            LayoutInflater inflater = (LayoutInflater) c
825                    .getSystemService(Context.LAYOUT_INFLATER_SERVICE);
826    
827            inflater = inflater.cloneInContext(c);
828            inflater.setFilter(this);
829    
830            result = inflater.inflate(mLayoutId, parent, false);
831    
832            performApply(result);
833    
834            return result;
835        }
836        
837        /**
838         * Applies all of the actions to the provided view.
839         *
840         * <p><strong>Caller beware: this may throw</strong>
841         * 
842         * @param v The view to apply the actions to.  This should be the result of
843         * the {@link #apply(Context,ViewGroup)} call.
844         */
845        public void reapply(Context context, View v) {
846            prepareContext(context);
847            performApply(v);
848        }
849    
850        private void performApply(View v) {
851            if (mActions != null) {
852                final int count = mActions.size();
853                for (int i = 0; i < count; i++) {
854                    Action a = mActions.get(i);
855                    a.apply(v);
856                }
857            }
858        }
859    
860        private Context prepareContext(Context context) {
861            Context c = null;
862            String packageName = mPackage;
863    
864            if (packageName != null) {
865                try {
866                    c = context.createPackageContext(packageName, 0);
867                } catch (NameNotFoundException e) {
868                    Log.e(LOG_TAG, "Package name " + packageName + " not found");
869                    c = context;
870                }
871            } else {
872                c = context;
873            }
874    
875            mContext = c;
876    
877            return c;
878        }

893        public void writeToParcel(Parcel dest, int flags) {
894            dest.writeString(mPackage);
895            dest.writeInt(mLayoutId);
896            int count;
897            if (mActions != null) {
898                count = mActions.size();
899            } else {
900                count = 0;
901            }
902            dest.writeInt(count);
903            for (int i=0; i<count; i++) {
904                Action a = mActions.get(i);
905                a.writeToParcel(dest, 0);
906            }
907        }
