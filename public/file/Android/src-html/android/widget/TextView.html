4025        public boolean onKeyMultiple(int keyCode, int repeatCount, KeyEvent event) {
4026            KeyEvent down = KeyEvent.changeAction(event, KeyEvent.ACTION_DOWN);
4027    
4028            int which = doKeyDown(keyCode, down, event);
4029            if (which == 0) {
4030                // Go through default dispatching.
4031                return super.onKeyMultiple(keyCode, repeatCount, event);
4032            }
4033            if (which == -1) {
4034                // Consumed the whole thing.
4035                return true;
4036            }
4037    
4038            repeatCount--;
4039            
4040            // We are going to dispatch the remaining events to either the input
4041            // or movement method.  To do this, we will just send a repeated stream
4042            // of down and up events until we have done the complete repeatCount.
4043            // It would be nice if those interfaces had an onKeyMultiple() method,
4044            // but adding that is a more complicated change.
4045            KeyEvent up = KeyEvent.changeAction(event, KeyEvent.ACTION_UP);
4046            if (which == 1) {
4047                mInput.onKeyUp(this, (Editable)mText, keyCode, up);
4048                while (--repeatCount > 0) {
4049                    mInput.onKeyDown(this, (Editable)mText, keyCode, down);
4050                    mInput.onKeyUp(this, (Editable)mText, keyCode, up);
4051                }
4052                if (mError != null && !mErrorWasChanged) {
4053                    setError(null, null);
4054                }
4055    
4056            } else if (which == 2) {
4057                mMovement.onKeyUp(this, (Spannable)mText, keyCode, up);
4058                while (--repeatCount > 0) {
4059                    mMovement.onKeyDown(this, (Spannable)mText, keyCode, down);
4060                    mMovement.onKeyUp(this, (Spannable)mText, keyCode, up);
4061                }
4062            }
4063    
4064            return true;
4065        }

4094        private int doKeyDown(int keyCode, KeyEvent event, KeyEvent otherEvent) {
4095            if (!isEnabled()) {
4096                return 0;
4097            }
4098    
4099            switch (keyCode) {
4100                case KeyEvent.KEYCODE_ENTER:
4101                    // If ALT modifier is held, then we always insert a
4102                    // newline character.
4103                    if ((event.getMetaState()&KeyEvent.META_ALT_ON) == 0) {
4104                        
4105                        // When mInputContentType is set, we know that we are
4106                        // running in a "modern" cupcake environment, so don't need
4107                        // to worry about the application trying to capture
4108                        // enter key events.
4109                        if (mInputContentType != null) {
4110                            
4111                            // If there is an action listener, given them a
4112                            // chance to consume the event.
4113                            if (mInputContentType.onEditorActionListener != null &&
4114                                    mInputContentType.onEditorActionListener.onEditorAction(
4115                                    this, EditorInfo.IME_NULL, event)) {
4116                                mInputContentType.enterDown = true;
4117                                // We are consuming the enter key for them.
4118                                return -1;
4119                            }
4120                        }
4121                        
4122                        // If our editor should move focus when enter is pressed, or
4123                        // this is a generated event from an IME action button, then
4124                        // don't let it be inserted into the text.
4125                        if ((event.getFlags()&KeyEvent.FLAG_EDITOR_ACTION) != 0
4126                                || shouldAdvanceFocusOnEnter()) {
4127                            return -1;
4128                        }
4129                    }
4130                    break;
4131                    
4132                case KeyEvent.KEYCODE_DPAD_CENTER:
4133                    if (shouldAdvanceFocusOnEnter()) {
4134                        return 0;
4135                    }
4136            }
4137    
4138            if (mInput != null) {
4139                /*
4140                 * Keep track of what the error was before doing the input
4141                 * so that if an input filter changed the error, we leave
4142                 * that error showing.  Otherwise, we take down whatever
4143                 * error was showing when the user types something.
4144                 */
4145                mErrorWasChanged = false;
4146    
4147                boolean doDown = true;
4148                if (otherEvent != null) {
4149                    try {
4150                        beginBatchEdit();
4151                        boolean handled = mInput.onKeyOther(this, (Editable) mText,
4152                                otherEvent);
4153                        if (mError != null && !mErrorWasChanged) {
4154                            setError(null, null);
4155                        }
4156                        doDown = false;
4157                        if (handled) {
4158                            return -1;
4159                        }
4160                    } catch (AbstractMethodError e) {
4161                        // onKeyOther was added after 1.0, so if it isn't
4162                        // implemented we need to try to dispatch as a regular down.
4163                    } finally {
4164                        endBatchEdit();
4165                    }
4166                }
4167                
4168                if (doDown) {
4169                    beginBatchEdit();
4170                    if (mInput.onKeyDown(this, (Editable) mText, keyCode, event)) {
4171                        endBatchEdit();
4172                        if (mError != null && !mErrorWasChanged) {
4173                            setError(null, null);
4174                        }
4175                        return 1;
4176                    }
4177                    endBatchEdit();
4178                }
4179            }
4180    
4181            // bug 650865: sometimes we get a key event before a layout.
4182            // don't try to move around if we don't know the layout.
4183    
4184            if (mMovement != null && mLayout != null) {
4185                boolean doDown = true;
4186                if (otherEvent != null) {
4187                    try {
4188                        boolean handled = mMovement.onKeyOther(this, (Spannable) mText,
4189                                otherEvent);
4190                        doDown = false;
4191                        if (handled) {
4192                            return -1;
4193                        }
4194                    } catch (AbstractMethodError e) {
4195                        // onKeyOther was added after 1.0, so if it isn't
4196                        // implemented we need to try to dispatch as a regular down.
4197                    }
4198                }
4199                if (doDown) {
4200                    if (mMovement.onKeyDown(this, (Spannable)mText, keyCode, event))
4201                        return 2;
4202                }
4203            }
4204    
4205            return 0;
4206        }

4294                if (mMovement.onKeyUp(this, (Spannable) mText, keyCode, event))

4367        boolean extractTextInternal(ExtractedTextRequest request,
4368                int partialStartOffset, int partialEndOffset, int delta,
4369                ExtractedText outText) {
4370            final CharSequence content = mText;
4371            if (content != null) {
4372                if (partialStartOffset != EXTRACT_NOTHING) {
4373                    final int N = content.length();
4374                    if (partialStartOffset < 0) {
4375                        outText.partialStartOffset = outText.partialEndOffset = -1;
4376                        partialStartOffset = 0;
4377                        partialEndOffset = N;
4378                    } else {
4379                        // Adjust offsets to ensure we contain full spans.
4380                        if (content instanceof Spanned) {
4381                            Spanned spanned = (Spanned)content;
4382                            Object[] spans = spanned.getSpans(partialStartOffset,
4383                                    partialEndOffset, ParcelableSpan.class);
4384                            int i = spans.length;
4385                            while (i > 0) {
4386                                i--;
4387                                int j = spanned.getSpanStart(spans[i]);
4388                                if (j < partialStartOffset) partialStartOffset = j;
4389                                j = spanned.getSpanEnd(spans[i]);
4390                                if (j > partialEndOffset) partialEndOffset = j;
4391                            }
4392                        }
4393                        outText.partialStartOffset = partialStartOffset;
4394                        outText.partialEndOffset = partialEndOffset;
4395                        // Now use the delta to determine the actual amount of text
4396                        // we need.
4397                        partialEndOffset += delta;
4398                        if (partialEndOffset > N) {
4399                            partialEndOffset = N;
4400                        } else if (partialEndOffset < 0) {
4401                            partialEndOffset = 0;
4402                        }
4403                    }
4404                    if ((request.flags&InputConnection.GET_TEXT_WITH_STYLES) != 0) {
4405                        outText.text = content.subSequence(partialStartOffset,
4406                                partialEndOffset);
4407                    } else {
4408                        outText.text = TextUtils.substring(content, partialStartOffset,
4409                                partialEndOffset);
4410                    }
4411                }
4412                outText.flags = 0;
4413                if (MetaKeyKeyListener.getMetaState(mText, MetaKeyKeyListener.META_SELECTING) != 0) {
4414                    outText.flags |= ExtractedText.FLAG_SELECTING;
4415                }
4416                if (mSingleLine) {
4417                    outText.flags |= ExtractedText.FLAG_SINGLE_LINE;
4418                }
4419                outText.startOffset = 0;
4420                outText.selectionStart = Selection.getSelectionStart(content);
4421                outText.selectionEnd = Selection.getSelectionEnd(content);
4422                return true;
4423            }
4424            return false;
4425        }

5322                vs = bottom - (vspace - vslack);
5323            if (ht - vs < vspace)
5324                vs = ht - vspace;
5325            if (0 - vs > 0)
5326                vs = 0;
5327    
5328            if (grav != 0) {
5329                if (x - hs < hslack) {
5330                    hs = x - hslack;
5331                }
5332                if (x - hs > hspace - hslack) {
5333                    hs = x - (hspace - hslack);
5334                }
5335            }
5336    
5337            if (grav < 0) {
5338                if (left - hs > 0)
5339                    hs = left;
5340                if (right - hs < hspace)
5341                    hs = right - hspace;
5342            } else if (grav > 0) {
5343                if (right - hs < hspace)
5344                    hs = right - hspace;
5345                if (left - hs > 0)
5346                    hs = left;
5347            } else /* grav == 0 */ {
5348                if (right - left <= hspace) {
5349                    /*
5350                     * If the entire text fits, center it exactly.
5351                     */
5352                    hs = left - (hspace - (right - left)) / 2;
5353                } else if (x > right - hslack) {
5354                    /*
5355                     * If we are near the right edge, keep the right edge
5356                     * at the edge of the view.
5357                     */
5358                    hs = right - hspace;
5359                } else if (x < left + hslack) {
5360                    /*
5361                     * If we are near the left edge, keep the left edge
5362                     * at the edge of the view.
5363                     */
5364                    hs = left;
5365                } else if (left > hs) {
5366                    /*
5367                     * Is there whitespace visible at the left?  Fix it if so.
5368                     */
5369                    hs = left;
5370                } else if (right < hs + hspace) {
5371                    /*
5372                     * Is there whitespace visible at the right?  Fix it if so.
5373                     */
5374                    hs = right - hspace;
5375                } else {
5376                    /*
5377                     * Otherwise, float as needed.
5378                     */
5379                    if (x - hs < hslack) {
5380                        hs = x - hslack;
5381                    }
5382                    if (x - hs > hspace - hslack) {
5383                        hs = x - (hspace - hslack);
5384                    }
5385                }
5386            }
5387    
5388            if (hs != mScrollX || vs != mScrollY) {
5389                if (mScroller == null) {
5390                    scrollTo(hs, vs);
5391                } else {
5392                    long duration = AnimationUtils.currentAnimationTimeMillis() - mLastScroll;
5393                    int dx = hs - mScrollX;
5394                    int dy = vs - mScrollY;
5395    
5396                    if (duration > ANIMATED_SCROLL_GAP) {
5397                        mScroller.startScroll(mScrollX, mScrollY, dx, dy);
5398                        invalidate();
5399                    } else {
5400                        if (!mScroller.isFinished()) {
5401                            mScroller.abortAnimation();
5402                        }
5403    
5404                        scrollBy(dx, dy);
5405                    }
5406    
5407                    mLastScroll = AnimationUtils.currentAnimationTimeMillis();
5408                }
5409    
5410                changed = true;
5411            }
5412    
5413            if (isFocused()) {
5414                // This offsets because getInterestingRect() is in terms of
5415                // viewport coordinates, but requestRectangleOnScreen()
5416                // is in terms of content coordinates.
5417    
5418                Rect r = new Rect();
5419                getInterestingRect(r, x, top, bottom, line);
5420                r.offset(mScrollX, mScrollY);
5421    
5422                if (requestRectangleOnScreen(r)) {
5423                    changed = true;
5424                }
5425            }
5426    
5427            return changed;
5428        }

5433         * one character (a selection range).  This will only work if the

5437            if (!(mText instanceof Spannable)) {
5438                return false;
5439            }
5440            int start = Selection.getSelectionStart(mText);
5441            int end = Selection.getSelectionEnd(mText);
5442            if (start != end) {

5444            }

5447            
5448            int line = mLayout.getLineForOffset(start);
5449

5451            final int bottom = mLayout.getLineTop(line+1);

5465

6858         * Called when a context menu option for the text view is selected.  Currently
6859         * this will be one of: {@link android.R.id#selectAll},
6860         * {@link android.R.id#startSelectingText}, {@link android.R.id#stopSelectingText},
6861         * {@link android.R.id#cut}, {@link android.R.id#copy},
6862         * {@link android.R.id#paste}, {@link android.R.id#copyUrl},
6863         * or {@link android.R.id#switchInputMethod}.
6864         */
6865        public boolean onTextContextMenuItem(int id) {
6866            int selStart = getSelectionStart();
6867            int selEnd = getSelectionEnd();
6868    
6869            if (!isFocused()) {
6870                selStart = 0;
6871                selEnd = mText.length();
6872            }
6873    
6874            int min = Math.min(selStart, selEnd);
6875            int max = Math.max(selStart, selEnd);
6876    
6877            if (min < 0) {
6878                min = 0;
6879            }
6880            if (max < 0) {
6881                max = 0;
6882            }
6883    
6884            ClipboardManager clip = (ClipboardManager)getContext()
6885                    .getSystemService(Context.CLIPBOARD_SERVICE);
6886    
6887            switch (id) {
6888                case ID_SELECT_ALL:
6889                    Selection.setSelection((Spannable) mText, 0,
6890                            mText.length());
6891                    return true;
6892    
6893                case ID_START_SELECTING_TEXT:
6894                    MetaKeyKeyListener.startSelecting(this, (Spannable) mText);
6895                    return true;
6896    
6897                case ID_STOP_SELECTING_TEXT:
6898                    MetaKeyKeyListener.stopSelecting(this, (Spannable) mText);
6899                    Selection.setSelection((Spannable) mText, getSelectionEnd());
6900                    return true;
6901    
6902                case ID_CUT:
6903                    MetaKeyKeyListener.stopSelecting(this, (Spannable) mText);
6904    
6905                    if (min == max) {
6906                        min = 0;
6907                        max = mText.length();
6908                    }
6909    
6910                    clip.setText(mTransformed.subSequence(min, max));
6911                    ((Editable) mText).delete(min, max);
6912                    return true;
6913    
6914                case ID_COPY:
6915                    MetaKeyKeyListener.stopSelecting(this, (Spannable) mText);
6916    
6917                    if (min == max) {
6918                        min = 0;
6919                        max = mText.length();
6920                    }
6921    
6922                    clip.setText(mTransformed.subSequence(min, max));
6923                    return true;
6924    
6925                case ID_PASTE:
6926                    MetaKeyKeyListener.stopSelecting(this, (Spannable) mText);
6927    
6928                    CharSequence paste = clip.getText();
6929    
6930                    if (paste != null) {
6931                        Selection.setSelection((Spannable) mText, max);
6932                        ((Editable) mText).replace(min, max, paste);
6933                    }
6934    
6935                    return true;
6936    
6937                case ID_COPY_URL:
6938                    MetaKeyKeyListener.stopSelecting(this, (Spannable) mText);
6939    
6940                    URLSpan[] urls = ((Spanned) mText).getSpans(min, max,
6941                                                           URLSpan.class);
6942                    if (urls.length == 1) {
6943                        clip.setText(urls[0].getURL());
6944                    }
6945    
6946                    return true;
6947    
6948                case ID_SWITCH_INPUT_METHOD:
6949                    InputMethodManager imm = InputMethodManager.peekInstance();
6950                    if (imm != null) {

6952                    }
6953                    return true;
