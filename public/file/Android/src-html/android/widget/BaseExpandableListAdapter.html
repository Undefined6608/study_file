017    package android.widget;
018    
019    import android.database.DataSetObservable;
020    import android.database.DataSetObserver;
021    import android.view.KeyEvent;
022    
023    /**
024     * Base class for a {@link ExpandableListAdapter} used to provide data and Views
025     * from some data to an expandable list view.
026     * <p>
027     * Adapters inheriting this class should verify that the base implementations of
028     * {@link #getCombinedChildId(long, long)} and {@link #getCombinedGroupId(long)}
029     * are correct in generating unique IDs from the group/children IDs.
030     * <p>
031     * @see SimpleExpandableListAdapter
032     * @see SimpleCursorTreeAdapter
033     */
034    public abstract class BaseExpandableListAdapter implements ExpandableListAdapter {
035        private final DataSetObservable mDataSetObservable = new DataSetObservable();
036        
037        public void registerDataSetObserver(DataSetObserver observer) {
038            mDataSetObservable.registerObserver(observer);
039        }
040    
041        public void unregisterDataSetObserver(DataSetObserver observer) {
042            mDataSetObservable.unregisterObserver(observer);
043        }
044        
045        /**
046         * {@see DataSetObservable#notifyInvalidated()}
047         */
048        public void notifyDataSetInvalidated() {
049            mDataSetObservable.notifyInvalidated();
050        }
051        
052        /**
053         * {@see DataSetObservable#notifyChanged()}
054         */
055        public void notifyDataSetChanged() {
056            mDataSetObservable.notifyChanged();
057        }
058    
059        public boolean areAllItemsEnabled() {
060            return true;
061        }
062    
063        public void onGroupCollapsed(int groupPosition) {
064        }
065    
066        public void onGroupExpanded(int groupPosition) {
067        }
068    
069        /**
070         * Override this method if you foresee a clash in IDs based on this scheme:
071         * <p>
072         * Base implementation returns a long:
073         * <li> bit 0: Whether this ID points to a child (unset) or group (set), so for this method
074         *             this bit will be 1.
075         * <li> bit 1-31: Lower 31 bits of the groupId
076         * <li> bit 32-63: Lower 32 bits of the childId.
077         * <p> 
078         * {@inheritDoc}
079         */
080        public long getCombinedChildId(long groupId, long childId) {
081            return 0x8000000000000000L | ((groupId & 0x7FFFFFFF) << 32) | (childId & 0xFFFFFFFF);
082        }
083    
084        /**
085         * Override this method if you foresee a clash in IDs based on this scheme:
086         * <p>
087         * Base implementation returns a long:
088         * <li> bit 0: Whether this ID points to a child (unset) or group (set), so for this method
089         *             this bit will be 0.
090         * <li> bit 1-31: Lower 31 bits of the groupId
091         * <li> bit 32-63: Lower 32 bits of the childId.
092         * <p> 
093         * {@inheritDoc}
094         */
095        public long getCombinedGroupId(long groupId) {
096            return (groupId & 0x7FFFFFFF) << 32;
097        }
098    
099        /**
100         * {@inheritDoc}
101         */
102        public boolean isEmpty() {
103            return getGroupCount() == 0;
104        }
105        
106    }
