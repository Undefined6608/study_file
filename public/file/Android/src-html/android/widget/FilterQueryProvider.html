017    package android.widget;
018    
019    import android.database.Cursor;
020    
021    /**
022     * This class can be used by external clients of CursorAdapter and
023     * CursorTreeAdapter to define how the content of the adapter should be
024     * filtered.
025     * 
026     * @see #runQuery(CharSequence)
027     */
028    public interface FilterQueryProvider {
029        /**
030         * Runs a query with the specified constraint. This query is requested
031         * by the filter attached to this adapter.
032         *
033         * Contract: when constraint is null or empty, the original results,
034         * prior to any filtering, must be returned.
035         *
036         * @param constraint the constraint with which the query must
037         *        be filtered
038         *
039         * @return a Cursor representing the results of the new query
040         */
041        Cursor runQuery(CharSequence constraint);
042    }
