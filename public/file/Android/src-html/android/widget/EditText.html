017    package android.widget;
018    
019    import android.text.*;
020    import android.text.method.*;
021    import android.content.Context;
022    import android.util.AttributeSet;
023    
024    
025    /*
026     * This is supposed to be a *very* thin veneer over TextView.
027     * Do not make any changes here that do anything that a TextView
028     * with a key listener and a movement method wouldn't do!
029     */
030    
031    /**
032     * EditText is a thin veneer over TextView that configures itself
033     * to be editable.
034     * <p>
035     * <b>XML attributes</b>
036     * <p>
037     * See {@link android.R.styleable#EditText EditText Attributes},
038     * {@link android.R.styleable#TextView TextView Attributes},
039     * {@link android.R.styleable#View View Attributes}
040     */
041    public class EditText extends TextView {
042        public EditText(Context context) {
043            this(context, null);
044        }
045    
046        public EditText(Context context, AttributeSet attrs) {
047            this(context, attrs, com.android.internal.R.attr.editTextStyle);
048        }
049    
050        public EditText(Context context, AttributeSet attrs, int defStyle) {
051            super(context, attrs, defStyle);
052        }
053    
054        @Override
055        protected boolean getDefaultEditable() {
056            return true;
057        }
058    
059        @Override
060        protected MovementMethod getDefaultMovementMethod() {
061            return ArrowKeyMovementMethod.getInstance();
062        }
063    
064        @Override
065        public Editable getText() {
066            return (Editable) super.getText();
067        }
068    
069        @Override
070        public void setText(CharSequence text, BufferType type) {
071            super.setText(text, BufferType.EDITABLE);
072        }
073    
074        /**
075         * Convenience for {@link Selection#setSelection(Spannable, int, int)}.
076         */
077        public void setSelection(int start, int stop) {
078            Selection.setSelection(getText(), start, stop);
079        }
080    
081        /**
082         * Convenience for {@link Selection#setSelection(Spannable, int)}.
083         */
084        public void setSelection(int index) {
085            Selection.setSelection(getText(), index);
086        }
087    
088        /**
089         * Convenience for {@link Selection#selectAll}.
090         */
091        public void selectAll() {
092            Selection.selectAll(getText());
093        }
094    
095        /**
096         * Convenience for {@link Selection#extendSelection}.
097         */
098        public void extendSelection(int index) {
099            Selection.extendSelection(getText(), index);
100        }
101    
102        @Override
103        public void setEllipsize(TextUtils.TruncateAt ellipsis) {
104            if (ellipsis == TextUtils.TruncateAt.MARQUEE) {
105                throw new IllegalArgumentException("EditText cannot use the ellipsize mode "
106                        + "TextUtils.TruncateAt.MARQUEE");
107            }
108            super.setEllipsize(ellipsis);
109        }
110    }
