036    /**
037     * Container for a tabbed window view. This object holds two children: a set of tab labels that the
038     * user clicks to select a specific tab, and a FrameLayout object that displays the contents of that
039     * page. The individual elements are typically controlled using this container object, rather than
040     * setting values on the child elements themselves.
041     */
042    public class TabHost extends FrameLayout implements ViewTreeObserver.OnTouchModeChangeListener {
043    
044        private TabWidget mTabWidget;
045        private FrameLayout mTabContent;
046        private List<TabSpec> mTabSpecs = new ArrayList<TabSpec>(2);
047        /**
048         * This field should be made private, so it is hidden from the SDK.
049         * {@hide}
050         */
051        protected int mCurrentTab = -1;
052        private View mCurrentView = null;
053        /**
054         * This field should be made private, so it is hidden from the SDK.
055         * {@hide}
056         */
057        protected LocalActivityManager mLocalActivityManager = null;
058        private OnTabChangeListener mOnTabChangeListener;
059        private OnKeyListener mTabKeyListener;

071        private final void initTabHost() {
072            setFocusableInTouchMode(true);
073            setDescendantFocusability(FOCUS_AFTER_DESCENDANTS);
074    
075            mCurrentTab = -1;
076            mCurrentView = null;
077        }

089        /**
090          * <p>Call setup() before adding tabs if loading TabHost using findViewById(). <i><b>However</i></b>: You do
091          * not need to call setup() after getTabHost() in {@link android.app.TabActivity TabActivity}.
092          * Example:</p>
093    <pre>mTabHost = (TabHost)findViewById(R.id.tabhost);
094    mTabHost.setup();
095    mTabHost.addTab(TAB_TAG_1, "Hello, world!", "Tab 1");
096          */
097        public void setup() {
098            mTabWidget = (TabWidget) findViewById(com.android.internal.R.id.tabs);
099            if (mTabWidget == null) {
100                throw new RuntimeException(
101                        "Your TabHost must have a TabWidget whose id attribute is 'android.R.id.tabs'");
102            }
103            
104            // KeyListener to attach to all tabs. Detects non-navigation keys 
105            // and relays them to the tab content.
106            mTabKeyListener = new OnKeyListener() {
107                public boolean onKey(View v, int keyCode, KeyEvent event) {
108                    switch (keyCode) {
109                        case KeyEvent.KEYCODE_DPAD_CENTER:
110                        case KeyEvent.KEYCODE_DPAD_LEFT:
111                        case KeyEvent.KEYCODE_DPAD_RIGHT:
112                        case KeyEvent.KEYCODE_DPAD_UP:
113                        case KeyEvent.KEYCODE_DPAD_DOWN:
114                        case KeyEvent.KEYCODE_ENTER:
115                            return false;
116                        
117                    }
118                    mTabContent.requestFocus(View.FOCUS_FORWARD);
119                    return mTabContent.dispatchKeyEvent(event);
120                }
121                
122            };
123            
124            mTabWidget.setTabSelectionListener(new TabWidget.OnTabSelectionChanged() {
125                public void onTabSelectionChanged(int tabIndex, boolean clicked) {
126                    setCurrentTab(tabIndex);
127                    if (clicked) {
128                        mTabContent.requestFocus(View.FOCUS_FORWARD);
129                    }
130                }
131            });
132    
133            mTabContent = (FrameLayout) findViewById(com.android.internal.R.id.tabcontent);
134            if (mTabContent == null) {
135                throw new RuntimeException(
136                        "Your TabHost must have a FrameLayout whose id attribute is 'android.R.id.tabcontent'");
137            }
138        }

141         * If you are using {@link TabSpec#setContent(android.content.Intent)}, this
142         * must be called since the activityGroup is needed to launch the local activity.
143         *
144         * This is done for you if you extend {@link android.app.TabActivity}.
145         * @param activityGroup Used to launch activities for tab content.
146         */
147        public void setup(LocalActivityManager activityGroup) {
148            setup();
149            mLocalActivityManager = activityGroup;

171        /**
172         * {@inheritDoc}
173         */
174        public void onTouchModeChanged(boolean isInTouchMode) {
175            if (!isInTouchMode) {
176                // leaving touch mode.. if nothing has focus, let's give it to
177                // the indicator of the current tab
178                if (!mCurrentView.hasFocus() || mCurrentView.isFocused()) {
179                    mTabWidget.getChildAt(mCurrentTab).requestFocus();
180                }
181            }
182        }

184        /**
185         * Add a tab.
186         * @param tabSpec Specifies how to create the indicator and content.
187         */
188        public void addTab(TabSpec tabSpec) {
189    
190            if (tabSpec.mIndicatorStrategy == null) {
191                throw new IllegalArgumentException("you must specify a way to create the tab indicator.");
192            }
193    
194            if (tabSpec.mContentStrategy == null) {
195                throw new IllegalArgumentException("you must specify a way to create the tab content");
196            }
197            View tabIndicator = tabSpec.mIndicatorStrategy.createIndicatorView();
198            tabIndicator.setOnKeyListener(mTabKeyListener);
199            mTabWidget.addView(tabIndicator);
200            mTabSpecs.add(tabSpec);
201    
202            if (mCurrentTab == -1) {
203                setCurrentTab(0);
204            }
205        }

209         * Removes all tabs from the tab widget associated with this tab host.
210         */
211        public void clearAllTabs() {
212            mTabWidget.removeAllViews();
213            initTabHost();
214            mTabContent.removeAllViews();
215            mTabSpecs.clear();
216            requestLayout();
217            invalidate();
218        }

246        public void setCurrentTabByTag(String tag) {
247            int i;
248            for (i = 0; i < mTabSpecs.size(); i++) {
249                if (mTabSpecs.get(i).getTag().equals(tag)) {
250                    setCurrentTab(i);
251                    break;
252                }
253            }

263        @Override
264        public boolean dispatchKeyEvent(KeyEvent event) {
265            final boolean handled = super.dispatchKeyEvent(event);
266    
267            // unhandled key ups change focus to tab indicator for embedded activities
268            // when there is nothing that will take focus from default focus searching
269            if (!handled
270                    && (event.getAction() == KeyEvent.ACTION_DOWN)
271                    && (event.getKeyCode() == KeyEvent.KEYCODE_DPAD_UP)
272                    && (mCurrentView.isRootNamespace())
273                    && (mCurrentView.hasFocus())
274                    && (mCurrentView.findFocus().focusSearch(View.FOCUS_UP) == null)) {
275                mTabWidget.getChildAt(mCurrentTab).requestFocus();
276                playSoundEffect(SoundEffectConstants.NAVIGATION_UP);
277                return true;
278            }
279            return handled;        
280        }

288        public void setCurrentTab(int index) {
289            if (index < 0 || index >= mTabSpecs.size()) {
290                return;
291            }
292    
293            if (index == mCurrentTab) {
294                return;
295            }
296    
297            // notify old tab content
298            if (mCurrentTab != -1) {
299                mTabSpecs.get(mCurrentTab).mContentStrategy.tabClosed();
300            }
301    
302            mCurrentTab = index;
303            final TabHost.TabSpec spec = mTabSpecs.get(index);
304    
305            // Call the tab widget's focusCurrentTab(), instead of just
306            // selecting the tab.
307            mTabWidget.focusCurrentTab(mCurrentTab);
308            
309            // tab content
310            mCurrentView = spec.mContentStrategy.getContentView();
311    
312            if (mCurrentView.getParent() == null) {
313                mTabContent
314                        .addView(
315                                mCurrentView,
316                                new ViewGroup.LayoutParams(
317                                        ViewGroup.LayoutParams.FILL_PARENT,
318                                        ViewGroup.LayoutParams.FILL_PARENT));
319            }
320    
321            if (!mTabWidget.hasFocus()) {
322                // if the tab widget didn't take focus (likely because we're in touch mode)
323                // give the current tab content view a shot
324                mCurrentView.requestFocus();
325            }
326    
327            //mTabContent.requestFocus(View.FOCUS_FORWARD);
328            invokeOnTabChangeListener();
329        }

372        /**
373         * A tab has a tab indictor, content, and a tag that is used to keep
374         * track of it.  This builder helps choose among these options.
375         *
376         * For the tab indicator, your choices are:
377         * 1) set a label
378         * 2) set a label and an icon
379         *
380         * For the tab content, your choices are:
381         * 1) the id of a {@link View}
382         * 2) a {@link TabContentFactory} that creates the {@link View} content.
383         * 3) an {@link Intent} that launches an {@link android.app.Activity}.
384         */
385        public class TabSpec {
386    
387            private String mTag;
388    
389            private IndicatorStrategy mIndicatorStrategy;
390            private ContentStrategy mContentStrategy;
391    
392            private TabSpec(String tag) {
393                mTag = tag;
394            }
395    
396            /**
397             * Specify a label as the tab indicator.
398             */
399            public TabSpec setIndicator(CharSequence label) {
400                mIndicatorStrategy = new LabelIndicatorStrategy(label);
401                return this;
402            }
403    
404            /**
405             * Specify a label and icon as the tab indicator.
406             */
407            public TabSpec setIndicator(CharSequence label, Drawable icon) {
408                mIndicatorStrategy = new LabelAndIconIndicatorStrategy(label, icon);
409                return this;
410            }
411    
412            /**
413             * Specify the id of the view that should be used as the content
414             * of the tab.
415             */
416            public TabSpec setContent(int viewId) {
417                mContentStrategy = new ViewIdContentStrategy(viewId);
418                return this;
419            }
420    
421            /**
422             * Specify a {@link android.widget.TabHost.TabContentFactory} to use to
423             * create the content of the tab.
424             */
425            public TabSpec setContent(TabContentFactory contentFactory) {
426                mContentStrategy = new FactoryContentStrategy(mTag, contentFactory);
427                return this;
428            }
429    
430            /**
431             * Specify an intent to use to launch an activity as the tab content.
432             */
433            public TabSpec setContent(Intent intent) {
434                mContentStrategy = new IntentContentStrategy(mTag, intent);
435                return this;
436            }
437    
438    
439            String getTag() {
440                return mTag;
441            }
442        }

455        /**
456         * Specifies what you do to manage the tab content.
457         */
458        private static interface ContentStrategy {
459    
460            /**
461             * Return the content view.  The view should may be cached locally.
462             */
463            View getContentView();
464    
465            /**
466             * Perhaps do something when the tab associated with this content has
467             * been closed (i.e make it invisible, or remove it).
468             */
469            void tabClosed();
470        }

472        /**
473         * How to create a tab indicator that just has a label.
474         */
475        private class LabelIndicatorStrategy implements IndicatorStrategy {
476    
477            private final CharSequence mLabel;
478    
479            private LabelIndicatorStrategy(CharSequence label) {
480                mLabel = label;
481            }
482    
483            public View createIndicatorView() {
484                LayoutInflater inflater =
485                        (LayoutInflater) getContext().getSystemService(Context.LAYOUT_INFLATER_SERVICE);
486                View tabIndicator = inflater.inflate(R.layout.tab_indicator,
487                        mTabWidget, // tab widget is the parent
488                        false); // no inflate params
489    
490                final TextView tv = (TextView) tabIndicator.findViewById(R.id.title);
491                tv.setText(mLabel);
492    
493                return tabIndicator;
494            }
495        }
496    
497        /**
498         * How we create a tab indicator that has a label and an icon
499         */
500        private class LabelAndIconIndicatorStrategy implements IndicatorStrategy {
501    
502            private final CharSequence mLabel;
503            private final Drawable mIcon;
504    
505            private LabelAndIconIndicatorStrategy(CharSequence label, Drawable icon) {
506                mLabel = label;
507                mIcon = icon;
508            }
509    
510            public View createIndicatorView() {
511                LayoutInflater inflater =
512                        (LayoutInflater) getContext().getSystemService(Context.LAYOUT_INFLATER_SERVICE);
513                View tabIndicator = inflater.inflate(R.layout.tab_indicator,
514                        mTabWidget, // tab widget is the parent
515                        false); // no inflate params
516    
517                final TextView tv = (TextView) tabIndicator.findViewById(R.id.title);
518                tv.setText(mLabel);
519    
520                final ImageView iconView = (ImageView) tabIndicator.findViewById(R.id.icon);
521                iconView.setImageDrawable(mIcon);
522    
523                return tabIndicator;
524            }
525        }

527        /**
528         * How to create the tab content via a view id.
529         */
530        private class ViewIdContentStrategy implements ContentStrategy {
531    
532            private final View mView;
533    
534            private ViewIdContentStrategy(int viewId) {
535                mView = mTabContent.findViewById(viewId);
536                if (mView != null) {
537                    mView.setVisibility(View.GONE);
538                } else {
539                    throw new RuntimeException("Could not create tab content because " +
540                            "could not find view with id " + viewId);
541                }
542            }
543    
544            public View getContentView() {
545                mView.setVisibility(View.VISIBLE);
546                return mView;
547            }
548    
549            public void tabClosed() {
550                mView.setVisibility(View.GONE);
551            }
552        }
553    
554        /**
555         * How tab content is managed using {@link TabContentFactory}.
556         */
557        private class FactoryContentStrategy implements ContentStrategy {
558            private View mTabContent;
559            private final CharSequence mTag;
560            private TabContentFactory mFactory;
561    
562            public FactoryContentStrategy(CharSequence tag, TabContentFactory factory) {
563                mTag = tag;
564                mFactory = factory;
565            }
566    
567            public View getContentView() {
568                if (mTabContent == null) {
569                    mTabContent = mFactory.createTabContent(mTag.toString());
570                }
571                mTabContent.setVisibility(View.VISIBLE);
572                return mTabContent;
573            }
574    
575            public void tabClosed() {
576                mTabContent.setVisibility(View.INVISIBLE);
577            }
578        }
579    
580        /**
581         * How tab content is managed via an {@link Intent}: the content view is the
582         * decorview of the launched activity.
583         */
584        private class IntentContentStrategy implements ContentStrategy {
585    
586            private final String mTag;
587            private final Intent mIntent;
588    
589            private View mLaunchedView;
590    
591            private IntentContentStrategy(String tag, Intent intent) {
592                mTag = tag;
593                mIntent = intent;
594            }
595    
596            public View getContentView() {
597                if (mLocalActivityManager == null) {
598                    throw new IllegalStateException("Did you forget to call 'public void setup(LocalActivityManager activityGroup)'?");
599                }
600                final Window w = mLocalActivityManager.startActivity(
601                        mTag, mIntent);
602                final View wd = w != null ? w.getDecorView() : null;
603                if (mLaunchedView != wd && mLaunchedView != null) {
604                    if (mLaunchedView.getParent() != null) {
605                        mTabContent.removeView(mLaunchedView);
606                    }
607                }
608                mLaunchedView = wd;
609                
610                // XXX Set FOCUS_AFTER_DESCENDANTS on embedded activies for now so they can get
611                // focus if none of their children have it. They need focus to be able to
612                // display menu items.
613                //
614                // Replace this with something better when Bug 628886 is fixed...
615                //
616                if (mLaunchedView != null) {
617                    mLaunchedView.setVisibility(View.VISIBLE);
618                    mLaunchedView.setFocusableInTouchMode(true);
619                    ((ViewGroup) mLaunchedView).setDescendantFocusability(
620                            FOCUS_AFTER_DESCENDANTS);
621                }
622                return mLaunchedView;
623            }
624    
625            public void tabClosed() {
626                if (mLaunchedView != null) {
627                    mLaunchedView.setVisibility(View.GONE);
628                }
629            }
630        }
