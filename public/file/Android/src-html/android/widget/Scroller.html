017    package android.widget;
018    
019    import android.content.Context;
020    import android.view.ViewConfiguration;
021    import android.view.animation.AnimationUtils;
022    import android.view.animation.Interpolator;

025    /**
026     * This class encapsulates scrolling.  The duration of the scroll
027     * can be passed in the constructor and specifies the maximum time that
028     * the scrolling animation should take.  Past this time, the scrolling is 
029     * automatically moved to its final stage and computeScrollOffset()
030     * will always return false to indicate that scrolling is over.
031     */
032    public class Scroller  {
033        private int mMode;
034    
035        private int mStartX;
036        private int mStartY;
037        private int mFinalX;
038        private int mFinalY;
039    
040        private int mMinX;
041        private int mMaxX;
042        private int mMinY;
043        private int mMaxY;
044    
045        private int mCurrX;
046        private int mCurrY;
047        private long mStartTime;
048        private int mDuration;
049        private float mDurationReciprocal;
050        private float mDeltaX;
051        private float mDeltaY;
052        private float mViscousFluidScale;
053        private float mViscousFluidNormalize;
054        private boolean mFinished;
055        private Interpolator mInterpolator;
056    
057        private float mCoeffX = 0.0f;
058        private float mCoeffY = 1.0f;
059        private float mVelocity;
060    
061        private static final int DEFAULT_DURATION = 250;
062        private static final int SCROLL_MODE = 0;
063        private static final int FLING_MODE = 1;
064    
065        private final float mDeceleration;
066    
067        /**
068         * Create a Scroller with the default duration and interpolator.
069         */
070        public Scroller(Context context) {
071            this(context, null);
072        }
073    
074        /**
075         * Create a Scroller with the specified interpolator. If the interpolator is
076         * null, the default (viscous) interpolator will be used.
077         */
078        public Scroller(Context context, Interpolator interpolator) {
079            mFinished = true;
080            mInterpolator = interpolator;
081            float ppi = context.getResources().getDisplayMetrics().density * 160.0f;
082            mDeceleration = 9.8f   // g (m/s^2)
083                          * 39.37f // inch/meter
084                          * ppi    // pixels per inch
085                          * ViewConfiguration.getScrollFriction();
086        }
087

089         * 
090         * Returns whether the scroller has finished scrolling.
091         * 
092         * @return True if the scroller has finished scrolling, false otherwise.
093         */

095            return mFinished;
096        }

098        /**
099         * Force the finished field to a particular value.
100         *  
101         * @param finished The new finished value.
102         */
103        public final void forceFinished(boolean finished) {
104            mFinished = finished;
105        }

108         * Returns how long the scroll event will take, in milliseconds.
109         * 
110         * @return The duration of the scroll in milliseconds.
111         */
112        public final int getDuration() {
113            return mDuration;
114        }

117         * Returns the current X offset in the scroll. 
118         * 
119         * @return The new X offset as an absolute distance from the origin.
120         */
121        public final int getCurrX() {
122            return mCurrX;
123        }
124        
125        /**
126         * Returns the current Y offset in the scroll. 
127         * 
128         * @return The new Y offset as an absolute distance from the origin.
129         */
130        public final int getCurrY() {
131            return mCurrY;
132        }
133        
134        /**
135         * Returns the start X offset in the scroll. 
136         * 
137         * @return The start X offset as an absolute distance from the origin.
138         */
139        public final int getStartX() {
140            return mStartX;
141        }
142        
143        /**
144         * Returns the start Y offset in the scroll. 
145         * 
146         * @return The start Y offset as an absolute distance from the origin.
147         */
148        public final int getStartY() {
149            return mStartY;
150        }
151        
152        /**
153         * Returns where the scroll will end. Valid only for "fling" scrolls.
154         * 
155         * @return The final X offset as an absolute distance from the origin.
156         */
157        public final int getFinalX() {
158            return mFinalX;
159        }
160        
161        /**
162         * Returns where the scroll will end. Valid only for "fling" scrolls.
163         * 
164         * @return The final Y offset as an absolute distance from the origin.
165         */
166        public final int getFinalY() {
167            return mFinalY;
168        }
169    
170        /**
171         * Call this when you want to know the new location.  If it returns true,
172         * the animation is not yet finished.  loc will be altered to provide the
173         * new location.
174         */ 
175        public boolean computeScrollOffset() {
176            if (mFinished) {
177                return false;
178            }
179    
180            int timePassed = (int)(AnimationUtils.currentAnimationTimeMillis() - mStartTime);
181        
182            if (timePassed < mDuration) {
183                switch (mMode) {
184                case SCROLL_MODE:
185                    float x = (float)timePassed * mDurationReciprocal;
186        
187                    if (mInterpolator == null)
188                        x = viscousFluid(x); 
189                    else
190                        x = mInterpolator.getInterpolation(x);
191        
192                    mCurrX = mStartX + Math.round(x * mDeltaX);
193                    mCurrY = mStartY + Math.round(x * mDeltaY);
194                    if ((mCurrX == mFinalX) && (mCurrY == mFinalY)) {
195                        mFinished = true;
196                    }
197                    break;
198                case FLING_MODE:
199                    float timePassedSeconds = timePassed / 1000.0f;
200                    float distance = (mVelocity * timePassedSeconds)
201                            - (mDeceleration * timePassedSeconds * timePassedSeconds / 2.0f);
202                    
203                    mCurrX = mStartX + Math.round(distance * mCoeffX);
204                    // Pin to mMinX <= mCurrX <= mMaxX
205                    mCurrX = Math.min(mCurrX, mMaxX);
206                    mCurrX = Math.max(mCurrX, mMinX);
207                    
208                    mCurrY = mStartY + Math.round(distance * mCoeffY);
209                    // Pin to mMinY <= mCurrY <= mMaxY
210                    mCurrY = Math.min(mCurrY, mMaxY);
211                    mCurrY = Math.max(mCurrY, mMinY);
212    
213                    if (mCurrX == mFinalX && mCurrY == mFinalY) {
214                        mFinished = true;
215                    }
216                    
217                    break;
218                }
219            }
220            else {
221                mCurrX = mFinalX;
222                mCurrY = mFinalY;
223                mFinished = true;
224            }
225            return true;
226        }

228        /**
229         * Start scrolling by providing a starting point and the distance to travel.
230         * The scroll will use the default value of 250 milliseconds for the
231         * duration.
232         * 
233         * @param startX Starting horizontal scroll offset in pixels. Positive
234         *        numbers will scroll the content to the left.
235         * @param startY Starting vertical scroll offset in pixels. Positive numbers
236         *        will scroll the content up.
237         * @param dx Horizontal distance to travel. Positive numbers will scroll the
238         *        content to the left.
239         * @param dy Vertical distance to travel. Positive numbers will scroll the
240         *        content up.
241         */
242        public void startScroll(int startX, int startY, int dx, int dy) {
243            startScroll(startX, startY, dx, dy, DEFAULT_DURATION);
244        }
245    
246        /**
247         * Start scrolling by providing a starting point and the distance to travel.
248         * 
249         * @param startX Starting horizontal scroll offset in pixels. Positive
250         *        numbers will scroll the content to the left.
251         * @param startY Starting vertical scroll offset in pixels. Positive numbers
252         *        will scroll the content up.
253         * @param dx Horizontal distance to travel. Positive numbers will scroll the
254         *        content to the left.
255         * @param dy Vertical distance to travel. Positive numbers will scroll the
256         *        content up.
257         * @param duration Duration of the scroll in milliseconds.
258         */
259        public void startScroll(int startX, int startY, int dx, int dy, int duration) {
260            mMode = SCROLL_MODE;
261            mFinished = false;
262            mDuration = duration;
263            mStartTime = AnimationUtils.currentAnimationTimeMillis();
264            mStartX = startX;
265            mStartY = startY;
266            mFinalX = startX + dx;
267            mFinalY = startY + dy;
268            mDeltaX = dx;
269            mDeltaY = dy;
270            mDurationReciprocal = 1.0f / (float) mDuration;
271            // This controls the viscous fluid effect (how much of it)
272            mViscousFluidScale = 8.0f;
273            // must be set to 1.0 (used in viscousFluid())
274            mViscousFluidNormalize = 1.0f;
275            mViscousFluidNormalize = 1.0f / viscousFluid(1.0f);
276        }
277    
278        /**
279         * Start scrolling based on a fling gesture. The distance travelled will
280         * depend on the initial velocity of the fling.
281         * 
282         * @param startX Starting point of the scroll (X)
283         * @param startY Starting point of the scroll (Y)
284         * @param velocityX Initial velocity of the fling (X) measured in pixels per
285         *        second.
286         * @param velocityY Initial velocity of the fling (Y) measured in pixels per
287         *        second
288         * @param minX Minimum X value. The scroller will not scroll past this
289         *        point.
290         * @param maxX Maximum X value. The scroller will not scroll past this
291         *        point.
292         * @param minY Minimum Y value. The scroller will not scroll past this
293         *        point.
294         * @param maxY Maximum Y value. The scroller will not scroll past this
295         *        point.
296         */
297        public void fling(int startX, int startY, int velocityX, int velocityY,
298                int minX, int maxX, int minY, int maxY) {
299            mMode = FLING_MODE;
300            mFinished = false;
301    
302            float velocity = (float)Math.hypot(velocityX, velocityY);
303         
304            mVelocity = velocity;
305            mDuration = (int) (1000 * velocity / mDeceleration); // Duration is in
306                                                                // milliseconds
307            mStartTime = AnimationUtils.currentAnimationTimeMillis();
308            mStartX = startX;
309            mStartY = startY;
310    
311            mCoeffX = velocity == 0 ? 1.0f : velocityX / velocity; 
312            mCoeffY = velocity == 0 ? 1.0f : velocityY / velocity;
313    
314            int totalDistance = (int) ((velocity * velocity) / (2 * mDeceleration));
315            
316            mMinX = minX;
317            mMaxX = maxX;
318            mMinY = minY;
319            mMaxY = maxY;
320            
321            
322            mFinalX = startX + Math.round(totalDistance * mCoeffX);
323            // Pin to mMinX <= mFinalX <= mMaxX
324            mFinalX = Math.min(mFinalX, mMaxX);
325            mFinalX = Math.max(mFinalX, mMinX);
326            
327            mFinalY = startY + Math.round(totalDistance * mCoeffY);
328            // Pin to mMinY <= mFinalY <= mMaxY
329            mFinalY = Math.min(mFinalY, mMaxY);
330            mFinalY = Math.max(mFinalY, mMinY);
331        }
332        
333        
334        
335        private float viscousFluid(float x)
336        {
337            x *= mViscousFluidScale;
338            if (x < 1.0f) {
339                x -= (1.0f - (float)Math.exp(-x));
340            } else {
341                float start = 0.36787944117f;   // 1/e == exp(-1)
342                x = 1.0f - (float)Math.exp(1.0f - x);
343                x = start + x * (1.0f - start);
344            }
345            x *= mViscousFluidNormalize;
346            return x;
347        }

349        /**
350         * 
351         */
352        public void abortAnimation() {
353            mCurrX = mFinalX;
354            mCurrY = mFinalY;
355            mFinished = true;
356        }
357        
358        /**
359         * Extend the scroll animation. This allows a running animation to 
360         * scroll further and longer, when used with setFinalX() or setFinalY().
361         *
362         * @param extend Additional time to scroll in milliseconds.
363         */
364        public void extendDuration(int extend) {
365            int passed = timePassed();
366            mDuration = passed + extend;
367            mDurationReciprocal = 1.0f / (float)mDuration;
368            mFinished = false;
369        }
370        
371        public int timePassed() {
372            return (int)(AnimationUtils.currentAnimationTimeMillis() - mStartTime);
373        }
374        
375        public void setFinalX(int newX) {
376            mFinalX = newX;
377            mDeltaX = mFinalX - mStartX;
378            mFinished = false;
379        }
380    
381       public void setFinalY(int newY) {
382            mFinalY = newY;
383            mDeltaY = mFinalY - mStartY;
384            mFinished = false;
385        }
