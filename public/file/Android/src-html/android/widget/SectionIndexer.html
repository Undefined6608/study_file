017    package android.widget;
018    
019    /**
020     * Interface that should be implemented on Adapters to enable fast scrolling 
021     * in an {@link AbsListView} between sections of the list. A section is a group of list items
022     * to jump to that have something in common. For example, they may begin with the
023     * same letter or they may be songs from the same artist. 
024     */
025    public interface SectionIndexer {
026        /**
027         * This provides the list view with an array of section objects. In the simplest
028         * case these are Strings, each containing one letter of the alphabet.
029         * They could be more complex objects that indicate the grouping for the adapter's
030         * consumption. The list view will call toString() on the objects to get the
031         * preview letter to display while scrolling.
032         * @return the array of objects that indicate the different sections of the list.
033         */
034        Object[] getSections();
035        
036        /**
037         * Provides the starting index in the list for a given section.
038         * @param section the index of the section to jump to.
039         * @return the starting position of that section. If the section is out of bounds, the
040         * position must be clipped to fall within the size of the list.
041         */
042        int getPositionForSection(int section);
043        
044        /**
045         * This is a reverse mapping to fetch the section index for a given position
046         * in the list.
047         * @param position the position for which to return the section
048         * @return the section index. If the position is out of bounds, the section index
049         * must be clipped to fall within the size of the section array.
050         */
051        int getSectionForPosition(int position);    
052    }
