031    /**
032     * A ListAdapter that manages a ListView backed by an array of arbitrary
033     * objects.  By default this class expects that the provided resource id references
034     * a single TextView.  If you want to use a more complex layout, use the constructors that
035     * also takes a field id.  That field id should reference a TextView in the larger layout
036     * resource.
037     *
038     * However the TextView is referenced, it will be filled with the toString() of each object in
039     * the array. You can add lists or arrays of custom objects. Override the toString() method
040     * of your objects to determine what text will be displayed for the item in the list.
041     *
042     * To use something other than TextViews for the array display, for instance, ImageViews,
043     * or to have some of data besides toString() results fill the views,
044     * override {@link #getView(int, View, ViewGroup)} to return the type of view you want.
045     */
046    public class ArrayAdapter<T> extends BaseAdapter implements Filterable {
047        /**
048         * Contains the list of objects that represent the data of this ArrayAdapter.
049         * The content of this list is referred to as "the array" in the documentation.
050         */
051        private List<T> mObjects;
052    
053        /**
054         * Lock used to modify the content of {@link #mObjects}. Any write operation
055         * performed on the array should be synchronized on this lock. This lock is also
056         * used by the filter (see {@link #getFilter()} to make a synchronized copy of
057         * the original array of data.
058         */
059        private final Object mLock = new Object();
060    
061        /**
062         * The resource indicating what views to inflate to display the content of this
063         * array adapter.
064         */
065        private int mResource;
066    
067        /**
068         * The resource indicating what views to inflate to display the content of this
069         * array adapter in a drop down widget.
070         */
071        private int mDropDownResource;
072    
073        /**
074         * If the inflated resource is not a TextView, {@link #mFieldId} is used to find
075         * a TextView inside the inflated views hierarchy. This field must contain the
076         * identifier that matches the one defined in the resource file.
077         */
078        private int mFieldId = 0;
079    
080        /**
081         * Indicates whether or not {@link #notifyDataSetChanged()} must be called whenever
082         * {@link #mObjects} is modified.
083         */
084        private boolean mNotifyOnChange = true;
085    
086        private Context mContext;    
087    
088        private ArrayList<T> mOriginalValues;
089        private ArrayFilter mFilter;
090    
091        private LayoutInflater mInflater;

166        /**
167         * Adds the specified object at the end of the array.
168         *
169         * @param object The object to add at the end of the array.
170         */
171        public void add(T object) {
172            if (mOriginalValues != null) {
173                synchronized (mLock) {
174                    mOriginalValues.add(object);
175                    if (mNotifyOnChange) notifyDataSetChanged();
176                }
177            } else {
178                mObjects.add(object);
179                if (mNotifyOnChange) notifyDataSetChanged();
180            }
181        }
182    
183        /**
184         * Inserts the specified object at the specified index in the array.
185         *
186         * @param object The object to insert into the array.
187         * @param index The index at which the object must be inserted.
188         */
189        public void insert(T object, int index) {
190            if (mOriginalValues != null) {
191                synchronized (mLock) {
192                    mOriginalValues.add(index, object);
193                    if (mNotifyOnChange) notifyDataSetChanged();
194                }
195            } else {
196                mObjects.add(index, object);
197                if (mNotifyOnChange) notifyDataSetChanged();
198            }
199        }
200    
201        /**
202         * Removes the specified object from the array.
203         *
204         * @param object The object to remove.
205         */
206        public void remove(T object) {
207            if (mOriginalValues != null) {
208                synchronized (mLock) {
209                    mOriginalValues.remove(object);
210                }
211            } else {
212                mObjects.remove(object);
213            }
214            if (mNotifyOnChange) notifyDataSetChanged();
215        }
216    
217        /**
218         * Remove all elements from the list.
219         */
220        public void clear() {
221            if (mOriginalValues != null) {
222                synchronized (mLock) {
223                    mOriginalValues.clear();
224                }
225            } else {
226                mObjects.clear();
227            }
228            if (mNotifyOnChange) notifyDataSetChanged();
229        }

231        /**
232         * Sorts the content of this adapter using the specified comparator.
233         *
234         * @param comparator The comparator used to sort the objects contained
235         *        in this adapter.
236         */
237        public void sort(Comparator<? super T> comparator) {
238            Collections.sort(mObjects, comparator);
239            if (mNotifyOnChange) notifyDataSetChanged();        
240        }

246        public void notifyDataSetChanged() {
247            super.notifyDataSetChanged();
248            mNotifyOnChange = true;
249        }

251        /**
252         * Control whether methods that change the list ({@link #add},
253         * {@link #insert}, {@link #remove}, {@link #clear}) automatically call
254         * {@link #notifyDataSetChanged}.  If set to false, caller must
255         * manually call notifyDataSetChanged() to have the changes
256         * reflected in the attached view.
257         *
258         * The default is true, and calling notifyDataSetChanged()
259         * resets the flag to true.
260         *
261         * @param notifyOnChange if true, modifications to the list will
262         *                       automatically call {@link
263         *                       #notifyDataSetChanged}
264         */
265        public void setNotifyOnChange(boolean notifyOnChange) {
266            mNotifyOnChange = notifyOnChange;
267        }

269        private void init(Context context, int resource, int textViewResourceId, List<T> objects) {
270            mContext = context;
271            mInflater = (LayoutInflater)context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
272            mResource = mDropDownResource = resource;
273            mObjects = objects;
274            mFieldId = textViewResourceId;
275        }

326        private View createViewFromResource(int position, View convertView, ViewGroup parent,
327                int resource) {
328            View view;
329            TextView text;
330    
331            if (convertView == null) {
332                view = mInflater.inflate(resource, parent, false);
333            } else {
334                view = convertView;
335            }
336    
337            try {
338                if (mFieldId == 0) {
339                    //  If no custom field is assigned, assume the whole resource is a TextView
340                    text = (TextView) view;
341                } else {
342                    //  Otherwise, find the TextView field within the layout
343                    text = (TextView) view.findViewById(mFieldId);
344                }
345            } catch (ClassCastException e) {
346                Log.e("ArrayAdapter", "You must supply a resource ID for a TextView");
347                throw new IllegalStateException(
348                        "ArrayAdapter requires the resource ID to be a TextView", e);
349            }
350    
351            text.setText(getItem(position).toString());
352    
353            return view;
354        }

374        /**
375         * Creates a new ArrayAdapter from external resources. The content of the array is
376         * obtained through {@link android.content.res.Resources#getTextArray(int)}.
377         *
378         * @param context The application's environment.
379         * @param textArrayResId The identifier of the array to use as the data source.
380         * @param textViewResId The identifier of the layout used to create views.
381         *
382         * @return An ArrayAdapter<CharSequence>.
383         */
384        public static ArrayAdapter<CharSequence> createFromResource(Context context,
385                int textArrayResId, int textViewResId) {
386            CharSequence[] strings = context.getResources().getTextArray(textArrayResId);
387            return new ArrayAdapter<CharSequence>(context, textViewResId, strings);
388        }

400        /**
401         * <p>An array filter constrains the content of the array adapter with
402         * a prefix. Each item that does not start with the supplied prefix
403         * is removed from the list.</p>
404         */
405        private class ArrayFilter extends Filter {
406            @Override
407            protected FilterResults performFiltering(CharSequence prefix) {
408                FilterResults results = new FilterResults();
409    
410                if (mOriginalValues == null) {
411                    synchronized (mLock) {
412                        mOriginalValues = new ArrayList<T>(mObjects);
413                    }
414                }
415    
416                if (prefix == null || prefix.length() == 0) {
417                    synchronized (mLock) {
418                        ArrayList<T> list = new ArrayList<T>(mOriginalValues);
419                        results.values = list;
420                        results.count = list.size();
421                    }
422                } else {
423                    String prefixString = prefix.toString().toLowerCase();
424    
425                    final ArrayList<T> values = mOriginalValues;
426                    final int count = values.size();
427    
428                    final ArrayList<T> newValues = new ArrayList<T>(count);
429    
430                    for (int i = 0; i < count; i++) {
431                        final T value = values.get(i);
432                        final String valueText = value.toString().toLowerCase();
433    
434                        // First match against the whole, non-splitted value
435                        if (valueText.startsWith(prefixString)) {
436                            newValues.add(value);
437                        } else {
438                            final String[] words = valueText.split(" ");
439                            final int wordCount = words.length;
440    
441                            for (int k = 0; k < wordCount; k++) {
442                                if (words[k].startsWith(prefixString)) {
443                                    newValues.add(value);
444                                    break;
445                                }
446                            }
447                        }
448                    }
449    
450                    results.values = newValues;
451                    results.count = newValues.size();
452                }
453    
454                return results;
455            }
456    
457            @Override
458            protected void publishResults(CharSequence constraint, FilterResults results) {
459                //noinspection unchecked
460                mObjects = (List<T>) results.values;
461                if (results.count > 0) {
462                    notifyDataSetChanged();
463                } else {
464                    notifyDataSetInvalidated();
465                }
466            }
467        }
