017    package android.widget;
018    
019    
020    import android.content.Context;
021    import android.content.res.TypedArray;
022    import android.graphics.drawable.Drawable;
023    import android.graphics.drawable.LayerDrawable;
024    import android.util.AttributeSet;
025    
026    /**
027     * Displays checked/unchecked states as a button
028     * with a "light" indicator and by default accompanied with the text "ON" or "OFF".
029     * 
030     * @attr ref android.R.styleable#ToggleButton_textOn
031     * @attr ref android.R.styleable#ToggleButton_textOff
032     * @attr ref android.R.styleable#ToggleButton_disabledAlpha
033     */
034    public class ToggleButton extends CompoundButton {
035        private CharSequence mTextOn;
036        private CharSequence mTextOff;
037        
038        private Drawable mIndicatorDrawable;
039    
040        private static final int NO_ALPHA = 0xFF;
041        private float mDisabledAlpha;
042        
043        public ToggleButton(Context context, AttributeSet attrs, int defStyle) {
044            super(context, attrs, defStyle);
045            
046            TypedArray a =
047                context.obtainStyledAttributes(
048                        attrs, com.android.internal.R.styleable.ToggleButton, defStyle, 0);
049            mTextOn = a.getText(com.android.internal.R.styleable.ToggleButton_textOn);
050            mTextOff = a.getText(com.android.internal.R.styleable.ToggleButton_textOff);
051            mDisabledAlpha = a.getFloat(com.android.internal.R.styleable.ToggleButton_disabledAlpha, 0.5f);
052            syncTextState();
053            a.recycle();
054        }
055    
056        public ToggleButton(Context context, AttributeSet attrs) {
057            this(context, attrs, com.android.internal.R.attr.buttonStyleToggle);
058        }
059    
060        public ToggleButton(Context context) {
061            this(context, null);
062        }
063    
064        @Override
065        public void setChecked(boolean checked) {
066            super.setChecked(checked);
067            
068            syncTextState();
069        }
070    
071        private void syncTextState() {
072            boolean checked = isChecked();
073            if (checked && mTextOn != null) {
074                setText(mTextOn);
075            } else if (!checked && mTextOff != null) {
076                setText(mTextOff);
077            }
078        }
079    
080        /**
081         * Returns the text for when the button is in the checked state.
082         * 
083         * @return The text.
084         */
085        public CharSequence getTextOn() {
086            return mTextOn;
087        }
088    
089        /**
090         * Sets the text for when the button is in the checked state.
091         *  
092         * @param textOn The text.
093         */
094        public void setTextOn(CharSequence textOn) {
095            mTextOn = textOn;
096        }
097    
098        /**
099         * Returns the text for when the button is not in the checked state.
100         * 
101         * @return The text.
102         */
103        public CharSequence getTextOff() {
104            return mTextOff;
105        }
106    
107        /**
108         * Sets the text for when the button is not in the checked state.
109         * 
110         * @param textOff The text.
111         */
112        public void setTextOff(CharSequence textOff) {
113            mTextOff = textOff;
114        }
115    
116        @Override
117        protected void onFinishInflate() {
118            super.onFinishInflate();
119            
120            updateReferenceToIndicatorDrawable(getBackground());
121        }
122    
123        @Override
124        public void setBackgroundDrawable(Drawable d) {
125            super.setBackgroundDrawable(d);
126            
127            updateReferenceToIndicatorDrawable(d);
128        }
129    
130        private void updateReferenceToIndicatorDrawable(Drawable backgroundDrawable) {
131            if (backgroundDrawable instanceof LayerDrawable) {
132                LayerDrawable layerDrawable = (LayerDrawable) backgroundDrawable;
133                mIndicatorDrawable =
134                        layerDrawable.findDrawableByLayerId(com.android.internal.R.id.toggle);
135            }
136        }
137        
138        @Override
139        protected void drawableStateChanged() {
140            super.drawableStateChanged();
141            
142            if (mIndicatorDrawable != null) {
143                mIndicatorDrawable.setAlpha(isEnabled() ? NO_ALPHA : (int) (NO_ALPHA * mDisabledAlpha));
144            }
145        }
146        
147    }
