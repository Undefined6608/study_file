017    package android.widget;
018    
019    import android.content.Context;
020    import android.os.Handler;
021    import android.util.AttributeSet;
022    import android.view.KeyEvent;
023    import android.view.MotionEvent;
024    import android.view.View;
025    import android.view.View.OnLongClickListener;
026    
027    public class ZoomButton extends ImageButton implements OnLongClickListener {
028    
029        private final Handler mHandler;
030        private final Runnable mRunnable = new Runnable() {
031            public void run() {
032                if ((mOnClickListener != null) && mIsInLongpress && isEnabled()) {
033                    mOnClickListener.onClick(ZoomButton.this);
034                    mHandler.postDelayed(this, mZoomSpeed);
035                }
036            }
037        };
038        
039        private long mZoomSpeed = 1000;
040        private boolean mIsInLongpress;
041        
042        public ZoomButton(Context context) {
043            this(context, null);
044        }
045    
046        public ZoomButton(Context context, AttributeSet attrs) {
047            this(context, attrs, 0);
048        }
049        
050        public ZoomButton(Context context, AttributeSet attrs, int defStyle) {
051            super(context, attrs, defStyle);
052            mHandler = new Handler();
053            setOnLongClickListener(this);
054        }
055    
056        @Override
057        public boolean onTouchEvent(MotionEvent event) {
058            if ((event.getAction() == MotionEvent.ACTION_CANCEL)
059                    || (event.getAction() == MotionEvent.ACTION_UP)) {
060                mIsInLongpress = false;
061            }
062            return super.onTouchEvent(event);
063        }
064            
065        public void setZoomSpeed(long speed) {
066            mZoomSpeed = speed;
067        }
068    
069        public boolean onLongClick(View v) {
070            mIsInLongpress = true;
071            mHandler.post(mRunnable);
072            return true;
073        }
074            
075        @Override
076        public boolean onKeyUp(int keyCode, KeyEvent event) {
077            mIsInLongpress = false;
078            return super.onKeyUp(keyCode, event);
079        }
080        
081        @Override
082        public void setEnabled(boolean enabled) {
083            if (!enabled) {
084                
085                /* If we're being disabled reset the state back to unpressed
086                 * as disabled views don't get events and therefore we won't
087                 * get the up event to reset the state.
088                 */
089                setPressed(false);
090            }
091            super.setEnabled(enabled);
092        }
093        
094        @Override
095        public boolean dispatchUnhandledMove(View focused, int direction) {
096            clearFocus();
097            return super.dispatchUnhandledMove(focused, direction);
098        }
099    }
