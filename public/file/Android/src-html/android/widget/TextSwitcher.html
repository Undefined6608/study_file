017    package android.widget;
018    
019    
020    import android.content.Context;
021    import android.util.AttributeSet;
022    import android.view.View;
023    import android.view.ViewGroup;
024    
025    /**
026     * Specialized {@link android.widget.ViewSwitcher} that contains
027     * only children of type {@link android.widget.TextView}.
028     *
029     * A TextSwitcher is useful to animate a label on screen. Whenever
030     * {@link #setText(CharSequence)} is called, TextSwitcher animates the current text
031     * out and animates the new text in. 
032     */
033    public class TextSwitcher extends ViewSwitcher {
034        /**
035         * Creates a new empty TextSwitcher.
036         *
037         * @param context the application's environment
038         */
039        public TextSwitcher(Context context) {
040            super(context);
041        }
042    
043        /**
044         * Creates a new empty TextSwitcher for the given context and with the
045         * specified set attributes.
046         *
047         * @param context the application environment
048         * @param attrs a collection of attributes
049         */
050        public TextSwitcher(Context context, AttributeSet attrs) {
051            super(context, attrs);
052        }
053    
054        /**
055         * {@inheritDoc}
056         *
057         * @throws IllegalArgumentException if child is not an instance of
058         *         {@link android.widget.TextView}
059         */
060        @Override
061        public void addView(View child, int index, ViewGroup.LayoutParams params) {
062            if (!(child instanceof TextView)) {
063                throw new IllegalArgumentException(
064                        "TextSwitcher children must be instances of TextView");
065            }
066    
067            super.addView(child, index, params);
068        }
069    
070        /**
071         * Sets the text of the next view and switches to the next view. This can
072         * be used to animate the old text out and animate the next text in.
073         *
074         * @param text the new text to display
075         */
076        public void setText(CharSequence text) {
077            final TextView t = (TextView) getNextView();
078            t.setText(text);
079            showNext();
080        }
081    
082        /**
083         * Sets the text of the text view that is currently showing.  This does
084         * not perform the animations.
085         *
086         * @param text the new text to display
087         */
088        public void setCurrentText(CharSequence text) {
089            ((TextView)getCurrentView()).setText(text);
090        }
091    }
