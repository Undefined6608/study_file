032    /**
033     * A view that displays one child at a time and lets the user pick among them.
034     * The items in the Spinner come from the {@link Adapter} associated with
035     * this view.
036     * 
037     * @attr ref android.R.styleable#Spinner_prompt
038     */
039    @Widget
040    public class Spinner extends AbsSpinner implements OnClickListener {
041        
042        private CharSequence mPrompt;
043    
044        public Spinner(Context context) {
045            this(context, null);
046        }
047    
048        public Spinner(Context context, AttributeSet attrs) {
049            this(context, attrs, com.android.internal.R.attr.spinnerStyle);
050        }
051    
052        public Spinner(Context context, AttributeSet attrs, int defStyle) {
053            super(context, attrs, defStyle);
054    
055            TypedArray a = context.obtainStyledAttributes(attrs,
056                    com.android.internal.R.styleable.Spinner, defStyle, 0);
057            
058            mPrompt = a.getString(com.android.internal.R.styleable.Spinner_prompt);
059    
060            a.recycle();
061        }
062    
063        @Override
064        public int getBaseline() {
065            View child = null;
066    
067            if (getChildCount() > 0) {
068                child = getChildAt(0);
069            } else if (mAdapter != null && mAdapter.getCount() > 0) {
070                child = makeAndAddView(0);
071                // TODO: We should probably put the child in the recycler
072            }
073    
074            if (child != null) {
075                return child.getTop() + child.getBaseline();
076            } else {
077                return -1;
078            }
079        }
080    
081        /**
082         * <p>A spinner does not support item click events. Calling this method
083         * will raise an exception.</p>
084         *
085         * @param l this listener will be ignored
086         */
087        @Override
088        public void setOnItemClickListener(OnItemClickListener l) {
089            throw new RuntimeException("setOnItemClickListener cannot be used with a spinner.");
090        }
091    
092        /**
093         * @see android.view.View#onLayout(boolean,int,int,int,int)
094         *
095         * Creates and positions all views
096         *
097         */
098        @Override
099        protected void onLayout(boolean changed, int l, int t, int r, int b) {
100            super.onLayout(changed, l, t, r, b);
101            mInLayout = true;
102            layout(0, false);
103            mInLayout = false;
104        }
105    
106        /**
107         * Creates and positions all views for this Spinner.
108         *
109         * @param delta Change in the selected position. +1 moves selection is moving to the right,
110         * so views are scrolling to the left. -1 means selection is moving to the left.
111         */
112        @Override
113        void layout(int delta, boolean animate) {
114            int childrenLeft = mSpinnerPadding.left;
115            int childrenWidth = mRight - mLeft - mSpinnerPadding.left - mSpinnerPadding.right;
116    
117            if (mDataChanged) {
118                handleDataChanged();
119            }
120    
121            // Handle the empty set by removing all views
122            if (mItemCount == 0) {
123                resetList();
124                return;
125            }
126    
127            if (mNextSelectedPosition >= 0) {
128                setSelectedPositionInt(mNextSelectedPosition);
129            }
130    
131            recycleAllViews();
132    
133            // Clear out old views
134            removeAllViewsInLayout();
135    
136            // Make selected view and center it
137            mFirstPosition = mSelectedPosition;
138            View sel = makeAndAddView(mSelectedPosition);
139            int width = sel.getMeasuredWidth();
140            int selectedOffset = childrenLeft + (childrenWidth / 2) - (width / 2);
141            sel.offsetLeftAndRight(selectedOffset);
142    
143            // Flush any cached views that did not get reused above
144            mRecycler.clear();
145    
146            invalidate();
147    
148            checkSelectionChanged();
149    
150            mDataChanged = false;
151            mNeedSync = false;
152            setNextSelectedPositionInt(mSelectedPosition);
153        }
154    
155        /**
156         * Obtain a view, either by pulling an existing view from the recycler or
157         * by getting a new one from the adapter. If we are animating, make sure
158         * there is enough information in the view's layout parameters to animate
159         * from the old to new positions.
160         *
161         * @param position Position in the spinner for the view to obtain
162         * @return A view that has been added to the spinner
163         */
164        private View makeAndAddView(int position) {
165    
166            View child;
167    
168            if (!mDataChanged) {
169                child = mRecycler.get(position);
170                if (child != null) {
171                    // Position the view
172                    setUpChild(child);
173    
174                    return child;
175                }
176            }
177    
178            // Nothing found in the recycler -- ask the adapter for a view
179            child = mAdapter.getView(position, null, this);
180    
181            // Position the view
182            setUpChild(child);
183    
184            return child;
185        }
186

188    
189        /**
190         * Helper for makeAndAddView to set the position of a view
191         * and fill out its layout paramters.
192         *
193         * @param child The view to position
194         */
195        private void setUpChild(View child) {
196    
197            // Respect layout params that are already in the view. Otherwise
198            // make some up...
199            ViewGroup.LayoutParams lp = child.getLayoutParams();
200            if (lp == null) {
201                lp = generateDefaultLayoutParams();
202            }
203    
204            addViewInLayout(child, 0, lp);
205    
206            child.setSelected(hasFocus());
207    
208            // Get measure specs
209            int childHeightSpec = ViewGroup.getChildMeasureSpec(mHeightMeasureSpec,
210                    mSpinnerPadding.top + mSpinnerPadding.bottom, lp.height);
211            int childWidthSpec = ViewGroup.getChildMeasureSpec(mWidthMeasureSpec,
212                    mSpinnerPadding.left + mSpinnerPadding.right, lp.width);
213    
214            // Measure child
215            child.measure(childWidthSpec, childHeightSpec);
216    
217            int childLeft;
218            int childRight;
219    
220            // Position vertically based on gravity setting
221            int childTop = mSpinnerPadding.top
222                    + ((mMeasuredHeight - mSpinnerPadding.bottom - 
223                            mSpinnerPadding.top - child.getMeasuredHeight()) / 2);
224            int childBottom = childTop + child.getMeasuredHeight();
225    
226            int width = child.getMeasuredWidth();
227            childLeft = 0;
228            childRight = childLeft + width;
229    
230            child.layout(childLeft, childTop, childRight, childBottom);
231        }
232    
233        @Override
234        public boolean performClick() {
235            boolean handled = super.performClick();
236            
237            if (!handled) {
238                handled = true;
239                Context context = getContext();
240                
241                final DropDownAdapter adapter = new DropDownAdapter(getAdapter());
242    
243                AlertDialog.Builder builder = new AlertDialog.Builder(context);
244                if (mPrompt != null) {
245                    builder.setTitle(mPrompt);
246                }
247                builder.setSingleChoiceItems(adapter, getSelectedItemPosition(), this).show();
248            }
249    
250            return handled;
251        }
252        
253        public void onClick(DialogInterface dialog, int which) {
254            setSelection(which);
255            dialog.dismiss();
256        }

258        /**
259         * Sets the prompt to display when the dialog is shown.
260         * @param prompt the prompt to set
261         */
262        public void setPrompt(CharSequence prompt) {
263            mPrompt = prompt;
264        }
265    
266        /**
267         * Sets the prompt to display when the dialog is shown.
268         * @param promptId the resource ID of the prompt to display when the dialog is shown
269         */
270        public void setPromptId(int promptId) {
271            mPrompt = getContext().getText(promptId);
272        }
273    
274        /**
275         * @return The prompt to display when the dialog is shown
276         */
277        public CharSequence getPrompt() {
278            return mPrompt;
279        }

281        /**
282         * <p>Wrapper class for an Adapter. Transforms the embedded Adapter instance
283         * into a ListAdapter.</p>
284         */
285        private static class DropDownAdapter implements ListAdapter, SpinnerAdapter {
286            private SpinnerAdapter mAdapter;
287    
288            /**
289             * <p>Creates a new ListAddapter wrapper for the specified adapter.</p>
290             *
291             * @param adapter the Adapter to transform into a ListAdapter
292             */
293            public DropDownAdapter(SpinnerAdapter adapter) {
294                this.mAdapter = adapter;
295            }
296    
297            public int getCount() {
298                return mAdapter == null ? 0 : mAdapter.getCount();
299            }
300    
301            public Object getItem(int position) {
302                return mAdapter == null ? null : mAdapter.getItem(position);
303            }
304    
305            public long getItemId(int position) {
306                return mAdapter == null ? -1 : mAdapter.getItemId(position);
307            }
308    
309            public View getView(int position, View convertView, ViewGroup parent) {
310                return getDropDownView(position, convertView, parent);
311            }
312    
313            public View getDropDownView(int position, View convertView, ViewGroup parent) {
314                return mAdapter == null ? null :
315                        mAdapter.getDropDownView(position, convertView, parent);
316            }
317    
318            public boolean hasStableIds() {
319                return mAdapter != null && mAdapter.hasStableIds();
320            }
321    
322            public void registerDataSetObserver(DataSetObserver observer) {
323                if (mAdapter != null) {
324                    mAdapter.registerDataSetObserver(observer);
325                }
326            }
327    
328            public void unregisterDataSetObserver(DataSetObserver observer) {
329                if (mAdapter != null) {
330                    mAdapter.unregisterDataSetObserver(observer);
331                }
332            }
333    
334            /**
335             * <p>Always returns false.</p>
336             *
337             * @return false
338             */
339            public boolean areAllItemsEnabled() {
340                return true;
341            }
342    
343            /**
344             * <p>Always returns false.</p>
345             *
346             * @return false
347             */
348            public boolean isEnabled(int position) {
349                return true;
350            }
351    
352            public int getItemViewType(int position) {
353                return 0;
354            }
355    
356            public int getViewTypeCount() {
357                return 1;
358            }
359            
360            public boolean isEmpty() {
361                return getCount() == 0;
362            }
363        }
