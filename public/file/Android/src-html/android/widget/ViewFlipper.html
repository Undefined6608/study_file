017    package android.widget;
018    
019    
020    import android.content.Context;
021    import android.content.res.TypedArray;
022    import android.os.Handler;
023    import android.os.Message;
024    import android.util.AttributeSet;
025    import android.widget.RemoteViews.RemoteView;
026    
027    /**
028     * Simple {@link ViewAnimator} that will animate between two or more views
029     * that have been added to it.  Only one child is shown at a time.  If
030     * requested, can automatically flip between each child at a regular interval.
031     *
032     * @attr ref android.R.styleable#ViewFlipper_flipInterval
033     */
034    public class ViewFlipper extends ViewAnimator {
035        private int mFlipInterval = 3000;
036        private boolean mKeepFlipping = false;
037    
038        public ViewFlipper(Context context) {
039            super(context);
040        }
041    
042        public ViewFlipper(Context context, AttributeSet attrs) {
043            super(context, attrs);
044    
045            TypedArray a = context.obtainStyledAttributes(attrs,
046                    com.android.internal.R.styleable.ViewFlipper);
047            mFlipInterval = a.getInt(com.android.internal.R.styleable.ViewFlipper_flipInterval,
048                    3000);
049            a.recycle();
050        }
051    
052        /**
053         * How long to wait before flipping to the next view
054         * 
055         * @param milliseconds
056         *            time in milliseconds
057         */
058        @android.view.RemotableViewMethod
059        public void setFlipInterval(int milliseconds) {
060            mFlipInterval = milliseconds;
061        }
062    
063        /**
064         * Start a timer to cycle through child views
065         */
066        public void startFlipping() {
067            if (!mKeepFlipping) {
068                mKeepFlipping = true;
069                showOnly(mWhichChild);
070                Message msg = mHandler.obtainMessage(FLIP_MSG);
071                mHandler.sendMessageDelayed(msg, mFlipInterval);
072            }
073        }
074    
075        /**
076         * No more flips
077         */
078        public void stopFlipping() {
079            mKeepFlipping = false;
080        }
081    
082        /**
083         * Returns true if the child views are flipping.
084         */
085        public boolean isFlipping() {
086            return mKeepFlipping;
087        }
088    
089        private final int FLIP_MSG = 1;
090    
091        private final Handler mHandler = new Handler() {
092            @Override
093            public void handleMessage(Message msg) {
094                if (msg.what == FLIP_MSG) {
095                    if (mKeepFlipping) {
096                        showNext();
097                        msg = obtainMessage(FLIP_MSG);
098                        sendMessageDelayed(msg, mFlipInterval);
099                    }
100                }
101            }
102        };
103    }
