017    package android.opengl;
018    
019    /**
020     * A collection of utility methods for computing the visibility of triangle
021     * meshes.
022     *
023     */
024    public class Visibility {
025        /**
026         * Test whether a given triangle mesh is visible on the screen. The mesh
027         * is specified as an indexed triangle list.
028         *
029         * @param ws the world space to screen space transform matrix, as an OpenGL
030         * column matrix.
031         * @param wsOffset an index into the ws array where the data starts.
032         * @param positions the vertex positions (x, y, z).
033         * @param positionsOffset the index in the positions array where the data
034         *        starts.
035         * @param indices the indices of the triangle list. The indices are
036         * expressed as chars because they are unsigned 16-bit values.
037         * @param indicesOffset the index in the indices array where the index data
038         *        starts.
039         * @param indexCount the number of indices in use. Typically a multiple of
040         * three. If not a multiple of three, the remaining one or two indices will
041         * be ignored.
042         * @return 2 if all of the mesh is visible, 1 if some part of the mesh is
043         *         visible, 0 if no part is visible.
044         *
045         * @throws IllegalArgumentException if ws is null, wsOffset < 0,
046         * positions is null, positionsOffset < 0, indices is null,
047         * indicesOffset < 0, indicesOffset > indices.length - indexCount
048         */
049        public static native int visibilityTest(float[] ws, int wsOffset,
050                float[] positions, int positionsOffset, char[] indices,
051                int indicesOffset, int indexCount);
052    
053        /**
054         * Given an OpenGL ES ModelView-Projection matrix (which implicitly
055         * describes a frustum) and a list of spheres, determine which spheres
056         * intersect the frustum.
057         * <p>
058         * A ModelView-Projection matrix can be computed by multiplying the
059         * a Projection matrix by the a ModelView matrix (in that order.). There
060         * are several possible ways to obtain the current ModelView and
061         * Projection matrices. The most generally applicable way is to keep
062         * track of the current matrices in application code. If that is not
063         * convenient, there are two optional OpenGL ES extensions which may
064         * be used to read the current matrices from OpenGL ES:
065         * <ul>
066         * <li>GL10Ext.glQueryMatrixxOES
067         * <li>GL11.GL_MODELVIEW_MATRIX_FLOAT_AS_INT_BITS_OES and
068         * GL_PROJECTION_MATRIX_FLOAT_AS_INT_BITS_OES
069         * </ul>
070         * The problem with reading back the matrices is that your application
071         * will only work with devices that support the extension(s) that
072         * it uses.
073         * <p>
074         * A frustum is a six-sided truncated pyramid that defines the portion of
075         * world space that is visible in the view.
076         * <p>
077         * Spheres are described as four floating point values: x, y, z, and r, in
078         * world-space coordinates. R is the radius of the sphere.
079         * <p>
080         * @param mvp a float array containing the mode-view-projection matrix
081         * @param mvpOffset The offset of the mvp data within the mvp array.
082         * @param spheres a float array containing the sphere data.
083         * @param spheresOffset an offset into the sphere array where the sphere
084         *        data starts
085         * @param spheresCount the number of spheres to cull.
086         * @param results an integer array containing the indices of the spheres
087         * that are either contained entirely within or intersect the frustum.
088         * @param resultsOffset an offset into the results array where the results
089         *        start.
090         * @param resultsCapacity the number of array elements available for storing
091         *        results.
092         * @return the number of spheres that intersected the frustum. Can be
093         * larger than resultsCapacity, in which case only the first resultsCapacity
094         * results are written into the results array.
095         *
096         * @throws IllegalArgumentException if mvp is null, mvpOffset < 0,
097         * mvpOffset > mvp.length - 16, spheres is null, spheresOffset < 0,
098         * spheresOffset > spheres.length - sphereCount,
099         * results is null, resultsOffset < 0, resultsOffset > results.length -
100         * resultsCapacity.
101         */
102        public static native int frustumCullSpheres(float[] mvp, int mvpOffset,
103                float[] spheres, int spheresOffset, int spheresCount,
104                int[] results, int resultsOffset, int resultsCapacity);
105    
106        /**
107         * Compute a bounding sphere for a set of points. It is approximately the
108         * minimal bounding sphere of an axis-aligned box that bounds the points.
109         *
110         * @param positions positions in x, y, z triples
111         * @param positionsOffset offset into positions array
112         * @param positionsCount number of position triples to process
113         * @param sphere array containing the output as (x, y, z, r)
114         * @param sphereOffset offset where the sphere data will be written
115         *
116         * @throws IllegalArgumentException if positions is null,
117         * positionsOffset < 0, positionsOffset > positions.length - positionsCount,
118         * sphere is null, sphereOffset < 0, sphereOffset > sphere.length - 4.
119         */
120        public static native void computeBoundingSphere(float[] positions,
121                int positionsOffset, int positionsCount, float[] sphere,
122                int sphereOffset);
123    }
