017    package android.net;
018    
019    import java.util.HashMap;
020    import java.util.Map;
021    import java.util.Set;
022    
023    /**
024     *
025     * MailTo URL parser
026     *
027     * This class parses a mailto scheme URL and then can be queried for
028     * the parsed parameters. This implements RFC 2368.
029     *
030     */
031    public class MailTo {
032        
033        static public final String MAILTO_SCHEME = "mailto:";
034        
035        // All the parsed content is added to the headers.
036        private HashMap<String, String> mHeaders;
037        
038        // Well known headers
039        static private final String TO = "to";
040        static private final String BODY = "body";
041        static private final String CC = "cc";
042        static private final String SUBJECT = "subject";
043    
044        
045        /**
046         * Test to see if the given string is a mailto URL
047         * @param url string to be tested
048         * @return true if the string is a mailto URL
049         */
050        public static boolean isMailTo(String url) {
051            if (url != null && url.startsWith(MAILTO_SCHEME)) {
052                return true;
053            }
054            return false;
055        }
056        
057        /**
058         * Parse and decode a mailto scheme string. This parser implements
059         * RFC 2368. The returned object can be queried for the parsed parameters.
060         * @param url String containing a mailto URL
061         * @return MailTo object
062         * @exception ParseException if the scheme is not a mailto URL
063         */
064        public static MailTo parse(String url) throws ParseException {
065            if (url == null) {
066                throw new NullPointerException();
067            }
068            if (!isMailTo(url)) {
069                 throw new ParseException("Not a mailto scheme");
070            }
071            // Strip the scheme as the Uri parser can't cope with it.
072            String noScheme = url.substring(MAILTO_SCHEME.length());
073            Uri email = Uri.parse(noScheme);
074            MailTo m = new MailTo();
075            
076            // Parse out the query parameters
077            String query = email.getQuery();
078            if (query != null ) {
079                String[] queries = query.split("&");
080                for (String q : queries) {
081                    String[] nameval = q.split("=");
082                    if (nameval.length == 0) {
083                        continue;
084                    }
085                    // insert the headers with the name in lowercase so that
086                    // we can easily find common headers
087                    m.mHeaders.put(Uri.decode(nameval[0]).toLowerCase(), 
088                            nameval.length > 1 ? Uri.decode(nameval[1]) : null);
089                }
090            }
091            
092            // Address can be specified in both the headers and just after the
093            // mailto line. Join the two together.
094            String address = email.getPath();
095            if (address != null) {
096                String addr = m.getTo();
097                if (addr != null) {
098                    address += ", " + addr;
099                }
100                m.mHeaders.put(TO, address);
101            }
102            
103            return m;
104        }
105        
106        /**
107         * Retrieve the To address line from the parsed mailto URL. This could be
108         * several email address that are comma-space delimited.
109         * If no To line was specified, then null is return
110         * @return comma delimited email addresses or null
111         */
112        public String getTo() {
113            return mHeaders.get(TO);
114        }
115        
116        /**
117         * Retrieve the CC address line from the parsed mailto URL. This could be
118         * several email address that are comma-space delimited.
119         * If no CC line was specified, then null is return
120         * @return comma delimited email addresses or null
121         */
122        public String getCc() {
123            return mHeaders.get(CC);
124        }
125        
126        /**
127         * Retrieve the subject line from the parsed mailto URL.
128         * If no subject line was specified, then null is return
129         * @return subject or null
130         */
131        public String getSubject() {
132            return mHeaders.get(SUBJECT);
133        }
134        
135        /**
136         * Retrieve the body line from the parsed mailto URL.
137         * If no body line was specified, then null is return
138         * @return body or null
139         */
140        public String getBody() {
141            return mHeaders.get(BODY);
142        }
143        
144        /**
145         * Retrieve all the parsed email headers from the mailto URL
146         * @return map containing all parsed values
147         */
148        public Map<String, String> getHeaders() {
149            return mHeaders;
150        }
151        
152        @Override
153        public String toString() {
154            StringBuilder sb = new StringBuilder(MAILTO_SCHEME);
155            sb.append('?');
156            for (Map.Entry<String,String> header : mHeaders.entrySet()) {
157                sb.append(Uri.encode(header.getKey()));
158                sb.append('=');
159                sb.append(Uri.encode(header.getValue()));
160                sb.append('&');
161            }
162            return sb.toString();
163        }
164        
165        /**
166         * Private constructor. The only way to build a Mailto object is through
167         * the parse() method.
168         */
169        private MailTo() {
170            mHeaders = new HashMap<String, String>();
171        }
172    }
