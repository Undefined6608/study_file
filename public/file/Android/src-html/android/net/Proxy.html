017    package android.net;
018    
019    import android.content.ContentResolver;
020    import android.content.Context;
021    import android.os.SystemProperties;
022    import android.provider.Settings;
023    import android.util.Log;
024    
025    import junit.framework.Assert;
026    
027    /**
028     * A convenience class for accessing the user and default proxy
029     * settings.
030     */
031    final public class Proxy {
032    
033        static final public String PROXY_CHANGE_ACTION =
034            "android.intent.action.PROXY_CHANGE";
035    
036        /**
037         * Return the proxy host set by the user.
038         * @param ctx A Context used to get the settings for the proxy host.
039         * @return String containing the host name. If the user did not set a host
040         *         name it returns the default host. A null value means that no
041         *         host is to be used.
042         */
043        static final public String getHost(Context ctx) {
044            ContentResolver contentResolver = ctx.getContentResolver();
045            Assert.assertNotNull(contentResolver);
046            String host = Settings.Secure.getString(
047                    contentResolver,
048                    Settings.Secure.HTTP_PROXY);
049            if (host != null) {
050                int i = host.indexOf(':');
051                if (i == -1) {
052                    if (android.util.Config.DEBUG) {
053                        Assert.assertTrue(host.length() == 0);
054                    }
055                    return null;
056                }
057                return host.substring(0, i);
058            }
059            return getDefaultHost();
060        }
061    
062        /**
063         * Return the proxy port set by the user.
064         * @param ctx A Context used to get the settings for the proxy port.
065         * @return The port number to use or -1 if no proxy is to be used.
066         */
067        static final public int getPort(Context ctx) {
068            ContentResolver contentResolver = ctx.getContentResolver();
069            Assert.assertNotNull(contentResolver);
070            String host = Settings.Secure.getString(
071                    contentResolver,
072                    Settings.Secure.HTTP_PROXY);
073            if (host != null) {
074                int i = host.indexOf(':');
075                if (i == -1) {
076                    if (android.util.Config.DEBUG) {
077                        Assert.assertTrue(host.length() == 0);
078                    }
079                    return -1;
080                }
081                if (android.util.Config.DEBUG) {
082                    Assert.assertTrue(i < host.length());
083                }
084                return Integer.parseInt(host.substring(i+1));
085            }
086            return getDefaultPort();
087        }
088    
089        /**
090         * Return the default proxy host specified by the carrier.
091         * @return String containing the host name or null if there is no proxy for
092         * this carrier.
093         */
094        static final public String getDefaultHost() {
095            String host = SystemProperties.get("net.gprs.http-proxy");
096            if (host != null) {
097                Uri u = Uri.parse(host);
098                host = u.getHost();
099                return host;
100            } else {
101                return null;
102            }
103        }
104    
105        /**
106         * Return the default proxy port specified by the carrier.
107         * @return The port number to be used with the proxy host or -1 if there is
108         * no proxy for this carrier.
109         */
110        static final public int getDefaultPort() {
111            String host = SystemProperties.get("net.gprs.http-proxy");
112            if (host != null) {
113                Uri u = Uri.parse(host);
114                return u.getPort();
115            } else {
116                return -1;
117            }
118        }
119    
120    };
