017    package android.net;
018    
019    import android.os.Parcelable;
020    import android.os.Parcel;
021    
022    /**
023     * A simple object for retrieving the results of a DHCP request.
024     */
025    public class DhcpInfo implements Parcelable {
026        public int ipAddress;
027        public int gateway;
028        public int netmask;
029    
030        public int dns1;
031        public int dns2;
032    
033        public int serverAddress;
034        public int leaseDuration;
035    
036        public DhcpInfo() {
037            super();
038        }
039    
040        public String toString() {
041            StringBuffer str = new StringBuffer();
042    
043            str.append("ipaddr "); putAddress(str, ipAddress);
044            str.append(" gateway "); putAddress(str, gateway);
045            str.append(" netmask "); putAddress(str, netmask);
046            str.append(" dns1 "); putAddress(str, dns1);
047            str.append(" dns2 "); putAddress(str, dns2);
048            str.append(" DHCP server "); putAddress(str, serverAddress);
049            str.append(" lease ").append(leaseDuration).append(" seconds");
050    
051            return str.toString();
052        }
053    
054        private static void putAddress(StringBuffer buf, int addr) {
055            buf.append(addr  & 0xff).append('.').
056                append((addr >>>= 8) & 0xff).append('.').
057                append((addr >>>= 8) & 0xff).append('.').
058                append((addr >>>= 8) & 0xff);
059        }
060    
061        /** Implement the Parcelable interface {@hide} */
062        public int describeContents() {
063            return 0;
064        }
065    
066        /** Implement the Parcelable interface {@hide} */
067        public void writeToParcel(Parcel dest, int flags) {
068            dest.writeInt(ipAddress);
069            dest.writeInt(gateway);
070            dest.writeInt(netmask);
071            dest.writeInt(dns1);
072            dest.writeInt(dns2);
073            dest.writeInt(serverAddress);
074            dest.writeInt(leaseDuration);
075        }
076    
077        /** Implement the Parcelable interface {@hide} */
078        public static final Creator<DhcpInfo> CREATOR =
079            new Creator<DhcpInfo>() {
080                public DhcpInfo createFromParcel(Parcel in) {
081                    DhcpInfo info = new DhcpInfo();
082                    info.ipAddress = in.readInt();
083                    info.gateway = in.readInt();
084                    info.netmask = in.readInt();
085                    info.dns1 = in.readInt();
086                    info.dns2 = in.readInt();
087                    info.serverAddress = in.readInt();
088                    info.leaseDuration = in.readInt();
089                    return info;
090                }
091    
092                public DhcpInfo[] newArray(int size) {
093                    return new DhcpInfo[size];
094                }
095            };
096    }
