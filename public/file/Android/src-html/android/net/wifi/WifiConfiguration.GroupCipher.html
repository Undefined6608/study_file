024    /**
025     * A class representing a configured Wi-Fi network, including the
026     * security configuration. Android will not necessarily support
027     * all of these security schemes initially.
028     */

046        /**
047         * Recognized key management schemes.
048         */
049        public static class KeyMgmt {
050            private KeyMgmt() { }
051            
052            /** WPA is not used; plaintext or static WEP could be used. */
053            public static final int NONE = 0;
054            /** WPA pre-shared key (requires {@code preSharedKey} to be specified). */
055            public static final int WPA_PSK = 1;
056            /** WPA using EAP authentication. Generally used with an external authentication server. */
057            public static final int WPA_EAP = 2;
058            /** IEEE 802.1X using EAP authentication and (optionally) dynamically
059             * generated WEP keys. */
060            public static final int IEEE8021X = 3;
061    
062            public static final String varName = "key_mgmt";
063    
064            public static final String[] strings = { "NONE", "WPA_PSK", "WPA_EAP", "IEEE8021X" };
065        }

068         * Recognized security protocols.
069         */
070        public static class Protocol {

072    
073            /** WPA/IEEE 802.11i/D3.0 */
074            public static final int WPA = 0;
075            /** WPA2/IEEE 802.11i */
076            public static final int RSN = 1;
077

079

084         * Recognized IEEE 802.11 authentication algorithms.
085         */
086        public static class AuthAlgorithm {
087            private AuthAlgorithm() { }
088    
089            /** Open System authentication (required for WPA/WPA2) */
090            public static final int OPEN = 0;
091            /** Shared Key authentication (requires static WEP keys) */
092            public static final int SHARED = 1;
093            /** LEAP/Network EAP (only used with LEAP) */
094            public static final int LEAP = 2;
095    
096            public static final String varName = "auth_alg";
097    
098            public static final String[] strings = { "OPEN", "SHARED", "LEAP" };

101        /**
102         * Recognized pairwise ciphers for WPA.
103         */
104        public static class PairwiseCipher {
105            private PairwiseCipher() { }
106    
107            /** Use only Group keys (deprecated) */
108            public static final int NONE = 0;
109            /** Temporal Key Integrity Protocol [IEEE 802.11i/D7.0] */
110            public static final int TKIP = 1;
111            /** AES in Counter mode with CBC-MAC [RFC 3610, IEEE 802.11i/D7.0] */
112            public static final int CCMP = 2;
113    
114            public static final String varName = "pairwise";
115            
116            public static final String[] strings = { "NONE", "TKIP", "CCMP" };
117        }
118    
119        /**
120         * Recognized group ciphers.
121         * <pre>
122         * CCMP = AES in Counter mode with CBC-MAC [RFC 3610, IEEE 802.11i/D7.0]
123         * TKIP = Temporal Key Integrity Protocol [IEEE 802.11i/D7.0]
124         * WEP104 = WEP (Wired Equivalent Privacy) with 104-bit key
125         * WEP40 = WEP (Wired Equivalent Privacy) with 40-bit key (original 802.11)
126         * </pre>
127         */
128        public static class GroupCipher {
129            private GroupCipher() { }
130    
131            /** WEP40 = WEP (Wired Equivalent Privacy) with 40-bit key (original 802.11) */
132            public static final int WEP40 = 0;
133            /** WEP104 = WEP (Wired Equivalent Privacy) with 104-bit key */
134            public static final int WEP104 = 1;
135            /** Temporal Key Integrity Protocol [IEEE 802.11i/D7.0] */
136            public static final int TKIP = 2;
137            /** AES in Counter mode with CBC-MAC [RFC 3610, IEEE 802.11i/D7.0] */
138            public static final int CCMP = 3;
139    
140            public static final String varName = "group";
141    
142            public static final String[] strings = { "WEP40", "WEP104", "TKIP", "CCMP" };
143        }

145        /** Possible status of a network configuration. */
146        public static class Status {
147            private Status() { }
148    
149            /** this is the network we are currently connected to */
150            public static final int CURRENT = 0;
151            /** supplicant will not attempt to use this network */
152            public static final int DISABLED = 1;
153            /** supplicant will consider this network available for association */
154            public static final int ENABLED = 2;
155    
156            public static final String[] strings = { "current", "disabled", "enabled" };
157        }
158    
159        /**
160         * The ID number that the supplicant uses to identify this
161         * network configuration entry. This must be passed as an argument
162         * to most calls into the supplicant.
163         */
164        public int networkId;
165    
166        /**
167         * The current status of this network configuration entry.
168         * @see Status
169         */
170        public int status;
171        /**
172         * The network's SSID. Can either be an ASCII string,
173         * which must be enclosed in double quotation marks
174         * (e.g., {@code &quot;MyNetwork&quot;}, or a string of
175         * hex digits,which are not enclosed in quotes
176         * (e.g., {@code 01a243f405}).
177         */
178        public String SSID;
179        /**
180         * When set, this network configuration entry should only be used when
181         * associating with the AP having the specified BSSID. The value is
182         * a string in the format of an Ethernet MAC address, e.g.,
183         * <code>XX:XX:XX:XX:XX:XX</code> where each <code>X</code> is a hex digit.
184         */
185        public String BSSID;
186    
187        /**
188         * Pre-shared key for use with WPA-PSK.
189         * <p/>
190         * When the value of this key is read, the actual key is
191         * not returned, just a "*" if the key has a value, or the null
192         * string otherwise.
193         */
194        public String preSharedKey;
195        /**
196         * Up to four WEP keys. Either an ASCII string enclosed in double
197         * quotation marks (e.g., {@code &quot;abcdef&quot;} or a string
198         * of hex digits (e.g., {@code 0102030405}).
199         * <p/>
200         * When the value of one of these keys is read, the actual key is
201         * not returned, just a "*" if the key has a value, or the null
202         * string otherwise.
203         */
204        public String[] wepKeys;
205        
206        /** Default WEP key index, ranging from 0 to 3. */
207        public int wepTxKeyIndex;
208    
209        /**
210         * Priority determines the preference given to a network by {@code wpa_supplicant}
211         * when choosing an access point with which to associate.
212         */
213        public int priority;
214    
215        /**
216         * This is a network that does not broadcast its SSID, so an
217         * SSID-specific probe request must be used for scans.
218         */
219        public boolean hiddenSSID;
220    
221        /**
222         * The set of key management protocols supported by this configuration.
223         * See {@link KeyMgmt} for descriptions of the values.
224         * Defaults to WPA-PSK WPA-EAP.
225         */
226        public BitSet allowedKeyManagement;
227        /**
228         * The set of security protocols supported by this configuration.
229         * See {@link Protocol} for descriptions of the values.
230         * Defaults to WPA RSN.
231         */
232        public BitSet allowedProtocols;
233        /**
234         * The set of authentication protocols supported by this configuration.
235         * See {@link AuthAlgorithm} for descriptions of the values.
236         * Defaults to automatic selection.
237         */
238        public BitSet allowedAuthAlgorithms;
239        /**
240         * The set of pairwise ciphers for WPA supported by this configuration.
241         * See {@link PairwiseCipher} for descriptions of the values.
242         * Defaults to CCMP TKIP.
243         */
244        public BitSet allowedPairwiseCiphers;
245        /**
246         * The set of group ciphers supported by this configuration.
247         * See {@link GroupCipher} for descriptions of the values.
248         * Defaults to CCMP TKIP WEP104 WEP40.
249         */
250        public BitSet allowedGroupCiphers;
251    
252        public WifiConfiguration() {
253            networkId = -1;
254            SSID = null;
255            BSSID = null;
256            priority = 0;
257            hiddenSSID = false;
258            allowedKeyManagement = new BitSet();
259            allowedProtocols = new BitSet();
260            allowedAuthAlgorithms = new BitSet();
261            allowedPairwiseCiphers = new BitSet();
262            allowedGroupCiphers = new BitSet();
263            wepKeys = new String[4];
264            for (int i = 0; i < wepKeys.length; i++)
265                wepKeys[i] = null;
266        }
267    
268        public String toString() {
269            StringBuffer sbuf = new StringBuffer();
270            if (this.status == WifiConfiguration.Status.CURRENT) {
271                sbuf.append("* ");
272            } else if (this.status == WifiConfiguration.Status.DISABLED) {
273                sbuf.append("- ");
274            }
275            sbuf.append("ID: ").append(this.networkId).append(" SSID: ").append(this.SSID).
276                    append(" BSSID: ").append(this.BSSID).append(" PRIO: ").append(this.priority).
277                    append('\n');
278            sbuf.append(" KeyMgmt:");
279            for (int k = 0; k < this.allowedKeyManagement.size(); k++) {
280                if (this.allowedKeyManagement.get(k)) {
281                    sbuf.append(" ");
282                    if (k < KeyMgmt.strings.length) {
283                        sbuf.append(KeyMgmt.strings[k]);
284                    } else {
285                        sbuf.append("??");
286                    }
287                }
288            }
289            sbuf.append(" Protocols:");
290            for (int p = 0; p < this.allowedProtocols.size(); p++) {
291                if (this.allowedProtocols.get(p)) {
292                    sbuf.append(" ");
293                    if (p < Protocol.strings.length) {
294                        sbuf.append(Protocol.strings[p]);
295                    } else {
296                        sbuf.append("??");
297                    }
298                }
299            }
300            sbuf.append('\n');
301            sbuf.append(" AuthAlgorithms:");
302            for (int a = 0; a < this.allowedAuthAlgorithms.size(); a++) {
303                if (this.allowedAuthAlgorithms.get(a)) {
304                    sbuf.append(" ");
305                    if (a < AuthAlgorithm.strings.length) {
306                        sbuf.append(AuthAlgorithm.strings[a]);
307                    } else {
308                        sbuf.append("??");
309                    }
310                }
311            }
312            sbuf.append('\n');
313            sbuf.append(" PairwiseCiphers:");
314            for (int pc = 0; pc < this.allowedPairwiseCiphers.size(); pc++) {
315                if (this.allowedPairwiseCiphers.get(pc)) {
316                    sbuf.append(" ");
317                    if (pc < PairwiseCipher.strings.length) {
318                        sbuf.append(PairwiseCipher.strings[pc]);
319                    } else {
320                        sbuf.append("??");
321                    }
322                }
323            }
324            sbuf.append('\n');
325            sbuf.append(" GroupCiphers:");
326            for (int gc = 0; gc < this.allowedGroupCiphers.size(); gc++) {
327                if (this.allowedGroupCiphers.get(gc)) {
328                    sbuf.append(" ");
329                    if (gc < GroupCipher.strings.length) {
330                        sbuf.append(GroupCipher.strings[gc]);
331                    } else {
332                        sbuf.append("??");
333                    }
334                }
335            }
336            sbuf.append('\n');
337            if (this.preSharedKey != null) {
338                sbuf.append(" PSK: ").append('*');
339            }
340            return sbuf.toString();
341        }

355        private static BitSet readBitSet(Parcel src) {
356            int cardinality = src.readInt();
357    
358            BitSet set = new BitSet();
359            for (int i = 0; i < cardinality; i++)
360                set.set(src.readInt());
361    
362            return set;
363        }
364    
365        private static void writeBitSet(Parcel dest, BitSet set) {
366            int nextSetBit = -1;
367    
368            dest.writeInt(set.cardinality());
369    
370            while ((nextSetBit = set.nextSetBit(nextSetBit + 1)) != -1)
371                dest.writeInt(nextSetBit);
372        }

374        /** Implement the Parcelable interface {@hide} */
375        public int describeContents() {
376            return 0;
377        }
378    
379        /** Implement the Parcelable interface {@hide} */
380        public void writeToParcel(Parcel dest, int flags) {
381            dest.writeInt(networkId);
382            dest.writeInt(status);
383            dest.writeString(SSID);
384            dest.writeString(BSSID);
385            dest.writeString(preSharedKey);
386            for (String wepKey : wepKeys)
387                dest.writeString(wepKey);
388            dest.writeInt(wepTxKeyIndex);
389            dest.writeInt(priority);
390            dest.writeInt(hiddenSSID ? 1 : 0);
391    
392            writeBitSet(dest, allowedKeyManagement);
393            writeBitSet(dest, allowedProtocols);
394            writeBitSet(dest, allowedAuthAlgorithms);
395            writeBitSet(dest, allowedPairwiseCiphers);
396            writeBitSet(dest, allowedGroupCiphers);
397        }

399        /** Implement the Parcelable interface {@hide} */
400        public static final Creator<WifiConfiguration> CREATOR =
401            new Creator<WifiConfiguration>() {
402                public WifiConfiguration createFromParcel(Parcel in) {
403                    WifiConfiguration config = new WifiConfiguration();
404                    config.networkId = in.readInt();
405                    config.status = in.readInt();
406                    config.SSID = in.readString();
407                    config.BSSID = in.readString();
408                    config.preSharedKey = in.readString();
409                    for (int i = 0; i < config.wepKeys.length; i++)
410                        config.wepKeys[i] = in.readString();
411                    config.wepTxKeyIndex = in.readInt();
412                    config.priority = in.readInt();
413                    config.hiddenSSID = in.readInt() != 0;
414                    config.allowedKeyManagement   = readBitSet(in);
415                    config.allowedProtocols       = readBitSet(in);
416                    config.allowedAuthAlgorithms  = readBitSet(in);
417                    config.allowedPairwiseCiphers = readBitSet(in);
418                    config.allowedGroupCiphers    = readBitSet(in);
419                    return config;
420                }
421    
422                public WifiConfiguration[] newArray(int size) {
423                    return new WifiConfiguration[size];
424                }
425            };
