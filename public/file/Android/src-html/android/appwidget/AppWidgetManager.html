040        /**
041         * Send this from your {@link AppWidgetHost} activity when you want to pick an AppWidget to display.
042         * The AppWidget picker activity will be launched.
043         * <p>
044         * You must supply the following extras:
045         * <table>
046         *   <tr>
047         *     <td>{@link #EXTRA_APPWIDGET_ID}</td>
048         *     <td>A newly allocated appWidgetId, which will be bound to the AppWidget provider
049         *         once the user has selected one.</td>
050         *  </tr>
051         * </table>
052         *
053         * <p>
054         * The system will respond with an onActivityResult call with the following extras in
055         * the intent:
056         * <table>
057         *   <tr>
058         *     <td>{@link #EXTRA_APPWIDGET_ID}</td>
059         *     <td>The appWidgetId that you supplied in the original intent.</td>
060         *  </tr>
061         * </table>
062         * <p>
063         * When you receive the result from the AppWidget pick activity, if the resultCode is
064         * {@link android.app.Activity#RESULT_OK}, an AppWidget has been selected.  You should then
065         * check the AppWidgetProviderInfo for the returned AppWidget, and if it has one, launch its configuration
066         * activity.  If {@link android.app.Activity#RESULT_CANCELED} is returned, you should delete
067         * the appWidgetId.
068         *
069         * @see #ACTION_APPWIDGET_CONFIGURE
070         */
071        public static final String ACTION_APPWIDGET_PICK = "android.appwidget.action.APPWIDGET_PICK";
072    
073        /**
074         * Sent when it is time to configure your AppWidget while it is being added to a host.
075         * This action is not sent as a broadcast to the AppWidget provider, but as a startActivity
076         * to the activity specified in the {@link AppWidgetProviderInfo AppWidgetProviderInfo meta-data}.
077         *
078         * <p>
079         * The intent will contain the following extras:
080         * <table>
081         *   <tr>
082         *     <td>{@link #EXTRA_APPWIDGET_ID}</td>
083         *     <td>The appWidgetId to configure.</td>
084         *  </tr>
085         * </table>
086         *
087         * <p>If you return {@link android.app.Activity#RESULT_OK} using
088         * {@link android.app.Activity#setResult Activity.setResult()}, the AppWidget will be added,
089         * and you will receive an {@link #ACTION_APPWIDGET_UPDATE} broadcast for this AppWidget.
090         * If you return {@link android.app.Activity#RESULT_CANCELED}, the host will cancel the add
091         * and not display this AppWidget, and you will receive a {@link #ACTION_APPWIDGET_DELETED} broadcast.
092         */
093        public static final String ACTION_APPWIDGET_CONFIGURE = "android.appwidget.action.APPWIDGET_CONFIGURE";

185        static WeakHashMap<Context, WeakReference<AppWidgetManager>> sManagerCache = new WeakHashMap();
186        static IAppWidgetService sService;
187        
188        Context mContext;
189    
190        /**
191         * Get the AppWidgetManager instance to use for the supplied {@link android.content.Context
192         * Context} object.
193         */
194        public static AppWidgetManager getInstance(Context context) {
195            synchronized (sManagerCache) {
196                if (sService == null) {
197                    IBinder b = ServiceManager.getService(Context.APPWIDGET_SERVICE);
198                    sService = IAppWidgetService.Stub.asInterface(b);
199                }
200    
201                WeakReference<AppWidgetManager> ref = sManagerCache.get(context);
202                AppWidgetManager result = null;
203                if (ref != null) {
204                    result = ref.get();
205                }
206                if (result == null) {
207                    result = new AppWidgetManager(context);
208                    sManagerCache.put(context, new WeakReference(result));
209                }
210                return result;
211            }
212        }
213    
214        private AppWidgetManager(Context context) {
215            mContext = context;
216        }

218        /**
219         * Set the RemoteViews to use for the specified appWidgetIds.
220         *
221         * <p>
222         * It is okay to call this method both inside an {@link #ACTION_APPWIDGET_UPDATE} broadcast,
223         * and outside of the handler.
224         * This method will only work when called from the uid that owns the AppWidget provider.
225         *
226         * @param appWidgetIds     The AppWidget instances for which to set the RemoteViews.
227         * @param views         The RemoteViews object to show.
228         */
229        public void updateAppWidget(int[] appWidgetIds, RemoteViews views) {
230            try {
231                sService.updateAppWidgetIds(appWidgetIds, views);
232            }
233            catch (RemoteException e) {
234                throw new RuntimeException("system server dead?", e);
235            }
236        }
237    
238        /**
239         * Set the RemoteViews to use for the specified appWidgetId.
240         *
241         * <p>
242         * It is okay to call this method both inside an {@link #ACTION_APPWIDGET_UPDATE} broadcast,
243         * and outside of the handler.
244         * This method will only work when called from the uid that owns the AppWidget provider.
245         *
246         * @param appWidgetId      The AppWidget instance for which to set the RemoteViews.
247         * @param views         The RemoteViews object to show.
248         */
249        public void updateAppWidget(int appWidgetId, RemoteViews views) {
250            updateAppWidget(new int[] { appWidgetId }, views);
251        }

253        /**
254         * Set the RemoteViews to use for all AppWidget instances for the supplied AppWidget provider.
255         *
256         * <p>

258         * and outside of the handler.
259         * This method will only work when called from the uid that owns the AppWidget provider.
260         *
261         * @param provider      The {@link ComponentName} for the {@link
262         * android.content.BroadcastReceiver BroadcastReceiver} provider
263         *                      for your AppWidget.
264         * @param views         The RemoteViews object to show.
265         */

267            try {
268                sService.updateAppWidgetProvider(provider, views);
269            }

271                throw new RuntimeException("system server dead?", e);
272            }
273        }
