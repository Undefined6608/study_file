017    package android.appwidget;
018    
019    import android.os.Parcel;
020    import android.os.Parcelable;
021    import android.content.ComponentName;
022    
023    /**
024     * Describes the meta data for an installed AppWidget provider.  The fields in this class
025     * correspond to the fields in the <code>&lt;appwidget-provider&gt;</code> xml tag.
026     */
027    public class AppWidgetProviderInfo implements Parcelable {
028        /**
029         * Identity of this AppWidget component.  This component should be a {@link
030         * android.content.BroadcastReceiver}, and it will be sent the AppWidget intents
031         * {@link android.appwidget as described in the AppWidget package documentation}.
032         *
033         * <p>This field corresponds to the <code>android:name</code> attribute in
034         * the <code>&lt;receiver&gt;</code> element in the AndroidManifest.xml file.
035         */
036        public ComponentName provider;
037    
038        /**
039         * Minimum width of the AppWidget, in dp.
040         *
041         * <p>This field corresponds to the <code>android:minWidth</code> attribute in
042         * the AppWidget meta-data file.
043         */
044        public int minWidth;
045    
046        /**
047         * Minimum height of the AppWidget, in dp.
048         *
049         * <p>This field corresponds to the <code>android:minHeight</code> attribute in
050         * the AppWidget meta-data file.
051         */
052        public int minHeight;
053    
054        /**
055         * How often, in milliseconds, that this AppWidget wants to be updated.
056         * The AppWidget manager may place a limit on how often a AppWidget is updated.
057         *
058         * <p>This field corresponds to the <code>android:updatePeriodMillis</code> attribute in
059         * the AppWidget meta-data file.
060         */
061        public int updatePeriodMillis;
062    
063        /**
064         * The resource id of the initial layout for this AppWidget.  This should be
065         * displayed until the RemoteViews for the AppWidget is available.
066         *
067         * <p>This field corresponds to the <code>android:initialLayout</code> attribute in
068         * the AppWidget meta-data file.
069         */
070        public int initialLayout;
071    
072        /**
073         * The activity to launch that will configure the AppWidget.
074         *
075         * <p>This class name of field corresponds to the <code>android:configure</code> attribute in
076         * the AppWidget meta-data file.  The package name always corresponds to the package containing
077         * the AppWidget provider.
078         */
079        public ComponentName configure;
080    
081        /**
082         * The label to display to the user in the AppWidget picker.  If not supplied in the
083         * xml, the application label will be used.
084         *
085         * <p>This field corresponds to the <code>android:label</code> attribute in
086         * the <code>&lt;receiver&gt;</code> element in the AndroidManifest.xml file.
087         */
088        public String label;
089    
090        /**
091         * The icon to display for this AppWidget in the AppWidget picker.  If not supplied in the
092         * xml, the application icon will be used.
093         *
094         * <p>This field corresponds to the <code>android:icon</code> attribute in
095         * the <code>&lt;receiver&gt;</code> element in the AndroidManifest.xml file.
096         */
097        public int icon;
098    
099        public AppWidgetProviderInfo() {
100        }
101    
102        /**
103         * Unflatten the AppWidgetProviderInfo from a parcel.
104         */
105        public AppWidgetProviderInfo(Parcel in) {
106            if (0 != in.readInt()) {
107                this.provider = new ComponentName(in);
108            }
109            this.minWidth = in.readInt();
110            this.minHeight = in.readInt();
111            this.updatePeriodMillis = in.readInt();
112            this.initialLayout = in.readInt();
113            if (0 != in.readInt()) {
114                this.configure = new ComponentName(in);
115            }
116            this.label = in.readString();
117            this.icon = in.readInt();
118        }
119    
120    
121        public void writeToParcel(android.os.Parcel out, int flags) {
122            if (this.provider != null) {
123                out.writeInt(1);
124                this.provider.writeToParcel(out, flags);
125            } else {
126                out.writeInt(0);
127            }
128            out.writeInt(this.minWidth);
129            out.writeInt(this.minHeight);
130            out.writeInt(this.updatePeriodMillis);
131            out.writeInt(this.initialLayout);
132            if (this.configure != null) {
133                out.writeInt(1);
134                this.configure.writeToParcel(out, flags);
135            } else {
136                out.writeInt(0);
137            }
138            out.writeString(this.label);
139            out.writeInt(this.icon);
140        }
141    
142        public int describeContents() {
143            return 0;
144        }
145    
146        /**
147         * Parcelable.Creator that instantiates AppWidgetProviderInfo objects
148         */
149        public static final Parcelable.Creator<AppWidgetProviderInfo> CREATOR
150                = new Parcelable.Creator<AppWidgetProviderInfo>()
151        {
152            public AppWidgetProviderInfo createFromParcel(Parcel parcel)
153            {
154                return new AppWidgetProviderInfo(parcel);
155            }
156    
157            public AppWidgetProviderInfo[] newArray(int size)
158            {
159                return new AppWidgetProviderInfo[size];
160            }
161        };
162    
163        public String toString() {
164            return "AppWidgetProviderInfo(provider=" + this.provider + ")";
165        }
166    }
