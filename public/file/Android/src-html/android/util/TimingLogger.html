017    package android.util;
018    
019    import java.util.ArrayList;
020    
021    import android.os.SystemClock;
022    
023    /**
024     * A utility class to help log timings splits throughout a method call.
025     * Typical usage is:
026     *
027     * TimingLogger timings = new TimingLogger(TAG, "methodA");
028     * ... do some work A ...
029     * timings.addSplit("work A");
030     * ... do some work B ...
031     * timings.addSplit("work B");
032     * ... do some work C ...
033     * timings.addSplit("work C");
034     * timings.dumpToLog();
035     *
036     * The dumpToLog call would add the following to the log:
037     *
038     * D/TAG     ( 3459): methodA: begin
039     * D/TAG     ( 3459): methodA:      9 ms, work A
040     * D/TAG     ( 3459): methodA:      1 ms, work B
041     * D/TAG     ( 3459): methodA:      6 ms, work C
042     * D/TAG     ( 3459): methodA: end, 16 ms
043     */
044    public class TimingLogger {
045    
046        /**
047         * The Log tag to use for checking Log.isLoggable and for
048         * logging the timings.
049         */
050        private String mTag;
051    
052        /** A label to be included in every log. */
053        private String mLabel;
054    
055        /** Used to track whether Log.isLoggable was enabled at reset time. */
056        private boolean mDisabled;
057    
058        /** Stores the time of each split. */
059        ArrayList<Long> mSplits;
060    
061        /** Stores the labels for each split. */
062        ArrayList<String> mSplitLabels;
063    
064        /**
065         * Create and initialize a TimingLogger object that will log using
066         * the specific tag. If the Log.isLoggable is not enabled to at
067         * least the Log.VERBOSE level for that tag at creation time then
068         * the addSplit and dumpToLog call will do nothing.
069         * @param tag the log tag to use while logging the timings
070         * @param label a string to be displayed with each log
071         */
072        public TimingLogger(String tag, String label) {
073            reset(tag, label);
074        }
075    
076        /**
077         * Clear and initialize a TimingLogger object that will log using
078         * the specific tag. If the Log.isLoggable is not enabled to at
079         * least the Log.VERBOSE level for that tag at creation time then
080         * the addSplit and dumpToLog call will do nothing.
081         * @param tag the log tag to use while logging the timings
082         * @param label a string to be displayed with each log
083         */
084        public void reset(String tag, String label) {
085            mTag = tag;
086            mLabel = label;
087            reset();
088        }
089    
090        /**
091         * Clear and initialize a TimingLogger object that will log using
092         * the tag and label that was specified previously, either via
093         * the constructor or a call to reset(tag, label). If the
094         * Log.isLoggable is not enabled to at least the Log.VERBOSE
095         * level for that tag at creation time then the addSplit and
096         * dumpToLog call will do nothing.
097         */
098        public void reset() {
099            mDisabled = !Log.isLoggable(mTag, Log.VERBOSE);
100            if (mDisabled) return;
101            if (mSplits == null) {
102                mSplits = new ArrayList<Long>();
103                mSplitLabels = new ArrayList<String>();
104            } else {
105                mSplits.clear();
106                mSplitLabels.clear();
107            }
108            addSplit(null);
109        }
110    
111        /**
112         * Add a split for the current time, labeled with splitLabel. If
113         * Log.isLoggable was not enabled to at least the Log.VERBOSE for
114         * the specified tag at construction or reset() time then this
115         * call does nothing.
116         * @param splitLabel a label to associate with this split.
117         */
118        public void addSplit(String splitLabel) {
119            if (mDisabled) return;
120            long now = SystemClock.elapsedRealtime();
121            mSplits.add(now);
122            mSplitLabels.add(splitLabel);
123        }
124    
125        /**
126         * Dumps the timings to the log using Log.d(). If Log.isLoggable was
127         * not enabled to at least the Log.VERBOSE for the specified tag at
128         * construction or reset() time then this call does nothing.
129         */
130        public void dumpToLog() {
131            if (mDisabled) return;
132            Log.d(mTag, mLabel + ": begin");
133            final long first = mSplits.get(0);
134            long now = first;
135            for (int i = 1; i < mSplits.size(); i++) {
136                now = mSplits.get(i);
137                final String splitLabel = mSplitLabels.get(i);
138                final long prev = mSplits.get(i - 1);
139    
140                Log.d(mTag, mLabel + ":      " + (now - prev) + " ms, " + splitLabel);
141            }
142            Log.d(mTag, mLabel + ": end, " + (now - first) + " ms");
143        }
144    }
