017    package android.util;
018    
019    import android.os.*;
020    
021    
022    /**
023     * A structure describing general information about a display, such as its
024     * size, density, and font scaling.
025     */
026    public class DisplayMetrics {
027        /**
028         * The reference density used throughout the system.
029         * 
030         * @hide Pending API council approval
031         */
032        public static final int DEFAULT_DENSITY = 160;
033    
034        private static final int sLcdDensity = SystemProperties.getInt("ro.sf.lcd_density",
035                DEFAULT_DENSITY);
036    
037        /**
038         * The absolute width of the display in pixels.
039         */
040        public int widthPixels;
041        /**
042         * The absolute height of the display in pixels.
043         */
044        public int heightPixels;
045        /**
046         * The logical density of the display.  This is a scaling factor for the
047         * Density Independent Pixel unit, where one DIP is one pixel on an
048         * approximately 160 dpi screen (for example a 240x320, 1.5"x2" screen), 
049         * providing the baseline of the system's display. Thus on a 160dpi screen 
050         * this density value will be 1; on a 106 dpi screen it would be .75; etc.
051         *  
052         * <p>This value does not exactly follow the real screen size (as given by 
053         * {@link #xdpi} and {@link #ydpi}, but rather is used to scale the size of
054         * the overall UI in steps based on gross changes in the display dpi.  For 
055         * example, a 240x320 screen will have a density of 1 even if its width is 
056         * 1.8", 1.3", etc. However, if the screen resolution is increased to 
057         * 320x480 but the screen size remained 1.5"x2" then the density would be 
058         * increased (probably to 1.5).
059         *
060         * @see #DEFAULT_DENSITY
061         */
062        public float density;
063        /**
064         * A scaling factor for fonts displayed on the display.  This is the same
065         * as {@link #density}, except that it may be adjusted in smaller
066         * increments at runtime based on a user preference for the font size.
067         */
068        public float scaledDensity;
069        /**
070         * The exact physical pixels per inch of the screen in the X dimension.
071         */
072        public float xdpi;
073        /**
074         * The exact physical pixels per inch of the screen in the Y dimension.
075         */
076        public float ydpi;
077    
078        public DisplayMetrics() {
079        }
080        
081        public void setTo(DisplayMetrics o) {
082            widthPixels = o.widthPixels;
083            heightPixels = o.heightPixels;
084            density = o.density;
085            scaledDensity = o.scaledDensity;
086            xdpi = o.xdpi;
087            ydpi = o.ydpi;
088        }
089        
090        public void setToDefaults() {
091            widthPixels = 0;
092            heightPixels = 0;
093            density = sLcdDensity / (float) DEFAULT_DENSITY;
094            scaledDensity = density;
095            xdpi = sLcdDensity;
096            ydpi = sLcdDensity;
097        }
098    }
