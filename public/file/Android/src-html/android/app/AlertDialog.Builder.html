035    /**
036     * A subclass of Dialog that can display one, two or three buttons. If you only want to
037     * display a String in this dialog box, use the setMessage() method.  If you
038     * want to display a more complex view, look up the FrameLayout called "body"
039     * and add your view to it:
040     *
041     * <pre>
042     * FrameLayout fl = (FrameLayout) findViewById(R.id.body);
043     * fl.add(myView, new LayoutParams(FILL_PARENT, WRAP_CONTENT));
044     * </pre>
045     * 
046     * <p>The AlertDialog class takes care of automatically setting
047     * {@link WindowManager.LayoutParams#FLAG_ALT_FOCUSABLE_IM
048     * WindowManager.LayoutParams.FLAG_ALT_FOCUSABLE_IM} for you based on whether
049     * any views in the dialog return true from {@link View#onCheckIsTextEditor()
050     * View.onCheckIsTextEditor()}.  Generally you want this set for a Dialog
051     * without text editors, so that it will be placed on top of the current
052     * input method UI.  You can modify this behavior by forcing the flag to your
053     * desired mode after calling {@link #onCreate}.
054     */
055    public class AlertDialog extends Dialog implements DialogInterface {

780             * Creates a {@link AlertDialog} with the arguments supplied to this builder. It does not
781             * {@link Dialog#show()} the dialog. This allows the user to do any extra processing
782             * before displaying the dialog. Use {@link #show()} if you don't have any other processing
783             * to do and want this to be created and displayed.
784             */
785            public AlertDialog create() {
786                final AlertDialog dialog = new AlertDialog(P.mContext);
787                P.apply(dialog.mAlert);
788                dialog.setCancelable(P.mCancelable);
789                dialog.setOnCancelListener(P.mOnCancelListener);
790                if (P.mOnKeyListener != null) {
791                    dialog.setOnKeyListener(P.mOnKeyListener);
792                }
793                return dialog;
794            }

802                dialog.show();
