017    package android.app;
018    
019    import android.content.ComponentCallbacks;
020    import android.content.Context;
021    import android.content.ContextWrapper;
022    import android.content.res.Configuration;
023    
024    /**
025     * Base class for those who need to maintain global application state. You can
026     * provide your own implementation by specifying its name in your
027     * AndroidManifest.xml's &lt;application&gt; tag, which will cause that class
028     * to be instantiated for you when the process for your application/package is
029     * created.
030     */
031    public class Application extends ContextWrapper implements ComponentCallbacks {
032        
033        public Application() {
034            super(null);
035        }
036    
037        /**
038         * Called when the application is starting, before any other application
039         * objects have been created.  Implementations should be as quick as
040         * possible (for example using lazy initialization of state) since the time
041         * spent in this function directly impacts the performance of starting the
042         * first activity, service, or receiver in a process.
043         * If you override this method, be sure to call super.onCreate().
044         */
045        public void onCreate() {
046        }
047    
048        /**
049         * Called when the application is stopping.  There are no more application
050         * objects running and the process will exit.  <em>Note: never depend on
051         * this method being called; in many cases an unneeded application process
052         * will simply be killed by the kernel without executing any application
053         * code.</em>
054         * If you override this method, be sure to call super.onTerminate().
055         */
056        public void onTerminate() {
057        }
058        
059        public void onConfigurationChanged(Configuration newConfig) {
060        }
061        
062        public void onLowMemory() {
063        }
064        
065        // ------------------ Internal API ------------------
066        
067        /**
068         * @hide
069         */
070        /* package */ final void attach(Context context) {
071            attachBaseContext(context);
072        }
073    
074    }
