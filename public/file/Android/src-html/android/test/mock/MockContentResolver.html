017    package android.test.mock;
018    
019    import android.content.ContentProvider;
020    import android.content.ContentResolver;
021    import android.content.Context;
022    import android.content.IContentProvider;
023    import android.database.ContentObserver;
024    import android.net.Uri;
025    
026    import com.google.android.collect.Maps;
027    
028    import java.util.Map;
029    
030    /**
031     * A mock {@link android.content.ContentResolver} class that isolates the test code from the real
032     * content system.  All methods are non-functional and throw
033     * {@link java.lang.UnsupportedOperationException}.  
034     *
035     * <p>This only isolates the test code in ways that have proven useful so far. More should be
036     * added as they become a problem.
037     */
038    public class MockContentResolver extends ContentResolver {
039        Map<String, ContentProvider> mProviders;
040    
041        public MockContentResolver() {
042            super(null);
043            mProviders = Maps.newHashMap();
044        }
045    
046        public void addProvider(String name, ContentProvider provider) {
047            mProviders.put(name, provider);
048        }
049    
050        /** @hide */
051        @Override
052        protected IContentProvider acquireProvider(Context context, String name) {
053            final ContentProvider provider = mProviders.get(name);
054            if (provider != null) {
055                return provider.getIContentProvider();
056            } else {
057                return null;
058            }
059        }
060    
061        /** @hide */
062        @Override
063        public boolean releaseProvider(IContentProvider provider) {
064            return true;
065        }
066    
067        @Override
068        public void notifyChange(Uri uri, ContentObserver observer, boolean syncToNetwork) {
069        }
070    }
