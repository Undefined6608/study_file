004    
005    import android.content.Context;
006    import android.content.ContextWrapper;

008    import android.database.sqlite.SQLiteDatabase;

016    /**
017     * This is a class which delegates to the given context, but performs database
018     * and file operations with a renamed database/file name (prefixes default
019     * names with a given prefix).
020     */
021    public class RenamingDelegatingContext extends ContextWrapper {
022    
023        private Context mFileContext;
024        private String mFilePrefix = null;
025    
026        private Set<String> mDatabaseNames = Sets.newHashSet();
027        private Set<String> mFileNames = Sets.newHashSet();
028    
029        public static <T extends ContentProvider> T providerWithRenamedContext(
030                Class<T> contentProvider, Context c, String filePrefix)
031                throws IllegalAccessException, InstantiationException {
032            return providerWithRenamedContext(contentProvider, c, filePrefix, false);
033        }
034    
035        public static <T extends ContentProvider> T providerWithRenamedContext(
036                Class<T> contentProvider, Context c, String filePrefix,
037                boolean allowAccessToExistingFilesAndDbs)
038                throws IllegalAccessException, InstantiationException {
039            Class<T> mProviderClass = contentProvider;
040            T mProvider = mProviderClass.newInstance();
041            RenamingDelegatingContext mContext = new RenamingDelegatingContext(c, filePrefix);
042            if (allowAccessToExistingFilesAndDbs) {
043                mContext.makeExistingFilesAndDbsAccessible();
044            }
045            mProvider.attachInfo(mContext, null);
046            return mProvider;
047        }
048    
049        /**
050         * Makes accessible all files and databases whose names match the filePrefix that was passed to
051         * the constructor. Normally only files and databases that were created through this context are
052         * accessible.
053         */
054        public void makeExistingFilesAndDbsAccessible() {
055            String[] databaseList = mFileContext.databaseList();
056            for (String diskName : databaseList) {
057                if (shouldDiskNameBeVisible(diskName)) {
058                    mDatabaseNames.add(publicNameFromDiskName(diskName));
059                }
060            }
061            String[] fileList = mFileContext.fileList();
062            for (String diskName : fileList) {
063                if (shouldDiskNameBeVisible(diskName)) {
064                    mFileNames.add(publicNameFromDiskName(diskName));
065                }
066            }
067        }
068    
069        /**
070         * Returns if the given diskName starts with the given prefix or not.
071         * @param diskName name of the database/file.
072         */
073        boolean shouldDiskNameBeVisible(String diskName) {
074            return diskName.startsWith(mFilePrefix);
075        }
076    
077        /**
078         * Returns the public name (everything following the prefix) of the given diskName.
079         * @param diskName name of the database/file.
080         */
081        String publicNameFromDiskName(String diskName) {
082            if (!shouldDiskNameBeVisible(diskName)) {
083                throw new IllegalArgumentException("disk file should not be visible: " + diskName);
084            }
085            return diskName.substring(mFilePrefix.length(), diskName.length());
086        }
087    
088        /**
089         * @param context : the context that will be delagated.
090         * @param filePrefix : a prefix with which database and file names will be
091         * prefixed.
092         */
093        public RenamingDelegatingContext(Context context, String filePrefix) {
094            super(context);
095            mFileContext = context;
096            mFilePrefix = filePrefix;
097        }
098    
099        /**
100         * @param context : the context that will be delagated.
101         * @param fileContext : the context that file and db methods will be delgated to
102         * @param filePrefix : a prefix with which database and file names will be
103         * prefixed.
104         */
105        public RenamingDelegatingContext(Context context, Context fileContext, String filePrefix) {
106            super(context);
107            mFileContext = fileContext;
108            mFilePrefix = filePrefix;
109        }
110    
111        public String getDatabasePrefix() {
112            return mFilePrefix;
113        }
114    
115        private String renamedFileName(String name) {
116            return mFilePrefix + name;
117        }
118    
119        @Override
120        public SQLiteDatabase openOrCreateDatabase(String name,
121                int mode, SQLiteDatabase.CursorFactory factory) {
122            final String internalName = renamedFileName(name);
123            if (!mDatabaseNames.contains(name)) {
124                mDatabaseNames.add(name);
125                mFileContext.deleteDatabase(internalName);
126            }
127            return mFileContext.openOrCreateDatabase(internalName, mode, factory);
128        }
129    
130        @Override
131        public boolean deleteDatabase(String name) {
132            if (mDatabaseNames.contains(name)) {
133                mDatabaseNames.remove(name);
134                return mFileContext.deleteDatabase(renamedFileName(name));
135            } else {
136                return false;
137            }
138        }
139    
140        @Override
141        public String[] databaseList() {
142            return mDatabaseNames.toArray(new String[]{});
143        }
144    
145        @Override
146        public FileInputStream openFileInput(String name)
147                throws FileNotFoundException {
148            final String internalName = renamedFileName(name);
149            if (mFileNames.contains(name)) {
150                return mFileContext.openFileInput(internalName);
151            } else {
152                throw new FileNotFoundException(internalName);
153            }
154        }
155    
156        @Override
157        public FileOutputStream openFileOutput(String name, int mode)
158                throws FileNotFoundException {
159            mFileNames.add(name);
160            return mFileContext.openFileOutput(renamedFileName(name), mode);
161        }
162    
163        @Override
164        public File getFileStreamPath(String name) {
165            return mFileContext.getFileStreamPath(renamedFileName(name));
166        }
167    
168        @Override
169        public boolean deleteFile(String name) {
170            if (mFileNames.contains(name)) {
171                mFileNames.remove(name);
172                return mFileContext.deleteFile(renamedFileName(name));
173            } else {
174                return false;
175            }
176        }
177    
178        @Override
179        public String[] fileList() {
180            return mFileNames.toArray(new String[]{});
181        }
