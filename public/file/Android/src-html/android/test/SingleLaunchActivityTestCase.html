017    package android.test;
018    
019    import android.app.Activity;
020    import android.view.IWindowManager;
021    import android.os.ServiceManager;
022    
023    /**
024     * If you would like to test a single activity with an
025     * {@link android.test.InstrumentationTestCase}, this provides some of the boiler plate to
026     * launch and finish the activity in {@link #setUp} and {@link #tearDown}.
027     * 
028     * This launches the activity only once for the entire class instead of doing it 
029     * in every setup / teardown call.
030     */
031    public abstract class SingleLaunchActivityTestCase<T extends Activity>
032            extends InstrumentationTestCase {
033        
034        String mPackage;
035        Class<T> mActivityClass;
036        private static int sTestCaseCounter = 0;
037        private static boolean sActivityLaunchedFlag = false;
038    
039        /**
040         * <b>NOTE:</b> The parameter <i>pkg</i> must refer to the package identifier of the
041         * package hosting the activity to be launched, which is specified in the AndroidManifest.xml
042         * file.  This is not necessarily the same as the java package name.
043         *
044         * @param pkg The package hosting the activity to be launched.
045         * @param activityClass The activity to test.
046         */
047        public SingleLaunchActivityTestCase(String pkg, Class<T> activityClass) {
048            mPackage = pkg;
049            mActivityClass = activityClass;        
050            sTestCaseCounter ++;                
051        }
052        
053        /**
054         * The activity that will be set up for use in each test method.
055         */
056        private static Activity sActivity;
057    
058        public T getActivity() {
059            return (T) sActivity;
060        }
061    
062        @Override
063        protected void setUp() throws Exception {
064            super.setUp();
065            // If it is the first test case, launch the activity.
066            if (!sActivityLaunchedFlag) {
067                // by default, not in touch mode
068                getInstrumentation().setInTouchMode(false);
069                sActivity = launchActivity(mPackage, mActivityClass, null);
070                sActivityLaunchedFlag = true;
071            }                        
072        }
073    
074        @Override
075        protected void tearDown() throws Exception {
076            // If it is the last test case, call finish on the activity.
077            sTestCaseCounter --;
078            if (sTestCaseCounter == 1) {
079                sActivity.finish();
080            }        
081            super.tearDown();
082        }
083    
084        public void testActivityTestCaseSetUpProperly() throws Exception {
085            assertNotNull("activity should be launched successfully", sActivity);
086        }
087    }
