017    package android.hardware;
018    
019    /**
020     * Used for receiving notifications from the SensorManager when
021     * sensor values have changed.
022     * 
023     * This interface is deprecated, use 
024     * {@link android.hardware.SensorEventListener SensorEventListener} instead.
025     * 
026     */
027    @Deprecated
028    public interface SensorListener {
029    
030        /**
031         * <p>Called when sensor values have changed.
032         * The length and contents of the values array vary
033         * depending on which sensor is being monitored.
034         * See {@link android.hardware.SensorManager SensorManager}
035         * for details on possible sensor types.
036         *
037         * <p><u>Definition of the coordinate system used below.</u><p>
038         * <p>The X axis refers to the screen's horizontal axis
039         * (the small edge in portrait mode, the long edge in landscape mode) and
040         * points to the right. 
041         * <p>The Y axis refers to the screen's vertical axis and points towards
042         * the top of the screen (the origin is in the lower-left corner).
043         * <p>The Z axis points toward the sky when the device is lying on its back
044         * on a table.
045         * <p> <b>IMPORTANT NOTE:</b> The axis <b><u>are swapped</u></b> when the
046         * device's screen orientation changes. To access the unswapped values,
047         * use indices 3, 4 and 5 in values[].
048         * 
049         * <p>{@link android.hardware.SensorManager#SENSOR_ORIENTATION SENSOR_ORIENTATION},
050         * {@link android.hardware.SensorManager#SENSOR_ORIENTATION_RAW SENSOR_ORIENTATION_RAW}:<p>
051         *  All values are angles in degrees.
052         * 
053         * <p>values[0]: Azimuth, rotation around the Z axis (0<=azimuth<360).
054         * 0 = North, 90 = East, 180 = South, 270 = West
055         * 
056         * <p>values[1]: Pitch, rotation around X axis (-180<=pitch<=180), with positive
057         * values when the z-axis moves toward the y-axis.
058         *
059         * <p>values[2]: Roll, rotation around Y axis (-90<=roll<=90), with positive values 
060         * when the z-axis moves toward the x-axis.
061         *
062         * <p>Note that this definition of yaw, pitch and roll is different from the
063         * traditional definition used in aviation where the X axis is along the long
064         * side of the plane (tail to nose).
065         *
066         * <p>{@link android.hardware.SensorManager#SENSOR_ACCELEROMETER SENSOR_ACCELEROMETER}:<p>
067         *  All values are in SI units (m/s^2) and measure contact forces.
068         *  
069         *  <p>values[0]: force applied by the device on the x-axis 
070         *  <p>values[1]: force applied by the device on the y-axis 
071         *  <p>values[2]: force applied by the device on the z-axis
072         *  
073         *  <p><u>Examples</u>:
074         *    <li>When the device is pushed on its left side toward the right, the
075         *    x acceleration value is negative (the device applies a reaction force
076         *    to the push toward the left)</li>
077         *    
078         *    <li>When the device lies flat on a table, the acceleration value is 
079         *    {@link android.hardware.SensorManager#STANDARD_GRAVITY -STANDARD_GRAVITY},
080         *    which correspond to the force the device applies on the table in reaction
081         *    to gravity.</li>
082         *
083         * <p>{@link android.hardware.SensorManager#SENSOR_MAGNETIC_FIELD SENSOR_MAGNETIC_FIELD}:<p>
084         *  All values are in micro-Tesla (uT) and measure the ambient magnetic
085         *  field in the X, Y and -Z axis.
086         *  <p><b><u>Note:</u></b> the magnetic field's Z axis is inverted.
087         *  
088         * @param sensor The ID of the sensor being monitored
089         * @param values The new values for the sensor.
090         */
091        public void onSensorChanged(int sensor, float[] values);
092    
093        /**
094         * Called when the accuracy of a sensor has changed.
095         * See {@link android.hardware.SensorManager SensorManager}
096         * for details.
097         *
098         * @param sensor The ID of the sensor being monitored
099         * @param accuracy The new accuracy of this sensor.
100         */
101        public void onAccuracyChanged(int sensor, int accuracy);    
102    }
