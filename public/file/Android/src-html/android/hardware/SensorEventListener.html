017    package android.hardware;
018    
019    /**
020     * Used for receiving notifications from the SensorManager when
021     * sensor values have changed.
022     */
023    public interface SensorEventListener {
024    
025        /**
026         * Called when sensor values have changed.
027         * <p>See {@link android.hardware.SensorManager SensorManager}
028         * for details on possible sensor types.
029         * <p>See also {@link android.hardware.SensorEvent SensorEvent}.
030         * 
031         * <p><b>NOTE:</b> The application doesn't own the
032         * {@link android.hardware.SensorEvent event}
033         * object passed as a parameter and therefore cannot hold on o it.
034         * The object may be part of an internal pool and may be reused by
035         * the framework.
036         *
037         * @param event the {@link android.hardware.SensorEvent SensorEvent}. 
038         */
039        public void onSensorChanged(SensorEvent event);
040    
041        /**
042         * Called when the accuracy of a sensor has changed.
043         * <p>See {@link android.hardware.SensorManager SensorManager}
044         * for details.
045         *
046         * @param accuracy The new accuracy of this sensor
047         */
048        public void onAccuracyChanged(Sensor sensor, int accuracy);    
049    }
