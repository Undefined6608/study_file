017    package android.hardware;
018    
019    /**
020     * This class represents a sensor event and holds informations such as the
021     * sensor type (eg: accelerometer, orientation, etc...), the time-stamp, 
022     * accuracy and of course the sensor's {@link SensorEvent#values data}.
023     *
024     * <p><u>Definition of the coordinate system used by the SensorEvent API.</u><p>
025     * 
026     * <pre>
027     * The coordinate space is defined relative to the screen of the phone 
028     * in its default orientation. The axes are not swapped when the device's
029     * screen orientation changes.
030     * 
031     * The OpenGL ES coordinate system is used. The origin is in the
032     * lower-left corner  with respect to the screen, with the X axis horizontal
033     * and pointing  right, the Y axis vertical and pointing up and the Z axis
034     * pointing outside the front face of the screen. In this system, coordinates
035     * behind the screen have negative Z values.
036     * 
037     * <b>Note:</b> This coordinate system is different from the one used in the
038     * Android 2D APIs where the origin is in the top-left corner. 
039     *
040     *   x<0         x>0
041     *                ^
042     *                |
043     *    +-----------+-->  y>0
044     *    |           |
045     *    |           |
046     *    |           |
047     *    |           |   / z<0
048     *    |           |  /
049     *    |           | /
050     *    O-----------+/
051     *    |[]  [ ]  []/
052     *    +----------/+     y<0
053     *              /
054     *             /
055     *           |/ z>0 (toward the sky)
056     *
057     *    O: Origin (x=0,y=0,z=0)
058     * </pre>
059     */
060    
061    public class SensorEvent {
062        /**
063         * The length and contents of the values array vary depending on which
064         * sensor type is being monitored (see also {@link SensorEvent} for a 
065         * definition of the coordinate system used):
066         * 
067         * <p>{@link android.hardware.Sensor#TYPE_ORIENTATION Sensor.TYPE_ORIENTATION}:<p>
068         *  All values are angles in degrees.
069         * 
070         * <p>values[0]: Azimuth, angle between the magnetic north direction and
071         * the Y axis, around the Z axis (0 to 359). 
072         * 0=North, 90=East, 180=South, 270=West
073         * 
074         * <p>values[1]: Pitch, rotation around X axis (-180 to 180), 
075         * with positive values when the z-axis moves <b>toward</b> the y-axis.
076         *
077         * <p>values[2]: Roll, rotation around Y axis (-90 to 90), with 
078         * positive values  when the x-axis moves <b>away</b> from the z-axis.
079         * 
080         * <p><b>Note:</b> This definition is different from <b>yaw, pitch and 
081         * roll</b> used in aviation where the X axis is along the long side of
082         * the plane (tail to nose).
083         * 
084         * <p><b>Note:</b> It is preferable to use 
085         * {@link android.hardware.SensorManager#getRotationMatrix 
086         *      getRotationMatrix()} in conjunction with
087         * {@link android.hardware.SensorManager#remapCoordinateSystem 
088         *      remapCoordinateSystem()} and
089         * {@link android.hardware.SensorManager#getOrientation getOrientation()}
090         * to compute these values; while it may be more expensive, it is usually 
091         * more accurate.
092         *
093         * <p>{@link android.hardware.Sensor#TYPE_ACCELEROMETER Sensor.TYPE_ACCELEROMETER}:<p>
094         *  All values are in SI units (m/s^2) and measure the acceleration applied
095         *  to the phone minus the force of gravity.
096         *  
097         *  <p>values[0]: Acceleration minus Gx on the x-axis 
098         *  <p>values[1]: Acceleration minus Gy on the y-axis 
099         *  <p>values[2]: Acceleration minus Gz on the z-axis
100         *  
101         *  <p><u>Examples</u>:
102         *    <li>When the device lies flat on a table and is pushed on its left 
103         *    side toward the right, the x acceleration value is positive.</li>
104         *    
105         *    <li>When the device lies flat on a table, the acceleration value is 
106         *    +9.81, which correspond to the acceleration of the device (0 m/s^2) 
107         *    minus the force of gravity (-9.81 m/s^2).</li>
108         *    
109         *    <li>When the device lies flat on a table and is pushed toward the sky
110         *    with an acceleration of A m/s^2, the acceleration value is equal to 
111         *    A+9.81 which correspond to the acceleration of the 
112         *    device (+A m/s^2) minus the force of gravity (-9.81 m/s^2).</li>
113         * 
114         * 
115         * <p>{@link android.hardware.Sensor#TYPE_MAGNETIC_FIELD Sensor.TYPE_MAGNETIC_FIELD}:<p>
116         *  All values are in micro-Tesla (uT) and measure the ambient magnetic
117         *  field in the X, Y and Z axis.
118         *  
119         */    
120        public final float[] values;
121    
122        /**
123         * The sensor that generated this event.
124         * See {@link android.hardware.SensorManager SensorManager}
125         * for details.
126         */
127        public Sensor sensor;
128        
129        /**
130         * The accuracy of this event.
131         * See {@link android.hardware.SensorManager SensorManager}
132         * for details.
133         */
134        public int accuracy;
135        
136        
137        /**
138         * The time in nanosecond at which the event happened
139         */
140        public long timestamp;
141    
142        
143        SensorEvent(int size) {
144            values = new float[size];
145        }
146    }
