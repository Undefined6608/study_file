017    package android.graphics.drawable;
018    
019    import java.io.IOException;
020    
021    import org.xmlpull.v1.XmlPullParser;
022    import org.xmlpull.v1.XmlPullParserException;
023    
024    import android.content.res.Resources;
025    import android.content.res.TypedArray;
026    import android.util.AttributeSet;
027    
028    /**
029     * A resource that manages a number of alternate Drawables, each assigned a maximum numerical value.
030     * Setting the level value of the object with {@link #setLevel(int)} will load the image with the next
031     * greater or equal value assigned to its max attribute.
032     * A good example use of
033     * a LevelListDrawable would be a battery level indicator icon, with different images to indicate the current
034     * battery level.
035     * <p>
036     * It can be defined in an XML file with the <code>&lt;level-list></code> element.
037     * Each Drawable level is defined in a nested <code>&lt;item></code>. For example:
038     * </p>
039     * <pre>
040     * &lt;level-list xmlns:android="http://schemas.android.com/apk/res/android">
041     *  &lt;item android:maxLevel="0" android:drawable="@drawable/ic_wifi_signal_1" />
042     *  &lt;item android:maxLevel="1" android:drawable="@drawable/ic_wifi_signal_2" />
043     *  &lt;item android:maxLevel="2" android:drawable="@drawable/ic_wifi_signal_3" />
044     *  &lt;item android:maxLevel="3" android:drawable="@drawable/ic_wifi_signal_4" />
045     * &lt;/level-list>
046     *</pre>
047     * <p>With this XML saved into the res/drawable/ folder of the project, it can be referenced as
048     * the drawable for an {@link android.widget.ImageView}. The default image is the first in the list.
049     * It can then be changed to one of the other levels with
050     * {@link android.widget.ImageView#setImageLevel(int)}.</p>
051     * @attr ref android.R.styleable#LevelListDrawableItem_minLevel
052     * @attr ref android.R.styleable#LevelListDrawableItem_maxLevel
053     * @attr ref android.R.styleable#LevelListDrawableItem_drawable
054     */
055    public class LevelListDrawable extends DrawableContainer {
056        private final LevelListState mLevelListState;
057        private boolean mMutated;
058    
059        public LevelListDrawable() {
060            this(null);
061        }
062    
063        public void addLevel(int low, int high, Drawable drawable) {
064            if (drawable != null) {
065                mLevelListState.addLevel(low, high, drawable);
066                // in case the new state matches our current state...
067                onLevelChange(getLevel());
068            }
069        }
070    
071        // overrides from Drawable
072    
073        @Override
074        protected boolean onLevelChange(int level) {
075            int idx = mLevelListState.indexOfLevel(level);
076            if (selectDrawable(idx)) {
077                return true;
078            }
079            return super.onLevelChange(level);
080        }
081    
082        @Override
083        public void inflate(Resources r, XmlPullParser parser, AttributeSet attrs)
084                throws XmlPullParserException, IOException {
085    
086            super.inflate(r, parser, attrs);
087    
088            int type;
089    
090            int low = 0;
091    
092            final int innerDepth = parser.getDepth() + 1;
093            int depth;
094            while ((type = parser.next()) != XmlPullParser.END_DOCUMENT
095                    && ((depth = parser.getDepth()) >= innerDepth
096                    || type != XmlPullParser.END_TAG)) {
097                if (type != XmlPullParser.START_TAG) {
098                    continue;
099                }
100    
101                if (depth > innerDepth || !parser.getName().equals("item")) {
102                    continue;
103                }
104    
105                TypedArray a = r.obtainAttributes(attrs,
106                        com.android.internal.R.styleable.LevelListDrawableItem);
107    
108                low = a.getInt(
109                        com.android.internal.R.styleable.LevelListDrawableItem_minLevel, 0);
110                int high = a.getInt(
111                        com.android.internal.R.styleable.LevelListDrawableItem_maxLevel, 0);
112                int drawableRes = a.getResourceId(
113                        com.android.internal.R.styleable.LevelListDrawableItem_drawable, 0);
114    
115                a.recycle();
116    
117                if (high < 0) {
118                    throw new XmlPullParserException(parser.getPositionDescription()
119                            + ": <item> tag requires a 'maxLevel' attribute");
120                }
121    
122                Drawable dr;
123                if (drawableRes != 0) {
124                    dr = r.getDrawable(drawableRes);
125                } else {
126                    while ((type = parser.next()) == XmlPullParser.TEXT) {
127                    }
128                    if (type != XmlPullParser.START_TAG) {
129                        throw new XmlPullParserException(
130                                parser.getPositionDescription()
131                                        + ": <item> tag requires a 'drawable' attribute or "
132                                        + "child tag defining a drawable");
133                    }
134                    dr = Drawable.createFromXmlInner(r, parser, attrs);
135                }
136    
137                mLevelListState.addLevel(low, high, dr);
138            }
139    
140            onLevelChange(getLevel());
141        }
142    
143        @Override
144        public Drawable mutate() {
145            if (!mMutated && super.mutate() == this) {
146                mLevelListState.mLows = mLevelListState.mLows.clone();
147                mLevelListState.mHighs = mLevelListState.mHighs.clone();
148                mMutated = true;
149            }
150            return this;
151        }
152    
153        private final static class LevelListState extends DrawableContainerState {
154            private int[] mLows;
155            private int[] mHighs;
156    
157            LevelListState(LevelListState orig, LevelListDrawable owner) {
158                super(orig, owner);
159    
160                if (orig != null) {
161                    mLows = orig.mLows;
162                    mHighs = orig.mHighs;
163                } else {
164                    mLows = new int[getChildren().length];
165                    mHighs = new int[getChildren().length];
166                }
167            }
168    
169            public void addLevel(int low, int high, Drawable drawable) {
170                int pos = addChild(drawable);
171                mLows[pos] = low;
172                mHighs[pos] = high;
173            }
174    
175            public int indexOfLevel(int level) {
176                final int[] lows = mLows;
177                final int[] highs = mHighs;
178                final int N = getChildCount();
179                for (int i = 0; i < N; i++) {
180                    if (level >= lows[i] && level <= highs[i]) {
181                        return i;
182                    }
183                }
184                return -1;
185            }
186    
187            @Override
188            public Drawable newDrawable() {
189                return new LevelListDrawable(this);
190            }
191    
192            @Override
193            public void growArray(int oldSize, int newSize) {
194                super.growArray(oldSize, newSize);
195                int[] newInts = new int[newSize];
196                System.arraycopy(mLows, 0, newInts, 0, oldSize);
197                mLows = newInts;
198                newInts = new int[newSize];
199                System.arraycopy(mHighs, 0, newInts, 0, oldSize);
200                mHighs = newInts;
201            }
202        }
203    
204        private LevelListDrawable(LevelListState state) {
205            LevelListState as = new LevelListState(state, this);
206            mLevelListState = as;
207            setConstantState(as);
208            onLevelChange(getLevel());
209        }
210    }
