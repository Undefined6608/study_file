039    /**
040     * A Drawable that wraps a bitmap and can be tiled, stretched, or aligned. You can create a
041     * BitmapDrawable from a file path, an input stream, through XML inflation, or from
042     * a {@link android.graphics.Bitmap} object.
043     * <p>It can be defined in an XML file with the <code>&lt;bitmap></code> element.</p>
044     * <p>
045     * Also see the {@link android.graphics.Bitmap} class, which handles the management and 
046     * transformation of raw bitmap graphics, and should be used when drawing to a 
047     * {@link android.graphics.Canvas}.
048     * </p>
049     *
050     * @attr ref android.R.styleable#BitmapDrawable_src
051     * @attr ref android.R.styleable#BitmapDrawable_antialias
052     * @attr ref android.R.styleable#BitmapDrawable_filter
053     * @attr ref android.R.styleable#BitmapDrawable_dither
054     * @attr ref android.R.styleable#BitmapDrawable_gravity
055     * @attr ref android.R.styleable#BitmapDrawable_tileMode
056     */
057    public class BitmapDrawable extends Drawable {

059        private static final int DEFAULT_PAINT_FLAGS = Paint.FILTER_BITMAP_FLAG;
060        private BitmapState mBitmapState;
061        private Bitmap mBitmap;
062        private final Rect mDstRect = new Rect();   // Gravity.apply() sets this

064        private boolean mApplyGravity;

066        private int mBitmapWidth;
067        private int mBitmapHeight;
068        private boolean mMutated;

070        public BitmapDrawable() {

076        }

078        public BitmapDrawable(String filepath) {
079            this(new BitmapState(BitmapFactory.decodeFile(filepath)));
080            if (mBitmap == null) {
081                android.util.Log.w("BitmapDrawable", "BitmapDrawable cannot decode " + filepath);
082            }
083        }

086            this(new BitmapState(BitmapFactory.decodeStream(is)));

089            }
090        }

100        private void setBitmap(Bitmap bitmap) {
101            mBitmap = bitmap;
102            if (bitmap != null) {
103                mBitmapWidth = bitmap.getWidth();
104                mBitmapHeight = bitmap.getHeight();
105            } else {
106                mBitmapWidth = mBitmapHeight = -1;

202        public void setTileModeXY(Shader.TileMode xmode, Shader.TileMode ymode) {
203            final BitmapState state = mBitmapState;
204            if (state.mTileModeX != xmode || state.mTileModeY != ymode) {
205                state.mTileModeX = xmode;
206                state.mTileModeY = ymode;
207                mRebuildShader = true;
208            }
209        }

218            super.onBoundsChange(bounds);

222        @Override
223        public void draw(Canvas canvas) {
224            Bitmap bitmap = mBitmap;
225            if (bitmap != null) {
226                final BitmapState state = mBitmapState;
227                if (mRebuildShader) {
228                    Shader.TileMode tmx = state.mTileModeX;
229                    Shader.TileMode tmy = state.mTileModeY;
230    
231                    if (tmx == null && tmy == null) {
232                        state.mPaint.setShader(null);
233                    } else {
234                        Shader s = new BitmapShader(bitmap,
235                                tmx == null ? Shader.TileMode.CLAMP : tmx,
236                                tmy == null ? Shader.TileMode.CLAMP : tmy);
237                        state.mPaint.setShader(s);
238                    }
239                    mRebuildShader = false;
240                    copyBounds(mDstRect);
241                }
242    
243                Shader shader = state.mPaint.getShader();
244                if (shader == null) {
245                    if (mApplyGravity) {
246                        Gravity.apply(state.mGravity, mBitmapWidth, mBitmapHeight,
247                                getBounds(), mDstRect);
248                        mApplyGravity = false;
249                    }
250                    canvas.drawBitmap(bitmap, null, mDstRect, state.mPaint);
251                } else {
252                    if (mApplyGravity) {
253                        mDstRect.set(getBounds());
254                        mApplyGravity = false;
255                    }
256                    canvas.drawRect(mDstRect, state.mPaint);
257                }
258            }
259        }

271        /**
272         * A mutable BitmapDrawable still shares its Bitmap with any other Drawable
273         * that comes from the same resource.
274         *
275         * @return This drawable.
276         */
277        @Override
278        public Drawable mutate() {
279            if (!mMutated && super.mutate() == this) {
280                mBitmapState = new BitmapState(mBitmapState);
281                mMutated = true;
282            }
283            return this;
284        }

286        @Override
287        public void inflate(Resources r, XmlPullParser parser, AttributeSet attrs)
288                throws XmlPullParserException, IOException {
289            super.inflate(r, parser, attrs);
290    
291            TypedArray a = r.obtainAttributes(attrs, com.android.internal.R.styleable.BitmapDrawable);
292    
293            final int id = a.getResourceId(com.android.internal.R.styleable.BitmapDrawable_src, 0);
294            if (id == 0) {
295                throw new XmlPullParserException(parser.getPositionDescription() +
296                        ": <bitmap> requires a valid src attribute");
297            }
298            final Bitmap bitmap = BitmapFactory.decodeResource(r, id);
299            if (bitmap == null) {
300                throw new XmlPullParserException(parser.getPositionDescription() +
301                        ": <bitmap> requires a valid src attribute");
302            }
303            mBitmapState.mBitmap = bitmap;
304            setBitmap(bitmap);
305            setDensityScale(r.getDisplayMetrics());
306    
307            final Paint paint = mBitmapState.mPaint;
308            paint.setAntiAlias(a.getBoolean(com.android.internal.R.styleable.BitmapDrawable_antialias,
309                    paint.isAntiAlias()));
310            paint.setFilterBitmap(a.getBoolean(com.android.internal.R.styleable.BitmapDrawable_filter,
311                    paint.isFilterBitmap()));
312            paint.setDither(a.getBoolean(com.android.internal.R.styleable.BitmapDrawable_dither,
313                    paint.isDither()));
314            setGravity(a.getInt(com.android.internal.R.styleable.BitmapDrawable_gravity, Gravity.FILL));
315            int tileMode = a.getInt(com.android.internal.R.styleable.BitmapDrawable_tileMode, -1);
316            if (tileMode != -1) {
317                switch (tileMode) {
318                    case 0:
319                        setTileModeXY(Shader.TileMode.CLAMP, Shader.TileMode.CLAMP);
320                        break;
321                    case 1:
322                        setTileModeXY(Shader.TileMode.REPEAT, Shader.TileMode.REPEAT);
323                        break;
324                    case 2:
325                        setTileModeXY(Shader.TileMode.MIRROR, Shader.TileMode.MIRROR);
326                        break;
327                }
328            }
329    
330            a.recycle();
331        }

333        @Override
334        public int getIntrinsicWidth() {
335            final Bitmap bitmap = mBitmap;
336            final BitmapState state = mBitmapState;
337    
338            if (!state.mAutoScale || state.mBitmapScale == Bitmap.DENSITY_SCALE_UNKNOWN) {
339                return mBitmapWidth;
340            } else {
341                return bitmap != null ? (int) (mBitmapWidth /
342                        (state.mBitmapScale / state.mTargetDensityScale) + 0.5f) : -1;
343    
344            }
345        }
346    
347        @Override
348        public int getIntrinsicHeight() {
349            final Bitmap bitmap = mBitmap;
350            final BitmapState state = mBitmapState;
351    
352            if (!state.mAutoScale || state.mBitmapScale == Bitmap.DENSITY_SCALE_UNKNOWN) {
353                return mBitmapHeight;
354            } else {
355                return bitmap != null ? (int) (mBitmapHeight /
356                        (state.mBitmapScale / state.mTargetDensityScale) + 0.5f) : -1;
357            }
358        }
359    
360        @Override
361        public int getOpacity() {
362            if (mBitmapState.mGravity != Gravity.FILL) {
363                return PixelFormat.TRANSLUCENT;
364            }
365            Bitmap bm = mBitmap;
366            return (bm == null || bm.hasAlpha() || mBitmapState.mPaint.getAlpha() < 255) ?
367                    PixelFormat.TRANSLUCENT : PixelFormat.OPAQUE;
368        }

370        @Override
371        public final ConstantState getConstantState() {
372            mBitmapState.mChangingConfigurations = super.getChangingConfigurations();
373            return mBitmapState;
374        }
375    
376        final static class BitmapState extends ConstantState {
377            Bitmap mBitmap;
378            int mChangingConfigurations;
379            int mGravity = Gravity.FILL;
380            Paint mPaint = new Paint(DEFAULT_PAINT_FLAGS);
381            Shader.TileMode mTileModeX;
382            Shader.TileMode mTileModeY;
383            boolean mAutoScale;
384            float mBitmapScale;
385            float mTargetDensityScale = 1.0f;
386    
387            BitmapState(Bitmap bitmap) {
388                mBitmap = bitmap;
389                if (bitmap != null) {
390                    mBitmapScale = bitmap.getDensityScale();
391                    mAutoScale = bitmap.isAutoScalingEnabled();
392                } else {
393                    mBitmapScale = 1.0f;
394                    mAutoScale = false;
395                }
396            }
397    
398            BitmapState(BitmapState bitmapState) {
399                this(bitmapState.mBitmap);
400                mChangingConfigurations = bitmapState.mChangingConfigurations;
401                mGravity = bitmapState.mGravity;
402                mTileModeX = bitmapState.mTileModeX;
403                mTileModeY = bitmapState.mTileModeY;
404                mTargetDensityScale = bitmapState.mTargetDensityScale;
405                mPaint = new Paint(bitmapState.mPaint);
406            }
407    
408            @Override
409            public Drawable newDrawable() {
410                return new BitmapDrawable(this);
411            }
412            
413            @Override
414            public int getChangingConfigurations() {
415                return mChangingConfigurations;
416            }
417        }
418    
419        private BitmapDrawable(BitmapState state) {
420            mBitmapState = state;
421            setBitmap(state.mBitmap);
422        }
