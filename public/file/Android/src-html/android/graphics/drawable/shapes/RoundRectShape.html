017    package android.graphics.drawable.shapes;
018    
019    import android.graphics.Canvas;
020    import android.graphics.Paint;
021    import android.graphics.Path;
022    import android.graphics.RectF;
023    
024    /**
025     * Creates a rounded-corner rectangle. Optionally, an inset (rounded) rectangle
026     * can be included (to make a sort of "O" shape). 
027     * The rounded rectangle can be drawn to a Canvas with its own draw() method,
028     * but more graphical control is available if you instead pass
029     * the RoundRectShape to a {@link android.graphics.drawable.ShapeDrawable}.
030     */
031    public class RoundRectShape extends RectShape {
032        private float[] mOuterRadii;
033        private RectF   mInset;
034        private float[] mInnerRadii;
035        
036        private RectF mInnerRect;
037        private Path  mPath;    // this is what we actually draw
038        
039        /**
040         * RoundRectShape constructor.
041         * Specifies an outer (round)rect and an optional inner (round)rect.
042         *
043         * @param outerRadii An array of 8 radius values, for the outer roundrect. 
044         *                   The first two floats are for the 
045         *                   top-left corner (remaining pairs correspond clockwise). 
046         *                   For no rounded corners on the outer rectangle, 
047         *                   pass null.
048         * @param inset      A RectF that specifies the distance from the inner 
049         *                   rect to each side of the outer rect. 
050         *                   For no inner, pass null.
051         * @param innerRadii An array of 8 radius values, for the inner roundrect.
052         *                   The first two floats are for the 
053         *                   top-left corner (remaining pairs correspond clockwise). 
054         *                   For no rounded corners on the inner rectangle, 
055         *                   pass null.
056         *                   If inset parameter is null, this parameter is ignored. 
057         */
058        public RoundRectShape(float[] outerRadii, RectF inset,
059                              float[] innerRadii) {
060            if (outerRadii.length < 8) {
061                throw new ArrayIndexOutOfBoundsException(
062                                            "outer radii must have >= 8 values");
063            }
064            if (innerRadii != null && innerRadii.length < 8) {
065                throw new ArrayIndexOutOfBoundsException(
066                                            "inner radii must have >= 8 values");
067            }
068            mOuterRadii = outerRadii;
069            mInset = inset;
070            mInnerRadii = innerRadii;
071            
072            if (inset != null) {
073                mInnerRect = new RectF();
074            }
075            mPath = new Path();
076        }
077        
078        @Override
079        public void draw(Canvas canvas, Paint paint) {
080            canvas.drawPath(mPath, paint);
081        }
082        
083        @Override
084        protected void onResize(float w, float h) {
085            super.onResize(w, h);
086            
087            RectF r = rect();
088            mPath.reset();
089    
090            if (mOuterRadii != null) {
091                mPath.addRoundRect(r, mOuterRadii, Path.Direction.CW);
092            } else {
093                mPath.addRect(r, Path.Direction.CW);
094            }
095            if (mInnerRect != null) {
096                mInnerRect.set(r.left + mInset.left, r.top + mInset.top,
097                               r.right - mInset.right, r.bottom - mInset.bottom);
098                if (mInnerRect.width() < w && mInnerRect.height() < h) {
099                    if (mInnerRadii != null) {
100                        mPath.addRoundRect(mInnerRect, mInnerRadii,
101                                           Path.Direction.CCW);
102                    } else {
103                        mPath.addRect(mInnerRect, Path.Direction.CCW);
104                    }
105                }
106            }
107        }
108    
109        @Override
110        public RoundRectShape clone() throws CloneNotSupportedException {
111            RoundRectShape shape = (RoundRectShape) super.clone();
112            shape.mOuterRadii = mOuterRadii.clone();
113            shape.mInnerRadii = mInnerRadii.clone();
114            shape.mInset = new RectF(mInset);
115            shape.mInnerRect = new RectF(mInnerRect);
116            shape.mPath = new Path(mPath);
117            return shape;
118        }
119    }
