017    package android.graphics.drawable;
018    
019    import android.graphics.*;
020    import android.content.res.Resources;
021    import android.content.res.TypedArray;
022    import android.util.AttributeSet;
023    import android.util.TypedValue;
024    import org.xmlpull.v1.XmlPullParser;
025    import org.xmlpull.v1.XmlPullParserException;
026    
027    import java.io.IOException;
028    import java.io.InputStream;
029    
030    /**
031     * 
032     * A resizeable bitmap, with stretchable areas that you define. This type of image
033     * is defined in a .png file with a special format, described in <a link="../../../resources.html#ninepatch">
034     * Resources</a>.
035     *
036     */
037    public class NinePatchDrawable extends Drawable {
038        private NinePatchState mNinePatchState;
039        private NinePatch mNinePatch;
040        private Rect mPadding;
041        private Paint mPaint;
042        private boolean mMutated;
043    
044        NinePatchDrawable() {
045        }
046    
047        public NinePatchDrawable(Bitmap bitmap, byte[] chunk, Rect padding, String srcName) {
048            this(new NinePatchState(new NinePatch(bitmap, chunk, srcName), padding));
049        }
050        
051        public NinePatchDrawable(NinePatch patch) {
052            this(new NinePatchState(patch, null));
053        }
054    
055        private void setNinePatchState(NinePatchState state) {
056            mNinePatchState = state;
057            mNinePatch = state.mNinePatch;
058            mPadding = state.mPadding;
059            if (state.mDither) setDither(state.mDither);
060        }
061    
062        // overrides
063    
064        @Override
065        public void draw(Canvas canvas) {
066            mNinePatch.draw(canvas, getBounds(), mPaint);
067        }
068    
069        @Override
070        public int getChangingConfigurations() {
071            return super.getChangingConfigurations() | mNinePatchState.mChangingConfigurations;
072        }
073        
074        @Override
075        public boolean getPadding(Rect padding) {
076            padding.set(mPadding);
077            return true;
078        }
079    
080        @Override
081        public void setAlpha(int alpha) {
082            getPaint().setAlpha(alpha);
083        }
084        
085        @Override
086        public void setColorFilter(ColorFilter cf) {
087            getPaint().setColorFilter(cf);
088        }
089    
090        @Override
091        public void setDither(boolean dither) {
092            getPaint().setDither(dither);
093        }
094    
095        @Override
096        public void inflate(Resources r, XmlPullParser parser, AttributeSet attrs)
097                throws XmlPullParserException, IOException {
098            super.inflate(r, parser, attrs);
099    
100            TypedArray a = r.obtainAttributes(attrs, com.android.internal.R.styleable.NinePatchDrawable);
101    
102            final int id = a.getResourceId(com.android.internal.R.styleable.NinePatchDrawable_src, 0);
103            if (id == 0) {
104                throw new XmlPullParserException(parser.getPositionDescription() +
105                        ": <nine-patch> requires a valid src attribute");
106            }
107    
108            final boolean dither = a.getBoolean(
109                    com.android.internal.R.styleable.NinePatchDrawable_dither, false);
110            final BitmapFactory.Options options = new BitmapFactory.Options();
111            if (dither) {
112                options.inDither = false;
113            }
114    
115            final Rect padding = new Rect();        
116            Bitmap bitmap = null;
117    
118            try {
119                final TypedValue value = new TypedValue();
120                final InputStream is = r.openRawResource(id, value);
121    
122                bitmap = BitmapFactory.decodeStream(r, value, is, padding, options);
123    
124                is.close();
125            } catch (IOException e) {
126                // Ignore
127            }
128    
129            if (bitmap == null) {
130                throw new XmlPullParserException(parser.getPositionDescription() +
131                        ": <nine-patch> requires a valid src attribute");
132            } else if (bitmap.getNinePatchChunk() == null) {
133                throw new XmlPullParserException(parser.getPositionDescription() +
134                        ": <nine-patch> requires a valid 9-patch source image");
135            }
136    
137            setNinePatchState(new NinePatchState(
138                    new NinePatch(bitmap, bitmap.getNinePatchChunk(), "XML 9-patch"), padding, dither));
139    
140            a.recycle();
141        }
142    
143    
144        public Paint getPaint() {
145            if (mPaint == null) {
146                mPaint = new Paint();
147            }
148            return mPaint;
149        }
150    
151        /**
152         * Retrieves the width of the source .png file (before resizing).
153         */
154        @Override
155        public int getIntrinsicWidth() {
156            return mNinePatch.getWidth();
157        }
158    
159        /**
160         * Retrieves the height of the source .png file (before resizing).
161         */
162        @Override
163        public int getIntrinsicHeight() {
164            return mNinePatch.getHeight();
165        }
166    
167        @Override
168        public int getMinimumWidth() {
169            return mNinePatch.getWidth();
170        }
171    
172        @Override
173        public int getMinimumHeight() {
174            return mNinePatch.getHeight();
175        }
176    
177        /**
178         * Returns a {@link android.graphics.PixelFormat graphics.PixelFormat}
179         * value of OPAQUE or TRANSLUCENT.
180         */
181        @Override
182        public int getOpacity() {
183            return mNinePatch.hasAlpha() || (mPaint != null && mPaint.getAlpha() < 255) ?
184                    PixelFormat.TRANSLUCENT : PixelFormat.OPAQUE;
185        }
186    
187        @Override
188        public Region getTransparentRegion() {
189            return mNinePatch.getTransparentRegion(getBounds());
190        }
191        
192        @Override
193        public ConstantState getConstantState() {
194            mNinePatchState.mChangingConfigurations = super.getChangingConfigurations();
195            return mNinePatchState;
196        }
197    
198        @Override
199        public Drawable mutate() {
200            if (!mMutated && super.mutate() == this) {
201                mNinePatchState = new NinePatchState(mNinePatchState);
202                mNinePatch = mNinePatchState.mNinePatch;
203                mPadding = mNinePatchState.mPadding;
204                mMutated = true;
205            }
206            return this;
207        }
208    
209        final static class NinePatchState extends ConstantState {
210            final NinePatch mNinePatch;
211            final Rect mPadding;
212            final boolean mDither;
213            int mChangingConfigurations;
214    
215            NinePatchState(NinePatch ninePatch, Rect padding) {
216                this(ninePatch, padding, false);
217            }
218    
219            NinePatchState(NinePatch ninePatch, Rect rect, boolean dither) {
220                mNinePatch = ninePatch;
221                mPadding = rect;
222                mDither = dither;
223            }
224    
225            NinePatchState(NinePatchState state) {
226                mNinePatch = new NinePatch(state.mNinePatch);
227                mPadding = new Rect(state.mPadding);
228                mChangingConfigurations = state.mChangingConfigurations;
229                mDither = state.mDither;
230            }
231    
232            @Override
233            public Drawable newDrawable() {
234                return new NinePatchDrawable(this);
235            }
236            
237            @Override
238            public int getChangingConfigurations() {
239                return mChangingConfigurations;
240            }
241        }
242    
243        private NinePatchDrawable(NinePatchState state) {
244            setNinePatchState(state);
245        }
246    }
