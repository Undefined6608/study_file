017    package android.graphics.drawable;
018    
019    import android.graphics.drawable.shapes.RoundRectShape;
020    import android.content.res.Resources;
021    import android.content.res.TypedArray;
022    import android.util.AttributeSet;
023    
024    import org.xmlpull.v1.XmlPullParser;
025    
026    /**
027     * Drawable that draws its bounds in the given paint, with optional
028     * rounded corners.
029    */
030    public class PaintDrawable extends ShapeDrawable {
031    
032        public PaintDrawable() {
033        }
034    
035        public PaintDrawable(int color) {
036            getPaint().setColor(color);
037        }
038        
039        /**
040         * Specify radius for the corners of the rectangle. If this is > 0, then the
041         * drawable is drawn in a round-rectangle, rather than a rectangle.
042         * @param radius the radius for the corners of the rectangle
043         */
044        public void setCornerRadius(float radius) {
045            float[] radii = null;
046            if (radius > 0) {
047                radii = new float[8];
048                for (int i = 0; i < 8; i++) {
049                    radii[i] = radius;
050                }
051            }
052            setCornerRadii(radii);
053        }
054        
055        /**
056         * Specify radii for each of the 4 corners. For each corner, the array
057         * contains 2 values, [X_radius, Y_radius]. The corners are ordered
058         * top-left, top-right, bottom-right, bottom-left
059         * @param radii the x and y radii of the corners
060         */
061        public void setCornerRadii(float[] radii) {
062            if (radii == null) {
063                if (getShape() != null) {
064                    setShape(null);
065                }
066            } else {
067                setShape(new RoundRectShape(radii, null, null));
068            }
069        }
070    
071        @Override
072        protected boolean inflateTag(String name, Resources r, XmlPullParser parser,
073                                     AttributeSet attrs) {
074            if (name.equals("corners")) {
075                TypedArray a = r.obtainAttributes(attrs,
076                                            com.android.internal.R.styleable.DrawableCorners);
077                int radius = a.getDimensionPixelSize(
078                                    com.android.internal.R.styleable.DrawableCorners_radius, 0);
079                setCornerRadius(radius);
080                
081                // now check of they have any per-corner radii
082                
083                int topLeftRadius = a.getDimensionPixelSize(
084                        com.android.internal.R.styleable.DrawableCorners_topLeftRadius, radius);
085                int topRightRadius = a.getDimensionPixelSize(
086                        com.android.internal.R.styleable.DrawableCorners_topRightRadius, radius);
087                int bottomLeftRadius = a.getDimensionPixelSize(
088                    com.android.internal.R.styleable.DrawableCorners_bottomLeftRadius, radius);
089                int bottomRightRadius = a.getDimensionPixelSize(
090                    com.android.internal.R.styleable.DrawableCorners_bottomRightRadius, radius);
091    
092                if (topLeftRadius != radius || topRightRadius != radius ||
093                        bottomLeftRadius != radius || bottomRightRadius != radius) {
094                    setCornerRadii(new float[] {
095                                   topLeftRadius, topLeftRadius,
096                                   topRightRadius, topRightRadius,
097                                   bottomLeftRadius, bottomLeftRadius,
098                                   bottomRightRadius, bottomRightRadius
099                                   });
100                }
101                a.recycle();
102                return true;
103            }
104            return super.inflateTag(name, r, parser, attrs);
105        }
106    }
