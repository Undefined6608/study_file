017    package android.graphics;
018    
019    import android.content.res.AssetManager;
020    
021    /**
022     * The Typeface class specifies the typeface and intrinsic style of a font.
023     * This is used in the paint, along with optionally Paint settings like
024     * textSize, textSkewX, textScaleX to specify
025     * how text appears when drawn (and measured).
026     */
027    public class Typeface {
028    
029        /** The default NORMAL typeface object */
030        public static final Typeface DEFAULT;
031        /**
032         * The default BOLD typeface object. Note: this may be not actually be
033         * bold, depending on what fonts are installed. Call getStyle() to know
034         * for sure.
035         */
036        public static final Typeface DEFAULT_BOLD;
037        /** The NORMAL style of the default sans serif typeface. */
038        public static final Typeface SANS_SERIF;
039        /** The NORMAL style of the default serif typeface. */
040        public static final Typeface SERIF;
041        /** The NORMAL style of the default monospace typeface. */
042        public static final Typeface MONOSPACE;
043    
044        private static Typeface[] sDefaults;
045        
046        /* package */ int native_instance;
047    
048        // Style
049        public static final int NORMAL = 0;
050        public static final int BOLD = 1;
051        public static final int ITALIC = 2;
052        public static final int BOLD_ITALIC = 3;
053    
054        /** Returns the typeface's intrinsic style attributes */
055        public int getStyle() {
056            return nativeGetStyle(native_instance);
057        }
058    
059        /** Returns true if getStyle() has the BOLD bit set. */
060        public final boolean isBold() {
061            return (getStyle() & BOLD) != 0;
062        }
063    
064        /** Returns true if getStyle() has the ITALIC bit set. */
065        public final boolean isItalic() {
066            return (getStyle() & ITALIC) != 0;
067        }
068    
069        /**
070         * Create a typeface object given a family name, and option style information.
071         * If null is passed for the name, then the "default" font will be chosen.
072         * The resulting typeface object can be queried (getStyle()) to discover what
073         * its "real" style characteristics are.
074         *
075         * @param familyName May be null. The name of the font family.
076         * @param style  The style (normal, bold, italic) of the typeface.
077         *               e.g. NORMAL, BOLD, ITALIC, BOLD_ITALIC
078         * @return The best matching typeface.
079         */
080        public static Typeface create(String familyName, int style) {
081            return new Typeface(nativeCreate(familyName, style));
082        }
083    
084        /**
085         * Create a typeface object that best matches the specified existing
086         * typeface and the specified Style. Use this call if you want to pick a new
087         * style from the same family of an existing typeface object. If family is
088         * null, this selects from the default font's family.
089         *
090         * @param family May be null. The name of the existing type face.
091         * @param style  The style (normal, bold, italic) of the typeface.
092         *               e.g. NORMAL, BOLD, ITALIC, BOLD_ITALIC
093         * @return The best matching typeface.
094         */
095        public static Typeface create(Typeface family, int style) {
096            int ni = 0;        
097            if (family != null) {
098                ni = family.native_instance;
099            }
100            return new Typeface(nativeCreateFromTypeface(ni, style));
101        }
102    
103        /**
104         * Returns one of the default typeface objects, based on the specified style
105         *
106         * @return the default typeface that corresponds to the style
107         */
108        public static Typeface defaultFromStyle(int style) {
109            return sDefaults[style];
110        }
111        
112        /**
113         * Create a new typeface from the specified font data.
114         * @param mgr The application's asset manager
115         * @param path  The file name of the font data in the assets directory
116         * @return The new typeface.
117         */
118        public static Typeface createFromAsset(AssetManager mgr, String path) {
119            return new Typeface(nativeCreateFromAsset(mgr, path));
120        }
121        
122        // don't allow clients to call this directly
123        private Typeface(int ni) {
124            native_instance = ni;
125        }
126        
127        static {
128            DEFAULT         = create((String)null, 0);
129            DEFAULT_BOLD    = create((String)null, Typeface.BOLD);
130            SANS_SERIF      = create("sans-serif", 0);
131            SERIF           = create("serif", 0);
132            MONOSPACE       = create("monospace", 0);
133            
134            sDefaults = new Typeface[] {
135                DEFAULT,
136                DEFAULT_BOLD,
137                create((String)null, Typeface.ITALIC),
138                create((String)null, Typeface.BOLD_ITALIC),
139            };
140        }
141    
142        protected void finalize() throws Throwable {
143            nativeUnref(native_instance);
144        }
145    
146        private static native int  nativeCreate(String familyName, int style);
147        private static native int  nativeCreateFromTypeface(int native_instance,
148                                                            int style);
149        private static native void nativeUnref(int native_instance);
150        private static native int  nativeGetStyle(int native_instance);
151        private static native int  nativeCreateFromAsset(AssetManager mgr,
152                                                         String path);
153    }
