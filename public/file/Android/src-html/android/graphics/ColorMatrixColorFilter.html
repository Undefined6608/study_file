017    package android.graphics;
018    
019    public class ColorMatrixColorFilter extends ColorFilter {
020        /**
021         * Create a colorfilter that transforms colors through a 4x5 color matrix.
022         *
023         * @param matrix 4x5 matrix used to transform colors. It is copied into
024         *               the filter, so changes made to the matrix after the filter
025         *               is constructed will not be reflected in the filter.
026         */
027        public ColorMatrixColorFilter(ColorMatrix matrix) {
028            native_instance = nativeColorMatrixFilter(matrix.getArray());
029        }
030    
031        /**
032        * Create a colorfilter that transforms colors through a 4x5 color matrix.
033         *
034         * @param array array of floats used to transform colors, treated as a 4x5
035         *              matrix. The first 20 entries of the array are copied into
036         *              the filter. See ColorMatrix.
037         */
038        public ColorMatrixColorFilter(float[] array) {
039            if (array.length < 20) {
040                throw new ArrayIndexOutOfBoundsException();
041            }
042            native_instance = nativeColorMatrixFilter(array);
043        }
044    
045        private static native int nativeColorMatrixFilter(float[] array);
046    }
