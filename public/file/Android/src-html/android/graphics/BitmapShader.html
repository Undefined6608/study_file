017    package android.graphics;
018    
019    public class BitmapShader extends Shader {
020    
021        // we hold on just for the GC, since our native counterpart is using it
022        private Bitmap mBitmap;
023    
024        /**
025         * Call this to create a new shader that will draw with a bitmap.
026         *
027         * @param bitmap            The bitmap to use inside the shader
028         * @param tileX             The tiling mode for x to draw the bitmap in.
029         * @param tileY             The tiling mode for y to draw the bitmap in.
030         */
031        public BitmapShader(Bitmap bitmap, TileMode tileX, TileMode tileY) {
032            mBitmap = bitmap;
033            native_instance = nativeCreate(bitmap.ni(),
034                                           tileX.nativeInt, tileY.nativeInt);
035        }
036    
037        private static native int nativeCreate(int native_bitmap,
038                                               int shaderTileModeX,
039                                               int shaderTileModeY);    
040    }
