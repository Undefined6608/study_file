017    package android.graphics;
018    
019    public class PathDashPathEffect extends PathEffect {
020    
021        public enum Style {
022            TRANSLATE(0),   //!< translate the shape to each position
023            ROTATE(1),      //!< rotate the shape about its center
024            MORPH(2);       //!< transform each point, and turn lines into curves
025            
026            Style(int value) {
027                native_style = value;
028            }
029            int native_style;
030        }
031    
032        /**
033         * Dash the drawn path by stamping it with the specified shape. This only
034         * applies to drawings when the paint's style is STROKE or STROKE_AND_FILL.
035         * If the paint's style is FILL, then this effect is ignored. The paint's
036         * strokeWidth does not affect the results.
037         * @param shape The path to stamp along
038         * @param advance spacing between each stamp of shape
039         * @param phase amount to offset before the first shape is stamped
040         * @param style how to transform the shape at each position as it is stamped
041         */
042        public PathDashPathEffect(Path shape, float advance, float phase,
043                                  Style style) {
044            native_instance = nativeCreate(shape.ni(), advance, phase,
045                                           style.native_style);
046        }
047        
048        private static native int nativeCreate(int native_path, float advance,
049                                               float phase, int native_style);
050    }
