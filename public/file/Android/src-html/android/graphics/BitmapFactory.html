030    /**
031     * Creates Bitmap objects from various sources, including files, streams, 
032     * and byte-arrays. 
033     */
034    public class BitmapFactory {
035        public static class Options {
036            /**
037             * Create a default Options object, which if left unchanged will give
038             * the same result from the decoder as if null were passed.
039             */
040            public Options() {
041                inDither = true;
042                inDensity = 0;
043                inScaled = true;
044            }
045    
046            /**
047             * If set to true, the decoder will return null (no bitmap), but
048             * the out... fields will still be set, allowing the caller to query
049             * the bitmap without having to allocate the memory for its pixels.
050             */
051            public boolean inJustDecodeBounds;
052    
053            /**
054             * If set to a value > 1, requests the decoder to subsample the original
055             * image, returning a smaller image to save memory. The sample size is
056             * the number of pixels in either dimension that correspond to a single
057             * pixel in the decoded bitmap. For example, inSampleSize == 4 returns
058             * an image that is 1/4 the width/height of the original, and 1/16 the
059             * number of pixels. Any value <= 1 is treated the same as 1. Note: the
060             * decoder will try to fulfill this request, but the resulting bitmap
061             * may have different dimensions that precisely what has been requested.
062             * Also, powers of 2 are often faster/easier for the decoder to honor.
063             */
064            public int inSampleSize;
065            
066            /**
067             * If this is non-null, the decoder will try to decode into this
068             * internal configuration. If it is null, or the request cannot be met,
069             * the decoder will try to pick the best matching config based on the
070             * system's screen depth, and characteristics of the original image such
071             * as if it has per-pixel alpha (requiring a config that also does).
072             */
073            public Bitmap.Config inPreferredConfig;
074            
075            /**
076             * If dither is true, the decoder will atttempt to dither the decoded
077             * image.
078             */
079            public boolean inDither;
080    
081            /**
082             * The desired pixel density of the bitmap.
083             *
084             * @see android.util.DisplayMetrics#DEFAULT_DENSITY
085             * @see android.util.DisplayMetrics#density
086             *
087             * @hide pending API council approval
088             */
089            public int inDensity;
090    
091            /**
092             * </p>If the bitmap is loaded from {@link android.content.res.Resources} and
093             * this flag is turned on, the bitmap will be scaled to match the default
094             * display's pixel density.</p>
095             *
096             * </p>This flag is turned on by default and should be turned off if you need
097             * a non-scaled version of the bitmap. In this case,
098             * {@link android.graphics.Bitmap#setAutoScalingEnabled(boolean)} can be used
099             * to properly scale the bitmap at drawing time.</p>
100             *
101             * @hide pending API council approval
102             */
103            public boolean inScaled;
104    
105            /**
106             * The resulting width of the bitmap, set independent of the state of
107             * inJustDecodeBounds. However, if there is an error trying to decode,
108             * outWidth will be set to -1.
109             */
110            public int outWidth;
111    
112            /**
113             * The resulting height of the bitmap, set independent of the state of
114             * inJustDecodeBounds. However, if there is an error trying to decode,
115             * outHeight will be set to -1.
116             */
117            public int outHeight;
118    
119            /**
120             * If known, this string is set to the mimetype of the decoded image.
121             * If not know, or there is an error, it is set to null.
122             */
123            public String outMimeType;
124            
125            /**
126             * Temp storage to use for decoding.  Suggest 16K or so.
127             */
128            public byte[] inTempStorage;
129            
130            private native void requestCancel();
131    
132            /**
133             * Flag to indicate that cancel has been called on this object.  This
134             * is useful if there's an intermediary that wants to first decode the
135             * bounds and then decode the image.  In that case the intermediary
136             * can check, inbetween the bounds decode and the image decode, to see
137             * if the operation is canceled.
138             */
139            public boolean mCancel;
140            
141            /**
142             *  This can be called from another thread while this options object is
143             *  inside a decode... call. Calling this will notify the decoder that
144             *  it should cancel its operation. This is not guaranteed to cancel
145             *  the decode, but if it does, the decoder... operation will return
146             *  null, or if inJustDecodeBounds is true, will set outWidth/outHeight
147             *  to -1
148             */
149            public void requestCancelDecode() {
150                mCancel = true;
151                requestCancel();
152            }
153        }

155        /**
156         * Decode a file path into a bitmap. If the specified file name is null,
157         * or cannot be decoded into a bitmap, the function returns null.
158         *
159         * @param pathName complete path name for the file to be decoded.
160         * @param opts null-ok; Options that control downsampling and whether the
161         *             image should be completely decoded, or just is size returned.
162         * @return The decoded bitmap, or null if the image data could not be
163         *         decoded, or, if opts is non-null, if opts requested only the
164         *         size be returned (in opts.outWidth and opts.outHeight)
165         */
166        public static Bitmap decodeFile(String pathName, Options opts) {
167            Bitmap bm = null;
168            InputStream stream = null;
169            try {
170                stream = new FileInputStream(pathName);
171                bm = decodeStream(stream, null, opts);
172            } catch (Exception e) {
173                /*  do nothing.
174                    If the exception happened on open, bm will be null.
175                */
176            } finally {
177                if (stream != null) {
178                    try {
179                        stream.close();
180                    } catch (IOException e) {
181                        // do nothing here
182                    }
183                }
184            }
185            return bm;
186        }

193         * @return the resulting decoded bitmap, or null if it could not be decoded.
194         */
195        public static Bitmap decodeFile(String pathName) {

199        /**
200         * Decode a new Bitmap from an InputStream. This InputStream was obtained from
201         * resources, which we pass to be able to scale the bitmap accordingly.
202         *
203         * @hide
204         */
205        public static Bitmap decodeStream(Resources res, TypedValue value, InputStream is,
206                Rect pad, Options opts) {
207    
208            if (opts == null) {
209                opts = new Options();
210            }
211    
212            Bitmap bm = decodeStream(is, pad, opts);
213    
214            if (bm != null && res != null && value != null) {
215                byte[] np = bm.getNinePatchChunk();
216                final boolean isNinePatch = np != null && NinePatch.isNinePatchChunk(np);
217    
218                final int density = value.density;
219                if (opts.inDensity == 0) {
220                    opts.inDensity = density == TypedValue.DENSITY_DEFAULT ?
221                            DisplayMetrics.DEFAULT_DENSITY : density;
222                }            
223                float scale = opts.inDensity / (float) DisplayMetrics.DEFAULT_DENSITY;
224    
225                if (opts.inScaled || isNinePatch) {
226                    bm.setDensityScale(1.0f);
227                    bm.setAutoScalingEnabled(false);
228                    // Assume we are going to prescale for the screen
229                    scale = res.getDisplayMetrics().density / scale;
230                    if (scale != 1.0f) {
231                        // TODO: This is very inefficient and should be done in native by Skia
232                        final Bitmap oldBitmap = bm;
233                        bm = Bitmap.createScaledBitmap(oldBitmap, (int) (bm.getWidth() * scale + 0.5f),
234                                (int) (bm.getHeight() * scale + 0.5f), true);
235                        oldBitmap.recycle();
236    
237                        if (isNinePatch) {
238                            np = nativeScaleNinePatch(np, scale, pad);
239                            bm.setNinePatchChunk(np);
240                        }
241                    }
242                } else {
243                    bm.setDensityScale(scale);
244                    bm.setAutoScalingEnabled(true);
245                }
246            }
247    
248            return bm;
249        }

251        /**
252         * Decode an image referenced by a resource ID.
253         *
254         * @param res   The resources object containing the image data
255         * @param id The resource id of the image data
256         * @param opts null-ok; Options that control downsampling and whether the
257         *             image should be completely decoded, or just is size returned.
258         * @return The decoded bitmap, or null if the image data could not be
259         *         decoded, or, if opts is non-null, if opts requested only the
260         *         size be returned (in opts.outWidth and opts.outHeight)
261         */
262        public static Bitmap decodeResource(Resources res, int id, Options opts) {
263            Bitmap bm = null;
264            
265            try {
266                final TypedValue value = new TypedValue();
267                final InputStream is = res.openRawResource(id, value);
268    
269                bm = decodeStream(res, value, is, null, opts);
270                is.close();
271            } catch (java.io.IOException e) {
272                /*  do nothing.
273                    If the exception happened on open, bm will be null.
274                    If it happened on close, bm is still valid.
275                */
276            }
277            return bm;
278        }

292         * Decode an immutable bitmap from the specified byte array.
293         *
294         * @param data byte array of compressed image data
295         * @param offset offset into imageData for where the decoder should begin
296         *               parsing.
297         * @param length the number of bytes, beginning at offset, to parse
298         * @param opts null-ok; Options that control downsampling and whether the
299         *             image should be completely decoded, or just is size returned.

301         *         decoded, or, if opts is non-null, if opts requested only the
302         *         size be returned (in opts.outWidth and opts.outHeight)
303         */
304        public static Bitmap decodeByteArray(byte[] data, int offset, int length, Options opts) {
305            if ((offset | length) < 0 || data.length < offset + length) {
306                throw new ArrayIndexOutOfBoundsException();
307            }
308            return nativeDecodeByteArray(data, offset, length, opts);
309        }

324        /**
325         * Decode an input stream into a bitmap. If the input stream is null, or
326         * cannot be used to decode a bitmap, the function returns null.
327         * The stream's position will be where ever it was after the encoded data
328         * was read.
329         * 
330         * @param is The input stream that holds the raw data to be decoded into a
331         *           bitmap.
332         * @param outPadding If not null, return the padding rect for the bitmap if
333         *                   it exists, otherwise set padding to [-1,-1,-1,-1]. If
334         *                   no bitmap is returned (null) then padding is
335         *                   unchanged.
336         * @param opts null-ok; Options that control downsampling and whether the
337         *             image should be completely decoded, or just is size returned.
338         * @return The decoded bitmap, or null if the image data could not be
339         *         decoded, or, if opts is non-null, if opts requested only the
340         *         size be returned (in opts.outWidth and opts.outHeight)
341         */
342        public static Bitmap decodeStream(InputStream is, Rect outPadding, Options opts) {
343            // we don't throw in this case, thus allowing the caller to only check
344            // the cache, and not force the image to be decoded.
345            if (is == null) {
346                return null;
347            }
348            
349            // we need mark/reset to work properly
350    
351            if (!is.markSupported()) {
352                is = new BufferedInputStream(is, 16 * 1024);
353            }
354    
355            // so we can call reset() if a given codec gives up after reading up to
356            // this many bytes. FIXME: need to find out from the codecs what this
357            // value should be.
358            is.mark(1024);
359    
360            Bitmap  bm;
361    
362            if (is instanceof AssetManager.AssetInputStream) {
363                bm = nativeDecodeAsset(((AssetManager.AssetInputStream) is).getAssetInt(),
364                        outPadding, opts);
365            } else {
366                // pass some temp storage down to the native code. 1024 is made up,
367                // but should be large enough to avoid too many small calls back
368                // into is.read(...) This number is not related to the value passed
369                // to mark(...) above.
370                byte [] tempStorage = null;
371                if (opts != null)
372                    tempStorage = opts.inTempStorage;
373                if (tempStorage == null)
374                    tempStorage = new byte[16 * 1024];
375                bm = nativeDecodeStream(is, tempStorage, outPadding, opts);
376            }
377    
378            return bm;
379        }
