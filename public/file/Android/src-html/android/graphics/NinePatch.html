017    package android.graphics;
018    
019    
020    /**
021     * The NinePatch class permits drawing a bitmap in nine sections.
022     * The four corners are unscaled; the four edges are scaled in one axis,
023     * and the middle is scaled in both axes. Normally, the middle is 
024     * transparent so that the patch can provide a selection about a rectangle.
025     * Essentially, it allows the creation of custom graphics that will scale the 
026     * way that you define, when content added within the image exceeds the normal
027     * bounds of the graphic. For a thorough explanation of a NinePatch image, 
028     * read the discussion in the 
029     * <a href="{@docRoot}guide/topics/graphics/2d-graphics.html#nine-patch">2D
030     * Graphics</a> document.
031     * <p>
032     * The <a href="{@docRoot}guide/developing/tools/draw9patch.html">Draw 9-Patch</a> 
033     * tool offers an extremely handy way to create your NinePatch images,
034     * using a WYSIWYG graphics editor.
035     * </p>
036     */
037    public class NinePatch {
038        /** 
039         * Create a drawable projection from a bitmap to nine patches.
040         *
041         * @param bitmap    The bitmap describing the patches.
042         * @param chunk     The 9-patch data chunk describing how the underlying
043         *                  bitmap is split apart and drawn.
044         * @param srcName   The name of the source for the bitmap. Might be null.
045         */
046        public NinePatch(Bitmap bitmap, byte[] chunk, String srcName) {
047            mBitmap = bitmap;
048            mChunk = chunk;
049            mSrcName = srcName;
050            validateNinePatchChunk(mBitmap.ni(), chunk);
051        }
052    
053        /**
054         * @hide
055         */
056        public NinePatch(NinePatch patch) {
057            mBitmap = patch.mBitmap;
058            mChunk = patch.mChunk;
059            mSrcName = patch.mSrcName;
060            mPaint = new Paint(patch.mPaint);
061            validateNinePatchChunk(mBitmap.ni(), mChunk);
062        }
063    
064        public void setPaint(Paint p) {
065            mPaint = p;
066        }
067        
068        /** 
069         * Draw a bitmap to nine patches.
070         *
071         * @param canvas    A container for the current matrix and clip used to draw the bitmap.
072         * @param location  Where to draw the bitmap.
073         */
074        public void draw(Canvas canvas, RectF location) {
075            nativeDraw(canvas.mNativeCanvas, location,
076                       mBitmap.ni(), mChunk,
077                       mPaint != null ? mPaint.mNativePaint : 0);
078        }
079        
080        /** 
081         * Draw a bitmap to nine patches.
082         *
083         * @param canvas    A container for the current matrix and clip used to draw the bitmap.
084         * @param location  Where to draw the bitmap.
085         */
086        public void draw(Canvas canvas, Rect location) {
087            nativeDraw(canvas.mNativeCanvas, location,
088                       mBitmap.ni(), mChunk,
089                       mPaint != null ? mPaint.mNativePaint : 0);
090        }
091    
092        /** 
093         * Draw a bitmap to nine patches.
094         *
095         * @param canvas    A container for the current matrix and clip used to draw the bitmap.
096         * @param location  Where to draw the bitmap.
097         * @param paint     The Paint to draw through.
098         */
099        public void draw(Canvas canvas, Rect location, Paint paint) {
100            nativeDraw(canvas.mNativeCanvas, location,
101                       mBitmap.ni(), mChunk, paint != null ? paint.mNativePaint : 0);
102        }
103    
104        public int getWidth() {
105            return mBitmap.getWidth();
106        }
107    
108        public int getHeight() {
109            return mBitmap.getHeight();
110        }
111    
112        public final boolean hasAlpha() {
113            return mBitmap.hasAlpha();
114        }
115    
116        public final Region getTransparentRegion(Rect location) {
117            int r = nativeGetTransparentRegion(mBitmap.ni(), mChunk, location);
118            return r != 0 ? new Region(r) : null;
119        }
120        
121        public native static boolean isNinePatchChunk(byte[] chunk);
122    
123        private final Rect   mRect = new Rect();
124        private final Bitmap mBitmap;
125        private final byte[] mChunk;
126        private Paint        mPaint;
127        private String       mSrcName;  // Useful for debugging
128    
129        private static native void validateNinePatchChunk(int bitmap, byte[] chunk);
130        private static native void nativeDraw(int canvas_instance, RectF loc, int bitmap_instance,
131                                              byte[] c, int paint_instance_or_null);
132        private static native void nativeDraw(int canvas_instance, Rect loc, int bitmap_instance,
133                                              byte[] c, int paint_instance_or_null);
134        private static native int nativeGetTransparentRegion(
135                int bitmap, byte[] chunk, Rect location);
136    }
