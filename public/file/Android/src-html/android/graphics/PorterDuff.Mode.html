017    package android.graphics;
018    
019    public class PorterDuff {
020    
021        // these value must match their native equivalents. See SkPorterDuff.h
022        public enum Mode {
023            /** [0, 0] */
024            CLEAR       (0),
025            /** [Sa, Sc] */
026            SRC         (1),
027            /** [Da, Dc] */
028            DST         (2),
029            /** [Sa + (1 - Sa)*Da, Rc = Sc + (1 - Sa)*Dc] */
030            SRC_OVER    (3),
031            /** [Sa + (1 - Sa)*Da, Rc = Dc + (1 - Da)*Sc] */
032            DST_OVER    (4),
033            /** [Sa * Da, Sc * Da] */
034            SRC_IN      (5),
035            /** [Sa * Da, Sa * Dc] */
036            DST_IN      (6),
037            /** [Sa * (1 - Da), Sc * (1 - Da)] */
038            SRC_OUT     (7),
039            /** [Da * (1 - Sa), Dc * (1 - Sa)] */
040            DST_OUT     (8),
041            /** [Da, Sc * Da + (1 - Sa) * Dc] */
042            SRC_ATOP    (9),
043            /** [Sa, Sa * Dc + Sc * (1 - Da)] */
044            DST_ATOP    (10),
045            /** [Sa + Da - 2 * Sa * Da, Sc * (1 - Da) + (1 - Sa) * Dc] */
046            XOR         (11),
047            /** [Sa + Da - Sa*Da,
048                 Sc*(1 - Da) + Dc*(1 - Sa) + min(Sc, Dc)] */
049            DARKEN      (12),
050            /** [Sa + Da - Sa*Da,
051                 Sc*(1 - Da) + Dc*(1 - Sa) + max(Sc, Dc)] */
052            LIGHTEN     (13),
053            /** [Sa * Da, Sc * Dc] */
054            MULTIPLY    (14),
055            /** [Sa + Da - Sa * Da, Sc + Dc - Sc * Dc] */
056            SCREEN      (15);
057    
058            Mode(int nativeInt) {
059                this.nativeInt = nativeInt;
060            }
061            final int nativeInt;
062        }
063    }
