017    package android.graphics;
018    
019    public class PixelFormat
020    {
021        /* these constants need to match those
022           in ui/PixelFormat.h & pixelflinger/format.h */
023        
024        public static final int UNKNOWN     = 0;
025    
026        /** System chooses a format that supports translucency (many alpha bits) */
027        public static final int TRANSLUCENT = -3;
028    
029        /** 
030         * System chooses a format that supports transparency
031         * (at least 1 alpha bit) 
032         */    
033        public static final int TRANSPARENT = -2;
034    
035        /** System chooses an opaque format (no alpha bits required) */
036        public static final int OPAQUE      = -1;
037    
038        public static final int RGBA_8888   = 1;
039        public static final int RGBX_8888   = 2;
040        public static final int RGB_888     = 3;
041        public static final int RGB_565     = 4;
042    
043        public static final int RGBA_5551   = 6;
044        public static final int RGBA_4444   = 7;
045        public static final int A_8         = 8;
046        public static final int L_8         = 9;
047        public static final int LA_88       = 0xA;
048        public static final int RGB_332     = 0xB;
049        
050        /**
051         * YCbCr formats, used for video. These are not necessarily supported
052         * by the hardware.
053         */
054        public static final int YCbCr_422_SP= 0x10;
055        public static final int YCbCr_420_SP= 0x11;
056    
057        /**
058         * Encoded formats.  These are not necessarily supported by the hardware.
059         */
060        public static final int JPEG        = 0x100;
061    
062        /*
063         * We use a class initializer to allow the native code to cache some
064         * field offsets.
065         */
066        native private static void nativeClassInit();
067        static { nativeClassInit(); }
068    
069        public static native void getPixelFormatInfo(int format, PixelFormat info);
070        public static boolean formatHasAlpha(int format) {
071            switch (format) {
072                case PixelFormat.A_8:
073                case PixelFormat.LA_88:
074                case PixelFormat.RGBA_4444:
075                case PixelFormat.RGBA_5551:
076                case PixelFormat.RGBA_8888:
077                case PixelFormat.TRANSLUCENT:
078                case PixelFormat.TRANSPARENT:
079                    return true;
080            }
081            return false;
082        }
083        
084        public int  bytesPerPixel;
085        public int  bitsPerPixel;
086    }
