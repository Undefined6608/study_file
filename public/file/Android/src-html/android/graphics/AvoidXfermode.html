017    package android.graphics;
018    
019    /**
020     * AvoidXfermode xfermode will draw the src everywhere except on top of the
021     * opColor or, depending on the Mode, draw only on top of the opColor.
022     */
023    public class AvoidXfermode extends Xfermode {
024    
025        // these need to match the enum in SkAvoidXfermode.h on the native side
026        public enum Mode {
027            AVOID   (0),    //!< draw everywhere except on the opColor
028            TARGET  (1);    //!< draw only on top of the opColor
029            
030            Mode(int nativeInt) {
031                this.nativeInt = nativeInt;
032            }
033            final int nativeInt;
034        }
035        
036        /**
037         * This xfermode will draw the src everywhere except on top of the opColor
038         * or, depending on the Mode, draw only on top of the opColor.
039         *
040         * @param opColor The color to avoid (or to target depending on Mode). Note
041         *                that the alpha in opColor is ignored.
042         * @param tolerance How closely we compare a pixel to the opColor.
043         *                  0 - only operate if exact match
044         *                  255 - maximum gradation (blending) based on how
045         *                  similar the pixel is to our opColor (max tolerance)
046         * @param mode If we should avoid or target the opColor
047         */
048        public AvoidXfermode(int opColor, int tolerance, Mode mode) {
049            if (tolerance < 0 || tolerance > 255) {
050                throw new IllegalArgumentException("tolerance must be 0..255");
051            }
052            native_instance = nativeCreate(opColor, tolerance, mode.nativeInt);
053        }
054    
055        private static native int nativeCreate(int opColor, int tolerance,
056                                               int nativeMode);
057    }
