017    package android.graphics;
018    
019    /**
020     * Shader is the based class for objects that return horizontal spans of colors
021     * during drawing. A subclass of Shader is installed in a Paint calling
022     * paint.setShader(shader). After that any object (other than a bitmap) that is
023     * drawn with that paint will get its color(s) from the shader.
024     */
025    public class Shader {
026    
027        // this is set by subclasses, but don't make it public
028        /* package */ int native_instance;
029    
030        public enum TileMode {
031            /**
032             * replicate the edge color if the shader draws outside of its
033             * original bounds
034             */
035            CLAMP   (0),
036            /**
037             * repeat the shader's image horizontally and vertically
038             */
039            REPEAT  (1),
040            /**
041             * repeat the shader's image horizontally and vertically, alternating
042             * mirror images so that adjacent images always seam
043             */
044            MIRROR  (2);
045        
046            TileMode(int nativeInt) {
047                this.nativeInt = nativeInt;
048            }
049            final int nativeInt;
050        }
051    
052        /**
053         * Return true if the shader has a non-identity local matrix.
054         * @param localM If not null, it is set to the shader's local matrix.
055         * @return true if the shader has a non-identity local matrix
056         */
057        public boolean getLocalMatrix(Matrix localM) {
058            return nativeGetLocalMatrix(native_instance, localM.native_instance);
059        }
060    
061        /**
062         * Set the shader's local matrix. Passing null will reset the shader's
063         * matrix to identity
064         * @param localM The shader's new local matrix, or null to specify identity
065         */
066        public void setLocalMatrix(Matrix localM) {
067            nativeSetLocalMatrix(native_instance,
068                                 localM != null ? localM.native_instance : 0);
069        }
070    
071        protected void finalize() throws Throwable {
072            nativeDestructor(native_instance);
073        }
074    
075        private static native void nativeDestructor(int native_shader);
076        private static native boolean nativeGetLocalMatrix(int native_shader,
077                                                           int matrix_instance);
078        private static native void nativeSetLocalMatrix(int native_shader,
079                                                        int matrix_instance);
080    }
