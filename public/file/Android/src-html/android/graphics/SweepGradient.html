017    package android.graphics;
018    
019    public class SweepGradient extends Shader {
020    
021        /**
022         * A subclass of Shader that draws a sweep gradient around a center point.
023         *
024         * @param cx       The x-coordinate of the center
025         * @param cy       The y-coordinate of the center
026         * @param colors   The colors to be distributed between around the center.
027         *                 There must be at least 2 colors in the array.
028         * @param positions May be NULL. The relative position of
029         *                 each corresponding color in the colors array, beginning
030         *                 with 0 and ending with 1.0. If the values are not
031         *                 monotonic, the drawing may produce unexpected results.
032         *                 If positions is NULL, then the colors are automatically
033         *                 spaced evenly.
034         */
035        public SweepGradient(float cx, float cy,
036                             int colors[], float positions[]) {
037            if (colors.length < 2) {
038                throw new IllegalArgumentException("needs >= 2 number of colors");
039            }
040            if (positions != null && colors.length != positions.length) {
041                throw new IllegalArgumentException(
042                            "color and position arrays must be of equal length");
043            }
044            native_instance = nativeCreate1(cx, cy, colors, positions);
045        }
046    
047        /**
048         * A subclass of Shader that draws a sweep gradient around a center point.
049         *
050         * @param cx       The x-coordinate of the center
051         * @param cy       The y-coordinate of the center
052         * @param color0   The color to use at the start of the sweep
053         * @param color1   The color to use at the end of the sweep
054         */
055        public SweepGradient(float cx, float cy, int color0, int color1) {
056            native_instance = nativeCreate2(cx, cy, color0, color1);
057        }
058    
059        private static native int nativeCreate1(float x, float y,
060                                                int colors[], float positions[]);
061        private static native int nativeCreate2(float x, float y,
062                                                int color0, int color1);
063    }
