017    package android.graphics;
018    
019    public class RadialGradient extends Shader {
020    
021            /**     Create a shader that draws a radial gradient given the center and radius.
022            @param x        The x-coordinate of the center of the radius
023            @param y        The y-coordinate of the center of the radius
024                    @param radius   Must be positive. The radius of the circle for this gradient
025            @param colors   The colors to be distributed between the center and edge of the circle
026            @param positions May be NULL. The relative position of
027                            each corresponding color in the colors array. If this is NULL,
028                            the the colors are distributed evenly between the center and edge of the circle.
029            @param  tile    The Shader tiling mode
030            */
031            public RadialGradient(float x, float y, float radius,
032                              int colors[], float positions[], TileMode tile) {
033            if (radius <= 0) {
034                throw new IllegalArgumentException("radius must be > 0");
035            }
036            if (colors.length < 2) {
037                throw new IllegalArgumentException("needs >= 2 number of colors");
038            }
039            if (positions != null && colors.length != positions.length) {
040                throw new IllegalArgumentException("color and position arrays must be of equal length");
041            }
042            native_instance = nativeCreate1(x, y, radius, colors, positions, tile.nativeInt);
043        }
044    
045            /**     Create a shader that draws a radial gradient given the center and radius.
046            @param x        The x-coordinate of the center of the radius
047            @param y        The y-coordinate of the center of the radius
048                    @param radius   Must be positive. The radius of the circle for this gradient
049            @param color0   The color at the center of the circle.
050            @param color1   The color at the edge of the circle.
051            @param tile     The Shader tiling mode
052            */
053            public RadialGradient(float x, float y, float radius,
054                              int color0, int color1, TileMode tile) {
055            if (radius <= 0) {
056                throw new IllegalArgumentException("radius must be > 0");
057            }
058            native_instance = nativeCreate2(x, y, radius, color0, color1, tile.nativeInt);
059        }
060    
061            private static native int nativeCreate1(float x, float y, float radius,
062                                                int colors[], float positions[], int tileMode);
063            private static native int nativeCreate2(float x, float y, float radius,
064                                                int color0, int color1, int tileMode);
065    }
