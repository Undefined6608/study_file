017    package android.graphics;
018    
019    /** A subclass of shader that returns the coposition of two other shaders, combined by
020        an {@link android.graphics.Xfermode} subclass.
021    */
022    public class ComposeShader extends Shader {
023        /** Create a new compose shader, given shaders A, B, and a combining mode.
024            When the mode is applied, it will be given the result from shader A as its
025            "dst", and the result of from shader B as its "src".
026            @param shaderA  The colors from this shader are seen as the "dst" by the mode
027            @param shaderB  The colors from this shader are seen as the "src" by the mode
028            @param mode     The mode that combines the colors from the two shaders. If mode
029                            is null, then SRC_OVER is assumed.
030        */
031        public ComposeShader(Shader shaderA, Shader shaderB, Xfermode mode) {
032            native_instance = nativeCreate1(shaderA.native_instance, shaderB.native_instance,
033                                            (mode != null) ? mode.native_instance : 0);
034        }
035    
036        /** Create a new compose shader, given shaders A, B, and a combining PorterDuff mode.
037            When the mode is applied, it will be given the result from shader A as its
038            "dst", and the result of from shader B as its "src".
039            @param shaderA  The colors from this shader are seen as the "dst" by the mode
040            @param shaderB  The colors from this shader are seen as the "src" by the mode
041            @param mode     The PorterDuff mode that combines the colors from the two shaders.
042        */
043        public ComposeShader(Shader shaderA, Shader shaderB, PorterDuff.Mode mode) {
044            native_instance = nativeCreate2(shaderA.native_instance, shaderB.native_instance,
045                                            mode.nativeInt);
046        }
047    
048        private static native int nativeCreate1(int native_shaderA, int native_shaderB, int native_mode);
049        private static native int nativeCreate2(int native_shaderA, int native_shaderB, int porterDuffMode);
050    }
