017    package android.graphics;
018    
019    import android.os.Parcel;
020    import android.os.Parcelable;
021    
022    public class Region implements Parcelable {
023    
024        // the native values for these must match up with the enum in SkRegion.h
025        public enum Op {
026            DIFFERENCE(0),
027            INTERSECT(1),
028            UNION(2),
029            XOR(3),
030            REVERSE_DIFFERENCE(4),
031            REPLACE(5);
032    
033            Op(int nativeInt) {
034                this.nativeInt = nativeInt;
035            }
036            final int nativeInt;
037        }
038    
039        /** Create an empty region
040        */
041        public Region() {
042            this(nativeConstructor());
043        }
044    
045        /** Return a copy of the specified region
046        */
047        public Region(Region region) {
048            this(nativeConstructor());
049            nativeSetRegion(mNativeRegion, region.mNativeRegion);
050        }
051    
052        /** Return a region set to the specified rectangle
053        */
054        public Region(Rect r) {
055            mNativeRegion = nativeConstructor();
056            nativeSetRect(mNativeRegion, r.left, r.top, r.right, r.bottom);
057        }
058    
059        /** Return a region set to the specified rectangle
060        */
061        public Region(int left, int top, int right, int bottom) {
062            mNativeRegion = nativeConstructor();
063            nativeSetRect(mNativeRegion, left, top, right, bottom);
064        }
065    
066        /** Set the region to the empty region
067        */
068        public void setEmpty() {
069            nativeSetRect(mNativeRegion, 0, 0, 0, 0);
070        }
071    
072        /** Set the region to the specified region.
073        */
074        public boolean set(Region region) {
075            return nativeSetRegion(mNativeRegion, region.mNativeRegion);
076        }
077    
078        /** Set the region to the specified rectangle
079        */
080        public boolean set(Rect r) {
081            return nativeSetRect(mNativeRegion, r.left, r.top, r.right, r.bottom);
082        }
083        
084        /** Set the region to the specified rectangle
085        */
086        public boolean set(int left, int top, int right, int bottom) {
087            return nativeSetRect(mNativeRegion, left, top, right, bottom);
088        }
089    
090        /**
091         * Set the region to the area described by the path and clip.
092         * Return true if the resulting region is non-empty. This produces a region
093         * that is identical to the pixels that would be drawn by the path
094         * (with no antialiasing).
095         */
096        public boolean setPath(Path path, Region clip) {
097            return nativeSetPath(mNativeRegion, path.ni(), clip.mNativeRegion);
098        }
099    
100        /**
101         * Return true if this region is empty
102         */
103        public native boolean isEmpty();
104        
105        /**
106         * Return true if the region contains a single rectangle
107         */
108        public native boolean isRect();
109        
110        /**
111         * Return true if the region contains more than one rectangle
112         */
113        public native boolean isComplex();
114    
115        /**
116         * Return a new Rect set to the bounds of the region. If the region is
117         * empty, the Rect will be set to [0, 0, 0, 0]
118         */
119        public Rect getBounds() {
120            Rect r = new Rect();
121            nativeGetBounds(mNativeRegion, r);
122            return r;
123        }
124        
125        /**
126         * Set the Rect to the bounds of the region. If the region is empty, the
127         * Rect will be set to [0, 0, 0, 0]
128         */
129        public boolean getBounds(Rect r) {
130            if (r == null) {
131                throw new NullPointerException();
132            }
133            return nativeGetBounds(mNativeRegion, r);
134        }
135    
136        /**
137         * Return the boundary of the region as a new Path. If the region is empty,
138         * the path will also be empty.
139         */
140        public Path getBoundaryPath() {
141            Path path = new Path();
142            nativeGetBoundaryPath(mNativeRegion, path.ni());
143            return path;
144        }
145    
146        /**
147         * Set the path to the boundary of the region. If the region is empty, the
148         * path will also be empty.
149         */
150        public boolean getBoundaryPath(Path path) {
151            return nativeGetBoundaryPath(mNativeRegion, path.ni());
152        }
153            
154        /**
155         * Return true if the region contains the specified point
156         */
157        public native boolean contains(int x, int y);
158    
159        /**
160         * Return true if the region is a single rectangle (not complex) and it
161         * contains the specified rectangle. Returning false is not a guarantee
162         * that the rectangle is not contained by this region, but return true is a
163         * guarantee that the rectangle is contained by this region.
164         */
165        public boolean quickContains(Rect r) {
166            return quickContains(r.left, r.top, r.right, r.bottom);
167        }
168    
169        /**
170         * Return true if the region is a single rectangle (not complex) and it
171         * contains the specified rectangle. Returning false is not a guarantee
172         * that the rectangle is not contained by this region, but return true is a
173         * guarantee that the rectangle is contained by this region.
174         */
175        public native boolean quickContains(int left, int top, int right,
176                                            int bottom);
177    
178        /**
179         * Return true if the region is empty, or if the specified rectangle does
180         * not intersect the region. Returning false is not a guarantee that they
181         * intersect, but returning true is a guarantee that they do not.
182         */
183        public boolean quickReject(Rect r) {
184            return quickReject(r.left, r.top, r.right, r.bottom);
185        }
186    
187        /**
188         * Return true if the region is empty, or if the specified rectangle does
189         * not intersect the region. Returning false is not a guarantee that they
190         * intersect, but returning true is a guarantee that they do not.
191         */
192        public native boolean quickReject(int left, int top, int right, int bottom);
193    
194        /**
195         * Return true if the region is empty, or if the specified region does not
196         * intersect the region. Returning false is not a guarantee that they
197         * intersect, but returning true is a guarantee that they do not.
198         */
199        public native boolean quickReject(Region rgn);
200    
201        /**
202         * Translate the region by [dx, dy]. If the region is empty, do nothing.
203         */
204        public void translate(int dx, int dy) {
205            translate(dx, dy, null);
206        }
207    
208        /**
209         * Set the dst region to the result of translating this region by [dx, dy].
210         * If this region is empty, then dst will be set to empty.
211         */
212        public native void translate(int dx, int dy, Region dst);
213    
214        public final boolean union(Rect r) {
215            return op(r, Op.UNION);
216        }
217    
218        /**
219         * Perform the specified Op on this region and the specified rect. Return
220         * true if the result of the op is not empty.
221         */
222        public boolean op(Rect r, Op op) {
223            return nativeOp(mNativeRegion, r.left, r.top, r.right, r.bottom,
224                            op.nativeInt);
225        }
226    
227        /**
228         * Perform the specified Op on this region and the specified rect. Return
229         * true if the result of the op is not empty.
230         */
231        public boolean op(int left, int top, int right, int bottom, Op op) {
232            return nativeOp(mNativeRegion, left, top, right, bottom,
233                            op.nativeInt);
234        }
235    
236        /**
237         * Perform the specified Op on this region and the specified region. Return
238         * true if the result of the op is not empty.
239         */
240        public boolean op(Region region, Op op) {
241            return op(this, region, op);
242        }
243    
244        /**
245         * Set this region to the result of performing the Op on the specified rect
246         * and region. Return true if the result is not empty.
247         */
248        public boolean op(Rect rect, Region region, Op op) {
249            return nativeOp(mNativeRegion, rect, region.mNativeRegion,
250                            op.nativeInt);
251        }
252    
253        /**
254         * Set this region to the result of performing the Op on the specified
255         * regions. Return true if the result is not empty.
256         */
257        public boolean op(Region region1, Region region2, Op op) {
258            return nativeOp(mNativeRegion, region1.mNativeRegion,
259                            region2.mNativeRegion, op.nativeInt);
260        }
261    
262        //////////////////////////////////////////////////////////////////////////
263        
264        public static final Parcelable.Creator<Region> CREATOR
265            = new Parcelable.Creator<Region>() {
266                /**
267                * Rebuild a Region previously stored with writeToParcel().
268                 * @param p    Parcel object to read the region from
269                 * @return a new region created from the data in the parcel
270                 */
271                public Region createFromParcel(Parcel p) {
272                    int ni = nativeCreateFromParcel(p);
273                    if (ni == 0) {
274                        throw new RuntimeException();
275                    }
276                    return new Region(ni);
277                }
278                public Region[] newArray(int size) {
279                    return new Region[size];
280                }
281        };
282        
283        public int describeContents() {
284            return 0;
285        }
286    
287        /**
288         * Write the region and its pixels to the parcel. The region can be
289         * rebuilt from the parcel by calling CREATOR.createFromParcel().
290         * @param p    Parcel object to write the region data into
291         */
292        public void writeToParcel(Parcel p, int flags) {
293            if (!nativeWriteToParcel(mNativeRegion, p)) {
294                throw new RuntimeException();
295            }
296        }
297        
298        protected void finalize() throws Throwable {
299            nativeDestructor(mNativeRegion);
300        }
301        
302        /*package*/ Region(int ni) {
303            if (ni == 0) {
304                throw new RuntimeException();
305            }
306            mNativeRegion = ni;
307        }
308    
309        /* add dummy parameter so constructor can be called from jni without
310           triggering 'not cloneable' exception */
311        private Region(int ni, int dummy) {
312            this(ni);
313        }
314    
315        /*package*/ final int ni() {
316            return mNativeRegion;
317        }
318    
319        private static native int nativeConstructor();
320        private static native void nativeDestructor(int native_region);
321    
322        private static native boolean nativeSetRegion(int native_dst,
323                                                      int native_src);
324        private static native boolean nativeSetRect(int native_dst, int left,
325                                                    int top, int right, int bottom);
326        private static native boolean nativeSetPath(int native_dst, int native_path,
327                                                    int native_clip);
328        private static native boolean nativeGetBounds(int native_region, Rect rect);
329        private static native boolean nativeGetBoundaryPath(int native_region,
330                                                            int native_path);
331    
332        private static native boolean nativeOp(int native_dst, int left, int top,
333                                               int right, int bottom, int op);
334        private static native boolean nativeOp(int native_dst, Rect rect,
335                                               int native_region, int op);
336        private static native boolean nativeOp(int native_dst, int native_region1,
337                                               int native_region2, int op);
338    
339        private static native int nativeCreateFromParcel(Parcel p);
340        private static native boolean nativeWriteToParcel(int native_region,
341                                                          Parcel p);
342    
343        private final int mNativeRegion;
344    }
