017    package android.telephony;
018    
019    import android.os.Bundle;
020    import android.os.RemoteException;
021    import android.os.ServiceManager;
022    import android.telephony.gsm.GsmCellLocation;
023    import com.android.internal.telephony.ITelephony;
024    
025    /**
026     * Abstract class that represents the location of the device.  Currently the only
027     * subclass is {@link android.telephony.gsm.GsmCellLocation}.  {@more}
028     */
029    public abstract class CellLocation {
030    
031        /**
032         * Request an update of the current location.  If the location has changed,
033         * a broadcast will be sent to everyone registered with {@link
034         * PhoneStateListener#LISTEN_CELL_LOCATION}.
035         */
036        public static void requestLocationUpdate() {
037            try {
038                ITelephony phone = ITelephony.Stub.asInterface(ServiceManager.getService("phone"));
039                if (phone != null) {
040                    phone.updateServiceLocation();
041                }
042            } catch (RemoteException ex) {
043                // ignore it
044            }
045        }
046    
047        /**
048         * Create a new CellLocation from a intent notifier Bundle
049         *
050         * This method is used by PhoneStateIntentReceiver and maybe by
051         * external applications.
052         *
053         * @param bundle Bundle from intent notifier
054         * @return newly created CellLocation
055         *
056         * @hide
057         */
058        public static CellLocation newFromBundle(Bundle bundle) {
059            return new GsmCellLocation(bundle);
060        }
061    
062        /**
063         * @hide
064         */
065        public abstract void fillInNotifierBundle(Bundle bundle);
066    
067        /**
068         * Return a new CellLocation object representing an unknown location.
069         */
070        public static CellLocation getEmpty() {
071            return new GsmCellLocation();
072        }
073    }
