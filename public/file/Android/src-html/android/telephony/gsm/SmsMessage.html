034    class SmsAddress {
035        // From TS 23.040 9.1.2.5 and TS 24.008 table 10.5.118
036        static final int TON_UNKNOWN = 0;
037    
038        static final int TON_INTERNATIONAL = 1;
039

045    
046        static final int TON_ALPHANUMERIC = 5;
047    
048        static final int TON_APPREVIATED = 6;
049    
050        static final int OFFSET_ADDRESS_LENGTH = 0;
051    
052        static final int OFFSET_TOA = 1;
053    
054        static final int OFFSET_ADDRESS_VALUE = 2;
055    
056        int ton;
057    
058        String address;
059    
060        byte[] origBytes;
061    
062        /**
063         * New SmsAddress from TS 23.040 9.1.2.5 Address Field
064         *
065         * @param offset the offset of the Address-Length byte
066         * @param length the length in bytes rounded up, e.g. "2 +
067         *        (addressLength + 1) / 2"
068         */
069    
070        SmsAddress(byte[] data, int offset, int length) {
071            origBytes = new byte[length];
072            System.arraycopy(data, offset, origBytes, 0, length);
073    
074            // addressLength is the count of semi-octets, not bytes
075            int addressLength = origBytes[OFFSET_ADDRESS_LENGTH] & 0xff;
076    
077            int toa = origBytes[OFFSET_TOA] & 0xff;
078            ton = 0x7 & (toa >> 4);
079    
080            // TOA must have its high bit set
081            if ((toa & 0x80) != 0x80) {
082                throw new RuntimeException("Invalid TOA - high bit must be set");
083            }
084    
085            if (isAlphanumeric()) {
086                // An alphanumeric address
087                int countSeptets = addressLength * 4 / 7;
088    
089                address = GsmAlphabet.gsm7BitPackedToString(origBytes,
090                        OFFSET_ADDRESS_VALUE, countSeptets);
091            } else {
092                // TS 23.040 9.1.2.5 says
093                // that "the MS shall interpret reserved values as 'Unknown'
094                // but shall store them exactly as received"
095    
096                byte lastByte = origBytes[length - 1];
097    
098                if ((addressLength & 1) == 1) {
099                    // Make sure the final unused BCD digit is 0xf
100                    origBytes[length - 1] |= 0xf0;
101                }
102                address = PhoneNumberUtils.calledPartyBCDToString(origBytes,
103                        OFFSET_TOA, length - OFFSET_TOA);
104    
105                // And restore origBytes
106                origBytes[length - 1] = lastByte;
107            }
108        }

111            return address;
112        }

125        /**
126         * Returns true of this is a valid CPHS voice message waiting indicator
127         * address
128         */
129        public boolean isCphsVoiceMessageIndicatorAddress() {
130            // CPHS-style MWI message
131            // See CPHS 4.7 B.4.2.1
132            //
133            // Basically:
134            //
135            // - Originating address should be 4 bytes long and alphanumeric
136            // - Decode will result with two chars:
137            // - Char 1
138            // 76543210
139            // ^ set/clear indicator (0 = clear)
140            // ^^^ type of indicator (000 = voice)
141            // ^^^^ must be equal to 0001
142            // - Char 2:
143            // 76543210
144            // ^ line number (0 = line 1)
145            // ^^^^^^^ set to 0
146            //
147            // Remember, since the alpha address is stored in 7-bit compact form,
148            // the "line number" is really the top bit of the first address value
149            // byte
150    
151            return (origBytes[OFFSET_ADDRESS_LENGTH] & 0xff) == 4
152                    && isAlphanumeric() && (origBytes[OFFSET_TOA] & 0x0f) == 0;
153        }
154    
155        /**
156         * Returns true if this is a valid CPHS voice message waiting indicator
157         * address indicating a "set" of "indicator 1" of type "voice message
158         * waiting"
159         */
160        public boolean isCphsVoiceMessageSet() {
161            // 0x11 means "set" "voice message waiting" "indicator 1"
162            return isCphsVoiceMessageIndicatorAddress()
163                    && (origBytes[OFFSET_ADDRESS_VALUE] & 0xff) == 0x11;
164    
165        }
166    
167        /**
168         * Returns true if this is a valid CPHS voice message waiting indicator
169         * address indicationg a "clear" of "indicator 1" of type "voice message
170         * waiting"
171         */
172        public boolean isCphsVoiceMessageClear() {
173            // 0x10 means "clear" "voice message waiting" "indicator 1"
174            return isCphsVoiceMessageIndicatorAddress()
175                    && (origBytes[OFFSET_ADDRESS_VALUE] & 0xff) == 0x10;
176    
177        }
178    
179        public boolean couldBeEmailGateway() {
180            // Some carriers seems to send email gateway messages in this form:
181            // from: an UNKNOWN TON, 3 or 4 digits long, beginning with a 5
182            // PID: 0x00, Data coding scheme 0x03
183            // So we just attempt to treat any message from an address length <= 4
184            // as an email gateway
185    
186            return address.length() <= 4;
187        }

401        /**
402         * Create an SmsMessage from an SMS EF record.
403         *
404         * @param index Index of SMS record. This should be index in ArrayList
405         *              returned by SmsManager.getAllMessagesFromSim + 1.
406         * @param data Record data.
407         * @return An SmsMessage representing the record.
408         * 
409         * @hide
410         */
411        public static SmsMessage createFromEfRecord(int index, byte[] data) {
412            try {
413                SmsMessage msg = new SmsMessage();
414    
415                msg.indexOnSim = index;
416    
417                // First byte is status: RECEIVED_READ, RECEIVED_UNREAD, STORED_SENT,
418                // or STORED_UNSENT
419                // See TS 51.011 10.5.3
420                if ((data[0] & 1) == 0) {
421                    Log.w(LOG_TAG,
422                            "SMS parsing failed: Trying to parse a free record");
423                    return null;
424                } else {
425                    msg.statusOnSim = data[0] & 0x07;
426                }
427    
428                int size = data.length - 1;
429    
430                // Note: Data may include trailing FF's.  That's OK; message
431                // should still parse correctly.
432                byte[] pdu = new byte[size];
433                System.arraycopy(data, 1, pdu, 0, size);
434                msg.parsePdu(pdu);
435                return msg;
436            } catch (RuntimeException ex) {
437                Log.e(LOG_TAG, "SMS PDU parsing failed: ", ex);
438                return null;
439            }
440        }

455        /**
456         * Calculates the number of SMS's required to encode the message body and
457         * the number of characters remaining until the next message, given the
458         * current encoding.
459         *
460         * @param messageBody the message to encode
461         * @param use7bitOnly if true, characters that are not part of the GSM
462         *         alphabet are counted as a single space char.  If false, a
463         *         messageBody containing non-GSM alphabet characters is calculated
464         *         for 16-bit encoding.
465         * @return an int[4] with int[0] being the number of SMS's required, int[1]
466         *         the number of code units used, and int[2] is the number of code
467         *         units remaining until the next message. int[3] is the encoding
468         *         type that should be used for the message.
469         */
470        public static int[] calculateLength(CharSequence messageBody, boolean use7bitOnly) {
471            int ret[] = new int[4];
472    
473            try {
474                // Try GSM alphabet
475                int septets = GsmAlphabet.countGsmSeptets(messageBody, !use7bitOnly);
476                ret[1] = septets;
477                if (septets > MAX_USER_DATA_SEPTETS) {
478                    ret[0] = (septets / MAX_USER_DATA_SEPTETS_WITH_HEADER) + 1;
479                    ret[2] = MAX_USER_DATA_SEPTETS_WITH_HEADER
480                                - (septets % MAX_USER_DATA_SEPTETS_WITH_HEADER);
481                } else {
482                    ret[0] = 1;
483                    ret[2] = MAX_USER_DATA_SEPTETS - septets;
484                }
485                ret[3] = ENCODING_7BIT;
486            } catch (EncodeException ex) {
487                // fall back to UCS-2
488                int octets = messageBody.length() * 2;
489                ret[1] = messageBody.length();
490                if (octets > MAX_USER_DATA_BYTES) {
491                    // 6 is the size of the user data header
492                    ret[0] = (octets / MAX_USER_DATA_BYTES_WITH_HEADER) + 1;
493                    ret[2] = (MAX_USER_DATA_BYTES_WITH_HEADER
494                                - (octets % MAX_USER_DATA_BYTES_WITH_HEADER))/2;
495                } else {
496                    ret[0] = 1;
497                    ret[2] = (MAX_USER_DATA_BYTES - octets)/2;
498                }
499                ret[3] = ENCODING_16BIT;
500            }
501    
502            return ret;
503        }

525        /**
526         * Get an SMS-SUBMIT PDU for a destination address and a message
527         *
528         * @param scAddress Service Centre address.  Null means use default.
529         * @return a <code>SubmitPdu</code> containing the encoded SC
530         *         address, if applicable, and the encoded message.
531         *         Returns null on encode error.
532         * @hide
533         */
534        public static SubmitPdu getSubmitPdu(String scAddress,
535                String destinationAddress, String message,
536                boolean statusReportRequested, byte[] header) {
537    
538            // Perform null parameter checks.
539            if (message == null || destinationAddress == null) {
540                return null;
541            }
542    
543            SubmitPdu ret = new SubmitPdu();
544            // MTI = SMS-SUBMIT, UDHI = header != null
545            byte mtiByte = (byte)(0x01 | (header != null ? 0x40 : 0x00));
546            ByteArrayOutputStream bo = getSubmitPduHead(
547                    scAddress, destinationAddress, mtiByte,
548                    statusReportRequested, ret);
549    
550            try {
551                // First, try encoding it with the GSM alphabet
552    
553                // User Data (and length)
554                byte[] userData = GsmAlphabet.stringToGsm7BitPackedWithHeader(message, header);
555    
556                if ((0xff & userData[0]) > MAX_USER_DATA_SEPTETS) {
557                    // Message too long
558                    return null;
559                }
560    
561                // TP-Data-Coding-Scheme
562                // Default encoding, uncompressed
563                // To test writing messages to the SIM card, change this value 0x00 to 0x12, which
564                // means "bits 1 and 0 contain message class, and the class is 2". Note that this
565                // takes effect for the sender. In other words, messages sent by the phone with this
566                // change will end up on the receiver's SIM card. You can then send messages to
567                // yourself (on a phone with this change) and they'll end up on the SIM card.
568                bo.write(0x00); 
569    
570                // (no TP-Validity-Period)
571    
572                bo.write(userData, 0, userData.length);
573            } catch (EncodeException ex) {
574                byte[] userData, textPart;
575                // Encoding to the 7-bit alphabet failed. Let's see if we can
576                // send it as a UCS-2 encoded message
577    
578                try {
579                    textPart = message.getBytes("utf-16be");
580                } catch (UnsupportedEncodingException uex) {
581                    Log.e(LOG_TAG,
582                          "Implausible UnsupportedEncodingException ",
583                          uex);
584                    return null;
585                }
586                
587                if (header != null) {
588                    userData = new byte[header.length + textPart.length];
589                    
590                    System.arraycopy(header, 0, userData, 0, header.length);
591                    System.arraycopy(textPart, 0, userData, header.length, textPart.length);
592                } else {
593                    userData = textPart;
594                }
595    
596                if (userData.length > MAX_USER_DATA_BYTES) {
597                    // Message too long
598                    return null;
599                }
600    
601                // TP-Data-Coding-Scheme
602                // Class 3, UCS-2 encoding, uncompressed
603                bo.write(0x0b);
604    
605                // (no TP-Validity-Period)
606                
607                // TP-UDL
608                bo.write(userData.length);
609    
610                bo.write(userData, 0, userData.length);
611            }
612    
613            ret.encodedMessage = bo.toByteArray();
614            return ret;
615        }

633        /**
634         * Get an SMS-SUBMIT PDU for a data message to a destination address &amp; port
635         *
636         * @param scAddress Service Centre address. null == use default
637         * @param destinationAddress the address of the destination for the message
638         * @param destinationPort the port to deliver the message to at the
639         *        destination
640         * @param data the dat for the message
641         * @return a <code>SubmitPdu</code> containing the encoded SC
642         *         address, if applicable, and the encoded message.
643         *         Returns null on encode error.
644         */
645        public static SubmitPdu getSubmitPdu(String scAddress,
646                String destinationAddress, short destinationPort, byte[] data,
647                boolean statusReportRequested) {
648            if (data.length > (MAX_USER_DATA_BYTES - 7 /* UDH size */)) {
649                Log.e(LOG_TAG, "SMS data message may only contain "
650                        + (MAX_USER_DATA_BYTES - 7) + " bytes");
651                return null;
652            }
653    
654            SubmitPdu ret = new SubmitPdu();
655            ByteArrayOutputStream bo = getSubmitPduHead(
656                    scAddress, destinationAddress, (byte) 0x41, // MTI = SMS-SUBMIT,
657                                                                // TP-UDHI = true
658                    statusReportRequested, ret);
659    
660            // TP-Data-Coding-Scheme
661            // No class, 8 bit data
662            bo.write(0x04);
663    
664            // (no TP-Validity-Period)
665    
666            // User data size
667            bo.write(data.length + 7);
668    
669            // User data header size
670            bo.write(0x06); // header is 6 octets
671    
672            // User data header, indicating the destination port
673            bo.write(SmsHeader.APPLICATION_PORT_ADDRESSING_16_BIT); // port
674                                                                    // addressing
675                                                                    // header
676            bo.write(0x04); // each port is 2 octets
677            bo.write((destinationPort >> 8) & 0xFF); // MSB of destination port
678            bo.write(destinationPort & 0xFF); // LSB of destination port
679            bo.write(0x00); // MSB of originating port
680            bo.write(0x00); // LSB of originating port
681    
682            // User data
683            bo.write(data, 0, data.length);
684    
685            ret.encodedMessage = bo.toByteArray();
686            return ret;
687        }

689        /**
690         * Create the beginning of a SUBMIT PDU.  This is the part of the
691         * SUBMIT PDU that is common to the two versions of {@link #getSubmitPdu},
692         * one of which takes a byte array and the other of which takes a
693         * <code>String</code>.
694         *
695         * @param scAddress Service Centre address. null == use default
696         * @param destinationAddress the address of the destination for the message
697         * @param mtiByte
698         * @param ret <code>SubmitPdu</code> containing the encoded SC
699         *        address, if applicable, and the encoded message
700         */
701        private static ByteArrayOutputStream getSubmitPduHead(
702                String scAddress, String destinationAddress, byte mtiByte,
703                boolean statusReportRequested, SubmitPdu ret) {
704            ByteArrayOutputStream bo = new ByteArrayOutputStream(
705                    MAX_USER_DATA_BYTES + 40);
706    
707            // SMSC address with length octet, or 0
708            if (scAddress == null) {
709                ret.encodedScAddress = null;
710            } else {
711                ret.encodedScAddress = PhoneNumberUtils.networkPortionToCalledPartyBCDWithLength(
712                        scAddress);
713            }
714    
715            // TP-Message-Type-Indicator (and friends)
716            if (statusReportRequested) {
717                // Set TP-Status-Report-Request bit.
718                mtiByte |= 0x20;
719                if (Config.LOGD) Log.d(LOG_TAG, "SMS status report requested");
720            }
721            bo.write(mtiByte);
722    
723            // space for TP-Message-Reference
724            bo.write(0);
725    
726            byte[] daBytes;
727    
728            daBytes = PhoneNumberUtils.networkPortionToCalledPartyBCD(destinationAddress);
729    
730            // destination address length in BCD digits, ignoring TON byte and pad
731            // TODO Should be better.
732            bo.write((daBytes.length - 1) * 2
733                    - ((daBytes[daBytes.length - 1] & 0xf0) == 0xf0 ? 1 : 0));
734    
735            // destination address
736            bo.write(daBytes, 0, daBytes.length);
737    
738            // TP-Protocol-Identifier
739            bo.write(0);
740            return bo;
741        }

824            /**
825             * Parses an SC timestamp and returns a currentTimeMillis()-style
826             * timestamp
827             */
828    
829            long getSCTimestampMillis() {
830                // TP-Service-Centre-Time-Stamp
831                int year = SimUtils.bcdByteToInt(pdu[cur++]);
832                int month = SimUtils.bcdByteToInt(pdu[cur++]);
833                int day = SimUtils.bcdByteToInt(pdu[cur++]);
834                int hour = SimUtils.bcdByteToInt(pdu[cur++]);
835                int minute = SimUtils.bcdByteToInt(pdu[cur++]);
836                int second = SimUtils.bcdByteToInt(pdu[cur++]);
837    
838                // For the timezone, the most significant bit of the
839                // least signficant nibble is the sign byte
840                // (meaning the max range of this field is 79 quarter-hours,
841                // which is more than enough)
842    
843                byte tzByte = pdu[cur++];
844    
845                // Mask out sign bit.
846                int timezoneOffset = SimUtils
847                        .bcdByteToInt((byte) (tzByte & (~0x08)));
848    
849                timezoneOffset = ((tzByte & 0x08) == 0) ? timezoneOffset
850                        : -timezoneOffset;
851    
852                Time time = new Time(Time.TIMEZONE_UTC);
853    
854                // It's 2006.  Should I really support years < 2000?
855                time.year = year >= 90 ? year + 1900 : year + 2000;
856                time.month = month - 1;
857                time.monthDay = day;
858                time.hour = hour;
859                time.minute = minute;
860                time.second = second;
861    
862                // Timezone offset is in quarter hours.
863                return time.toMillis(true) - (timezoneOffset * 15 * 60 * 1000);
864            }

866            /**
867             * Pulls the user data out of the PDU, and separates the payload from
868             * the header if there is one.
869             *
870             * @param hasUserDataHeader true if there is a user data header
871             * @param dataInSeptets true if the data payload is in septets instead
872             *  of octets
873             * @return the number of septets or octets in the user data payload
874             */
875            int constructUserData(boolean hasUserDataHeader, boolean dataInSeptets)
876            {
877                int offset = cur;
878                int userDataLength = pdu[offset++] & 0xff;
879                int headerSeptets = 0;
880    
881                if (hasUserDataHeader) {
882                    int userDataHeaderLength = pdu[offset++] & 0xff;
883    
884                    byte[] udh = new byte[userDataHeaderLength];
885                    System.arraycopy(pdu, offset, udh, 0, userDataHeaderLength);
886                    userDataHeader = SmsHeader.parse(udh);
887                    offset += userDataHeaderLength;
888    
889                    int headerBits = (userDataHeaderLength + 1) * 8;
890                    headerSeptets = headerBits / 7;
891                    headerSeptets += (headerBits % 7) > 0 ? 1 : 0;
892                    mUserDataSeptetPadding = (headerSeptets * 7) - headerBits;
893                }
894    
895                /*
896                 * Here we just create the user data length to be the remainder of
897                 * the pdu minus the user data hearder. This is because the count
898                 * could mean the number of uncompressed sepets if the userdata is
899                 * encoded in 7-bit.
900                 */
901                userData = new byte[pdu.length - offset];
902                System.arraycopy(pdu, offset, userData, 0, userData.length);
903                cur = offset;
904    
905                if (dataInSeptets) {
906                    // Return the number of septets
907                    int count = userDataLength - headerSeptets;
908                    // If count < 0, return 0 (means UDL was probably incorrect)
909                    return count < 0 ? 0 : count;
910                } else {
911                    // Return the number of octets
912                    return userData.length;
913                }
914            }

1300        /**
1301         * TS 27.005 3.1, <pdu> definition "In the case of SMS: 3GPP TS 24.011 [6]
1302         * SC address followed by 3GPP TS 23.040 [3] TPDU in hexadecimal format:
1303         * ME/TA converts each octet of TP data unit into two IRA character long
1304         * hexad number (e.g. octet with integer value 42 is presented to TE as two
1305         * characters 2A (IRA 50 and 65))" ...in the case of cell broadcast,
1306         * something else...
1307         */
1308        private void parsePdu(byte[] pdu) {
1309            mPdu = pdu;
1310            // Log.d(LOG_TAG, "raw sms mesage:");
1311            // Log.d(LOG_TAG, s);
1312    
1313            PduParser p = new PduParser(pdu);
1314    
1315            scAddress = p.getSCAddress();
1316    
1317            if (scAddress != null) {
1318                if (Config.LOGD) Log.d(LOG_TAG, "SMS SC address: " + scAddress);
1319            }
1320    
1321            // TODO(mkf) support reply path, user data header indicator
1322    
1323            // TP-Message-Type-Indicator
1324            // 9.2.3
1325            int firstByte = p.getByte();
1326    
1327            mti = firstByte & 0x3;
1328            switch (mti) {
1329            // TP-Message-Type-Indicator
1330            // 9.2.3
1331            case 0:
1332                parseSmsDeliver(p, firstByte);
1333                break;
1334            case 2:
1335                parseSmsStatusReport(p, firstByte);
1336                break;
1337            default:
1338                // TODO(mkf) the rest of these
1339                throw new RuntimeException("Unsupported message type");
1340            }
1341        }

1343        /**
1344         * Parses a SMS-STATUS-REPORT message.
1345         *
1346         * @param p A PduParser, cued past the first byte.
1347         * @param firstByte The first byte of the PDU, which contains MTI, etc.
1348         */
1349        private void parseSmsStatusReport(PduParser p, int firstByte) {
1350            isStatusReportMessage = true;
1351    
1352            // TP-Status-Report-Qualifier bit == 0 for SUBMIT
1353            forSubmit = (firstByte & 0x20) == 0x00;
1354            // TP-Message-Reference
1355            messageRef = p.getByte();
1356            // TP-Recipient-Address
1357            recipientAddress = p.getAddress();
1358            // TP-Service-Centre-Time-Stamp
1359            scTimeMillis = p.getSCTimestampMillis();
1360            // TP-Discharge-Time
1361            dischargeTimeMillis = p.getSCTimestampMillis();
1362            // TP-Status
1363            status = p.getByte();
1364    
1365            // The following are optional fields that may or may not be present.
1366            if (p.moreDataPresent()) {
1367                // TP-Parameter-Indicator
1368                int extraParams = p.getByte();
1369                int moreExtraParams = extraParams;
1370                while ((moreExtraParams & 0x80) != 0) {
1371                    // We only know how to parse a few extra parameters, all
1372                    // indicated in the first TP-PI octet, so skip over any
1373                    // additional TP-PI octets.
1374                    moreExtraParams = p.getByte();
1375                }
1376                // TP-Protocol-Identifier
1377                if ((extraParams & 0x01) != 0) {
1378                    protocolIdentifier = p.getByte();
1379                }
1380                // TP-Data-Coding-Scheme
1381                if ((extraParams & 0x02) != 0) {
1382                    dataCodingScheme = p.getByte();
1383                }
1384                // TP-User-Data-Length (implies existence of TP-User-Data)
1385                if ((extraParams & 0x04) != 0) {
1386                    boolean hasUserDataHeader = (firstByte & 0x40) == 0x40;
1387                    parseUserData(p, hasUserDataHeader);
1388                }
1389            }
1390        }

1392        private void parseSmsDeliver(PduParser p, int firstByte) {
1393            replyPathPresent = (firstByte & 0x80) == 0x80;
1394    
1395            originatingAddress = p.getAddress();
1396    
1397            if (originatingAddress != null) {
1398                if (Config.LOGV) Log.v(LOG_TAG, "SMS originating address: "
1399                        + originatingAddress.address);
1400            }
1401    
1402            // TP-Protocol-Identifier (TP-PID)
1403            // TS 23.040 9.2.3.9
1404            protocolIdentifier = p.getByte();
1405    
1406            // TP-Data-Coding-Scheme
1407            // see TS 23.038
1408            dataCodingScheme = p.getByte();
1409    
1410            if (Config.LOGV) {
1411                Log.v(LOG_TAG, "SMS TP-PID:" + protocolIdentifier
1412                        + " data coding scheme: " + dataCodingScheme);
1413            }
1414    
1415            scTimeMillis = p.getSCTimestampMillis();
1416    
1417            if (Config.LOGD) Log.d(LOG_TAG, "SMS SC timestamp: " + scTimeMillis);
1418    
1419            boolean hasUserDataHeader = (firstByte & 0x40) == 0x40;
1420    
1421            parseUserData(p, hasUserDataHeader);
1422        }

1424        /**
1425         * Parses the User Data of an SMS.
1426         *
1427         * @param p The current PduParser.
1428         * @param hasUserDataHeader Indicates whether a header is present in the
1429         *                          User Data.
1430         */
1431        private void parseUserData(PduParser p, boolean hasUserDataHeader) {
1432            boolean hasMessageClass = false;
1433            boolean userDataCompressed = false;
1434    
1435            int encodingType = ENCODING_UNKNOWN;
1436    
1437            // Look up the data encoding scheme
1438            if ((dataCodingScheme & 0x80) == 0) {
1439                // Bits 7..4 == 0xxx
1440                automaticDeletion = (0 != (dataCodingScheme & 0x40));
1441                userDataCompressed = (0 != (dataCodingScheme & 0x20));
1442                hasMessageClass = (0 != (dataCodingScheme & 0x10));
1443    
1444                if (userDataCompressed) {
1445                    Log.w(LOG_TAG, "4 - Unsupported SMS data coding scheme "
1446                            + "(compression) " + (dataCodingScheme & 0xff));
1447                } else {
1448                    switch ((dataCodingScheme >> 2) & 0x3) {
1449                    case 0: // GSM 7 bit default alphabet
1450                        encodingType = ENCODING_7BIT;
1451                        break;
1452    
1453                    case 2: // UCS 2 (16bit)
1454                        encodingType = ENCODING_16BIT;
1455                        break;
1456    
1457                    case 1: // 8 bit data
1458                    case 3: // reserved
1459                        Log.w(LOG_TAG, "1 - Unsupported SMS data coding scheme "
1460                                + (dataCodingScheme & 0xff));
1461                        encodingType = ENCODING_8BIT;
1462                        break;
1463                    }
1464                }
1465            } else if ((dataCodingScheme & 0xf0) == 0xf0) {
1466                automaticDeletion = false;
1467                hasMessageClass = true;
1468                userDataCompressed = false;
1469    
1470                if (0 == (dataCodingScheme & 0x04)) {
1471                    // GSM 7 bit default alphabet
1472                    encodingType = ENCODING_7BIT;
1473                } else {
1474                    // 8 bit data
1475                    encodingType = ENCODING_8BIT;
1476                }
1477            } else if ((dataCodingScheme & 0xF0) == 0xC0
1478                    || (dataCodingScheme & 0xF0) == 0xD0
1479                    || (dataCodingScheme & 0xF0) == 0xE0) {
1480                // 3GPP TS 23.038 V7.0.0 (2006-03) section 4
1481    
1482                // 0xC0 == 7 bit, don't store
1483                // 0xD0 == 7 bit, store
1484                // 0xE0 == UCS-2, store
1485    
1486                if ((dataCodingScheme & 0xF0) == 0xE0) {
1487                    encodingType = ENCODING_16BIT;
1488                } else {
1489                    encodingType = ENCODING_7BIT;
1490                }
1491    
1492                userDataCompressed = false;
1493                boolean active = ((dataCodingScheme & 0x08) == 0x08);
1494    
1495                // bit 0x04 reserved
1496    
1497                if ((dataCodingScheme & 0x03) == 0x00) {
1498                    isMwi = true;
1499                    mwiSense = active;
1500                    mwiDontStore = ((dataCodingScheme & 0xF0) == 0xC0);
1501                } else {
1502                    isMwi = false;
1503    
1504                    Log.w(LOG_TAG, "MWI for fax, email, or other "
1505                            + (dataCodingScheme & 0xff));
1506                }
1507            } else {
1508                Log.w(LOG_TAG, "3 - Unsupported SMS data coding scheme "
1509                        + (dataCodingScheme & 0xff));
1510            }
1511    
1512            // set both the user data and the user data header.
1513            int count = p.constructUserData(hasUserDataHeader,
1514                    encodingType == ENCODING_7BIT);
1515            this.userData = p.getUserData();
1516            this.userDataHeader = p.getUserDataHeader();
1517    
1518            switch (encodingType) {
1519            case ENCODING_UNKNOWN:
1520            case ENCODING_8BIT:
1521                messageBody = null;
1522                break;
1523    
1524            case ENCODING_7BIT:
1525                messageBody = p.getUserDataGSM7Bit(count);
1526                break;
1527    
1528            case ENCODING_16BIT:
1529                messageBody = p.getUserDataUCS2(count);
1530                break;
1531            }
1532    
1533            if (Config.LOGV) Log.v(LOG_TAG, "SMS message body (raw): '" + messageBody + "'");
1534    
1535            if (messageBody != null) {
1536                parseMessageBody();
1537            }
1538    
1539            if (!hasMessageClass) {
1540                messageClass = MessageClass.UNKNOWN;
1541            } else {
1542                switch (dataCodingScheme & 0x3) {
1543                case 0:
1544                    messageClass = MessageClass.CLASS_0;
1545                    break;
1546                case 1:
1547                    messageClass = MessageClass.CLASS_1;
1548                    break;
1549                case 2:
1550                    messageClass = MessageClass.CLASS_2;
1551                    break;
1552                case 3:
1553                    messageClass = MessageClass.CLASS_3;
1554                    break;
1555                }
1556            }
1557        }

1560            if (originatingAddress.couldBeEmailGateway()) {
1561                extractEmailAddressFromMessageBody();

1565        /**
1566         * Try to parse this message as an email gateway message -> Neither
1567         * of the standard ways are currently supported: There are two ways
1568         * specified in TS 23.040 Section 3.8 (not supported via this mechanism) -
1569         * SMS message "may have its TP-PID set for internet electronic mail - MT
1570         * SMS format: [<from-address><space>]<message> - "Depending on the
1571         * nature of the gateway, the destination/origination address is either
1572         * derived from the content of the SMS TP-OA or TP-DA field, or the
1573         * TP-OA/TP-DA field contains a generic gateway address and the to/from
1574         * address is added at the beginning as shown above." - multiple addreses
1575         * separated by commas, no spaces - subject field delimited by '()' or '##'
1576         * and '#' Section 9.2.3.24.11
1577         */
1578        private void extractEmailAddressFromMessageBody() {
1579    
1580            /*
1581             * a little guesswork here. I haven't found doc for this.
1582             * the format could be either
1583             *
1584             * 1. [x@y][ ]/[subject][ ]/[body]
1585             * -or-
1586             * 2. [x@y][ ]/[body]
1587             */
1588            int slash = 0, slash2 = 0, atSymbol = 0;
1589    
1590            try {
1591                slash = messageBody.indexOf(" /");
1592                if (slash == -1) {
1593                    return;
1594                }
1595    
1596                atSymbol = messageBody.indexOf('@');
1597                if (atSymbol == -1 || atSymbol > slash) {
1598                    return;
1599                }
1600    
1601                emailFrom = messageBody.substring(0, slash);
1602    
1603                slash2 = messageBody.indexOf(" /", slash + 2);
1604    
1605                if (slash2 == -1) {
1606                    pseudoSubject = null;
1607                    emailBody = messageBody.substring(slash + 2);
1608                } else {
1609                    pseudoSubject = messageBody.substring(slash + 2, slash2);
1610                    emailBody = messageBody.substring(slash2 + 2);
1611                }
1612    
1613                isEmail = true;
1614            } catch (Exception ex) {
1615                Log.w(LOG_TAG,
1616                        "extractEmailAddressFromMessageBody: exception slash="
1617                        + slash + ", atSymbol=" + atSymbol + ", slash2="
1618                        + slash2, ex);
1619            }
1620        }
