043        /**
044         * Send a text based SMS.
045         *
046         * @param destinationAddress the address to send the message to
047         * @param scAddress is the service center address or null to use
048         *  the current default SMSC
049         * @param text the body of the message to send
050         * @param sentIntent if not NULL this <code>PendingIntent</code> is
051         *  broadcast when the message is sucessfully sent, or failed.
052         *  The result code will be <code>Activity.RESULT_OK<code> for success,
053         *  or one of these errors:
054         *  <code>RESULT_ERROR_GENERIC_FAILURE</code>
055         *  <code>RESULT_ERROR_RADIO_OFF</code>
056         *  <code>RESULT_ERROR_NULL_PDU</code>.
057         *  The per-application based SMS control checks sentIntent. If sentIntent
058         *  is NULL the caller will be checked against all unknown applicaitons,
059         *  which cause smaller number of SMS to be sent in checking period.
060         * @param deliveryIntent if not NULL this <code>PendingIntent</code> is
061         *  broadcast when the message is delivered to the recipient.  The
062         *  raw pdu of the status report is in the extended data ("pdu").
063         *
064         * @throws IllegalArgumentException if destinationAddress or text are empty
065         */
066        public void sendTextMessage(
067                String destinationAddress, String scAddress, String text,
068                PendingIntent sentIntent, PendingIntent deliveryIntent) {
069            if (TextUtils.isEmpty(destinationAddress)) {
070                throw new IllegalArgumentException("Invalid destinationAddress");
071            }
072    
073            if (TextUtils.isEmpty(text)) {
074                throw new IllegalArgumentException("Invalid message body");
075            }
076    
077            SmsMessage.SubmitPdu pdus = SmsMessage.getSubmitPdu(
078                    scAddress, destinationAddress, text, (deliveryIntent != null));
079            sendRawPdu(pdus.encodedScAddress, pdus.encodedMessage, sentIntent, deliveryIntent);
080        }

082        /**
083         * Divide a text message into several messages, none bigger than
084         * the maximum SMS message size.
085         *
086         * @param text the original message.  Must not be null.
087         * @return an <code>ArrayList</code> of strings that, in order,
088         *   comprise the original message
089         */
090        public ArrayList<String> divideMessage(String text) {
091            int size = text.length();
092            int[] params = SmsMessage.calculateLength(text, false);
093                /* SmsMessage.calculateLength returns an int[4] with:
094                 *   int[0] being the number of SMS's required,
095                 *   int[1] the number of code units used,
096                 *   int[2] is the number of code units remaining until the next message.
097                 *   int[3] is the encoding type that should be used for the message.
098                 */
099            int messageCount = params[0];
100            int encodingType = params[3];
101            ArrayList<String> result = new ArrayList<String>(messageCount);
102    
103            int start = 0;
104            int limit;
105            
106            if (messageCount > 1) {
107                limit = (encodingType == SmsMessage.ENCODING_7BIT) ?
108                        SmsMessage.MAX_USER_DATA_SEPTETS_WITH_HEADER :
109                            SmsMessage.MAX_USER_DATA_BYTES_WITH_HEADER;            
110            } else {
111                limit = (encodingType == SmsMessage.ENCODING_7BIT) ?
112                    SmsMessage.MAX_USER_DATA_SEPTETS : SmsMessage.MAX_USER_DATA_BYTES;            
113            }
114    
115            try {
116                while (start < size) {
117                    int end = GsmAlphabet.findLimitIndex(text, start, limit, encodingType);
118                    result.add(text.substring(start, end));
119                    start = end;
120                }
121            } catch (EncodeException e) {
122                // ignore it.
123            }
124            return result;
125        }

127        /**
128         * Send a multi-part text based SMS.  The callee should have already
129         * divided the message into correctly sized parts by calling
130         * <code>divideMessage</code>.
131         * 
132         * @param destinationAddress the address to send the message to
133         * @param scAddress is the service center address or null to use
134         *   the current default SMSC
135         * @param parts an <code>ArrayList</code> of strings that, in order,
136         *   comprise the original message
137         * @param sentIntents if not null, an <code>ArrayList</code> of 
138         *   <code>PendingIntent</code>s (one for each message part) that is
139         *   broadcast when the corresponding message part has been sent.
140         *   The result code will be <code>Activity.RESULT_OK<code> for success,
141         *   or one of these errors:
142         *   <code>RESULT_ERROR_GENERIC_FAILURE</code>
143         *   <code>RESULT_ERROR_RADIO_OFF</code>
144         *   <code>RESULT_ERROR_NULL_PDU</code>.
145         *   The per-application based SMS control checks sentIntent. If sentIntent
146         *   is NULL the caller will be checked against all unknown applicaitons,
147         *   which cause smaller number of SMS to be sent in checking period.
148         * @param deliveryIntents if not null, an <code>ArrayList</code> of 
149         *   <code>PendingIntent</code>s (one for each message part) that is
150         *   broadcast when the corresponding message part has been delivered
151         *   to the recipient.  The raw pdu of the status report is in the
152         *   extended data ("pdu").
153         */
154        public void sendMultipartTextMessage(
155                String destinationAddress, String scAddress, ArrayList<String> parts,
156                ArrayList<PendingIntent> sentIntents, ArrayList<PendingIntent> deliveryIntents) {
157            if (TextUtils.isEmpty(destinationAddress)) {
158                throw new IllegalArgumentException("Invalid destinationAddress");
159            }
160            if (parts == null || parts.size() < 1) {
161                throw new IllegalArgumentException("Invalid message body");
162            }
163            
164            if (parts.size() > 1) {
165                try {
166                    ISms simISms = ISms.Stub.asInterface(ServiceManager.getService("isms"));
167                    if (simISms != null) {
168                        simISms.sendMultipartText(destinationAddress, scAddress, parts,
169                                sentIntents, deliveryIntents);
170                    }
171                } catch (RemoteException ex) {
172                    // ignore it
173                }
174            } else {
175                PendingIntent sentIntent = null;
176                PendingIntent deliveryIntent = null;
177                if (sentIntents != null && sentIntents.size() > 0) {
178                    sentIntent = sentIntents.get(0);
179                }
180                if (deliveryIntents != null && deliveryIntents.size() > 0) {
181                    deliveryIntent = deliveryIntents.get(0);
182                }
183                sendTextMessage(destinationAddress, scAddress, parts.get(0),
184                        sentIntent, deliveryIntent);
185            }
186        }

189         * Send a data based SMS to a specific application port.
190         *
191         * @param destinationAddress the address to send the message to
192         * @param scAddress is the service center address or null to use
193         *  the current default SMSC
194         * @param destinationPort the port to deliver the message to
195         * @param data the body of the message to send

198         *  The result code will be <code>Activity.RESULT_OK<code> for success,
199         *  or one of these errors:
200         *  <code>RESULT_ERROR_GENERIC_FAILURE</code>
201         *  <code>RESULT_ERROR_RADIO_OFF</code>
202         *  <code>RESULT_ERROR_NULL_PDU</code>.
203         *  The per-application based SMS control checks sentIntent. If sentIntent
204         *  is NULL the caller will be checked against all unknown applicaitons,
205         *  which cause smaller number of SMS to be sent in checking period.
206         * @param deliveryIntent if not NULL this <code>PendingIntent</code> is
207         *  broadcast when the message is delivered to the recipient.  The
208         *  raw pdu of the status report is in the extended data ("pdu").
209         *
210         * @throws IllegalArgumentException if destinationAddress or data are empty
211         */
212        public void sendDataMessage(
213                String destinationAddress, String scAddress, short destinationPort,
214                byte[] data, PendingIntent sentIntent, PendingIntent deliveryIntent) {
215            if (TextUtils.isEmpty(destinationAddress)) {
216                throw new IllegalArgumentException("Invalid destinationAddress");

218    
219            if (data == null || data.length == 0) {
220                throw new IllegalArgumentException("Invalid message data");
221            }

223            SmsMessage.SubmitPdu pdus = SmsMessage.getSubmitPdu(scAddress, destinationAddress,
224                    destinationPort, data, (deliveryIntent != null));
225            sendRawPdu(pdus.encodedScAddress, pdus.encodedMessage, sentIntent, deliveryIntent);
226        }

304         * Delete the specified message from the SIM.
305         *
306         * @param messageIndex is the record index of the message on SIM
307         * @return true for success
308         *
309         * {@hide}
310         */
311        public boolean
312        deleteMessageFromSim(int messageIndex) {
313            boolean success = false;
314            byte[] pdu = new byte[SimConstants.SMS_RECORD_LENGTH-1];
315            Arrays.fill(pdu, (byte)0xff);
316    
317            try {
318                ISms simISms = ISms.Stub.asInterface(ServiceManager.getService("isms"));
319                if (simISms != null) {
320                    success = simISms.updateMessageOnSimEf(messageIndex,
321                            STATUS_ON_SIM_FREE, pdu);
322                }
323            } catch (RemoteException ex) {
324                // ignore it
325            }
326    
327            return success;
328        }

359        /**
360         * Retrieves all messages currently stored on SIM.
361         *
362         * @return <code>ArrayList</code> of <code>SmsMessage</code> objects
363         *
364         * {@hide}
365         */
366        public ArrayList<SmsMessage> getAllMessagesFromSim() {
367            List<SmsRawData> records = null;
368    
369            try {
370                ISms simISms = ISms.Stub.asInterface(ServiceManager.getService("isms"));
371                if (simISms != null) {
372                    records = simISms.getAllMessagesFromSimEf();
373                }
374            } catch (RemoteException ex) {
375                // ignore it
376            }
377            
378            return createMessageListFromRawRecords(records); 
379       }
380    
381        /**
382         * Create a list of <code>SmsMessage</code>s from a list of RawSmsData
383         * records returned by <code>getAllMessagesFromSim()</code>
384         *
385         * @param records SMS EF records, returned by
386         *   <code>getAllMessagesFromSim</code>
387         * @return <code>ArrayList</code> of <code>SmsMessage</code> objects.
388         */
389        private ArrayList<SmsMessage> createMessageListFromRawRecords(List records) {
390            ArrayList<SmsMessage> messages = new ArrayList<SmsMessage>();
391            if (records != null) {
392                int count = records.size();
393                for (int i = 0; i < count; i++) {
394                    SmsRawData data = (SmsRawData)records.get(i);
395                    // List contains all records, including "free" records (null)
396                    if (data != null) {
397                        SmsMessage sms =
398                                SmsMessage.createFromEfRecord(i+1, data.getBytes());
399                        messages.add(sms);
400                    }
401                }
402            }
403            return messages;
404        }
