025    /**
026     * Objects implementing the <code>NamedNodeMap</code> interface are used to 
027     * represent collections of nodes that can be accessed by name. Note that 
028     * <code>NamedNodeMap</code> does not inherit from <code>NodeList</code>; 
029     * <code>NamedNodeMaps</code> are not maintained in any particular order. 
030     * Objects contained in an object implementing <code>NamedNodeMap</code> may 
031     * also be accessed by an ordinal index, but this is simply to allow 
032     * convenient enumeration of the contents of a <code>NamedNodeMap</code>, 
033     * and does not imply that the DOM specifies an order to these Nodes. 
034     * <p><code>NamedNodeMap</code> objects in the DOM are live.
035     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
036     */
037    public interface NamedNodeMap {
038        /**
039         * Retrieves a node specified by name.
040         * @param name The <code>nodeName</code> of a node to retrieve.
041         * @return A <code>Node</code> (of any type) with the specified 
042         *   <code>nodeName</code>, or <code>null</code> if it does not identify 
043         *   any node in this map.
044         */
045        public Node getNamedItem(String name);
046    
047        /**
048         * Adds a node using its <code>nodeName</code> attribute. If a node with 
049         * that name is already present in this map, it is replaced by the new 
050         * one.
051         * <br>As the <code>nodeName</code> attribute is used to derive the name 
052         * which the node must be stored under, multiple nodes of certain types 
053         * (those that have a "special" string value) cannot be stored as the 
054         * names would clash. This is seen as preferable to allowing nodes to be 
055         * aliased.
056         * @param arg A node to store in this map. The node will later be 
057         *   accessible using the value of its <code>nodeName</code> attribute.
058         * @return If the new <code>Node</code> replaces an existing node the 
059         *   replaced <code>Node</code> is returned, otherwise <code>null</code> 
060         *   is returned.
061         * @exception DOMException
062         *   WRONG_DOCUMENT_ERR: Raised if <code>arg</code> was created from a 
063         *   different document than the one that created this map.
064         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this map is readonly.
065         *   <br>INUSE_ATTRIBUTE_ERR: Raised if <code>arg</code> is an 
066         *   <code>Attr</code> that is already an attribute of another 
067         *   <code>Element</code> object. The DOM user must explicitly clone 
068         *   <code>Attr</code> nodes to re-use them in other elements.
069         */
070        public Node setNamedItem(Node arg)
071                                 throws DOMException;
072    
073        /**
074         * Removes a node specified by name. When this map contains the attributes 
075         * attached to an element, if the removed attribute is known to have a 
076         * default value, an attribute immediately appears containing the 
077         * default value as well as the corresponding namespace URI, local name, 
078         * and prefix when applicable.
079         * @param name The <code>nodeName</code> of the node to remove.
080         * @return The node removed from this map if a node with such a name 
081         *   exists.
082         * @exception DOMException
083         *   NOT_FOUND_ERR: Raised if there is no node named <code>name</code> in 
084         *   this map.
085         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this map is readonly.
086         */
087        public Node removeNamedItem(String name)
088                                    throws DOMException;
089    
090        /**
091         * Returns the <code>index</code>th item in the map. If <code>index</code> 
092         * is greater than or equal to the number of nodes in this map, this 
093         * returns <code>null</code>.
094         * @param index Index into this map.
095         * @return The node at the <code>index</code>th position in the map, or 
096         *   <code>null</code> if that is not a valid index.
097         */
098        public Node item(int index);
099    
100        /**
101         * The number of nodes in this map. The range of valid child node indices 
102         * is <code>0</code> to <code>length-1</code> inclusive.
103         * 
104         * @return the length of the map.
105         */
106        public int getLength();
107    
108        /**
109         * Retrieves a node specified by local name and namespace URI. HTML-only 
110         * DOM implementations do not need to implement this method.
111         * @param namespaceURI The namespace URI of the node to retrieve.
112         * @param localName The local name of the node to retrieve.
113         * @return A <code>Node</code> (of any type) with the specified local 
114         *   name and namespace URI, or <code>null</code> if they do not 
115         *   identify any node in this map.
116         * @since DOM Level 2
117         */
118        public Node getNamedItemNS(String namespaceURI, 
119                                   String localName);
120    
121        /**
122         * Adds a node using its <code>namespaceURI</code> and 
123         * <code>localName</code>. If a node with that namespace URI and that 
124         * local name is already present in this map, it is replaced by the new 
125         * one.
126         * <br>HTML-only DOM implementations do not need to implement this method.
127         * @param arg A node to store in this map. The node will later be 
128         *   accessible using the value of its <code>namespaceURI</code> and 
129         *   <code>localName</code> attributes.
130         * @return If the new <code>Node</code> replaces an existing node the 
131         *   replaced <code>Node</code> is returned, otherwise <code>null</code> 
132         *   is returned.
133         * @exception DOMException
134         *   WRONG_DOCUMENT_ERR: Raised if <code>arg</code> was created from a 
135         *   different document than the one that created this map.
136         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this map is readonly.
137         *   <br>INUSE_ATTRIBUTE_ERR: Raised if <code>arg</code> is an 
138         *   <code>Attr</code> that is already an attribute of another 
139         *   <code>Element</code> object. The DOM user must explicitly clone 
140         *   <code>Attr</code> nodes to re-use them in other elements.
141         * @since DOM Level 2
142         */
143        public Node setNamedItemNS(Node arg)
144                                   throws DOMException;
145    
146        /**
147         * Removes a node specified by local name and namespace URI. A removed 
148         * attribute may be known to have a default value when this map contains 
149         * the attributes attached to an element, as returned by the attributes 
150         * attribute of the <code>Node</code> interface. If so, an attribute 
151         * immediately appears containing the default value as well as the 
152         * corresponding namespace URI, local name, and prefix when applicable.
153         * <br>HTML-only DOM implementations do not need to implement this method.
154         * @param namespaceURI The namespace URI of the node to remove.
155         * @param localName The local name of the node to remove.
156         * @return The node removed from this map if a node with such a local 
157         *   name and namespace URI exists.
158         * @exception DOMException
159         *   NOT_FOUND_ERR: Raised if there is no node with the specified 
160         *   <code>namespaceURI</code> and <code>localName</code> in this map.
161         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this map is readonly.
162         * @since DOM Level 2
163         */
164        public Node removeNamedItemNS(String namespaceURI, 
165                                      String localName)
166                                      throws DOMException;
167    
168    }
