025    /**
026     * The <code>Document</code> interface represents the entire HTML or XML 
027     * document. Conceptually, it is the root of the document tree, and provides 
028     * the primary access to the document's data.
029     * <p>Since elements, text nodes, comments, processing instructions, etc. 
030     * cannot exist outside the context of a <code>Document</code>, the 
031     * <code>Document</code> interface also contains the factory methods needed 
032     * to create these objects. The <code>Node</code> objects created have a 
033     * <code>ownerDocument</code> attribute which associates them with the 
034     * <code>Document</code> within whose context they were created.
035     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
036     */
037    public interface Document extends Node {
038        /**
039         * The Document Type Declaration (see <code>DocumentType</code>) 
040         * associated with this document. For HTML documents as well as XML 
041         * documents without a document type declaration this returns 
042         * <code>null</code>. The DOM Level 2 does not support editing the 
043         * Document Type Declaration. <code>docType</code> cannot be altered in 
044         * any way, including through the use of methods inherited from the 
045         * <code>Node</code> interface, such as <code>insertNode</code> or 
046         * <code>removeNode</code>.
047         * 
048         * @return the Document Type Declaration associated with this document, if
049         * any.
050         */
051        public DocumentType getDoctype();
052    
053        /**
054         * The <code>DOMImplementation</code> object that handles this document. A 
055         * DOM application may use objects from multiple implementations.
056         * 
057         * @return <code>DOMImplementation</code> object that handles this document.
058         */
059        public DOMImplementation getImplementation();
060    
061        /**
062         * This is a convenience attribute that allows direct access to the child 
063         * node that is the root element of the document. For HTML documents, 
064         * this is the element with the tagName "HTML".
065         * 
066         * @return the root element of this document. 
067         */
068        public Element getDocumentElement();
069    
070        /**
071         * Creates an element of the type specified. Note that the instance 
072         * returned implements the <code>Element</code> interface, so attributes 
073         * can be specified directly on the returned object.
074         * <br>In addition, if there are known attributes with default values, 
075         * <code>Attr</code> nodes representing them are automatically created 
076         * and attached to the element.
077         * <br>To create an element with a qualified name and namespace URI, use 
078         * the <code>createElementNS</code> method.
079         * @param tagName The name of the element type to instantiate. For XML, 
080         *   this is case-sensitive. For HTML, the <code>tagName</code> 
081         *   parameter may be provided in any case, but it must be mapped to the 
082         *   canonical uppercase form by the DOM implementation. 
083         * @return A new <code>Element</code> object with the 
084         *   <code>nodeName</code> attribute set to <code>tagName</code>, and 
085         *   <code>localName</code>, <code>prefix</code>, and 
086         *   <code>namespaceURI</code> set to <code>null</code>.
087         * @exception DOMException
088         *   INVALID_CHARACTER_ERR: Raised if the specified name contains an 
089         *   illegal character.
090         */
091        public Element createElement(String tagName)
092                                     throws DOMException;
093    
094        /**
095         * Creates an empty <code>DocumentFragment</code> object. 
096         * @return A new <code>DocumentFragment</code>.
097         */
098        public DocumentFragment createDocumentFragment();
099    
100        /**
101         * Creates a <code>Text</code> node given the specified string.
102         * @param data The data for the node.
103         * @return The new <code>Text</code> object.
104         */
105        public Text createTextNode(String data);
106    
107        /**
108         * Creates a <code>Comment</code> node given the specified string.
109         * @param data The data for the node.
110         * @return The new <code>Comment</code> object.
111         */
112        public Comment createComment(String data);
113    
114        /**
115         * Creates a <code>CDATASection</code> node whose value is the specified 
116         * string.
117         * @param data The data for the <code>CDATASection</code> contents.
118         * @return The new <code>CDATASection</code> object.
119         * @exception DOMException
120         *   NOT_SUPPORTED_ERR: Raised if this document is an HTML document.
121         */
122        public CDATASection createCDATASection(String data)
123                                               throws DOMException;
124    
125        /**
126         * Creates a <code>ProcessingInstruction</code> node given the specified 
127         * name and data strings.
128         * @param target The target part of the processing instruction.
129         * @param data The data for the node.
130         * @return The new <code>ProcessingInstruction</code> object.
131         * @exception DOMException
132         *   INVALID_CHARACTER_ERR: Raised if the specified target contains an 
133         *   illegal character.
134         *   <br>NOT_SUPPORTED_ERR: Raised if this document is an HTML document.
135         */
136        public ProcessingInstruction createProcessingInstruction(String target, 
137                                                                 String data)
138                                                                 throws DOMException;
139    
140        /**
141         * Creates an <code>Attr</code> of the given name. Note that the 
142         * <code>Attr</code> instance can then be set on an <code>Element</code> 
143         * using the <code>setAttributeNode</code> method. 
144         * <br>To create an attribute with a qualified name and namespace URI, use 
145         * the <code>createAttributeNS</code> method.
146         * @param name The name of the attribute.
147         * @return A new <code>Attr</code> object with the <code>nodeName</code> 
148         *   attribute set to <code>name</code>, and <code>localName</code>, 
149         *   <code>prefix</code>, and <code>namespaceURI</code> set to 
150         *   <code>null</code>. The value of the attribute is the empty string.
151         * @exception DOMException
152         *   INVALID_CHARACTER_ERR: Raised if the specified name contains an 
153         *   illegal character.
154         */
155        public Attr createAttribute(String name)
156                                    throws DOMException;
157    
158        /**
159         * Creates an <code>EntityReference</code> object. In addition, if the 
160         * referenced entity is known, the child list of the 
161         * <code>EntityReference</code> node is made the same as that of the 
162         * corresponding <code>Entity</code> node.If any descendant of the 
163         * <code>Entity</code> node has an unbound namespace prefix, the 
164         * corresponding descendant of the created <code>EntityReference</code> 
165         * node is also unbound; (its <code>namespaceURI</code> is 
166         * <code>null</code>). The DOM Level 2 does not support any mechanism to 
167         * resolve namespace prefixes.
168         * @param name The name of the entity to reference. 
169         * @return The new <code>EntityReference</code> object.
170         * @exception DOMException
171         *   INVALID_CHARACTER_ERR: Raised if the specified name contains an 
172         *   illegal character.
173         *   <br>NOT_SUPPORTED_ERR: Raised if this document is an HTML document.
174         */
175        public EntityReference createEntityReference(String name)
176                                                     throws DOMException;
177    
178        /**
179         * Returns a <code>NodeList</code> of all the <code>Elements</code> with a 
180         * given tag name in the order in which they are encountered in a 
181         * preorder traversal of the <code>Document</code> tree. 
182         * @param tagname The name of the tag to match on. The special value "*" 
183         *   matches all tags.
184         * @return A new <code>NodeList</code> object containing all the matched 
185         *   <code>Elements</code>.
186         */
187        public NodeList getElementsByTagName(String tagname);
188    
189        /**
190         * Imports a node from another document to this document. The returned 
191         * node has no parent; (<code>parentNode</code> is <code>null</code>). 
192         * The source node is not altered or removed from the original document; 
193         * this method creates a new copy of the source node.
194         * <br>For all nodes, importing a node creates a node object owned by the 
195         * importing document, with attribute values identical to the source 
196         * node's <code>nodeName</code> and <code>nodeType</code>, plus the 
197         * attributes related to namespaces (<code>prefix</code>, 
198         * <code>localName</code>, and <code>namespaceURI</code>). As in the 
199         * <code>cloneNode</code> operation on a <code>Node</code>, the source 
200         * node is not altered.
201         * <br>Additional information is copied as appropriate to the 
202         * <code>nodeType</code>, attempting to mirror the behavior expected if 
203         * a fragment of XML or HTML source was copied from one document to 
204         * another, recognizing that the two documents may have different DTDs 
205         * in the XML case. The following list describes the specifics for each 
206         * type of node. 
207         * <dl>
208         * <dt>ATTRIBUTE_NODE</dt>
209         * <dd>The <code>ownerElement</code> attribute 
210         * is set to <code>null</code> and the <code>specified</code> flag is 
211         * set to <code>true</code> on the generated <code>Attr</code>. The 
212         * descendants of the source <code>Attr</code> are recursively imported 
213         * and the resulting nodes reassembled to form the corresponding subtree.
214         * Note that the <code>deep</code> parameter has no effect on 
215         * <code>Attr</code> nodes; they always carry their children with them 
216         * when imported.</dd>
217         * <dt>DOCUMENT_FRAGMENT_NODE</dt>
218         * <dd>If the <code>deep</code> option 
219         * was set to <code>true</code>, the descendants of the source element 
220         * are recursively imported and the resulting nodes reassembled to form 
221         * the corresponding subtree. Otherwise, this simply generates an empty 
222         * <code>DocumentFragment</code>.</dd>
223         * <dt>DOCUMENT_NODE</dt>
224         * <dd><code>Document</code> 
225         * nodes cannot be imported.</dd>
226         * <dt>DOCUMENT_TYPE_NODE</dt>
227         * <dd><code>DocumentType</code> 
228         * nodes cannot be imported.</dd>
229         * <dt>ELEMENT_NODE</dt>
230         * <dd>Specified attribute nodes of the 
231         * source element are imported, and the generated <code>Attr</code> 
232         * nodes are attached to the generated <code>Element</code>. Default 
233         * attributes are not copied, though if the document being imported into 
234         * defines default attributes for this element name, those are assigned. 
235         * If the <code>importNode</code> <code>deep</code> parameter was set to 
236         * <code>true</code>, the descendants of the source element are 
237         * recursively imported and the resulting nodes reassembled to form the 
238         * corresponding subtree.</dd>
239         * <dt>ENTITY_NODE</dt>
240         * <dd><code>Entity</code> nodes can be 
241         * imported, however in the current release of the DOM the 
242         * <code>DocumentType</code> is readonly. Ability to add these imported 
243         * nodes to a <code>DocumentType</code> will be considered for addition 
244         * to a future release of the DOM.On import, the <code>publicId</code>, 
245         * <code>systemId</code>, and <code>notationName</code> attributes are 
246         * copied. If a <code>deep</code> import is requested, the descendants 
247         * of the the source <code>Entity</code> are recursively imported and 
248         * the resulting nodes reassembled to form the corresponding subtree.</dd>
249         * <dt>
250         * ENTITY_REFERENCE_NODE</dt>
251         * <dd>Only the <code>EntityReference</code> itself is 
252         * copied, even if a <code>deep</code> import is requested, since the 
253         * source and destination documents might have defined the entity 
254         * differently. If the document being imported into provides a 
255         * definition for this entity name, its value is assigned.</dd>
256         * <dt>NOTATION_NODE</dt>
257         * <dd>
258         * <code>Notation</code> nodes can be imported, however in the current 
259         * release of the DOM the <code>DocumentType</code> is readonly. Ability 
260         * to add these imported nodes to a <code>DocumentType</code> will be 
261         * considered for addition to a future release of the DOM.On import, the 
262         * <code>publicId</code> and <code>systemId</code> attributes are copied.
263         * Note that the <code>deep</code> parameter has no effect on 
264         * <code>Notation</code> nodes since they never have any children.</dd>
265         * <dt>
266         * PROCESSING_INSTRUCTION_NODE</dt>
267         * <dd>The imported node copies its 
268         * <code>target</code> and <code>data</code> values from those of the 
269         * source node.</dd>
270         * <dt>TEXT_NODE, CDATA_SECTION_NODE, COMMENT_NODE</dt>
271         * <dd>These three 
272         * types of nodes inheriting from <code>CharacterData</code> copy their 
273         * <code>data</code> and <code>length</code> attributes from those of 
274         * the source node.</dd>
275         *  
276         * @param importedNode The node to import.
277         * @param deep If <code>true</code>, recursively import the subtree under 
278         *   the specified node; if <code>false</code>, import only the node 
279         *   itself, as explained above. This has no effect on <code>Attr</code>
280         *   , <code>EntityReference</code>, and <code>Notation</code> nodes.
281         * @return The imported node that belongs to this <code>Document</code>.
282         * @exception DOMException
283         *   NOT_SUPPORTED_ERR: Raised if the type of node being imported is not 
284         *   supported.
285         * @since DOM Level 2
286         */
287        public Node importNode(Node importedNode, 
288                               boolean deep)
289                               throws DOMException;
290    
291        /**
292         * Creates an element of the given qualified name and namespace URI. 
293         * HTML-only DOM implementations do not need to implement this method.
294         * @param namespaceURI The namespace URI of the element to create.
295         * @param qualifiedName The qualified name of the element type to 
296         *   instantiate.
297         * @return A new <code>Element</code> object with the following 
298         *   attributes:AttributeValue<code>Node.nodeName</code>
299         *   <code>qualifiedName</code><code>Node.namespaceURI</code>
300         *   <code>namespaceURI</code><code>Node.prefix</code>prefix, extracted 
301         *   from <code>qualifiedName</code>, or <code>null</code> if there is 
302         *   no prefix<code>Node.localName</code>local name, extracted from 
303         *   <code>qualifiedName</code><code>Element.tagName</code>
304         *   <code>qualifiedName</code>
305         * @exception DOMException
306         *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
307         *   contains an illegal character.
308         *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
309         *   malformed, if the <code>qualifiedName</code> has a prefix and the 
310         *   <code>namespaceURI</code> is <code>null</code>, or if the 
311         *   <code>qualifiedName</code> has a prefix that is "xml" and the 
312         *   <code>namespaceURI</code> is different from "
313         *   http://www.w3.org/XML/1998/namespace" .
314         * @since DOM Level 2
315         */
316        public Element createElementNS(String namespaceURI, 
317                                       String qualifiedName)
318                                       throws DOMException;
319    
320        /**
321         * Creates an attribute of the given qualified name and namespace URI. 
322         * HTML-only DOM implementations do not need to implement this method.
323         * @param namespaceURI The namespace URI of the attribute to create.
324         * @param qualifiedName The qualified name of the attribute to instantiate.
325         * @return A new <code>Attr</code> object with the following attributes:
326         *   AttributeValue<code>Node.nodeName</code>qualifiedName
327         *   <code>Node.namespaceURI</code><code>namespaceURI</code>
328         *   <code>Node.prefix</code>prefix, extracted from 
329         *   <code>qualifiedName</code>, or <code>null</code> if there is no 
330         *   prefix<code>Node.localName</code>local name, extracted from 
331         *   <code>qualifiedName</code><code>Attr.name</code>
332         *   <code>qualifiedName</code><code>Node.nodeValue</code>the empty 
333         *   string
334         * @exception DOMException
335         *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
336         *   contains an illegal character.
337         *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
338         *   malformed, if the <code>qualifiedName</code> has a prefix and the 
339         *   <code>namespaceURI</code> is <code>null</code>, if the 
340         *   <code>qualifiedName</code> has a prefix that is "xml" and the 
341         *   <code>namespaceURI</code> is different from "
342         *   http://www.w3.org/XML/1998/namespace", or if the 
343         *   <code>qualifiedName</code> is "xmlns" and the 
344         *   <code>namespaceURI</code> is different from "
345         *   http://www.w3.org/2000/xmlns/".
346         * @since DOM Level 2
347         */
348        public Attr createAttributeNS(String namespaceURI, 
349                                      String qualifiedName)
350                                      throws DOMException;
351    
352        /**
353         * Returns a <code>NodeList</code> of all the <code>Elements</code> with a 
354         * given local name and namespace URI in the order in which they are 
355         * encountered in a preorder traversal of the <code>Document</code> tree.
356         * @param namespaceURI The namespace URI of the elements to match on. The 
357         *   special value "*" matches all namespaces.
358         * @param localName The local name of the elements to match on. The 
359         *   special value "*" matches all local names.
360         * @return A new <code>NodeList</code> object containing all the matched 
361         *   <code>Elements</code>.
362         * @since DOM Level 2
363         */
364        public NodeList getElementsByTagNameNS(String namespaceURI, 
365                                               String localName);
366    
367        /**
368         * Returns the <code>Element</code> whose <code>ID</code> is given by 
369         * <code>elementId</code>. If no such element exists, returns 
370         * <code>null</code>. Behavior is not defined if more than one element 
371         * has this <code>ID</code>. The DOM implementation must have 
372         * information that says which attributes are of type ID. Attributes 
373         * with the name "ID" are not of type ID unless so defined. 
374         * Implementations that do not know whether attributes are of type ID or 
375         * not are expected to return <code>null</code>.
376         * @param elementId The unique <code>id</code> value for an element.
377         * @return The matching element.
378         * @since DOM Level 2
379         */
380        public Element getElementById(String elementId);
