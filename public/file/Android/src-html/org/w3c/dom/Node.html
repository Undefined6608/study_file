025    /**
026     * The <code>Node</code> interface is the primary datatype for the entire 
027     * Document Object Model. It represents a single node in the document tree. 
028     * While all objects implementing the <code>Node</code> interface expose 
029     * methods for dealing with children, not all objects implementing the 
030     * <code>Node</code> interface may have children. For example, 
031     * <code>Text</code> nodes may not have children, and adding children to 
032     * such nodes results in a <code>DOMException</code> being raised.
033     * <p>The attributes <code>nodeName</code>, <code>nodeValue</code> and 
034     * <code>attributes</code> are included as a mechanism to get at node 
035     * information without casting down to the specific derived interface. In 
036     * cases where there is no obvious mapping of these attributes for a 
037     * specific <code>nodeType</code> (e.g., <code>nodeValue</code> for an 
038     * <code>Element</code> or <code>attributes</code> for a <code>Comment</code>
039     * ), this returns <code>null</code>. Note that the specialized interfaces 
040     * may contain additional and more convenient mechanisms to get and set the 
041     * relevant information.
042     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
043     */
044    public interface Node {
045        // NodeType
046        /**
047         * The node is an <code>Element</code>.
048         */
049        public static final short ELEMENT_NODE              = 1;
050        /**
051         * The node is an <code>Attr</code>.
052         */
053        public static final short ATTRIBUTE_NODE            = 2;
054        /**
055         * The node is a <code>Text</code> node.
056         */
057        public static final short TEXT_NODE                 = 3;
058        /**
059         * The node is a <code>CDATASection</code>.
060         */
061        public static final short CDATA_SECTION_NODE        = 4;
062        /**
063         * The node is an <code>EntityReference</code>.
064         */
065        public static final short ENTITY_REFERENCE_NODE     = 5;
066        /**
067         * The node is an <code>Entity</code>.
068         */
069        public static final short ENTITY_NODE               = 6;
070        /**
071         * The node is a <code>ProcessingInstruction</code>.
072         */
073        public static final short PROCESSING_INSTRUCTION_NODE = 7;
074        /**
075         * The node is a <code>Comment</code>.
076         */
077        public static final short COMMENT_NODE              = 8;
078        /**
079         * The node is a <code>Document</code>.
080         */
081        public static final short DOCUMENT_NODE             = 9;
082        /**
083         * The node is a <code>DocumentType</code>.
084         */
085        public static final short DOCUMENT_TYPE_NODE        = 10;
086        /**
087         * The node is a <code>DocumentFragment</code>.
088         */
089        public static final short DOCUMENT_FRAGMENT_NODE    = 11;
090        /**
091         * The node is a <code>Notation</code>.
092         */
093        public static final short NOTATION_NODE             = 12;
094    
095        /**
096         * The name of this node, depending on its type; see the table above. 
097         * 
098         * @return the name of the node.
099         */
100        public String getNodeName();
101    
102        /**
103         * Returns the value of this node, depending on its type; see the table
104         * above.
105         * 
106         * @return the value of the node.
107         * 
108         * @exception DOMException
109         *   DOMSTRING_SIZE_ERR: Raised when it would return more characters than 
110         *   fit in a <code>DOMString</code> variable on the implementation 
111         *   platform.
112         */
113        public String getNodeValue()
114                                     throws DOMException;
115        /**
116         * Sets the value of this node, depending on its type; see the table above. 
117         * When it is defined to be <code>null</code>, setting it has no effect.
118         * 
119         * @param nodeValue the new value of the node.
120         * 
121         * @exception DOMException
122         *   NO_MODIFICATION_ALLOWED_ERR: Raised when the node is readonly.
123         */
124        public void setNodeValue(String nodeValue)
125                                     throws DOMException;
126    
127        /**
128         * A code representing the type of the underlying object, as defined above.
129         * 
130         * @return the type of the node.
131         */
132        public short getNodeType();
133    
134        /**
135         * The parent of this node. All nodes, except <code>Attr</code>, 
136         * <code>Document</code>, <code>DocumentFragment</code>, 
137         * <code>Entity</code>, and <code>Notation</code> may have a parent. 
138         * However, if a node has just been created and not yet added to the 
139         * tree, or if it has been removed from the tree, this is 
140         * <code>null</code>.
141         * 
142         * @return the parent node, if any.
143         */
144        public Node getParentNode();
145    
146        /**
147         * A <code>NodeList</code> that contains all children of this node. If 
148         * there are no children, this is a <code>NodeList</code> containing no 
149         * nodes.
150         * 
151         * @return the list of children, which may be empty.
152         */
153        public NodeList getChildNodes();
154    
155        /**
156         * The first child of this node. If there is no such node, this returns 
157         * <code>null</code>.
158         * 
159         * @return the first child of the node, if any.
160         */
161        public Node getFirstChild();
162    
163        /**
164         * The last child of this node. If there is no such node, this returns 
165         * <code>null</code>.
166         * 
167         * @return the last child of the node, if any.
168         */
169        public Node getLastChild();
170    
171        /**
172         * The node immediately preceding this node. If there is no such node, 
173         * this returns <code>null</code>.
174         * 
175         * @return the preceding node, if any.
176         */
177        public Node getPreviousSibling();
178    
179        /**
180         * The node immediately following this node. If there is no such node, 
181         * this returns <code>null</code>.
182         * 
183         * @return the following node, if any.
184         */
185        public Node getNextSibling();
186    
187        /**
188         * A <code>NamedNodeMap</code> containing the attributes of this node (if 
189         * it is an <code>Element</code>) or <code>null</code> otherwise. 
190         * 
191         * @return the attributes of the node, which may be an empty map, or
192         * <code>null</code>, if this the node cannot have any attributes.
193         */
194        public NamedNodeMap getAttributes();
195    
196        /**
197         * The <code>Document</code> object associated with this node. This is 
198         * also the <code>Document</code> object used to create new nodes. When 
199         * this node is a <code>Document</code> or a <code>DocumentType</code> 
200         * which is not used with any <code>Document</code> yet, this is 
201         * <code>null</code>.
202         * 
203         * @return the document this node belongs to, if any.
204         * 
205         * @version DOM Level 2
206         */
207        public Document getOwnerDocument();
208    
209        /**
210         * Inserts the node <code>newChild</code> before the existing child node 
211         * <code>refChild</code>. If <code>refChild</code> is <code>null</code>, 
212         * insert <code>newChild</code> at the end of the list of children.
213         * <br>If <code>newChild</code> is a <code>DocumentFragment</code> object, 
214         * all of its children are inserted, in the same order, before 
215         * <code>refChild</code>. If the <code>newChild</code> is already in the 
216         * tree, it is first removed.
217         * @param newChild The node to insert.
218         * @param refChild The reference node, i.e., the node before which the new 
219         *   node must be inserted.
220         * @return The node being inserted.
221         * @exception DOMException
222         *   HIERARCHY_REQUEST_ERR: Raised if this node is of a type that does not 
223         *   allow children of the type of the <code>newChild</code> node, or if 
224         *   the node to insert is one of this node's ancestors.
225         *   <br>WRONG_DOCUMENT_ERR: Raised if <code>newChild</code> was created 
226         *   from a different document than the one that created this node.
227         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly or 
228         *   if the parent of the node being inserted is readonly.
229         *   <br>NOT_FOUND_ERR: Raised if <code>refChild</code> is not a child of 
230         *   this node.
231         */
232        public Node insertBefore(Node newChild, 
233                                 Node refChild)
234                                 throws DOMException;
235    
236        /**
237         * Replaces the child node <code>oldChild</code> with <code>newChild</code>
238         *  in the list of children, and returns the <code>oldChild</code> node.
239         * <br>If <code>newChild</code> is a <code>DocumentFragment</code> object, 
240         * <code>oldChild</code> is replaced by all of the 
241         * <code>DocumentFragment</code> children, which are inserted in the 
242         * same order. If the <code>newChild</code> is already in the tree, it 
243         * is first removed.
244         * @param newChild The new node to put in the child list.
245         * @param oldChild The node being replaced in the list.
246         * @return The node replaced.
247         * @exception DOMException
248         *   HIERARCHY_REQUEST_ERR: Raised if this node is of a type that does not 
249         *   allow children of the type of the <code>newChild</code> node, or if 
250         *   the node to put in is one of this node's ancestors.
251         *   <br>WRONG_DOCUMENT_ERR: Raised if <code>newChild</code> was created 
252         *   from a different document than the one that created this node.
253         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node or the parent of 
254         *   the new node is readonly.
255         *   <br>NOT_FOUND_ERR: Raised if <code>oldChild</code> is not a child of 
256         *   this node.
257         */
258        public Node replaceChild(Node newChild, 
259                                 Node oldChild)
260                                 throws DOMException;
261    
262        /**
263         * Removes the child node indicated by <code>oldChild</code> from the list 
264         * of children, and returns it.
265         * @param oldChild The node being removed.
266         * @return The node removed.
267         * @exception DOMException
268         *   NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
269         *   <br>NOT_FOUND_ERR: Raised if <code>oldChild</code> is not a child of 
270         *   this node.
271         */
272        public Node removeChild(Node oldChild)
273                                throws DOMException;
274    
275        /**
276         * Adds the node <code>newChild</code> to the end of the list of children 
277         * of this node. If the <code>newChild</code> is already in the tree, it 
278         * is first removed.
279         * @param newChild The node to add.If it is a <code>DocumentFragment</code>
280         *    object, the entire contents of the document fragment are moved 
281         *   into the child list of this node
282         * @return The node added.
283         * @exception DOMException
284         *   HIERARCHY_REQUEST_ERR: Raised if this node is of a type that does not 
285         *   allow children of the type of the <code>newChild</code> node, or if 
286         *   the node to append is one of this node's ancestors.
287         *   <br>WRONG_DOCUMENT_ERR: Raised if <code>newChild</code> was created 
288         *   from a different document than the one that created this node.
289         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
290         */
291        public Node appendChild(Node newChild)
292                                throws DOMException;
293    
294        /**
295         * Returns whether this node has any children.
296         * @return  <code>true</code> if this node has any children, 
297         *   <code>false</code> otherwise.
298         */
299        public boolean hasChildNodes();
300    
301        /**
302         * Returns a duplicate of this node, i.e., serves as a generic copy 
303         * constructor for nodes. The duplicate node has no parent; (
304         * <code>parentNode</code> is <code>null</code>.).
305         * <br>Cloning an <code>Element</code> copies all attributes and their 
306         * values, including those generated by the XML processor to represent 
307         * defaulted attributes, but this method does not copy any text it 
308         * contains unless it is a deep clone, since the text is contained in a 
309         * child <code>Text</code> node. Cloning an <code>Attribute</code> 
310         * directly, as opposed to be cloned as part of an <code>Element</code> 
311         * cloning operation, returns a specified attribute (
312         * <code>specified</code> is <code>true</code>). Cloning any other type 
313         * of node simply returns a copy of this node.
314         * <br>Note that cloning an immutable subtree results in a mutable copy, 
315         * but the children of an <code>EntityReference</code> clone are readonly
316         * . In addition, clones of unspecified <code>Attr</code> nodes are 
317         * specified. And, cloning <code>Document</code>, 
318         * <code>DocumentType</code>, <code>Entity</code>, and 
319         * <code>Notation</code> nodes is implementation dependent.
320         * @param deep If <code>true</code>, recursively clone the subtree under 
321         *   the specified node; if <code>false</code>, clone only the node 
322         *   itself (and its attributes, if it is an <code>Element</code>). 
323         * @return The duplicate node.
324         */
325        public Node cloneNode(boolean deep);
326    
327        /**
328         * Puts all <code>Text</code> nodes in the full depth of the sub-tree 
329         * underneath this <code>Node</code>, including attribute nodes, into a 
330         * "normal" form where only structure (e.g., elements, comments, 
331         * processing instructions, CDATA sections, and entity references) 
332         * separates <code>Text</code> nodes, i.e., there are neither adjacent 
333         * <code>Text</code> nodes nor empty <code>Text</code> nodes. This can 
334         * be used to ensure that the DOM view of a document is the same as if 
335         * it were saved and re-loaded, and is useful when operations (such as 
336         * XPointer  lookups) that depend on a particular document tree 
337         * structure are to be used.In cases where the document contains 
338         * <code>CDATASections</code>, the normalize operation alone may not be 
339         * sufficient, since XPointers do not differentiate between 
340         * <code>Text</code> nodes and <code>CDATASection</code> nodes.
341         * @version DOM Level 2
342         */
343        public void normalize();
344    
345        /**
346         * Tests whether the DOM implementation implements a specific feature and 
347         * that feature is supported by this node.
348         * @param feature The name of the feature to test. This is the same name 
349         *   which can be passed to the method <code>hasFeature</code> on 
350         *   <code>DOMImplementation</code>.
351         * @param version This is the version number of the feature to test. In 
352         *   Level 2, version 1, this is the string "2.0". If the version is not 
353         *   specified, supporting any version of the feature will cause the 
354         *   method to return <code>true</code>.
355         * @return Returns <code>true</code> if the specified feature is 
356         *   supported on this node, <code>false</code> otherwise.
357         * @since DOM Level 2
358         */
359        public boolean isSupported(String feature, 
360                                   String version);
361    
362        /**
363         * The namespace URI of this node, or <code>null</code> if it is 
364         * unspecified.
365         * <br>This is not a computed value that is the result of a namespace 
366         * lookup based on an examination of the namespace declarations in 
367         * scope. It is merely the namespace URI given at creation time.
368         * <br>For nodes of any type other than <code>ELEMENT_NODE</code> and 
369         * <code>ATTRIBUTE_NODE</code> and nodes created with a DOM Level 1 
370         * method, such as <code>createElement</code> from the 
371         * <code>Document</code> interface, this is always <code>null</code>.Per 
372         * the Namespaces in XML Specification  an attribute does not inherit 
373         * its namespace from the element it is attached to. If an attribute is 
374         * not explicitly given a namespace, it simply has no namespace.
375         * 
376         * @return the namespace URI, if any.
377         * 
378         * @since DOM Level 2
379         */
380        public String getNamespaceURI();
381    
382        /**
383         * Returns the namespace prefix of this node, or <code>null</code> if it is 
384         * unspecified.
385         * <br>For nodes of any type other than <code>ELEMENT_NODE</code> and 
386         * <code>ATTRIBUTE_NODE</code> and nodes created with a DOM Level 1 
387         * method, such as <code>createElement</code> from the 
388         * <code>Document</code> interface, this is always <code>null</code>.
389         * 
390         * @return the namespace prefix, if any.
391         * 
392         * @exception DOMException
393         * @since DOM Level 2
394         */
395        public String getPrefix();
396        
397        /**
398         * Sets the namespace prefix of this node.
399         * <br>Note that setting this attribute, when permitted, changes the 
400         * <code>nodeName</code> attribute, which holds the qualified name, as 
401         * well as the <code>tagName</code> and <code>name</code> attributes of 
402         * the <code>Element</code> and <code>Attr</code> interfaces, when 
403         * applicable.
404         * <br>Note also that changing the prefix of an attribute that is known to 
405         * have a default value, does not make a new attribute with the default 
406         * value and the original prefix appear, since the 
407         * <code>namespaceURI</code> and <code>localName</code> do not change.
408         * <br>For nodes of any type other than <code>ELEMENT_NODE</code> and 
409         * <code>ATTRIBUTE_NODE</code> and nodes created with a DOM Level 1 
410         * method, such as <code>createElement</code> from the 
411         * <code>Document</code> interface, this is always <code>null</code>.
412         * @exception DOMException
413         *   INVALID_CHARACTER_ERR: Raised if the specified prefix contains an 
414         *   illegal character.
415         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
416         *   <br>NAMESPACE_ERR: Raised if the specified <code>prefix</code> is 
417         *   malformed, if the <code>namespaceURI</code> of this node is 
418         *   <code>null</code>, if the specified prefix is "xml" and the 
419         *   <code>namespaceURI</code> of this node is different from "
420         *   http://www.w3.org/XML/1998/namespace", if this node is an attribute 
421         *   and the specified prefix is "xmlns" and the 
422         *   <code>namespaceURI</code> of this node is different from "
423         *   http://www.w3.org/2000/xmlns/", or if this node is an attribute and 
424         *   the <code>qualifiedName</code> of this node is "xmlns" .
425         *   
426         * @param prefix the new namespace prefix.
427         * 
428         * @since DOM Level 2
429         */
430        public void setPrefix(String prefix)
431                                   throws DOMException;
432    
433        /**
434         * Returns the local part of the qualified name of this node.
435         * <br>For nodes of any type other than <code>ELEMENT_NODE</code> and 
436         * <code>ATTRIBUTE_NODE</code> and nodes created with a DOM Level 1 
437         * method, such as <code>createElement</code> from the 
438         * <code>Document</code> interface, this is always <code>null</code>.
439         * 
440         * @return the local name, if any.
441         * 
442         * @since DOM Level 2
443         */
444        public String getLocalName();
445    
446        /**
447         * Returns whether this node (if it is an element) has any attributes.
448         * @return <code>true</code> if this node has any attributes, 
449         *   <code>false</code> otherwise.
450         * @since DOM Level 2
451         */
452        public boolean hasAttributes();
