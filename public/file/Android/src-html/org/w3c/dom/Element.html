025    /**
026     * The <code>Element</code> interface represents an element in an HTML or XML 
027     * document. Elements may have attributes associated with them; since the 
028     * <code>Element</code> interface inherits from <code>Node</code>, the 
029     * generic <code>Node</code> interface attribute <code>attributes</code> may 
030     * be used to retrieve the set of all attributes for an element. There are 
031     * methods on the <code>Element</code> interface to retrieve either an 
032     * <code>Attr</code> object by name or an attribute value by name. In XML, 
033     * where an attribute value may contain entity references, an 
034     * <code>Attr</code> object should be retrieved to examine the possibly 
035     * fairly complex sub-tree representing the attribute value. On the other 
036     * hand, in HTML, where all attributes have simple string values, methods to 
037     * directly access an attribute value can safely be used as a convenience.In 
038     * DOM Level 2, the method <code>normalize</code> is inherited from the 
039     * <code>Node</code> interface where it was moved.
040     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
041     */
042    public interface Element extends Node {
043        /**
044         * The name of the element. For example, in: 
045         * <pre> &lt;elementExample 
046         * id="demo"&gt; ... &lt;/elementExample&gt; , </pre>
047         *  <code>tagName</code> has 
048         * the value <code>"elementExample"</code>. Note that this is 
049         * case-preserving in XML, as are all of the operations of the DOM. The 
050         * HTML DOM returns the <code>tagName</code> of an HTML element in the 
051         * canonical uppercase form, regardless of the case in the source HTML 
052         * document.
053         * 
054         * @return the name of the element.
055         */
056        public String getTagName();
057    
058        /**
059         * Retrieves an attribute value by name.
060         * @param name The name of the attribute to retrieve.
061         * @return The <code>Attr</code> value as a string, or the empty string 
062         *   if that attribute does not have a specified or default value.
063         */
064        public String getAttribute(String name);
065    
066        /**
067         * Adds a new attribute. If an attribute with that name is already present 
068         * in the element, its value is changed to be that of the value 
069         * parameter. This value is a simple string; it is not parsed as it is 
070         * being set. So any markup (such as syntax to be recognized as an 
071         * entity reference) is treated as literal text, and needs to be 
072         * appropriately escaped by the implementation when it is written out. 
073         * In order to assign an attribute value that contains entity 
074         * references, the user must create an <code>Attr</code> node plus any 
075         * <code>Text</code> and <code>EntityReference</code> nodes, build the 
076         * appropriate subtree, and use <code>setAttributeNode</code> to assign 
077         * it as the value of an attribute.
078         * <br>To set an attribute with a qualified name and namespace URI, use 
079         * the <code>setAttributeNS</code> method.
080         * @param name The name of the attribute to create or alter.
081         * @param value Value to set in string form.
082         * @exception DOMException
083         *   INVALID_CHARACTER_ERR: Raised if the specified name contains an 
084         *   illegal character.
085         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
086         */
087        public void setAttribute(String name, 
088                                 String value)
089                                 throws DOMException;
090    
091        /**
092         * Removes an attribute by name. If the removed attribute is known to have 
093         * a default value, an attribute immediately appears containing the 
094         * default value as well as the corresponding namespace URI, local name, 
095         * and prefix when applicable.
096         * <br>To remove an attribute by local name and namespace URI, use the 
097         * <code>removeAttributeNS</code> method.
098         * @param name The name of the attribute to remove.
099         * @exception DOMException
100         *   NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
101         */
102        public void removeAttribute(String name)
103                                    throws DOMException;
104    
105        /**
106         * Retrieves an attribute node by name.
107         * <br>To retrieve an attribute node by qualified name and namespace URI, 
108         * use the <code>getAttributeNodeNS</code> method.
109         * @param name The name (<code>nodeName</code>) of the attribute to 
110         *   retrieve.
111         * @return The <code>Attr</code> node with the specified name (
112         *   <code>nodeName</code>) or <code>null</code> if there is no such 
113         *   attribute.
114         */
115        public Attr getAttributeNode(String name);
116    
117        /**
118         * Adds a new attribute node. If an attribute with that name (
119         * <code>nodeName</code>) is already present in the element, it is 
120         * replaced by the new one.
121         * <br>To add a new attribute node with a qualified name and namespace 
122         * URI, use the <code>setAttributeNodeNS</code> method.
123         * @param newAttr The <code>Attr</code> node to add to the attribute list.
124         * @return If the <code>newAttr</code> attribute replaces an existing 
125         *   attribute, the replaced <code>Attr</code> node is returned, 
126         *   otherwise <code>null</code> is returned.
127         * @exception DOMException
128         *   WRONG_DOCUMENT_ERR: Raised if <code>newAttr</code> was created from a 
129         *   different document than the one that created the element.
130         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
131         *   <br>INUSE_ATTRIBUTE_ERR: Raised if <code>newAttr</code> is already an 
132         *   attribute of another <code>Element</code> object. The DOM user must 
133         *   explicitly clone <code>Attr</code> nodes to re-use them in other 
134         *   elements.
135         */
136        public Attr setAttributeNode(Attr newAttr)
137                                     throws DOMException;
138    
139        /**
140         * Removes the specified attribute node. If the removed <code>Attr</code> 
141         * has a default value it is immediately replaced. The replacing 
142         * attribute has the same namespace URI and local name, as well as the 
143         * original prefix, when applicable.
144         * @param oldAttr The <code>Attr</code> node to remove from the attribute 
145         *   list.
146         * @return The <code>Attr</code> node that was removed.
147         * @exception DOMException
148         *   NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
149         *   <br>NOT_FOUND_ERR: Raised if <code>oldAttr</code> is not an attribute 
150         *   of the element.
151         */
152        public Attr removeAttributeNode(Attr oldAttr)
153                                        throws DOMException;
154    
155        /**
156         * Returns a <code>NodeList</code> of all descendant <code>Elements</code> 
157         * with a given tag name, in the order in which they are encountered in 
158         * a preorder traversal of this <code>Element</code> tree.
159         * @param name The name of the tag to match on. The special value "*" 
160         *   matches all tags.
161         * @return A list of matching <code>Element</code> nodes.
162         */
163        public NodeList getElementsByTagName(String name);
164    
165        /**
166         * Retrieves an attribute value by local name and namespace URI. HTML-only 
167         * DOM implementations do not need to implement this method.
168         * @param namespaceURI The namespace URI of the attribute to retrieve.
169         * @param localName The local name of the attribute to retrieve.
170         * @return The <code>Attr</code> value as a string, or the empty string 
171         *   if that attribute does not have a specified or default value.
172         * @since DOM Level 2
173         */
174        public String getAttributeNS(String namespaceURI, 
175                                     String localName);
176    
177        /**
178         * Adds a new attribute. If an attribute with the same local name and 
179         * namespace URI is already present on the element, its prefix is 
180         * changed to be the prefix part of the <code>qualifiedName</code>, and 
181         * its value is changed to be the <code>value</code> parameter. This 
182         * value is a simple string; it is not parsed as it is being set. So any 
183         * markup (such as syntax to be recognized as an entity reference) is 
184         * treated as literal text, and needs to be appropriately escaped by the 
185         * implementation when it is written out. In order to assign an 
186         * attribute value that contains entity references, the user must create 
187         * an <code>Attr</code> node plus any <code>Text</code> and 
188         * <code>EntityReference</code> nodes, build the appropriate subtree, 
189         * and use <code>setAttributeNodeNS</code> or 
190         * <code>setAttributeNode</code> to assign it as the value of an 
191         * attribute.
192         * <br>HTML-only DOM implementations do not need to implement this method.
193         * @param namespaceURI The namespace URI of the attribute to create or 
194         *   alter.
195         * @param qualifiedName The qualified name of the attribute to create or 
196         *   alter.
197         * @param value The value to set in string form.
198         * @exception DOMException
199         *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
200         *   contains an illegal character.
201         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
202         *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
203         *   malformed, if the <code>qualifiedName</code> has a prefix and the 
204         *   <code>namespaceURI</code> is <code>null</code>, if the 
205         *   <code>qualifiedName</code> has a prefix that is "xml" and the 
206         *   <code>namespaceURI</code> is different from "
207         *   http://www.w3.org/XML/1998/namespace", or if the 
208         *   <code>qualifiedName</code> is "xmlns" and the 
209         *   <code>namespaceURI</code> is different from "
210         *   http://www.w3.org/2000/xmlns/".
211         * @since DOM Level 2
212         */
213        public void setAttributeNS(String namespaceURI, 
214                                   String qualifiedName, 
215                                   String value)
216                                   throws DOMException;
217    
218        /**
219         * Removes an attribute by local name and namespace URI. If the removed 
220         * attribute has a default value it is immediately replaced. The 
221         * replacing attribute has the same namespace URI and local name, as 
222         * well as the original prefix.
223         * <br>HTML-only DOM implementations do not need to implement this method.
224         * @param namespaceURI The namespace URI of the attribute to remove.
225         * @param localName The local name of the attribute to remove.
226         * @exception DOMException
227         *   NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
228         * @since DOM Level 2
229         */
230        public void removeAttributeNS(String namespaceURI, 
231                                      String localName)
232                                      throws DOMException;
233    
234        /**
235         * Retrieves an <code>Attr</code> node by local name and namespace URI. 
236         * HTML-only DOM implementations do not need to implement this method.
237         * @param namespaceURI The namespace URI of the attribute to retrieve.
238         * @param localName The local name of the attribute to retrieve.
239         * @return The <code>Attr</code> node with the specified attribute local 
240         *   name and namespace URI or <code>null</code> if there is no such 
241         *   attribute.
242         * @since DOM Level 2
243         */
244        public Attr getAttributeNodeNS(String namespaceURI, 
245                                       String localName);
246    
247        /**
248         * Adds a new attribute. If an attribute with that local name and that 
249         * namespace URI is already present in the element, it is replaced by 
250         * the new one.
251         * <br>HTML-only DOM implementations do not need to implement this method.
252         * @param newAttr The <code>Attr</code> node to add to the attribute list.
253         * @return If the <code>newAttr</code> attribute replaces an existing 
254         *   attribute with the same local name and namespace URI, the replaced 
255         *   <code>Attr</code> node is returned, otherwise <code>null</code> is 
256         *   returned.
257         * @exception DOMException
258         *   WRONG_DOCUMENT_ERR: Raised if <code>newAttr</code> was created from a 
259         *   different document than the one that created the element.
260         *   <br>NO_MODIFICATION_ALLOWED_ERR: Raised if this node is readonly.
261         *   <br>INUSE_ATTRIBUTE_ERR: Raised if <code>newAttr</code> is already an 
262         *   attribute of another <code>Element</code> object. The DOM user must 
263         *   explicitly clone <code>Attr</code> nodes to re-use them in other 
264         *   elements.
265         * @since DOM Level 2
266         */
267        public Attr setAttributeNodeNS(Attr newAttr)
268                                       throws DOMException;
269    
270        /**
271         * Returns a <code>NodeList</code> of all the descendant 
272         * <code>Elements</code> with a given local name and namespace URI in 
273         * the order in which they are encountered in a preorder traversal of 
274         * this <code>Element</code> tree.
275         * <br>HTML-only DOM implementations do not need to implement this method.
276         * @param namespaceURI The namespace URI of the elements to match on. The 
277         *   special value "*" matches all namespaces.
278         * @param localName The local name of the elements to match on. The 
279         *   special value "*" matches all local names.
280         * @return A new <code>NodeList</code> object containing all the matched 
281         *   <code>Elements</code>.
282         * @since DOM Level 2
283         */
284        public NodeList getElementsByTagNameNS(String namespaceURI, 
285                                               String localName);
286    
287        /**
288         * Returns <code>true</code> when an attribute with a given name is 
289         * specified on this element or has a default value, <code>false</code> 
290         * otherwise.
291         * @param name The name of the attribute to look for.
292         * @return <code>true</code> if an attribute with the given name is 
293         *   specified on this element or has a default value, <code>false</code>
294         *    otherwise.
295         * @since DOM Level 2
296         */
297        public boolean hasAttribute(String name);
298    
299        /**
300         * Returns <code>true</code> when an attribute with a given local name and 
301         * namespace URI is specified on this element or has a default value, 
302         * <code>false</code> otherwise. HTML-only DOM implementations do not 
303         * need to implement this method.
304         * @param namespaceURI The namespace URI of the attribute to look for.
305         * @param localName The local name of the attribute to look for.
306         * @return <code>true</code> if an attribute with the given local name 
307         *   and namespace URI is specified or has a default value on this 
308         *   element, <code>false</code> otherwise.
309         * @since DOM Level 2
310         */
311        public boolean hasAttributeNS(String namespaceURI, 
312                                      String localName);
313
