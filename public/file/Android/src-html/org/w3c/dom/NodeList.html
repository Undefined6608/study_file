025    /**
026     * The <code>NodeList</code> interface provides the abstraction of an ordered 
027     * collection of nodes, without defining or constraining how this collection 
028     * is implemented. <code>NodeList</code> objects in the DOM are live.
029     * <p>The items in the <code>NodeList</code> are accessible via an integral 
030     * index, starting from 0.
031     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
032     */
033    public interface NodeList {
034        /**
035         * Returns the <code>index</code>th item in the collection. If 
036         * <code>index</code> is greater than or equal to the number of nodes in 
037         * the list, this returns <code>null</code>.
038         * @param index Index into the collection.
039         * @return The node at the <code>index</code>th position in the 
040         *   <code>NodeList</code>, or <code>null</code> if that is not a valid 
041         *   index.
042         */
043        public Node item(int index);
044    
045        /**
046         * The number of nodes in the list. The range of valid child node indices 
047         * is 0 to <code>length-1</code> inclusive.
048         * 
049         * @return the length of the list. 
050         */
051        public int getLength();
052    
053    }
