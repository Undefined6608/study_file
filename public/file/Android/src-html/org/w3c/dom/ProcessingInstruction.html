020    /**
021     * The <code>ProcessingInstruction</code> interface represents a "processing 
022     * instruction", used in XML as a way to keep processor-specific information 
023     * in the text of the document.
024     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
025     */
026    public interface ProcessingInstruction extends Node {
027        /**
028         * The target of this processing instruction. XML defines this as being 
029         * the first token following the markup that begins the processing 
030         * instruction.
031         * 
032         * @return the target of this processing instruction.
033         */
034        public String getTarget();
035    
036        /**
037         * Returns the content of this processing instruction. This is from the
038         * first non white space character after the target to the character
039         * immediately preceding the <code>?&gt;</code>.
040         * 
041         * @return the data of this processing instruction.
042         */
043        public String getData();
044    
045        /**
046         * Sets the content of this processing instruction. This is from the first
047         * non white space character after the target to the character immediately 
048         * preceding the <code>?&gt;</code>.
049         * 
050         * @param data the new data of the processing instruction.
051         * 
052         * @exception DOMException
053         *   NO_MODIFICATION_ALLOWED_ERR: Raised when the node is readonly.
054         */
055        public void setData(String data)
056                              throws DOMException;
057    
058    }
