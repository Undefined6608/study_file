020    /**
021     * The <code>DOMImplementation</code> interface provides a number of methods 
022     * for performing operations that are independent of any particular instance 
023     * of the document object model.
024     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
025     */
026    public interface DOMImplementation {
027        /**
028         * Test if the DOM implementation implements a specific feature.
029         * @param feature The name of the feature to test (case-insensitive). The 
030         *   values used by DOM features are defined throughout the DOM Level 2 
031         *   specifications and listed in the  section. The name must be an XML 
032         *   name. To avoid possible conflicts, as a convention, names referring 
033         *   to features defined outside the DOM specification should be made 
034         *   unique by reversing the name of the Internet domain name of the 
035         *   person (or the organization that the person belongs to) who defines 
036         *   the feature, component by component, and using this as a prefix. 
037         *   For instance, the W3C SVG Working Group defines the feature 
038         *   "org.w3c.dom.svg".
039         * @param version This is the version number of the feature to test. In 
040         *   Level 2, the string can be either "2.0" or "1.0". If the version is 
041         *   not specified, supporting any version of the feature causes the 
042         *   method to return <code>true</code>.
043         * @return <code>true</code> if the feature is implemented in the 
044         *   specified version, <code>false</code> otherwise.
045         */
046        public boolean hasFeature(String feature, 
047                                  String version);
048    
049        /**
050         * Creates an empty <code>DocumentType</code> node. Entity declarations 
051         * and notations are not made available. Entity reference expansions and 
052         * default attribute additions do not occur. It is expected that a 
053         * future version of the DOM will provide a way for populating a 
054         * <code>DocumentType</code>.
055         * <br>HTML-only DOM implementations do not need to implement this method.
056         * @param qualifiedName The qualified name of the document type to be 
057         *   created. 
058         * @param publicId The external subset public identifier.
059         * @param systemId The external subset system identifier.
060         * @return A new <code>DocumentType</code> node with 
061         *   <code>Node.ownerDocument</code> set to <code>null</code>.
062         * @exception DOMException
063         *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
064         *   contains an illegal character.
065         *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
066         *   malformed.
067         * @since DOM Level 2
068         */
069        public DocumentType createDocumentType(String qualifiedName, 
070                                               String publicId, 
071                                               String systemId)
072                                               throws DOMException;
073    
074        /**
075         * Creates an XML <code>Document</code> object of the specified type with 
076         * its document element. HTML-only DOM implementations do not need to 
077         * implement this method.
078         * @param namespaceURI The namespace URI of the document element to create.
079         * @param qualifiedName The qualified name of the document element to be 
080         *   created.
081         * @param doctype The type of document to be created or <code>null</code>.
082         *   When <code>doctype</code> is not <code>null</code>, its 
083         *   <code>Node.ownerDocument</code> attribute is set to the document 
084         *   being created.
085         * @return A new <code>Document</code> object.
086         * @exception DOMException
087         *   INVALID_CHARACTER_ERR: Raised if the specified qualified name 
088         *   contains an illegal character.
089         *   <br>NAMESPACE_ERR: Raised if the <code>qualifiedName</code> is 
090         *   malformed, if the <code>qualifiedName</code> has a prefix and the 
091         *   <code>namespaceURI</code> is <code>null</code>, or if the 
092         *   <code>qualifiedName</code> has a prefix that is "xml" and the 
093         *   <code>namespaceURI</code> is different from "
094         *   http://www.w3.org/XML/1998/namespace" .
095         *   <br>WRONG_DOCUMENT_ERR: Raised if <code>doctype</code> has already 
096         *   been used with a different document or was created from a different 
097         *   implementation.
098         * @since DOM Level 2
099         */
100        public Document createDocument(String namespaceURI, 
101                                       String qualifiedName, 
102                                       DocumentType doctype)
103                                       throws DOMException;
104    
105    }
