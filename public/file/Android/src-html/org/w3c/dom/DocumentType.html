020    /**
021     * Each <code>Document</code> has a <code>doctype</code> attribute whose value 
022     * is either <code>null</code> or a <code>DocumentType</code> object. The 
023     * <code>DocumentType</code> interface in the DOM Core provides an interface 
024     * to the list of entities that are defined for the document, and little 
025     * else because the effect of namespaces and the various XML schema efforts 
026     * on DTD representation are not clearly understood as of this writing.
027     * <p>The DOM Level 2 doesn't support editing <code>DocumentType</code> nodes.
028     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
029     */
030    public interface DocumentType extends Node {
031        /**
032         * The name of DTD; i.e., the name immediately following the 
033         * <code>DOCTYPE</code> keyword.
034         * 
035         * @return the name of the DTD.
036         */
037        public String getName();
038    
039        /**
040         * A <code>NamedNodeMap</code> containing the general entities, both 
041         * external and internal, declared in the DTD. Parameter entities are 
042         * not contained. Duplicates are discarded. For example in: 
043         * <pre>&lt;!DOCTYPE 
044         * ex SYSTEM "ex.dtd" [ &lt;!ENTITY foo "foo"&gt; &lt;!ENTITY bar 
045         * "bar"&gt; &lt;!ENTITY bar "bar2"&gt; &lt;!ENTITY % baz "baz"&gt; 
046         * ]&gt; &lt;ex/&gt;</pre>
047         *  the interface provides access to <code>foo</code> 
048         * and the first declaration of <code>bar</code> but not the second 
049         * declaration of <code>bar</code> or <code>baz</code>. Every node in 
050         * this map also implements the <code>Entity</code> interface.
051         * <br>The DOM Level 2 does not support editing entities, therefore 
052         * <code>entities</code> cannot be altered in any way.
053         * 
054         * @return the entities declared in this DTD.
055         */
056        public NamedNodeMap getEntities();
057    
058        /**
059         * A <code>NamedNodeMap</code> containing the notations declared in the 
060         * DTD. Duplicates are discarded. Every node in this map also implements 
061         * the <code>Notation</code> interface.
062         * <br>The DOM Level 2 does not support editing notations, therefore 
063         * <code>notations</code> cannot be altered in any way.
064         * 
065         * @return the notations declared in this DTD.
066         */
067        public NamedNodeMap getNotations();
068    
069        /**
070         * The public identifier of the external subset.
071         * 
072         * @return the public identifier.
073         * 
074         * @since DOM Level 2
075         */
076        public String getPublicId();
077    
078        /**
079         * The system identifier of the external subset.
080         * 
081         * @return the system identifier.
082         * 
083         * @since DOM Level 2
084         */
085        public String getSystemId();
086    
087        /**
088         * The internal subset as a string. The actual content returned depends on 
089         * how much information is available to the implementation. This may 
090         * vary depending on various parameters, including the XML processor 
091         * used to build the document.
092         * 
093         * @return the internal subset.
094         * 
095         * @since DOM Level 2
096         */
097        public String getInternalSubset();
098    
099    }
