020    /**
021     * This interface represents a notation declared in the DTD. A notation either 
022     * declares, by name, the format of an unparsed entity (see section 4.7 of 
023     * the XML 1.0 specification ), or is used for formal declaration of 
024     * processing instruction targets (see section 2.6 of the XML 1.0 
025     * specification ). The <code>nodeName</code> attribute inherited from 
026     * <code>Node</code> is set to the declared name of the notation.
027     * <p>The DOM Level 1 does not support editing <code>Notation</code> nodes; 
028     * they are therefore readonly.
029     * <p>A <code>Notation</code> node does not have any parent.
030     * <p>See also the <a href='http://www.w3.org/TR/2000/REC-DOM-Level-2-Core-20001113'>Document Object Model (DOM) Level 2 Core Specification</a>.
031     */
032    public interface Notation extends Node {
033        /**
034         * The public identifier of this notation. If the public identifier was 
035         * not specified, this is <code>null</code>.
036         * 
037         * @return the public identifier of this notation, if any.
038         */
039        public String getPublicId();
040    
041        /**
042         * The system identifier of this notation. If the system identifier was 
043         * not specified, this is <code>null</code>.
044         * 
045         * @return the system identifier of this notation, if any.
046         */
047        public String getSystemId();
048    
049    }
