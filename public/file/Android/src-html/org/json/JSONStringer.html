027    /**
028     * JSONStringer provides a quick and convenient way of producing JSON text.
029     * The texts produced strictly conform to JSON syntax rules. No whitespace is 
030     * added, so the results are ready for transmission or storage. Each instance of 
031     * JSONStringer can produce one JSON text.
032     * <p>
033     * A JSONStringer instance provides a <code>value</code> method for appending 
034     * values to the 
035     * text, and a <code>key</code> 
036     * method for adding keys before values in objects. There are <code>array</code>
037     * and <code>endArray</code> methods that make and bound array values, and 
038     * <code>object</code> and <code>endObject</code> methods which make and bound 
039     * object values. All of these methods return the JSONStringer instance, 
040     * permitting cascade style. For example, <pre>
041     * myString = new JSONStringer()
042     *     .object()
043     *         .key("JSON").value("Hello, World!")
044     *     .endObject()
045     *     .toString();</pre> which produces the string <pre>
046     * {"JSON":"Hello, World!"}</pre>
047     * <p>
048     * The first method called must be <code>array</code> or <code>object</code>. 
049     * There are no methods for adding commas or colons. JSONStringer adds them for 
050     * you. Objects and arrays can be nested up to 20 levels deep.
051     * <p>
052     * This can sometimes be easier than using a JSONObject to build a string.
053     * @author JSON.org
054     * @version 2
055     */
056    public class JSONStringer {
057        private static final int maxdepth = 20;
058        
059        /**
060         * The comma flag determines if a comma should be output before the next
061         * value.
062         */
063        private boolean comma;
064        
065        /**
066         * The current mode. Values: 
067         * 'a' (array), 
068         * 'd' (done), 
069         * 'i' (initial), 
070         * 'k' (key), 
071         * 'o' (object).
072         */
073        private char mode;
074        
075        /**
076         * The string buffer that holds the JSON text that is built.
077         */
078        private StringBuilder sb;
079        
080        /**
081         * The object/array stack.
082         */
083        private char stack[];
084        
085        /**
086         * The stack top index. A value of 0 indicates that the stack is empty.
087         */
088        private int top;
089        
090        /**
091         * Make a fresh JSONStringer. It can be used to build one JSON text.
092         */
093        public JSONStringer() {
094            this.sb = new StringBuilder();
095            this.stack = new char[maxdepth];
096            this.top = 0;
097            this.mode = 'i';
098            this.comma = false;
099        }
100        
101        /**
102         * Append a value.
103         * @param s A string value.
104         * @return this
105         * @throws JSONException If the value is out of sequence.
106         */
107        private JSONStringer append(String s) 
108                throws JSONException {
109            if (s == null) {
110                throw new JSONException("Null pointer");
111            }
112            if (this.mode == 'o' || this.mode == 'a') {
113                if (this.comma && this.mode == 'a') {
114                    this.sb.append(',');
115                }
116                this.sb.append(s);
117                if (this.mode == 'o') {
118                    this.mode = 'k';
119                }
120                this.comma = true;
121                return this;
122            } 
123            throw new JSONException("Value out of sequence.");    
124        }
125        
126        /**
127         * Begin appending a new array. All values until the balancing 
128         * <code>endArray</code> will be appended to this array. The 
129         * <code>endArray</code> method must be called to mark the array's end.
130         * @return this
131         * @throws JSONException If the nesting is too deep, or if the object is 
132         * started in the wrong place (for example as a key or after the end of the 
133         * outermost array or object).
134         */
135        public JSONStringer array() throws JSONException {
136            if (this.mode == 'i' || this.mode == 'o' || this.mode == 'a') {
137                push('a');
138                this.append("[");
139                this.comma = false;
140                return this;
141            } 
142            throw new JSONException("Misplaced array.");
143        }
144        
145        /**
146         * End something. 
147         * @param m Mode
148         * @param c Closing character
149         * @return this
150         * @throws JSONException If unbalanced.
151         */
152        private JSONStringer end(char m, char c) throws JSONException {
153            if (this.mode != m) {
154                throw new JSONException(m == 'o' ? "Misplaced endObject." :
155                    "Misplaced endArray.");            
156            }
157            pop(m);
158            this.sb.append(c);
159            this.comma = true;
160            return this;
161        }
162        
163        /**
164         * End an array. This method most be called to balance calls to 
165         * <code>array</code>.
166         * @return this
167         * @throws JSONException If incorrectly nested.
168         */
169        public JSONStringer endArray() throws JSONException {
170            return end('a', ']');
171        }
172        
173        /**
174         * End an object. This method most be called to balance calls to 
175         * <code>object</code>.
176         * @return this
177         * @throws JSONException If incorrectly nested.
178         */
179        public JSONStringer endObject() throws JSONException {
180            return end('k', '}');
181        }
182        
183        /**
184         * Append a key. The key will be associated with the next value. In an
185         * object, every value must be preceded by a key.
186         * @param s A key string.
187         * @return this
188         * @throws JSONException If the key is out of place. For example, keys
189         *     do not belong in arrays or if the key is null.
190         */
191        public JSONStringer key(String s) 
192                throws JSONException {
193            if (s == null) {
194                throw new JSONException("Null key.");
195            }
196            if (this.mode == 'k') {
197                if (this.comma) {
198                    this.sb.append(',');
199                }
200                this.sb.append(JSONObject.quote(s));
201                this.sb.append(':');
202                this.comma = false;
203                this.mode = 'o';
204                return this;
205            } 
206            throw new JSONException("Misplaced key.");    
207        }
208    
209        
210        /**
211         * Begin appending a new object. All keys and values until the balancing 
212         * <code>endObject</code> will be appended to this object. The 
213         * <code>endObject</code> method must be called to mark the object's end.
214         * @return this
215         * @throws JSONException If the nesting is too deep, or if the object is 
216         * started in the wrong place (for example as a key or after the end of the 
217         * outermost array or object).
218         */
219        public JSONStringer object() throws JSONException {
220            if (this.mode == 'i') {
221                this.mode = 'o';
222            }
223            if (this.mode == 'o' || this.mode == 'a') {
224                this.append("{");
225                push('k');
226                this.comma = false;
227                return this;
228            } 
229            throw new JSONException("Misplaced object.");
230            
231        }
232        
233        
234        /**
235         * Pop an array or object scope.
236         * @param c The scope to close.
237         * @throws JSONException If nesting is wrong.
238         */
239        private void pop(char c) throws JSONException {
240            if (this.top <= 0 || this.stack[this.top - 1] != c) {
241                throw new JSONException("Nesting error.");
242            }
243            this.top -= 1;
244            this.mode = this.top == 0 ? 'd' : this.stack[this.top - 1];
245        }
246        
247        /**
248         * Push an array or object scope.
249         * @param c The scope to open.
250         * @throws JSONException If nesting is too deep.
251         */
252        private void push(char c) throws JSONException {
253            if (this.top >= maxdepth) {
254                throw new JSONException("Nesting too deep.");
255            }
256            this.stack[this.top] = c;
257            this.mode = c;
258            this.top += 1;
259        }
260        
261        
262        /**
263         * Append either the value <code>true</code> or the value 
264         * <code>false</code>. 
265         * @param b A boolean.
266         * @return this
267         * @throws JSONException
268         */
269        public JSONStringer value(boolean b) throws JSONException {
270            return this.append(b ? "true" : "false");
271        }
272        
273        /**
274         * Append a double value.
275         * @param d A double.
276         * @return this
277         * @throws JSONException If the number is not finite.
278         */
279        public JSONStringer value(double d) throws JSONException {
280            return this.value(new Double(d));
281        }
282        
283        /**
284         * Append a long value.
285         * @param l A long.
286         * @return this
287         * @throws JSONException
288         */
289        public JSONStringer value(long l) throws JSONException {
290            return this.append(Long.toString(l));
291        }
292        
293        
294        /**
295         * Append an object value.
296         * @param o The object to append. It can be null, or a Boolean, Number,
297         *   String, JSONObject, or JSONArray.
298         * @return this
299         * @throws JSONException If the value is out of sequence.
300         */
301        public JSONStringer value(Object o) throws JSONException {
302            if (JSONObject.NULL.equals(o)) {
303                return this.append("null");
304            }
305            if (o instanceof Number) {
306                JSONObject.testValidity(o);
307                return this.append(JSONObject.numberToString((Number)o));
308            }
309            if (o instanceof Boolean || 
310                    o instanceof JSONArray || o instanceof JSONObject) {
311                return this.append(o.toString());
312            }
313            return this.append(JSONObject.quote(o.toString()));
314        }
315        
316        /**
317         * Return the JSON text. This method is used to obtain the product of the
318         * JSONStringer instance. It will return <code>null</code> if there was a 
319         * problem in the construction of the JSON text (such as the calls to 
320         * <code>array</code> were not properly balanced with calls to 
321         * <code>endArray</code>).
322         * @return The JSON text.
323         */
324        public String toString() {
325            return this.mode == 'd' ? this.sb.toString() : null;
326        }
327    }
