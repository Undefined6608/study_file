032    package org.apache.http.message;
033    
034    import org.apache.http.HttpRequest;
035    import org.apache.http.ProtocolVersion;
036    import org.apache.http.RequestLine;
037    import org.apache.http.params.HttpProtocolParams;
038    
039    /**
040     * Basic implementation of an HTTP request that can be modified.
041     *
042     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
043     *
044     * @version $Revision: 573864 $
045     * 
046     * @since 4.0
047     */
048    public class BasicHttpRequest extends AbstractHttpMessage implements HttpRequest {
049        
050        private final RequestLine requestline;
051        private final String method;
052        private final String uri;
053            
054        public BasicHttpRequest(final String method, final String uri) {
055            super();
056            if (method == null) {
057                throw new IllegalArgumentException("Method name may not be null");
058            }
059            if (uri == null) {
060                throw new IllegalArgumentException("Request URI may not be null");
061            }
062            this.method = method;
063            this.uri = uri;
064            this.requestline = null;
065        }
066    
067        public BasicHttpRequest(final String method, final String uri, final ProtocolVersion ver) {
068            this(new BasicRequestLine(method, uri, ver));
069        }
070    
071        public BasicHttpRequest(final RequestLine requestline) {
072            super();
073            if (requestline == null) {
074                throw new IllegalArgumentException("Request line may not be null");
075            }
076            this.requestline = requestline;
077            this.method = requestline.getMethod();
078            this.uri = requestline.getUri();
079        }
080    
081        public ProtocolVersion getProtocolVersion() {
082            if (this.requestline != null) {
083                return this.requestline.getProtocolVersion();
084            } else {
085                return HttpProtocolParams.getVersion(getParams());
086            }
087        }
088        
089        public RequestLine getRequestLine() {
090            if (this.requestline != null) {
091                return this.requestline;
092            } else {
093                ProtocolVersion ver = HttpProtocolParams.getVersion(getParams());
094                return new BasicRequestLine(this.method, this.uri, ver);
095            }
096        }
097        
098    }
