032    package org.apache.http.message;
033    
034    
035    import org.apache.http.HeaderElement;
036    import org.apache.http.NameValuePair;
037    import org.apache.http.util.CharArrayBuffer;
038    
039    
040    
041    /**
042     * Interface for formatting elements of a header value.
043     * This is the complement to {@link HeaderValueParser}.
044     * Instances of this interface are expected to be stateless and thread-safe.
045     *
046     * <p>
047     * All formatting methods accept an optional buffer argument.
048     * If a buffer is passed in, the formatted element will be appended
049     * and the modified buffer is returned. If no buffer is passed in,
050     * a new buffer will be created and filled with the formatted element.
051     * In both cases, the caller is allowed to modify the returned buffer.
052     * </p>
053     *
054     *
055     * <!-- empty lines above to avoid 'svn diff' context problems -->
056     * @version $Revision: 571954 $
057     *
058     * @since 4.0
059     */
060    public interface HeaderValueFormatter {
061    
062        /**
063         * Formats an array of header elements.
064         *
065         * @param buffer    the buffer to append to, or
066         *                  <code>null</code> to create a new buffer
067         * @param elems     the header elements to format
068         * @param quote     <code>true</code> to always format with quoted values,
069         *                  <code>false</code> to use quotes only when necessary
070         *
071         * @return  a buffer with the formatted header elements.
072         *          If the <code>buffer</code> argument was not <code>null</code>,
073         *          that buffer will be used and returned.
074         */
075        CharArrayBuffer formatElements(CharArrayBuffer buffer,
076                                       HeaderElement[] elems,
077                                       boolean quote)
078            ;
079    
080    
081        /**
082         * Formats one header element.
083         *
084         * @param buffer    the buffer to append to, or
085         *                  <code>null</code> to create a new buffer
086         * @param elem      the header element to format
087         * @param quote     <code>true</code> to always format with quoted values,
088         *                  <code>false</code> to use quotes only when necessary
089         *
090         * @return  a buffer with the formatted header element.
091         *          If the <code>buffer</code> argument was not <code>null</code>,
092         *          that buffer will be used and returned.
093         */
094        CharArrayBuffer formatHeaderElement(CharArrayBuffer buffer,
095                                            HeaderElement elem,
096                                            boolean quote)
097            ;
098    
099    
100    
101        /**
102         * Formats the parameters of a header element.
103         * That's a list of name-value pairs, to be separated by semicolons.
104         * This method will <i>not</i> generate a leading semicolon.
105         *
106         * @param buffer    the buffer to append to, or
107         *                  <code>null</code> to create a new buffer
108         * @param nvps      the parameters (name-value pairs) to format
109         * @param quote     <code>true</code> to always format with quoted values,
110         *                  <code>false</code> to use quotes only when necessary
111         *
112         * @return  a buffer with the formatted parameters.
113         *          If the <code>buffer</code> argument was not <code>null</code>,
114         *          that buffer will be used and returned.
115         */
116        CharArrayBuffer formatParameters(CharArrayBuffer buffer,
117                                         NameValuePair[] nvps,
118                                         boolean quote)
119            ;
120    
121    
122        /**
123         * Formats one name-value pair, where the value is optional.
124         *
125         * @param buffer    the buffer to append to, or
126         *                  <code>null</code> to create a new buffer
127         * @param nvp       the name-value pair to format
128         * @param quote     <code>true</code> to always format with a quoted value,
129         *                  <code>false</code> to use quotes only when necessary
130         *
131         * @return  a buffer with the formatted name-value pair.
132         *          If the <code>buffer</code> argument was not <code>null</code>,
133         *          that buffer will be used and returned.
134         */
135        CharArrayBuffer formatNameValuePair(CharArrayBuffer buffer,
136                                            NameValuePair nvp,
137                                            boolean quote)
138            ;
139    
140    }
