032    package org.apache.http.message;
033    
034    import org.apache.http.HttpStatus;
035    import org.apache.http.ProtocolVersion;
036    import org.apache.http.StatusLine;
037    
038    
039    
040    /**
041     * Represents a status line as returned from a HTTP server.
042     * See <a href="http://www.ietf.org/rfc/rfc2616.txt">RFC2616</a> section 6.1.
043     * This class is immutable and therefore inherently thread safe.
044     *
045     * @see HttpStatus
046     * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
047     * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
048     *
049     * @version $Id: BasicStatusLine.java 604625 2007-12-16 14:11:11Z olegk $
050     * 
051     * @since 4.0
052     */
053    public class BasicStatusLine implements StatusLine, Cloneable {
054    
055        // ----------------------------------------------------- Instance Variables
056    
057        /** The protocol version. */
058        private final ProtocolVersion protoVersion;
059    
060        /** The status code. */
061        private final int statusCode;
062    
063        /** The reason phrase. */
064        private final String reasonPhrase;
065    
066        // ----------------------------------------------------------- Constructors
067        /**
068         * Creates a new status line with the given version, status, and reason.
069         *
070         * @param version           the protocol version of the response
071         * @param statusCode        the status code of the response
072         * @param reasonPhrase      the reason phrase to the status code, or
073         *                          <code>null</code>
074         */
075        public BasicStatusLine(final ProtocolVersion version, int statusCode,
076                               final String reasonPhrase) {
077            super();
078            if (version == null) {
079                throw new IllegalArgumentException
080                    ("Protocol version may not be null.");
081            }
082            if (statusCode < 0) {
083                throw new IllegalArgumentException
084                    ("Status code may not be negative.");
085            }
086            this.protoVersion = version;
087            this.statusCode   = statusCode;
088            this.reasonPhrase = reasonPhrase;
089        }
090    
091        // --------------------------------------------------------- Public Methods
092    
093        /**
094         * @return the Status-Code
095         */
096        public int getStatusCode() {
097            return this.statusCode;
098        }
099    
100        /**
101         * @return the HTTP-Version
102         */
103        public ProtocolVersion getProtocolVersion() {
104            return this.protoVersion;
105        }
106    
107        /**
108         * @return the Reason-Phrase
109         */
110        public String getReasonPhrase() {
111            return this.reasonPhrase;
112        }
113    
114        public String toString() {
115            // no need for non-default formatting in toString()
116            return BasicLineFormatter.DEFAULT
117                .formatStatusLine(null, this).toString();
118        }
119        
120        public Object clone() throws CloneNotSupportedException {
121            return super.clone();
122        }
123        
124    }
