032    package org.apache.http.cookie;
033    
034    import java.io.Serializable;
035    import java.util.Comparator;
036    
037    /**
038     * This cookie comparator ensures that multiple cookies satisfying 
039     * a common criteria are ordered in the <tt>Cookie</tt> header such
040     * that those with more specific Path attributes precede those with
041     * less specific.
042     *  
043     * <p>
044     * This comparator assumes that Path attributes of two cookies 
045     * path-match a commmon request-URI. Otherwise, the result of the
046     * comparison is undefined.
047     * </p>
048     * 
049     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
050     */
051    public class CookiePathComparator implements Serializable, Comparator<Cookie> {
052    
053        private static final long serialVersionUID = 7523645369616405818L;
054    
055        private String normalizePath(final Cookie cookie) {
056            String path = cookie.getPath();
057            if (path == null) {
058                path = "/";
059            }
060            if (!path.endsWith("/")) {
061                path = path + '/';
062            }
063            return path;
064        }
065        
066        public int compare(final Cookie c1, final Cookie c2) {
067            String path1 = normalizePath(c1);
068            String path2 = normalizePath(c2);
069            if (path1.equals(path2)) {
070                return 0;
071            } else if (path1.startsWith(path2)) {
072                return -1;
073            } else if (path2.startsWith(path1)) {
074                return 1;
075            } else {
076                // Does not really matter
077                return 0;
078            }
079        }
080    
081    }
