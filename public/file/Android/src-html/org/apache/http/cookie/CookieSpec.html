032    package org.apache.http.cookie;
033    
034    import java.util.List;
035    
036    import org.apache.http.Header;
037    
038    /**
039     * Defines the cookie management specification.
040     * <p>Cookie management specification must define
041     * <ul>
042     *   <li> rules of parsing "Set-Cookie" header
043     *   <li> rules of validation of parsed cookies
044     *   <li>  formatting of "Cookie" header 
045     * </ul>
046     * for a given host, port and path of origin
047     * 
048     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
049     * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
050     *
051     * @since 4.0
052     */
053    public interface CookieSpec {    
054    
055        /**
056         * Returns version of the state management this cookie specification
057         * conforms to.
058         * 
059         * @return version of the state management specification
060         */
061        int getVersion();
062        
063        /**
064          * Parse the <tt>"Set-Cookie"</tt> Header into an array of Cookies.
065          *
066          * <p>This method will not perform the validation of the resultant
067          * {@link Cookie}s</p> 
068          *
069          * @see #validate
070          *
071          * @param header the <tt>Set-Cookie</tt> received from the server
072          * @param origin details of the cookie origin
073          * @return an array of <tt>Cookie</tt>s parsed from the header
074          * @throws MalformedCookieException if an exception occurs during parsing
075          */
076        List<Cookie> parse(Header header, CookieOrigin origin) throws MalformedCookieException;
077    
078        /**
079          * Validate the cookie according to validation rules defined by the 
080          *  cookie specification.
081          *
082          * @param cookie the Cookie to validate
083          * @param origin details of the cookie origin
084          * @throws MalformedCookieException if the cookie is invalid
085          */
086        void validate(Cookie cookie, CookieOrigin origin) throws MalformedCookieException;
087        
088        /**
089         * Determines if a Cookie matches the target location.
090         *
091         * @param cookie the Cookie to be matched
092         * @param origin the target to test against
093         * 
094         * @return <tt>true</tt> if the cookie should be submitted with a request 
095         *  with given attributes, <tt>false</tt> otherwise.
096         */
097        boolean match(Cookie cookie, CookieOrigin origin);
098    
099        /**
100         * Create <tt>"Cookie"</tt> headers for an array of Cookies.
101         *
102         * @param cookies the Cookies format into a Cookie header
103         * @return a Header for the given Cookies.
104         * @throws IllegalArgumentException if an input parameter is illegal
105         */
106        List<Header> formatCookies(List<Cookie> cookies);
107    
108        /**
109         * Returns a request header identifying what version of the state management 
110         * specification is understood. May be <code>null</code> if the cookie 
111         * specification does not support <tt>Cookie2</tt> header.
112         */
113        Header getVersionHeader();
114        
115    }
