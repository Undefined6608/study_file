032    package org.apache.http.conn;
033    
034    import java.util.concurrent.TimeUnit;
035    
036    /**
037     * Encapsulates a request for a {@link ManagedClientConnection}.
038     */
039    public interface ClientConnectionRequest {
040        
041        /**
042         * Obtains a connection within a given time.
043         * This method will block until a connection becomes available,
044         * the timeout expires, or the connection manager is
045         * {@link ClientConnectionManager#shutdown() shut down}.
046         * Timeouts are handled with millisecond precision.
047         * 
048         * If {@link #abortRequest()} is called while this is blocking or
049         * before this began, an {@link InterruptedException} will
050         * be thrown.
051         * 
052         * @param timeout   the timeout, 0 or negative for no timeout
053         * @param tunit     the unit for the <code>timeout</code>,
054         *                  may be <code>null</code> only if there is no timeout
055         *
056         * @return  a connection that can be used to communicate
057         *          along the given route
058         *
059         * @throws ConnectionPoolTimeoutException
060         *         in case of a timeout
061         * @throws InterruptedException
062         *         if the calling thread is interrupted while waiting
063         */
064        ManagedClientConnection getConnection(long timeout, TimeUnit tunit) 
065            throws InterruptedException, ConnectionPoolTimeoutException;
066        
067        /**
068         * Aborts the call to {@link #getConnection(long, TimeUnit)},
069         * causing it to throw an {@link InterruptedException}.
070         */
071        void abortRequest();
072    
073    }
