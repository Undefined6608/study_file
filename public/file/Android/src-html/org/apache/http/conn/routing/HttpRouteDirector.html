032    package org.apache.http.conn.routing;
033    
034    
035    
036    /**
037     * Provides directions on establishing a route.
038     * Implementations of this interface compare a planned route with
039     * a tracked route and indicate the next step required.
040     * 
041     * @author <a href="mailto:rolandw at apache.org">Roland Weber</a>
042     *
043     *
044     * <!-- empty lines to avoid svn diff problems -->
045     * @version $Revision: 620255 $
046     *
047     * @since 4.0
048     */
049    public interface HttpRouteDirector {
050    
051        /** Indicates that the route can not be established at all. */
052        public final static int UNREACHABLE = -1;
053    
054        /** Indicates that the route is complete. */
055        public final static int COMPLETE = 0;
056    
057        /** Step: open connection to target. */
058        public final static int CONNECT_TARGET = 1;
059    
060        /** Step: open connection to proxy. */
061        public final static int CONNECT_PROXY = 2;
062    
063        /** Step: tunnel through proxy to target. */
064        public final static int TUNNEL_TARGET = 3;
065    
066        /** Step: tunnel through proxy to other proxy. */
067        public final static int TUNNEL_PROXY = 4;
068    
069        /** Step: layer protocol (over tunnel). */
070        public final static int LAYER_PROTOCOL = 5;
071    
072    
073        /**
074         * Provides the next step.
075         *
076         * @param plan      the planned route
077         * @param fact      the currently established route, or
078         *                  <code>null</code> if nothing is established
079         *
080         * @return  one of the constants defined in this interface, indicating
081         *          either the next step to perform, or success, or failure.
082         *          0 is for success, a negative value for failure.
083         */
084        public int nextStep(RouteInfo plan, RouteInfo fact)
085            ;
086    
087    
088    } // interface HttpRouteDirector
