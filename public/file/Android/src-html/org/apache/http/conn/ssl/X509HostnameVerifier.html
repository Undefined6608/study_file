032    package org.apache.http.conn.ssl;
033    
034    import javax.net.ssl.HostnameVerifier;
035    import javax.net.ssl.SSLException;
036    import javax.net.ssl.SSLSession;
037    import javax.net.ssl.SSLSocket;
038    import java.io.IOException;
039    import java.security.cert.X509Certificate;
040    
041    /**
042     * Interface for checking if a hostname matches the names stored inside the
043     * server's X.509 certificate.  Implements javax.net.ssl.HostnameVerifier, but
044     * we don't actually use that interface.  Instead we added some methods that
045     * take String parameters (instead of javax.net.ssl.HostnameVerifier's
046     * SSLSession).  JUnit is a lot easier this way!  :-)
047     * <p/>
048     * We provide the HostnameVerifier.DEFAULT, HostnameVerifier.STRICT, and
049     * HostnameVerifier.ALLOW_ALL implementations.  But feel free to define
050     * your own implementation!
051     * <p/>
052     * Inspired by Sebastian Hauer's original StrictSSLProtocolSocketFactory in the
053     * HttpClient "contrib" repository.
054     *
055     * @author Julius Davies
056     * @author <a href="mailto:hauer@psicode.com">Sebastian Hauer</a>
057     *
058     * @since 4.0 (8-Dec-2006)
059     */
060    public interface X509HostnameVerifier extends HostnameVerifier {
061    
062        boolean verify(String host, SSLSession session);
063    
064        void verify(String host, SSLSocket ssl) throws IOException;
065    
066        void verify(String host, X509Certificate cert) throws SSLException;
067    
068        /**
069         * Checks to see if the supplied hostname matches any of the supplied CNs
070         * or "DNS" Subject-Alts.  Most implementations only look at the first CN,
071         * and ignore any additional CNs.  Most implementations do look at all of
072         * the "DNS" Subject-Alts. The CNs or Subject-Alts may contain wildcards
073         * according to RFC 2818.
074         *
075         * @param cns         CN fields, in order, as extracted from the X.509
076         *                    certificate.
077         * @param subjectAlts Subject-Alt fields of type 2 ("DNS"), as extracted
078         *                    from the X.509 certificate.
079         * @param host        The hostname to verify.
080         * @throws SSLException If verification failed.
081         */
082        void verify(String host, String[] cns, String[] subjectAlts)
083              throws SSLException;
084    
085    
086    }
