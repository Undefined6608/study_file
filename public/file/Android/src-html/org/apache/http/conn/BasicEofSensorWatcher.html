031    package org.apache.http.conn;
032    
033    import java.io.InputStream;
034    import java.io.IOException;
035    
036    
037    /**
038     * Basic implementation of {@link EofSensorWatcher EofSensorWatcher}.
039     * The underlying connection is released on close or EOF.
040     *
041     * @author <a href="mailto:rolandw at apache.org">Roland Weber</a>
042     *
043     *
044     * <!-- empty lines to avoid svn diff problems -->
045     * @version $Revision: 672367 $
046     *
047     * @since 4.0
048     */
049    public class BasicEofSensorWatcher implements EofSensorWatcher {
050    
051    
052        /** The connection to auto-release. */
053        protected ManagedClientConnection managedConn;
054    
055        /** Whether to keep the connection alive. */
056        protected boolean attemptReuse;
057    
058    
059    
060        /**
061         * Creates a new watcher for auto-releasing a connection.
062         *
063         * @param conn      the connection to auto-release
064         * @param reuse     whether the connection should be re-used
065         */
066        public BasicEofSensorWatcher(ManagedClientConnection conn,
067                                     boolean reuse) {
068            if (conn == null)
069                throw new IllegalArgumentException
070                    ("Connection may not be null.");
071    
072            managedConn = conn;
073            attemptReuse = reuse;
074        }
075    
076    
077        // non-javadoc, see interface EofSensorWatcher
078        public boolean eofDetected(InputStream wrapped)
079            throws IOException {
080    
081            try {
082                if (attemptReuse) {
083                    // there may be some cleanup required, such as
084                    // reading trailers after the response body:
085                    wrapped.close();
086                    managedConn.markReusable();
087                }
088            } finally {
089                managedConn.releaseConnection();
090            }
091            return false;
092        }
093    
094    
095        // non-javadoc, see interface EofSensorWatcher
096        public boolean streamClosed(InputStream wrapped)
097            throws IOException {
098    
099            try {
100                if (attemptReuse) {
101                    // this assumes that closing the stream will
102                    // consume the remainder of the response body:
103                    wrapped.close();
104                    managedConn.markReusable();
105                }
106            } finally {
107                managedConn.releaseConnection();
108            }
109            return false;
110        }
111    
112    
113        // non-javadoc, see interface EofSensorWatcher
114        public boolean streamAbort(InputStream wrapped)
115            throws IOException {
116    
117            managedConn.abortConnection();
118            return false;
119        }
120    
121    } // class BasicEofSensorWatcher
