031    package org.apache.http.conn.params;
032    
033    import java.util.HashMap;
034    import java.util.Map;
035    
036    import org.apache.http.conn.routing.HttpRoute;
037    
038    /**
039     * This class maintains a map of HTTP routes to maximum number of connections allowed 
040     * for those routes. This class can be used by pooling 
041     * {@link org.apache.http.conn.ClientConnectionManager connection managers} for 
042     * a fine-grained control of connections on a per route basis. 
043     * 
044     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
045     * 
046     * @version $Revision: 652947 $
047     * 
048     * @since 4.0
049     */
050    public final class ConnPerRouteBean implements ConnPerRoute {
051    
052        /** The default maximum number of connections allowed per host */
053        public static final int DEFAULT_MAX_CONNECTIONS_PER_ROUTE = 2;   // Per RFC 2616 sec 8.1.4
054        
055        private final Map<HttpRoute, Integer> maxPerHostMap;
056        
057        private int defaultMax;
058        
059        public ConnPerRouteBean(int defaultMax) {
060            super();
061            this.maxPerHostMap = new HashMap<HttpRoute, Integer>();
062            setDefaultMaxPerRoute(defaultMax);
063        }
064        
065        public ConnPerRouteBean() {
066            this(DEFAULT_MAX_CONNECTIONS_PER_ROUTE);
067        }
068        
069        public int getDefaultMax() {
070            return this.defaultMax;
071        }
072    
073        public void setDefaultMaxPerRoute(int max) {
074            if (max < 1) {
075                throw new IllegalArgumentException
076                    ("The maximum must be greater than 0.");
077            }
078            this.defaultMax = max;
079        }
080    
081        public void setMaxForRoute(final HttpRoute route, int max) {
082            if (route == null) {
083                throw new IllegalArgumentException
084                    ("HTTP route may not be null.");
085            }
086            if (max < 1) {
087                throw new IllegalArgumentException
088                    ("The maximum must be greater than 0.");
089            }
090            this.maxPerHostMap.put(route, Integer.valueOf(max));
091        }
092    
093        public int getMaxForRoute(final HttpRoute route) {
094            if (route == null) {
095                throw new IllegalArgumentException
096                    ("HTTP route may not be null.");
097            }
098            Integer max = this.maxPerHostMap.get(route);
099            if (max != null) {
100                return max.intValue();
101            } else {
102                return this.defaultMax;
103            }
104        }
105        
106        public void setMaxForRoutes(final Map<HttpRoute, Integer> map) {
107            if (map == null) {
108                return;
109            }
110            this.maxPerHostMap.clear();
111            this.maxPerHostMap.putAll(map);
112        }
113        
114    }
