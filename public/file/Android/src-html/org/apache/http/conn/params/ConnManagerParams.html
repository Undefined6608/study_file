031    package org.apache.http.conn.params;
032    
033    import org.apache.http.conn.routing.HttpRoute;
034    import org.apache.http.params.HttpParams;
035    
036    /**
037     * This class represents a collection of HTTP protocol parameters applicable
038     * to client-side
039     * {@link org.apache.http.conn.ClientConnectionManager connection managers}. 
040     * 
041     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
042     * @author Michael Becke
043     * 
044     * @version $Revision: 658785 $
045     * 
046     * @since 4.0
047     *
048     * @see ConnManagerPNames
049     */
050    public final class ConnManagerParams implements ConnManagerPNames {
051    
052        /** The default maximum number of connections allowed overall */
053        public static final int DEFAULT_MAX_TOTAL_CONNECTIONS = 20;
054    
055        /**
056         * Returns the timeout in milliseconds used when retrieving a
057         * {@link org.apache.http.conn.ManagedClientConnection} from the
058         * {@link org.apache.http.conn.ClientConnectionManager}.
059         * 
060         * @return timeout in milliseconds.
061         */ 
062        public static long getTimeout(final HttpParams params) {
063            if (params == null) {
064                throw new IllegalArgumentException("HTTP parameters may not be null");
065            }
066            return params.getLongParameter(TIMEOUT, 0);
067        }
068    
069        /**
070         * Sets the timeout in milliseconds used when retrieving a
071         * {@link org.apache.http.conn.ManagedClientConnection} from the
072         * {@link org.apache.http.conn.ClientConnectionManager}.
073         * 
074         * @param timeout the timeout in milliseconds
075         */ 
076        public static void setTimeout(final HttpParams params, long timeout) {
077            if (params == null) {
078                throw new IllegalArgumentException("HTTP parameters may not be null");
079            }
080            params.setLongParameter(TIMEOUT, timeout);
081        }
082    
083        /** The default maximum number of connections allowed per host */
084        private static final ConnPerRoute DEFAULT_CONN_PER_ROUTE = new ConnPerRoute() {
085            
086            public int getMaxForRoute(HttpRoute route) {
087                return ConnPerRouteBean.DEFAULT_MAX_CONNECTIONS_PER_ROUTE;
088            }
089            
090        };
091        
092        /**
093         * Sets lookup interface for maximum number of connections allowed per route.
094         *
095         * @param params HTTP parameters
096         * @param connPerRoute lookup interface for maximum number of connections allowed 
097         *        per route
098         * 
099         * @see ConnManagerPNames#MAX_CONNECTIONS_PER_ROUTE
100         */
101        public static void setMaxConnectionsPerRoute(final HttpParams params,
102                                                    final ConnPerRoute connPerRoute) {
103            if (params == null) {
104                throw new IllegalArgumentException
105                    ("HTTP parameters must not be null.");
106            }
107            params.setParameter(MAX_CONNECTIONS_PER_ROUTE, connPerRoute);
108        }
109    
110        /**
111         * Returns lookup interface for maximum number of connections allowed per route.
112         *
113         * @param params HTTP parameters
114         * 
115         * @return lookup interface for maximum number of connections allowed per route.
116         * 
117         * @see ConnManagerPNames#MAX_CONNECTIONS_PER_ROUTE
118         */
119        public static ConnPerRoute getMaxConnectionsPerRoute(final HttpParams params) {
120            if (params == null) {
121                throw new IllegalArgumentException
122                    ("HTTP parameters must not be null.");
123            }
124            ConnPerRoute connPerRoute = (ConnPerRoute) params.getParameter(MAX_CONNECTIONS_PER_ROUTE);
125            if (connPerRoute == null) {
126                connPerRoute = DEFAULT_CONN_PER_ROUTE;
127            }
128            return connPerRoute;
129        }
130    
131    
132        /**
133         * Sets the maximum number of connections allowed.
134         *
135         * @param params HTTP parameters
136         * @param maxTotalConnections The maximum number of connections allowed.
137         * 
138         * @see ConnManagerPNames#MAX_TOTAL_CONNECTIONS
139         */
140        public static void setMaxTotalConnections(
141                final HttpParams params,
142                int maxTotalConnections) {
143            if (params == null) {
144                throw new IllegalArgumentException
145                    ("HTTP parameters must not be null.");
146            }
147            params.setIntParameter(MAX_TOTAL_CONNECTIONS, maxTotalConnections);
148        }
149    
150        /**
151         * Gets the maximum number of connections allowed.
152         *
153         * @param params HTTP parameters
154         *
155         * @return The maximum number of connections allowed.
156         * 
157         * @see ConnManagerPNames#MAX_TOTAL_CONNECTIONS
158         */
159        public static int getMaxTotalConnections(
160                final HttpParams params) {
161            if (params == null) {
162                throw new IllegalArgumentException
163                    ("HTTP parameters must not be null.");
164            }
165            return params.getIntParameter(MAX_TOTAL_CONNECTIONS, DEFAULT_MAX_TOTAL_CONNECTIONS);
166        }
167    
168        
169    }
