031    package org.apache.http.conn;
032    
033    import java.io.InputStream;
034    import java.io.IOException;
035    
036    
037    /**
038     * A watcher for {@link EofSensorInputStream EofSensorInputStream}.
039     * Each stream will notify it's watcher at most once.
040     *
041     * @author <a href="mailto:rolandw at apache.org">Roland Weber</a>
042     *
043     *
044     * <!-- empty lines to avoid svn diff problems -->
045     * @version $Revision: 552264 $
046     *
047     * @since 4.0
048     */
049    public interface EofSensorWatcher {
050    
051        /**
052         * Indicates that EOF is detected.
053         *
054         * @param wrapped   the underlying stream which has reached EOF
055         *
056         * @return  <code>true</code> if <code>wrapped</code> should be closed,
057         *          <code>false</code> if it should be left alone
058         *
059         * @throws IOException
060         *         in case of an IO problem, for example if the watcher itself
061         *         closes the underlying stream. The caller will leave the
062         *         wrapped stream alone, as if <code>false</code> was returned.
063         */
064        boolean eofDetected(InputStream wrapped)
065            throws IOException
066            ;
067    
068    
069        /**
070         * Indicates that the {@link EofSensorInputStream stream} is closed.
071         * This method will be called only if EOF was <i>not</i> detected
072         * before closing. Otherwise, {@link #eofDetected eofDetected} is called.
073         *
074         * @param wrapped   the underlying stream which has not reached EOF
075         *
076         * @return  <code>true</code> if <code>wrapped</code> should be closed,
077         *          <code>false</code> if it should be left alone
078         *
079         * @throws IOException
080         *         in case of an IO problem, for example if the watcher itself
081         *         closes the underlying stream. The caller will leave the
082         *         wrapped stream alone, as if <code>false</code> was returned.
083         */
084        boolean streamClosed(InputStream wrapped)
085            throws IOException
086            ;
087    
088    
089        /**
090         * Indicates that the {@link EofSensorInputStream stream} is aborted.
091         * This method will be called only if EOF was <i>not</i> detected
092         * before aborting. Otherwise, {@link #eofDetected eofDetected} is called.
093         * <p/>
094         * This method will also be invoked when an input operation causes an
095         * IOException to be thrown to make sure the input stream gets shut down. 
096         *
097         * @param wrapped   the underlying stream which has not reached EOF
098         *
099         * @return  <code>true</code> if <code>wrapped</code> should be closed,
100         *          <code>false</code> if it should be left alone
101         *
102         * @throws IOException
103         *         in case of an IO problem, for example if the watcher itself
104         *         closes the underlying stream. The caller will leave the
105         *         wrapped stream alone, as if <code>false</code> was returned.
106         */
107        boolean streamAbort(InputStream wrapped)
108            throws IOException
109            ;
110    
111    
112    } // interface EofSensorWatcher
