032    package org.apache.http.conn.ssl;
033    
034    import javax.net.ssl.SSLException;
035    
036    /**
037     * The Strict HostnameVerifier works the same way as Sun Java 1.4, Sun
038     * Java 5, Sun Java 6-rc.  It's also pretty close to IE6.  This
039     * implementation appears to be compliant with RFC 2818 for dealing with
040     * wildcards.
041     * <p/>
042     * The hostname must match either the first CN, or any of the subject-alts.
043     * A wildcard can occur in the CN, and in any of the subject-alts.  The
044     * one divergence from IE6 is how we only check the first CN.  IE6 allows
045     * a match against any of the CNs present.  We decided to follow in
046     * Sun Java 1.4's footsteps and only check the first CN.  (If you need
047     * to check all the CN's, feel free to write your own implementation!).
048     * <p/>
049     * A wildcard such as "*.foo.com" matches only subdomains in the same
050     * level, for example "a.foo.com".  It does not match deeper subdomains
051     * such as "a.b.foo.com".
052     * 
053     * @author Julius Davies
054     */
055    public class StrictHostnameVerifier extends AbstractVerifier {
056    
057        public final void verify(
058                final String host, 
059                final String[] cns,
060                final String[] subjectAlts) throws SSLException {
061            verify(host, cns, subjectAlts, true);
062        }
063    
064        @Override
065        public final String toString() { 
066            return "STRICT"; 
067        }
068        
069    }
