031    package org.apache.http.conn;
032    
033    import java.io.IOException;
034    
035    
036    /**
037     * Interface for releasing a connection.
038     * This can be implemented by various "trigger" objects which are
039     * associated with a connection, for example
040     * a {@link EofSensorInputStream stream}
041     * or an {@link BasicManagedEntity entity}
042     * or the {@link ManagedClientConnection connection} itself.
043     * <br/>
044     * The methods in this interface can safely be called multiple times.
045     * The first invocation releases the connection, subsequent calls
046     * are ignored.
047     *
048     * @author <a href="mailto:rolandw at apache.org">Roland Weber</a>
049     *
050     *
051     * <!-- empty lines to avoid svn diff problems -->
052     * @version $Revision: 672367 $
053     *
054     * @since 4.0
055     */
056    public interface ConnectionReleaseTrigger {
057    
058        /**
059         * Releases the connection with the option of keep-alive. This is a
060         * "graceful" release and may cause IO operations for consuming the
061         * remainder of a response entity. Use
062         * {@link #abortConnection abortConnection} for a hard release. The
063         * connection may be reused as specified by the duration.
064         * 
065         * @throws IOException
066         *             in case of an IO problem. The connection will be released
067         *             anyway.
068         */
069        void releaseConnection()
070            throws IOException
071            ;
072    
073        /**
074         * Releases the connection without the option of keep-alive.
075         * This is a "hard" release that implies a shutdown of the connection.
076         * Use {@link #releaseConnection releaseConnection} for a graceful release.
077         *
078         * @throws IOException      in case of an IO problem.
079         *         The connection will be released anyway.
080         */
081        void abortConnection()
082            throws IOException
083            ;
084    
085    
086    } // interface ConnectionReleaseTrigger
