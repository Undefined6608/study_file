032    package org.apache.http.util;
033    
034    /**
035     * A resizable byte array.
036     *
037     * @author <a href="mailto:oleg@ural.ru">Oleg Kalnichevski</a>
038     * 
039     * @version $Revision: 496070 $
040     * 
041     * @since 4.0
042     */
043    public final class ByteArrayBuffer  {
044        
045        private byte[] buffer;
046        private int len;
047    
048        public ByteArrayBuffer(int capacity) {
049            super();
050            if (capacity < 0) {
051                throw new IllegalArgumentException("Buffer capacity may not be negative");
052            }
053            this.buffer = new byte[capacity]; 
054        }
055    
056        private void expand(int newlen) {
057            byte newbuffer[] = new byte[Math.max(this.buffer.length << 1, newlen)];
058            System.arraycopy(this.buffer, 0, newbuffer, 0, this.len);
059            this.buffer = newbuffer;
060        }
061        
062        public void append(final byte[] b, int off, int len) {
063            if (b == null) {
064                return;
065            }
066            if ((off < 0) || (off > b.length) || (len < 0) ||
067                    ((off + len) < 0) || ((off + len) > b.length)) {
068                throw new IndexOutOfBoundsException();
069            }
070            if (len == 0) {
071                return;
072            }
073            int newlen = this.len + len;
074            if (newlen > this.buffer.length) {
075                expand(newlen);
076            }
077            System.arraycopy(b, off, this.buffer, this.len, len);
078            this.len = newlen;
079        }
080    
081        public void append(int b) {
082            int newlen = this.len + 1;
083            if (newlen > this.buffer.length) {
084                expand(newlen);
085            }
086            this.buffer[this.len] = (byte)b;
087            this.len = newlen;
088        }
089    
090        public void append(final char[] b, int off, int len) {
091            if (b == null) {
092                return;
093            }
094            if ((off < 0) || (off > b.length) || (len < 0) ||
095                    ((off + len) < 0) || ((off + len) > b.length)) {
096                throw new IndexOutOfBoundsException();
097            }
098            if (len == 0) {
099                return;
100            }
101            int oldlen = this.len;
102            int newlen = oldlen + len;
103            if (newlen > this.buffer.length) {
104                expand(newlen);
105            }
106            for (int i1 = off, i2 = oldlen; i2 < newlen; i1++, i2++) {
107                this.buffer[i2] = (byte) b[i1];
108            }
109            this.len = newlen;
110        }
111    
112        public void append(final CharArrayBuffer b, int off, int len) {
113            if (b == null) {
114                return;
115            }
116            append(b.buffer(), off, len);
117        }
118        
119        public void clear() {
120            this.len = 0;
121        }
122        
123        public byte[] toByteArray() {
124            byte[] b = new byte[this.len]; 
125            if (this.len > 0) {
126                System.arraycopy(this.buffer, 0, b, 0, this.len);
127            }
128            return b;
129        }
130        
131        public int byteAt(int i) {
132            return this.buffer[i];
133        }
134        
135        public int capacity() {
136            return this.buffer.length;
137        }
138        
139        public int length() {
140            return this.len;
141        }
142    
143        public byte[] buffer() {
144            return this.buffer;
145        }
146            
147        public void setLength(int len) {
148            if (len < 0 || len > this.buffer.length) {
149                throw new IndexOutOfBoundsException();
150            }
151            this.len = len;
152        }
153        
154        public boolean isEmpty() {
155            return this.len == 0; 
156        }
157        
158        public boolean isFull() {
159            return this.len == this.buffer.length; 
160        }
161        
162    }
