032    package org.apache.http.util;
033    
034    /**
035     * A set of utility methods to help produce consistent
036     * {@link Object#equals equals} and {@link Object#hashCode hashCode} methods.
037     *  
038     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
039     * 
040     * @since 4.0
041     */
042    public final class LangUtils {
043        
044        public static final int HASH_SEED = 17;
045        public static final int HASH_OFFSET = 37;
046    
047        /** Disabled default constructor. */
048        private LangUtils() {
049        }
050    
051        public static int hashCode(final int seed, final int hashcode) {
052            return seed * HASH_OFFSET + hashcode;
053        }
054    
055        public static int hashCode(final int seed, final boolean b) {
056            return hashCode(seed, b ? 1 : 0);
057        }
058        
059        public static int hashCode(final int seed, final Object obj) {
060            return hashCode(seed, obj != null ? obj.hashCode() : 0);
061        }
062        
063        public static boolean equals(final Object obj1, final Object obj2) {
064            return obj1 == null ? obj2 == null : obj1.equals(obj2);
065        }
066    
067        public static boolean equals(final Object[] a1, final Object[] a2) {
068            if (a1 == null) {
069                if (a2 == null) {
070                    return true;
071                } else {
072                    return false;
073                }
074            } else {
075                if (a2 != null && a1.length == a2.length) {
076                    for (int i = 0; i < a1.length; i++) {
077                        if (!equals(a1[i], a2[i])) {
078                            return false;
079                        }
080                    }
081                    return true;
082                } else {
083                    return false;
084                }
085            }
086        }
087        
088    }
