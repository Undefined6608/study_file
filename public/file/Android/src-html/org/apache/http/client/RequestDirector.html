032    package org.apache.http.client;
033    
034    import java.io.IOException;
035    
036    import org.apache.http.HttpHost;
037    import org.apache.http.HttpRequest;
038    import org.apache.http.HttpResponse;
039    import org.apache.http.HttpException;
040    import org.apache.http.protocol.HttpContext;
041    
042    /**
043     * A client-side request director.
044     * The director decides which steps are necessary to execute a request.
045     * It establishes connections and optionally processes redirects and
046     * authentication challenges. The director may therefore generate and
047     * send a sequence of requests in order to execute one initial request.
048     *
049     * <br/><b>Note:</b>
050     * It is most likely that implementations of this interface will
051     * allocate connections, and return responses that depend on those
052     * connections for reading the response entity. Such connections
053     * MUST be released, but that is out of the scope of a request director.
054     *
055     * @author <a href="mailto:rolandw at apache.org">Roland Weber</a>
056     *
057     *
058     * <!-- empty lines to avoid svn diff problems -->
059     * @version $Revision: 676020 $
060     *
061     * @since 4.0
062     */
063    public interface RequestDirector {
064    
065    
066        /**
067         * Executes a request.
068         * <br/><b>Note:</b>
069         * For the time being, a new director is instantiated for each request.
070         * This is the same behavior as for <code>HttpMethodDirector</code>
071         * in HttpClient 3.
072         *
073         * @param target    the target host for the request.
074         *                  Implementations may accept <code>null</code>
075         *                  if they can still determine a route, for example
076         *                  to a default target or by inspecting the request.
077         * @param request   the request to execute
078         * @param context   the context for executing the request
079         *
080         * @return  the final response to the request.
081         *          This is never an intermediate response with status code 1xx.
082         *
083         * @throws HttpException            in case of a problem
084         * @throws IOException              in case of an IO problem
085         *                                     or if the connection was aborted
086         */
087        HttpResponse execute(HttpHost target, HttpRequest request,
088                             HttpContext context)
089            throws HttpException, IOException
090            ;
091    
092    } // class ClientRequestDirector
