032    package org.apache.http.client.methods;
033    
034    import java.net.URI;
035    
036    /**
037     * HTTP POST method.
038     * <p>
039     * The HTTP POST method is defined in section 9.5 of 
040     * <a href="http://www.ietf.org/rfc/rfc2616.txt">RFC2616</a>:
041     * <blockquote>
042     * The POST method is used to request that the origin server accept the entity
043     * enclosed in the request as a new subordinate of the resource identified by
044     * the Request-URI in the Request-Line. POST is designed to allow a uniform
045     * method to cover the following functions:
046     * <ul>
047     *   <li>Annotation of existing resources</li>
048     *   <li>Posting a message to a bulletin board, newsgroup, mailing list, or 
049     *     similar group of articles</li>
050     *   <li>Providing a block of data, such as the result of submitting a form,
051     *     to a data-handling process</li>
052     *   <li>Extending a database through an append operation</li>
053     * </ul>
054     * </blockquote>
055     * </p>
056     *
057     * @version $Revision: 664505 $
058     * 
059     * @since 4.0
060     */
061    public class HttpPost extends HttpEntityEnclosingRequestBase {
062    
063        public final static String METHOD_NAME = "POST";
064        
065        public HttpPost() {
066            super();
067        }
068        
069        public HttpPost(final URI uri) {
070            super();
071            setURI(uri);
072        }
073        
074        /**
075         * @throws IllegalArgumentException if the uri is invalid. 
076         */
077        public HttpPost(final String uri) {
078            super();
079            setURI(URI.create(uri));
080        }
081    
082        @Override
083        public String getMethod() {
084            return METHOD_NAME;
085        }
086        
087    }
