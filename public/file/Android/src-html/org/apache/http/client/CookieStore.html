031    package org.apache.http.client;
032    
033    import java.util.Date;
034    import java.util.List;
035    
036    import org.apache.http.cookie.Cookie;
037    
038    /**
039     * Abstract cookie store.
040     * 
041     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
042     * 
043     * @since 4.0
044     */
045    public interface CookieStore {
046    
047        /**
048         * Adds an {@link Cookie HTTP cookie}, replacing any existing equivalent cookies.
049         * If the given cookie has already expired it will not be added, but existing 
050         * values will still be removed.
051         * 
052         * @param cookie the {@link Cookie cookie} to be added
053         */
054        void addCookie(Cookie cookie);
055    
056        /**
057         * Returns all cookies contained in this store.
058         * 
059         * @return all cookies
060         */
061        List<Cookie> getCookies();
062    
063        /**
064         * Removes all of {@link Cookie cookies} in this store that have expired by 
065         * the specified {@link java.util.Date date}. 
066         * 
067         * @return true if any cookies were purged.
068         */
069        boolean clearExpired(Date date);
070    
071        /**
072         * Clears all cookies.
073         */
074        void clear();
075        
076    }
