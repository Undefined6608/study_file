032    package org.apache.http.client.methods;
033    
034    import java.io.IOException;
035    
036    import org.apache.http.client.HttpClient;
037    import org.apache.http.conn.ClientConnectionManager;
038    import org.apache.http.conn.ClientConnectionRequest;
039    import org.apache.http.conn.ConnectionReleaseTrigger;
040    import org.apache.http.conn.ManagedClientConnection;
041    import org.apache.http.impl.conn.tsccm.ThreadSafeClientConnManager;
042    
043    /**
044     * Interface representing an HTTP request that can be aborted by shutting 
045     * down the underlying HTTP connection.
046     *
047     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
048     *
049     * <!-- empty lines to avoid svn diff problems -->
050     * @version   $Revision: 639600 $
051     *
052     * @since 4.0
053     */
054    public interface AbortableHttpRequest {
055    
056        /**
057         * Sets the {@link ClientConnectionRequest} callback that can be
058         * used to abort a long-lived request for a connection.
059         * If the request is already aborted, throws an {@link IOException}.
060         * 
061         * @see ClientConnectionManager
062         * @see ThreadSafeClientConnManager
063         */
064        void setConnectionRequest(ClientConnectionRequest connRequest) throws IOException;
065        
066        /**
067         * Sets the {@link ConnectionReleaseTrigger} callback that can
068         * be used to abort an active connection.
069         * Typically, this will be the {@link ManagedClientConnection} itself.
070         * If the request is already aborted, throws an {@link IOException}.
071         */
072        void setReleaseTrigger(ConnectionReleaseTrigger releaseTrigger) throws IOException;
073    
074        /**
075         * Aborts this http request. Any active execution of this method should
076         * return immediately. If the request has not started, it will abort after
077         * the next execution. Aborting this request will cause all subsequent
078         * executions with this request to fail.
079         * 
080         * @see HttpClient#execute(HttpUriRequest)
081         * @see HttpClient#execute(org.apache.http.HttpHost,
082         *      org.apache.http.HttpRequest)
083         * @see HttpClient#execute(HttpUriRequest,
084         *      org.apache.http.protocol.HttpContext)
085         * @see HttpClient#execute(org.apache.http.HttpHost,
086         *      org.apache.http.HttpRequest, org.apache.http.protocol.HttpContext)
087         */
088        void abort();
089    
090    }
