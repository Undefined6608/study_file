031    package org.apache.http.client;
032    
033    /**
034     * Signals a circular redirect
035     * 
036     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
037     * 
038     * @since 4.0
039     */
040    public class CircularRedirectException extends RedirectException {
041    
042        private static final long serialVersionUID = 6830063487001091803L;
043    
044        /**
045         * Creates a new CircularRedirectException with a <tt>null</tt> detail message. 
046         */
047        public CircularRedirectException() {
048            super();
049        }
050    
051        /**
052         * Creates a new CircularRedirectException with the specified detail message.
053         * 
054         * @param message The exception detail message
055         */
056        public CircularRedirectException(String message) {
057            super(message);
058        }
059    
060        /**
061         * Creates a new CircularRedirectException with the specified detail message and cause.
062         * 
063         * @param message the exception detail message
064         * @param cause the <tt>Throwable</tt> that caused this exception, or <tt>null</tt>
065         * if the cause is unavailable, unknown, or not a <tt>Throwable</tt>
066         */
067        public CircularRedirectException(String message, Throwable cause) {
068            super(message, cause);
069        }
070    }
