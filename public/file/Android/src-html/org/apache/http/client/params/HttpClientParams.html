031    package org.apache.http.client.params;
032    
033    import org.apache.http.params.HttpParams;
034    
035    /**
036     * An adaptor for accessing HTTP client parameters in {@link HttpParams}.
037     * 
038     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
039     * 
040     * @version $Revision: 659595 $
041     * 
042     * @since 4.0
043     */
044    public class HttpClientParams {
045    
046        private HttpClientParams() {
047            super();
048        }
049    
050        public static boolean isRedirecting(final HttpParams params) {
051            if (params == null) {
052                throw new IllegalArgumentException("HTTP parameters may not be null");
053            }
054            return params.getBooleanParameter
055                (ClientPNames.HANDLE_REDIRECTS, true); 
056        }
057    
058        public static void setRedirecting(final HttpParams params, boolean value) {
059            if (params == null) {
060                throw new IllegalArgumentException("HTTP parameters may not be null");
061            }
062            params.setBooleanParameter
063                (ClientPNames.HANDLE_REDIRECTS, value); 
064        }
065        
066        public static boolean isAuthenticating(final HttpParams params) {
067            if (params == null) {
068                throw new IllegalArgumentException("HTTP parameters may not be null");
069            }
070            return params.getBooleanParameter
071                (ClientPNames.HANDLE_AUTHENTICATION, true); 
072        }
073    
074        public static void setAuthenticating(final HttpParams params, boolean value) {
075            if (params == null) {
076                throw new IllegalArgumentException("HTTP parameters may not be null");
077            }
078            params.setBooleanParameter
079                (ClientPNames.HANDLE_AUTHENTICATION, value); 
080        }
081        
082        public static String getCookiePolicy(final HttpParams params) { 
083            if (params == null) {
084                throw new IllegalArgumentException("HTTP parameters may not be null");
085            }
086            String cookiePolicy = (String)
087                params.getParameter(ClientPNames.COOKIE_POLICY);
088            if (cookiePolicy == null) {
089                return CookiePolicy.BEST_MATCH;
090            }
091            return cookiePolicy;
092        }
093        
094        public static void setCookiePolicy(final HttpParams params, final String cookiePolicy) {
095            if (params == null) {
096                throw new IllegalArgumentException("HTTP parameters may not be null");
097            }
098            params.setParameter(ClientPNames.COOKIE_POLICY, cookiePolicy);
099        }
100    
101    }
