032    package org.apache.http.client.methods;
033    
034    import java.net.URI;
035    
036    /**
037     * HTTP GET method.
038     * <p>
039     * The HTTP GET method is defined in section 9.3 of
040     * <a href="http://www.ietf.org/rfc/rfc2616.txt">RFC2616</a>:
041     * <blockquote>
042     * The GET method means retrieve whatever information (in the form of an
043     * entity) is identified by the Request-URI. If the Request-URI refers
044     * to a data-producing process, it is the produced data which shall be
045     * returned as the entity in the response and not the source text of the
046     * process, unless that text happens to be the output of the process.
047     * </blockquote>
048     * </p>
049     * <p>
050     * GetMethods will follow redirect requests from the http server by default.
051     * This behavour can be disabled by calling setFollowRedirects(false).</p>
052     * 
053     * @version $Revision: 664505 $
054     * 
055     * @since 4.0
056     */
057    public class HttpGet extends HttpRequestBase {
058    
059        public final static String METHOD_NAME = "GET";
060        
061        public HttpGet() {
062            super();
063        }
064    
065        public HttpGet(final URI uri) {
066            super();
067            setURI(uri);
068        }
069    
070        /**
071         * @throws IllegalArgumentException if the uri is invalid. 
072         */
073        public HttpGet(final String uri) {
074            super();
075            setURI(URI.create(uri));
076        }
077    
078        @Override
079        public String getMethod() {
080            return METHOD_NAME;
081        }
082        
083    }
