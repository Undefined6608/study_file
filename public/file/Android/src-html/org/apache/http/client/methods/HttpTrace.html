032    package org.apache.http.client.methods;
033    
034    import java.net.URI;
035    
036    /**
037     * HTTP TRACE method.
038     * <p>
039     * The HTTP TRACE method is defined in section 9.6 of 
040     * <a href="http://www.ietf.org/rfc/rfc2616.txt">RFC2616</a>:
041     * <blockquote>
042     *  The TRACE method is used to invoke a remote, application-layer loop-
043     *  back of the request message. The final recipient of the request
044     *  SHOULD reflect the message received back to the client as the
045     *  entity-body of a 200 (OK) response. The final recipient is either the
046     *  origin server or the first proxy or gateway to receive a Max-Forwards
047     *  value of zero (0) in the request (see section 14.31). A TRACE request
048     *  MUST NOT include an entity.
049     * </blockquote>
050     * </p>
051     * 
052     * @version $Revision: 664505 $
053     * 
054     * @since 4.0
055     */
056    public class HttpTrace extends HttpRequestBase {
057    
058        public final static String METHOD_NAME = "TRACE";
059        
060        public HttpTrace() {
061            super();
062        }
063    
064        public HttpTrace(final URI uri) {
065            super();
066            setURI(uri);
067        }
068        
069        /**
070         * @throws IllegalArgumentException if the uri is invalid. 
071         */
072        public HttpTrace(final String uri) {
073            super();
074            setURI(URI.create(uri));
075        }
076    
077        @Override
078        public String getMethod() {
079            return METHOD_NAME;
080        }
081        
082    }
