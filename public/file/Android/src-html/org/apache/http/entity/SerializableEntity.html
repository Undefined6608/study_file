032    package org.apache.http.entity;
033    
034    import java.io.ByteArrayInputStream;
035    import java.io.ByteArrayOutputStream;
036    import java.io.IOException;
037    import java.io.InputStream;
038    import java.io.ObjectOutputStream;
039    import java.io.OutputStream;
040    import java.io.Serializable;
041    
042    public class SerializableEntity extends AbstractHttpEntity {
043        
044        private byte[] objSer;
045        
046        private Serializable objRef;
047        
048        public SerializableEntity(Serializable ser, boolean bufferize) throws IOException {
049            super();
050            if (ser == null) {
051                throw new IllegalArgumentException("Source object may not be null");
052            }
053            
054            if (bufferize) {
055                createBytes(ser);
056            } else {
057                this.objRef = ser;
058            }
059        }
060        
061        private void createBytes(Serializable ser) throws IOException {
062            ByteArrayOutputStream baos = new ByteArrayOutputStream();
063            ObjectOutputStream out = new ObjectOutputStream(baos);
064            out.writeObject(ser);
065            out.flush();    
066            this.objSer = baos.toByteArray();
067        }
068        
069        public InputStream getContent() throws IOException, IllegalStateException {
070            if (this.objSer == null) {
071                createBytes(this.objRef);
072            }
073            return new ByteArrayInputStream(this.objSer);
074        }
075    
076        public long getContentLength() {
077            if (this.objSer ==  null) { 
078                return -1;
079            } else {
080                return this.objSer.length;
081            }
082        }
083    
084        public boolean isRepeatable() {
085            return true;
086        }
087    
088        public boolean isStreaming() {
089            return this.objSer == null;
090        }
091    
092        public void writeTo(OutputStream outstream) throws IOException {
093            if (outstream == null) {
094                throw new IllegalArgumentException("Output stream may not be null");
095            }
096            
097            if (this.objSer == null) {
098                ObjectOutputStream out = new ObjectOutputStream(outstream);
099                out.writeObject(this.objRef);
100                out.flush();
101            } else {
102                outstream.write(this.objSer);
103                outstream.flush();
104            }
105        }
106    
107    }
