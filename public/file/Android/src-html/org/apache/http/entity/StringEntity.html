032    package org.apache.http.entity;
033    
034    import java.io.ByteArrayInputStream;
035    import java.io.IOException;
036    import java.io.InputStream;
037    import java.io.OutputStream;
038    import java.io.UnsupportedEncodingException;
039    
040    import org.apache.http.protocol.HTTP;
041    
042    /**
043     *  An entity whose content is retrieved from a string.
044     *
045     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
046     *
047     * @version $Revision: 618367 $
048     * 
049     * @since 4.0
050     */
051    public class StringEntity extends AbstractHttpEntity implements Cloneable {
052    
053        protected final byte[] content;
054    
055        public StringEntity(final String s, String charset) 
056                throws UnsupportedEncodingException {
057            super();
058            if (s == null) {
059                throw new IllegalArgumentException("Source string may not be null");
060            }
061            if (charset == null) {
062                charset = HTTP.DEFAULT_CONTENT_CHARSET;
063            }
064            this.content = s.getBytes(charset);
065            setContentType(HTTP.PLAIN_TEXT_TYPE + HTTP.CHARSET_PARAM + charset);
066        }
067    
068        public StringEntity(final String s) 
069                throws UnsupportedEncodingException {
070            this(s, null);
071        }
072    
073        public boolean isRepeatable() {
074            return true;
075        }
076    
077        public long getContentLength() {
078            return this.content.length;
079        }
080        
081        public InputStream getContent() throws IOException {
082            return new ByteArrayInputStream(this.content);
083        }
084        
085        public void writeTo(final OutputStream outstream) throws IOException {
086            if (outstream == null) {
087                throw new IllegalArgumentException("Output stream may not be null");
088            }
089            outstream.write(this.content);
090            outstream.flush();
091        }
092    
093        /**
094         * Tells that this entity is not streaming.
095         *
096         * @return <code>false</code>
097         */
098        public boolean isStreaming() {
099            return false;
100        }
101    
102        public Object clone() throws CloneNotSupportedException {
103            return super.clone();
104        }
105    
106    } // class StringEntity
