032    package org.apache.http.entity;
033    
034    import java.io.IOException;
035    import java.io.InputStream;
036    import java.io.OutputStream;
037    
038    /**
039     * Entity that delegates the process of content generation to an abstract
040     * content producer.
041     *
042     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
043     *
044     * @version $Revision: 496070 $
045     * 
046     * @since 4.0
047     */
048    public class EntityTemplate extends AbstractHttpEntity {
049    
050        private final ContentProducer contentproducer;
051        
052        public EntityTemplate(final ContentProducer contentproducer) {
053            super();
054            if (contentproducer == null) {
055                throw new IllegalArgumentException("Content producer may not be null");
056            }
057            this.contentproducer = contentproducer; 
058        }
059    
060        public long getContentLength() {
061            return -1;
062        }
063    
064        public InputStream getContent() {
065            throw new UnsupportedOperationException("Entity template does not implement getContent()");
066        }
067    
068        public boolean isRepeatable() {
069            return true;
070        }
071    
072        public void writeTo(final OutputStream outstream) throws IOException {
073            if (outstream == null) {
074                throw new IllegalArgumentException("Output stream may not be null");
075            }
076            this.contentproducer.writeTo(outstream);
077        }
078    
079        public boolean isStreaming() {
080            return true;
081        }
082    
083        public void consumeContent() throws IOException {
084        }
085        
086    }
