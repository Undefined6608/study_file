032    package org.apache.http.protocol;
033    
034    import java.util.HashMap;
035    import java.util.Map;
036    
037    /**
038     * Default implementation of the {@link HttpContext HttpContext}.
039     *
040     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
041     *
042     * @version $Revision: 654882 $
043     * 
044     * @since 4.0
045     */
046    public class BasicHttpContext implements HttpContext {
047        
048        private final HttpContext parentContext;
049        private Map map = null;
050        
051        public BasicHttpContext() {
052            this(null);
053        }
054        
055        public BasicHttpContext(final HttpContext parentContext) {
056            super();
057            this.parentContext = parentContext;
058        }
059        
060        public Object getAttribute(final String id) {
061            if (id == null) {
062                throw new IllegalArgumentException("Id may not be null");
063            }
064            Object obj = null;
065            if (this.map != null) {
066                obj = this.map.get(id);
067            }
068            if (obj == null && this.parentContext != null) {
069                obj = this.parentContext.getAttribute(id);
070            }
071            return obj;
072        }
073    
074        public void setAttribute(final String id, final Object obj) {
075            if (id == null) {
076                throw new IllegalArgumentException("Id may not be null");
077            }
078            if (this.map == null) {
079                this.map = new HashMap();
080            }
081            this.map.put(id, obj);
082        }
083        
084        public Object removeAttribute(final String id) {
085            if (id == null) {
086                throw new IllegalArgumentException("Id may not be null");
087            }
088            if (this.map != null) {
089                return this.map.remove(id);
090            } else {
091                return null;
092            }
093        }
094    
095    }
