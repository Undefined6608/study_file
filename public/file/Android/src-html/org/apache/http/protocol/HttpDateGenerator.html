032    package org.apache.http.protocol;
033    
034    import java.text.DateFormat;
035    import java.text.SimpleDateFormat;
036    import java.util.Date;
037    import java.util.Locale;
038    import java.util.TimeZone;
039    
040    
041    /**
042     * Generates a date in the format required by the HTTP protocol.
043     *
044     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
045     *
046     * @version $Revision: 548066 $
047     * 
048     * @since 4.0
049     */
050    public class HttpDateGenerator {
051    
052        /** Date format pattern used to generate the header in RFC 1123 format. */
053        public static final
054            String PATTERN_RFC1123 = "EEE, dd MMM yyyy HH:mm:ss zzz";
055    
056        /** The time zone to use in the date header. */
057        public static final TimeZone GMT = TimeZone.getTimeZone("GMT");
058    
059    
060        private final DateFormat dateformat;
061        
062        private long dateAsLong = 0L;
063        private String dateAsText = null;
064    
065        public HttpDateGenerator() {
066            super();
067            this.dateformat = new SimpleDateFormat(PATTERN_RFC1123, Locale.US);
068            this.dateformat.setTimeZone(GMT);
069        }
070        
071        public synchronized String getCurrentDate() {
072            long now = System.currentTimeMillis();
073            if (now - this.dateAsLong > 1000) {
074                // Generate new date string
075                this.dateAsText = this.dateformat.format(new Date(now));
076                this.dateAsLong = now;
077            }
078            return this.dateAsText;
079        }
080        
081    }
