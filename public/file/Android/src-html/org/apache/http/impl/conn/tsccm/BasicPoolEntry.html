031    package org.apache.http.impl.conn.tsccm;
032    
033    
034    import java.lang.ref.ReferenceQueue;
035    
036    import org.apache.http.conn.OperatedClientConnection;
037    import org.apache.http.conn.ClientConnectionOperator;
038    import org.apache.http.conn.routing.HttpRoute;
039    import org.apache.http.impl.conn.AbstractPoolEntry;
040    
041    
042    
043    /**
044     * Basic implementation of a connection pool entry.
045     */
046    public class BasicPoolEntry extends AbstractPoolEntry {
047    
048        /**
049         * A weak reference to <code>this</code> used to detect GC of entries.
050         * Pool entries can only be GCed when they are allocated by an application
051         * and therefore not referenced with a hard link in the manager.
052         */
053        private final BasicPoolEntryRef reference;
054    
055        /**
056         * Creates a new pool entry.
057         *
058         * @param op      the connection operator
059         * @param route   the planned route for the connection
060         * @param queue   the reference queue for tracking GC of this entry,
061         *                or <code>null</code>
062         */
063        public BasicPoolEntry(ClientConnectionOperator op,
064                              HttpRoute route,
065                              ReferenceQueue<Object> queue) {
066            super(op, route);
067            if (route == null) {
068                throw new IllegalArgumentException("HTTP route may not be null");
069            }
070            this.reference = new BasicPoolEntryRef(this, queue);
071        }
072    
073        protected final OperatedClientConnection getConnection() {
074            return super.connection;
075        }
076    
077        protected final HttpRoute getPlannedRoute() {
078            return super.route;
079        }
080    
081        protected final BasicPoolEntryRef getWeakRef() {
082            return this.reference;
083        }
084    
085    
086    } // class BasicPoolEntry
