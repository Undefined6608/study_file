032    package org.apache.http.impl.cookie;
033    
034    import java.util.Date;
035    
036    import org.apache.http.cookie.SetCookie2;
037    
038    /**
039     * HTTP "magic-cookie" represents a piece of state information
040     * that the HTTP agent and the target server can exchange to maintain 
041     * a session as specified by RFC2965.
042     * 
043     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
044     */
045    public class BasicClientCookie2 extends BasicClientCookie implements SetCookie2 {
046    
047        private String commentURL;
048        private int[] ports;
049        private boolean discard;
050        
051        /**
052         * Default Constructor taking a name and a value. The value may be null.
053         * 
054         * @param name The name.
055         * @param value The value.
056         */
057        public BasicClientCookie2(final String name, final String value) {
058            super(name, value);
059        }
060    
061        @Override
062        public int[] getPorts() {
063            return this.ports;
064        }
065    
066        public void setPorts(final int[] ports) {
067            this.ports = ports;
068        }
069        
070        @Override
071        public String getCommentURL() {
072            return this.commentURL;
073        }
074    
075        public void setCommentURL(final String commentURL) {
076            this.commentURL = commentURL;
077        }
078    
079        public void setDiscard(boolean discard) {
080            this.discard = discard;
081        }
082    
083        @Override
084        public boolean isPersistent() {
085            return !this.discard && super.isPersistent();
086        }
087    
088        @Override
089        public boolean isExpired(final Date date) {
090            return this.discard || super.isExpired(date);
091        }
092     
093        @Override
094        public Object clone() throws CloneNotSupportedException {
095            BasicClientCookie2 clone = (BasicClientCookie2) super.clone();
096            clone.ports = this.ports.clone();
097            return clone;
098        }
099        
100    }
