031    package org.apache.http.impl.client;
032    
033    import java.util.HashMap;
034    
035    import org.apache.http.auth.AuthScope;
036    import org.apache.http.auth.Credentials;
037    import org.apache.http.client.CredentialsProvider;
038    
039    /**
040     * Default implementation of {@link CredentialsProvider}
041     * 
042     * @author <a href="mailto:remm@apache.org">Remy Maucherat</a>
043     * @author Rodney Waldhoff
044     * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
045     * @author Sean C. Sullivan
046     * @author <a href="mailto:becke@u.washington.edu">Michael Becke</a>
047     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
048     * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
049     * @author <a href="mailto:adrian@intencha.com">Adrian Sutton</a>
050     * 
051     * @since 4.0
052     */
053    public class BasicCredentialsProvider implements CredentialsProvider {
054    
055        private final HashMap<AuthScope, Credentials> credMap;
056    
057        /**
058         * Default constructor.
059         */
060        public BasicCredentialsProvider() {
061            super();
062            this.credMap = new HashMap<AuthScope, Credentials>();
063        }
064    
065        /** 
066         * Sets the {@link Credentials credentials} for the given authentication 
067         * scope. Any previous credentials for the given scope will be overwritten.
068         * 
069         * @param authscope the {@link AuthScope authentication scope}
070         * @param credentials the authentication {@link Credentials credentials} 
071         * for the given scope.
072         * 
073         * @see #getCredentials(AuthScope)
074         */
075        public synchronized void setCredentials(
076                final AuthScope authscope, 
077                final Credentials credentials) {
078            if (authscope == null) {
079                throw new IllegalArgumentException("Authentication scope may not be null");
080            }
081            credMap.put(authscope, credentials);
082        }
083    
084        /**
085         * Find matching {@link Credentials credentials} for the given authentication scope.
086         *
087         * @param map the credentials hash map
088         * @param authscope the {@link AuthScope authentication scope}
089         * @return the credentials 
090         * 
091         */
092        private static Credentials matchCredentials(
093                final HashMap<AuthScope, Credentials> map, 
094                final AuthScope authscope) {
095            // see if we get a direct hit
096            Credentials creds = map.get(authscope);
097            if (creds == null) {
098                // Nope.
099                // Do a full scan
100                int bestMatchFactor  = -1;
101                AuthScope bestMatch  = null;
102                for (AuthScope current: map.keySet()) {
103                    int factor = authscope.match(current);
104                    if (factor > bestMatchFactor) {
105                        bestMatchFactor = factor;
106                        bestMatch = current;
107                    }
108                }
109                if (bestMatch != null) {
110                    creds = map.get(bestMatch);
111                }
112            }
113            return creds;
114        }
115        
116        /**
117         * Get the {@link Credentials credentials} for the given authentication scope.
118         *
119         * @param authscope the {@link AuthScope authentication scope}
120         * @return the credentials 
121         * 
122         * @see #setCredentials(AuthScope, Credentials)
123         */
124        public synchronized Credentials getCredentials(final AuthScope authscope) {
125            if (authscope == null) {
126                throw new IllegalArgumentException("Authentication scope may not be null");
127            }
128            return matchCredentials(this.credMap, authscope);
129        }
130    
131        @Override
132        public String toString() {
133            return credMap.toString();
134        }
135        
136        /**
137         * Clears all credentials.
138         */
139        public synchronized void clear() {
140            this.credMap.clear();
141        }
142        
143    }
