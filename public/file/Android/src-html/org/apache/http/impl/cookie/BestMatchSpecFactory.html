032    package org.apache.http.impl.cookie;
033    
034    import org.apache.http.cookie.CookieSpec;
035    import org.apache.http.cookie.CookieSpecFactory;
036    import org.apache.http.cookie.params.CookieSpecPNames;
037    import org.apache.http.params.HttpParams;
038    
039    /**
040     * 
041     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
042     *
043     * @since 4.0
044     */
045    public class BestMatchSpecFactory implements CookieSpecFactory {    
046    
047        public CookieSpec newInstance(final HttpParams params) {
048            if (params != null) {
049                return new BestMatchSpec(
050                        (String []) params.getParameter(CookieSpecPNames.DATE_PATTERNS), 
051                        params.getBooleanParameter(CookieSpecPNames.SINGLE_COOKIE_HEADER, false));
052            } else {
053                return new BestMatchSpec();
054            }
055        }
056    
057    }
