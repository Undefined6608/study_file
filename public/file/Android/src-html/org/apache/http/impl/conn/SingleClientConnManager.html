049    /**
050     * A connection "manager" for a single connection.
051     * This manager is good only for single-threaded use.
052     * Allocation <i>always</i> returns the connection immediately,
053     * even if it has not been released after the previous allocation.
054     * In that case, a {@link #MISUSE_MESSAGE warning} is logged
055     * and the previously issued connection is revoked.
056     * <p>
057     * This class is derived from <code>SimpleHttpConnectionManager</code>
058     * in HttpClient 3. See there for original authors.
059     * </p>
060     *
061     * @author <a href="mailto:rolandw at apache.org">Roland Weber</a>
062     * @author <a href="mailto:becke@u.washington.edu">Michael Becke</a>
063     *
064     *
065     * <!-- empty lines to avoid svn diff problems -->
066     * @version   $Revision: 673450 $
067     *
068     * @since 4.0
069     */
070    public class SingleClientConnManager implements ClientConnectionManager {
071    
072        private final Log log = LogFactory.getLog(getClass());
073    
074        /** The message to be logged on multiple allocation. */
075        public final static String MISUSE_MESSAGE =
076        "Invalid use of SingleClientConnManager: connection still allocated.\n" +
077        "Make sure to release the connection before allocating another one.";
078    
079    
080        /** The schemes supported by this connection manager. */
081        protected SchemeRegistry schemeRegistry; 
082        
083        /** The operator for opening and updating connections. */
084        protected ClientConnectionOperator connOperator;
085    
086        /** The one and only entry in this pool. */
087        protected PoolEntry uniquePoolEntry;
088    
089        /** The currently issued managed connection, if any. */
090        protected ConnAdapter managedConn;
091    
092        /** The time of the last connection release, or -1. */
093        protected long lastReleaseTime;
094        
095        /** The time the last released connection expires and shouldn't be reused. */
096        protected long connectionExpiresTime;
097    
098        /** Whether the connection should be shut down  on release. */
099        protected boolean alwaysShutDown;
100    
101        /** Indicates whether this connection manager is shut down. */
102        protected volatile boolean isShutDown;
103    
104    
105    
106    
107        /**
108         * Creates a new simple connection manager.
109         *
110         * @param params    the parameters for this manager
111         * @param schreg    the scheme registry, or
112         *                  <code>null</code> for the default registry
113         */
114        public SingleClientConnManager(HttpParams params,
115                                       SchemeRegistry schreg) {
116    
117            if (schreg == null) {
118                throw new IllegalArgumentException
119                    ("Scheme registry must not be null.");
120            }
121            this.schemeRegistry  = schreg;
122            this.connOperator    = createConnectionOperator(schreg);
123            this.uniquePoolEntry = new PoolEntry();
124            this.managedConn     = null;
125            this.lastReleaseTime = -1L;
126            this.alwaysShutDown  = false; //@@@ from params? as argument?
127            this.isShutDown      = false;
128    
129        } // <constructor>
130    
131    
132        @Override
133        protected void finalize() throws Throwable {
134            shutdown();
135            super.finalize();
136        }
137

139        // non-javadoc, see interface ClientConnectionManager
140        public SchemeRegistry getSchemeRegistry() {
141            return this.schemeRegistry;
142        }

145        /**
146         * Hook for creating the connection operator.
147         * It is called by the constructor.
148         * Derived classes can override this method to change the
149         * instantiation of the operator.
150         * The default implementation here instantiates
151         * {@link DefaultClientConnectionOperator DefaultClientConnectionOperator}.
152         *
153         * @param schreg    the scheme registry to use, or <code>null</code>
154         *
155         * @return  the connection operator to use
156         */
157        protected ClientConnectionOperator
158            createConnectionOperator(SchemeRegistry schreg) {
159    
160            return new DefaultClientConnectionOperator(schreg);
161        }

164        /**
165         * Asserts that this manager is not shut down.
166         *
167         * @throws IllegalStateException    if this manager is shut down
168         */
169        protected final void assertStillUp()
170            throws IllegalStateException {
171    
172            if (this.isShutDown)
173                throw new IllegalStateException("Manager is shut down.");
174        }

177        public final ClientConnectionRequest requestConnection(
178                final HttpRoute route,
179                final Object state) {
180            
181            return new ClientConnectionRequest() {
182                
183                public void abortRequest() {
184                    // Nothing to abort, since requests are immediate.
185                }
186                
187                public ManagedClientConnection getConnection(
188                        long timeout, TimeUnit tunit) {
189                    return SingleClientConnManager.this.getConnection(
190                            route, state);
191                }
192                
193            };
194        }
195    
196    
197        /**
198         * Obtains a connection.
199         * This method does not block.
200         *
201         * @param route     where the connection should point to
202         *
203         * @return  a connection that can be used to communicate
204         *          along the given route
205         */
206        public ManagedClientConnection getConnection(HttpRoute route, Object state) {
207    
208            if (route == null) {
209                throw new IllegalArgumentException("Route may not be null.");
210            }
211            assertStillUp();
212    
213            if (log.isDebugEnabled()) {
214                log.debug("Get connection for route " + route);
215            }
216    
217            if (managedConn != null)
218                revokeConnection();
219    
220            // check re-usability of the connection
221            boolean recreate = false;
222            boolean shutdown = false;
223            
224            // Kill the connection if it expired.
225            closeExpiredConnections();
226            
227            if (uniquePoolEntry.connection.isOpen()) {
228                RouteTracker tracker = uniquePoolEntry.tracker;
229                shutdown = (tracker == null || // can happen if method is aborted
230                            !tracker.toRoute().equals(route));
231            } else {
232                // If the connection is not open, create a new PoolEntry,
233                // as the connection may have been marked not reusable,
234                // due to aborts -- and the PoolEntry should not be reused
235                // either.  There's no harm in recreating an entry if
236                // the connection is closed.
237                recreate = true;
238            }
239    
240            if (shutdown) {
241                recreate = true;
242                try {
243                    uniquePoolEntry.shutdown();
244                } catch (IOException iox) {
245                    log.debug("Problem shutting down connection.", iox);
246                }
247            }
248            
249            if (recreate)
250                uniquePoolEntry = new PoolEntry();
251    
252            managedConn = new ConnAdapter(uniquePoolEntry, route);
253    
254            return managedConn;
255        }

258        // non-javadoc, see interface ClientConnectionManager
259        public void releaseConnection(ManagedClientConnection conn, long validDuration, TimeUnit timeUnit) {
260            assertStillUp();
261    
262            if (!(conn instanceof ConnAdapter)) {
263                throw new IllegalArgumentException
264                    ("Connection class mismatch, " +
265                     "connection not obtained from this manager.");
266            }
267            
268            if (log.isDebugEnabled()) {
269                log.debug("Releasing connection " + conn);
270            }
271    
272            ConnAdapter sca = (ConnAdapter) conn;
273            if (sca.poolEntry == null)
274                return; // already released
275            ClientConnectionManager manager = sca.getManager();
276            if (manager != null && manager != this) {
277                throw new IllegalArgumentException
278                    ("Connection not obtained from this manager.");
279            }
280    
281            try {
282                // make sure that the response has been read completely
283                if (sca.isOpen() && (this.alwaysShutDown ||
284                                     !sca.isMarkedReusable())
285                    ) {
286                    if (log.isDebugEnabled()) {
287                        log.debug
288                            ("Released connection open but not reusable.");
289                    }
290    
291                    // make sure this connection will not be re-used
292                    // we might have gotten here because of a shutdown trigger
293                    // shutdown of the adapter also clears the tracked route
294                    sca.shutdown();
295                }
296            } catch (IOException iox) {
297                //@@@ log as warning? let pass?
298                if (log.isDebugEnabled())
299                    log.debug("Exception shutting down released connection.",
300                              iox);
301            } finally {
302                sca.detach();
303                managedConn = null;
304                lastReleaseTime = System.currentTimeMillis();
305                if(validDuration > 0)
306                    connectionExpiresTime = timeUnit.toMillis(validDuration) + lastReleaseTime;
307                else
308                    connectionExpiresTime = Long.MAX_VALUE;
309            }
310        } // releaseConnection
311        
312        public void closeExpiredConnections() {
313            if(System.currentTimeMillis() >= connectionExpiresTime) {
314                closeIdleConnections(0, TimeUnit.MILLISECONDS);
315            }
316        }
317    
318    
319        // non-javadoc, see interface ClientConnectionManager
320        public void closeIdleConnections(long idletime, TimeUnit tunit) {
321            assertStillUp();
322    
323            // idletime can be 0 or negative, no problem there
324            if (tunit == null) {
325                throw new IllegalArgumentException("Time unit must not be null.");
326            }
327    
328            if ((managedConn == null) && uniquePoolEntry.connection.isOpen()) {
329                final long cutoff =
330                    System.currentTimeMillis() - tunit.toMillis(idletime);
331                if (lastReleaseTime <= cutoff) {
332                    try {
333                        uniquePoolEntry.close();
334                    } catch (IOException iox) {
335                        // ignore
336                        log.debug("Problem closing idle connection.", iox);
337                    }
338                }
339            }
340        }
341    
342    
343        // non-javadoc, see interface ClientConnectionManager
344        public void shutdown() {
345    
346            this.isShutDown = true;
347    
348            if (managedConn != null)
349                managedConn.detach();
350    
351            try {
352                if (uniquePoolEntry != null) // and connection open?
353                    uniquePoolEntry.shutdown();
354            } catch (IOException iox) {
355                // ignore
356                log.debug("Problem while shutting down manager.", iox);
357            } finally {
358                uniquePoolEntry = null;
359            }
360        }
361    
362    
363        /**
364         * Revokes the currently issued connection.
365         * The adapter gets disconnected, the connection will be shut down.
366         */
367        protected void revokeConnection() {
368            if (managedConn == null)
369                return;
370    
371            log.warn(MISUSE_MESSAGE);
372    
373            managedConn.detach();
374    
375            try {
376                uniquePoolEntry.shutdown();
377            } catch (IOException iox) {
378                // ignore
379                log.debug("Problem while shutting down connection.", iox);
380            }
381        }

384        /**
385         * The pool entry for this connection manager.
386         */
387        protected class PoolEntry extends AbstractPoolEntry {
388    
389            /**
390             * Creates a new pool entry.
391             *
392             */
393            protected PoolEntry() {
394                super(SingleClientConnManager.this.connOperator, null);
395            }
396    
397            /**
398             * Closes the connection in this pool entry.
399             */
400            protected void close()
401                throws IOException {
402    
403                shutdownEntry();
404                if (connection.isOpen())
405                    connection.close();
406            }
407    
408    
409            /**
410             * Shuts down the connection in this pool entry.
411             */
412            protected void shutdown()
413                throws IOException {
414    
415                shutdownEntry();
416                if (connection.isOpen())
417                    connection.shutdown();
418            }
419    
420        } // class PoolEntry

424        /**
425         * The connection adapter used by this manager.
426         */
427        protected class ConnAdapter extends AbstractPooledConnAdapter {
428    
429            /**
430             * Creates a new connection adapter.
431             *
432             * @param entry   the pool entry for the connection being wrapped
433             * @param route   the planned route for this connection
434             */
435            protected ConnAdapter(PoolEntry entry, HttpRoute route) {
436                super(SingleClientConnManager.this, entry);
437                markReusable();
438                entry.route = route;
439            }
440
