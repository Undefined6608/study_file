032    package org.apache.http.impl.client;
033    
034    import java.io.IOException;
035    
036    import org.apache.http.HttpEntity;
037    import org.apache.http.HttpResponse;
038    import org.apache.http.StatusLine;
039    import org.apache.http.client.ResponseHandler;
040    import org.apache.http.client.HttpResponseException;
041    import org.apache.http.util.EntityUtils;
042    
043    /**
044     * A {@link ResponseHandler} that returns the response body as a String
045     * for successful (2xx) responses. If the response code was >= 300, the response
046     * body is consumed and an {@link HttpResponseException} is thrown.
047     * 
048     * If this is used with
049     * {@link org.apache.http.client.HttpClient#execute(
050     *  org.apache.http.client.methods.HttpUriRequest, ResponseHandler),
051     * HttpClient may handle redirects (3xx responses) internally.
052     * 
053     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
054     * 
055     * @version $Revision: 677240 $
056     * 
057     * @since 4.0
058     */
059    public class BasicResponseHandler implements ResponseHandler<String> {
060    
061        /**
062         * Returns the response body as a String if the response was successful (a
063         * 2xx status code). If no response body exists, this returns null. If the
064         * response was unsuccessful (>= 300 status code), throws an
065         * {@link HttpResponseException}.
066         */
067        public String handleResponse(final HttpResponse response)
068                throws HttpResponseException, IOException {
069            StatusLine statusLine = response.getStatusLine();
070            if (statusLine.getStatusCode() >= 300) {
071                throw new HttpResponseException(statusLine.getStatusCode(),
072                        statusLine.getReasonPhrase());
073            }
074    
075            HttpEntity entity = response.getEntity();
076            return entity == null ? null : EntityUtils.toString(entity);
077        }
078    
079    }
