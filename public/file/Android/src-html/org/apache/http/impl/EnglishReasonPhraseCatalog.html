040    /**
041     * English reason phrases for HTTP status codes.
042     * All status codes defined in RFC1945 (HTTP/1.0), RFC2616 (HTTP/1.1), and
043     * RFC2518 (WebDAV) are supported.
044     * 
045     * @author Unascribed
046     * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
047     * @author <a href="mailto:jsdever@apache.org">Jeff Dever</a>
048     * 
049     * @version $Revision: 505744 $
050     */
051    public class EnglishReasonPhraseCatalog
052        implements ReasonPhraseCatalog {
053    
054        // static array with english reason phrases defined below
055    
056        /**
057         * The default instance of this catalog.
058         * This catalog is thread safe, so there typically
059         * is no need to create other instances.
060         */
061        public final static EnglishReasonPhraseCatalog INSTANCE =
062            new EnglishReasonPhraseCatalog();
063    
064    
065        /**
066         * Restricted default constructor, for derived classes.
067         * If you need an instance of this class, use {@link #INSTANCE INSTANCE}.
068         */
069        protected EnglishReasonPhraseCatalog() {
070            // no body
071        }
072    
073    
074        /**
075         * Obtains the reason phrase for a status code.
076         *
077         * @param status    the status code, in the range 100-599
078         * @param loc       ignored
079         *
080         * @return  the reason phrase, or <code>null</code>
081         */
082        public String getReason(int status, Locale loc) {
083            if ((status < 100) || (status >= 600)) {
084                throw new IllegalArgumentException
085                    ("Unknown category for status code " + status + ".");
086            }
087    
088            final int category = status / 100;
089            final int subcode  = status - 100*category;
090    
091            String reason = null;
092            if (REASON_PHRASES[category].length > subcode)
093                reason = REASON_PHRASES[category][subcode];
094    
095            return reason;
096        }
097    
098    
099        /** Reason phrases lookup table. */
100        private static final String[][] REASON_PHRASES = new String[][]{
101            null,
102            new String[3],  // 1xx
103            new String[8],  // 2xx
104            new String[8],  // 3xx
105            new String[25], // 4xx
106            new String[8]   // 5xx
107        };
108    
109    
110    
111        /**
112         * Stores the given reason phrase, by status code.
113         * Helper method to initialize the static lookup table.
114         *
115         * @param status    the status code for which to define the phrase
116         * @param reason    the reason phrase for this status code
117         */
118        private static void setReason(int status, String reason) {
119            final int category = status / 100;
120            final int subcode  = status - 100*category;
121            REASON_PHRASES[category][subcode] = reason;
122        }
