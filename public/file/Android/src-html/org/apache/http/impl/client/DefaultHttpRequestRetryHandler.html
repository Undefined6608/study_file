032    package org.apache.http.impl.client;
033    
034    import java.io.IOException;
035    import java.io.InterruptedIOException;
036    import java.net.UnknownHostException;
037    
038    import javax.net.ssl.SSLHandshakeException;
039    
040    import org.apache.http.NoHttpResponseException;
041    import org.apache.http.client.HttpRequestRetryHandler;
042    import org.apache.http.protocol.HttpContext;
043    import org.apache.http.protocol.ExecutionContext;
044    
045    /**
046     * The default {@link HttpRequestRetryHandler} used by request executors.
047     * 
048     * @author Michael Becke
049     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
050     */
051    public class DefaultHttpRequestRetryHandler implements HttpRequestRetryHandler {
052    
053        /** the number of times a method will be retried */
054        private final int retryCount;
055        
056        /** Whether or not methods that have successfully sent their request will be retried */
057        private final boolean requestSentRetryEnabled;
058        
059        /**
060         * Default constructor
061         */
062        public DefaultHttpRequestRetryHandler(int retryCount, boolean requestSentRetryEnabled) {
063            super();
064            this.retryCount = retryCount;
065            this.requestSentRetryEnabled = requestSentRetryEnabled;
066        }
067        
068        /**
069         * Default constructor
070         */
071        public DefaultHttpRequestRetryHandler() {
072            this(3, false);
073        }
074        /** 
075         * Used <code>retryCount</code> and <code>requestSentRetryEnabled</code> to determine
076         * if the given method should be retried.
077         */
078        public boolean retryRequest(
079                final IOException exception, 
080                int executionCount,
081                final HttpContext context) {
082            if (exception == null) {
083                throw new IllegalArgumentException("Exception parameter may not be null");
084            }
085            if (context == null) {
086                throw new IllegalArgumentException("HTTP context may not be null");
087            }
088            if (executionCount > this.retryCount) {
089                // Do not retry if over max retry count
090                return false;
091            }
092            if (exception instanceof NoHttpResponseException) {
093                // Retry if the server dropped connection on us
094                return true;
095            }
096            if (exception instanceof InterruptedIOException) {
097                // Timeout
098                return false;
099            }
100            if (exception instanceof UnknownHostException) {
101                // Unknown host
102                return false;
103            }
104            if (exception instanceof SSLHandshakeException) {
105                // SSL handshake exception
106                return false;
107            }
108            Boolean b = (Boolean)
109                context.getAttribute(ExecutionContext.HTTP_REQ_SENT);
110            boolean sent = (b != null && b.booleanValue());
111            if (!sent || this.requestSentRetryEnabled) {
112                // Retry if the request has not been sent fully or
113                // if it's OK to retry methods that have been sent
114                return true;
115            }
116            // otherwise do not retry
117            return false;
118        }
119        
120        /**
121         * @return <code>true</code> if this handler will retry methods that have 
122         * successfully sent their request, <code>false</code> otherwise
123         */
124        public boolean isRequestSentRetryEnabled() {
125            return requestSentRetryEnabled;
126        }
127    
128        /**
129         * @return the maximum number of times a method will be retried
130         */
131        public int getRetryCount() {
132            return retryCount;
133        }
134    }
