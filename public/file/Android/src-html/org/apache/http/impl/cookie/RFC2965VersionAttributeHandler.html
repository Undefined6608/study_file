032    package org.apache.http.impl.cookie;
033    
034    import org.apache.http.cookie.ClientCookie;
035    import org.apache.http.cookie.Cookie;
036    import org.apache.http.cookie.CookieAttributeHandler;
037    import org.apache.http.cookie.CookieOrigin;
038    import org.apache.http.cookie.MalformedCookieException;
039    import org.apache.http.cookie.SetCookie;
040    import org.apache.http.cookie.SetCookie2;
041    
042    /**
043     * <tt>"Version"</tt> cookie attribute handler for RFC 2965 cookie spec.
044     */
045    public class RFC2965VersionAttributeHandler implements CookieAttributeHandler {
046    
047        public RFC2965VersionAttributeHandler() {
048            super();
049        }
050        
051        /**
052         * Parse cookie version attribute.
053         */
054        public void parse(final SetCookie cookie, final String value)
055                throws MalformedCookieException {
056            if (cookie == null) {
057                throw new IllegalArgumentException("Cookie may not be null");
058            }
059            if (value == null) {
060                throw new MalformedCookieException(
061                        "Missing value for version attribute");
062            }
063            int version = -1;
064            try {
065                version = Integer.parseInt(value);
066            } catch (NumberFormatException e) {
067                version = -1;
068            }
069            if (version < 0) {
070                throw new MalformedCookieException("Invalid cookie version.");
071            }
072            cookie.setVersion(version);
073        }
074    
075        /**
076         * validate cookie version attribute. Version attribute is REQUIRED.
077         */
078        public void validate(final Cookie cookie, final CookieOrigin origin)
079                throws MalformedCookieException {
080            if (cookie == null) {
081                throw new IllegalArgumentException("Cookie may not be null");
082            }
083            if (cookie instanceof SetCookie2) {
084                if (cookie instanceof ClientCookie 
085                        && !((ClientCookie) cookie).containsAttribute(ClientCookie.VERSION_ATTR)) {
086                    throw new MalformedCookieException(
087                            "Violates RFC 2965. Version attribute is required.");
088                }
089            }
090        }
091    
092        public boolean match(final Cookie cookie, final CookieOrigin origin) {
093            return true;
094        }
095    
096    }
