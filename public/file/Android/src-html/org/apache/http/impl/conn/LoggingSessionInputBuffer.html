031    package org.apache.http.impl.conn;
032    
033    import java.io.IOException;
034    
035    import org.apache.http.io.HttpTransportMetrics;
036    import org.apache.http.io.SessionInputBuffer;
037    import org.apache.http.util.CharArrayBuffer;
038    
039    /**
040     * Logs all data read to the wire LOG.
041     *
042     * @author Ortwin Glueck
043     * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
044     * @author <a href="mailto:oleg at ural.ru">Oleg Kalnichevski</a>
045     * 
046     * @since 4.0
047     */
048    public class LoggingSessionInputBuffer implements SessionInputBuffer {
049         
050        /** Original session input buffer. */
051        private final SessionInputBuffer in;
052    
053        /** The wire log to use for writing. */
054        private final Wire wire;
055        
056        /**
057         * Create an instance that wraps the specified session input buffer.
058         * @param in The session input buffer.
059         * @param wire The wire log to use.
060         */
061        public LoggingSessionInputBuffer(final SessionInputBuffer in, final Wire wire) {
062            super();
063            this.in = in;
064            this.wire = wire;
065        }
066    
067        public boolean isDataAvailable(int timeout) throws IOException {
068            return this.in.isDataAvailable(timeout);
069        }
070    
071        public int read(byte[] b, int off, int len) throws IOException {
072            int l = this.in.read(b,  off,  len);
073            if (this.wire.enabled() && l > 0) {
074                this.wire.input(b, off, l);
075            }
076            return l;
077        }
078    
079        public int read() throws IOException {
080            int l = this.in.read();
081            if (this.wire.enabled() && l > 0) { 
082                this.wire.input(l);
083            }
084            return l;
085        }
086    
087        public int read(byte[] b) throws IOException {
088            int l = this.in.read(b);
089            if (this.wire.enabled() && l > 0) {
090                this.wire.input(b, 0, l);
091            }
092            return l;
093        }
094    
095        public String readLine() throws IOException {
096            String s = this.in.readLine();
097            if (this.wire.enabled() && s != null) {
098                this.wire.input(s + "[EOL]");
099            }
100            return s;
101        }
102    
103        public int readLine(final CharArrayBuffer buffer) throws IOException {
104            int l = this.in.readLine(buffer);
105            if (this.wire.enabled() && l > 0) {
106                int pos = buffer.length() - l;
107                String s = new String(buffer.buffer(), pos, l);
108                this.wire.input(s + "[EOL]");
109            }
110            return l;
111        }
112    
113        public HttpTransportMetrics getMetrics() {
114            return this.in.getMetrics();
115        }
116    
117    }
