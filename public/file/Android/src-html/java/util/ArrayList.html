028    /**
029     * ArrayList is an implementation of {@link List}, backed by an array. All
030     * optional operations adding, removing, and replacing are supported. The
031     * elements can be any objects.

077        /**
078         * Constructs a new instance of {@code ArrayList} containing the elements of
079         * the specified collection. The initial size of the {@code ArrayList} will
080         * be 10% higher than the size of the specified collection.
081         * 
082         * @param collection
083         *            the collection of elements to add.
084         * @since Android 1.0
085         */
086        public ArrayList(Collection<? extends E> collection) {
087            int size = collection.size();
088            firstIndex = lastIndex = 0;
089            array = newElementArray(size + (size / 10));
090            addAll(collection);
091        }

093        @SuppressWarnings("unchecked")
094        private E[] newElementArray(int size) {
095            // BEGIN android-added
096            if (size == 0) {
097                return (E[])emptyArray;
098            }
099            // END android-added
100    
101            return (E[])new Object[size];
102        }

104        /**
105         * Inserts the specified object into this {@code ArrayList} at the specified
106         * location. The object is inserted before any previous element at the
107         * specified location. If the location is equal to the size of this
108         * {@code ArrayList}, the object is added at the end.
109         * 
110         * @param location
111         *            the index at which to insert the object.
112         * @param object
113         *            the object to add.
114         * @throws IndexOutOfBoundsException
115         *             when {@code location < 0 || > size()}
116         * @since Android 1.0
117         */
118        @Override
119        public void add(int location, E object) {
120            // BEGIN android-changed: slight performance improvement
121            int size = lastIndex - firstIndex;
122            // END android-changed
123            if (0 < location && location < size) {
124                if (firstIndex == 0 && lastIndex == array.length) {
125                    growForInsert(location, 1);
126                } else if ((location < size / 2 && firstIndex > 0)
127                        || lastIndex == array.length) {
128                    System.arraycopy(array, firstIndex, array, --firstIndex,
129                            location);
130                } else {
131                    int index = location + firstIndex;
132                    System.arraycopy(array, index, array, index + 1, size
133                            - location);
134                    lastIndex++;
135                }
136                array[location + firstIndex] = object;
137            } else if (location == 0) {
138                if (firstIndex == 0) {
139                    growAtFront(1);
140                }
141                array[--firstIndex] = object;
142            } else if (location == size) {
143                if (lastIndex == array.length) {
144                    growAtEnd(1);
145                }
146                array[lastIndex++] = object;
147            } else {
148                throw new IndexOutOfBoundsException();
149            }
150    
151            modCount++;
152        }
153    
154        /**
155         * Adds the specified object at the end of this {@code ArrayList}.
156         * 
157         * @param object
158         *            the object to add.
159         * @return always true
160         * @since Android 1.0
161         */
162        @Override
163        public boolean add(E object) {
164            if (lastIndex == array.length) {
165                growAtEnd(1);
166            }
167            array[lastIndex++] = object;
168            modCount++;
169            return true;
170        }

172        /**
173         * Inserts the objects in the specified collection at the specified location
174         * in this List. The objects are added in the order they are returned from
175         * the collection's iterator.
176         * 
177         * @param location
178         *            the index at which to insert.
179         * @param collection
180         *            the collection of objects.
181         * @return {@code true} if this {@code ArrayList} is modified, {@code false}
182         *         otherwise.
183         * @throws IndexOutOfBoundsException
184         *             when {@code location < 0 || > size()}
185         * @since Android 1.0
186         */
187        @Override
188        public boolean addAll(int location, Collection<? extends E> collection) {
189            int size = size();
190            if (location < 0 || location > size) {
191                throw new IndexOutOfBoundsException();
192            }
193            int growSize = collection.size();
194            if (0 < location && location < size) {
195                if (array.length - size < growSize) {
196                    growForInsert(location, growSize);
197                } else if ((location < size / 2 && firstIndex > 0)
198                        || lastIndex > array.length - growSize) {
199                    int newFirst = firstIndex - growSize;
200                    if (newFirst < 0) {
201                        int index = location + firstIndex;
202                        System.arraycopy(array, index, array, index - newFirst,
203                                size - location);
204                        lastIndex -= newFirst;
205                        newFirst = 0;
206                    }
207                    System.arraycopy(array, firstIndex, array, newFirst, location);
208                    firstIndex = newFirst;
209                } else {
210                    int index = location + firstIndex;
211                    System.arraycopy(array, index, array, index + growSize, size
212                            - location);
213                    lastIndex += growSize;
214                }
215            } else if (location == 0) {
216                growAtFront(growSize);
217                firstIndex -= growSize;
218            } else if (location == size) {
219                if (lastIndex > array.length - growSize) {
220                    growAtEnd(growSize);
221                }
222                lastIndex += growSize;
223            }
224    
225            if (growSize > 0) {
226                Iterator<? extends E> it = collection.iterator();
227                int index = location + firstIndex;
228                int end = index + growSize;
229                while (index < end) {
230                    array[index++] = it.next();
231                }
232                modCount++;
233                return true;
234            }
235            return false;
236        }

238        /**
239         * Adds the objects in the specified collection to this {@code ArrayList}.
240         * 
241         * @param collection
242         *            the collection of objects.
243         * @return {@code true} if this {@code ArrayList} is modified, {@code false}
244         *         otherwise.
245         * @since Android 1.0
246         */
247        @Override
248        public boolean addAll(Collection<? extends E> collection) {
249            int growSize = collection.size();
250            if (growSize > 0) {
251                if (lastIndex > array.length - growSize) {
252                    growAtEnd(growSize);
253                }
254                Iterator<? extends E> it = collection.iterator();
255                int end = lastIndex + growSize;
256                while (lastIndex < end) {
257                    array[lastIndex++] = it.next();
258                }
259                modCount++;
260                return true;
261            }
262            return false;
263        }

266         * Removes all elements from this {@code ArrayList}, leaving it empty.
267         * 
268         * @see #isEmpty
269         * @see #size
270         * @since Android 1.0
271         */
272        @Override
273        public void clear() {
274            if (firstIndex != lastIndex) {
275                Arrays.fill(array, firstIndex, lastIndex, null);
276                firstIndex = lastIndex = 0;
277                modCount++;
278            }

302         * Searches this {@code ArrayList} for the specified object.
303         * 
304         * @param object
305         *            the object to search for.
306         * @return {@code true} if {@code object} is an element of this
307         *         {@code ArrayList}, {@code false} otherwise
308         * @since Android 1.0
309         */
310        @Override
311        public boolean contains(Object object) {
312            if (object != null) {
313                for (int i = firstIndex; i < lastIndex; i++) {
314                    if (object.equals(array[i])) {
315                        return true;
316                    }
317                }
318            } else {
319                for (int i = firstIndex; i < lastIndex; i++) {
320                    if (array[i] == null) {
321                        return true;
322                    }
323                }
324            }
325            return false;
326        }

328        /**
329         * Ensures that after this operation the {@code ArrayList} can hold the
330         * specified number of elements without further growing.
331         * 
332         * @param minimumCapacity
333         *            the minimum capacity asked for.
334         * @since Android 1.0
335         */
336        public void ensureCapacity(int minimumCapacity) {
337            if (array.length < minimumCapacity) {
338                if (firstIndex > 0) {
339                    growAtFront(minimumCapacity - array.length);
340                } else {
341                    growAtEnd(minimumCapacity - array.length);
342                }
343            }
344        }

346        @Override
347        public E get(int location) {
348            // BEGIN android-changed: slight performance improvement
349            int _firstIndex = firstIndex;
350            if (0 <= location && location < lastIndex - _firstIndex) {
351                return array[_firstIndex + location];
352            }
353            throw new IndexOutOfBoundsException("Invalid location " + location
354                    + ", size is " + (lastIndex - _firstIndex));
355            // END android-changed
356        }

358        private void growAtEnd(int required) {
359            int size = size();
360            if (firstIndex >= required - (array.length - lastIndex)) {
361                int newLast = lastIndex - firstIndex;
362                if (size > 0) {
363                    System.arraycopy(array, firstIndex, array, 0, size);
364                    int start = newLast < firstIndex ? firstIndex : newLast;
365                    Arrays.fill(array, start, array.length, null);
366                }
367                firstIndex = 0;
368                lastIndex = newLast;
369            } else {
370                int increment = size / 2;
371                if (required > increment) {
372                    increment = required;
373                }
374                if (increment < 12) {
375                    increment = 12;
376                }
377                E[] newArray = newElementArray(size + increment);
378                if (size > 0) {
379                    System.arraycopy(array, firstIndex, newArray, 0, size);
380                    firstIndex = 0;
381                    lastIndex = size;
382                }
383                array = newArray;
384            }
385        }
386    
387        private void growAtFront(int required) {
388            int size = size();
389            if (array.length - lastIndex >= required) {
390                int newFirst = array.length - size;
391                if (size > 0) {
392                    System.arraycopy(array, firstIndex, array, newFirst, size);
393                    int length = firstIndex + size > newFirst ? newFirst
394                            : firstIndex + size;
395                    Arrays.fill(array, firstIndex, length, null);
396                }
397                firstIndex = newFirst;
398                lastIndex = array.length;
399            } else {
400                int increment = size / 2;
401                if (required > increment) {
402                    increment = required;
403                }
404                if (increment < 12) {
405                    increment = 12;
406                }
407                E[] newArray = newElementArray(size + increment);
408                if (size > 0) {
409                    System.arraycopy(array, firstIndex, newArray, newArray.length
410                            - size, size);
411                }
412                firstIndex = newArray.length - size;
413                lastIndex = newArray.length;
414                array = newArray;
415            }
416        }
417    
418        private void growForInsert(int location, int required) {
419            int size = size(), increment = size / 2;
420            if (required > increment) {
421                increment = required;
422            }
423            if (increment < 12) {
424                increment = 12;
425            }
426            E[] newArray = newElementArray(size + increment);
427            if (location < size / 2) {
428                int newFirst = newArray.length - (size + required);
429                System.arraycopy(array, location, newArray, location + increment,
430                        size - location);
431                System.arraycopy(array, firstIndex, newArray, newFirst, location);
432                firstIndex = newFirst;
433                lastIndex = newArray.length;
434            } else {
435                System.arraycopy(array, firstIndex, newArray, 0, location);
436                System.arraycopy(array, location, newArray, location + required,
437                        size - location);
438                firstIndex = 0;
439                lastIndex += required;
440            }
441            array = newArray;
442        }

444        @Override
445        public int indexOf(Object object) {
446            if (object != null) {
447                for (int i = firstIndex; i < lastIndex; i++) {
448                    if (object.equals(array[i])) {
449                        return i - firstIndex;
450                    }
451                }
452            } else {
453                for (int i = firstIndex; i < lastIndex; i++) {
454                    if (array[i] == null) {
455                        return i - firstIndex;
456                    }
457                }
458            }
459            return -1;
460        }
461    
462        @Override
463        public boolean isEmpty() {
464            return lastIndex == firstIndex;
465        }
466    
467        @Override
468        public int lastIndexOf(Object object) {
469            if (object != null) {
470                for (int i = lastIndex - 1; i >= firstIndex; i--) {
471                    if (object.equals(array[i])) {
472                        return i - firstIndex;
473                    }
474                }
475            } else {
476                for (int i = lastIndex - 1; i >= firstIndex; i--) {
477                    if (array[i] == null) {
478                        return i - firstIndex;
479                    }
480                }
481            }
482            return -1;
483        }

485        /**
486         * Removes the object at the specified location from this list.
487         * 
488         * @param location
489         *            the index of the object to remove.
490         * @return the removed object.
491         * @throws IndexOutOfBoundsException
492         *             when {@code location < 0 || >= size()}
493         * @since Android 1.0
494         */
495        @Override
496        public E remove(int location) {
497            E result;
498            // BEGIN android-changed: slight performance improvement
499            int size = lastIndex - firstIndex;
500            // END android-changed
501            if (0 <= location && location < size) {
502                if (location == size - 1) {
503                    result = array[--lastIndex];
504                    array[lastIndex] = null;
505                } else if (location == 0) {
506                    result = array[firstIndex];
507                    array[firstIndex++] = null;
508                } else {
509                    int elementIndex = firstIndex + location;
510                    result = array[elementIndex];
511                    if (location < size / 2) {
512                        System.arraycopy(array, firstIndex, array, firstIndex + 1,
513                                location);
514                        array[firstIndex++] = null;
515                    } else {
516                        System.arraycopy(array, elementIndex + 1, array,
517                                elementIndex, size - location - 1);
518                        array[--lastIndex] = null;
519                    }
520                }
521            } else {
522                throw new IndexOutOfBoundsException();
523            }
524    
525            modCount++;
526            return result;
527        }

531         * The remove(Object) implementation from AbstractCollection creates
532         * a new Iterator on every remove and ends up calling remove(int).
533         */
534        @Override
535        public boolean remove(Object object) {
536            int index = indexOf(object);
537            if (index >= 0) {
538                remove(index);
539                return true;
540            }
541            return false;
542        }

545        /**
546         * Removes the objects in the specified range from the start to the end, but
547         * not including the end index.
548         * 
549         * @param start
550         *            the index at which to start removing.
551         * @param end
552         *            the index one after the end of the range to remove.
553         * @throws IndexOutOfBoundsException
554         *             when {@code start < 0, start > end} or {@code end > size()}
555         * @since Android 1.0
556         */
557        @Override
558        protected void removeRange(int start, int end) {
559            if (start >= 0 && start <= end && end <= size()) {
560                if (start == end) {
561                    return;
562                }
563                int size = size();
564                if (end == size) {
565                    Arrays.fill(array, firstIndex + start, lastIndex, null);
566                    lastIndex = firstIndex + start;
567                } else if (start == 0) {
568                    Arrays.fill(array, firstIndex, firstIndex + end, null);
569                    firstIndex += end;
570                } else {
571                    System.arraycopy(array, firstIndex + end, array, firstIndex
572                            + start, size - end);
573                    int newLast = lastIndex + start - end;
574                    Arrays.fill(array, newLast, lastIndex, null);
575                    lastIndex = newLast;
576                }
577                modCount++;
578            } else {
579                throw new IndexOutOfBoundsException();
580            }
581        }

598            // BEGIN android-changed: slight performance improvement
599            if (0 <= location && location < (lastIndex - firstIndex)) {
600            // END android-changed
601                E result = array[firstIndex + location];
602                array[firstIndex + location] = object;
603                return result;
604            }
605            throw new IndexOutOfBoundsException();

634        /**
635         * Returns an array containing all elements contained in this
636         * {@code ArrayList}. If the specified array is large enough to hold the
637         * elements, the specified array is used, otherwise an array of the same
638         * type is created. If the specified array is used and is larger than this
639         * {@code ArrayList}, the array element following the collection elements
640         * is set to null.
641         * 
642         * @param contents
643         *            the array.
644         * @return an array of the elements from this {@code ArrayList}.
645         * @throws ArrayStoreException
646         *             when the type of an element in this {@code ArrayList} cannot
647         *             be stored in the type of the specified array.
648         * @since Android 1.0
649         */
650        @Override
651        @SuppressWarnings("unchecked")
652        public <T> T[] toArray(T[] contents) {
653            int size = size();
654            if (size > contents.length) {
655                Class<?> ct = contents.getClass().getComponentType();
656                contents = (T[]) Array.newInstance(ct, size);
657            }
658            System.arraycopy(array, firstIndex, contents, 0, size);
659            if (size < contents.length) {
660                contents[size] = null;
661            }
662            return contents;
663        }

665        /**
666         * Sets the capacity of this {@code ArrayList} to be the same as the current
667         * size.
668         * 
669         * @see #size
670         * @since Android 1.0
671         */
672        public void trimToSize() {
673            int size = size();
674            E[] newArray = newElementArray(size);
675            System.arraycopy(array, firstIndex, newArray, 0, size);
676            array = newArray;
677            firstIndex = 0;
678            lastIndex = array.length;
679        }

681        private static final ObjectStreamField[] serialPersistentFields = { new ObjectStreamField(
682                "size", Integer.TYPE) }; //$NON-NLS-1$
683    
684        private void writeObject(ObjectOutputStream stream) throws IOException {
685            ObjectOutputStream.PutField fields = stream.putFields();
686            fields.put("size", size()); //$NON-NLS-1$
687            stream.writeFields();
688            stream.writeInt(array.length);
689            Iterator<?> it = iterator();
690            while (it.hasNext()) {
691                stream.writeObject(it.next());
692            }
693        }
694    
695        @SuppressWarnings("unchecked")
696        private void readObject(ObjectInputStream stream) throws IOException,
697                ClassNotFoundException {
698            ObjectInputStream.GetField fields = stream.readFields();
699            lastIndex = fields.get("size", 0); //$NON-NLS-1$
700            array = newElementArray(stream.readInt());
701            for (int i = 0; i < lastIndex; i++) {
702                array[i] = (E)stream.readObject();
703            }
704        }
