018    package java.util;
019    
020    
021    /**
022     * A {@code Map} is a data structure consisting of a set of keys and values
023     * in which each key is mapped to a single value.  The class of the objects
024     * used as keys is declared when the {@code Map} is declared, as is the 
025     * class of the corresponding values.
026     * <p>A {@code Map} provides helper methods to iterate through all of the 
027     * keys contained in it, as well as various methods to access and update 
028     * the key/value pairs.  
029     * </p>  
030     *  
031     * @since Android 1.0
032     */
033    public interface Map<K,V> {
034    
035        /**
036         * {@code Map.Entry} is a key/value mapping contained in a {@code Map}.
037         *      
038         * @since Android 1.0
039         */
040        public static interface Entry<K,V> {
041            /**
042             * Compares the specified object to this {@code Map.Entry} and returns if they
043             * are equal. To be equal, the object must be an instance of {@code Map.Entry} and have the
044             * same key and value.
045             * 
046             * @param object
047             *            the {@code Object} to compare with this {@code Object}.
048             * @return {@code true} if the specified {@code Object} is equal to this
049             *         {@code Map.Entry}, {@code false} otherwise.
050             * @see #hashCode()
051             * @since Android 1.0
052             */
053            public boolean equals(Object object);
054    
055            /**
056             * Returns the key.
057             * 
058             * @return the key
059             * @since Android 1.0
060             */
061            public K getKey();
062    
063            /**
064             * Returns the value.
065             * 
066             * @return the value
067             * @since Android 1.0
068             */
069            public V getValue();
070    
071            /**
072             * Returns an integer hash code for the receiver. {@code Object} which are
073             * equal return the same value for this method.
074             * 
075             * @return the receiver's hash code.
076             * @see #equals(Object)
077             * @since Android 1.0
078             */
079            public int hashCode();
080    
081            /**
082             * Sets the value of this entry to the specified value, replacing any
083             * existing value.
084             * 
085             * @param object
086             *            the new value to set.
087             * @return object the replaced value of this entry.
088             * @since Android 1.0
089             */
090            public V setValue(V object);
091        };
092    
093        /**
094         * Removes all elements from this {@code Map}, leaving it empty.
095         * 
096         * @throws UnsupportedOperationException
097         *                if removing elements from this {@code Map} is not supported.
098         * @see #isEmpty()
099         * @see #size()
100         * @since Android 1.0
101         */
102        public void clear();
103    
104        /**
105         * Returns whether this {@code Map} contains the specified key.
106         * 
107         * @param key
108         *            the key to search for.
109         * @return {@code true} if this map contains the specified key,
110         *         {@code false} otherwise.
111         * @since Android 1.0
112         */
113        public boolean containsKey(Object key);
114    
115        /**
116         * Returns whether this {@code Map} contains the specified value.
117         * 
118         * @param value
119         *            the value to search for.
120         * @return {@code true} if this map contains the specified value,
121         *         {@code false} otherwise.
122         * @since Android 1.0
123         */
124        public boolean containsValue(Object value);
125    
126        /**
127         * Returns a {@code Set} containing all of the mappings in this {@code Map}. Each mapping is
128         * an instance of {@link Map.Entry}. As the {@code Set} is backed by this {@code Map},
129         * changes in one will be reflected in the other.
130         * 
131         * @return a set of the mappings
132         * @since Android 1.0
133         */
134        public Set<Map.Entry<K,V>> entrySet();
135    
136        /**
137         * Compares the argument to the receiver, and returns {@code true} if the
138         * specified object is a {@code Map} and both {@code Map}s contain the same mappings.
139         * 
140         * @param object
141         *            the {@code Object} to compare with this {@code Object}.
142         * @return boolean {@code true} if the {@code Object} is the same as this {@code Object}
143         *         {@code false} if it is different from this {@code Object}.
144         * @see #hashCode()
145         * @see #entrySet()
146         * @since Android 1.0
147         */
148        public boolean equals(Object object);
149    
150        /**
151         * Returns the value of the mapping with the specified key.
152         * 
153         * @param key
154         *            the key.
155         * @return the value of the mapping with the specified key, or {@code null}
156         *         if no mapping for the specified key is found.
157         * @since Android 1.0
158         */
159        public V get(Object key);
160    
161        /**
162         * Returns an integer hash code for the receiver. {@code Object}s which are equal
163         * return the same value for this method.
164         * 
165         * @return the receiver's hash.
166         * @see #equals(Object)
167         * @since Android 1.0
168         */
169        public int hashCode();
170    
171        /**
172         * Returns whether this map is empty.
173         * 
174         * @return {@code true} if this map has no elements, {@code false}
175         *         otherwise.
176         * @see #size()
177         * @since Android 1.0
178         */
179        public boolean isEmpty();
180    
181        /**
182         * Returns a set of the keys contained in this {@code Map}. The {@code Set} is backed by
183         * this {@code Map} so changes to one are reflected by the other. The {@code Set} does not
184         * support adding.
185         * 
186         * @return a set of the keys.
187         * @since Android 1.0
188         */
189        public Set<K> keySet();
190    
191        /**
192         * Maps the specified key to the specified value.
193         * 
194         * @param key
195         *            the key.
196         * @param value
197         *            the value.
198         * @return the value of any previous mapping with the specified key or
199         *         {@code null} if there was no mapping.
200         * @throws UnsupportedOperationException
201         *                if adding to this {@code Map} is not supported.
202         * @throws ClassCastException
203         *                if the class of the key or value is inappropriate for
204         *                this {@code Map}.
205         * @throws IllegalArgumentException
206         *                if the key or value cannot be added to this {@code Map}.
207         * @throws NullPointerException
208         *                if the key or value is {@code null} and this {@code Map} does
209         *                not support {@code null} keys or values.
210         * @since Android 1.0
211         */
212        public V put(K key, V value);
213    
214        /**
215         * Copies every mapping in the specified {@code Map} to this {@code Map}.
216         * 
217         * @param map
218         *            the {@code Map} to copy mappings from.
219         * @throws UnsupportedOperationException
220         *                if adding to this {@code Map} is not supported.
221         * @throws ClassCastException
222         *                if the class of a key or a value of the specified {@code Map} is
223         *                inappropriate for this {@code Map}.
224         * @throws IllegalArgumentException
225         *                if a key or value cannot be added to this {@code Map}.
226         * @throws NullPointerException
227         *                if a key or value is {@code null} and this {@code Map} does not
228         *                support {@code null} keys or values.
229         * @since Android 1.0
230         */
231        public void putAll(Map<? extends K,? extends V> map);
232    
233        /**
234         * Removes a mapping with the specified key from this {@code Map}.
235         * 
236         * @param key
237         *            the key of the mapping to remove.
238         * @return the value of the removed mapping or {@code null} if no mapping
239         *         for the specified key was found.
240         * @throws UnsupportedOperationException
241         *                if removing from this {@code Map} is not supported.
242         * @since Android 1.0
243         */
244        public V remove(Object key);
245    
246        /**
247         * Returns the number of mappings in this {@code Map}.
248         * 
249         * @return the number of mappings in this {@code Map}.
250         * @since Android 1.0
251         */
252        public int size();
253    
254        /**
255         * Returns a {@code Collection} of the values contained in this {@code Map}. The {@code Collection}
256         * is backed by this {@code Map} so changes to one are reflected by the other. The
257         * {@code Collection} supports {@link Collection#remove}, {@link Collection#removeAll}, 
258         * {@link Collection#retainAll}, and {@link Collection#clear} operations,
259         * and it does not support {@link Collection#add} or {@link Collection#addAll} operations.
260         * <p>
261         * This method returns a {@code Collection} which is the subclass of
262         * {@link AbstractCollection}. The {@link AbstractCollection#iterator} method of this subclass returns a
263         * "wrapper object" over the iterator of this {@code Map}'s {@link #entrySet()}. The {@link AbstractCollection#size} method
264         * wraps this {@code Map}'s {@link #size} method and the {@link AbstractCollection#contains} method wraps this {@code Map}'s
265         * {@link #containsValue} method.
266         * </p>
267         * <p>
268         * The collection is created when this method is called at first time and
269         * returned in response to all subsequent calls. This method may return
270         * different Collection when multiple calls to this method, since it has no
271         * synchronization performed.
272         * </p>
273         * 
274         * @return a collection of the values contained in this map.
275         * @since Android 1.0
276         */
277        public Collection<V> values();
278    }
