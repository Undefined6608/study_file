024    /**
025     * An EnumSet is a specialized Set to be used with enums as keys.
026     * 
027     * @since Android 1.0
028     */
029    public abstract class EnumSet<E extends Enum<E>> extends AbstractSet<E>
030            implements Cloneable, Serializable {
031        // BEGIN android-added
032        /*
033         * null-ok; package access to {@code java.lang}, set during
034         * first need. This shouldn't be used directly. Instead, use {@link
035         * SpecialAccess#LANG}, which is guaranteed to be initialized.
036         */
037        static /*package*/ LangAccess LANG_BOOTSTRAP = null;
038        // END android-added
039    
040        private static final long serialVersionUID = 4782406773684236311L;
041    
042        final Class<E> elementClass;
043    
044        EnumSet(Class<E> cls) {
045            elementClass = cls;
046        }
047    
048        /**
049         * Creates an empty enum set. The permitted elements are of type
050         * Class&lt;E&gt;.
051         * 
052         * @param elementType
053         *            the class object for the elements contained.
054         * @return an empty enum set, with permitted elements of type {@code
055         *         elementType}.
056         * @throws ClassCastException
057         *             if the specified element type is not and enum type.
058         * @since Android 1.0
059         */
060        public static <E extends Enum<E>> EnumSet<E> noneOf(Class<E> elementType) {
061            if (!elementType.isEnum()) {
062                throw new ClassCastException();
063            }
064            // BEGIN android-changed
065            E[] enums = SpecialAccess.LANG.getEnumValuesInOrder(elementType);
066            if (enums.length <= 64) {
067                return new MiniEnumSet<E>(elementType, enums);
068            }
069            return new HugeEnumSet<E>(elementType, enums);
070            // END android-changed
071        }
072    
073        /**
074         * Creates an enum set filled with all the enum elements of the specified
075         * {@code elementType}.
076         * 
077         * @param elementType
078         *            the class object for the elements contained.
079         * @return an enum set with elements solely from the specified element type.
080         * @throws ClassCastException
081         *             if the specified element type is not and enum type.
082         * @since Android 1.0
083         */
084        public static <E extends Enum<E>> EnumSet<E> allOf(Class<E> elementType) {
085            EnumSet<E> set = noneOf(elementType);
086            set.complement();
087            return set;
088        }
089    
090        /**
091         * Creates an enum set. All the contained elements are of type
092         * Class&lt;E&gt;, and the contained elements are the same as those
093         * contained in {@code s}.
094         * 
095         * @param s
096         *            the enum set from which to copy.
097         * @return an enum set with all the elements from the specified enum set.
098         * @throws ClassCastException
099         *             if the specified element type is not and enum type.
100         * @since Android 1.0
101         */
102        public static <E extends Enum<E>> EnumSet<E> copyOf(EnumSet<E> s) {
103            EnumSet<E> set = EnumSet.noneOf(s.elementClass);
104            set.addAll(s);
105            return set;
106        }
107    
108        /**
109         * Creates an enum set. The contained elements are the same as those
110         * contained in collection {@code c}. If c is an enum set, invoking this
111         * method is the same as invoking {@link #copyOf(EnumSet)}.
112         * 
113         * @param c
114         *            the collection from which to copy. if it is not an enum set,
115         *            it must not be empty.
116         * @return an enum set with all the elements from the specified collection.
117         * @throws IllegalArgumentException
118         *             if c is not an enum set and contains no elements at all.
119         * @since Android 1.0
120         */
121        public static <E extends Enum<E>> EnumSet<E> copyOf(Collection<E> c) {
122            if (c instanceof EnumSet) {
123                return copyOf((EnumSet<E>) c);
124            }
125            if (0 == c.size()) {
126                throw new IllegalArgumentException();
127            }
128            Iterator<E> iterator = c.iterator();
129            E element = iterator.next();
130            EnumSet<E> set = EnumSet.noneOf(element.getDeclaringClass());
131            set.add(element);
132            while (iterator.hasNext()) {
133                set.add(iterator.next());
134            }
135            return set;
136        }
137    
138        /**
139         * Creates an enum set. All the contained elements complement those from the
140         * specified enum set.
141         * 
142         * @param s
143         *            the specified enum set.
144         * @return an enum set with all the elements complementary to those from the
145         *         specified enum set.
146         * @since Android 1.0
147         */
148        public static <E extends Enum<E>> EnumSet<E> complementOf(EnumSet<E> s) {
149            EnumSet<E> set = EnumSet.noneOf(s.elementClass);
150            set.addAll(s);
151            set.complement();
152            return set;
153        }
154    
155        abstract void complement();
156    
157        /**
158         * Creates a new enum set, containing only the specified element. There are
159         * six overloadings of the method. They accept from one to five elements
160         * respectively. The sixth one receives an arbitrary number of elements, and
161         * runs slower than those that only receive a fixed number of elements.
162         * 
163         * @param e
164         *            the element to be initially contained.
165         * @return an enum set containing the specified element.
166         * @since Android 1.0
167         */
168        public static <E extends Enum<E>> EnumSet<E> of(E e) {
169            EnumSet<E> set = EnumSet.noneOf(e.getDeclaringClass());
170            set.add(e);
171            return set;
172        }
173    
174        /**
175         * Creates a new enum set, containing only the specified element. There are
176         * six overloadings of the method. They accept from one to five elements
177         * respectively. The sixth one receives an arbitrary number of elements, and
178         * runs slower than those that only receive a fixed number of elements.
179         * 
180         * @param e1
181         *            the initially contained element.
182         * @param e2
183         *            another initially contained element.
184         * @return an enum set containing the specified element.
185         * @since Android 1.0
186         */
187        public static <E extends Enum<E>> EnumSet<E> of(E e1, E e2) {
188            EnumSet<E> set = of(e1);
189            set.add(e2);
190            return set;
191        }
192    
193        /**
194         * Creates a new enum set, containing only the specified element. There are
195         * six overloadings of the method. They accept from one to five elements
196         * respectively. The sixth one receives an arbitrary number of elements, and
197         * runs slower than those that only receive a fixed number of elements.
198         * 
199         * @param e1
200         *            the initially contained element.
201         * @param e2
202         *            another initially contained element.
203         * @param e3
204         *            another initially contained element.
205         * @return an enum set containing the specified element.
206         * @since Android 1.0
207         */
208        public static <E extends Enum<E>> EnumSet<E> of(E e1, E e2, E e3) {
209            EnumSet<E> set = of(e1, e2);
210            set.add(e3);
211            return set;
212        }
213    
214        /**
215         * Creates a new enum set, containing only the specified element. There are
216         * six overloadings of the method. They accept from one to five elements
217         * respectively. The sixth one receives an arbitrary number of elements, and
218         * runs slower than those that only receive a fixed number of elements.
219         * 
220         * @param e1
221         *            the initially contained element.
222         * @param e2
223         *            another initially contained element.
224         * @param e3
225         *            another initially contained element.
226         * @param e4
227         *            another initially contained element.
228         * @return an enum set containing the specified element.
229         * @since Android 1.0
230         */
231        public static <E extends Enum<E>> EnumSet<E> of(E e1, E e2, E e3, E e4) {
232            EnumSet<E> set = of(e1, e2, e3);
233            set.add(e4);
234            return set;
235        }
236    
237        /**
238         * Creates a new enum set, containing only the specified element. There are
239         * six overloadings of the method. They accept from one to five elements
240         * respectively. The sixth one receives an arbitrary number of elements, and
241         * runs slower than those that only receive a fixed number of elements.
242         * 
243         * @param e1
244         *            the initially contained element.
245         * @param e2
246         *            another initially contained element.
247         * @param e3
248         *            another initially contained element.
249         * @param e4
250         *            another initially contained element.
251         * @param e5
252         *            another initially contained element.
253         * @return an enum set containing the specified element.
254         * @since Android 1.0
255         */
256        public static <E extends Enum<E>> EnumSet<E> of(E e1, E e2, E e3, E e4, E e5) {
257            EnumSet<E> set = of(e1, e2, e3, e4);
258            set.add(e5);
259            return set;
260        }
261    
262        /**
263         * Creates a new enum set, containing only the specified elements. It can
264         * receive an arbitrary number of elements, and runs slower than those only
265         * receiving a fixed number of elements.
266         * 
267         * @param start
268         *            the first initially contained element.
269         * @param others
270         *            the other initially contained elements.
271         * @return an enum set containing the specified elements.
272         * @since Android 1.0
273         */
274        public static <E extends Enum<E>> EnumSet<E> of(E start, E... others) {
275            EnumSet<E> set = of(start);
276            for (E e : others) {
277                set.add(e);
278            }
279            return set;
280        }
281    
282        /**
283         * Creates an enum set containing all the elements within the range defined
284         * by {@code start} and {@code end} (inclusive). All the elements must be in
285         * order.
286         * 
287         * @param start
288         *            the element used to define the beginning of the range.
289         * @param end
290         *            the element used to define the end of the range.
291         * @return an enum set with elements in the range from start to end.
292         * @throws IllegalArgumentException
293         *             if {@code start} is behind {@code end}.
294         * @since Android 1.0
295         */
296        public static <E extends Enum<E>> EnumSet<E> range(E start, E end) {
297            if (start.compareTo(end) > 0) {
298                throw new IllegalArgumentException();
299            }
300            EnumSet<E> set = EnumSet.noneOf(start.getDeclaringClass());
301            set.setRange(start, end);
302            return set;
303        }
304    
305        abstract void setRange(E start, E end);
306    
307        /**
308         * Creates a new enum set with the same elements as those contained in this
309         * enum set.
310         * 
311         * @return a new enum set with the same elements as those contained in this
312         *         enum set.
313         * @since Android 1.0
314         */
315        @SuppressWarnings("unchecked")
316        @Override
317        public EnumSet<E> clone() {
318            try {
319                Object set = super.clone();
320                return (EnumSet<E>) set;
321            } catch (CloneNotSupportedException e) {
322                return null;
323            }
324        }
325    
326        @SuppressWarnings("unchecked")
327        boolean isValidType(Class cls) {
328            return cls == elementClass || cls.getSuperclass() == elementClass;
329        }
330    
331        private static class SerializationProxy<E extends Enum<E>> implements
332                Serializable {
333    
334            private static final long serialVersionUID = 362491234563181265L;
335    
336            private Class<E> elementType;
337    
338            private E[] elements;
339    
340            private Object readResolve() {
341                EnumSet<E> set = EnumSet.noneOf(elementType);
342                for (E e : elements) {
343                    set.add(e);
344                }
345                return set;
346            }
347        }
348    
349        @SuppressWarnings("unchecked")
350        Object writeReplace() {
351            SerializationProxy proxy = new SerializationProxy();
352            proxy.elements = toArray(new Enum[0]);
353            proxy.elementType = elementClass;
354            return proxy;
355        }
