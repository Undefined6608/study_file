025    /**
026     * IdentityHashMap is a variant on HashMap which tests equality by reference
027     * instead of equality by value. Basically, keys and values are compared for
028     * equality by checking if their references are equal rather than by calling the
029     * "equals" function.
030     * <p>
031     * <b>Note: This class intentionally violates the general contract of {@code
032     * Map}'s on comparing objects by their {@code equals} method.</b>
033     * </p>
034     * <p>
035     * IdentityHashMap uses open addressing (linear probing in particular) for
036     * collision resolution. This is different from HashMap which uses Chaining.
037     * </p>
038     * Like HashMap, IdentityHashMap is not thread safe, so access by multiple
039     * threads must be synchronized by an external mechanism such as
040     * Collections.synchronizedMap.
041     * 
042     * @since Android 1.0
043     */
044    public class IdentityHashMap<K, V> extends AbstractMap<K, V> implements
045            Map<K, V>, Serializable, Cloneable {
046    
047        private static final long serialVersionUID = 8188218128353913216L;
048    
049        /*
050         * The internal data structure to hold key value pairs This array holds keys
051         * and values in an alternating fashion.
052         */
053        transient Object[] elementData;
054    
055        /* Actual number of key-value pairs. */
056        int size;
057    
058        /*
059         * maximum number of elements that can be put in this map before having to
060         * rehash.
061         */
062        transient int threshold;
063    
064        /*
065         * default threshold value that an IdentityHashMap created using the default
066         * constructor would have.
067         */
068        private static final int DEFAULT_MAX_SIZE = 21;
069    
070        /* Default load factor of 0.75; */
071        private static final int loadFactor = 7500;
072    
073        /*
074         * modification count, to keep track of structural modifications between the
075         * IdentityHashMap and the iterator
076         */
077        transient int modCount = 0;
078    
079        /*
080         * Object used to represent null keys and values. This is used to
081         * differentiate a literal 'null' key value pair from an empty spot in the
082         * map.
083         */
084        private static final Object NULL_OBJECT = new Object();  //$NON-LOCK-1$
085    
086        static class IdentityHashMapEntry<K, V> extends MapEntry<K, V> {
087            IdentityHashMapEntry(K theKey, V theValue) {
088                super(theKey, theValue);
089            }
090    
091            @Override
092            public Object clone() {
093                return super.clone();
094            }
095    
096            @Override
097            public boolean equals(Object object) {
098                if (this == object) {
099                    return true;
100                }
101                if (object instanceof Map.Entry) {
102                    Map.Entry<?, ?> entry = (Map.Entry) object;
103                    return (key == entry.getKey()) && (value == entry.getValue());
104                }
105                return false;
106            }
107    
108            @Override
109            public int hashCode() {
110                return System.identityHashCode(key)
111                        ^ System.identityHashCode(value);
112            }
113    
114            @Override
115            public String toString() {
116                return key + "=" + value; //$NON-NLS-1$
117            }
118        }

120        static class IdentityHashMapIterator<E, KT, VT> implements Iterator<E> {
121            private int position = 0; // the current position
122    
123            // the position of the entry that was last returned from next()
124            private int lastPosition = 0;
125    
126            final IdentityHashMap<KT, VT> associatedMap;
127    
128            int expectedModCount;
129    
130            final MapEntry.Type<E, KT, VT> type;
131    
132            boolean canRemove = false;
133    
134            IdentityHashMapIterator(MapEntry.Type<E, KT, VT> value,
135                    IdentityHashMap<KT, VT> hm) {
136                associatedMap = hm;
137                type = value;
138                expectedModCount = hm.modCount;
139            }
140    
141            public boolean hasNext() {
142                while (position < associatedMap.elementData.length) {
143                    // if this is an empty spot, go to the next one
144                    if (associatedMap.elementData[position] == null) {
145                        position += 2;
146                    } else {
147                        return true;
148                    }
149                }
150                return false;
151            }
152    
153            void checkConcurrentMod() throws ConcurrentModificationException {
154                if (expectedModCount != associatedMap.modCount) {
155                    throw new ConcurrentModificationException();
156                }
157            }
158    
159            public E next() {
160                checkConcurrentMod();
161                if (!hasNext()) {
162                    throw new NoSuchElementException();
163                }
164    
165                IdentityHashMapEntry<KT, VT> result = associatedMap
166                        .getEntry(position);
167                lastPosition = position;
168                position += 2;
169    
170                canRemove = true;
171                return type.get(result);
172            }
173    
174            public void remove() {
175                checkConcurrentMod();
176                if (!canRemove) {
177                    throw new IllegalStateException();
178                }
179    
180                canRemove = false;
181                associatedMap.remove(associatedMap.elementData[lastPosition]);
182                position = lastPosition;
183                expectedModCount++;
184            }
185        }

187        static class IdentityHashMapEntrySet<KT, VT> extends
188                AbstractSet<Map.Entry<KT, VT>> {
189            private final IdentityHashMap<KT, VT> associatedMap;
190    
191            public IdentityHashMapEntrySet(IdentityHashMap<KT, VT> hm) {
192                associatedMap = hm;
193            }
194    
195            IdentityHashMap<KT, VT> hashMap() {
196                return associatedMap;
197            }
198    
199            @Override
200            public int size() {
201                return associatedMap.size;
202            }
203    
204            @Override
205            public void clear() {
206                associatedMap.clear();
207            }
208    
209            @Override
210            public boolean remove(Object object) {
211                if (contains(object)) {
212                    associatedMap.remove(((Map.Entry) object).getKey());
213                    return true;
214                }
215                return false;
216            }
217    
218            @Override
219            public boolean contains(Object object) {
220                if (object instanceof Map.Entry) {
221                    IdentityHashMapEntry<?, ?> entry = associatedMap
222                            .getEntry(((Map.Entry) object).getKey());
223                    // we must call equals on the entry obtained from "this"
224                    return entry != null && entry.equals(object);
225                }
226                return false;
227            }
228    
229            @Override
230            public Iterator<Map.Entry<KT, VT>> iterator() {
231                return new IdentityHashMapIterator<Map.Entry<KT, VT>, KT, VT>(
232                        new MapEntry.Type<Map.Entry<KT, VT>, KT, VT>() {
233                            public Map.Entry<KT, VT> get(MapEntry<KT, VT> entry) {
234                                return entry;
235                            }
236                        }, associatedMap);
237            }
238        }

245        public IdentityHashMap() {
246            this(DEFAULT_MAX_SIZE);
247        }

249        /**
250         * Creates an IdentityHashMap with the specified maximum size parameter.
251         * 
252         * @param maxSize
253         *            The estimated maximum number of entries that will be put in
254         *            this map.
255         * @since Android 1.0
256         */
257        public IdentityHashMap(int maxSize) {
258            if (maxSize >= 0) {
259                this.size = 0;
260                threshold = getThreshold(maxSize);
261                elementData = newElementArray(computeElementArraySize());
262            } else {
263                throw new IllegalArgumentException();
264            }
265        }
266    
267        private int getThreshold(int maxSize) {
268            // assign the threshold to maxSize initially, this will change to a
269            // higher value if rehashing occurs.
270            return maxSize > 3 ? maxSize : 3;
271        }
272    
273        private int computeElementArraySize() {
274            return (int) (((long) threshold * 10000) / loadFactor) * 2;
275        }
276    
277        private Object[] newElementArray(int s) {
278            return new Object[s];
279        }

282         * Creates an IdentityHashMap using the given map as initial values.
283         * 
284         * @param map
285         *            A map of (key,value) pairs to copy into the IdentityHashMap.
286         * @since Android 1.0

288        public IdentityHashMap(Map<? extends K, ? extends V> map) {
289            this(map.size() < 6 ? 11 : map.size() * 2);
290            putAllImpl(map);

293        @SuppressWarnings("unchecked")
294        private V massageValue(Object value) {
295            return (V) ((value == NULL_OBJECT) ? null : value);
296        }

298        /**
299         * Removes all elements from this map, leaving it empty.
300         * 
301         * @see #isEmpty()
302         * @see #size()
303         * @since Android 1.0
304         */
305        @Override
306        public void clear() {
307            size = 0;
308            for (int i = 0; i < elementData.length; i++) {
309                elementData[i] = null;
310            }
311            modCount++;
312        }

314        /**
315         * Returns whether this map contains the specified key.
316         * 
317         * @param key
318         *            the key to search for.
319         * @return {@code true} if this map contains the specified key,
320         *         {@code false} otherwise.
321         * @since Android 1.0
322         */
323        @Override
324        public boolean containsKey(Object key) {
325            if (key == null) {
326                key = NULL_OBJECT;
327            }
328    
329            int index = findIndex(key, elementData);
330            return elementData[index] == key;
331        }
332    
333        /**
334         * Returns whether this map contains the specified value.
335         * 
336         * @param value
337         *            the value to search for.
338         * @return {@code true} if this map contains the specified value,
339         *         {@code false} otherwise.
340         * @since Android 1.0
341         */
342        @Override
343        public boolean containsValue(Object value) {
344            if (value == null) {
345                value = NULL_OBJECT;
346            }
347    
348            for (int i = 1; i < elementData.length; i = i + 2) {
349                if (elementData[i] == value) {
350                    return true;
351                }
352            }
353            return false;
354        }

356        /**
357         * Returns the value of the mapping with the specified key.
358         * 
359         * @param key
360         *            the key.
361         * @return the value of the mapping with the specified key.
362         * @since Android 1.0
363         */
364        @Override
365        public V get(Object key) {
366            if (key == null) {
367                key = NULL_OBJECT;
368            }
369    
370            int index = findIndex(key, elementData);
371    
372            if (elementData[index] == key) {
373                Object result = elementData[index + 1];
374                return massageValue(result);
375            }
376    
377            return null;
378        }

380        private IdentityHashMapEntry<K, V> getEntry(Object key) {
381            if (key == null) {
382                key = NULL_OBJECT;
383            }
384    
385            int index = findIndex(key, elementData);
386            if (elementData[index] == key) {
387                return getEntry(index);
388            }
389    
390            return null;
391        }
392    
393        @SuppressWarnings("unchecked")
394        private IdentityHashMapEntry<K, V> getEntry(int index) {
395            Object key = elementData[index];
396            Object value = elementData[index + 1];
397    
398            if (key == NULL_OBJECT) {
399                key = null;
400            }
401            if (value == NULL_OBJECT) {
402                value = null;
403            }
404    
405            return new IdentityHashMapEntry<K, V>((K) key, (V) value);
406        }
407    
408        private int findIndex(Object key, Object[] array) {
409            int length = array.length;
410            int index = getModuloHash(key, length);
411            int last = (index + length - 2) % length;
412            while (index != last) {
413                if (array[index] == key || (array[index] == null)) {
414                    /*
415                     * Found the key, or the next empty spot (which means key is not
416                     * in the table)
417                     */
418                    break;
419                }
420                index = (index + 2) % length;
421            }
422            return index;
423        }
424    
425        private int getModuloHash(Object key, int length) {
426            return ((System.identityHashCode(key) & 0x7FFFFFFF) % (length / 2)) * 2;
427        }

429        /**
430         * Maps the specified key to the specified value.
431         * 
432         * @param key
433         *            the key.
434         * @param value
435         *            the value.
436         * @return the value of any previous mapping with the specified key or
437         *         {@code null} if there was no such mapping.
438         * @since Android 1.0
439         */
440        @Override
441        public V put(K key, V value) {
442            Object _key = key;
443            Object _value = value;
444            if (_key == null) {
445                _key = NULL_OBJECT;
446            }
447    
448            if (_value == null) {
449                _value = NULL_OBJECT;
450            }
451    
452            int index = findIndex(_key, elementData);
453    
454            // if the key doesn't exist in the table
455            if (elementData[index] != _key) {
456                modCount++;
457                if (++size > threshold) {
458                    rehash();
459                    index = findIndex(_key, elementData);
460                }
461    
462                // insert the key and assign the value to null initially
463                elementData[index] = _key;
464                elementData[index + 1] = null;
465            }
466    
467            // insert value to where it needs to go, return the old value
468            Object result = elementData[index + 1];
469            elementData[index + 1] = _value;
470    
471            return massageValue(result);
472        }

475         * Copies all the mappings in the specified map to this map. These mappings
476         * will replace all mappings that this map had for any of the keys currently
477         * in the given map.

480         *            the map to copy mappings from.

488        private void rehash() {
489            int newlength = elementData.length << 1;
490            if (newlength == 0) {
491                newlength = 1;
492            }
493            Object[] newData = newElementArray(newlength);
494            for (int i = 0; i < elementData.length; i = i + 2) {
495                Object key = elementData[i];
496                if (key != null) {
497                    // if not empty
498                    int index = findIndex(key, newData);
499                    newData[index] = key;
500                    newData[index + 1] = elementData[i + 1];
501                }
502            }
503            elementData = newData;
504            computeMaxSize();
505        }
506    
507        private void computeMaxSize() {
508            threshold = (int) ((long) (elementData.length / 2) * loadFactor / 10000);
509        }

511        /**
512         * Removes the mapping with the specified key from this map.
513         * 
514         * @param key
515         *            the key of the mapping to remove.
516         * @return the value of the removed mapping, or {@code null} if no mapping
517         *         for the specified key was found.
518         * @since Android 1.0
519         */
520        @Override
521        public V remove(Object key) {
522            if (key == null) {
523                key = NULL_OBJECT;
524            }
525    
526            boolean hashedOk;
527            int index, next, hash;
528            Object result, object;
529            index = next = findIndex(key, elementData);
530    
531            if (elementData[index] != key) {
532                return null;
533            }
534    
535            // store the value for this key
536            result = elementData[index + 1];
537    
538            // shift the following elements up if needed
539            // until we reach an empty spot
540            int length = elementData.length;
541            while (true) {
542                next = (next + 2) % length;
543                object = elementData[next];
544                if (object == null) {
545                    break;
546                }
547    
548                hash = getModuloHash(object, length);
549                hashedOk = hash > index;
550                if (next < index) {
551                    hashedOk = hashedOk || (hash <= next);
552                } else {
553                    hashedOk = hashedOk && (hash <= next);
554                }
555                if (!hashedOk) {
556                    elementData[index] = object;
557                    elementData[index + 1] = elementData[next + 1];
558                    index = next;
559                }
560            }
561    
562            size--;
563            modCount++;
564    
565            // clear both the key and the value
566            elementData[index] = null;
567            elementData[index + 1] = null;
568    
569            return massageValue(result);
570        }

635        /**
636         * Returns a collection of the values contained in this map. The collection
637         * is backed by this map so changes to one are reflected by the other. The
638         * collection supports remove, removeAll, retainAll and clear operations,
639         * and it does not support add or addAll operations.
640         * <p>
641         * This method returns a collection which is the subclass of
642         * AbstractCollection. The iterator method of this subclass returns a
643         * "wrapper object" over the iterator of map's entrySet(). The {@code size}
644         * method wraps the map's size method and the {@code contains} method wraps
645         * the map's containsValue method.
646         * </p>
647         * <p>
648         * The collection is created when this method is called for the first time
649         * and returned in response to all subsequent calls. This method may return
650         * different collections when multiple concurrent calls occur, since no
651         * synchronization is performed.
652         * </p>
653         * 
654         * @return a collection of the values contained in this map.
655         * @since Android 1.0
656         */
657        @Override
658        public Collection<V> values() {
659            if (valuesCollection == null) {
660                valuesCollection = new AbstractCollection<V>() {
661                    @Override
662                    public boolean contains(Object object) {
663                        return containsValue(object);
664                    }
665    
666                    @Override
667                    public int size() {
668                        return IdentityHashMap.this.size();
669                    }
670    
671                    @Override
672                    public void clear() {
673                        IdentityHashMap.this.clear();
674                    }
675    
676                    @Override
677                    public Iterator<V> iterator() {
678                        return new IdentityHashMapIterator<V, K, V>(
679                                new MapEntry.Type<V, K, V>() {
680                                    public V get(MapEntry<K, V> entry) {
681                                        return entry.value;
682                                    }
683                                }, IdentityHashMap.this);
684                    }
685    
686                    @Override
687                    public boolean remove(Object object) {
688                        Iterator<?> it = iterator();
689                        while (it.hasNext()) {
690                            if (object == it.next()) {
691                                it.remove();
692                                return true;
693                            }
694                        }
695                        return false;
696                    }
697                };
698            }
699            return valuesCollection;
700        }

702        /**
703         * Compares this map with other objects. This map is equal to another map is
704         * it represents the same set of mappings. With this map, two mappings are
705         * the same if both the key and the value are equal by reference. When
706         * compared with a map that is not an IdentityHashMap, the equals method is
707         * neither necessarily symmetric (a.equals(b) implies b.equals(a)) nor
708         * transitive (a.equals(b) and b.equals(c) implies a.equals(c)).
709         * 
710         * @param object
711         *            the object to compare to.
712         * @return whether the argument object is equal to this object.
713         * @since Android 1.0
714         */
715        @Override
716        public boolean equals(Object object) {
717            /*
718             * We need to override the equals method in AbstractMap because
719             * AbstractMap.equals will call ((Map) object).entrySet().contains() to
720             * determine equality of the entries, so it will defer to the argument
721             * for comparison, meaning that reference-based comparison will not take
722             * place. We must ensure that all comparison is implemented by methods
723             * in this class (or in one of our inner classes) for reference-based
724             * comparison to take place.
725             */
726            if (this == object) {
727                return true;
728            }
729            if (object instanceof Map) {
730                Map<?, ?> map = (Map) object;
731                if (size() != map.size()) {
732                    return false;
733                }
734    
735                Set<Map.Entry<K, V>> set = entrySet();
736                // ensure we use the equals method of the set created by "this"
737                return set.equals(map.entrySet());
738            }
739            return false;
740        }

783        private void writeObject(ObjectOutputStream stream) throws IOException {
784            stream.defaultWriteObject();
785            stream.writeInt(size);
786            Iterator<?> iterator = entrySet().iterator();
787            while (iterator.hasNext()) {
788                MapEntry<?, ?> entry = (MapEntry) iterator.next();
789                stream.writeObject(entry.key);
790                stream.writeObject(entry.value);
791            }
792        }

794        @SuppressWarnings("unchecked")
795        private void readObject(ObjectInputStream stream) throws IOException,
796                ClassNotFoundException {
797            stream.defaultReadObject();
798            int savedSize = stream.readInt();
799            threshold = getThreshold(DEFAULT_MAX_SIZE);
800            elementData = newElementArray(computeElementArraySize());
801            for (int i = savedSize; --i >= 0;) {
802                K key = (K) stream.readObject();
803                put(key, (V) stream.readObject());
804            }
805            size = savedSize;
806        }

808        private void putAllImpl(Map<? extends K, ? extends V> map) {
