018    package java.util;
019    
020    import org.apache.harmony.luni.util.Msg;

022    /**
023     * {@code Timer}s are used to schedule jobs for execution in a background process. A
024     * single thread is used for the scheduling and this thread has the option of
025     * being a daemon thread. By calling {@code cancel} you can terminate a
026     * {@code Timer} and its associated thread. All tasks which are scheduled to run after
027     * this point are cancelled. Tasks are executed sequentially but are subject to
028     * the delays from other tasks run methods. If a specific task takes an
029     * excessive amount of time to run it may impact the time at which subsequent
030     * tasks may run.
031     * <p>
032     * 
033     * The {@code TimerTask} does not offer any guarantees about the real-time nature of
034     * scheduling tasks as its underlying implementation relies on the
035     * {@code Object.wait(long)} method.
036     * </p>
037     * <p>
038     * Multiple threads can share a single {@code Timer} without the need for their own
039     * synchronization.</p>
040     * <p>
041     * A {@code Timer} can be set to schedule tasks either at a fixed rate or 
042     * with a fixed period. Fixed-period execution is the default.</p>
043     * <p>
044     * The difference between fixed-rate and fixed-period execution 
045     * is the following:  With fixed-rate execution, the start time of each 
046     * successive run of the task is scheduled in absolute terms without regard for when the previous
047     * task run actually took place. This can result in a series of bunched-up runs 
048     * (one launched immediately after another) if busy resources or other 
049     * system delays prevent the {@code Timer} from firing for an extended time.
050     * With fixed-period execution, each successive run of the 
051     * task is scheduled relative to the start time of the previous run of the 
052     * task, so two runs of the task are never fired closer together in time than 
053     * the specified {@code period}.</p>
054     * </p>
055     * 
056     * @see TimerTask
057     * @see java.lang.Object#wait(long)
058     * @since Android 1.0
059     */
060    public class Timer {
061    
062        private static final class TimerImpl extends Thread {
063    
064            private static final class TimerNode {
065                TimerNode parent, left, right;
066    
067                TimerTask task;
068    
069                public TimerNode(TimerTask value) {
070                    this.task = value;
071                }
072    
073                public void deleteIfCancelled(TimerTree tasks) {
074                    /*
075                     * All changes in the tree structure during deleting this node
076                     * affect only the structure of the subtree having this node as
077                     * its root
078                     */
079                    if (left != null) {
080                        left.deleteIfCancelled(tasks);
081                    }
082                    if (right != null) {
083                        right.deleteIfCancelled(tasks);
084                    }
085                    if (task.cancelled) {
086                        tasks.delete(this);
087                        tasks.deletedCancelledNumber++;
088                    }
089                }
090            }
091    
092            private static final class TimerTree {
093    
094                int deletedCancelledNumber;
095    
096                TimerNode root;
097    
098                boolean isEmpty() {
099                    return root == null;
100                }
101    
102                void insert(TimerNode z) {
103                    TimerNode y = null, x = root;
104                    while (x != null) {
105                        y = x;
106                        if (z.task.getWhen() < x.task.getWhen()) {
107                            x = x.left;
108                        } else {
109                            x = x.right;
110                        }
111                    }
112                    z.parent = y;
113                    if (y == null) {
114                        root = z;
115                    } else if (z.task.getWhen() < y.task.getWhen()) {
116                        y.left = z;
117                    } else {
118                        y.right = z;
119                    }
120                }
121    
122                void delete(TimerNode z) {
123                    TimerNode y = null, x = null;
124                    if (z.left == null || z.right == null) {
125                        y = z;
126                    } else {
127                        y = successor(z);
128                    }
129                    if (y.left != null) {
130                        x = y.left;
131                    } else {
132                        x = y.right;
133                    }
134                    if (x != null) {
135                        x.parent = y.parent;
136                    }
137                    if (y.parent == null) {
138                        root = x;
139                    } else if (y == y.parent.left) {
140                        y.parent.left = x;
141                    } else {
142                        y.parent.right = x;
143                    }
144                    if (y != z) {
145                        z.task = y.task;
146                    }
147                }
148    
149                private TimerNode successor(TimerNode x) {
150                    if (x.right != null) {
151                        return minimum(x.right);
152                    }
153                    TimerNode y = x.parent;
154                    while (y != null && x == y.right) {
155                        x = y;
156                        y = y.parent;
157                    }
158                    return y;
159                }
160    
161                private TimerNode minimum(TimerNode x) {
162                    while (x.left != null) {
163                        x = x.left;
164                    }
165                    return x;
166                }
167    
168                TimerNode minimum() {
169                    return minimum(root);
170                }
171            }

173            /**
174             * True if the method cancel() of the Timer was called or the !!!stop()
175             * method was invoked
176             */
177            private boolean cancelled;
178    
179            /**
180             * True if the Timer has become garbage
181             */
182            private boolean finished;
183    
184            /**
185             * Vector consists of scheduled events, sorted according to
186             * {@code when} field of TaskScheduled object.
187             */
188            private TimerTree tasks = new TimerTree();
189    
190            /**
191             * Starts a new timer.
192             * 
193             * @param isDaemon
194             */
195            TimerImpl(boolean isDaemon) {
196                this.setDaemon(isDaemon);
197                this.start();
198            }

200            TimerImpl(String name, boolean isDaemon) {
201                this.setName(name);
202                this.setDaemon(isDaemon);
203                this.start();
204            }

206            /**
207             * This method will be launched on separate thread for each Timer
208             * object.
209             */
210            @Override
211            public void run() {
212                while (true) {
213                    TimerTask task;
214                    synchronized (this) {
215                        // need to check cancelled inside the synchronized block
216                        if (cancelled) {
217                            return;
218                        }
219                        if (tasks.isEmpty()) {
220                            if (finished) {
221                                return;
222                            }
223                            // no tasks scheduled -- sleep until any task appear
224                            try {
225                                this.wait();
226                            } catch (InterruptedException e) {
227                            }
228                            continue;
229                        }
230    
231                        long currentTime = System.currentTimeMillis();
232    
233                        TimerNode taskNode = tasks.minimum();
234                        task = taskNode.task;
235                        long timeToSleep;
236    
237                        synchronized (task.lock) {
238                            if (task.cancelled) {
239                                tasks.delete(taskNode);
240                                continue;
241                            }
242    
243                            // check the time to sleep for the first task scheduled
244                            timeToSleep = task.when - currentTime;
245                        }
246    
247                        if (timeToSleep > 0) {
248                            // sleep!
249                            try {
250                                this.wait(timeToSleep);
251                            } catch (InterruptedException e) {
252                                // Ignored
253                            }
254                            continue;
255                        }
256    
257                        // no sleep is necessary before launching the task
258    
259                        synchronized (task.lock) {
260                            if (task.cancelled) {
261                                tasks.delete(taskNode);
262                                continue;
263                            }
264    
265                            // set time to schedule
266                            task.setScheduledTime(task.when);
267    
268                            // remove task from queue
269                            tasks.delete(taskNode);
270    
271                            // set when the next task should be launched
272                            if (task.period >= 0) {
273                                // this is a repeating task,
274                                if (task.fixedRate) {
275                                    // task is scheduled at fixed rate
276                                    task.when = task.when + task.period;
277                                } else {
278                                    // task is scheduled at fixed delay
279                                    task.when = System.currentTimeMillis()
280                                            + task.period;
281                                }
282    
283                                // insert this task into queue
284                                insertTask(task);
285                            } else {
286                                task.when = 0;
287                            }
288                        }
289                    }
290    
291                    // run the task
292                    try {
293                        task.run();
294                    } catch (Exception e) {
295                        // Ignored
296                    }
297                }
298            }
299    
300            private void insertTask(TimerTask newTask) {
301                // callers are synchronized
302                tasks.insert(new TimerNode(newTask));
303                this.notify();
304            }
305    
306            /**
307             * Cancels timer.
308             */
309            public synchronized void cancel() {
310                cancelled = true;
311                tasks = new TimerTree();
312                this.notify();
313            }
314    
315            public int purge() {
316                if (tasks.isEmpty()) {
317                    return 0;
318                }
319                // callers are synchronized
320                tasks.deletedCancelledNumber = 0;
321                tasks.root.deleteIfCancelled(tasks);
322                return tasks.deletedCancelledNumber;
323            }
324    
325        }

332        private Object finalizer = new Object() { // $NON-LOCK-1$
333            @Override
334            protected void finalize() {
335                synchronized (impl) {
336                    impl.finished = true;
337                    impl.notify();
338                }
339            }

417        /**
418         * Schedule a task for single execution. If {@code when} is less than the
419         * current time, it will be scheduled to be executed as soon as possible.
420         * 
421         * @param task
422         *            the task to schedule.
423         * @param when
424         *            time of execution.
425         * @exception IllegalArgumentException
426         *                if {@code when.getTime() < 0}.
427         * @exception IllegalStateException
428         *                if the {@code Timer} has been canceled, or if the task has been
429         *                scheduled or canceled.
430         * @since Android 1.0
431         */
432        public void schedule(TimerTask task, Date when) {
433            if (when.getTime() < 0) {
434                throw new IllegalArgumentException();
435            }
436            long delay = when.getTime() - System.currentTimeMillis();
437            scheduleImpl(task, delay < 0 ? 0 : delay, -1, false);
438        }

558        /**
559         * Schedule a task.
560         * 
561         * @param task
562         * @param delay
563         * @param period
564         * @param fixed
565         */
566        private void scheduleImpl(TimerTask task, long delay, long period,
567                boolean fixed) {
568            synchronized (impl) {
569                if (impl.cancelled) {
570                    throw new IllegalStateException(Msg.getString("K00f3")); //$NON-NLS-1$
571                }
572    
573                long when = delay + System.currentTimeMillis();
574    
575                if (when < 0) {
576                    throw new IllegalArgumentException(Msg.getString("K00f5")); //$NON-NLS-1$
577                }
578    
579                synchronized (task.lock) {
580                    if (task.isScheduled()) {
581                        throw new IllegalStateException(Msg.getString("K00f6")); //$NON-NLS-1$
582                    }
583    
584                    if (task.cancelled) {
585                        throw new IllegalStateException(Msg.getString("K00f7")); //$NON-NLS-1$
586                    }
587    
588                    task.when = when;
589                    task.period = period;
590                    task.fixedRate = fixed;
591                }
592    
593                // insert the newTask into queue
594                impl.insertTask(task);
595            }
596        }
