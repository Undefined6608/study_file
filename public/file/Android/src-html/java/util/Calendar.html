038    /**
039     * {@code Calendar} is an abstract base class for converting between a
040     * {@code Date} object and a set of integer fields such as
041     * {@code YEAR}, {@code MONTH}, {@code DAY},
042     * {@code HOUR}, and so on. (A {@code Date} object represents a
043     * specific instant in time with millisecond precision. See {@link Date} for
044     * information about the {@code Date} class.)
045     * 
046     * <p>
047     * Subclasses of {@code Calendar} interpret a {@code Date}
048     * according to the rules of a specific calendar system.
049     * 
050     * <p>
051     * Like other locale-sensitive classes, {@code Calendar} provides a class
052     * method, {@code getInstance}, for getting a default instance of
053     * this class for general use. {@code Calendar}'s {@code getInstance} method
054     * returns a calendar whose locale is based on system settings and whose time fields
055     * have been initialized with the current date and time: <blockquote>
056     * 
057     * <pre>Calendar rightNow = Calendar.getInstance()</pre>
058     * 
059     * </blockquote>
060     * 
061     * <p>
062     * A {@code Calendar} object can produce all the time field values needed
063     * to implement the date-time formatting for a particular language and calendar
064     * style (for example, Japanese-Gregorian, Japanese-Traditional).
065     * {@code Calendar} defines the range of values returned by certain
066     * fields, as well as their meaning. For example, the first month of the year
067     * has value {@code MONTH} == {@code JANUARY} for all calendars.
068     * Other values are defined by the concrete subclass, such as {@code ERA}
069     * and {@code YEAR}. See individual field documentation and subclass
070     * documentation for details.
071     * 
072     * <p>
073     * When a {@code Calendar} is <em>lenient</em>, it accepts a wider
074     * range of field values than it produces. For example, a lenient
075     * {@code GregorianCalendar} interprets {@code MONTH} ==
076     * {@code JANUARY}, {@code DAY_OF_MONTH} == 32 as February 1. A
077     * non-lenient {@code GregorianCalendar} throws an exception when given
078     * out-of-range field settings. When calendars recompute field values for return
079     * by {@code get()}, they normalize them. For example, a
080     * {@code GregorianCalendar} always produces {@code DAY_OF_MONTH}
081     * values between 1 and the length of the month.
082     * 
083     * <p>
084     * {@code Calendar} defines a locale-specific seven day week using two
085     * parameters: the first day of the week and the minimal days in first week
086     * (from 1 to 7). These numbers are taken from the locale resource data when a
087     * {@code Calendar} is constructed. They may also be specified explicitly
088     * through the API.
089     * 
090     * <p>
091     * When setting or getting the {@code WEEK_OF_MONTH} or
092     * {@code WEEK_OF_YEAR} fields, {@code Calendar} must determine
093     * the first week of the month or year as a reference point. The first week of a
094     * month or year is defined as the earliest seven day period beginning on
095     * {@code getFirstDayOfWeek()} and containing at least
096     * {@code getMinimalDaysInFirstWeek()} days of that month or year. Weeks
097     * numbered ..., -1, 0 precede the first week; weeks numbered 2, 3,... follow
098     * it. Note that the normalized numbering returned by {@code get()} may
099     * be different. For example, a specific {@code Calendar} subclass may
100     * designate the week before week 1 of a year as week <em>n</em> of the
101     * previous year.
102     * 
103     * <p>
104     * When computing a {@code Date} from time fields, two special
105     * circumstances may arise: there may be insufficient information to compute the
106     * {@code Date} (such as only year and month but no day in the month), or
107     * there may be inconsistent information (such as "Tuesday, July 15, 1996" --
108     * July 15, 1996 is actually a Monday).
109     * 
110     * <p>
111     * <strong>Insufficient information.</strong> The calendar will use default
112     * information to specify the missing fields. This may vary by calendar; for the
113     * Gregorian calendar, the default for a field is the same as that of the start
114     * of the epoch: i.e., YEAR = 1970, MONTH = JANUARY, DATE = 1, etc.
115     * 
116     * <p>
117     * <strong>Inconsistent information.</strong> If fields conflict, the calendar
118     * will give preference to fields set more recently. For example, when
119     * determining the day, the calendar will look for one of the following
120     * combinations of fields. The most recent combination, as determined by the
121     * most recently set single field, will be used.
122     * 
123     * <blockquote>
124     * 
125     * <pre>
126     * MONTH + DAY_OF_MONTH
127     * MONTH + WEEK_OF_MONTH + DAY_OF_WEEK
128     * MONTH + DAY_OF_WEEK_IN_MONTH + DAY_OF_WEEK
129     * DAY_OF_YEAR
130     * DAY_OF_WEEK + WEEK_OF_YEAR</pre>
131     * 
132     * </blockquote>
133     * 
134     * For the time of day:
135     * 
136     * <blockquote>
137     * 
138     * <pre>
139     * HOUR_OF_DAY
140     * AM_PM + HOUR</pre>
141     * 
142     * </blockquote>
143     * 
144     * <p>
145     * <strong>Note:</strong> There are certain possible ambiguities in
146     * interpretation of certain singular times, which are resolved in the following
147     * ways:
148     * <ol>
149     * <li> 24:00:00 "belongs" to the following day. That is, 23:59 on Dec 31, 1969
150     * &lt; 24:00 on Jan 1, 1970 &lt; 24:01:00 on Jan 1, 1970 form a sequence of 
151     * three consecutive minutes in time. 
152     * 
153     * <li> Although historically not precise, midnight also belongs to "am", and
154     * noon belongs to "pm", so on the same day, we have 12:00 am (midnight) &lt; 12:01 am,
155     * and 12:00 pm (noon) &lt; 12:01 pm
156     * </ol>
157     * 
158     * <p>
159     * The date or time format strings are not part of the definition of a calendar,
160     * as those must be modifiable or overridable by the user at runtime. Use
161     * {@link java.text.DateFormat} to format dates.
162     * 
163     * <p>
164     * <strong>Field manipulation methods</strong>
165     * </p>
166     * 
167     * <p>
168     * {@code Calendar} fields can be changed using three methods:
169     * {@code set()}, {@code add()}, and {@code roll()}.
170     * </p>
171     * 
172     * <p>
173     * <strong>{@code set(f, value)}</strong> changes field {@code f}
174     * to {@code value}. In addition, it sets an internal member variable to
175     * indicate that field {@code f} has been changed. Although field
176     * {@code f} is changed immediately, the calendar's milliseconds is not
177     * recomputed until the next call to {@code get()},
178     * {@code getTime()}, or {@code getTimeInMillis()} is made. Thus,
179     * multiple calls to {@code set()} do not trigger multiple, unnecessary
180     * computations. As a result of changing a field using {@code set()},
181     * other fields may also change, depending on the field, the field value, and
182     * the calendar system. In addition, {@code get(f)} will not necessarily
183     * return {@code value} after the fields have been recomputed. The
184     * specifics are determined by the concrete calendar class.
185     * </p>
186     * 
187     * <p>
188     * <em>Example</em>: Consider a {@code GregorianCalendar} originally
189     * set to August 31, 1999. Calling <code>set(Calendar.MONTH,
190     * Calendar.SEPTEMBER)</code>
191     * sets the calendar to September 31, 1999. This is a temporary internal
192     * representation that resolves to October 1, 1999 if {@code getTime()}is
193     * then called. However, a call to {@code set(Calendar.DAY_OF_MONTH, 30)}
194     * before the call to {@code getTime()} sets the calendar to September
195     * 30, 1999, since no recomputation occurs after {@code set()} itself.
196     * </p>
197     * 
198     * <p>
199     * <strong>{@code add(f, delta)}</strong> adds {@code delta} to
200     * field {@code f}. This is equivalent to calling <code>set(f,
201     * get(f) + delta)</code>
202     * with two adjustments:
203     * </p>
204     * 
205     * <blockquote>
206     * <p>
207     * <strong>Add rule 1</strong>. The value of field {@code f} after the
208     * call minus the value of field {@code f} before the call is
209     * {@code delta}, modulo any overflow that has occurred in field
210     * {@code f}. Overflow occurs when a field value exceeds its range and,
211     * as a result, the next larger field is incremented or decremented and the
212     * field value is adjusted back into its range.
213     * </p>
214     * 
215     * <p>
216     * <strong>Add rule 2</strong>. If a smaller field is expected to be invariant,
217     * but &nbsp; it is impossible for it to be equal to its prior value because of
218     * changes in its minimum or maximum after field {@code f} is changed,
219     * then its value is adjusted to be as close as possible to its expected value.
220     * A smaller field represents a smaller unit of time. {@code HOUR} is a
221     * smaller field than {@code DAY_OF_MONTH}. No adjustment is made to
222     * smaller fields that are not expected to be invariant. The calendar system
223     * determines what fields are expected to be invariant.
224     * </p>
225     * </blockquote>
226     * 
227     * <p>
228     * In addition, unlike {@code set()}, {@code add()} forces an
229     * immediate recomputation of the calendar's milliseconds and all fields.
230     * </p>
231     * 
232     * <p>
233     * <em>Example</em>: Consider a {@code GregorianCalendar} originally
234     * set to August 31, 1999. Calling {@code add(Calendar.MONTH, 13)} sets
235     * the calendar to September 30, 2000. <strong>Add rule 1</strong> sets the
236     * {@code MONTH} field to September, since adding 13 months to August
237     * gives September of the next year. Since {@code DAY_OF_MONTH} cannot be
238     * 31 in September in a {@code GregorianCalendar}, <strong>add rule 2</strong>
239     * sets the {@code DAY_OF_MONTH} to 30, the closest possible value.
240     * Although it is a smaller field, {@code DAY_OF_WEEK} is not adjusted by
241     * rule 2, since it is expected to change when the month changes in a
242     * {@code GregorianCalendar}.
243     * </p>
244     * 
245     * <p>
246     * <strong>{@code roll(f, delta)}</strong> adds {@code delta} to
247     * field {@code f} without changing larger fields. This is equivalent to
248     * calling {@code add(f, delta)} with the following adjustment:
249     * </p>
250     * 
251     * <blockquote>
252     * <p>
253     * <strong>Roll rule</strong>. Larger fields are unchanged after the call. A
254     * larger field represents a larger unit of time. {@code DAY_OF_MONTH} is
255     * a larger field than {@code HOUR}.
256     * </p>
257     * </blockquote>
258     * 
259     * <p>
260     * <em>Example</em>: Consider a {@code GregorianCalendar} originally
261     * set to August 31, 1999. Calling <code>roll(Calendar.MONTH,
262     * 8)</code> sets
263     * the calendar to April 30, <strong>1999</strong>. Add rule 1 sets the
264     * {@code MONTH} field to April. Using a {@code GregorianCalendar},
265     * the {@code DAY_OF_MONTH} cannot be 31 in the month April. Add rule 2
266     * sets it to the closest possible value, 30. Finally, the <strong>roll rule</strong>
267     * maintains the {@code YEAR} field value of 1999.
268     * </p>
269     * 
270     * <p>
271     * <em>Example</em>: Consider a {@code GregorianCalendar} originally
272     * set to Sunday June 6, 1999. Calling
273     * {@code roll(Calendar.WEEK_OF_MONTH, -1)} sets the calendar to Tuesday
274     * June 1, 1999, whereas calling {@code add(Calendar.WEEK_OF_MONTH, -1)}
275     * sets the calendar to Sunday May 30, 1999. This is because the roll rule
276     * imposes an additional constraint: The {@code MONTH} must not change
277     * when the {@code WEEK_OF_MONTH} is rolled. Taken together with add rule
278     * 1, the resultant date must be between Tuesday June 1 and Saturday June 5.
279     * According to add rule 2, the {@code DAY_OF_WEEK}, an invariant when
280     * changing the {@code WEEK_OF_MONTH}, is set to Tuesday, the closest
281     * possible value to Sunday (where Sunday is the first day of the week).
282     * </p>
283     * 
284     * <p>
285     * <strong>Usage model</strong>. To motivate the behavior of {@code add()}
286     * and {@code roll()}, consider a user interface component with
287     * increment and decrement buttons for the month, day, and year, and an
288     * underlying {@code GregorianCalendar}. If the interface reads January
289     * 31, 1999 and the user presses the month increment button, what should it
290     * read? If the underlying implementation uses {@code set()}, it might
291     * read March 3, 1999. A better result would be February 28, 1999. Furthermore,
292     * if the user presses the month increment button again, it should read March
293     * 31, 1999, not March 28, 1999. By saving the original date and using either
294     * {@code add()} or {@code roll()}, depending on whether larger
295     * fields should be affected, the user interface can behave as most users will
296     * intuitively expect.
297     * </p>
298     * 
299     * <p>
300     * <b>Note:</b> You should always use {@code roll} and {@code add} rather than
301     * attempting to perform arithmetic operations directly on the fields of a
302     * <tt>Calendar</tt>. It is quite possible for <tt>Calendar</tt> subclasses
303     * to have fields with non-linear behavior, for example missing months or days
304     * during non-leap years. The subclasses' <tt>add</tt> and <tt>roll</tt>
305     * methods will take this into account, while simple arithmetic manipulations
306     * may give invalid results.
307     * 
308     * @see Date
309     * @see GregorianCalendar
310     * @see TimeZone
311     * @since Android 1.0
312     */

654         * {@code getMinimalDaysInFirstWeek()}. {@code DAY_OF_MONTH 1}
655         * through {@code 7} always correspond to <code>DAY_OF_WEEK_IN_MONTH
656         * 1</code>;
657         * {@code 8} through {@code 15} correspond to
658         * {@code DAY_OF_WEEK_IN_MONTH 2}, and so on.
659         * {@code DAY_OF_WEEK_IN_MONTH 0} indicates the week before
660         * {@code DAY_OF_WEEK_IN_MONTH 1}. Negative values count back from
661         * the end of the month, so the last Sunday of a month is specified as
662         * {@code DAY_OF_WEEK = SUNDAY, DAY_OF_WEEK_IN_MONTH = -1}. Because
663         * negative values count backward they will usually be aligned differently
664         * within the month than positive values. For example, if a month has 31
665         * days, {@code DAY_OF_WEEK_IN_MONTH -1} will overlap
666         * {@code DAY_OF_WEEK_IN_MONTH 5} and the end of {@code 4}.
667         * 
668         * @see #DAY_OF_WEEK

998        /**
999         * Gets the maximum value of the specified field for the current date.
1000         * 
1001         * @param field
1002         *            the field.
1003         * @return the maximum value of the specified field.
1004         * @since Android 1.0
1005         */
1006        public int getActualMaximum(int field) {
1007            int value, next;
1008            if (getMaximum(field) == (next = getLeastMaximum(field))) {
1009                return next;
1010            }
1011            complete();
1012            long orgTime = time;
1013            set(field, next);
1014            do {
1015                value = next;
1016                roll(field, true);
1017                next = get(field);
1018            } while (next > value);
1019            time = orgTime;
1020            areFieldsSet = false;
1021            return value;
1022        }

1025         * Gets the minimum value of the specified field for the current date.
1026         * 
1027         * @param field
1028         *            the field.
1029         * @return the minimum value of the specified field.
1030         * @since Android 1.0
1031         */
1032        public int getActualMinimum(int field) {
1033            int value, next;
1034            if (getMinimum(field) == (next = getGreatestMinimum(field))) {
1035                return next;
1036            }
1037            complete();
1038            long orgTime = time;
1039            set(field, next);
1040            do {
1041                value = next;
1042                roll(field, false);
1043                next = get(field);
1044            } while (next < value);
1045            time = orgTime;
1046            areFieldsSet = false;
1047            return value;
1048        }

1506         *             if the argument is null.
1507         * @throws IllegalArgumentException
1508         *             if the argument does not include a valid time

1511         */
1512        public int compareTo(Calendar anotherCalendar) {
1513            if (null == anotherCalendar) {
1514                throw new NullPointerException();
1515            }
1516            long timeInMillis = getTimeInMillis();
1517            long anotherTimeInMillis = anotherCalendar.getTimeInMillis();
1518            if (timeInMillis > anotherTimeInMillis) {
1519                return 1;
