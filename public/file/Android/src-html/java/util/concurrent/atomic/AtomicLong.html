007    package java.util.concurrent.atomic;
008    import sun.misc.Unsafe;
009    
010    /**
011     * A <tt>long</tt> value that may be updated atomically.  See the
012     * {@link java.util.concurrent.atomic} package specification for
013     * description of the properties of atomic variables. An
014     * <tt>AtomicLong</tt> is used in applications such as atomically
015     * incremented sequence numbers, and cannot be used as a replacement
016     * for a {@link java.lang.Long}. However, this class does extend
017     * <tt>Number</tt> to allow uniform access by tools and utilities that
018     * deal with numerically-based classes.
019     *
020     * @since 1.5
021     * @author Doug Lea
022     */
023    public class AtomicLong extends Number implements java.io.Serializable { 
024        private static final long serialVersionUID = 1927816293512124184L;
025    
026        // setup to use Unsafe.compareAndSwapLong for updates
027        // BEGIN android-changed
028        private static final Unsafe unsafe = UnsafeAccess.THE_ONE;
029        // END android-changed
030        private static final long valueOffset;
031    
032        /**
033         * Record whether the underlying JVM supports lockless
034         * CompareAndSet for longs. While the unsafe.CompareAndSetLong
035         * method works in either case, some constructions should be
036         * handled at Java level to avoid locking user-visible locks.
037         */
038        static final boolean VM_SUPPORTS_LONG_CAS = VMSupportsCS8();
039    
040        /**
041         * Returns whether underlying JVM supports lockless CompareAndSet
042         * for longs. Called only once and cached in VM_SUPPORTS_LONG_CAS.
043         */
044        private static native boolean VMSupportsCS8();
045    
046        static {
047          try {
048            valueOffset = unsafe.objectFieldOffset
049                (AtomicLong.class.getDeclaredField("value"));
050          } catch(Exception ex) { throw new Error(ex); }
051        }
052    
053        private volatile long value;
054    
055        /**
056         * Create a new AtomicLong with the given initial value.
057         *
058         * @param initialValue the initial value
059         */
060        public AtomicLong(long initialValue) {
061            value = initialValue;
062        }
063    
064        /**
065         * Create a new AtomicLong with initial value <tt>0</tt>.
066         */
067        public AtomicLong() {
068        }
069      
070        /**
071         * Get the current value.
072         *
073         * @return the current value
074         */
075        public final long get() {
076            return value;
077        }
078     
079        /**
080         * Set to the given value.
081         *
082         * @param newValue the new value
083         */
084        public final void set(long newValue) {
085            value = newValue;
086        }
087      
088        /**
089         * Set to the give value and return the old value.
090         *
091         * @param newValue the new value
092         * @return the previous value
093         */
094        public final long getAndSet(long newValue) {
095            while (true) {
096                long current = get();
097                if (compareAndSet(current, newValue))
098                    return current;
099            }
100        }
101      
102        /**
103         * Atomically set the value to the given updated value
104         * if the current value <tt>==</tt> the expected value.
105         * @param expect the expected value
106         * @param update the new value
107         * @return true if successful. False return indicates that
108         * the actual value was not equal to the expected value.
109         */
110        public final boolean compareAndSet(long expect, long update) {
111          return unsafe.compareAndSwapLong(this, valueOffset, expect, update);
112        }
113    
114        /**
115         * Atomically set the value to the given updated value
116         * if the current value <tt>==</tt> the expected value.
117         * May fail spuriously.
118         * @param expect the expected value
119         * @param update the new value
120         * @return true if successful.
121         */
122        public final boolean weakCompareAndSet(long expect, long update) {
123          return unsafe.compareAndSwapLong(this, valueOffset, expect, update);
124        }
125      
126        /**
127         * Atomically increment by one the current value.
128         * @return the previous value
129         */
130        public final long getAndIncrement() {
131            while (true) {
132                long current = get();
133                long next = current + 1;
134                if (compareAndSet(current, next))
135                    return current;
136            }
137        }
138      
139      
140        /**
141         * Atomically decrement by one the current value.
142         * @return the previous value
143         */
144        public final long getAndDecrement() {
145            while (true) {
146                long current = get();
147                long next = current - 1;
148                if (compareAndSet(current, next))
149                    return current;
150            }
151        }
152      
153      
154        /**
155         * Atomically add the given value to current value.
156         * @param delta the value to add
157         * @return the previous value
158         */
159        public final long getAndAdd(long delta) {
160            while (true) {
161                long current = get();
162                long next = current + delta;
163                if (compareAndSet(current, next))
164                    return current;
165            }
166        }
167      
168        /**
169         * Atomically increment by one the current value.
170         * @return the updated value
171         */
172        public final long incrementAndGet() {
173            for (;;) {
174                long current = get();
175                long next = current + 1;
176                if (compareAndSet(current, next))
177                    return next;
178            }
179        }
180        
181        /**
182         * Atomically decrement by one the current value.
183         * @return the updated value
184         */
185        public final long decrementAndGet() {
186            for (;;) {
187                long current = get();
188                long next = current - 1;
189                if (compareAndSet(current, next))
190                    return next;
191            }
192        }
193      
194      
195        /**
196         * Atomically add the given value to current value.
197         * @param delta the value to add
198         * @return the updated value
199         */
200        public final long addAndGet(long delta) {
201            for (;;) {
202                long current = get();
203                long next = current + delta;
204                if (compareAndSet(current, next))
205                    return next;
206            }
207        }
208      
209        /**
210         * Returns the String representation of the current value.
211         * @return the String representation of the current value.
212         */
213        public String toString() {
214            return Long.toString(get());
215        }
216    
217    
218        public int intValue() {
219            return (int)get();
220        }
221    
222        public long longValue() {
223            return (long)get();
224        }
225    
226        public float floatValue() {
227            return (float)get();
228        }
229    
230        public double doubleValue() {
231            return (double)get();
232        }
233      
234    }
