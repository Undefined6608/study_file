007    package java.util.concurrent.atomic;
008    import sun.misc.Unsafe;
009    import java.util.*;
010    
011    /**
012     * A <tt>long</tt> array in which elements may be updated atomically.
013     * See the {@link java.util.concurrent.atomic} package specification
014     * for description of the properties of atomic variables.
015     * @since 1.5
016     * @author Doug Lea
017     */
018    public class AtomicLongArray implements java.io.Serializable { 
019        private static final long serialVersionUID = -2308431214976778248L;
020    
021        // setup to use Unsafe.compareAndSwapInt for updates
022        // BEGIN android-changed
023        private static final Unsafe unsafe = UnsafeAccess.THE_ONE;
024        // END android-changed
025        private static final int base = unsafe.arrayBaseOffset(long[].class);
026        private static final int scale = unsafe.arrayIndexScale(long[].class);
027        private final long[] array;
028    
029        private long rawIndex(int i) {
030            if (i < 0 || i >= array.length)
031                throw new IndexOutOfBoundsException("index " + i);
032            return base + i * scale;
033        }
034    
035        /**
036         * Create a new AtomicLongArray of given length.
037         * @param length the length of the array
038         */
039        public AtomicLongArray(int length) {
040            array = new long[length];
041            // must perform at least one volatile write to conform to JMM
042            if (length > 0) 
043                unsafe.putLongVolatile(array, rawIndex(0), 0);
044        }
045    
046        /**
047         * Create a new AtomicLongArray with the same length as, and
048         * all elements copied from, the given array.
049         *
050         * @param array the array to copy elements from
051         * @throws NullPointerException if array is null
052         */
053        public AtomicLongArray(long[] array) {
054            if (array == null) 
055                throw new NullPointerException();
056            int length = array.length;
057            this.array = new long[length];
058            if (length > 0) {
059                int last = length-1;
060                for (int i = 0; i < last; ++i)
061                    this.array[i] = array[i];
062                // Do the last write as volatile
063                unsafe.putLongVolatile(this.array, rawIndex(last), array[last]);
064            }
065        }
066    
067        /**
068         * Returns the length of the array.
069         *
070         * @return the length of the array
071         */
072        public final int length() {
073            return array.length;
074        }
075    
076        /**
077         * Get the current value at position <tt>i</tt>.
078         *
079         * @param i the index
080         * @return the current value
081         */
082        public final long get(int i) {
083            return unsafe.getLongVolatile(array, rawIndex(i));
084        }
085     
086        /**
087         * Set the element at position <tt>i</tt> to the given value.
088         *
089         * @param i the index
090         * @param newValue the new value
091         */
092        public final void set(int i, long newValue) {
093            unsafe.putLongVolatile(array, rawIndex(i), newValue);
094        }
095      
096        /**
097         * Set the element at position <tt>i</tt> to the given value and return the
098         * old value.
099         *
100         * @param i the index
101         * @param newValue the new value
102         * @return the previous value
103         */
104        public final long getAndSet(int i, long newValue) {
105            while (true) {
106                long current = get(i);
107                if (compareAndSet(i, current, newValue))
108                    return current;
109            }
110        }
111      
112        /**
113         * Atomically set the value to the given updated value
114         * if the current value <tt>==</tt> the expected value.
115         * @param i the index
116         * @param expect the expected value
117         * @param update the new value
118         * @return true if successful. False return indicates that
119         * the actual value was not equal to the expected value.
120         */
121        public final boolean compareAndSet(int i, long expect, long update) {
122            return unsafe.compareAndSwapLong(array, rawIndex(i), 
123                                             expect, update);
124        }
125    
126        /**
127         * Atomically set the value to the given updated value
128         * if the current value <tt>==</tt> the expected value.
129         * May fail spuriously.
130         * @param i the index
131         * @param expect the expected value
132         * @param update the new value
133         * @return true if successful.
134         */
135        public final boolean weakCompareAndSet(int i, long expect, long update) {
136            return compareAndSet(i, expect, update);
137        }
138    
139        /**
140         * Atomically increment by one the element at index <tt>i</tt>.
141         *
142         * @param i the index
143         * @return the previous value;
144         */
145        public final long getAndIncrement(int i) {
146            while (true) {
147                long current = get(i);
148                long next = current + 1;
149                if (compareAndSet(i, current, next))
150                    return current;
151            }
152        }
153      
154        /**
155         * Atomically decrement by one the element at index <tt>i</tt>.
156         *
157         * @param i the index
158         * @return the previous value;
159         */
160        public final long getAndDecrement(int i) {
161            while (true) {
162                long current = get(i);
163                long next = current - 1;
164                if (compareAndSet(i, current, next))
165                    return current;
166            }
167        }
168      
169        /**
170         * Atomically add the given value to element at index <tt>i</tt>.
171         *
172         * @param i the index
173         * @param delta the value to add
174         * @return the previous value;
175         */
176        public final long getAndAdd(int i, long delta) {
177            while (true) {
178                long current = get(i);
179                long next = current + delta;
180                if (compareAndSet(i, current, next))
181                    return current;
182            }
183        }
184      
185    
186        /**
187         * Atomically increment the element at index <tt>i</tt>.
188         *
189         * @param i the index
190         * @return the updated value;
191         */
192        public final long incrementAndGet(int i) {
193            while (true) {
194                long current = get(i);
195                long next = current + 1;
196                if (compareAndSet(i, current, next))
197                    return next;
198            }
199        }
200      
201        /**
202         * Atomically decrement the element at index <tt>i</tt>.
203         *
204         * @param i the index
205         * @return the updated value;
206         */
207        public final long decrementAndGet(int i) {
208            while (true) {
209                long current = get(i);
210                long next = current - 1;
211                if (compareAndSet(i, current, next))
212                    return next;
213            }
214        }
215      
216        /**
217         * Atomically add the given value to element at index <tt>i</tt>.
218         *
219         * @param i the index
220         * @param delta the value to add
221         * @return the updated value;
222         */
223        public long addAndGet(int i, long delta) {
224            while (true) {
225                long current = get(i);
226                long next = current + delta;
227                if (compareAndSet(i, current, next))
228                    return next;
229            }
230        }
231    
232        /**
233         * Returns the String representation of the current values of array.
234         * @return the String representation of the current values of array.
235         */
236        public String toString() {
237            if (array.length > 0) // force volatile read
238                get(0);
239            return Arrays.toString(array);
240        }
241      
242    }
