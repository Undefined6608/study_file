007    package java.util.concurrent.atomic;
008    import sun.misc.Unsafe;
009    
010    /**
011     * An object reference that may be updated atomically. See the {@link
012     * java.util.concurrent.atomic} package specification for description
013     * of the properties of atomic variables.
014     * @since 1.5
015     * @author Doug Lea
016     * @param <V> The type of object referred to by this reference
017     */
018    public class AtomicReference<V>  implements java.io.Serializable { 
019        private static final long serialVersionUID = -1848883965231344442L;
020    
021        // BEGIN android-changed
022        private static final Unsafe unsafe = UnsafeAccess.THE_ONE;
023        // END android-changed
024        private static final long valueOffset;
025    
026        static {
027          try {
028            valueOffset = unsafe.objectFieldOffset
029                (AtomicReference.class.getDeclaredField("value"));
030          } catch(Exception ex) { throw new Error(ex); }
031        }
032    
033        private volatile V value;
034    
035        /**
036         * Create a new AtomicReference with the given initial value.
037         *
038         * @param initialValue the initial value
039         */
040        public AtomicReference(V initialValue) {
041            value = initialValue;
042        }
043    
044        /**
045         * Create a new AtomicReference with null initial value.
046         */
047        public AtomicReference() {
048        }
049      
050        /**
051         * Get the current value.
052         *
053         * @return the current value
054         */
055        public final V get() {
056            return value;
057        }
058      
059        /**
060         * Set to the given value.
061         *
062         * @param newValue the new value
063         */
064        public final void set(V newValue) {
065            value = newValue;
066        }
067      
068        /**
069         * Atomically set the value to the given updated value
070         * if the current value <tt>==</tt> the expected value.
071         * @param expect the expected value
072         * @param update the new value
073         * @return true if successful. False return indicates that
074         * the actual value was not equal to the expected value.
075         */
076        public final boolean compareAndSet(V expect, V update) {
077          return unsafe.compareAndSwapObject(this, valueOffset, expect, update);
078        }
079    
080        /**
081         * Atomically set the value to the given updated value
082         * if the current value <tt>==</tt> the expected value.
083         * May fail spuriously.
084         * @param expect the expected value
085         * @param update the new value
086         * @return true if successful.
087         */
088        public final boolean weakCompareAndSet(V expect, V update) {
089          return unsafe.compareAndSwapObject(this, valueOffset, expect, update);
090        }
091    
092        /**
093         * Set to the given value and return the old value.
094         *
095         * @param newValue the new value
096         * @return the previous value
097         */
098        public final V getAndSet(V newValue) {
099            while (true) {
100                V x = get();
101                if (compareAndSet(x, newValue))
102                    return x;
103            }
104        }
105    
106        /**
107         * Returns the String representation of the current value.
108         * @return the String representation of the current value.
109         */
110        public String toString() {
111            return String.valueOf(get());
112        }
113    
114    }
