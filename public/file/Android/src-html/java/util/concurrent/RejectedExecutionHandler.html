007    package java.util.concurrent;
008    
009    /**
010     * A handler for tasks that cannot be executed by a {@link
011     * ThreadPoolExecutor}.
012     *
013     * @since 1.5
014     * @author Doug Lea
015     */
016    public interface RejectedExecutionHandler {
017    
018        /**
019         * Method that may be invoked by a {@link ThreadPoolExecutor} when
020         * <tt>execute</tt> cannot accept a task. This may occur when no
021         * more threads or queue slots are available because their bounds
022         * would be exceeded, or upon shutdown of the Executor.
023         *
024         * In the absence other alternatives, the method may throw an
025         * unchecked {@link RejectedExecutionException}, which will be
026         * propagated to the caller of <tt>execute</tt>.
027         *
028         * @param r the runnable task requested to be executed
029         * @param executor the executor attempting to execute this task
030         * @throws RejectedExecutionException if there is no remedy
031         */
032        void rejectedExecution(Runnable r, ThreadPoolExecutor executor);
033    }
