008    import java.util.*;
009    import java.util.concurrent.*;

012    /**
013     * A reentrant mutual exclusion {@link Lock} with the same basic
014     * behavior and semantics as the implicit monitor lock accessed using
015     * <tt>synchronized</tt> methods and statements, but with extended
016     * capabilities.
017     *
018     * <p> A <tt>ReentrantLock</tt> is <em>owned</em> by the thread last
019     * successfully locking, but not yet unlocking it. A thread invoking
020     * <tt>lock</tt> will return, successfully acquiring the lock, when
021     * the lock is not owned by another thread. The method will return
022     * immediately if the current thread already owns the lock. This can
023     * be checked using methods {@link #isHeldByCurrentThread}, and {@link
024     * #getHoldCount}.  
025     *
026     * <p> The constructor for this class accepts an optional
027     * <em>fairness</em> parameter.  When set <tt>true</tt>, under
028     * contention, locks favor granting access to the longest-waiting
029     * thread.  Otherwise this lock does not guarantee any particular
030     * access order.  Programs using fair locks accessed by many threads
031     * may display lower overall throughput (i.e., are slower; often much
032     * slower) than those using the default setting, but have smaller
033     * variances in times to obtain locks and guarantee lack of
034     * starvation. Note however, that fairness of locks does not guarantee
035     * fairness of thread scheduling. Thus, one of many threads using a
036     * fair lock may obtain it multiple times in succession while other
037     * active threads are not progressing and not currently holding the
038     * lock.
039     *
040     * <p> It is recommended practice to <em>always</em> immediately
041     * follow a call to <tt>lock</tt> with a <tt>try</tt> block, most
042     * typically in a before/after construction such as:
043     *
044     * <pre>
045     * class X {
046     *   private final ReentrantLock lock = new ReentrantLock();
047     *   // ...
048     *
049     *   public void m() { 
050     *     lock.lock();  // block until condition holds
051     *     try {
052     *       // ... method body
053     *     } finally {
054     *       lock.unlock()
055     *     }
056     *   }
057     * }
058     * </pre>
059     *
060     * <p>In addition to implementing the {@link Lock} interface, this
061     * class defines methods <tt>isLocked</tt> and
062     * <tt>getLockQueueLength</tt>, as well as some associated
063     * <tt>protected</tt> access methods that may be useful for
064     * instrumentation and monitoring.
065     *
066     * <p> Serialization of this class behaves in the same way as built-in
067     * locks: a deserialized lock is in the unlocked state, regardless of
068     * its state when serialized.
069     *
070     * <p> This lock supports a maximum of 2147483648 recursive locks by
071     * the same thread. 
072     *
073     * @since 1.5
074     * @author Doug Lea
075     * 
076     */
077    public class ReentrantLock implements Lock, java.io.Serializable {
078        private static final long serialVersionUID = 7373984872572414699L;
079        /** Synchronizer providing all implementation mechanics */
080        private final Sync sync;
081    
082        /**
083         * Base of synchronization control for this lock. Subclassed
084         * into fair and nonfair versions below. Uses AQS state to
085         * represent the number of holds on the lock.
086         */
087        static abstract class Sync  extends AbstractQueuedSynchronizer {
088            /** Current owner thread */
089            transient Thread owner;
090    
091            /**
092             * Perform {@link Lock#lock}. The main reason for subclassing
093             * is to allow fast path for nonfair version.
094             */
095            abstract void lock();
096    
097            /** 
098             * Perform non-fair tryLock.  tryAcquire is
099             * implemented in subclasses, but both need nonfair
100             * try for trylock method
101             */
102            final boolean nonfairTryAcquire(int acquires) { 
103                final Thread current = Thread.currentThread();
104                int c = getState();
105                if (c == 0) {
106                    if (compareAndSetState(0, acquires)) {
107                        owner = current;
108                        return true;
109                    }
110                }
111                else if (current == owner) {
112                    setState(c+acquires);
113                    return true;
114                }
115                return false;
116            }
117    
118            protected final boolean tryRelease(int releases) {
119                int c = getState() - releases;
120                if (Thread.currentThread() != owner)
121                    throw new IllegalMonitorStateException();
122                boolean free = false;
123                if (c == 0) {
124                    free = true;
125                    owner = null;
126                }
127                setState(c);
128                return free;
129            }
130    
131            protected final boolean isHeldExclusively() {
132                return getState() != 0 && owner == Thread.currentThread();
133            }
134    
135            final ConditionObject newCondition() {
136                return new ConditionObject();
137            }
138    
139            // Methods relayed from outer class
140    
141            final Thread getOwner() {
142                int c = getState();
143                Thread o = owner;
144                return (c == 0)? null : o;
145            }
146            
147            final int getHoldCount() {
148                int c = getState();
149                Thread o = owner;
150                return (o == Thread.currentThread())? c : 0;
151            }
152            
153            final boolean isLocked() {
154                return getState() != 0;
155            }
156    
157            /**
158             * Reconstitute this lock instance from a stream
159             * @param s the stream
160             */
161            private void readObject(java.io.ObjectInputStream s)
162                throws java.io.IOException, ClassNotFoundException {
163                s.defaultReadObject();
164                setState(0); // reset to unlocked state
165            }
166        }

168        /**
169         * Sync object for non-fair locks
170         */
171        final static class NonfairSync extends Sync {
172            /**
173             * Perform lock.  Try immediate barge, backing up to normal
174             * acquire on failure.
175             */
176            final void lock() {
177                if (compareAndSetState(0, 1))
178                    owner = Thread.currentThread();
179                else
180                    acquire(1);
181            }
182    
183            protected final boolean tryAcquire(int acquires) { 
184                return nonfairTryAcquire(acquires);
185            }
186        }
187    
188        /**
189         * Sync object for fair locks
190         */
191        final static class FairSync  extends Sync {
192            final void lock() { 
193                acquire(1); 
194            }
195    
196            /**
197             * Fair version of tryAcquire.  Don't grant access unless
198             * recursive call or no waiters or is first.
199             */
200            protected final boolean tryAcquire(int acquires) { 
201                final Thread current = Thread.currentThread();
202                int c = getState();
203                if (c == 0) {
204                    Thread first = getFirstQueuedThread();
205                    if ((first == null || first == current) && 
206                        compareAndSetState(0, acquires)) {
207                        owner = current;
208                        return true;
209                    }
210                }
211                else if (current == owner) {
212                    setState(c+acquires);
213                    return true;
214                }
215                return false;
216            }
217        }

236        /**
237         * Acquires the lock. 
238         *
239         * <p>Acquires the lock if it is not held by another thread and returns 
240         * immediately, setting the lock hold count to one.
241         *
242         * <p>If the current thread
243         * already holds the lock then the hold count is incremented by one and
244         * the method returns immediately.
245         *
246         * <p>If the lock is held by another thread then the
247         * current thread becomes disabled for thread scheduling 
248         * purposes and lies dormant until the lock has been acquired,
249         * at which time the lock hold count is set to one. 
250         */
251        public void lock() {
252            sync.lock();
253        }

255        /**
256         * Acquires the lock unless the current thread is 
257         * {@link Thread#interrupt interrupted}.
258         *
259         * <p>Acquires the lock if it is not held by another thread and returns 
260         * immediately, setting the lock hold count to one.
261         *
262         * <p>If the current thread already holds this lock then the hold count 
263         * is incremented by one and the method returns immediately.
264         *
265         * <p>If the lock is held by another thread then the
266         * current thread becomes disabled for thread scheduling 
267         * purposes and lies dormant until one of two things happens:
268         *
269         * <ul>
270         *
271         * <li>The lock is acquired by the current thread; or
272         *
273         * <li>Some other thread {@link Thread#interrupt interrupts} the current
274         * thread.
275         *
276         * </ul>
277         *
278         * <p>If the lock is acquired by the current thread then the lock hold 
279         * count is set to one.
280         *
281         * <p>If the current thread:
282         *
283         * <ul>
284         *
285         * <li>has its interrupted status set on entry to this method; or 
286         *
287         * <li>is {@link Thread#interrupt interrupted} while acquiring 
288         * the lock,
289         *
290         * </ul>
291         *
292         * then {@link InterruptedException} is thrown and the current thread's 
293         * interrupted status is cleared. 
294         *
295         * <p>In this implementation, as this method is an explicit interruption 
296         * point, preference is 
297         * given to responding to the interrupt over normal or reentrant 
298         * acquisition of the lock.
299         *
300         * @throws InterruptedException if the current thread is interrupted
301         */
302        public void lockInterruptibly() throws InterruptedException { 
303            sync.acquireInterruptibly(1);
304        }

306        /**
307         * Acquires the lock only if it is not held by another thread at the time
308         * of invocation.
309         *
310         * <p>Acquires the lock if it is not held by another thread and
311         * returns immediately with the value <tt>true</tt>, setting the
312         * lock hold count to one. Even when this lock has been set to use a
313         * fair ordering policy, a call to <tt>tryLock()</tt> <em>will</em>
314         * immediately acquire the lock if it is available, whether or not
315         * other threads are currently waiting for the lock. 
316         * This &quot;barging&quot; behavior can be useful in certain 
317         * circumstances, even though it breaks fairness. If you want to honor
318         * the fairness setting for this lock, then use 
319         * {@link #tryLock(long, TimeUnit) tryLock(0, TimeUnit.SECONDS) }
320         * which is almost equivalent (it also detects interruption).
321         *
322         * <p> If the current thread
323         * already holds this lock then the hold count is incremented by one and
324         * the method returns <tt>true</tt>.
325         *
326         * <p>If the lock is held by another thread then this method will return 
327         * immediately with the value <tt>false</tt>.  
328         *
329         * @return <tt>true</tt> if the lock was free and was acquired by the
330         * current thread, or the lock was already held by the current thread; and
331         * <tt>false</tt> otherwise.
332         */
333        public boolean tryLock() {
334            return sync.nonfairTryAcquire(1);
335        }
336    
337        /**
338         * Acquires the lock if it is not held by another thread within the given 
339         * waiting time and the current thread has not been 
340         * {@link Thread#interrupt interrupted}.
341         *
342         * <p>Acquires the lock if it is not held by another thread and returns 
343         * immediately with the value <tt>true</tt>, setting the lock hold count 
344         * to one. If this lock has been set to use a fair ordering policy then
345         * an available lock <em>will not</em> be acquired if any other threads
346         * are waiting for the lock. This is in contrast to the {@link #tryLock()}
347         * method. If you want a timed <tt>tryLock</tt> that does permit barging on
348         * a fair lock then combine the timed and un-timed forms together:
349         *
350         * <pre>if (lock.tryLock() || lock.tryLock(timeout, unit) ) { ... }
351         * </pre>
352         *
353         * <p>If the current thread
354         * already holds this lock then the hold count is incremented by one and
355         * the method returns <tt>true</tt>.
356         *
357         * <p>If the lock is held by another thread then the
358         * current thread becomes disabled for thread scheduling 
359         * purposes and lies dormant until one of three things happens:
360         *
361         * <ul>
362         *
363         * <li>The lock is acquired by the current thread; or
364         *
365         * <li>Some other thread {@link Thread#interrupt interrupts} the current
366         * thread; or
367         *
368         * <li>The specified waiting time elapses
369         *
370         * </ul>
371         *
372         * <p>If the lock is acquired then the value <tt>true</tt> is returned and
373         * the lock hold count is set to one.
374         *
375         * <p>If the current thread:
376         *
377         * <ul>
378         *
379         * <li>has its interrupted status set on entry to this method; or 
380         *
381         * <li>is {@link Thread#interrupt interrupted} while acquiring
382         * the lock,
383         *
384         * </ul>
385         * then {@link InterruptedException} is thrown and the current thread's 
386         * interrupted status is cleared. 
387         *
388         * <p>If the specified waiting time elapses then the value <tt>false</tt>
389         * is returned.
390         * If the time is 
391         * less than or equal to zero, the method will not wait at all.
392         *
393         * <p>In this implementation, as this method is an explicit interruption 
394         * point, preference is 
395         * given to responding to the interrupt over normal or reentrant 
396         * acquisition of the lock, and over reporting the elapse of the waiting
397         * time.
398         *
399         * @param timeout the time to wait for the lock
400         * @param unit the time unit of the timeout argument
401         *
402         * @return <tt>true</tt> if the lock was free and was acquired by the
403         * current thread, or the lock was already held by the current thread; and
404         * <tt>false</tt> if the waiting time elapsed before the lock could be 
405         * acquired.
406         *
407         * @throws InterruptedException if the current thread is interrupted
408         * @throws NullPointerException if unit is null
409         *
410         */
411        public boolean tryLock(long timeout, TimeUnit unit) throws InterruptedException {
412            return sync.tryAcquireNanos(1, unit.toNanos(timeout));
413        }

415        /**
416         * Attempts to release this lock.  
417         *
418         * <p>If the current thread is the
419         * holder of this lock then the hold count is decremented. If the
420         * hold count is now zero then the lock is released.  If the
421         * current thread is not the holder of this lock then {@link
422         * IllegalMonitorStateException} is thrown.
423         * @throws IllegalMonitorStateException if the current thread does not
424         * hold this lock.
425         */
426        public void unlock() {
427            sync.release(1);
428        }

430        /**
431         * Returns a {@link Condition} instance for use with this 
432         * {@link Lock} instance.
433         *
434         * <p>The returned {@link Condition} instance supports the same
435         * usages as do the {@link Object} monitor methods ({@link
436         * Object#wait() wait}, {@link Object#notify notify}, and {@link
437         * Object#notifyAll notifyAll}) when used with the built-in
438         * monitor lock.
439         *
440         * <ul>
441         *
442         * <li>If this lock is not held when any of the {@link Condition}
443         * {@link Condition#await() waiting} or {@link Condition#signal
444         * signalling} methods are called, then an {@link
445         * IllegalMonitorStateException} is thrown.
446         *
447         * <li>When the condition {@link Condition#await() waiting}
448         * methods are called the lock is released and, before they
449         * return, the lock is reacquired and the lock hold count restored
450         * to what it was when the method was called.
451         *
452         * <li>If a thread is {@link Thread#interrupt interrupted} while
453         * waiting then the wait will terminate, an {@link
454         * InterruptedException} will be thrown, and the thread's
455         * interrupted status will be cleared.
456         *
457         * <li> Waiting threads are signalled in FIFO order
458         *
459         * <li>The ordering of lock reacquisition for threads returning
460         * from waiting methods is the same as for threads initially
461         * acquiring the lock, which is in the default case not specified,
462         * but for <em>fair</em> locks favors those threads that have been
463         * waiting the longest.
464         * 
465         * </ul>
466         *
467         * @return the Condition object
468         */
469        public Condition newCondition() {
470            return sync.newCondition();
471        }

495         *     }
496         *   }

717         * &quot;Unlocked&quot; or the String &quot;Locked by&quot;
718         * followed by the {@link Thread#getName} of the owning thread.
719         * @return a string identifying this lock, as well as its lock state.
720         */
721        public String toString() {
722            Thread owner = sync.getOwner();
723            return super.toString() + ((owner == null) ?
724                                       "[Unlocked]" :

726        }
