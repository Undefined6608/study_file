007    package java.util.concurrent.locks;
008    import java.util.concurrent.*;
009    import sun.misc.Unsafe;
010    
011    
012    /**
013     * Basic thread blocking primitives for creating locks and other
014     * synchronization classes.
015     *
016     * <p>This class associates with each thread that uses it, a permit
017     * (in the sense of the {@link java.util.concurrent.Semaphore
018     * Semaphore} class). A call to <tt>park</tt> will return immediately
019     * if the permit is available, consuming it in the process; otherwise
020     * it <em>may</em> block.  A call to <tt>unpark</tt> makes the permit
021     * available, if it was not already available. (Unlike with Semaphores
022     * though, permits do not accumulate. There is at most one.)
023     *
024     * <p>Methods <tt>park</tt> and <tt>unpark</tt> provide efficient
025     * means of blocking and unblocking threads that do not encounter the
026     * problems that cause the deprecated methods <tt>Thread.suspend</tt>
027     * and <tt>Thread.resume</tt> to be unusable for such purposes: Races
028     * between one thread invoking <tt>park</tt> and another thread trying
029     * to <tt>unpark</tt> it will preserve liveness, due to the
030     * permit. Additionally, <tt>park</tt> will return if the caller's
031     * thread was interrupted, and timeout versions are supported. The
032     * <tt>park</tt> method may also return at any other time, for "no
033     * reason", so in general must be invoked within a loop that rechecks
034     * conditions upon return. In this sense <tt>park</tt> serves as an
035     * optimization of a "busy wait" that does not waste as much time
036     * spinning, but must be paired with an <tt>unpark</tt> to be
037     * effective.
038     *
039     * <p>These methods are designed to be used as tools for creating
040     * higher-level synchronization utilities, and are not in themselves
041     * useful for most concurrency control applications.
042     *
043     * <p><b>Sample Usage.</b> Here is a sketch of a First-in-first-out
044     * non-reentrant lock class.
045     * <pre>
046     *   private AtomicBoolean locked = new AtomicBoolean(false);
047     *   private Queue&lt;Thread&gt; waiters = new ConcurrentLinkedQueue&lt;Thread&gt;();
048     *
049     *   public void lock() { 
050     *     boolean wasInterrupted = false;
051     *     Thread current = Thread.currentThread();
052     *     waiters.add(current);
053     *
054     *     // Block while not first in queue or cannot acquire lock
055     *     while (waiters.peek() != current || 
056     *            !locked.compareAndSet(false, true)) { 
057     *        LockSupport.park();
058     *        if (Thread.interrupted()) // ignore interrupts while waiting
059     *          wasInterrupted = true;
060     *     }
061     *
062     *     waiters.remove();
063     *     if (wasInterrupted)          // reassert interrupt status on exit
064     *        current.interrupt();
065     *   }
066     *
067     *   public void unlock() {
068     *     locked.set(false);
069     *     LockSupport.unpark(waiters.peek());
070     *   } 
071     * }
072     * </pre>
073     */
074    @SuppressWarnings("all")
075    public class LockSupport {
076        private LockSupport() {} // Cannot be instantiated.
077    
078        // BEGIN android-changed
079        private static final Unsafe unsafe = UnsafeAccess.THE_ONE;
080        // END android-changed
081    
082        /**
083         * Make available the permit for the given thread, if it
084         * was not already available.  If the thread was blocked on
085         * <tt>park</tt> then it will unblock.  Otherwise, its next call
086         * to <tt>park</tt> is guaranteed not to block. This operation
087         * is not guaranteed to have any effect at all if the given
088         * thread has not been started.
089         * @param thread the thread to unpark, or <tt>null</tt>, in which case
090         * this operation has no effect. 
091         */
092        public static void unpark(Thread thread) {
093            if (thread != null)
094                unsafe.unpark(thread);
095        }
096    
097        /**
098         * Disables the current thread for thread scheduling purposes unless the
099         * permit is available.
100         * <p>If the permit is available then it is consumed and the call returns
101         * immediately; otherwise 
102         * the current thread becomes disabled for thread scheduling 
103         * purposes and lies dormant until one of three things happens:
104         * <ul>
105         * <li>Some other thread invokes <tt>unpark</tt> with the current thread 
106         * as the target; or
107         * <li>Some other thread {@link Thread#interrupt interrupts} the current
108         * thread; or
109         * <li>The call spuriously (that is, for no reason) returns.
110         * </ul>
111         * <p>This method does <em>not</em> report which of these caused the 
112         * method to return. Callers should re-check the conditions which caused 
113         * the thread to park in the first place. Callers may also determine, 
114         * for example, the interrupt status of the thread upon return.
115         */
116        public static void park() {
117            unsafe.park(false, 0L);
118        }
119    
120        /**
121         * Disables the current thread for thread scheduling purposes, for up to
122         * the specified waiting time, unless the permit is available.
123         * <p>If the permit is available then it is consumed and the call returns
124         * immediately; otherwise 
125         * the current thread becomes disabled for thread scheduling 
126         * purposes and lies dormant until one of four things happens:
127         * <ul>
128         * <li>Some other thread invokes <tt>unpark</tt> with the current thread 
129         * as the target; or
130         * <li>Some other thread {@link Thread#interrupt interrupts} the current
131         * thread; or
132         * <li>The specified waiting time elapses; or
133         * <li>The call spuriously (that is, for no reason) returns.
134         * </ul>
135         * <p>This method does <em>not</em> report which of these caused the 
136         * method to return. Callers should re-check the conditions which caused 
137         * the thread to park in the first place. Callers may also determine, 
138         * for example, the interrupt status of the thread, or the elapsed time
139         * upon return.
140         *
141         * @param nanos the maximum number of nanoseconds to wait
142         */
143        public static void parkNanos(long nanos) {
144            if (nanos > 0)
145                unsafe.park(false, nanos);   
146        }
147    
148        /**
149         * Disables the current thread for thread scheduling purposes, until
150         * the specified deadline, unless the permit is available.
151         * <p>If the permit is available then it is consumed and the call returns
152         * immediately; otherwise 
153         * the current thread becomes disabled for thread scheduling 
154         * purposes and lies dormant until one of four things happens:
155         * <ul>
156         * <li>Some other thread invokes <tt>unpark</tt> with the current thread 
157         * as the target; or
158         * <li>Some other thread {@link Thread#interrupt interrupts} the current
159         * thread; or
160         * <li>The specified deadline passes; or
161         * <li>The call spuriously (that is, for no reason) returns.
162         * </ul>
163         * <p>This method does <em>not</em> report which of these caused the 
164         * method to return. Callers should re-check the conditions which caused 
165         * the thread to park in the first place. Callers may also determine, 
166         * for example, the interrupt status of the thread, or the current time
167         * upon return.
168         *
169         * @param deadline the absolute time, in milliseconds from the Epoch, to
170         * wait until
171         */
172        public static void parkUntil(long deadline) {
173            unsafe.park(true, deadline);   
174        }
175    
176    }
