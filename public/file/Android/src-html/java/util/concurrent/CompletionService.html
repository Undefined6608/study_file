007    package java.util.concurrent;
008    
009    /**
010     * A service that decouples the production of new asynchronous tasks
011     * from the consumption of the results of completed tasks.  Producers
012     * <tt>submit</tt> tasks for execution. Consumers <tt>take</tt>
013     * completed tasks and process their results in the order they
014     * complete.  A <tt>CompletionService</tt> can for example be used to
015     * manage asynchronous IO, in which tasks that perform reads are
016     * submitted in one part of a program or system, and then acted upon
017     * in a different part of the program when the reads complete,
018     * possibly in a different order than they were requested.
019    
020     * <p>
021     *
022     * Typically, a <tt>CompletionService</tt> relies on a separate {@link
023     * Executor} to actually execute the tasks, in which case the
024     * <tt>CompletionService</tt> only manages an internal completion
025     * queue. The {@link ExecutorCompletionService} class provides an
026     * implementation of this approach.
027     *
028     */
029    public interface CompletionService<V> {
030        /**
031         * Submits a value-returning task for execution and returns a Future
032         * representing the pending results of the task. Upon completion,
033         * this task may be taken or polled.
034         *
035         * @param task the task to submit
036         * @return a Future representing pending completion of the task
037         * @throws RejectedExecutionException if task cannot be scheduled
038         * for execution
039         * @throws NullPointerException if task null     
040         */
041        Future<V> submit(Callable<V> task);
042    
043    
044        /**
045         * Submits a Runnable task for execution and returns a Future 
046         * representing that task. Upon completion,
047         * this task may be taken or polled.
048         *
049         * @param task the task to submit
050         * @param result the result to return upon successful completion
051         * @return a Future representing pending completion of the task,
052         * and whose <tt>get()</tt> method will return the given result value 
053         * upon completion
054         * @throws RejectedExecutionException if task cannot be scheduled
055         * for execution
056         * @throws NullPointerException if task null     
057         */
058        Future<V> submit(Runnable task, V result);
059    
060        /**
061         * Retrieves and removes the Future representing the next
062         * completed task, waiting if none are yet present.
063         * @return the Future representing the next completed task
064         * @throws InterruptedException if interrupted while waiting.
065         */
066        Future<V> take() throws InterruptedException;
067    
068    
069        /**
070         * Retrieves and removes the Future representing the next
071         * completed task or <tt>null</tt> if none are present.
072         *
073         * @return the Future representing the next completed task, or
074         * <tt>null</tt> if none are present.
075         */
076        Future<V> poll();
077    
078        /**
079         * Retrieves and removes the Future representing the next
080         * completed task, waiting if necessary up to the specified wait
081         * time if none are yet present.
082         * @param timeout how long to wait before giving up, in units of
083         * <tt>unit</tt>
084         * @param unit a <tt>TimeUnit</tt> determining how to interpret the
085         * <tt>timeout</tt> parameter
086         * @return the Future representing the next completed task or
087         * <tt>null</tt> if the specified waiting time elapses before one
088         * is present.
089         * @throws InterruptedException if interrupted while waiting.
090         */
091        Future<V> poll(long timeout, TimeUnit unit) throws InterruptedException;
092    }
