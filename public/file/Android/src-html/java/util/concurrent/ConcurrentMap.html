008    import java.util.Map;

014    /**
015     * A {@link java.util.Map} providing additional atomic
016     * <tt>putIfAbsent</tt>, <tt>remove</tt>, and <tt>replace</tt> methods.
017     *
018     * @since 1.5
019     * @author Doug Lea
020     * @param <K> the type of keys maintained by this map
021     * @param <V> the type of mapped values 
022     */
023    public interface ConcurrentMap<K, V> extends Map<K, V> {
024        /**
025         * If the specified key is not already associated
026         * with a value, associate it with the given value.
027         * This is equivalent to
028         * <pre>
029         *   if (!map.containsKey(key)) 
030         *      return map.put(key, value);
031         *   else
032         *      return map.get(key);
033         * </pre>
034         * Except that the action is performed atomically.
035         * @param key key with which the specified value is to be associated.
036         * @param value value to be associated with the specified key.
037         * @return previous value associated with specified key, or <tt>null</tt>
038         *         if there was no mapping for key.  A <tt>null</tt> return can
039         *         also indicate that the map previously associated <tt>null</tt>
040         *         with the specified key, if the implementation supports
041         *         <tt>null</tt> values.
042         *
043         * @throws UnsupportedOperationException if the <tt>put</tt> operation is
044         *            not supported by this map.
045         * @throws ClassCastException if the class of the specified key or value
046         *            prevents it from being stored in this map.
047         * @throws IllegalArgumentException if some aspect of this key or value
048         *            prevents it from being stored in this map.
049         * @throws NullPointerException if this map does not permit <tt>null</tt>
050         *            keys or values, and the specified key or value is
051         *            <tt>null</tt>.
052         *
053         */
054        V putIfAbsent(K key, V value);
055    
056        /**
057         * Remove entry for key only if currently mapped to given value.
058         * Acts as
059         * <pre> 
060         *  if ((map.containsKey(key) && map.get(key).equals(value)) {
061         *     map.remove(key);
062         *     return true;
063         * } else return false;
064         * </pre>
065         * except that the action is performed atomically.
066         * @param key key with which the specified value is associated.
067         * @param value value associated with the specified key.
068         * @return true if the value was removed, false otherwise
069         * @throws NullPointerException if this map does not permit <tt>null</tt>
070         *            keys or values, and the specified key or value is
071         *            <tt>null</tt>.
072         */
073        boolean remove(Object key, Object value);
074    
075    
076        /**
077         * Replace entry for key only if currently mapped to given value.
078         * Acts as
079         * <pre> 
080         *  if ((map.containsKey(key) && map.get(key).equals(oldValue)) {
081         *     map.put(key, newValue);
082         *     return true;
083         * } else return false;
084         * </pre>
085         * except that the action is performed atomically.
086         * @param key key with which the specified value is associated.
087         * @param oldValue value expected to be associated with the specified key.
088         * @param newValue value to be associated with the specified key.
089         * @return true if the value was replaced
090         * @throws NullPointerException if this map does not permit <tt>null</tt>
091         *            keys or values, and the specified key or value is
092         *            <tt>null</tt>.
093         */
094        boolean replace(K key, V oldValue, V newValue);
095    
096        /**
097         * Replace entry for key only if currently mapped to some value.
098         * Acts as
099         * <pre> 
100         *  if ((map.containsKey(key)) {
101         *     return map.put(key, value);
102         * } else return null;
103         * </pre>
104         * except that the action is performed atomically.
105         * @param key key with which the specified value is associated.
106         * @param value value to be associated with the specified key.
107         * @return previous value associated with specified key, or <tt>null</tt>
108         *         if there was no mapping for key.  A <tt>null</tt> return can
109         *         also indicate that the map previously associated <tt>null</tt>
110         *         with the specified key, if the implementation supports
111         *         <tt>null</tt> values.
112         * @throws NullPointerException if this map does not permit <tt>null</tt>
113         *            keys or values, and the specified key or value is
114         *            <tt>null</tt>.
115         */
116        V replace(K key, V value);
117    
118    }
