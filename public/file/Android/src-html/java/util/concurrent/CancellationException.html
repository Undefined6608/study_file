007    package java.util.concurrent;
008    
009    /**
010     * Exception indicating that the result of a value-producing task,
011     * such as a {@link FutureTask}, cannot be retrieved because the task
012     * was cancelled.
013     *
014     * @since 1.5
015     * @author Doug Lea
016     */
017    public class CancellationException extends IllegalStateException {
018        private static final long serialVersionUID = -9202173006928992231L;
019    
020        /**
021         * Constructs a <tt>CancellationException</tt> with no detail message.
022         */
023        public CancellationException() {}
024    
025        /**
026         * Constructs a <tt>CancellationException</tt> with the specified detail
027         * message.
028         *
029         * @param message the detail message
030         */
031        public CancellationException(String message) {
032            super(message);
033        }
034    }
