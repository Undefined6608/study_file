007    package java.util.concurrent;
008    
009    /**
010     * A task that returns a result and may throw an exception.
011     * Implementors define a single method with no arguments called
012     * <tt>call</tt>.
013     *
014     * <p>The <tt>Callable</tt> interface is similar to {@link
015     * java.lang.Runnable}, in that both are designed for classes whose
016     * instances are potentially executed by another thread.  A
017     * <tt>Runnable</tt>, however, does not return a result and cannot
018     * throw a checked exception.
019     *
020     * <p> The {@link Executors} class contains utility methods to
021     * convert from other common forms to <tt>Callable</tt> classes.
022     *
023     * @see Executor
024     * @since 1.5
025     * @author Doug Lea
026     * @param <V> the result type of method <tt>call</tt>
027     */
028    public interface Callable<V> {
029        /**
030         * Computes a result, or throws an exception if unable to do so.
031         *
032         * @return computed result
033         * @throws Exception if unable to compute a result
034         */
035        V call() throws Exception;
036    }
