007    package java.util.concurrent;
008    
009    /**
010     * Exception thrown when a thread tries to wait upon a barrier that is
011     * in a broken state, or which enters the broken state while the thread
012     * is waiting.
013     *
014     * @see CyclicBarrier
015     *
016     * @since 1.5
017     * @author Doug Lea
018     *
019     */
020    public class BrokenBarrierException extends Exception {
021        private static final long serialVersionUID = 7117394618823254244L;
022    
023        /**
024         * Constructs a <tt>BrokenBarrierException</tt> with no specified detail
025         * message.
026         */
027        public BrokenBarrierException() {}
028    
029        /**
030         * Constructs a <tt>BrokenBarrierException</tt> with the specified
031         * detail message.
032         *
033         * @param message the detail message
034         */
035        public BrokenBarrierException(String message) {
036            super(message);
037        }
038    }
