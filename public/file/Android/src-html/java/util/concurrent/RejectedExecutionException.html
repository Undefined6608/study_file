007    package java.util.concurrent;
008    
009    /**
010     * Exception thrown by an {@link Executor} when a task cannot be
011     * accepted for execution.
012     * 
013     * @since 1.5
014     * @author Doug Lea
015     */
016    public class RejectedExecutionException extends RuntimeException {
017        private static final long serialVersionUID = -375805702767069545L;
018    
019        /**
020         * Constructs a <tt>RejectedExecutionException</tt> with no detail message.
021         * The cause is not initialized, and may subsequently be
022         * initialized by a call to {@link #initCause(Throwable) initCause}.
023         */
024        public RejectedExecutionException() { }
025    
026        /**
027         * Constructs a <tt>RejectedExecutionException</tt> with the
028         * specified detail message. The cause is not initialized, and may
029         * subsequently be initialized by a call to {@link
030         * #initCause(Throwable) initCause}.
031         *
032         * @param message the detail message
033         */
034        public RejectedExecutionException(String message) {
035            super(message);
036        }
037    
038        /**
039         * Constructs a <tt>RejectedExecutionException</tt> with the
040         * specified detail message and cause.
041         *
042         * @param  message the detail message
043         * @param  cause the cause (which is saved for later retrieval by the
044         *         {@link #getCause()} method)
045         */
046        public RejectedExecutionException(String message, Throwable cause) {
047            super(message, cause);
048        }
049    
050        /**
051         * Constructs a <tt>RejectedExecutionException</tt> with the
052         * specified cause.  The detail message is set to: <pre> (cause ==
053         * null ? null : cause.toString())</pre> (which typically contains
054         * the class and detail message of <tt>cause</tt>).
055         *
056         * @param  cause the cause (which is saved for later retrieval by the
057         *         {@link #getCause()} method)
058         */
059        public RejectedExecutionException(Throwable cause) {
060            super(cause);
061        }
062    }
