007    package java.util.concurrent;
008    
009    /**
010     * An object that creates new threads on demand.  Using thread factories
011     * removes hardwiring of calls to {@link Thread#Thread(Runnable) new Thread},
012     * enabling applications to use special thread subclasses, priorities, etc.
013     *
014     * <p> 
015     * The simplest implementation of this interface is just:
016     * <pre>
017     * class SimpleThreadFactory implements ThreadFactory {
018     *   public Thread newThread(Runnable r) {
019     *     return new Thread(r);
020     *   }
021     * }
022     * </pre>
023     *
024     * The {@link Executors#defaultThreadFactory} method provides a more
025     * useful simple implementation, that sets the created thread context
026     * to known values before returning it. 
027     * @since 1.5
028     * @author Doug Lea
029     */
030    public interface ThreadFactory { 
031    
032        /**
033         * Constructs a new <tt>Thread</tt>.  Implementations may also initialize
034         * priority, name, daemon status, <tt>ThreadGroup</tt>, etc.
035         *
036         * @param r a runnable to be executed by new thread instance
037         * @return constructed thread
038         */
039        Thread newThread(Runnable r);
040    }
