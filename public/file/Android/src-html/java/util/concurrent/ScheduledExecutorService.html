007    package java.util.concurrent;
008    import java.util.concurrent.atomic.*;
009    import java.util.*;
010    
011    /**
012     * An {@link ExecutorService} that can schedule commands to run after a given
013     * delay, or to execute periodically. 
014     *
015     * <p> The <tt>schedule</tt> methods create tasks with various delays
016     * and return a task object that can be used to cancel or check
017     * execution. The <tt>scheduleAtFixedRate</tt> and
018     * <tt>scheduleWithFixedDelay</tt> methods create and execute tasks
019     * that run periodically until cancelled.  
020     *
021     * <p> Commands submitted using the {@link Executor#execute} and
022     * {@link ExecutorService} <tt>submit</tt> methods are scheduled with
023     * a requested delay of zero. Zero and negative delays (but not
024     * periods) are also allowed in <tt>schedule</tt> methods, and are
025     * treated as requests for immediate execution.
026     *
027     * <p>All <tt>schedule</tt> methods accept <em>relative</em> delays and
028     * periods as arguments, not absolute times or dates. It is a simple
029     * matter to transform an absolute time represented as a {@link
030     * java.util.Date} to the required form. For example, to schedule at
031     * a certain future <tt>date</tt>, you can use: <tt>schedule(task,
032     * date.getTime() - System.currentTimeMillis(),
033     * TimeUnit.MILLISECONDS)</tt>. Beware however that expiration of a
034     * relative delay need not coincide with the current <tt>Date</tt> at
035     * which the task is enabled due to network time synchronization
036     * protocols, clock drift, or other factors. 
037     *
038     * The {@link Executors} class provides convenient factory methods for
039     * the ScheduledExecutorService implementations provided in this package.
040     *
041     * <h3>Usage Example</h3>
042     * 
043     * Here is a class with a method that sets up a ScheduledExecutorService
044     * to beep every ten seconds for an hour:
045     *
046     * <pre>
047     * import static java.util.concurrent.TimeUnit;
048     * class BeeperControl {
049     *    private final ScheduledExecutorService scheduler = 
050     *       Executors.newScheduledThreadPool(1);
051     *
052     *    public void beepForAnHour() {
053     *        final Runnable beeper = new Runnable() {
054     *                public void run() { System.out.println("beep"); }
055     *            };
056     *        final ScheduledFuture&lt;?&gt; beeperHandle = 
057     *            scheduler.scheduleAtFixedRate(beeper, 10, 10, SECONDS);
058     *        scheduler.schedule(new Runnable() {
059     *                public void run() { beeperHandle.cancel(true); }
060     *            }, 60 * 60, SECONDS);
061     *    }
062     * }
063     * </pre>
064     *
065     * @since 1.5
066     * @author Doug Lea
067     */
068    public interface ScheduledExecutorService extends ExecutorService {
069    
070        /**
071         * Creates and executes a one-shot action that becomes enabled
072         * after the given delay.
073         * @param command the task to execute.
074         * @param delay the time from now to delay execution.
075         * @param unit the time unit of the delay parameter.
076         * @return a Future representing pending completion of the task,
077         * and whose <tt>get()</tt> method will return <tt>null</tt>
078         * upon completion.
079         * @throws RejectedExecutionException if task cannot be scheduled
080         * for execution.
081         * @throws NullPointerException if command is null
082         */
083        public ScheduledFuture<?> schedule(Runnable command, long delay,  TimeUnit unit);
084    
085        /**
086         * Creates and executes a ScheduledFuture that becomes enabled after the
087         * given delay.
088         * @param callable the function to execute.
089         * @param delay the time from now to delay execution.
090         * @param unit the time unit of the delay parameter.
091         * @return a ScheduledFuture that can be used to extract result or cancel.
092         * @throws RejectedExecutionException if task cannot be scheduled
093         * for execution.
094         * @throws NullPointerException if callable is null
095         */
096        public <V> ScheduledFuture<V> schedule(Callable<V> callable, long delay, TimeUnit unit);
097    
098        /**
099         * Creates and executes a periodic action that becomes enabled first
100         * after the given initial delay, and subsequently with the given
101         * period; that is executions will commence after
102         * <tt>initialDelay</tt> then <tt>initialDelay+period</tt>, then
103         * <tt>initialDelay + 2 * period</tt>, and so on.  
104         * If any execution of the task
105         * encounters an exception, subsequent executions are suppressed.
106         * Otherwise, the task will only terminate via cancellation or
107         * termination of the executor.
108         * @param command the task to execute.
109         * @param initialDelay the time to delay first execution.
110         * @param period the period between successive executions.
111         * @param unit the time unit of the initialDelay and period parameters
112         * @return a Future representing pending completion of the task,
113         * and whose <tt>get()</tt> method will throw an exception upon
114         * cancellation.
115         * @throws RejectedExecutionException if task cannot be scheduled
116         * for execution.
117         * @throws NullPointerException if command is null
118         * @throws IllegalArgumentException if period less than or equal to zero.
119         */
120        public ScheduledFuture<?> scheduleAtFixedRate(Runnable command, long initialDelay,  long period, TimeUnit unit);
121        
122        /**
123         * Creates and executes a periodic action that becomes enabled first
124         * after the given initial delay, and subsequently with the
125         * given delay between the termination of one execution and the
126         * commencement of the next. If any execution of the task
127         * encounters an exception, subsequent executions are suppressed.
128         * Otherwise, the task will only terminate via cancellation or
129         * termination of the executor.
130         * @param command the task to execute.
131         * @param initialDelay the time to delay first execution.
132         * @param delay the delay between the termination of one
133         * execution and the commencement of the next.
134         * @param unit the time unit of the initialDelay and delay parameters
135         * @return a Future representing pending completion of the task,
136         * and whose <tt>get()</tt> method will throw an exception upon
137         * cancellation.
138         * @throws RejectedExecutionException if task cannot be scheduled
139         * for execution.
140         * @throws NullPointerException if command is null
141         * @throws IllegalArgumentException if delay less than or equal to zero.
142         */
143        public ScheduledFuture<?> scheduleWithFixedDelay(Runnable command, long initialDelay,  long delay, TimeUnit unit);
144    
145    }
