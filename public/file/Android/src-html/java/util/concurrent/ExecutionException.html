007    package java.util.concurrent;
008    
009    /**
010     * Exception thrown when attempting to retrieve the result of a task
011     * that aborted by throwing an exception. This exception can be
012     * inspected using the {@link #getCause()} method.
013     *
014     * @see Future
015     * @since 1.5
016     * @author Doug Lea
017     */
018    public class ExecutionException extends Exception {
019        private static final long serialVersionUID = 7830266012832686185L;
020    
021        /**
022         * Constructs a <tt>ExecutionException</tt> with no detail message.
023         * The cause is not initialized, and may subsequently be
024         * initialized by a call to {@link #initCause(Throwable) initCause}.
025         */
026        protected ExecutionException() { }
027    
028        /**
029         * Constructs a <tt>ExecutionException</tt> with the specified detail
030         * message. The cause is not initialized, and may subsequently be
031         * initialized by a call to {@link #initCause(Throwable) initCause}.
032         *
033         * @param message the detail message
034         */
035        protected ExecutionException(String message) {
036            super(message);
037        }
038    
039        /**
040         * Constructs a <tt>ExecutionException</tt> with the specified detail
041         * message and cause.
042         *
043         * @param  message the detail message
044         * @param  cause the cause (which is saved for later retrieval by the
045         *         {@link #getCause()} method)
046         */
047        public ExecutionException(String message, Throwable cause) {
048            super(message, cause);
049        }
050    
051        /**
052         * Constructs a <tt>ExecutionException</tt> with the specified cause.
053         * The detail message is set to:
054         * <pre>
055         *  (cause == null ? null : cause.toString())</pre>
056         * (which typically contains the class and detail message of
057         * <tt>cause</tt>).
058         *
059         * @param  cause the cause (which is saved for later retrieval by the
060         *         {@link #getCause()} method)
061         */
062        public ExecutionException(Throwable cause) {
063            super(cause);
064        }
065    }
