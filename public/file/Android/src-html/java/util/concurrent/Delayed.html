009    import java.util.*;
010    
011    // BEGIN android-note
012    // Added generic type Delayed to Comparable to be closer to the RI.
013    // END android-note
014    
015    /**
016     * A mix-in style interface for marking objects that should be
017     * acted upon after a given delay.
018     *
019     * <p>An implementation of this interface must define a
020     * <tt>compareTo</tt> method that provides an ordering consistent with
021     * its <tt>getDelay</tt> method.
022     *
023     * @since 1.5
024     * @author Doug Lea
025     */
026    public interface Delayed extends Comparable<Delayed> {
027    
028        /**
029         * Returns the delay associated with this object, in the given time unit.
030         *
031         * @param unit the time unit
032         * @return the delay; zero or negative values indicate that the
033         * delay has already elapsed
034         */
035        long getDelay(TimeUnit unit);
036    }
