016    /**
017     * An optionally-bounded {@linkplain BlockingQueue blocking queue} based on
018     * linked nodes.
019     * This queue orders elements FIFO (first-in-first-out).
020     * The <em>head</em> of the queue is that element that has been on the
021     * queue the longest time.
022     * The <em>tail</em> of the queue is that element that has been on the
023     * queue the shortest time. New elements
024     * are inserted at the tail of the queue, and the queue retrieval
025     * operations obtain elements at the head of the queue.
026     * Linked queues typically have higher throughput than array-based queues but
027     * less predictable performance in most concurrent applications.
028     *
029     * <p> The optional capacity bound constructor argument serves as a
030     * way to prevent excessive queue expansion. The capacity, if unspecified,
031     * is equal to {@link Integer#MAX_VALUE}.  Linked nodes are
032     * dynamically created upon each insertion unless this would bring the
033     * queue above capacity.
034     *
035     * <p>This class implements all of the <em>optional</em> methods
036     * of the {@link Collection} and {@link Iterator} interfaces.
037     *
038     * @since 1.5
039     * @author Doug Lea
040     * @param <E> the type of elements held in this collection
041     *
042     **/
043    public class LinkedBlockingQueue<E> extends AbstractQueue<E>
044            implements BlockingQueue<E>, java.io.Serializable {
045        private static final long serialVersionUID = -6903933977591709194L;
046    
047        /*
048         * A variant of the "two lock queue" algorithm.  The putLock gates
049         * entry to put (and offer), and has an associated condition for
050         * waiting puts.  Similarly for the takeLock.  The "count" field
051         * that they both rely on is maintained as an atomic to avoid
052         * needing to get both locks in most cases. Also, to minimize need
053         * for puts to get takeLock and vice-versa, cascading notifies are
054         * used. When a put notices that it has enabled at least one take,
055         * it signals taker. That taker in turn signals others if more
056         * items have been entered since the signal. And symmetrically for
057         * takes signalling puts. Operations such as remove(Object) and
058         * iterators acquire both locks.
059        */
060    
061        /**
062         * Linked list node class
063         */
064        static class Node<E> {
065            /** The item, volatile to ensure barrier separating write and read */
066            volatile E item;
067            Node<E> next;
068            Node(E x) { item = x; }
069        }
070    
071        /** The capacity bound, or Integer.MAX_VALUE if none */
072        private final int capacity;
073    
074        /** Current number of elements */
075        private final AtomicInteger count = new AtomicInteger(0);
076    
077        /** Head of linked list */
078        private transient Node<E> head;
079    
080        /** Tail of linked list */
081        private transient Node<E> last;
082    
083        /** Lock held by take, poll, etc */
084        private final ReentrantLock takeLock = new ReentrantLock();
085    
086        /** Wait queue for waiting takes */
087        private final Condition notEmpty = takeLock.newCondition();
088    
089        /** Lock held by put, offer, etc */
090        private final ReentrantLock putLock = new ReentrantLock();
091    
092        /** Wait queue for waiting puts */
093        private final Condition notFull = putLock.newCondition();
094    
095        /**
096         * Signal a waiting take. Called only from put/offer (which do not
097         * otherwise ordinarily lock takeLock.)
098         */
099        private void signalNotEmpty() {
100            final ReentrantLock takeLock = this.takeLock;
101            takeLock.lock();
102            try {
103                notEmpty.signal();
104            } finally {
105                takeLock.unlock();
106            }
107        }
108    
109        /**
110         * Signal a waiting put. Called only from take/poll.
111         */
112        private void signalNotFull() {
113            final ReentrantLock putLock = this.putLock;
114            putLock.lock();
115            try {
116                notFull.signal();
117            } finally {
118                putLock.unlock();
119            }
120        }
121    
122        /**
123         * Create a node and link it at end of queue
124         * @param x the item
125         */
126        private void insert(E x) {
127            last = last.next = new Node<E>(x);
128        }
129    
130        /**
131         * Remove a node from head of queue,
132         * @return the node
133         */
134        private E extract() {
135            Node<E> first = head.next;
136            head = first;
137            E x = first.item;
138            first.item = null;
139            return x;
140        }
141    
142        /**
143         * Lock to prevent both puts and takes.
144         */
145        private void fullyLock() {
146            putLock.lock();
147            takeLock.lock();
148        }
149    
150        /**
151         * Unlock to allow both puts and takes.
152         */
153        private void fullyUnlock() {
154            takeLock.unlock();
155            putLock.unlock();
156        }

159        /**
160         * Creates a <tt>LinkedBlockingQueue</tt> with a capacity of
161         * {@link Integer#MAX_VALUE}.
162         */
163        public LinkedBlockingQueue() {
164            this(Integer.MAX_VALUE);
165        }

167        /**
168         * Creates a <tt>LinkedBlockingQueue</tt> with the given (fixed) capacity.
169         *
170         * @param capacity the capacity of this queue.
171         * @throws IllegalArgumentException if <tt>capacity</tt> is not greater
172         *         than zero.
173         */
174        public LinkedBlockingQueue(int capacity) {
175            if (capacity <= 0) throw new IllegalArgumentException();
176            this.capacity = capacity;
177            last = head = new Node<E>(null);
178        }

180        /**
181         * Creates a <tt>LinkedBlockingQueue</tt> with a capacity of
182         * {@link Integer#MAX_VALUE}, initially containing the elements of the
183         * given collection,
184         * added in traversal order of the collection's iterator.
185         * @param c the collection of elements to initially contain
186         * @throws NullPointerException if <tt>c</tt> or any element within it
187         * is <tt>null</tt>
188         */
189        public LinkedBlockingQueue(Collection<? extends E> c) {
190            this(Integer.MAX_VALUE);
191            for (Iterator<? extends E> it = c.iterator(); it.hasNext();)
192                add(it.next());
193        }

209        /**
210         * Returns the number of elements that this queue can ideally (in
211         * the absence of memory or resource constraints) accept without
212         * blocking. This is always equal to the initial capacity of this queue
213         * less the current <tt>size</tt> of this queue.
214         * <p>Note that you <em>cannot</em> always tell if
215         * an attempt to <tt>add</tt> an element will succeed by
216         * inspecting <tt>remainingCapacity</tt> because it may be the
217         * case that a waiting consumer is ready to <tt>take</tt> an
218         * element out of an otherwise full queue.
219         * 
220         * @return the remaining capacity
221         */
222        public int remainingCapacity() {
223            return capacity - count.get();
224        }

226        /**
227         * Adds the specified element to the tail of this queue, waiting if
228         * necessary for space to become available.
229         * @param o the element to add
230         * @throws InterruptedException if interrupted while waiting.
231         * @throws NullPointerException if the specified element is <tt>null</tt>.
232         */
233        public void put(E o) throws InterruptedException {
234            if (o == null) throw new NullPointerException();
235            // Note: convention in all put/take/etc is to preset
236            // local var holding count  negative to indicate failure unless set.
237            int c = -1;
238            final ReentrantLock putLock = this.putLock;
239            final AtomicInteger count = this.count;
240            putLock.lockInterruptibly();
241            try {
242                /*
243                 * Note that count is used in wait guard even though it is
244                 * not protected by lock. This works because count can
245                 * only decrease at this point (all other puts are shut
246                 * out by lock), and we (or some other waiting put) are
247                 * signalled if it ever changes from
248                 * capacity. Similarly for all other uses of count in
249                 * other wait guards.
250                 */
251                try {
252                    while (count.get() == capacity)
253                        notFull.await();
254                } catch (InterruptedException ie) {
255                    notFull.signal(); // propagate to a non-interrupted thread
256                    throw ie;
257                }
258                insert(o);
259                c = count.getAndIncrement();
260                if (c + 1 < capacity)
261                    notFull.signal();
262            } finally {
263                putLock.unlock();
264            }
265            if (c == 0)
266                signalNotEmpty();
267        }
268    
269        /**
270         * Inserts the specified element at the tail of this queue, waiting if
271         * necessary up to the specified wait time for space to become available.
272         * @param o the element to add
273         * @param timeout how long to wait before giving up, in units of
274         * <tt>unit</tt>
275         * @param unit a <tt>TimeUnit</tt> determining how to interpret the
276         * <tt>timeout</tt> parameter
277         * @return <tt>true</tt> if successful, or <tt>false</tt> if
278         * the specified waiting time elapses before space is available.
279         * @throws InterruptedException if interrupted while waiting.
280         * @throws NullPointerException if the specified element is <tt>null</tt>.
281         */
282        public boolean offer(E o, long timeout, TimeUnit unit)
283            throws InterruptedException {
284    
285            if (o == null) throw new NullPointerException();
286            long nanos = unit.toNanos(timeout);
287            int c = -1;
288            final ReentrantLock putLock = this.putLock;
289            final AtomicInteger count = this.count;
290            putLock.lockInterruptibly();
291            try {
292                for (;;) {
293                    if (count.get() < capacity) {
294                        insert(o);
295                        c = count.getAndIncrement();
296                        if (c + 1 < capacity)
297                            notFull.signal();
298                        break;
299                    }
300                    if (nanos <= 0)
301                        return false;
302                    try {
303                        nanos = notFull.awaitNanos(nanos);
304                    } catch (InterruptedException ie) {
305                        notFull.signal(); // propagate to a non-interrupted thread
306                        throw ie;
307                    }
308                }
309            } finally {
310                putLock.unlock();
311            }
312            if (c == 0)
313                signalNotEmpty();
314            return true;
315        }
316    
317        /**
318         * Inserts the specified element at the tail of this queue if possible,
319         * returning immediately if this queue is full.
320         *
321         * @param o the element to add.
322         * @return <tt>true</tt> if it was possible to add the element to
323         *         this queue, else <tt>false</tt>
324         * @throws NullPointerException if the specified element is <tt>null</tt>
325         */
326        public boolean offer(E o) {
327            if (o == null) throw new NullPointerException();
328            final AtomicInteger count = this.count;
329            if (count.get() == capacity)
330                return false;
331            int c = -1;
332            final ReentrantLock putLock = this.putLock;
333            putLock.lock();
334            try {
335                if (count.get() < capacity) {
336                    insert(o);
337                    c = count.getAndIncrement();
338                    if (c + 1 < capacity)
339                        notFull.signal();
340                }
341            } finally {
342                putLock.unlock();
343            }
344            if (c == 0)
345                signalNotEmpty();
346            return c >= 0;
347        }

350        public E take() throws InterruptedException {
351            E x;
352            int c = -1;
353            final AtomicInteger count = this.count;
354            final ReentrantLock takeLock = this.takeLock;
355            takeLock.lockInterruptibly();
356            try {
357                try {
358                    while (count.get() == 0)
359                        notEmpty.await();
360                } catch (InterruptedException ie) {
361                    notEmpty.signal(); // propagate to a non-interrupted thread
362                    throw ie;
363                }
364    
365                x = extract();
366                c = count.getAndDecrement();
367                if (c > 1)
368                    notEmpty.signal();
369            } finally {
370                takeLock.unlock();
371            }
372            if (c == capacity)
373                signalNotFull();
374            return x;
375        }

377        public E poll(long timeout, TimeUnit unit) throws InterruptedException {
378            E x = null;
379            int c = -1;
380            long nanos = unit.toNanos(timeout);
381            final AtomicInteger count = this.count;
382            final ReentrantLock takeLock = this.takeLock;
383            takeLock.lockInterruptibly();
384            try {
385                for (;;) {
386                    if (count.get() > 0) {
387                        x = extract();
388                        c = count.getAndDecrement();
389                        if (c > 1)
390                            notEmpty.signal();
391                        break;
392                    }
393                    if (nanos <= 0)
394                        return null;
395                    try {
396                        nanos = notEmpty.awaitNanos(nanos);
397                    } catch (InterruptedException ie) {
398                        notEmpty.signal(); // propagate to a non-interrupted thread
399                        throw ie;
400                    }
401                }
402            } finally {
403                takeLock.unlock();
404            }
405            if (c == capacity)
406                signalNotFull();
407            return x;
408        }
409    
410        public E poll() {
411            final AtomicInteger count = this.count;
412            if (count.get() == 0)
413                return null;
414            E x = null;
415            int c = -1;
416            final ReentrantLock takeLock = this.takeLock;
417            takeLock.lock();
418            try {
419                if (count.get() > 0) {
420                    x = extract();
421                    c = count.getAndDecrement();
422                    if (c > 1)
423                        notEmpty.signal();
424                }
425            } finally {
426                takeLock.unlock();
427            }
428            if (c == capacity)
429                signalNotFull();
430            return x;
431        }

434        public E peek() {
435            if (count.get() == 0)
436                return null;
437            final ReentrantLock takeLock = this.takeLock;
438            takeLock.lock();
439            try {
440                Node<E> first = head.next;
441                if (first == null)
442                    return null;
443                else
444                    return first.item;
445            } finally {
446                takeLock.unlock();
447            }
448        }

450        public boolean remove(Object o) {
451            if (o == null) return false;
452            boolean removed = false;
453            fullyLock();
454            try {
455                Node<E> trail = head;
456                Node<E> p = head.next;
457                while (p != null) {
458                    if (o.equals(p.item)) {
459                        removed = true;
460                        break;
461                    }
462                    trail = p;
463                    p = p.next;
464                }
465                if (removed) {
466                    p.item = null;
467                    trail.next = p.next;
468                    if (count.getAndDecrement() == capacity)
469                        notFull.signalAll();
470                }
471            } finally {
472                fullyUnlock();
473            }
474            return removed;
475        }
476    
477        public Object[] toArray() {
478            fullyLock();
479            try {
480                int size = count.get();
481                Object[] a = new Object[size];
482                int k = 0;
483                for (Node<E> p = head.next; p != null; p = p.next)
484                    a[k++] = p.item;
485                return a;
486            } finally {
487                fullyUnlock();
488            }
489        }
490    
491        public <T> T[] toArray(T[] a) {
492            fullyLock();
493            try {
494                int size = count.get();
495                if (a.length < size)
496                    a = (T[])java.lang.reflect.Array.newInstance
497                        (a.getClass().getComponentType(), size);
498    
499                int k = 0;
500                for (Node p = head.next; p != null; p = p.next)
501                    a[k++] = (T)p.item;
502                return a;
503            } finally {
504                fullyUnlock();
505            }
506        }

508        public String toString() {
509            fullyLock();
510            try {
511                return super.toString();
512            } finally {
513                fullyUnlock();
514            }
515        }

517        public void clear() {
518            fullyLock();
519            try {
520                head.next = null;
521                if (count.getAndSet(0) == capacity)
522                    notFull.signalAll();
523            } finally {
524                fullyUnlock();
525            }
526        }

528        public int drainTo(Collection<? super E> c) {
529            if (c == null)
530                throw new NullPointerException();
531            if (c == this)
532                throw new IllegalArgumentException();
533            Node first;
534            fullyLock();
535            try {
536                first = head.next;
537                head.next = null;
538                if (count.getAndSet(0) == capacity)
539                    notFull.signalAll();
540            } finally {
541                fullyUnlock();
542            }
543            // Transfer the elements outside of locks
544            int n = 0;
545            for (Node<E> p = first; p != null; p = p.next) {
546                c.add(p.item);
547                p.item = null;
548                ++n;
549            }
550            return n;
551        }
552            
553        public int drainTo(Collection<? super E> c, int maxElements) {
554            if (c == null)
555                throw new NullPointerException();
556            if (c == this)
557                throw new IllegalArgumentException();
558            if (maxElements <= 0)
559                return 0;
560            fullyLock();
561            try {
562                int n = 0;
563                Node<E> p = head.next;
564                while (p != null && n < maxElements) {
565                    c.add(p.item);
566                    p.item = null;
567                    p = p.next;
568                    ++n;
569                }
570                if (n != 0) {
571                    head.next = p;
572                    if (count.getAndAdd(-n) == capacity)
573                        notFull.signalAll();
574                }
575                return n;
576            } finally {
577                fullyUnlock();
578            }
579        }

581        /**
582         * Returns an iterator over the elements in this queue in proper sequence.
583         * The returned <tt>Iterator</tt> is a "weakly consistent" iterator that
584         * will never throw {@link java.util.ConcurrentModificationException},
585         * and guarantees to traverse elements as they existed upon
586         * construction of the iterator, and may (but is not guaranteed to)
587         * reflect any modifications subsequent to construction.
588         *
589         * @return an iterator over the elements in this queue in proper sequence.
590         */
591        public Iterator<E> iterator() {
592          return new Itr();
593        }
594    
595        private class Itr implements Iterator<E> {
596            /*
597             * Basic weak-consistent iterator.  At all times hold the next
598             * item to hand out so that if hasNext() reports true, we will
599             * still have it to return even if lost race with a take etc.
600             */
601            private Node<E> current;
602            private Node<E> lastRet;
603            private E currentElement;
604    
605            Itr() {
606                final ReentrantLock putLock = LinkedBlockingQueue.this.putLock;
607                final ReentrantLock takeLock = LinkedBlockingQueue.this.takeLock;
608                putLock.lock();
609                takeLock.lock();
610                try {
611                    current = head.next;
612                    if (current != null)
613                        currentElement = current.item;
614                } finally {
615                    takeLock.unlock();
616                    putLock.unlock();
617                }
618            }
619    
620            public boolean hasNext() {
621                return current != null;
622            }
623    
624            public E next() {
625                final ReentrantLock putLock = LinkedBlockingQueue.this.putLock;
626                final ReentrantLock takeLock = LinkedBlockingQueue.this.takeLock;
627                putLock.lock();
628                takeLock.lock();
629                try {
630                    if (current == null)
631                        throw new NoSuchElementException();
632                    E x = currentElement;
633                    lastRet = current;
634                    current = current.next;
635                    if (current != null)
636                        currentElement = current.item;
637                    return x;
638                } finally {
639                    takeLock.unlock();
640                    putLock.unlock();
641                }
642            }
643    
644            public void remove() {
645                if (lastRet == null)
646                    throw new IllegalStateException();
647                final ReentrantLock putLock = LinkedBlockingQueue.this.putLock;
648                final ReentrantLock takeLock = LinkedBlockingQueue.this.takeLock;
649                putLock.lock();
650                takeLock.lock();
651                try {
652                    Node<E> node = lastRet;
653                    lastRet = null;
654                    Node<E> trail = head;
655                    Node<E> p = head.next;
656                    while (p != null && p != node) {
657                        trail = p;
658                        p = p.next;
659                    }
660                    if (p == node) {
661                        p.item = null;
662                        trail.next = p.next;
663                        int c = count.getAndDecrement();
664                        if (c == capacity)
665                            notFull.signalAll();
666                    }
667                } finally {
668                    takeLock.unlock();
669                    putLock.unlock();
670                }
671            }
672        }

674        /**
675         * Save the state to a stream (that is, serialize it).
676         *
677         * @serialData The capacity is emitted (int), followed by all of
678         * its elements (each an <tt>Object</tt>) in the proper order,
679         * followed by a null
680         * @param s the stream
681         */
682        private void writeObject(java.io.ObjectOutputStream s)
683            throws java.io.IOException {
684    
685            fullyLock();
686            try {
687                // Write out any hidden stuff, plus capacity
688                s.defaultWriteObject();
689    
690                // Write out all elements in the proper order.
691                for (Node<E> p = head.next; p != null; p = p.next)
692                    s.writeObject(p.item);
693    
694                // Use trailing null as sentinel
695                s.writeObject(null);
696            } finally {
697                fullyUnlock();
698            }
699        }
700    
701        /**
702         * Reconstitute this queue instance from a stream (that is,
703         * deserialize it).
704         * @param s the stream
705         */
706        private void readObject(java.io.ObjectInputStream s)
707            throws java.io.IOException, ClassNotFoundException {
708            // Read in capacity, and any hidden stuff
709            s.defaultReadObject();
710    
711            count.set(0);
712            last = head = new Node<E>(null);
713    
714            // Read in all elements and place in queue
715            for (;;) {
716                E item = (E)s.readObject();
717                if (item == null)
718                    break;
719                add(item);
720            }
721        }
