018    package java.util.jar;
019    
020    import java.io.IOException;
021    import java.security.CodeSigner;
022    import java.security.cert.CertPath;
023    import java.security.cert.Certificate;
024    import java.security.cert.CertificateException;
025    import java.security.cert.CertificateFactory;
026    import java.security.cert.X509Certificate;
027    import java.util.ArrayList;
028    import java.util.List;
029    import java.util.zip.ZipEntry;
030    
031    import javax.security.auth.x500.X500Principal;
032    
033    /**
034     * Represents a single file in a JAR archive together with the manifest
035     * attributes and digital signatures associated with it.
036     * 
037     * @since Android 1.0
038     */
039    public class JarEntry extends ZipEntry {
040        private Attributes attributes;
041    
042        JarFile parentJar;
043       
044        CodeSigner signers[];
045    
046        // Cached factory used to build CertPath-s in <code>getCodeSigners()</code>.
047        private CertificateFactory factory;
048    
049        private boolean isFactoryChecked = false;     
050    
051        /**
052         * Creates a new {@code JarEntry} named name.
053         * 
054         * @param name
055         *            The name of the new {@code JarEntry}.
056         * @since Android 1.0
057         */
058        public JarEntry(String name) {
059            super(name);
060        }
061    
062        /**
063         * Creates a new {@code JarEntry} using the values obtained from entry.
064         * 
065         * @param entry
066         *            The ZipEntry to obtain values from.
067         * @since Android 1.0
068         */
069        public JarEntry(ZipEntry entry) {
070            super(entry);
071        }
072    
073        /**
074         * Returns the {@code Attributes} object associated with this entry or
075         * {@code null} if none exists.
076         * 
077         * @return the {@code Attributes} for this entry.
078         * @exception IOException
079         *                If an error occurs obtaining the {@code Attributes}.
080         * @see Attributes
081         * @since Android 1.0
082         */
083        public Attributes getAttributes() throws IOException {
084            if (attributes != null || parentJar == null) {
085                return attributes;
086            }
087            Manifest manifest = parentJar.getManifest();
088            if (manifest == null) {
089                return null;
090            }
091            return attributes = manifest.getAttributes(getName());
092        }
093    
094        /**
095         * Returns an array of {@code Certificate} Objects associated with this
096         * entry or {@code null} if none exists. Make sure that the everything is
097         * read from the input stream before calling this method, or else the method
098         * returns {@code null}.
099         * 
100         * @return the certificate for this entry.
101         * @see java.security.cert.Certificate
102         * @since Android 1.0
103         */
104        public Certificate[] getCertificates() {
105            if (null == parentJar) {
106                return null;
107            }
108            JarVerifier jarVerifier = parentJar.verifier;
109            if (null == jarVerifier) {
110                return null;
111            }
112            return jarVerifier.getCertificates(getName());
113        }
114    
115        void setAttributes(Attributes attrib) {
116            attributes = attrib;
117        }
118    
119        /**
120         * Create a new {@code JarEntry} using the values obtained from the
121         * argument.
122         * 
123         * @param je
124         *            The {@code JarEntry} to obtain values from.
125         * @since Android 1.0
126         */
127        public JarEntry(JarEntry je) {
128            super(je);
129            parentJar = je.parentJar;
130            attributes = je.attributes;        
131            signers = je.signers;
132        }
133    
134        /**
135         * Returns the code signers for the digital signatures associated with the
136         * JAR file. If there is no such code signer, it returns {@code null}. Make
137         * sure that the everything is read from the input stream before calling
138         * this method, or else the method returns {@code null}.
139         * 
140         * @return the code signers for the JAR entry.
141         * @see CodeSigner
142         * @since Android 1.0
143         */
144        public CodeSigner[] getCodeSigners() {
145            if (null == signers) {
146                signers = getCodeSigners(getCertificates());
147            }
148            if (null == signers) {
149                return null;
150            }
151    
152            CodeSigner[] tmp = new CodeSigner[signers.length];
153            System.arraycopy(signers, 0, tmp, 0, tmp.length);
154            return tmp;
155        }
156    
157        private CodeSigner[] getCodeSigners(Certificate[] certs) {
158            if(null == certs) {
159                return null;
160            }
161    
162            X500Principal prevIssuer = null;
163            ArrayList<Certificate> list = new ArrayList<Certificate>(certs.length);
164            ArrayList<CodeSigner> asigners = new ArrayList<CodeSigner>();
165    
166            for (Certificate element : certs) {
167                if (!(element instanceof X509Certificate)) {
168                    // Only X509Certificate-s are taken into account - see API spec.
169                    continue;
170                }
171                X509Certificate x509 = (X509Certificate) element;
172                if (null != prevIssuer) {
173                    X500Principal subj = x509.getSubjectX500Principal();
174                    if (!prevIssuer.equals(subj)) {
175                        // Ok, this ends the previous chain,
176                        // so transform this one into CertPath ...
177                        addCodeSigner(asigners, list);
178                        // ... and start a new one
179                        list.clear();
180                    }// else { it's still the same chain }
181    
182                }
183                prevIssuer = x509.getIssuerX500Principal();
184                list.add(x509);
185            }
186            if (!list.isEmpty()) {
187                addCodeSigner(asigners, list);
188            }
189            if (asigners.isEmpty()) {
190                // 'signers' is 'null' already
191                return null;
192            }
193    
194            CodeSigner[] tmp = new CodeSigner[asigners.size()];
195            asigners.toArray(tmp);
196            return tmp;
197    
198        }
199    
200        private void addCodeSigner(ArrayList<CodeSigner> asigners,
201                List<Certificate> list) {
202            CertPath certPath = null;
203            if (!isFactoryChecked) {
204                try {
205                    factory = CertificateFactory.getInstance("X.509"); //$NON-NLS-1$
206                } catch (CertificateException ex) {
207                    // do nothing
208                } finally {
209                    isFactoryChecked = true;
210                }
211            }
212            if (null == factory) {
213                return;
214            }
215            try {
216                certPath = factory.generateCertPath(list);
217            } catch (CertificateException ex) {
218                // do nothing
219            }
220            if (null != certPath) {
221                asigners.add(new CodeSigner(certPath, null));
222            }
223        }
224    }
