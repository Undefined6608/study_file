017    package java.util.regex;
018    
019    import com.ibm.icu4jni.regex.NativeRegEx;
020    
021    /**
022     * Provides a means of matching regular expressions against a given input,
023     * finding occurrences of regular expressions in a given input, or replacing
024     * parts of a given input. A {@code Matcher} instance has an associated {@link
025     * Pattern} instance and an input text. A typical use case is to
026     * iteratively find all occurrences of the {@code Pattern}, until the end of
027     * the input is reached, as the following example illustrates:
028     * 
029     * <p/>
030     * 
031     * <pre>
032     * Pattern p = Pattern.compile("[A-Za-z]+");
033     *  
034     * Matcher m = p.matcher("Hello, Android!");
035     * while (m.find()) {
036     *     System.out.println(m.group()); // prints "Hello" and "Android"
037     * }
038     * </pre>
039     * 
040     * <p/>
041     * 
042     * The {@code Matcher} has a state that results from the previous operations.
043     * For example, it knows whether the most recent attempt to find the
044     * {@code Pattern} was successful and at which position the next attempt would
045     * resume the search. Depending on the application's needs, it may become
046     * necessary to explicitly {@link #reset()} this state from time to time.
047     * 
048     * @since Android 1.0
049     */
050    public final class Matcher implements MatchResult {
051    
052        /**
053         * Holds the pattern, that is, the compiled regular expression.
054         */
055        private Pattern pattern;
056    
057        /**
058         * Holds the handle for the native version of the pattern.
059         */
060        private int nativePattern;
061        
062        /**
063         * Holds the input text.
064         */
065        private String input = "";
066    
067        /**
068         * Holds the start of the region, or 0 if the matching should start at the
069         * beginning of the text.
070         */
071        private int regionStart;
072        
073        /**
074         * Holds the end of the region, or input.length() if the matching should
075         * go until the end of the input.
076         */
077        private int regionEnd;
078    
079        /**
080         * Reflects whether we just reset the matcher or whether we already
081         * started some find/replace operations.
082         */
083        private boolean searching;
084        
085        /**
086         * Holds the position where the next find operation will take place. 
087         */
088        private int findPos;
089        
090        /**
091         * Holds the position where the next append operation will take place. 
092         */
093        private int appendPos;
094        
095        /**
096         * Reflects whether a match has been found during the most recent find
097         * operation.
098         */
099        private boolean matchFound;
100    
101        /**
102         * Holds the offsets for the most recent match.
103         */
104        private int[] matchOffsets;
105    
106        /**
107         * Reflects whether the bounds of the region are anchoring.
108         */
109        private boolean anchoringBounds = true;
110        
111        /**
112         * Reflects whether the bounds of the region are transparent.
113         */
114        private boolean transparentBounds;
115    
116        /**
117         * Creates a matcher for a given combination of pattern and input. Both
118         * elements can be changed later on.
119         * 
120         * @param pattern
121         *            the pattern to use.
122         * @param input
123         *            the input to use.
124         */
125        Matcher(Pattern pattern, CharSequence input) {
126            usePattern(pattern);
127            reset(input);
128        }
129    
130        /**
131         * Resets the Matcher. A new input sequence and a new region can be
132         * specified. Results of a previous find get lost. The next attempt to find
133         * an occurrence of the Pattern in the string will start at the beginning of
134         * the region. This is the internal version of reset() to which the several
135         * public versions delegate.
136         * 
137         * @param input
138         *            the input sequence.
139         * @param start
140         *            the start of the region.
141         * @param end
142         *            the end of the region.
143         * 
144         * @return the matcher itself.
145         */
146        private Matcher reset(CharSequence input, int start, int end) {
147            if (input == null) {
148                throw new IllegalArgumentException();
149            }
150            
151            if (start < 0 || end < 0 || start > input.length() || 
152                    end > input.length() || start > end) {
153                throw new IllegalArgumentException();
154            }
155    
156            // Maybe should have a reset() here, but it makes thing worse...
157            // NativeRegEx.reset(nativePattern, 0);
158            
159            if (!input.equals(this.input)) {
160                this.input = input.toString();
161                
162                NativeRegEx.setText(nativePattern, this.input);
163                
164                regionStart = 0;
165                regionEnd = input.length();
166            }
167    
168            if (start != regionStart || end != regionEnd) {
169                regionStart = start;
170                regionEnd = end;
171    
172                NativeRegEx.setRegion(nativePattern, regionStart, regionEnd);
173            }
174    
175            searching = false;
176            matchFound = false;
177            findPos = regionStart;
178            appendPos = 0;
179            
180            return this;
181        }
182    
183        /**
184         * Resets the {@code Matcher}. This results in the region being set to the
185         * whole input. Results of a previous find get lost. The next attempt to
186         * find an occurrence of the {@link Pattern} in the string will start at the
187         * beginning of the input.
188         * 
189         * @return the {@code Matcher} itself.
190         * 
191         * @since Android 1.0
192         */
193        public Matcher reset() {
194            return reset(input, 0, input.length());
195        }
196    
197        /**
198         * Provides a new input and resets the {@code Matcher}. This results in the
199         * region being set to the whole input. Results of a previous find get lost.
200         * The next attempt to find an occurrence of the {@link Pattern} in the
201         * string will start at the beginning of the input.
202         * 
203         * @param input
204         *            the new input sequence.
205         * 
206         * @return the {@code Matcher} itself.
207         * 
208         * @since Android 1.0
209         */
210        public Matcher reset(CharSequence input) {
211            return reset(input, 0, input.length());
212        }
213    
214        /**
215         * Sets a new pattern for the {@code Matcher}. Results of a previous find
216         * get lost. The next attempt to find an occurrence of the {@link Pattern}
217         * in the string will start at the beginning of the input.
218         * 
219         * @param pattern
220         *            the new {@code Pattern}.
221         * 
222         * @return the {@code Matcher} itself.
223         * 
224         * @since Android 1.0
225         */
226        public Matcher usePattern(Pattern pattern) {
227            if (pattern == null) {
228                throw new IllegalArgumentException();
229            }
230            
231            this.pattern = pattern;
232            
233            if (nativePattern != 0) {
234                NativeRegEx.close(nativePattern);
235            }
236            nativePattern = NativeRegEx.clone(pattern.mNativePattern);
237    
238            if (input != null) {
239                NativeRegEx.setText(nativePattern, input);
240                NativeRegEx.setRegion(nativePattern, regionStart, regionEnd);
241                NativeRegEx.useAnchoringBounds(nativePattern, anchoringBounds);
242                NativeRegEx.useTransparentBounds(nativePattern, transparentBounds);
243            }
244            
245            matchOffsets = new int[(this.pattern.mGroupCount + 1) * 2];
246            matchFound = false;
247            return this;
248        }
249    
250        /**
251         * Returns the {@link Pattern} instance used inside this matcher.
252         * 
253         * @return the {@code Pattern} instance.
254         * 
255         * @since Android 1.0
256         */
257        public Pattern pattern() {
258            return pattern;
259        }
260    
261        /**
262         * Returns the number of groups in the results, which is always equal to
263         * the number of groups in the original regular expression.
264         * 
265         * @return the number of groups.
266         * 
267         * @since Android 1.0
268         */
269        public int groupCount() {
270            return pattern.mGroupCount;
271        }

273        /**
274         * Resets this matcher and sets a region. Only characters inside the region

276         * 
277         * @param start

281         * @return the {@code Matcher} itself.
282         * @since Android 1.0

284        public Matcher region(int start, int end) {
285            return reset(input, start, end);
286        }

296            return regionStart;
297        }

300         * Returns this matcher's region end, that is, the first character that is

303         * @return the end of the region.

307            return regionEnd;
308        }

310        /**
311         * Determines whether this matcher has anchoring bounds enabled or not. When
312         * anchoring bounds are enabled, the start and end of the input match the
313         * '^' and '$' meta-characters, otherwise not. Anchoring bounds are enabled

315         * 
316         * @param value
317         *            the new value for anchoring bounds.
318         * @return the {@code Matcher} itself.
319         * @since Android 1.0

322            anchoringBounds = value;
323            NativeRegEx.useAnchoringBounds(nativePattern, value);
324            return this;
325        }

327        /**
328         * Indicates whether this matcher has anchoring bounds enabled. When
329         * anchoring bounds are enabled, the start and end of the input match the
330         * '^' and '$' meta-characters, otherwise not. Anchoring bounds are enabled
331         * by default.
332         * 
333         * @return true if (and only if) the {@code Matcher} uses anchoring bounds.
334         * @since Android 1.0
335         */
336        public boolean hasAnchoringBounds() {
337            return anchoringBounds;
338        }

340        /**
341         * Determines whether this matcher has transparent bounds enabled or not.
342         * When transparent bounds are enabled, the parts of the input outside the
343         * region are subject to lookahead and lookbehind, otherwise they are not.
344         * Transparent bounds are disabled by default.
345         * 
346         * @param value
347         *            the new value for transparent bounds.
348         * @return the {@code Matcher} itself.
349         * @since Android 1.0
350         */
351        public Matcher useTransparentBounds(boolean value) {
352            transparentBounds = value;
353            NativeRegEx.useTransparentBounds(nativePattern, value);
354            return this;
355        }

357        /**
358         * Indicates whether this matcher has transparent bounds enabled. When
359         * transparent bounds are enabled, the parts of the input outside the region
360         * are subject to lookahead and lookbehind, otherwise they are not.
361         * Transparent bounds are disabled by default.
362         * 
363         * @return true if (and only if) the {@code Matcher} uses anchoring bounds.
364         * @since Android 1.0
365         */
366        public boolean hasTransparentBounds() {
367            return transparentBounds;
368        }
369        
370        /**
371         * Makes sure that a successful match has been made. Is invoked internally
372         * from various places in the class.
373         * 
374         * @throws IllegalStateException
375         *             if no successful match has been made.
376         * 
377         * @since Android 1.0
378         */
379        private void ensureMatch() throws IllegalStateException {
380            if (!matchFound) {
381                throw new IllegalStateException("No successful match so far");
382            }
383        }
384        
385        /**
386         * Returns the next occurrence of the {@link Pattern} in the input. If a
387         * previous match was successful, the method continues the search from the
388         * first character following that match in the input. Otherwise it searches
389         * either from the region start (if one has been set), or from position 0.
390         * 
391         * @return true if (and only if) a match has been found.
392         * @since Android 1.0
393         */
394        public boolean find() {
395            if (!searching) {
396                searching = true;
397                matchFound = NativeRegEx.find(nativePattern, -1);
398            } else {
399                matchFound = NativeRegEx.findNext(nativePattern);
400            }
401    
402            if (matchFound) {
403                NativeRegEx.startEnd(nativePattern, matchOffsets);
404                findPos = matchOffsets[1];
405            }
406            
407            return matchFound;
408        }
409    
410        /**
411         * Returns the next occurrence of the {@link Pattern} in the input. The
412         * method starts the search from the given character in the input.
413         * 
414         * @param start
415         *            The index in the input at which the find operation is to
416         *            begin. If this is less than the start of the region, it is
417         *            automatically adjusted to that value. If it is beyond the end
418         *            of the region, the method will fail.
419         * @return true if (and only if) a match has been found.
420         * @since Android 1.0
421         */
422        public boolean find(int start) {
423            findPos = start;
424            
425            if (findPos < regionStart) {
426                findPos = regionStart;
427            } else if (findPos >= regionEnd) {
428                matchFound = false;
429                return false;
430            }
431            
432            matchFound = NativeRegEx.find(nativePattern, findPos); 
433            if (matchFound) {
434                NativeRegEx.startEnd(nativePattern, matchOffsets);
435                findPos = matchOffsets[1];
436            }
437            
438            return matchFound;
439        }
440    
441        /**
442         * Tries to match the {@link Pattern} against the entire region (or the
443         * entire input, if no region has been set).
444         * 
445         * @return true if (and only if) the {@code Pattern} matches the entire
446         *         region.
447         * 
448         * @since Android 1.0
449         */
450        public boolean matches() {
451            matchFound = NativeRegEx.matches(nativePattern, -1); 
452            if (matchFound) {
453                NativeRegEx.startEnd(nativePattern, matchOffsets);
454                findPos = matchOffsets[1];
455            }
456            
457            return matchFound;
458        }
459    
460        /**
461         * Tries to match the {@link Pattern}, starting from the beginning of the
462         * region (or the beginning of the input, if no region has been set).
463         * Doesn't require the {@code Pattern} to match against the whole region.
464         * 
465         * @return true if (and only if) the {@code Pattern} matches.
466         * 
467         * @since Android 1.0
468         */
469        public boolean lookingAt() {
470            matchFound = NativeRegEx.lookingAt(nativePattern, -1); 
471            if (matchFound) {
472                NativeRegEx.startEnd(nativePattern, matchOffsets);
473                findPos = matchOffsets[1];
474            }
475            
476            return matchFound;
477        }

479        /**
480         * Returns the index of the first character of the text that matched the
481         * whole regular expression.
482         * 
483         * @return the character index.
484         * @throws IllegalStateException
485         *             if no successful match has been made.
486         * @since Android 1.0
487         */
488        public int start() throws IllegalStateException {
489            return start(0);
490        }
491    
492        /**
493         * Returns the index of the first character of the text that matched a given
494         * group.
495         * 
496         * @param group
497         *            the group, ranging from 0 to groupCount() - 1, with 0
498         *            representing the whole pattern.
499         * @return the character index.
500         * @throws IllegalStateException
501         *             if no successful match has been made.
502         * @since Android 1.0
503         */
504        public int start(int group) throws IllegalStateException {
505            ensureMatch();
506            return matchOffsets[group * 2];
507        }
508    
509        /**
510         * Returns the index of the first character following the text that matched
511         * the whole regular expression.
512         * 
513         * @return the character index.
514         * @throws IllegalStateException
515         *             if no successful match has been made.
516         * @since Android 1.0
517         */
518        public int end() {
519            return end(0);
520        }
521    
522        /**
523         * Returns the index of the first character following the text that matched
524         * a given group.
525         * 
526         * @param group
527         *            the group, ranging from 0 to groupCount() - 1, with 0
528         *            representing the whole pattern.
529         * @return the character index.
530         * @throws IllegalStateException
531         *             if no successful match has been made.
532         * @since Android 1.0
533         */
534        public int end(int group) {
535            ensureMatch();
536            return matchOffsets[(group * 2) + 1];
537        }
538    
539        /**
540         * Returns the text that matched the whole regular expression.
541         * 
542         * @return the text.
543         * @throws IllegalStateException
544         *             if no successful match has been made.
545         * @since Android 1.0
546         */
547        public String group() {
548            return group(0);
549        }
550    
551        /**
552         * Returns the text that matched a given group of the regular expression.
553         * 
554         * @param group
555         *            the group, ranging from 0 to groupCount() - 1, with 0
556         *            representing the whole pattern.
557         * @return the text that matched the group.
558         * @throws IllegalStateException
559         *             if no successful match has been made.
560         * @since Android 1.0
561         */
562        public String group(int group) {
563            ensureMatch();
564            int from = matchOffsets[group * 2];
565            int to = matchOffsets[(group * 2) + 1];
566            if (from == -1 || to == -1) {
567                return null;
568            } else {
569                return input.substring(from, to);
570            }
571        }

595        /**
596         * Converts the current match into a separate {@link MatchResult} instance
597         * that is independent from this matcher. The new object is unaffected when
598         * the state of this matcher changes.
599         * 
600         * @return the new {@code MatchResult}.
601         * @throws IllegalStateException

607            return new MatchResultImpl(input, matchOffsets);
608        }

610        /**
611         * Appends a literal part of the input plus a replacement for the current
612         * match to a given {@link StringBuffer}. The literal part is exactly the
613         * part of the input between the previous match and the current match. The
614         * method can be used in conjunction with {@link #find()} and
615         * {@link #appendTail(StringBuffer)} to walk through the input and replace
616         * all occurrences of the {@code Pattern} with something else.
617         * 
618         * @param buffer
619         *            the {@code StringBuffer} to append to.
620         * @param replacement
621         *            the replacement text.
622         * @return the {@code Matcher} itself.
623         * @throws IllegalStateException
624         *             if no successful match has been made.
625         * @since Android 1.0
626         */
627        public Matcher appendReplacement(StringBuffer buffer, String replacement)
628                throws IllegalStateException {
629            
630            buffer.append(input.substring(appendPos, start()));
631            appendEvaluated(buffer, replacement);
632            appendPos = end();
633            
634            return this;
635        }
636    
637        /**
638         * Appends the (unmatched) remainder of the input to the given
639         * {@link StringBuffer}. The method can be used in conjunction with
640         * {@link #find()} and {@link #appendReplacement(StringBuffer, String)} to
641         * walk through the input and replace all matches of the {@code Pattern}
642         * with something else.
643         * 
644         * @param buffer
645         *            the {@code StringBuffer} to append to.
646         * @return the {@code StringBuffer}.
647         * @throws IllegalStateException
648         *             if no successful match has been made.
649         * @since Android 1.0
650         */
651        public StringBuffer appendTail(StringBuffer buffer) {
652            if (appendPos < regionEnd) {
653                buffer.append(input.substring(appendPos, regionEnd));
654            }
655            
656            return buffer;
657        }
658    
659        /**
660         * Internal helper method to append a given string to a given string buffer.
661         * If the string contains any references to groups, these are replaced by
662         * the corresponding group's contents.
663         * 
664         * @param buffer
665         *            the string buffer.
666         * @param s
667         *            the string to append.
668         */
669        private void appendEvaluated(StringBuffer buffer, String s) {
670            boolean escape = false;
671            boolean dollar = false;
672            
673            for (int i = 0; i < s.length(); i++) {
674                char c = s.charAt(i);
675                if (c == '\\' && !escape) {
676                    escape = true;
677                } else if (c == '$' && !escape) {
678                    dollar = true;
679                } else if (c >= '0' && c <= '9' && dollar) {
680                    buffer.append(group(c - '0'));
681                    dollar = false;
682                } else {
683                    buffer.append(c);
684                    dollar = false;
685                    escape = false;
686                }
687            }
688            
689            // This seemingly stupid piece of code reproduces a JDK bug. 
690            if (escape) {
691                throw new ArrayIndexOutOfBoundsException(s.length());
692            }
693        }
694        
695        /**
696         * Replaces all occurrences of this matcher's pattern in the input with a
697         * given string.
698         * 
699         * @param replacement
700         *            the replacement text.
701         * @return the modified input string.
702         * @since Android 1.0
703         */
704        public String replaceAll(String replacement) {
705            StringBuffer buffer = new StringBuffer(input.length());
706            
707            findPos = 0;
708            appendPos = 0;
709            matchFound = false;
710            searching = false;
711            
712            while (find()) {
713                appendReplacement(buffer, replacement);
714            }
715            
716            return appendTail(buffer).toString();
717        }
718    
719        /**
720         * Replaces the first occurrence of this matcher's pattern in the input with
721         * a given string.
722         * 
723         * @param replacement
724         *            the replacement text.
725         * @return the modified input string.
726         * @since Android 1.0
727         */
728        public String replaceFirst(String replacement) {
729            StringBuffer buffer = new StringBuffer(input.length());
730    
731            findPos = 0;
732            appendPos = 0;
733            matchFound = false;
734            searching = false;
735            
736            if (find()) {
737                appendReplacement(buffer, replacement);
738            }
739            
740            return appendTail(buffer).toString();
741        }

743        /**
744         * Returns a replacement string for the given one that has all backslashes
745         * and dollar signs escaped.
746         * 
747         * @param s
748         *            the input string.
749         * @return the input string, with all backslashes and dollar signs having
750         *         been escaped.
751         * @since Android 1.0
752         */
753        public static String quoteReplacement(String s) {
754            StringBuffer buffer = new StringBuffer(s.length());
755            
756            for (int i = 0; i < s.length(); i++) {
757                char c = s.charAt(i);
758                if (c == '\\' || c == '$') {
759                    buffer.append('\\');
760                }
761                buffer.append(c);
762            }
763            
764            return buffer.toString();
765        }

767        @Override
768        protected void finalize() throws Throwable {
769            try {
770                if (nativePattern != 0) {
771                    NativeRegEx.close(nativePattern);
772                }
773            }
774            finally {
775                super.finalize();
776            }
777        }
