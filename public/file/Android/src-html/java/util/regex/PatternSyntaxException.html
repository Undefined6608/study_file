017    package java.util.regex;
018    
019    import java.util.Arrays;
020    
021    /**
022     * Encapsulates a syntax error that occurred during the compilation of a
023     * {@link Pattern}. Might include a detailed description, the original regular
024     * expression, and the index at which the error occurred.
025     * 
026     * @see Pattern#compile(String)
027     * @see Pattern#compile(java.lang.String,int)
028     * 
029     * @since Android 1.0
030     */
031    public class PatternSyntaxException extends IllegalArgumentException {
032        
033        private static final long serialVersionUID = -3864639126226059218L;
034        
035        /**
036         * Holds the syntactically incorrect regular expression, or null if the
037         * regular expression is not known.
038         */
039        private String pattern;
040    
041        /**
042         * Holds the description of the syntax error, or null if the description is
043         * not known.
044         */
045        private String description;
046    
047        /**
048         * Holds the index around which the error occured, or -1, in case it is
049         * unknown.
050         */
051        private int index = -1;
052    
053        /**
054         * Creates a new PatternSyntaxException for a given message, pattern, and
055         * error index.
056         * 
057         * @param description
058         *            the description of the syntax error, or {@code null} if the
059         *            description is not known.
060         * @param pattern
061         *            the syntactically incorrect regular expression, or
062         *            {@code null} if the regular expression is not known.
063         * @param index
064         *            the character index around which the error occurred, or -1 if
065         *            the index is not known.
066         * @since Android 1.0
067         */
068        public PatternSyntaxException(String description, String pattern, int index) {
069            this.pattern = pattern;
070            this.description = description;
071            this.index = index;
072        }
073    
074        /**
075         * Returns the syntactically incorrect regular expression.
076         * 
077         * @return the regular expression.
078         * 
079         * @since Android 1.0
080         */
081        public String getPattern() {
082            return pattern;
083        }
084    
085        /**
086         * Returns a detailed error message for the exception. The message is
087         * potentially multi-line, and it might include a detailed description, the
088         * original regular expression, and the index at which the error occured.
089         * 
090         * @return the error message.
091         * 
092         * @since Android 1.0
093         */
094        @Override
095        public String getMessage() {
096            StringBuilder builder = new StringBuilder("Syntax error");
097            
098            if (description != null) {
099                builder.append(' ');
100                builder.append(description);
101            }
102    
103            if (index >= 0) {
104                builder.append(" near index " + index + ":");
105            }
106            
107            if (pattern != null) {
108                builder.append('\n');
109                builder.append(pattern);
110                
111                if (index >= 0) {
112                    char[] spaces = new char[index];
113                    Arrays.fill(spaces, ' ');
114                    builder.append('\n');
115                    builder.append(spaces);
116                    builder.append('^');
117                }
118            }
119            
120            return builder.toString();
121        }
122    
123        /**
124         * Returns the description of the syntax error, or {@code null} if the
125         * description is not known.
126         * 
127         * @return the description.
128         * @since Android 1.0
129         */
130        public String getDescription() {
131            return description;
132        }
133    
134        /**
135         * Returns the character index around which the error occurred, or -1 if the
136         * index is not known.
137         * 
138         * @return the index.
139         * 
140         * @since Android 1.0
141         */
142        public int getIndex() {
143            return index;
144        }
145    
146    }
