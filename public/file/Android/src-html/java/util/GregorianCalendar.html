035    /**
036     * {@code GregorianCalendar} is a concrete subclass of {@link Calendar}
037     * and provides the standard calendar used by most of the world.
038     * 
039     * <p>
040     * The standard (Gregorian) calendar has 2 eras, BC and AD.
041     * 
042     * <p>
043     * This implementation handles a single discontinuity, which corresponds by
044     * default to the date the Gregorian calendar was instituted (October 15, 1582
045     * in some countries, later in others). The cutover date may be changed by the
046     * caller by calling {@code setGregorianChange()}.
047     * 
048     * <p>
049     * Historically, in those countries which adopted the Gregorian calendar first,
050     * October 4, 1582 was thus followed by October 15, 1582. This calendar models
051     * this correctly. Before the Gregorian cutover, {@code GregorianCalendar}
052     * implements the Julian calendar. The only difference between the Gregorian and
053     * the Julian calendar is the leap year rule. The Julian calendar specifies leap
054     * years every four years, whereas the Gregorian calendar omits century years
055     * which are not divisible by 400.
056     * 
057     * <p>
058     * {@code GregorianCalendar} implements <em>proleptic</em> Gregorian
059     * and Julian calendars. That is, dates are computed by extrapolating the
060     * current rules indefinitely far backward and forward in time. As a result,
061     * {@code GregorianCalendar} may be used for all years to generate
062     * meaningful and consistent results. However, dates obtained using
063     * {@code GregorianCalendar} are historically accurate only from March 1,
064     * 4 AD onward, when modern Julian calendar rules were adopted. Before this
065     * date, leap year rules were applied irregularly, and before 45 BC the Julian
066     * calendar did not even exist.
067     * 
068     * <p>
069     * Prior to the institution of the Gregorian calendar, New Year's Day was March
070     * 25. To avoid confusion, this calendar always uses January 1. A manual
071     * adjustment may be made if desired for dates that are prior to the Gregorian
072     * changeover and which fall between January 1 and March 24.
073     * 
074     * <p>
075     * Values calculated for the {@code WEEK_OF_YEAR} field range from 1 to
076     * 53. Week 1 for a year is the earliest seven day period starting on
077     * {@code getFirstDayOfWeek()} that contains at least
078     * {@code getMinimalDaysInFirstWeek()} days from that year. It thus
079     * depends on the values of {@code getMinimalDaysInFirstWeek()},
080     * {@code getFirstDayOfWeek()}, and the day of the week of January 1.
081     * Weeks between week 1 of one year and week 1 of the following year are
082     * numbered sequentially from 2 to 52 or 53 (as needed).
083     * 
084     * <p>
085     * For example, January 1, 1998 was a Thursday. If
086     * {@code getFirstDayOfWeek()} is {@code MONDAY} and
087     * {@code getMinimalDaysInFirstWeek()} is 4 (these are the values
088     * reflecting ISO 8601 and many national standards), then week 1 of 1998 starts
089     * on December 29, 1997, and ends on January 4, 1998. If, however,
090     * {@code getFirstDayOfWeek()} is {@code SUNDAY}, then week 1 of
091     * 1998 starts on January 4, 1998, and ends on January 10, 1998; the first three
092     * days of 1998 then are part of week 53 of 1997.
093     * 
094     * <p>
095     * Values calculated for the {@code WEEK_OF_MONTH} field range from 0 or
096     * 1 to 4 or 5. Week 1 of a month (the days with <code>WEEK_OF_MONTH =
097     * 1</code>)
098     * is the earliest set of at least {@code getMinimalDaysInFirstWeek()}
099     * contiguous days in that month, ending on the day before
100     * {@code getFirstDayOfWeek()}. Unlike week 1 of a year, week 1 of a
101     * month may be shorter than 7 days, need not start on
102     * {@code getFirstDayOfWeek()}, and will not include days of the
103     * previous month. Days of a month before week 1 have a
104     * {@code WEEK_OF_MONTH} of 0.
105     * 
106     * <p>
107     * For example, if {@code getFirstDayOfWeek()} is {@code SUNDAY}
108     * and {@code getMinimalDaysInFirstWeek()} is 4, then the first week of
109     * January 1998 is Sunday, January 4 through Saturday, January 10. These days
110     * have a {@code WEEK_OF_MONTH} of 1. Thursday, January 1 through
111     * Saturday, January 3 have a {@code WEEK_OF_MONTH} of 0. If
112     * {@code getMinimalDaysInFirstWeek()} is changed to 3, then January 1
113     * through January 3 have a {@code WEEK_OF_MONTH} of 1.
114     * 
115     * <p>
116     * <strong>Example:</strong> <blockquote>
117     * 
118     * <pre>
119     * // get the supported ids for GMT-08:00 (Pacific Standard Time)
120     * String[] ids = TimeZone.getAvailableIDs(-8 * 60 * 60 * 1000);
121     * // if no ids were returned, something is wrong. get out.
122     * if (ids.length == 0)
123     *     System.exit(0);
124     *
125     *  // begin output
126     * System.out.println("Current Time");
127     *
128     * // create a Pacific Standard Time time zone
129     * SimpleTimeZone pdt = new SimpleTimeZone(-8 * 60 * 60 * 1000, ids[0]);

166     * calendar.set(Calendar.HOUR, 3);
167     * System.out.println("ERA: " + calendar.get(Calendar.ERA));
168     * System.out.println("YEAR: " + calendar.get(Calendar.YEAR));

171     * System.out.println("WEEK_OF_MONTH: " + calendar.get(Calendar.WEEK_OF_MONTH));
172     * System.out.println("DATE: " + calendar.get(Calendar.DATE));
173     * System.out.println("DAY_OF_MONTH: " + calendar.get(Calendar.DAY_OF_MONTH));

175     * System.out.println("DAY_OF_WEEK: " + calendar.get(Calendar.DAY_OF_WEEK));
176     * System.out.println("DAY_OF_WEEK_IN_MONTH: "
177     *                    + calendar.get(Calendar.DAY_OF_WEEK_IN_MONTH));
178     * System.out.println("AM_PM: " + calendar.get(Calendar.AM_PM));
179     * System.out.println("HOUR: " + calendar.get(Calendar.HOUR));
180     * System.out.println("HOUR_OF_DAY: " + calendar.get(Calendar.HOUR_OF_DAY));

182     * System.out.println("SECOND: " + calendar.get(Calendar.SECOND));
183     * System.out.println("MILLISECOND: " + calendar.get(Calendar.MILLISECOND));
184     * System.out.println("ZONE_OFFSET: "
185     *        + (calendar.get(Calendar.ZONE_OFFSET)/(60*60*1000))); // in hours
186     * System.out.println("DST_OFFSET: "

373        /**
374         * Adds the specified amount to a {@code Calendar} field.
375         * 
376         * @param field
377         *            the {@code Calendar} field to modify.
378         * @param value
379         *            the amount to add to the field.
380         * 
381         * @exception IllegalArgumentException
382         *                when the specified field is DST_OFFSET or ZONE_OFFSET.
383         * @since Android 1.0
384         */
385        @Override
386        public void add(int field, int value) {
387            if (value == 0) {
388                return;
389            }
390            if (field < 0 || field >= ZONE_OFFSET) {
391                throw new IllegalArgumentException();
392            }
393    
394            isCached = false;
395    
396            if (field == ERA) {
397                complete();
398                if (fields[ERA] == AD) {
399                    if (value >= 0) {
400                        return;
401                    }
402                    set(ERA, BC);
403                } else {
404                    if (value <= 0) {
405                        return;
406                    }
407                    set(ERA, AD);
408                }
409                complete();
410                return;
411            }
412    
413            if (field == YEAR || field == MONTH) {
414                complete();
415                if (field == MONTH) {
416                    int month = fields[MONTH] + value;
417                    if (month < 0) {
418                        value = (month - 11) / 12;
419                        month = 12 + (month % 12);
420                    } else {
421                        value = month / 12;
422                    }
423                    set(MONTH, month % 12);
424                }
425                set(YEAR, fields[YEAR] + value);
426                int days = daysInMonth(isLeapYear(fields[YEAR]), fields[MONTH]);
427                if (fields[DATE] > days) {
428                    set(DATE, days);
429                }
430                complete();
431                return;
432            }
433    
434            long multiplier = 0;
435            getTimeInMillis(); // Update the time
436            switch (field) {
437                case MILLISECOND:
438                    time += value;
439                    break;
440                case SECOND:
441                    time += value * 1000L;
442                    break;
443                case MINUTE:
444                    time += value * 60000L;
445                    break;
446                case HOUR:
447                case HOUR_OF_DAY:
448                    time += value * 3600000L;
449                    break;
450                case AM_PM:
451                    multiplier = 43200000L;
452                    break;
453                case DATE:
454                case DAY_OF_YEAR:
455                case DAY_OF_WEEK:
456                    multiplier = 86400000L;
457                    break;
458                case WEEK_OF_YEAR:
459                case WEEK_OF_MONTH:
460                case DAY_OF_WEEK_IN_MONTH:
461                    multiplier = 604800000L;
462                    break;
463            }
464            if (multiplier > 0) {
465                int zoneOffset = getTimeZone().getRawOffset();
466                int offset = getOffset(time + zoneOffset);
467                time += value * multiplier;
468                int newOffset = getOffset(time + zoneOffset);
469                // Adjust for moving over a DST boundary
470                if (newOffset != offset) {
471                    time += offset - newOffset;
472                }
473            }
474            areFieldsSet = false;
475            complete();
476        }

491        private final void fullFieldsCalc(long timeVal, int millis, int zoneOffset) {
492            long days = timeVal / 86400000;
493    
494            if (millis < 0) {
495                millis += 86400000;
496                days--;
497            }
498            // Cannot add ZONE_OFFSET to time as it might overflow
499            millis += zoneOffset;
500            while (millis < 0) {
501                millis += 86400000;
502                days--;
503            }
504            while (millis >= 86400000) {
505                millis -= 86400000;
506                days++;
507            }
508    
509            int dayOfYear = computeYearAndDay(days, timeVal + zoneOffset);
510            fields[DAY_OF_YEAR] = dayOfYear;
511            if(fields[YEAR] == changeYear && gregorianCutover < timeVal + zoneOffset){
512                dayOfYear += currentYearSkew;
513            }
514            int month = dayOfYear / 32;
515            boolean leapYear = isLeapYear(fields[YEAR]);
516            int date = dayOfYear - daysInYear(leapYear, month);
517            if (date > daysInMonth(leapYear, month)) {
518                date -= daysInMonth(leapYear, month);
519                month++;
520            }
521            fields[DAY_OF_WEEK] = mod7(days - 3) + 1;
522            int dstOffset = fields[YEAR] <= 0 ? 0 : getTimeZone().getOffset(AD,
523                    fields[YEAR], month, date, fields[DAY_OF_WEEK], millis);
524            if (fields[YEAR] > 0) {
525                dstOffset -= zoneOffset;
526            }
527            fields[DST_OFFSET] = dstOffset;
528            if (dstOffset != 0) {
529                long oldDays = days;
530                millis += dstOffset;
531                if (millis < 0) {
532                    millis += 86400000;
533                    days--;
534                } else if (millis >= 86400000) {
535                    millis -= 86400000;
536                    days++;
537                }
538                if (oldDays != days) {
539                    dayOfYear = computeYearAndDay(days, timeVal - zoneOffset
540                            + dstOffset);
541                    fields[DAY_OF_YEAR] = dayOfYear;
542                    if(fields[YEAR] == changeYear && gregorianCutover < timeVal - zoneOffset + dstOffset){
543                        dayOfYear += currentYearSkew;
544                    }
545                    month = dayOfYear / 32;
546                    leapYear = isLeapYear(fields[YEAR]);
547                    date = dayOfYear - daysInYear(leapYear, month);
548                    if (date > daysInMonth(leapYear, month)) {
549                        date -= daysInMonth(leapYear, month);
550                        month++;
551                    }
552                    fields[DAY_OF_WEEK] = mod7(days - 3) + 1;
553                }
554            }
555    
556            fields[MILLISECOND] = (millis % 1000);
557            millis /= 1000;
558            fields[SECOND] = (millis % 60);
559            millis /= 60;
560            fields[MINUTE] = (millis % 60);
561            millis /= 60;
562            fields[HOUR_OF_DAY] = (millis % 24);
563            millis /= 24;
564            fields[AM_PM] = fields[HOUR_OF_DAY] > 11 ? 1 : 0;
565            fields[HOUR] = fields[HOUR_OF_DAY] % 12;
566    
567            if (fields[YEAR] <= 0) {
568                fields[ERA] = BC;
569                fields[YEAR] = -fields[YEAR] + 1;
570            } else {
571                fields[ERA] = AD;
572            }
573            fields[MONTH] = month;
574            fields[DATE] = date;
575            fields[DAY_OF_WEEK_IN_MONTH] = (date - 1) / 7 + 1;
576            fields[WEEK_OF_MONTH] = (date - 1 + mod7(days - date - 2
577                    - (getFirstDayOfWeek() - 1))) / 7 + 1;
578            int daysFromStart = mod7(days - 3 - (fields[DAY_OF_YEAR] - 1)
579                    - (getFirstDayOfWeek() - 1));
580            int week = (fields[DAY_OF_YEAR] - 1 + daysFromStart) / 7
581                    + (7 - daysFromStart >= getMinimalDaysInFirstWeek() ? 1 : 0);
582            if (week == 0) {
583                fields[WEEK_OF_YEAR] = 7 - mod7(daysFromStart
584                        - (isLeapYear(fields[YEAR] - 1) ? 2 : 1)) >= getMinimalDaysInFirstWeek() ? 53
585                        : 52;
586            } else if (fields[DAY_OF_YEAR] >= (leapYear ? 367 : 366)
587                    - mod7(daysFromStart + (leapYear ? 2 : 1))) {
588                fields[WEEK_OF_YEAR] = 7 - mod7(daysFromStart + (leapYear ? 2 : 1)) >= getMinimalDaysInFirstWeek() ? 1
589                        : week;
590            } else {
591                fields[WEEK_OF_YEAR] = week;
592            }
593        }

595        private final void cachedFieldsCheckAndGet(long timeVal,
596                long newTimeMillis, long newTimeMillisAdjusted, int millis,
597                int zoneOffset) {
598            int dstOffset = fields[DST_OFFSET];
599            if (!isCached
600                    || newTimeMillis >= nextMidnightMillis
601                    || newTimeMillis <= lastMidnightMillis
602                    || cachedFields[4] != zoneOffset
603                    || (dstOffset == 0 && (newTimeMillisAdjusted >= nextMidnightMillis))
604                    || (dstOffset != 0 && (newTimeMillisAdjusted <= lastMidnightMillis))) {
605                fullFieldsCalc(timeVal, millis, zoneOffset);
606                isCached = false;
607            } else {
608                fields[YEAR] = cachedFields[0];
609                fields[MONTH] = cachedFields[1];
610                fields[DATE] = cachedFields[2];
611                fields[DAY_OF_WEEK] = cachedFields[3];
612                fields[ERA] = cachedFields[5];
613                fields[WEEK_OF_YEAR] = cachedFields[6];
614                fields[WEEK_OF_MONTH] = cachedFields[7];
615                fields[DAY_OF_YEAR] = cachedFields[8];
616                fields[DAY_OF_WEEK_IN_MONTH] = cachedFields[9];
617            }

620        @Override
621        protected void computeFields() {
622            int zoneOffset = getTimeZone().getRawOffset();
623    
624            fields[ZONE_OFFSET] = zoneOffset;
625    
626            int millis = (int) (time % 86400000);
627            int savedMillis = millis;
628            int dstOffset = fields[DST_OFFSET];
629            // compute without a change in daylight saving time
630            int offset = zoneOffset + dstOffset;
631            long newTime = time + offset;
632    
633            if (time > 0L && newTime < 0L && offset > 0) {
634                newTime = 0x7fffffffffffffffL;
635            } else if (time < 0L && newTime > 0L && offset < 0) {
636                newTime = 0x8000000000000000L;
637            }
638    
639            if (isCached) {
640                if (millis < 0) {
641                    millis += 86400000;
642                }
643    
644                // Cannot add ZONE_OFFSET to time as it might overflow
645                millis += zoneOffset;
646                millis += dstOffset;
647    
648                if (millis < 0) {
649                    millis += 86400000;
650                } else if (millis >= 86400000) {
651                    millis -= 86400000;
652                }
653    
654                fields[MILLISECOND] = (millis % 1000);
655                millis /= 1000;
656                fields[SECOND] = (millis % 60);
657                millis /= 60;
658                fields[MINUTE] = (millis % 60);
659                millis /= 60;
660                fields[HOUR_OF_DAY] = (millis % 24);
661                millis /= 24;
662                fields[AM_PM] = fields[HOUR_OF_DAY] > 11 ? 1 : 0;
663                fields[HOUR] = fields[HOUR_OF_DAY] % 12;
664    
665                long newTimeAdjusted = newTime;
666                if (getTimeZone().useDaylightTime()) {
667                    // BEGIN android-changed: removed unnecessary cast
668                    int dstSavings = (/* (SimpleTimeZone) */ getTimeZone())
669                            .getDSTSavings();
670                    // END android-changed
671                    newTimeAdjusted += (dstOffset == 0) ? dstSavings : -dstSavings;
672                }
673    
674                if (newTime > 0L && newTimeAdjusted < 0L && dstOffset == 0) {
675                    newTimeAdjusted = 0x7fffffffffffffffL;
676                } else if (newTime < 0L && newTimeAdjusted > 0L && dstOffset != 0) {
677                    newTimeAdjusted = 0x8000000000000000L;
678                }
679    
680                cachedFieldsCheckAndGet(time, newTime, newTimeAdjusted,
681                        savedMillis, zoneOffset);
682            } else {
683                fullFieldsCalc(time, savedMillis, zoneOffset);
684            }
685    
686            for (int i = 0; i < FIELD_COUNT; i++) {
687                isSet[i] = true;
688            }
689    
690            // Caching
691            if (!isCached
692                    && newTime != 0x7fffffffffffffffL
693                    && newTime != 0x8000000000000000L
694                    && (!getTimeZone().useDaylightTime() || getTimeZone() instanceof SimpleTimeZone)) {
695                int cacheMillis = 0;
696    
697                cachedFields[0] = fields[YEAR];
698                cachedFields[1] = fields[MONTH];
699                cachedFields[2] = fields[DATE];
700                cachedFields[3] = fields[DAY_OF_WEEK];
701                cachedFields[4] = zoneOffset;
702                cachedFields[5] = fields[ERA];
703                cachedFields[6] = fields[WEEK_OF_YEAR];
704                cachedFields[7] = fields[WEEK_OF_MONTH];
705                cachedFields[8] = fields[DAY_OF_YEAR];
706                cachedFields[9] = fields[DAY_OF_WEEK_IN_MONTH];
707    
708                cacheMillis += (23 - fields[HOUR_OF_DAY]) * 60 * 60 * 1000;
709                cacheMillis += (59 - fields[MINUTE]) * 60 * 1000;
710                cacheMillis += (59 - fields[SECOND]) * 1000;
711                nextMidnightMillis = newTime + cacheMillis;
712    
713                cacheMillis = fields[HOUR_OF_DAY] * 60 * 60 * 1000;
714                cacheMillis += fields[MINUTE] * 60 * 1000;
715                cacheMillis += fields[SECOND] * 1000;
716                lastMidnightMillis = newTime - cacheMillis;
717    
718                isCached = true;
719            }
720        }

722        @Override
723        protected void computeTime() {
724            if (!isLenient()) {
725                if (isSet[HOUR_OF_DAY]) {
726                    if (fields[HOUR_OF_DAY] < 0 || fields[HOUR_OF_DAY] > 23) {
727                        throw new IllegalArgumentException();
728                    }
729                } else if (isSet[HOUR] && (fields[HOUR] < 0 || fields[HOUR] > 11)) {
730                    throw new IllegalArgumentException();
731                }
732                if (isSet[MINUTE] && (fields[MINUTE] < 0 || fields[MINUTE] > 59)) {
733                    throw new IllegalArgumentException();
734                }
735                if (isSet[SECOND] && (fields[SECOND] < 0 || fields[SECOND] > 59)) {
736                    throw new IllegalArgumentException();
737                }
738                if (isSet[MILLISECOND]
739                        && (fields[MILLISECOND] < 0 || fields[MILLISECOND] > 999)) {
740                    throw new IllegalArgumentException();
741                }
742                if (isSet[WEEK_OF_YEAR]
743                        && (fields[WEEK_OF_YEAR] < 1 || fields[WEEK_OF_YEAR] > 53)) {
744                    throw new IllegalArgumentException();
745                }
746                if (isSet[DAY_OF_WEEK]
747                        && (fields[DAY_OF_WEEK] < 1 || fields[DAY_OF_WEEK] > 7)) {
748                    throw new IllegalArgumentException();
749                }
750                if (isSet[DAY_OF_WEEK_IN_MONTH]
751                        && (fields[DAY_OF_WEEK_IN_MONTH] < 1 || fields[DAY_OF_WEEK_IN_MONTH] > 6)) {
752                    throw new IllegalArgumentException();
753                }
754                if (isSet[WEEK_OF_MONTH]
755                        && (fields[WEEK_OF_MONTH] < 1 || fields[WEEK_OF_MONTH] > 6)) {
756                    throw new IllegalArgumentException();
757                }
758                if (isSet[AM_PM] && fields[AM_PM] != AM && fields[AM_PM] != PM) {
759                    throw new IllegalArgumentException();
760                }
761                if (isSet[HOUR] && (fields[HOUR] < 0 || fields[HOUR] > 11)) {
762                    throw new IllegalArgumentException();
763                }
764                if (isSet[YEAR]) {
765                    if (isSet[ERA] && fields[ERA] == BC
766                            && (fields[YEAR] < 1 || fields[YEAR] > 292269054)) {
767                        throw new IllegalArgumentException();
768                    } else if (fields[YEAR] < 1 || fields[YEAR] > 292278994) {
769                        throw new IllegalArgumentException();
770                    }
771                }
772                if (isSet[MONTH] && (fields[MONTH] < 0 || fields[MONTH] > 11)) {
773                    throw new IllegalArgumentException();
774                }
775            }
776    
777            long timeVal;
778            long hour = 0;
779            if (isSet[HOUR_OF_DAY] && lastTimeFieldSet != HOUR) {
780                hour = fields[HOUR_OF_DAY];
781            } else if (isSet[HOUR]) {
782                hour = (fields[AM_PM] * 12) + fields[HOUR];
783            }
784            timeVal = hour * 3600000;
785    
786            if (isSet[MINUTE]) {
787                timeVal += ((long) fields[MINUTE]) * 60000;
788            }
789            if (isSet[SECOND]) {
790                timeVal += ((long) fields[SECOND]) * 1000;
791            }
792            if (isSet[MILLISECOND]) {
793                timeVal += fields[MILLISECOND];
794            }
795    
796            long days;
797            int year = isSet[YEAR] ? fields[YEAR] : 1970;
798            if (isSet[ERA]) {
799                // Always test for valid ERA, even if the Calendar is lenient
800                if (fields[ERA] != BC && fields[ERA] != AD) {
801                    throw new IllegalArgumentException();
802                }
803                if (fields[ERA] == BC) {
804                    year = 1 - year;
805                }
806            }
807    
808            boolean weekMonthSet = isSet[WEEK_OF_MONTH]
809                    || isSet[DAY_OF_WEEK_IN_MONTH];
810            boolean useMonth = (isSet[DATE] || isSet[MONTH] || weekMonthSet)
811                    && lastDateFieldSet != DAY_OF_YEAR;
812            if (useMonth
813                    && (lastDateFieldSet == DAY_OF_WEEK || lastDateFieldSet == WEEK_OF_YEAR)) {
814                if (isSet[WEEK_OF_YEAR] && isSet[DAY_OF_WEEK]) {
815                    useMonth = lastDateFieldSet != WEEK_OF_YEAR && weekMonthSet
816                            && isSet[DAY_OF_WEEK];
817                } else if (isSet[DAY_OF_YEAR]) {
818                    useMonth = isSet[DATE] && isSet[MONTH];
819                }
820            }
821    
822            if (useMonth) {
823                int month = fields[MONTH];
824                year += month / 12;
825                month %= 12;
826                if (month < 0) {
827                    year--;
828                    month += 12;
829                }
830                boolean leapYear = isLeapYear(year);
831                days = daysFromBaseYear(year) + daysInYear(leapYear, month);
832                boolean useDate = isSet[DATE];
833                if (useDate
834                        && (lastDateFieldSet == DAY_OF_WEEK
835                                || lastDateFieldSet == WEEK_OF_MONTH || lastDateFieldSet == DAY_OF_WEEK_IN_MONTH)) {
836                    useDate = !(isSet[DAY_OF_WEEK] && weekMonthSet);
837                }
838                if (useDate) {
839                    if (!isLenient()
840                            && (fields[DATE] < 1 || fields[DATE] > daysInMonth(
841                                    leapYear, month))) {
842                        throw new IllegalArgumentException();
843                    }
844                    days += fields[DATE] - 1;
845                } else {
846                    int dayOfWeek;
847                    if (isSet[DAY_OF_WEEK]) {
848                        dayOfWeek = fields[DAY_OF_WEEK] - 1;
849                    } else {
850                        dayOfWeek = getFirstDayOfWeek() - 1;
851                    }
852                    if (isSet[WEEK_OF_MONTH]
853                            && lastDateFieldSet != DAY_OF_WEEK_IN_MONTH) {
854                        int skew = mod7(days - 3 - (getFirstDayOfWeek() - 1));
855                        days += (fields[WEEK_OF_MONTH] - 1) * 7
856                                + mod7(skew + dayOfWeek - (days - 3)) - skew;
857                    } else if (isSet[DAY_OF_WEEK_IN_MONTH]) {
858                        if (fields[DAY_OF_WEEK_IN_MONTH] >= 0) {
859                            days += mod7(dayOfWeek - (days - 3))
860                                    + (fields[DAY_OF_WEEK_IN_MONTH] - 1) * 7;
861                        } else {
862                            days += daysInMonth(leapYear, month)
863                                    + mod7(dayOfWeek
864                                            - (days + daysInMonth(leapYear, month) - 3))
865                                    + fields[DAY_OF_WEEK_IN_MONTH] * 7;
866                        }
867                    }
868                }
869            } else {
870                boolean useWeekYear = isSet[WEEK_OF_YEAR]
871                        && lastDateFieldSet != DAY_OF_YEAR;
872                if (useWeekYear && isSet[DAY_OF_YEAR]) {
873                    useWeekYear = isSet[DAY_OF_WEEK];
874                }
875                days = daysFromBaseYear(year);
876                if (useWeekYear) {
877                    int dayOfWeek;
878                    if (isSet[DAY_OF_WEEK]) {
879                        dayOfWeek = fields[DAY_OF_WEEK] - 1;
880                    } else {
881                        dayOfWeek = getFirstDayOfWeek() - 1;
882                    }
883                    int skew = mod7(days - 3 - (getFirstDayOfWeek() - 1));
884                    days += (fields[WEEK_OF_YEAR] - 1) * 7
885                            + mod7(skew + dayOfWeek - (days - 3)) - skew;
886                    if (7 - skew < getMinimalDaysInFirstWeek()) {
887                        days += 7;
888                    }
889                } else if (isSet[DAY_OF_YEAR]) {
890                    if (!isLenient()
891                            && (fields[DAY_OF_YEAR] < 1 || fields[DAY_OF_YEAR] > (365 + (isLeapYear(year) ? 1
892                                    : 0)))) {
893                        throw new IllegalArgumentException();
894                    }
895                    days += fields[DAY_OF_YEAR] - 1;
896                } else if (isSet[DAY_OF_WEEK]) {
897                    days += mod7(fields[DAY_OF_WEEK] - 1 - (days - 3));
898                }
899            }
900            lastDateFieldSet = 0;
901    
902            timeVal += days * 86400000;
903            // Use local time to compare with the gregorian change
904            if (year == changeYear
905                    && timeVal >= gregorianCutover + julianError() * 86400000L) {
906                timeVal -= julianError() * 86400000L;
907            }
908    
909            // It is not possible to simply subtract getOffset(timeVal) from timeVal
910            // to get UTC.
911            // The trick is needed for the moment when DST transition occurs,
912            // say 1:00 is a transition time when DST offset becomes +1 hour,
913            // then wall time in the interval 1:00 - 2:00 is invalid and is
914            // treated as UTC time.
915            long timeValWithoutDST = timeVal - getOffset(timeVal)
916                    + getTimeZone().getRawOffset();
917            timeVal -= getOffset(timeValWithoutDST);
918            // Need to update wall time in fields, since it was invalid due to DST
919            // transition
920            this.time = timeVal;
921            if (timeValWithoutDST != timeVal) {
922                computeFields();
923                areFieldsSet = true;
924            }
925        }

927        private int computeYearAndDay(long dayCount, long localTime) {
928            int year = 1970;
929            long days = dayCount;
930            if (localTime < gregorianCutover) {
931                days -= julianSkew;
932            }
933            int approxYears;
934    
935            while ((approxYears = (int) (days / 365)) != 0) {
936                year = year + approxYears;
937                days = dayCount - daysFromBaseYear(year);
938            }
939            if (days < 0) {
940                year = year - 1;
941                days = days + daysInYear(year);
942            }
943            fields[YEAR] = year;
944            return (int) days + 1;
945        }

947        private long daysFromBaseYear(int iyear) {
948            long year = iyear;
949    
950            if (year >= 1970) {
951                long days = (year - 1970) * 365 + ((year - 1969) / 4);
952                if (year > changeYear) {
953                    days -= ((year - 1901) / 100) - ((year - 1601) / 400);
954                } else {
955                    if(year == changeYear){
956                        days += currentYearSkew;
957                    }else if(year == changeYear -1){
958                        days += lastYearSkew;
959                    }else{
960                        days += julianSkew;
961                    }
962                }
963                return days;
964            } else if (year <= changeYear) {
965                return (year - 1970) * 365 + ((year - 1972) / 4) + julianSkew;
966            }
967            return (year - 1970) * 365 + ((year - 1972) / 4)
968                    - ((year - 2000) / 100) + ((year - 2000) / 400);
969        }

976            if (leapYear && month == FEBRUARY) {
977                return DaysInMonth[month] + 1;
978            }
979    
980            return DaysInMonth[month];

983        private int daysInYear(int year) {
984            int daysInYear = isLeapYear(year) ? 366 : 365;
985            if (year == changeYear) {
986                daysInYear -= currentYearSkew;
987            }
988            if (year == changeYear - 1) {
989                daysInYear -= lastYearSkew;
990            }
991            return daysInYear;
992        }

995            if (leapYear && month > FEBRUARY) {
996                return DaysInYear[month] + 1;
997            }
998    
999            return DaysInYear[month];
1000        }

1024         * Gets the maximum value of the specified field for the current date. For
1025         * example, the maximum number of days in the current month.

1028         *            the field.
1029         * @return the maximum value of the specified field.

1032        @Override
1033        public int getActualMaximum(int field) {
1034            int value;
1035            if ((value = maximums[field]) == leastMaximums[field]) {
1036                return value;
1037            }
1038    
1039            switch (field) {
1040                case WEEK_OF_YEAR:
1041                case WEEK_OF_MONTH:
1042                    isCached = false;
1043                    break;
1044            }
1045    
1046            complete();
1047            long orgTime = time;
1048            int result = 0;

1060                case WEEK_OF_MONTH:

1067                case DAY_OF_YEAR:
1068                    return daysInYear(fields[YEAR]);

1083                        result--;
1084                    }
1085                    break;
1086                case DST_OFFSET:
1087                    result = getMaximum(DST_OFFSET);
1088                    break;
1089            }
1090            time = orgTime;

1092        }

1185        int getOffset(long localTime) {
1186            TimeZone timeZone = getTimeZone();
1187            if (!timeZone.useDaylightTime()) {
1188                return timeZone.getRawOffset();
1189            }
1190    
1191            long dayCount = localTime / 86400000;
1192            int millis = (int) (localTime % 86400000);
1193            if (millis < 0) {
1194                millis += 86400000;
1195                dayCount--;
1196            }
1197    
1198            int year = 1970;
1199            long days = dayCount;
1200            if (localTime < gregorianCutover) {
1201                days -= julianSkew;
1202            }
1203            int approxYears;
1204    
1205            while ((approxYears = (int) (days / 365)) != 0) {
1206                year = year + approxYears;
1207                days = dayCount - daysFromBaseYear(year);
1208            }
1209            if (days < 0) {
1210                year = year - 1;
1211                days = days + 365 + (isLeapYear(year) ? 1 : 0);
1212                if (year == changeYear && localTime < gregorianCutover) {
1213                    days -= julianError();
1214                }
1215            }
1216            if (year <= 0) {
1217                return timeZone.getRawOffset();
1218            }
1219            int dayOfYear = (int) days + 1;
1220    
1221            int month = dayOfYear / 32;
1222            boolean leapYear = isLeapYear(year);
1223            int date = dayOfYear - daysInYear(leapYear, month);
1224            if (date > daysInMonth(leapYear, month)) {
1225                date -= daysInMonth(leapYear, month);
1226                month++;
1227            }
1228            int dayOfWeek = mod7(dayCount - 3) + 1;
1229            int offset = timeZone.getOffset(AD, year, month, date, dayOfWeek,
1230                    millis);
1231            return offset;
1232        }

1260                return year % 4 == 0 && (year % 100 != 0 || year % 400 == 0);
1261            }

1271            int rem = value % mod;
1272            if (value < 0 && rem < 0) {
1273                return rem + mod;

1286        /**
1287         * Adds the specified amount the specified field and wraps the value of the
1288         * field when it goes beyond the maximum or minimum value for the current
1289         * date. Other fields will be adjusted as required to maintain a consistent
1290         * date.
1291         * 
1292         * @param field
1293         *            the field to roll.
1294         * @param value
1295         *            the amount to add.
1296         * 
1297         * @exception IllegalArgumentException
1298         *                when an invalid field is specified.
1299         * @since Android 1.0
1300         */
1301        @Override
1302        public void roll(int field, int value) {
1303            if (value == 0) {
1304                return;
1305            }
1306            if (field < 0 || field >= ZONE_OFFSET) {
1307                throw new IllegalArgumentException();
1308            }
1309    
1310            isCached = false;
1311    
1312            complete();
1313            int max = -1;
1314            switch (field) {
1315                case YEAR:
1316                    max = maximums[field];
1317                    break;
1318                case WEEK_OF_YEAR:
1319                case WEEK_OF_MONTH:
1320                    int days,
1321                    day;
1322                    if (field == WEEK_OF_YEAR) {
1323                        days = daysInYear(fields[YEAR]);
1324                        day = DAY_OF_YEAR;
1325                    } else {
1326                        days = daysInMonth();
1327                        day = DATE;
1328                    }
1329                    int mod = mod7(fields[DAY_OF_WEEK] - fields[day]
1330                            - (getFirstDayOfWeek() - 1));
1331                    int maxWeeks = (days - 1 + mod) / 7 + 1;
1332                    int newWeek = mod(fields[field] - 1 + value, maxWeeks) + 1;
1333                    if (newWeek == maxWeeks) {
1334                        if (fields[day] + (newWeek - fields[field]) * 7 > days) {
1335                            set(day, days);
1336                        } else {
1337                            set(field, newWeek);
1338                        }
1339                    } else if (newWeek == 1) {
1340                        int week = (fields[day] - ((fields[day] - 1) / 7 * 7) - 1 + mod) / 7 + 1;
1341                        if (week > 1) {
1342                            set(day, 1);
1343                        } else {
1344                            set(field, newWeek);
1345                        }
1346                    } else {
1347                        set(field, newWeek);
1348                    }
1349                    break;
1350                case DATE:
1351                    max = daysInMonth();
1352                    break;
1353                case DAY_OF_YEAR:
1354                    max = daysInYear(fields[YEAR]);
1355                    break;
1356                case DAY_OF_WEEK:
1357                    max = maximums[field];
1358                    lastDateFieldSet = WEEK_OF_MONTH;
1359                    break;
1360                case DAY_OF_WEEK_IN_MONTH:
1361                    max = (fields[DATE] + ((daysInMonth() - fields[DATE]) / 7 * 7) - 1) / 7 + 1;
1362                    break;
1363    
1364                case ERA:
1365                case MONTH:
1366                case AM_PM:
1367                case HOUR:
1368                case HOUR_OF_DAY:
1369                case MINUTE:
1370                case SECOND:
1371                case MILLISECOND:
1372                    set(field, mod(fields[field] + value, maximums[field] + 1));
1373                    if (field == MONTH && fields[DATE] > daysInMonth()) {
1374                        set(DATE, daysInMonth());
1375                    } else if (field == AM_PM) {
1376                        lastTimeFieldSet = HOUR;
1377                    }
1378                    break;
1379            }
1380            if (max != -1) {
1381                set(field, mod(fields[field] - 1 + value, max) + 1);
1382            }
1383            complete();
1384        }

1387         * Increments or decrements the specified field and wraps the value of the
1388         * field when it goes beyond the maximum or minimum value for the current
1389         * date. Other fields will be adjusted as required to maintain a consistent
1390         * date. For example, March 31 will roll to April 30 when rolling the month
1391         * field.
1392         * 
1393         * @param field
1394         *            the field to roll.
1395         * @param increment
1396         *            {@code true} to increment the field, {@code false} to
1397         *            decrement.
1398         * @exception IllegalArgumentException
1399         *                when an invalid field is specified.
1400         * @since Android 1.0
1401         */

1407        /**
1408         * Sets the gregorian change date of this calendar.
1409         * 
1410         * @param date
1411         *            a {@code Date} which represents the gregorian change date.
1412         * @since Android 1.0
1413         */
1414        public void setGregorianChange(Date date) {
1415            gregorianCutover = date.getTime();
1416            GregorianCalendar cal = new GregorianCalendar(TimeZone.GMT);
1417            cal.setTime(date);
1418            changeYear = cal.get(YEAR);
1419            if (cal.get(ERA) == BC) {
1420                changeYear = 1 - changeYear;
1421            }
1422            julianSkew = ((changeYear - 2000) / 400) + julianError()
1423                    - ((changeYear - 2000) / 100);
1424            isCached = false;
1425            int dayOfYear = cal.get(DAY_OF_YEAR);
1426            if (dayOfYear < julianSkew) {
1427                currentYearSkew = dayOfYear-1;
1428                lastYearSkew = julianSkew - dayOfYear + 1;
1429            } else {
1430                lastYearSkew = 0;
1431                currentYearSkew = julianSkew;
1432            }
1433            isCached = false;
1434        }
