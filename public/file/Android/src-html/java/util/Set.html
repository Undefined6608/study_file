018    package java.util;
019    
020    
021    /**
022     * A {@code Set} is a data structure which does not allow duplicate elements.
023     * 
024     * @since Android 1.0
025     */
026    public interface Set<E> extends Collection<E> {
027        
028        /**
029         * Adds the specified object to this set. The set is not modified if it
030         * already contains the object.
031         * 
032         * @param object
033         *            the object to add.
034         * @return {@code true} if this set is modified, {@code false} otherwise.
035         * @throws UnsupportedOperationException
036         *             when adding to this set is not supported.
037         * @throws ClassCastException
038         *             when the class of the object is inappropriate for this set.
039         * @throws IllegalArgumentException
040         *             when the object cannot be added to this set.
041         * @since Android 1.0
042         */
043        public boolean add(E object);
044    
045        /**
046         * Adds the objects in the specified collection which do not exist yet in
047         * this set.
048         * 
049         * @param collection
050         *            the collection of objects.
051         * @return {@code true} if this set is modified, {@code false} otherwise.
052         * @throws UnsupportedOperationException
053         *             when adding to this set is not supported.
054         * @throws ClassCastException
055         *             when the class of an object is inappropriate for this set.
056         * @throws IllegalArgumentException
057         *             when an object cannot be added to this set.
058         * @since Android 1.0
059         */
060        public boolean addAll(Collection<? extends E> collection);
061    
062        /**
063         * Removes all elements from this set, leaving it empty.
064         * 
065         * @throws UnsupportedOperationException
066         *             when removing from this set is not supported.
067         * @see #isEmpty
068         * @see #size
069         * @since Android 1.0
070         */
071        public void clear();
072    
073        /**
074         * Searches this set for the specified object.
075         * 
076         * @param object
077         *            the object to search for.
078         * @return {@code true} if object is an element of this set, {@code false}
079         *         otherwise.
080         * @since Android 1.0
081         */
082        public boolean contains(Object object);
083    
084        /**
085         * Searches this set for all objects in the specified collection.
086         * 
087         * @param collection
088         *            the collection of objects.
089         * @return {@code true} if all objects in the specified collection are
090         *         elements of this set, {@code false} otherwise.
091         * @since Android 1.0
092         */
093        public boolean containsAll(Collection<?> collection);
094    
095        /**
096         * Compares the specified object to this set, and returns true if they
097         * represent the <em>same</em> object using a class specific comparison.
098         * Equality for a set means that both sets have the same size and the same
099         * elements.
100         * 
101         * @param object
102         *            the object to compare with this object.
103         * @return boolean {@code true} if the object is the same as this object,
104         *         and {@code false} if it is different from this object.
105         * @see #hashCode
106         * @since Android 1.0
107         */
108        public boolean equals(Object object);
109    
110        /**
111         * Returns the hash code for this set. Two set which are equal must return
112         * the same value.
113         * 
114         * @return the hash code of this set.
115         * 
116         * @see #equals
117         * @since Android 1.0
118         */
119        public int hashCode();
120    
121        /**
122         * Returns true if this set has no elements.
123         * 
124         * @return {@code true} if this set has no elements, {@code false}
125         *         otherwise.
126         * @see #size
127         * @since Android 1.0
128         */
129        public boolean isEmpty();
130    
131        /**
132         * Returns an iterator on the elements of this set. The elements are
133         * unordered.
134         * 
135         * @return an iterator on the elements of this set.
136         * @see Iterator
137         * @since Android 1.0
138         */
139        public Iterator<E> iterator();
140    
141        /**
142         * Removes the specified object from this set.
143         * 
144         * @param object
145         *            the object to remove.
146         * @return {@code true} if this set was modified, {@code false} otherwise.
147         * @throws UnsupportedOperationException
148         *             when removing from this set is not supported.
149         * @since Android 1.0
150         */
151        public boolean remove(Object object);
152    
153        /**
154         * Removes all objects in the specified collection from this set.
155         * 
156         * @param collection
157         *            the collection of objects to remove.
158         * @return {@code true} if this set was modified, {@code false} otherwise.
159         * @throws UnsupportedOperationException
160         *             when removing from this set is not supported.
161         * @since Android 1.0
162         */
163        public boolean removeAll(Collection<?> collection);
164    
165        /**
166         * Removes all objects from this set that are not contained in the specified
167         * collection.
168         * 
169         * @param collection
170         *            the collection of objects to retain.
171         * @return {@code true} if this set was modified, {@code false} otherwise.
172         * @throws UnsupportedOperationException
173         *             when removing from this set is not supported.
174         * @since Android 1.0
175         */
176        public boolean retainAll(Collection<?> collection);
177    
178        /**
179         * Returns the number of elements in this set.
180         * 
181         * @return the number of elements in this set.
182         * @since Android 1.0
183         */
184        public int size();
185    
186        /**
187         * Returns an array containing all elements contained in this set.
188         * 
189         * @return an array of the elements from this set.
190         * @since Android 1.0
191         */
192        public Object[] toArray();
193    
194        /**
195         * Returns an array containing all elements contained in this set. If the
196         * specified array is large enough to hold the elements, the specified array
197         * is used, otherwise an array of the same type is created. If the specified
198         * array is used and is larger than this set, the array element following
199         * the collection elements is set to null.
200         * 
201         * @param array
202         *            the array.
203         * @return an array of the elements from this set.
204         * @throws ArrayStoreException
205         *             when the type of an element in this set cannot be stored in
206         *             the type of the specified array.
207         * @see Collection#toArray(Object[])
208         * @since Android 1.0
209         */
210        public <T> T[] toArray(T[] array);
211    }
