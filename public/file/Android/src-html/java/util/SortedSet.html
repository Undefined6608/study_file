018    package java.util;
019    
020    
021    /**
022     * SortedSet is a Set which iterates over its elements in a sorted order. The
023     * order is determined either by the elements natural ordering, or by a
024     * {@link Comparator} which is passed into a concrete implementation at
025     * construction time. All elements in this set must be mutually comparable. The
026     * ordering in this set must be consistent with {@code equals} of its elements.
027     * 
028     * @see Comparator
029     * @see Comparable
030     * @since Android 1.0
031     */
032    public interface SortedSet<E> extends Set<E> {
033        
034        /**
035         * Returns the comparator used to compare elements in this {@code SortedSet}.
036         * 
037         * @return a comparator or null if the natural ordering is used.
038         * @since Android 1.0
039         */
040        public Comparator<? super E> comparator();
041    
042        /**
043         * Returns the first element in this {@code SortedSet}. The first element
044         * is the lowest element.
045         * 
046         * @return the first element.
047         * @throws NoSuchElementException
048         *             when this {@code SortedSet} is empty.
049         * @since Android 1.0
050         */
051        public E first();
052    
053        /**
054         * Returns a {@code SortedSet} of the specified portion of this
055         * {@code SortedSet} which contains elements less than the end element. The
056         * returned {@code SortedSet} is backed by this {@code SortedSet} so changes
057         * to one set are reflected by the other.
058         * 
059         * @param end
060         *            the end element.
061         * @return a subset where the elements are less than {@code end}.
062         * @throws ClassCastException
063         *             when the class of the end element is inappropriate for this
064         *             SubSet.
065         * @throws NullPointerException
066         *             when the end element is null and this {@code SortedSet} does
067         *             not support null elements.
068         * @since Android 1.0
069         */
070        public SortedSet<E> headSet(E end);
071    
072        /**
073         * Returns the last element in this {@code SortedSet}. The last element is
074         * the highest element.
075         * 
076         * @return the last element.
077         * @throws NoSuchElementException
078         *             when this {@code SortedSet} is empty.
079         * @since Android 1.0
080         */
081        public E last();
082    
083        /**
084         * Returns a {@code SortedSet} of the specified portion of this
085         * {@code SortedSet} which contains elements greater or equal to the start
086         * element but less than the end element. The returned {@code SortedSet} is
087         * backed by this SortedMap so changes to one set are reflected by the
088         * other.
089         * 
090         * @param start
091         *            the start element.
092         * @param end
093         *            the end element.
094         * @return a subset where the elements are greater or equal to {@code start}
095         *         and less than {@code end}.
096         * @throws ClassCastException
097         *             when the class of the start or end element is inappropriate
098         *             for this SubSet.
099         * @throws NullPointerException
100         *             when the start or end element is null and this
101         *             {@code SortedSet} does not support null elements.
102         * @throws IllegalArgumentException
103         *             when the start element is greater than the end element.
104         * @since Android 1.0
105         */
106        public SortedSet<E> subSet(E start, E end);
107    
108        /**
109         * Returns a {@code SortedSet} of the specified portion of this
110         * {@code SortedSet} which contains elements greater or equal to the start
111         * element. The returned {@code SortedSet} is backed by this
112         * {@code SortedSet} so changes to one set are reflected by the other.
113         * 
114         * @param start
115         *            the start element.
116         * @return a subset where the elements are greater or equal to {@code start} .
117         * @throws ClassCastException
118         *             when the class of the start element is inappropriate for this
119         *             SubSet.
120         * @throws NullPointerException
121         *             when the start element is null and this {@code SortedSet}
122         *             does not support null elements.
123         * @since Android 1.0
124         */
125        public SortedSet<E> tailSet(E start);
126    }
