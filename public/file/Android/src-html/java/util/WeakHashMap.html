018    package java.util;
019    
020    import java.lang.ref.ReferenceQueue;
021    import java.lang.ref.WeakReference;
022    
023    /**
024     * WeakHashMap is an implementation of Map with keys which are WeakReferences. A
025     * key/value mapping is removed when the key is no longer referenced. All
026     * optional operations (adding and removing) are supported. Keys and values can
027     * be any objects. Note that the garbage collector acts similar to a second
028     * thread on this collection, possibly removing keys.
029     * 
030     * @see HashMap
031     * @see WeakReference
032     * @since Android 1.0
033     */
034    public class WeakHashMap<K, V> extends AbstractMap<K, V> implements Map<K, V> {
035    
036        private static final int DEFAULT_SIZE = 16;
037    
038        private final ReferenceQueue<K> referenceQueue;
039    
040        int elementCount;
041    
042        Entry<K, V>[] elementData;
043    
044        private final int loadFactor;
045    
046        private int threshold;
047    
048        volatile int modCount;
049    
050        // Simple utility method to isolate unchecked cast for array creation
051        @SuppressWarnings("unchecked")
052        private static <K, V> Entry<K, V>[] newEntryArray(int size) {
053            return new Entry[size];
054        }
055    
056        private static final class Entry<K, V> extends WeakReference<K> implements
057                Map.Entry<K, V> {
058            int hash;
059    
060            boolean isNull;
061    
062            V value;
063    
064            Entry<K, V> next;
065    
066            interface Type<R, K, V> {
067                R get(Map.Entry<K, V> entry);
068            }
069    
070            Entry(K key, V object, ReferenceQueue<K> queue) {
071                super(key, queue);
072                isNull = key == null;
073                hash = isNull ? 0 : key.hashCode();
074                value = object;
075            }
076    
077            public K getKey() {
078                return super.get();
079            }
080    
081            public V getValue() {
082                return value;
083            }
084    
085            public V setValue(V object) {
086                V result = value;
087                value = object;
088                return result;
089            }
090    
091            @Override
092            public boolean equals(Object other) {
093                if (!(other instanceof Map.Entry)) {
094                    return false;
095                }
096                Map.Entry<?, ?> entry = (Map.Entry<?, ?>) other;
097                Object key = super.get();
098                return (key == null ? key == entry.getKey() : key.equals(entry
099                        .getKey()))
100                        && (value == null ? value == entry.getValue() : value
101                                .equals(entry.getValue()));
102            }
103    
104            @Override
105            public int hashCode() {
106                return hash + (value == null ? 0 : value.hashCode());
107            }
108    
109            @Override
110            public String toString() {
111                return super.get() + "=" + value; //$NON-NLS-1$
112            }
113        }
114    
115        class HashIterator<R> implements Iterator<R> {
116            private int position = 0, expectedModCount;
117    
118            private Entry<K, V> currentEntry, nextEntry;
119    
120            private K nextKey;
121    
122            final Entry.Type<R, K, V> type;
123    
124            HashIterator(Entry.Type<R, K, V> type) {
125                this.type = type;
126                expectedModCount = modCount;
127            }
128    
129            public boolean hasNext() {
130                if (nextEntry != null) {
131                    return true;
132                }
133                while (true) {
134                    if (nextEntry == null) {
135                        while (position < elementData.length) {
136                            if ((nextEntry = elementData[position++]) != null) {
137                                break;
138                            }
139                        }
140                        if (nextEntry == null) {
141                            return false;
142                        }
143                    }
144                    // ensure key of next entry is not gc'ed
145                    nextKey = nextEntry.get();
146                    if (nextKey != null || nextEntry.isNull) {
147                        return true;
148                    }
149                    nextEntry = nextEntry.next;
150                }
151            }
152    
153            public R next() {
154                if (expectedModCount == modCount) {
155                    if (hasNext()) {
156                        currentEntry = nextEntry;
157                        nextEntry = currentEntry.next;
158                        R result = type.get(currentEntry);
159                        // free the key
160                        nextKey = null;
161                        return result;
162                    }
163                    throw new NoSuchElementException();
164                }
165                throw new ConcurrentModificationException();
166            }
167    
168            public void remove() {
169                if (expectedModCount == modCount) {
170                    if (currentEntry != null) {
171                        removeEntry(currentEntry);
172                        currentEntry = null;
173                        expectedModCount++;
174                        // cannot poll() as that would change the expectedModCount
175                    } else {
176                        throw new IllegalStateException();
177                    }
178                } else {
179                    throw new ConcurrentModificationException();
180                }
181            }
182        }

184        /**
185         * Constructs a new empty {@code WeakHashMap} instance.
186         * 
187         * @since Android 1.0
188         */
189        public WeakHashMap() {
190            this(DEFAULT_SIZE);
191        }
192    
193        /**
194         * Constructs a new {@code WeakHashMap} instance with the specified
195         * capacity.
196         * 
197         * @param capacity
198         *            the initial capacity of this map.
199         * @throws IllegalArgumentException
200         *                if the capacity is less than zero.
201         * @since Android 1.0
202         */
203        public WeakHashMap(int capacity) {
204            if (capacity >= 0) {
205                elementCount = 0;
206                elementData = newEntryArray(capacity == 0 ? 1 : capacity);
207                loadFactor = 7500; // Default load factor of 0.75
208                computeMaxSize();
209                referenceQueue = new ReferenceQueue<K>();
210            } else {
211                throw new IllegalArgumentException();
212            }
213        }
214    
215        /**
216         * Constructs a new {@code WeakHashMap} instance with the specified capacity
217         * and load factor.
218         * 
219         * @param capacity
220         *            the initial capacity of this map.
221         * @param loadFactor
222         *            the initial load factor.
223         * @throws IllegalArgumentException
224         *             if the capacity is less than zero or the load factor is less
225         *             or equal to zero.
226         * @since Android 1.0
227         */
228        public WeakHashMap(int capacity, float loadFactor) {
229            if (capacity >= 0 && loadFactor > 0) {
230                elementCount = 0;
231                elementData = newEntryArray(capacity == 0 ? 1 : capacity);
232                this.loadFactor = (int) (loadFactor * 10000);
233                computeMaxSize();
234                referenceQueue = new ReferenceQueue<K>();
235            } else {
236                throw new IllegalArgumentException();
237            }
238        }
239    
240        /**
241         * Constructs a new {@code WeakHashMap} instance containing the mappings
242         * from the specified map.
243         * 
244         * @param map
245         *            the mappings to add.
246         * @since Android 1.0
247         */
248        public WeakHashMap(Map<? extends K, ? extends V> map) {
249            this(map.size() < 6 ? 11 : map.size() * 2);
250            putAllImpl(map);
251        }
252    
253        /**
254         * Removes all mappings from this map, leaving it empty.
255         * 
256         * @see #isEmpty()
257         * @see #size()
258         * @since Android 1.0
259         */
260        @Override
261        public void clear() {
262            if (elementCount > 0) {
263                elementCount = 0;
264                Arrays.fill(elementData, null);
265                modCount++;
266                while (referenceQueue.poll() != null) {
267                    // do nothing
268                }
269            }
270        }
271    
272        private void computeMaxSize() {
273            threshold = (int) ((long) elementData.length * loadFactor / 10000);
274        }

276        /**
277         * Returns whether this map contains the specified key.
278         * 
279         * @param key
280         *            the key to search for.
281         * @return {@code true} if this map contains the specified key,
282         *         {@code false} otherwise.
283         * @since Android 1.0
284         */
285        @Override
286        public boolean containsKey(Object key) {
287            return getEntry(key) != null;
288        }
289    
290        /**
291         * Returns a set containing all of the mappings in this map. Each mapping is
292         * an instance of {@link Map.Entry}. As the set is backed by this map,
293         * changes in one will be reflected in the other. It does not support adding
294         * operations.
295         * 
296         * @return a set of the mappings.
297         * @since Android 1.0
298         */
299        @Override
300        public Set<Map.Entry<K, V>> entrySet() {
301            poll();
302            return new AbstractSet<Map.Entry<K, V>>() {
303                @Override
304                public int size() {
305                    return WeakHashMap.this.size();
306                }
307    
308                @Override
309                public void clear() {
310                    WeakHashMap.this.clear();
311                }
312    
313                @Override
314                public boolean remove(Object object) {
315                    if (contains(object)) {
316                        WeakHashMap.this
317                                .remove(((Map.Entry<?, ?>) object).getKey());
318                        return true;
319                    }
320                    return false;
321                }
322    
323                @Override
324                public boolean contains(Object object) {
325                    if (object instanceof Map.Entry) {
326                        Entry<?, ?> entry = getEntry(((Map.Entry<?, ?>) object)
327                                .getKey());
328                        if (entry != null) {
329                            Object key = entry.get();
330                            if (key != null || entry.isNull) {
331                                return object.equals(entry);
332                            }
333                        }
334                    }
335                    return false;
336                }
337    
338                @Override
339                public Iterator<Map.Entry<K, V>> iterator() {
340                    return new HashIterator<Map.Entry<K, V>>(
341                            new Entry.Type<Map.Entry<K, V>, K, V>() {
342                                public Map.Entry<K, V> get(Map.Entry<K, V> entry) {
343                                    return entry;
344                                }
345                            });
346                }
347            };
348        }

350        /**
351         * Returns a set of the keys contained in this map. The set is backed by
352         * this map so changes to one are reflected by the other. The set does not
353         * support adding.
354         * 
355         * @return a set of the keys.
356         * @since Android 1.0
357         */
358        @Override
359        public Set<K> keySet() {
360            poll();
361            if (keySet == null) {
362                keySet = new AbstractSet<K>() {
363                    @Override
364                    public boolean contains(Object object) {
365                        return containsKey(object);
366                    }
367    
368                    @Override
369                    public int size() {
370                        return WeakHashMap.this.size();
371                    }
372    
373                    @Override
374                    public void clear() {
375                        WeakHashMap.this.clear();
376                    }
377    
378                    @Override
379                    public boolean remove(Object key) {
380                        if (containsKey(key)) {
381                            WeakHashMap.this.remove(key);
382                            return true;
383                        }
384                        return false;
385                    }
386    
387                    @Override
388                    public Iterator<K> iterator() {
389                        return new HashIterator<K>(new Entry.Type<K, K, V>() {
390                            public K get(Map.Entry<K, V> entry) {
391                                return entry.getKey();
392                            }
393                        });
394                    }
395                };
396            }
397            return keySet;
398        }

400        /**
401         * Returns a collection of the values contained in this map. The collection
402         * is backed by this map so changes to one are reflected by the other. The
403         * collection supports remove, removeAll, retainAll and clear operations,
404         * and it does not support add or addAll operations.
405         * <p>
406         * This method returns a collection which is the subclass of
407         * AbstractCollection. The iterator method of this subclass returns a
408         * "wrapper object" over the iterator of map's entrySet(). The size method
409         * wraps the map's size method and the contains method wraps the map's
410         * containsValue method.
411         * </p>
412         * <p>
413         * The collection is created when this method is called at first time and
414         * returned in response to all subsequent calls. This method may return
415         * different Collection when multiple calls to this method, since it has no
416         * synchronization performed.
417         * </p>
418         * 
419         * @return a collection of the values contained in this map.
420         * @since Android 1.0
421         */
422        @Override
423        public Collection<V> values() {
424            poll();
425            if (valuesCollection == null) {
426                valuesCollection = new AbstractCollection<V>() {
427                    @Override
428                    public int size() {
429                        return WeakHashMap.this.size();
430                    }
431    
432                    @Override
433                    public void clear() {
434                        WeakHashMap.this.clear();
435                    }
436    
437                    @Override
438                    public boolean contains(Object object) {
439                        return containsValue(object);
440                    }
441    
442                    @Override
443                    public Iterator<V> iterator() {
444                        return new HashIterator<V>(new Entry.Type<V, K, V>() {
445                            public V get(Map.Entry<K, V> entry) {
446                                return entry.getValue();
447                            }
448                        });
449                    }
450                };
451            }
452            return valuesCollection;
453        }

455        /**
456         * Returns the value of the mapping with the specified key.
457         * 
458         * @param key
459         *            the key.
460         * @return the value of the mapping with the specified key, or {@code null}
461         *         if no mapping for the specified key is found.
462         * @since Android 1.0
463         */
464        @Override
465        public V get(Object key) {
466            poll();
467            if (key != null) {
468                int index = (key.hashCode() & 0x7FFFFFFF) % elementData.length;
469                Entry<K, V> entry = elementData[index];
470                while (entry != null) {
471                    if (key.equals(entry.get())) {
472                        return entry.value;
473                    }
474                    entry = entry.next;
475                }
476                return null;
477            }
478            Entry<K, V> entry = elementData[0];
479            while (entry != null) {
480                if (entry.isNull) {
481                    return entry.value;
482                }
483                entry = entry.next;
484            }
485            return null;
486        }
487    
488        Entry<K, V> getEntry(Object key) {
489            poll();
490            if (key != null) {
491                int index = (key.hashCode() & 0x7FFFFFFF) % elementData.length;
492                Entry<K, V> entry = elementData[index];
493                while (entry != null) {
494                    if (key.equals(entry.get())) {
495                        return entry;
496                    }
497                    entry = entry.next;
498                }
499                return null;
500            }
501            Entry<K, V> entry = elementData[0];
502            while (entry != null) {
503                if (entry.isNull) {
504                    return entry;
505                }
506                entry = entry.next;
507            }
508            return null;
509        }

511        /**
512         * Returns whether this map contains the specified value.
513         * 
514         * @param value
515         *            the value to search for.
516         * @return {@code true} if this map contains the specified value,
517         *         {@code false} otherwise.
518         * @since Android 1.0
519         */
520        @Override
521        public boolean containsValue(Object value) {
522            poll();
523            if (value != null) {
524                for (int i = elementData.length; --i >= 0;) {
525                    Entry<K, V> entry = elementData[i];
526                    while (entry != null) {
527                        K key = entry.get();
528                        if ((key != null || entry.isNull)
529                                && value.equals(entry.value)) {
530                            return true;
531                        }
532                        entry = entry.next;
533                    }
534                }
535            } else {
536                for (int i = elementData.length; --i >= 0;) {
537                    Entry<K, V> entry = elementData[i];
538                    while (entry != null) {
539                        K key = entry.get();
540                        if ((key != null || entry.isNull) && entry.value == null) {
541                            return true;
542                        }
543                        entry = entry.next;
544                    }
545                }
546            }
547            return false;
548        }

561        @SuppressWarnings("unchecked")
562        void poll() {
563            Entry<K, V> toRemove;
564            while ((toRemove = (Entry<K, V>) referenceQueue.poll()) != null) {
565                removeEntry(toRemove);
566            }
567        }
568    
569        void removeEntry(Entry<K, V> toRemove) {
570            Entry<K, V> entry, last = null;
571            int index = (toRemove.hash & 0x7FFFFFFF) % elementData.length;
572            entry = elementData[index];
573            // Ignore queued entries which cannot be found, the user could
574            // have removed them before they were queued, i.e. using clear()
575            while (entry != null) {
576                if (toRemove == entry) {
577                    modCount++;
578                    if (last == null) {
579                        elementData[index] = entry.next;
580                    } else {
581                        last.next = entry.next;
582                    }
583                    elementCount--;
584                    break;
585                }
586                last = entry;
587                entry = entry.next;
588            }
589        }

591        /**
592         * Maps the specified key to the specified value.
593         * 
594         * @param key
595         *            the key.
596         * @param value
597         *            the value.
598         * @return the value of any previous mapping with the specified key or
599         *         {@code null} if there was no mapping.
600         * @since Android 1.0
601         */
602        @Override
603        public V put(K key, V value) {
604            poll();
605            int index = 0;
606            Entry<K, V> entry;
607            if (key != null) {
608                index = (key.hashCode() & 0x7FFFFFFF) % elementData.length;
609                entry = elementData[index];
610                while (entry != null && !key.equals(entry.get())) {
611                    entry = entry.next;
612                }
613            } else {
614                entry = elementData[0];
615                while (entry != null && !entry.isNull) {
616                    entry = entry.next;
617                }
618            }
619            if (entry == null) {
620                modCount++;
621                if (++elementCount > threshold) {
622                    rehash();
623                    index = key == null ? 0 : (key.hashCode() & 0x7FFFFFFF)
624                            % elementData.length;
625                }
626                entry = new Entry<K, V>(key, value, referenceQueue);
627                entry.next = elementData[index];
628                elementData[index] = entry;
629                return null;
630            }
631            V result = entry.value;
632            entry.value = value;
633            return result;
634        }
635    
636        private void rehash() {
637            int length = elementData.length << 1;
638            if (length == 0) {
639                length = 1;
640            }
641            Entry<K, V>[] newData = newEntryArray(length);
642            for (int i = 0; i < elementData.length; i++) {
643                Entry<K, V> entry = elementData[i];
644                while (entry != null) {
645                    int index = entry.isNull ? 0 : (entry.hash & 0x7FFFFFFF)
646                            % length;
647                    Entry<K, V> next = entry.next;
648                    entry.next = newData[index];
649                    newData[index] = entry;
650                    entry = next;
651                }
652            }
653            elementData = newData;
654            computeMaxSize();
655        }

657        /**
658         * Copies all the mappings in the given map to this map. These mappings will
659         * replace all mappings that this map had for any of the keys currently in
660         * the given map.
661         * 
662         * @param map
663         *            the map to copy mappings from.
664         * @since Android 1.0
665         */
666        @Override
667        public void putAll(Map<? extends K, ? extends V> map) {
668            putAllImpl(map);
669        }

671        /**
672         * Removes the mapping with the specified key from this map.
673         * 
674         * @param key
675         *            the key of the mapping to remove.
676         * @return the value of the removed mapping or {@code null} if no mapping
677         *         for the specified key was found.
678         * @since Android 1.0
679         */
680        @Override
681        public V remove(Object key) {
682            poll();
683            int index = 0;
684            Entry<K, V> entry, last = null;
685            if (key != null) {
686                index = (key.hashCode() & 0x7FFFFFFF) % elementData.length;
687                entry = elementData[index];
688                while (entry != null && !key.equals(entry.get())) {
689                    last = entry;
690                    entry = entry.next;
691                }
692            } else {
693                entry = elementData[0];
694                while (entry != null && !entry.isNull) {
695                    last = entry;
696                    entry = entry.next;
697                }
698            }
699            if (entry != null) {
700                modCount++;
701                if (last == null) {
702                    elementData[index] = entry.next;
703                } else {
704                    last.next = entry.next;
705                }
706                elementCount--;
707                return entry.value;
708            }
709            return null;
710        }

713         * Returns the number of elements in this map.
714         * 
715         * @return the number of elements in this map.
716         * @since Android 1.0
717         */
718        @Override
719        public int size() {
720            poll();
721            return elementCount;
722        }

724        private void putAllImpl(Map<? extends K, ? extends V> map) {
725            if (map.entrySet() != null) {
726                super.putAll(map);
727            }
