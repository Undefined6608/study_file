029    /**
030     * <p>
031     * UUID is an immutable representation of a 128-bit universally unique
032     * identifier (UUID).
033     * </p>
034     * <p>
035     * There are multiple, variant layouts of UUIDs, but this class is based upon
036     * variant 2 of <a href="http://www.ietf.org/rfc/rfc4122.txt">RFC 4122</a>, the
037     * Leach-Salz variant. This class can be used to model alternate variants, but
038     * most of the methods will be unsupported in those cases; see each method for
039     * details.
040     * </p>
041     * 
042     * @since Android 1.0
043     */
044    public final class UUID implements Serializable, Comparable<UUID> {
045    
046        private static final long serialVersionUID = -4856846361193249489L;
047    
048        private static SecureRandom rng;
049    
050        private long mostSigBits;
051        private long leastSigBits;
052    
053        private transient int variant;
054        private transient int version;
055        private transient long timestamp;
056        private transient int clockSequence;
057        private transient long node;
058        private transient int hash;
059    
060        /**
061         * <p>
062         * Constructs an instance with the specified bits.
063         * </p>
064         * 
065         * @param mostSigBits
066         *            The 64 most significant bits of the UUID.
067         * @param leastSigBits
068         *            The 64 least significant bits of the UUID.
069         * @since Android 1.0
070         */
071        public UUID(long mostSigBits, long leastSigBits) {
072            super();
073            this.mostSigBits = mostSigBits;
074            this.leastSigBits = leastSigBits;
075            init();
076        }
077    
078        /**
079         * <p>
080         * Sets up the transient fields of this instance based on the current values
081         * of the {@code mostSigBits} and {@code leastSigBits} fields.
082         * </p>
083         */
084        private void init() {
085            // setup hash field
086            int msbHash = (int) (mostSigBits ^ (mostSigBits >>> 32));
087            int lsbHash = (int) (leastSigBits ^ (leastSigBits >>> 32));
088            hash = msbHash ^ lsbHash;
089    
090            // setup variant field
091            if ((leastSigBits & 0x8000000000000000L) == 0) {
092                // MSB0 not set, NCS backwards compatibility variant
093                variant = 0;
094            } else if ((leastSigBits & 0x4000000000000000L) != 0) {
095                // MSB1 set, either MS reserved or future reserved
096                variant = (int) ((leastSigBits & 0xE000000000000000L) >>> 61);
097            } else {
098                // MSB1 not set, RFC 4122 variant
099                variant = 2;
100            }
101    
102            // setup version field
103            version = (int) ((mostSigBits & 0x000000000000F000) >>> 12);
104    
105            if (variant != 2 && version != 1) {
106                return;
107            }
108    
109            // setup timestamp field
110            long timeLow = (mostSigBits & 0xFFFFFFFF00000000L) >>> 32;
111            long timeMid = (mostSigBits & 0x00000000FFFF0000L) << 16;
112            long timeHigh = (mostSigBits & 0x0000000000000FFFL) << 48;
113            timestamp = timeLow | timeMid | timeHigh;
114    
115            // setup clock sequence field
116            clockSequence = (int) ((leastSigBits & 0x3FFF000000000000L) >>> 48);
117    
118            // setup node field
119            node = (leastSigBits & 0x0000FFFFFFFFFFFFL);
120        }

122        /**
123         * <p>
124         * Generates a variant 2, version 4 (randomly generated number) UUID as per
125         * <a href="http://www.ietf.org/rfc/rfc4122.txt">RFC 4122</a>.
126         * </p>
127         * 
128         * @return an UUID instance.
129         * @since Android 1.0
130         */
131        public static UUID randomUUID() {
132            byte[] data;
133            // lock on the class to protect lazy init
134            synchronized (UUID.class) {
135                if (rng == null) {
136                    rng = new SecureRandom();
137                }
138                rng.nextBytes(data = new byte[16]);
139            }
140    
141            long msb = (data[0] & 0xFFL) << 56;
142            msb |= (data[1] & 0xFFL) << 48;
143            msb |= (data[2] & 0xFFL) << 40;
144            msb |= (data[3] & 0xFFL) << 32;
145            msb |= (data[4] & 0xFFL) << 24;
146            msb |= (data[5] & 0xFFL) << 16;
147            msb |= (data[6] & 0x0FL) << 8;
148            msb |= (0x4L << 12); // set the version to 4
149            msb |= (data[7] & 0xFFL);
150    
151            long lsb = (data[8] & 0x3FL) << 56;
152            lsb |= (0x2L << 62); // set the variant to bits 01
153            lsb |= (data[9] & 0xFFL) << 48;
154            lsb |= (data[10] & 0xFFL) << 40;
155            lsb |= (data[11] & 0xFFL) << 32;
156            lsb |= (data[12] & 0xFFL) << 24;
157            lsb |= (data[13] & 0xFFL) << 16;
158            lsb |= (data[14] & 0xFFL) << 8;
159            lsb |= (data[15] & 0xFFL);
160            return new UUID(msb, lsb);
161        }
162    
163        /**
164         * <p>
165         * Generates a variant 2, version 3 (name-based, MD5-hashed) UUID as per <a
166         * href="http://www.ietf.org/rfc/rfc4122.txt">RFC 4122</a>.
167         * </p>
168         * 
169         * @param name
170         *            the name used as byte array to create an UUID.
171         * @return an UUID instance.
172         * @since Android 1.0
173         */
174        public static UUID nameUUIDFromBytes(byte[] name) {
175            if (name == null) {
176                throw new NullPointerException();
177            }
178    
179            byte[] hash;
180            try {
181                MessageDigest md = MessageDigest.getInstance("MD5"); //$NON-NLS-1$
182                hash = md.digest(name);
183            } catch (NoSuchAlgorithmException e) {
184                throw new AssertionError(e);
185            }
186    
187            long msb = (hash[0] & 0xFFL) << 56;
188            msb |= (hash[1] & 0xFFL) << 48;
189            msb |= (hash[2] & 0xFFL) << 40;
190            msb |= (hash[3] & 0xFFL) << 32;
191            msb |= (hash[4] & 0xFFL) << 24;
192            msb |= (hash[5] & 0xFFL) << 16;
193            msb |= (hash[6] & 0x0FL) << 8;
194            msb |= (0x3L << 12); // set the version to 3
195            msb |= (hash[7] & 0xFFL);
196    
197            long lsb = (hash[8] & 0x3FL) << 56;
198            lsb |= (0x2L << 62); // set the variant to bits 01
199            lsb |= (hash[9] & 0xFFL) << 48;
200            lsb |= (hash[10] & 0xFFL) << 40;
201            lsb |= (hash[11] & 0xFFL) << 32;
202            lsb |= (hash[12] & 0xFFL) << 24;
203            lsb |= (hash[13] & 0xFFL) << 16;
204            lsb |= (hash[14] & 0xFFL) << 8;
205            lsb |= (hash[15] & 0xFFL);
206            return new UUID(msb, lsb);
207        }

209        /**
210         * <p>
211         * Parses a UUID string with the format defined by {@link #toString()}.
212         * </p>
213         * 
214         * @param uuid
215         *            the UUID string to parse.
216         * @return an UUID instance.
217         * @throws IllegalArgumentException
218         *             if {@code uuid} is not formatted correctly.
219         * @since Android 1.0
220         */
221        public static UUID fromString(String uuid) {
222            if (uuid == null) {
223                throw new NullPointerException();
224            }
225            
226            int[] position = new int[5];
227            int lastPosition = 1;
228            int startPosition = 0;
229            
230            int i = 0;
231            for (; i < position.length  && lastPosition > 0; i++) {
232                position[i] = uuid.indexOf("-", startPosition); //$NON-NLS-1$
233                lastPosition = position[i];
234                startPosition = position[i] + 1;
235            }
236    
237            // should have and only can have four "-" in UUID
238            if(i != position.length || lastPosition != -1)
239            {
240                throw new IllegalArgumentException(Msg.getString("KA014") + uuid); //$NON-NLS-1$
241            }
242    
243            long m1 = Long.parseLong(uuid.substring(0, position[0]), 16);
244            long m2 = Long.parseLong(uuid.substring(position[0]+ 1, position[1]), 16);
245            long m3 = Long.parseLong(uuid.substring(position[1] + 1, position[2]), 16);
246    
247            long lsb1 = Long.parseLong(uuid.substring(position[2] + 1, position[3]), 16);
248            long lsb2 = Long.parseLong(uuid.substring(position[3]+ 1), 16);
249    
250            long msb = (m1 << 32) | (m2 << 16) | m3;
251            long lsb = (lsb1 << 48) | lsb2;
252            
253            return new UUID(msb, lsb);
254        }

293         * 
294         * @return an {@code int} value.
295         * @since Android 1.0
296         */

299        }

374        /**
375         * <p>
376         * Compares this UUID to the specified UUID. The natural ordering of UUIDs
377         * is based upon the value of the bits from most significant to least
378         * significant.
379         * </p>
380         * 
381         * @param uuid
382         *            the UUID to compare to.
383         * @return a value of -1, 0 or 1 if this UUID is less than, equal to or
384         *         greater than {@code uuid}.
385         * @since Android 1.0
386         */
387        public int compareTo(UUID uuid) {
388            if (uuid == this) {
389                return 0;
390            }
391    
392            if (this.mostSigBits != uuid.mostSigBits) {
393                return this.mostSigBits < uuid.mostSigBits ? -1 : 1;
394            }
395    
396            assert this.mostSigBits == uuid.mostSigBits;
397    
398            if (this.leastSigBits != uuid.leastSigBits) {
399                return this.leastSigBits < uuid.leastSigBits ? -1 : 1;
400            }
401    
402            assert this.leastSigBits == uuid.leastSigBits;
403    
404            return 0;
405        }

407        /**
408         * <p>
409         * Compares this UUID to another object for equality. If {@code object}
410         * is not {@code null}, is a UUID instance, and all bits are equal, then
411         * {@code true} is returned.
412         * </p>
413         * 
414         * @param object
415         *            the {@code Object} to compare to.
416         * @return {@code true} if this UUID is equal to {@code object}
417         *         or {@code false} if not.
418         * @since Android 1.0
419         */
420        @Override
421        public boolean equals(Object object) {
422            if (object == null) {
423                return false;
424            }
425    
426            if (this == object) {
427                return true;
428            }
429    
430            if (!(object instanceof UUID)) {
431                return false;
432            }
433    
434            UUID that = (UUID) object;
435    
436            return (this.leastSigBits == that.leastSigBits)
437                    && (this.mostSigBits == that.mostSigBits);
438        }

454        /**
455         * <p>
456         * Returns a string representation of this UUID in the following format, as
457         * per <a href="http://www.ietf.org/rfc/rfc4122.txt">RFC 4122</a>.
458         * </p>
459         * 
460         * <pre>
461         *            UUID                   = time-low &quot;-&quot; time-mid &quot;-&quot;
462         *                                     time-high-and-version &quot;-&quot;
463         *                                     clock-seq-and-reserved
464         *                                     clock-seq-low &quot;-&quot; node
465         *            time-low               = 4hexOctet
466         *            time-mid               = 2hexOctet
467         *            time-high-and-version  = 2hexOctet
468         *            clock-seq-and-reserved = hexOctet
469         *            clock-seq-low          = hexOctet
470         *            node                   = 6hexOctet
471         *            hexOctet               = hexDigit hexDigit
472         *            hexDigit =
473         *                &quot;0&quot; / &quot;1&quot; / &quot;2&quot; / &quot;3&quot; / &quot;4&quot; / &quot;5&quot; / &quot;6&quot; / &quot;7&quot; / &quot;8&quot; / &quot;9&quot; /
474         *                &quot;a&quot; / &quot;b&quot; / &quot;c&quot; / &quot;d&quot; / &quot;e&quot; / &quot;f&quot; /
475         *                &quot;A&quot; / &quot;B&quot; / &quot;C&quot; / &quot;D&quot; / &quot;E&quot; / &quot;F&quot;
476         * </pre>
477         * 
478         * @return a String instance.
479         * @since Android 1.0
480         */
481        @Override
482        public String toString() {
483            StringBuilder builder = new StringBuilder(36);
484            String msbStr = Long.toHexString(mostSigBits);
485            if (msbStr.length() < 16) {
486                int diff = 16 - msbStr.length();
487                for (int i = 0; i < diff; i++) {
488                    builder.append('0');
489                }
490            }
491            builder.append(msbStr);
492            builder.insert(8, '-');
493            builder.insert(13, '-');
494            builder.append('-');
495            String lsbStr = Long.toHexString(leastSigBits);
496            if (lsbStr.length() < 16) {
497                int diff = 16 - lsbStr.length();
498                for (int i = 0; i < diff; i++) {
499                    builder.append('0');
500                }
501            }
502            builder.append(lsbStr);
503            builder.insert(23, '-');
504            return builder.toString();
505        }
