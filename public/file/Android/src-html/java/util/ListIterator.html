018    package java.util;
019    
020    
021    /**
022     * An ListIterator is used to sequence over a List of objects. ListIterator can
023     * move backwards or forwards through the list.
024     * 
025     * @since Android 1.0
026     */
027    public interface ListIterator<E> extends Iterator<E> {
028        
029        /**
030         * Inserts the specified object into the list between {@code next} and
031         * {@code previous}. The object inserted will be the previous object.
032         * 
033         * @param object
034         *            the object to insert.
035         * @throws UnsupportedOperationException
036         *             if adding is not supported by the list being iterated.
037         * @throws ClassCastException
038         *             if the class of the object is inappropriate for the list.
039         * @throws IllegalArgumentException
040         *             if the object cannot be added to the list.
041         * @since Android 1.0
042         */
043        void add(E object);
044    
045        /**
046         * Returns whether there are more elements to iterate.
047         * 
048         * @return {@code true} if there are more elements, {@code false} otherwise.
049         * @see #next
050         * @since Android 1.0
051         */
052        public boolean hasNext();
053    
054        /**
055         * Returns whether there are previous elements to iterate.
056         * 
057         * @return {@code true} if there are previous elements, {@code false}
058         *         otherwise.
059         * @see #previous
060         * @since Android 1.0
061         */
062        public boolean hasPrevious();
063    
064        /**
065         * Returns the next object in the iteration.
066         * 
067         * @return the next object.
068         * @throws NoSuchElementException
069         *             if there are no more elements.
070         * @see #hasNext
071         * @since Android 1.0
072         */
073        public E next();
074    
075        /**
076         * Returns the index of the next object in the iteration.
077         * 
078         * @return the index of the next object, or the size of the list if the
079         *         iterator is at the end.
080         * @throws NoSuchElementException
081         *             if there are no more elements.
082         * @see #next
083         * @since Android 1.0
084         */
085        public int nextIndex();
086    
087        /**
088         * Returns the previous object in the iteration.
089         * 
090         * @return the previous object.
091         * @throws NoSuchElementException
092         *             if there are no previous elements.
093         * @see #hasPrevious
094         * @since Android 1.0
095         */
096        public E previous();
097    
098        /**
099         * Returns the index of the previous object in the iteration.
100         * 
101         * @return the index of the previous object, or -1 if the iterator is at the
102         *         beginning.
103         * @throws NoSuchElementException
104         *             if there are no previous elements.
105         * @see #previous
106         * @since Android 1.0
107         */
108        public int previousIndex();
109    
110        /**
111         * Removes the last object returned by {@code next} or {@code previous} from
112         * the list.
113         * 
114         * @throws UnsupportedOperationException
115         *             if removing is not supported by the list being iterated.
116         * @throws IllegalStateException
117         *             if {@code next} or {@code previous} have not been called, or
118         *             {@code remove} or {@code add} have already been called after
119         *             the last call to {@code next} or {@code previous}.
120         * @since Android 1.0
121         */
122        public void remove();
123    
124        /**
125         * Replaces the last object returned by {@code next} or {@code previous}
126         * with the specified object.
127         * 
128         * @param object
129         *            the object to set.
130         * @throws UnsupportedOperationException
131         *             if setting is not supported by the list being iterated
132         * @throws ClassCastException
133         *             if the class of the object is inappropriate for the list.
134         * @throws IllegalArgumentException
135         *             if the object cannot be added to the list.
136         * @throws IllegalStateException
137         *             if {@code next} or {@code previous} have not been called, or
138         *             {@code remove} or {@code add} have already been called after
139         *             the last call to {@code next} or {@code previous}.
140         * @since Android 1.0
141         */
142        void set(E object);
143    }
