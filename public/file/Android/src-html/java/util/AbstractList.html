018    package java.util;

020    /**
021     * {@code AbstractList} is an abstract implementation of the {@code List} interface, optimized
022     * for a backing store which supports random access. This implementation does
023     * not support adding or replacing. A subclass must implement the abstract
024     * methods {@code get()} and {@code size()}, and to create a
025     * modifiable {@code List} it's necessary to override the {@code add()} method that
026     * currently throws an {@code UnsupportedOperationException}.
027     * 
028     * @since Android 1.0
029     */
030    public abstract class AbstractList<E> extends AbstractCollection<E> implements
031            List<E> {
032    
033        /**
034         * A counter for changes to the list.
035         * 
036         * @since Android 1.0
037         */
038        protected transient int modCount;
039    
040        private class SimpleListIterator implements Iterator<E> {
041            int pos = -1;
042    
043            int expectedModCount;
044    
045            int lastPosition = -1;
046    
047            SimpleListIterator() {
048                super();
049                expectedModCount = modCount;
050            }
051    
052            public boolean hasNext() {
053                return pos + 1 < size();
054            }
055    
056            public E next() {
057                if (expectedModCount == modCount) {
058                    try {
059                        E result = get(pos + 1);
060                        lastPosition = ++pos;
061                        return result;
062                    } catch (IndexOutOfBoundsException e) {
063                        throw new NoSuchElementException();
064                    }
065                }
066                throw new ConcurrentModificationException();
067            }
068    
069            public void remove() {
070                if (expectedModCount == modCount) {
071                    try {
072                        AbstractList.this.remove(lastPosition);
073                    } catch (IndexOutOfBoundsException e) {
074                        throw new IllegalStateException();
075                    }
076                    if (modCount != expectedModCount) {
077                        expectedModCount++;
078                    }
079                    if (pos == lastPosition) {
080                        pos--;
081                    }
082                    lastPosition = -1;
083                } else {
084                    throw new ConcurrentModificationException();
085                }
086            }
087        }
088    
089        private final class FullListIterator extends SimpleListIterator implements
090                ListIterator<E> {
091            FullListIterator(int start) {
092                super();
093                if (0 <= start && start <= size()) {
094                    pos = start - 1;
095                } else {
096                    throw new IndexOutOfBoundsException();
097                }
098            }
099    
100            public void add(E object) {
101                if (expectedModCount == modCount) {
102                    try {
103                        AbstractList.this.add(pos + 1, object);
104                    } catch (IndexOutOfBoundsException e) {
105                        throw new NoSuchElementException();
106                    }
107                    pos++;
108                    lastPosition = -1;
109                    if (modCount != expectedModCount) {
110                        expectedModCount++;
111                    }
112                } else {
113                    throw new ConcurrentModificationException();
114                }
115            }
116    
117            public boolean hasPrevious() {
118                return pos >= 0;
119            }
120    
121            public int nextIndex() {
122                return pos + 1;
123            }
124    
125            public E previous() {
126                if (expectedModCount == modCount) {
127                    try {
128                        E result = get(pos);
129                        lastPosition = pos;
130                        pos--;
131                        return result;
132                    } catch (IndexOutOfBoundsException e) {
133                        throw new NoSuchElementException();
134                    }
135                }
136                throw new ConcurrentModificationException();
137            }
138    
139            public int previousIndex() {
140                return pos;
141            }
142    
143            public void set(E object) {
144                if (expectedModCount == modCount) {
145                    try {
146                        AbstractList.this.set(lastPosition, object);
147                    } catch (IndexOutOfBoundsException e) {
148                        throw new IllegalStateException();
149                    }
150                } else {
151                    throw new ConcurrentModificationException();
152                }
153            }
154        }

156        private static final class SubAbstractListRandomAccess<E> extends
157                SubAbstractList<E> implements RandomAccess {
158            SubAbstractListRandomAccess(AbstractList<E> list, int start, int end) {
159                super(list, start, end);
160            }
161        }

163        private static class SubAbstractList<E> extends AbstractList<E> {
164            private final AbstractList<E> fullList;
165    
166            private int offset;
167    
168            private int size;
169    
170            private static final class SubAbstractListIterator<E> implements
171                    ListIterator<E> {
172                private final SubAbstractList<E> subList;
173    
174                private final ListIterator<E> iterator;
175    
176                private int start;
177    
178                private int end;
179    
180                SubAbstractListIterator(ListIterator<E> it,
181                        SubAbstractList<E> list, int offset, int length) {
182                    super();
183                    iterator = it;
184                    subList = list;
185                    start = offset;
186                    end = start + length;
187                }
188    
189                public void add(E object) {
190                    iterator.add(object);
191                    subList.sizeChanged(true);
192                    end++;
193                }
194    
195                public boolean hasNext() {
196                    return iterator.nextIndex() < end;
197                }
198    
199                public boolean hasPrevious() {
200                    return iterator.previousIndex() >= start;
201                }
202    
203                public E next() {
204                    if (iterator.nextIndex() < end) {
205                        return iterator.next();
206                    }
207                    throw new NoSuchElementException();
208                }
209    
210                public int nextIndex() {
211                    return iterator.nextIndex() - start;
212                }
213    
214                public E previous() {
215                    if (iterator.previousIndex() >= start) {
216                        return iterator.previous();
217                    }
218                    throw new NoSuchElementException();
219                }
220    
221                public int previousIndex() {
222                    int previous = iterator.previousIndex();
223                    if (previous >= start) {
224                        return previous - start;
225                    }
226                    return -1;
227                }
228    
229                public void remove() {
230                    iterator.remove();
231                    subList.sizeChanged(false);
232                    end--;
233                }
234    
235                public void set(E object) {
236                    iterator.set(object);
237                }
238            }
239    
240            SubAbstractList(AbstractList<E> list, int start, int end) {
241                super();
242                fullList = list;
243                modCount = fullList.modCount;
244                offset = start;
245                size = end - start;
246            }
247    
248            @Override
249            public void add(int location, E object) {
250                if (modCount == fullList.modCount) {
251                    if (0 <= location && location <= size) {
252                        fullList.add(location + offset, object);
253                        size++;
254                        modCount = fullList.modCount;
255                    } else {
256                        throw new IndexOutOfBoundsException();
257                    }
258                } else {
259                    throw new ConcurrentModificationException();
260                }
261            }
262    
263            @Override
264            public boolean addAll(int location, Collection<? extends E> collection) {
265                if (modCount == fullList.modCount) {
266                    if (0 <= location && location <= size) {
267                        boolean result = fullList.addAll(location + offset,
268                                collection);
269                        if (result) {
270                            size += collection.size();
271                            modCount = fullList.modCount;
272                        }
273                        return result;
274                    }
275                    throw new IndexOutOfBoundsException();
276                }
277                throw new ConcurrentModificationException();
278            }
279    
280            @Override
281            public boolean addAll(Collection<? extends E> collection) {
282                if (modCount == fullList.modCount) {
283                    boolean result = fullList.addAll(offset + size, collection);
284                    if (result) {
285                        size += collection.size();
286                        modCount = fullList.modCount;
287                    }
288                    return result;
289                }
290                throw new ConcurrentModificationException();
291            }
292    
293            @Override
294            public E get(int location) {
295                if (modCount == fullList.modCount) {
296                    if (0 <= location && location < size) {
297                        return fullList.get(location + offset);
298                    }
299                    throw new IndexOutOfBoundsException();
300                }
301                throw new ConcurrentModificationException();
302            }
303    
304            @Override
305            public Iterator<E> iterator() {
306                return listIterator(0);
307            }
308    
309            @Override
310            public ListIterator<E> listIterator(int location) {
311                if (modCount == fullList.modCount) {
312                    if (0 <= location && location <= size) {
313                        return new SubAbstractListIterator<E>(fullList
314                                .listIterator(location + offset), this, offset,
315                                size);
316                    }
317                    throw new IndexOutOfBoundsException();
318                }
319                throw new ConcurrentModificationException();
320            }
321    
322            @Override
323            public E remove(int location) {
324                if (modCount == fullList.modCount) {
325                    if (0 <= location && location < size) {
326                        E result = fullList.remove(location + offset);
327                        size--;
328                        modCount = fullList.modCount;
329                        return result;
330                    }
331                    throw new IndexOutOfBoundsException();
332                }
333                throw new ConcurrentModificationException();
334            }
335    
336            @Override
337            protected void removeRange(int start, int end) {
338                if (start != end) {
339                    if (modCount == fullList.modCount) {
340                        fullList.removeRange(start + offset, end + offset);
341                        size -= end - start;
342                        modCount = fullList.modCount;
343                    } else {
344                        throw new ConcurrentModificationException();
345                    }
346                }
347            }
348    
349            @Override
350            public E set(int location, E object) {
351                if (modCount == fullList.modCount) {
352                    if (0 <= location && location < size) {
353                        return fullList.set(location + offset, object);
354                    }
355                    throw new IndexOutOfBoundsException();
356                }
357                throw new ConcurrentModificationException();
358            }
359    
360            @Override
361            public int size() {
362                return size;
363            }
364    
365            void sizeChanged(boolean increment) {
366                if (increment) {
367                    size++;
368                } else {
369                    size--;
370                }
371                modCount = fullList.modCount;
372            }
373        }

384        /**
385         * Inserts the specified object into this List at the specified location.
386         * The object is inserted before any previous element at the specified
387         * location. If the location is equal to the size of this List, the object
388         * is added at the end.
389         * 
390         * Concrete implementations that would like to support the add functionality
391         * must override this method.          
392         * 
393         * @param location
394         *            the index at which to insert.
395         * @param object
396         *            the object to add.
397         * 
398         * @throws UnsupportedOperationException
399         *                if adding to this List is not supported.
400         * @throws ClassCastException
401         *                if the class of the object is inappropriate for this
402         *                List
403         * @throws IllegalArgumentException
404         *                if the object cannot be added to this List
405         * @throws IndexOutOfBoundsException
406         *                if {@code location < 0 || >= size()}
407         * @since Android 1.0
408         */
409        public void add(int location, E object) {
410            throw new UnsupportedOperationException();
411        }
412    
413        /**
414         * Adds the specified object at the end of this List.
415         * 
416         * 
417         * @param object
418         *            the object to add
419         * @return true
420         * 
421         * @throws UnsupportedOperationException
422         *                if adding to this List is not supported
423         * @throws ClassCastException
424         *                if the class of the object is inappropriate for this
425         *                List
426         * @throws IllegalArgumentException
427         *                if the object cannot be added to this List
428         * @since Android 1.0
429         */
430        @Override
431        public boolean add(E object) {
432            add(size(), object);
433            return true;
434        }
435    
436        /**
437         * Inserts the objects in the specified Collection at the specified location
438         * in this List. The objects are added in the order they are returned from
439         * the collection's iterator.
440         * 
441         * @param location
442         *            the index at which to insert.
443         * @param collection
444         *            the Collection of objects
445         * @return {@code true} if this List is modified, {@code false} otherwise.
446         * @throws UnsupportedOperationException
447         *             if adding to this list is not supported.
448         * @throws ClassCastException
449         *             if the class of an object is inappropriate for this list.
450         * @throws IllegalArgumentException
451         *             if an object cannot be added to this list.
452         * @throws IndexOutOfBoundsException
453         *             if {@code location < 0 || > size()}
454         * @since Android 1.0
455         */
456        public boolean addAll(int location, Collection<? extends E> collection) {
457            Iterator<? extends E> it = collection.iterator();
458            while (it.hasNext()) {
459                add(location++, it.next());
460            }
461            return !collection.isEmpty();
462        }

464        /**
465         * Removes all elements from this list, leaving it empty.
466         * 
467         * @throws UnsupportedOperationException
468         *             if removing from this list is not supported.
469         * @see List#isEmpty
470         * @see List#size
471         * @since Android 1.0
472         */
473        @Override
474        public void clear() {
475            removeRange(0, size());
476        }

478        /**
479         * Compares the specified object to this list and return true if they are
480         * equal. Two lists are equal when they both contain the same objects in the
481         * same order.
482         * 
483         * @param object
484         *            the object to compare to this object.
485         * @return {@code true} if the specified object is equal to this list,
486         *         {@code false} otherwise.
487         * @see #hashCode
488         * @since Android 1.0
489         */
490        @Override
491        public boolean equals(Object object) {
492            if (this == object) {
493                return true;
494            }
495            if (object instanceof List) {
496                List<?> list = (List<?>) object;
497                if (list.size() != size()) {
498                    return false;
499                }
500    
501                Iterator<?> it1 = iterator(), it2 = list.iterator();
502                while (it1.hasNext()) {
503                    Object e1 = it1.next(), e2 = it2.next();
504                    if (!(e1 == null ? e2 == null : e1.equals(e2))) {
505                        return false;
506                    }
507                }
508                return true;
509            }
510            return false;
511        }

514         * Returns the element at the specified location in this list.
515         * 
516         * @param location
517         *            the index of the element to return.
518         * @return the element at the specified index.
519         * @throws IndexOutOfBoundsException
520         *             if {@code location < 0 || >= size()}

525        /**
526         * Returns the hash code of this list. The hash code is calculated by taking
527         * each element's hashcode into account.
528         * 
529         * @return the hash code.
530         * @see #equals
531         * @see List#hashCode()
532         * @since Android 1.0
533         */
534        @Override
535        public int hashCode() {
536            int result = 1;
537            Iterator<?> it = iterator();
538            while (it.hasNext()) {
539                Object object = it.next();
540                result = (31 * result) + (object == null ? 0 : object.hashCode());
541            }
542            return result;
543        }

545        /**
546         * Searches this list for the specified object and returns the index of the
547         * first occurrence.
548         * 
549         * @param object
550         *            the object to search for.
551         * @return the index of the first occurrence of the object, or -1 if it was
552         *         not found.
553         * @since Android 1.0
554         */
555        public int indexOf(Object object) {
556            ListIterator<?> it = listIterator();
557            if (object != null) {
558                while (it.hasNext()) {
559                    if (object.equals(it.next())) {
560                        return it.previousIndex();
561                    }
562                }
563            } else {
564                while (it.hasNext()) {
565                    if (it.next() == null) {
566                        return it.previousIndex();
567                    }
568                }
569            }
570            return -1;
571        }

586        /**
587         * Searches this list for the specified object and returns the index of the
588         * last occurrence.
589         * 
590         * @param object
591         *            the object to search for.
592         * @return the index of the last occurrence of the object, or -1 if the
593         *         object was not found.
594         * @since Android 1.0
595         */
596        public int lastIndexOf(Object object) {
597            ListIterator<?> it = listIterator(size());
598            if (object != null) {
599                while (it.hasPrevious()) {
600                    if (object.equals(it.previous())) {
601                        return it.nextIndex();
602                    }
603                }
604            } else {
605                while (it.hasPrevious()) {
606                    if (it.previous() == null) {
607                        return it.nextIndex();
608                    }
609                }
610            }
611            return -1;
612        }

659        /**
660         * Removes the objects in the specified range from the start to the end
661         * index minus one.
662         * 
663         * @param start
664         *            the index at which to start removing.
665         * @param end
666         *            the index after the last element to remove.
667         * @throws UnsupportedOperationException
668         *             if removing from this list is not supported.
669         * @throws IndexOutOfBoundsException
670         *             if {@code start < 0} or {@code start >= size()}.
671         * @since Android 1.0
672         */
673        protected void removeRange(int start, int end) {
674            Iterator<?> it = listIterator(start);
675            for (int i = start; i < end; i++) {
676                it.next();
677                it.remove();
678            }
679        }

704        /**
705         * Returns a part of consecutive elements of this list as a view. The
706         * returned view will be of zero length if start equals end. Any change that
707         * occurs in the returned subList will be reflected to the original list,
708         * and vice-versa. All the supported optional operations by the original
709         * list will also be supported by this subList.
710         * <p>
711         * This method can be used as a handy method to do some operations on a sub
712         * range of the original list, for example
713         * {@code list.subList(from, to).clear();}
714         * <p>
715         * If the original list is modified in other ways than through the returned
716         * subList, the behavior of the returned subList becomes undefined.
717         * <p>
718         * The returned subList is a subclass of AbstractList. The subclass stores
719         * offset, size of itself, and modCount of the original list. If the
720         * original list implements RandomAccess interface, the returned subList
721         * also implements RandomAccess interface.
722         * <p>
723         * The subList's set(int, Object), get(int), add(int, Object), remove(int),
724         * addAll(int, Collection) and removeRange(int, int) methods first check the
725         * bounds, adjust offsets and then call the corresponding methods of the
726         * original AbstractList. addAll(Collection c) method of the returned
727         * subList calls the original addAll(offset + size, c).
728         * <p>
729         * The listIterator(int) method of the subList wraps the original list
730         * iterator. The iterator() method of the subList invokes the original
731         * listIterator() method, and the size() method merely returns the size of
732         * the subList.
733         * <p>
734         * All methods will throw a ConcurrentModificationException if the modCount
735         * of the original list is not equal to the expected value.
736         * 
737         * @param start
738         *            start index of the subList (inclusive).
739         * @param end
740         *            end index of the subList, (exclusive).
741         * @return a subList view of this list starting from {@code start}
742         *         (inclusive), and ending with {@code end} (exclusive)
743         * @throws IndexOutOfBoundsException
744         *             if (start < 0 || end > size())
745         * @throws IllegalArgumentException
746         *             if (start > end)
747         * @since Android 1.0
748         */
749        public List<E> subList(int start, int end) {
750            if (0 <= start && end <= size()) {
751                if (start <= end) {
752                    if (this instanceof RandomAccess) {
753                        return new SubAbstractListRandomAccess<E>(this, start, end);
754                    }
755                    return new SubAbstractList<E>(this, start, end);
756                }
757                throw new IllegalArgumentException();
758            }
759            throw new IndexOutOfBoundsException();
760        }
