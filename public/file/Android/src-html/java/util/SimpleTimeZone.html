038    /**
039     * {@code SimpleTimeZone} is a concrete subclass of {@code TimeZone}
040     * that represents a time zone for use with a Gregorian calendar. This class
041     * does not handle historical changes.
042     * <P>
043     * Use a negative value for {@code dayOfWeekInMonth} to indicate that
044     * {@code SimpleTimeZone} should count from the end of the month
045     * backwards. For example, Daylight Savings Time ends at the last
046     * (dayOfWeekInMonth = -1) Sunday in October, at 2 AM in standard time.
047     * 
048     * @see Calendar
049     * @see GregorianCalendar
050     * @see TimeZone
051     * @since Android 1.0
052     */
053    public class SimpleTimeZone extends TimeZone {

117        /**
118         * Constructs a {@code SimpleTimeZone} with the given base time zone offset from GMT,
119         * time zone ID, and times to start and end the daylight savings time. Timezone IDs can
120         * be obtained from {@code TimeZone.getAvailableIDs}. Normally you should use
121         * {@code TimeZone.getDefault} to create a {@code TimeZone}. For a time zone that does not
122         * use daylight saving time, do not use this constructor; instead you should
123         * use {@code SimpleTimeZone(rawOffset, ID)}.
124         * <p>
125         * By default, this constructor specifies day-of-week-in-month rules. That
126         * is, if the {@code startDay} is 1, and the {@code startDayOfWeek} is {@code SUNDAY}, then this
127         * indicates the first Sunday in the {@code startMonth}. A {@code startDay} of -1 likewise
128         * indicates the last Sunday. However, by using negative or zero values for
129         * certain parameters, other types of rules can be specified.
130         * <p>
131         * Day of month: To specify an exact day of the month, such as March 1, set
132         * {@code startDayOfWeek} to zero.
133         * <p>
134         * Day of week after day of month: To specify the first day of the week
135         * occurring on or after an exact day of the month, make the day of the week
136         * negative. For example, if {@code startDay} is 5 and {@code startDayOfWeek} is {@code -MONDAY},
137         * this indicates the first Monday on or after the 5th day of the
138         * {@code startMonth}.
139         * <p>
140         * Day of week before day of month: To specify the last day of the week
141         * occurring on or before an exact day of the month, make the day of the
142         * week and the day of the month negative. For example, if {@code startDay} is {@code -21}
143         * and {@code startDayOfWeek} is {@code -WEDNESDAY}, this indicates the last Wednesday on or
144         * before the 21st of the {@code startMonth}.
145         * <p>
146         * The above examples refer to the {@code startMonth}, {@code startDay}, and {@code startDayOfWeek};
147         * the same applies for the {@code endMonth}, {@code endDay}, and {@code endDayOfWeek}.
148         * <p>
149         * The daylight savings time difference is set to the default value: one hour.          
150         * 
151         * @param offset
152         *            the given base time zone offset to GMT.
153         * @param name
154         *            the time zone ID which is obtained from
155         *            {@code TimeZone.getAvailableIDs}.
156         * @param startMonth
157         *            the daylight savings starting month. The month indexing is 0-based. eg, 0
158         *            for January.
159         * @param startDay
160         *            the daylight savings starting day-of-week-in-month. Please see
161         *            the member description for an example.
162         * @param startDayOfWeek
163         *            the daylight savings starting day-of-week. Please see the
164         *            member description for an example.
165         * @param startTime
166         *            the daylight savings starting time in local wall time, which
167         *            is standard time in this case. Please see the member
168         *            description for an example.
169         * @param endMonth
170         *            the daylight savings ending month. The month indexing is 0-based. eg, 0 for
171         *            January.
172         * @param endDay
173         *            the daylight savings ending day-of-week-in-month. Please see
174         *            the member description for an example.
175         * @param endDayOfWeek
176         *            the daylight savings ending day-of-week. Please see the member
177         *            description for an example.
178         * @param endTime
179         *            the daylight savings ending time in local wall time, which is
180         *            daylight time in this case. Please see the member description
181         *            for an example.
182         * @throws IllegalArgumentException
183         *             if the month, day, dayOfWeek, or time parameters are out of
184         *             range for the start or end rule.
185         * @since Android 1.0
186         */
187        public SimpleTimeZone(int offset, String name, int startMonth,
188                int startDay, int startDayOfWeek, int startTime, int endMonth,
189                int endDay, int endDayOfWeek, int endTime) {
190            this(offset, name, startMonth, startDay, startDayOfWeek, startTime,
191                    endMonth, endDay, endDayOfWeek, endTime, 3600000);
192        }

364        @Override
365        public int getOffset(int era, int year, int month, int day, int dayOfWeek,
366                int time) {
367            if (era != GregorianCalendar.BC && era != GregorianCalendar.AD) {
368                throw new IllegalArgumentException(Msg.getString("K00ea", era)); //$NON-NLS-1$
369            }
370            checkRange(month, dayOfWeek, time);
371            if (month != Calendar.FEBRUARY || day != 29 || !isLeapYear(year)) {
372                checkDay(month, day);
373            }
374    
375            if (!useDaylightTime() || era != GregorianCalendar.AD
376                    || year < startYear) {
377                return rawOffset;
378            }
379            if (endMonth < startMonth) {
380                if (month > endMonth && month < startMonth) {
381                    return rawOffset;
382                }
383            } else {
384                if (month < startMonth || month > endMonth) {
385                    return rawOffset;
386                }
387            }
388    
389            int ruleDay = 0, daysInMonth, firstDayOfMonth = mod7(dayOfWeek - day);
390            if (month == startMonth) {
391                switch (startMode) {
392                case DOM_MODE:
393                    ruleDay = startDay;
394                    break;
395                case DOW_IN_MONTH_MODE:
396                    if (startDay >= 0) {
397                        ruleDay = mod7(startDayOfWeek - firstDayOfMonth) + 1
398                                + (startDay - 1) * 7;
399                    } else {
400                        daysInMonth = GregorianCalendar.DaysInMonth[startMonth];
401                        if (startMonth == Calendar.FEBRUARY && isLeapYear(year)) {
402                            daysInMonth += 1;
403                        }
404                        ruleDay = daysInMonth
405                                + 1
406                                + mod7(startDayOfWeek
407                                        - (firstDayOfMonth + daysInMonth))
408                                + startDay * 7;
409                    }
410                    break;
411                case DOW_GE_DOM_MODE:
412                    ruleDay = startDay
413                            + mod7(startDayOfWeek
414                                    - (firstDayOfMonth + startDay - 1));
415                    break;
416                case DOW_LE_DOM_MODE:
417                    ruleDay = startDay
418                            + mod7(startDayOfWeek
419                                    - (firstDayOfMonth + startDay - 1));
420                    if (ruleDay != startDay) {
421                        ruleDay -= 7;
422                    }
423                    break;
424                }
425                if (ruleDay > day || ruleDay == day && time < startTime) {
426                    return rawOffset;
427                }
428            }
429    
430            int ruleTime = endTime - dstSavings;
431            int nextMonth = (month + 1) % 12;
432            if (month == endMonth || (ruleTime < 0 && nextMonth == endMonth)) {
433                switch (endMode) {
434                case DOM_MODE:
435                    ruleDay = endDay;
436                    break;
437                case DOW_IN_MONTH_MODE:
438                    if (endDay >= 0) {
439                        ruleDay = mod7(endDayOfWeek - firstDayOfMonth) + 1
440                                + (endDay - 1) * 7;
441                    } else {
442                        daysInMonth = GregorianCalendar.DaysInMonth[endMonth];
443                        if (endMonth == Calendar.FEBRUARY && isLeapYear(year)) {
444                            daysInMonth++;
445                        }
446                        ruleDay = daysInMonth
447                                + 1
448                                + mod7(endDayOfWeek
449                                        - (firstDayOfMonth + daysInMonth)) + endDay
450                                * 7;
451                    }
452                    break;
453                case DOW_GE_DOM_MODE:
454                    ruleDay = endDay
455                            + mod7(endDayOfWeek - (firstDayOfMonth + endDay - 1));
456                    break;
457                case DOW_LE_DOM_MODE:
458                    ruleDay = endDay
459                            + mod7(endDayOfWeek - (firstDayOfMonth + endDay - 1));
460                    if (ruleDay != endDay) {
461                        ruleDay -= 7;
462                    }
463                    break;
464                }
465    
466                int ruleMonth = endMonth;
467                if (ruleTime < 0) {
468                    int changeDays = 1 - (ruleTime / 86400000);
469                    ruleTime = (ruleTime % 86400000) + 86400000;
470                    ruleDay -= changeDays;
471                    if (ruleDay <= 0) {
472                        if (--ruleMonth < Calendar.JANUARY) {
473                            ruleMonth = Calendar.DECEMBER;
474                        }
475                        ruleDay += GregorianCalendar.DaysInMonth[ruleMonth];
476                        if (ruleMonth == Calendar.FEBRUARY && isLeapYear(year)) {
477                            ruleDay++;
478                        }
479                    }
480                }
481    
482                if (month == ruleMonth) {
483                    if (ruleDay < day || ruleDay == day && time >= ruleTime) {
484                        return rawOffset;
485                    }
486                } else if (nextMonth != ruleMonth) {
487                    return rawOffset;
488                }
489            }
490            return rawOffset + dstSavings;
491        }

528        @Override
529        public boolean hasSameRules(TimeZone zone) {
530            if (!(zone instanceof SimpleTimeZone)) {
531                return false;
532            }
533            SimpleTimeZone tz = (SimpleTimeZone) zone;
534            if (useDaylight != tz.useDaylight) {
535                return false;
536            }
537            if (!useDaylight) {
538                return rawOffset == tz.rawOffset;
539            }
540            return rawOffset == tz.rawOffset && dstSavings == tz.dstSavings
541                    && startYear == tz.startYear && startMonth == tz.startMonth
542                    && startDay == tz.startDay && startMode == tz.startMode
543                    && startDayOfWeek == tz.startDayOfWeek
544                    && startTime == tz.startTime && endMonth == tz.endMonth
545                    && endDay == tz.endDay && endDayOfWeek == tz.endDayOfWeek
546                    && endTime == tz.endTime && endMode == tz.endMode;
547        }

549        @Override
550        public boolean inDaylightTime(Date time) {
551            // check for null pointer
552            long millis = time.getTime();
553            if (!useDaylightTime()) {
554                return false;
555            }
556            if (daylightSavings == null) {
557                daylightSavings = new GregorianCalendar(this);
558            }
559            return daylightSavings.getOffset(millis + rawOffset) != rawOffset;
560        }

562        private boolean isLeapYear(int year) {
563            if (year > 1582) {
564                return year % 4 == 0 && (year % 100 != 0 || year % 400 == 0);
565            }
566            return year % 4 == 0;
567        }

569        private int mod7(int num1) {
570            int rem = num1 % 7;
571            return (num1 < 0 && rem < 0) ? 7 + rem : rem;
572        }

589        private void checkRange(int month, int dayOfWeek, int time) {
590            if (month < Calendar.JANUARY || month > Calendar.DECEMBER) {
591                throw new IllegalArgumentException(Msg.getString("K00e5", month)); //$NON-NLS-1$
592            }
593            if (dayOfWeek < Calendar.SUNDAY || dayOfWeek > Calendar.SATURDAY) {
594                throw new IllegalArgumentException(Msg.getString("K00e7", dayOfWeek)); //$NON-NLS-1$
595            }
596            if (time < 0 || time >= 24 * 3600000) {
597                throw new IllegalArgumentException(Msg.getString("K00e8", time)); //$NON-NLS-1$
598            }
599        }
600    
601        private void checkDay(int month, int day) {
602            if (day <= 0 || day > GregorianCalendar.DaysInMonth[month]) {
603                throw new IllegalArgumentException(Msg.getString("K00e6", day)); //$NON-NLS-1$
604            }
605        }

607        private void setEndMode() {
608            if (endDayOfWeek == 0) {
609                endMode = DOM_MODE;
610            } else if (endDayOfWeek < 0) {
611                endDayOfWeek = -endDayOfWeek;
612                if (endDay < 0) {
613                    endDay = -endDay;
614                    endMode = DOW_LE_DOM_MODE;
615                } else {
616                    endMode = DOW_GE_DOM_MODE;
617                }
618            } else {
619                endMode = DOW_IN_MONTH_MODE;
620            }
621            useDaylight = startDay != 0 && endDay != 0;
622            if (endDay != 0) {
623                checkRange(endMonth, endMode == DOM_MODE ? 1 : endDayOfWeek,
624                        endTime);
625                if (endMode != DOW_IN_MONTH_MODE) {
626                    checkDay(endMonth, endDay);
627                } else {
628                    if (endDay < -5 || endDay > 5) {
629                        throw new IllegalArgumentException(Msg.getString("K00f8", endDay)); //$NON-NLS-1$
630                    }
631                }

721        private void setStartMode() {
722            if (startDayOfWeek == 0) {
723                startMode = DOM_MODE;
724            } else if (startDayOfWeek < 0) {
725                startDayOfWeek = -startDayOfWeek;
726                if (startDay < 0) {
727                    startDay = -startDay;
728                    startMode = DOW_LE_DOM_MODE;
729                } else {
730                    startMode = DOW_GE_DOM_MODE;
731                }
732            } else {
733                startMode = DOW_IN_MONTH_MODE;
734            }
735            useDaylight = startDay != 0 && endDay != 0;
736            if (startDay != 0) {
737                checkRange(startMonth, startMode == DOM_MODE ? 1 : startDayOfWeek,
738                        startTime);
739                if (startMode != DOW_IN_MONTH_MODE) {
740                    checkDay(startMonth, startDay);
741                } else {
742                    if (startDay < -5 || startDay > 5) {
743                        throw new IllegalArgumentException(Msg.getString("K00f8", startDay)); //$NON-NLS-1$
744                    }
745                }
746            }
747            if (startMode != DOM_MODE) {
748                startDayOfWeek--;
749            }
750        }

893        private void writeObject(ObjectOutputStream stream) throws IOException {
894            int sEndDay = endDay, sEndDayOfWeek = endDayOfWeek + 1, sStartDay = startDay, sStartDayOfWeek = startDayOfWeek + 1;
895            if (useDaylight
896                    && (startMode != DOW_IN_MONTH_MODE || endMode != DOW_IN_MONTH_MODE)) {
897                Calendar cal = new GregorianCalendar(this);
898                if (endMode != DOW_IN_MONTH_MODE) {
899                    cal.set(Calendar.MONTH, endMonth);
900                    cal.set(Calendar.DATE, endDay);
901                    sEndDay = cal.get(Calendar.DAY_OF_WEEK_IN_MONTH);
902                    if (endMode == DOM_MODE) {
903                        sEndDayOfWeek = cal.getFirstDayOfWeek();
904                    }
905                }
906                if (startMode != DOW_IN_MONTH_MODE) {
907                    cal.set(Calendar.MONTH, startMonth);
908                    cal.set(Calendar.DATE, startDay);
909                    sStartDay = cal.get(Calendar.DAY_OF_WEEK_IN_MONTH);
910                    if (startMode == DOM_MODE) {
911                        sStartDayOfWeek = cal.getFirstDayOfWeek();
912                    }
913                }
914            }
915            ObjectOutputStream.PutField fields = stream.putFields();
916            fields.put("dstSavings", dstSavings); //$NON-NLS-1$
917            fields.put("endDay", sEndDay); //$NON-NLS-1$
918            fields.put("endDayOfWeek", sEndDayOfWeek); //$NON-NLS-1$
919            fields.put("endMode", endMode); //$NON-NLS-1$
920            fields.put("endMonth", endMonth); //$NON-NLS-1$
921            fields.put("endTime", endTime); //$NON-NLS-1$
922            fields.put("monthLength", GregorianCalendar.DaysInMonth); //$NON-NLS-1$
923            fields.put("rawOffset", rawOffset); //$NON-NLS-1$
924            fields.put("serialVersionOnStream", 1); //$NON-NLS-1$
925            fields.put("startDay", sStartDay); //$NON-NLS-1$
926            fields.put("startDayOfWeek", sStartDayOfWeek); //$NON-NLS-1$
927            fields.put("startMode", startMode); //$NON-NLS-1$
928            fields.put("startMonth", startMonth); //$NON-NLS-1$
929            fields.put("startTime", startTime); //$NON-NLS-1$
930            fields.put("startYear", startYear); //$NON-NLS-1$
931            fields.put("useDaylight", useDaylight); //$NON-NLS-1$
932            stream.writeFields();
933            stream.writeInt(4);
934            byte[] values = new byte[4];
935            values[0] = (byte) startDay;
936            values[1] = (byte) (startMode == DOM_MODE ? 0 : startDayOfWeek + 1);
937            values[2] = (byte) endDay;
938            values[3] = (byte) (endMode == DOM_MODE ? 0 : endDayOfWeek + 1);
939            stream.write(values);
940        }

942        private void readObject(ObjectInputStream stream) throws IOException,
943                ClassNotFoundException {
944            ObjectInputStream.GetField fields = stream.readFields();
945            rawOffset = fields.get("rawOffset", 0); //$NON-NLS-1$
946            useDaylight = fields.get("useDaylight", false); //$NON-NLS-1$
947            if (useDaylight) {
948                endMonth = fields.get("endMonth", 0); //$NON-NLS-1$
949                endTime = fields.get("endTime", 0); //$NON-NLS-1$
950                startMonth = fields.get("startMonth", 0); //$NON-NLS-1$
951                startTime = fields.get("startTime", 0); //$NON-NLS-1$
952                startYear = fields.get("startYear", 0); //$NON-NLS-1$
953            }
954            if (fields.get("serialVersionOnStream", 0) == 0) { //$NON-NLS-1$
955                if (useDaylight) {
956                    startMode = endMode = DOW_IN_MONTH_MODE;
957                    endDay = fields.get("endDay", 0); //$NON-NLS-1$
958                    endDayOfWeek = fields.get("endDayOfWeek", 0) - 1; //$NON-NLS-1$
959                    startDay = fields.get("startDay", 0); //$NON-NLS-1$
960                    startDayOfWeek = fields.get("startDayOfWeek", 0) - 1; //$NON-NLS-1$
961                }
962            } else {
963                dstSavings = fields.get("dstSavings", 0); //$NON-NLS-1$
964                if (useDaylight) {
965                    endMode = fields.get("endMode", 0); //$NON-NLS-1$
966                    startMode = fields.get("startMode", 0); //$NON-NLS-1$
967                    int length = stream.readInt();
968                    byte[] values = new byte[length];
969                    stream.readFully(values);
970                    if (length >= 4) {
971                        startDay = values[0];
972                        startDayOfWeek = values[1];
973                        if (startMode != DOM_MODE) {
974                            startDayOfWeek--;
975                        }
976                        endDay = values[2];
977                        endDayOfWeek = values[3];
978                        if (endMode != DOM_MODE) {
979                            endDayOfWeek--;
980                        }
981                    }
982                }
983            }
984        }
