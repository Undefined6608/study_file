018    package java.util;
019    
020    /**
021     * A {@code Comparator} is used to compare two objects to determine their ordering with
022     * respect to each other. On a given {@code Collection}, a {@code Comparator} can be used to
023     * obtain a sorted {@code Collection} which is <i>totally ordered</i>. For a {@code Comparator}
024     * to be <i>consistent with equals</i>, its {code #compare(Object, Object)}
025     * method has to return zero for each pair of elements (a,b) where a.equals(b)
026     * holds true. It is recommended that a {@code Comparator} implements
027     * {@link java.io.Serializable}.
028     *  
029     * @since Android 1.0
030     */
031    public interface Comparator<T> {
032        /**
033         * Compares the two specified objects to determine their relative ordering. The ordering
034         * implied by the return value of this method for all possible pairs of
035         * {@code (object1, object2)} should form an <i>equivalence relation</i>.
036         * This means that
037         * <ul>
038         * <li>{@code compare(a,a)} returns zero for all {@code a}</li>
039         * <li>the sign of {@code compare(a,b)} must be the opposite of the sign of {@code
040         * compare(b,a)} for all pairs of (a,b)</li>
041         * <li>From {@code compare(a,b) > 0} and {@code compare(b,c) > 0} it must
042         * follow {@code compare(a,c) > 0} for all possible combinations of {@code
043         * (a,b,c)}</li>
044         * </ul>
045         * 
046         * @param object1
047         *            an {@code Object}.
048         * @param object2
049         *            a second {@code Object} to compare with {@code object1}.
050         * @return an integer < 0 if {@code object1} is less than {@code object2}, 0 if they are
051         *         equal, and > 0 if {@code object1} is greater than {@code object2}.
052         * @exception ClassCastException
053         *                when objects are not of the correct type.
054         */
055        public int compare(T object1, T object2);
056    
057        /**
058         * Compares this {@code Comparator} with the specified {@code Object} and indicates whether they
059         * are equal. In order to be equal, {@code object} must represent the same object
060         * as this instance using a class-specific comparison.
061         * <p>
062         * A {@code Comparator} never needs to override this method, but may choose so for
063         * performance reasons.
064         * 
065         * @param object
066         *            the {@code Object} to compare with this comparator.
067         * @return boolean {@code true} if specified {@code Object} is the same as this
068         *         {@code Object}, and {@code false} otherwise.
069         * @see Object#hashCode
070         * @see Object#equals
071         */
072        public boolean equals(Object object);
073    }
