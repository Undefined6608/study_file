018    package java.util;
019    
020    
021    /**
022     * Observable is used to notify a group of Observer objects when a change
023     * occurs. On creation, the set of observers is empty. After a change occurred,
024     * the application can call the {@link #notifyObservers()} method. This will
025     * cause the invocation of the {@code update()} method of all registered
026     * Observers. The order of invocation is not specified. This implementation will
027     * call the Observers in the order they registered. Subclasses are completely
028     * free in what order they call the update methods.
029     * 
030     * @see Observer
031     * 
032     * @since Android 1.0
033     */
034    public class Observable {
035        
036        Vector<Observer> observers = new Vector<Observer>();
037    
038        boolean changed = false;
039    
040        /**
041         * Constructs a new {@code Observable} object.
042         * 
043         * @since Android 1.0
044         */
045        public Observable() {
046            super();
047        }
048    
049        /**
050         * Adds the specified observer to the list of observers. If it is already
051         * registered, it is not added a second time.
052         * 
053         * @param observer
054         *            the Observer to add.
055         * @since Android 1.0
056         */
057        public synchronized void addObserver(Observer observer) {
058            if (observer == null) {
059                throw new NullPointerException();
060            }
061            if (!observers.contains(observer))
062                observers.addElement(observer);
063        }
064    
065        /**
066         * Clears the changed flag for this {@code Observable}. After calling
067         * {@code clearChanged()}, {@code hasChanged()} will return {@code false}.
068         * 
069         * @since Android 1.0
070         */
071        protected synchronized void clearChanged() {
072            changed = false;
073        }
074    
075        /**
076         * Returns the number of observers registered to this {@code Observable}.
077         * 
078         * @return the number of observers.
079         * @since Android 1.0
080         */
081        public synchronized int countObservers() {
082            return observers.size();
083        }
084    
085        /**
086         * Removes the specified observer from the list of observers. Passing null
087         * won't do anything.
088         * 
089         * @param observer
090         *            the observer to remove.
091         * @since Android 1.0
092         */
093        public synchronized void deleteObserver(Observer observer) {
094            observers.removeElement(observer);
095        }
096    
097        /**
098         * Removes all observers from the list of observers.
099         * 
100         * @since Android 1.0
101         */
102        public synchronized void deleteObservers() {
103            observers.setSize(0);
104        }
105    
106        /**
107         * Returns the changed flag for this {@code Observable}.
108         * 
109         * @return {@code true} when the changed flag for this {@code Observable} is
110         *         set, {@code false} otherwise.
111         * @since Android 1.0
112         */
113        public synchronized boolean hasChanged() {
114            return changed;
115        }
116    
117        /**
118         * If {@code hasChanged()} returns {@code true}, calls the {@code update()}
119         * method for every observer in the list of observers using null as the
120         * argument. Afterwards, calls {@code clearChanged()}.
121         * <p>
122         * Equivalent to calling {@code notifyObservers(null)}.
123         * </p>
124         * 
125         * @since Android 1.0
126         */
127        public void notifyObservers() {
128            notifyObservers(null);
129        }
130    
131        /**
132         * If {@code hasChanged()} returns {@code true}, calls the {@code update()}
133         * method for every Observer in the list of observers using the specified
134         * argument. Afterwards calls {@code clearChanged()}.
135         * 
136         * @param data
137         *            the argument passed to {@code update()}.
138         * @since Android 1.0
139         */
140        @SuppressWarnings("unchecked")
141        public void notifyObservers(Object data) {
142            if (hasChanged()) {
143                // Must clone the vector in case deleteObserver is called
144                Vector<Observer> clone = (Vector<Observer>)observers.clone();
145                int size = clone.size();
146                for (int i = 0; i < size; i++) {
147                    clone.elementAt(i).update(this, data);
148                }
149                clearChanged();
150            }
151        }
152    
153        /**
154         * Sets the changed flag for this {@code Observable}. After calling
155         * {@code setChanged()}, {@code hasChanged()} will return {@code true}.
156         * 
157         * @since Android 1.0
158         */
159        protected synchronized void setChanged() {
160            changed = true;
161        }
162    }
