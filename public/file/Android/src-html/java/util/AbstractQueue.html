017    package java.util;
018    
019    /**
020     * AbstractQueue is an abstract class which implements some of the methods in
021     * {@link Queue}. The provided implementations of {@code add, remove} and
022     * {@code element} are based on {@code offer, poll}, and {@code peek} except
023     * that they throw exceptions to indicate some error instead of returning true
024     * or false.
025     * 
026     * @param <E>
027     *            the type of the element in the collection.
028     * @since Android 1.0
029     */
030    public abstract class AbstractQueue<E> extends AbstractCollection<E> implements
031            Queue<E> {
032    
033        /**
034         * Constructor to be used by subclasses.
035         * 
036         * @since Android 1.0
037         */
038        protected AbstractQueue() {
039            super();
040        }
041    
042        /**
043         * Adds an element to the queue.
044         * 
045         * @param o
046         *            the element to be added to the queue.
047         * @return {@code true} if the operation succeeds, otherwise {@code false}.
048         * @throws IllegalStateException
049         *             if the element is not allowed to be added to the queue.
050         * @since Android 1.0
051         */
052        @Override
053        public boolean add(E o) {
054            if (null == o) {
055                throw new NullPointerException();
056            }
057            if (offer(o)) {
058                return true;
059            }
060            throw new IllegalStateException();
061        }
062    
063        /**
064         * Adds all the elements of a collection to the queue. If the collection is
065         * the queue itself, then an IllegalArgumentException will be thrown. If
066         * during the process, some runtime exception is thrown, then those elements
067         * in the collection which have already successfully been added will remain
068         * in the queue. The result of the method is undefined if the collection is
069         * modified during the process of the method.
070         * 
071         * @param c
072         *            the collection to be added to the queue.
073         * @return {@code true} if the operation succeeds, otherwise {@code false}.
074         * @throws NullPointerException
075         *             if the collection or any element of it is null.
076         * @throws IllegalArgumentException
077         *             If the collection to be added to the queue is the queue
078         *             itself.
079         * @since Android 1.0
080         */
081        @Override
082        public boolean addAll(Collection<? extends E> c) {
083            if (null == c) {
084                throw new NullPointerException();
085            }
086            if (this == c) {
087                throw new IllegalArgumentException();
088            }
089            return super.addAll(c);
090        }
091    
092        /**
093         * Removes the element at the head of the queue and returns it.
094         * 
095         * @return the element at the head of the queue.
096         * @throws NoSuchElementException
097         *             if the queue is empty.
098         * @since Android 1.0
099         */
100        public E remove() {
101            E o = poll();
102            if (null == o) {
103                throw new NoSuchElementException();
104            }
105            return o;
106        }
107    
108        /**
109         * Returns but does not remove the element at the head of the queue.
110         * 
111         * @return the element at the head of the queue.
112         * @throws NoSuchElementException
113         *             if the queue is empty.
114         * @since Android 1.0
115         */
116        public E element() {
117            E o = peek();
118            if (null == o) {
119                throw new NoSuchElementException();
120            }
121            return o;
122        }
123    
124        /**
125         * Removes all elements of the queue, leaving it empty.
126         * 
127         * @since Android 1.0
128         */
129        @Override
130        public void clear() {
131            E o;
132            do {
133                o = poll();
134            } while (null != o);
135        }
136    }
