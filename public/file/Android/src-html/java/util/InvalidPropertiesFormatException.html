018    package java.util;
019    
020    import java.io.NotSerializableException;
021    import java.io.ObjectInputStream;
022    import java.io.ObjectOutputStream;
023    import java.io.IOException;
024    
025    /**
026     * An {@code InvalidPropertiesFormatException} is thrown if loading the XML
027     * document defining the properties does not follow the {@code Properties}
028     * specification.
029     * 
030     * Even though this Exception inherits the {@code Serializable} interface, it is not
031     * serializable. The methods used for serialization throw
032     * {@code NotSerializableException}s.
033     * 
034     * @since Android 1.0
035     */
036    public class InvalidPropertiesFormatException extends IOException {
037        
038        private static final long serialVersionUID = 7763056076009360219L;
039    
040        /**
041         * Constructs a new {@code InvalidPropertiesFormatException} with the
042         * current stack trace and message filled in.
043         * 
044         * @param m
045         *           the detail message for the exception.
046         */
047        public InvalidPropertiesFormatException(String m) {
048            super(m);
049        }
050    
051        /**
052         * Constructs a new {@code InvalidPropertiesFormatException} with the cause
053         * for the Exception.
054         * 
055         * @param c
056         *           the cause for the Exception.
057         */
058        public InvalidPropertiesFormatException(Throwable c) {
059            initCause(c);
060        }
061        
062        private void writeObject(ObjectOutputStream out) 
063                throws NotSerializableException{
064            throw new NotSerializableException();        
065        }
066        
067        private void readObject(ObjectInputStream in) 
068                throws NotSerializableException{
069            throw new NotSerializableException();        
070        }
071    }
