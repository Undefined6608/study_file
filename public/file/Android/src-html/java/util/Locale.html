050     * @since Android 1.0

263        /**
264         * Constructs a new {@code Locale} using the specified language, country, and
265         * variant codes.
266         * 
267         * @param language
268         *            the language this {@code Locale} represents.
269         * @param country
270         *            the country this {@code Locale} represents.
271         * @param variant
272         *            the variant this {@code Locale} represents.
273         * @throws NullPointerException
274         *             if {@code language}, {@code country}, or
275         *             {@code variant} is {@code null}.
276         * @since Android 1.0
277         */
278        public Locale(String language, String country, String variant) {
279            if (language == null || country == null || variant == null) {
280                throw new NullPointerException();
281            }
282            languageCode = Util.toASCIILowerCase(language);
283            // Map new language codes to the obsolete language
284            // codes so the correct resource bundles will be used.
285            if (languageCode.equals("he")) {//$NON-NLS-1$
286                languageCode = "iw"; //$NON-NLS-1$
287            } else if (languageCode.equals("id")) {//$NON-NLS-1$
288                languageCode = "in"; //$NON-NLS-1$
289            } else if (languageCode.equals("yi")) {//$NON-NLS-1$
290                languageCode = "ji"; //$NON-NLS-1$
291            }
292    
293            // countryCode is defined in ASCII character set
294            countryCode = Util.toASCIIUpperCase(country);
295    
296            variantCode = variant;
297        }

316        /**
317         * Compares the specified object to this {@code Locale} and returns whether they are
318         * equal. The object must be an instance of {@code Locale} and have the same
319         * language, country and variant.
320         * 
321         * @param object
322         *            the object to compare with this object.
323         * @return {@code true} if the specified object is equal to this {@code Locale},
324         *         {@code false} otherwise.
325         * @see #hashCode
326         * @since Android 1.0
327         */
328        @Override
329        public boolean equals(Object object) {
330            if (object == this) {
331                return true;
332            }
333            if (object instanceof Locale) {
334                Locale o = (Locale) object;
335                return languageCode.equals(o.languageCode)
336                        && countryCode.equals(o.countryCode)
337                        && variantCode.equals(o.variantCode);
338            }
339            return false;
340        }

432        // BEGIN android-added
433        static Locale[] find() {
434            String[] locales = Resources.getAvailableLocales();
435            ArrayList<Locale> temp = new ArrayList<Locale>();
436            for (int i = 0; i < locales.length; i++) {
437                String s = locales[i];
438                int first = s.indexOf('_');
439                int second = s.indexOf('_', first + 1);
440                
441                if (first == -1) {
442                    // Language only
443                    temp.add(new Locale(s));
444                } else if (second == -1) {
445                    // Language and country
446                    temp.add(new Locale(s.substring(0, first), s.substring(first + 1)));
447                } else {
448                    // Language and country and variant
449                    temp.add(new Locale(s.substring(0, first), s.substring(first + 1, second), s.substring(second + 1)));
450                }
451            }
452            Locale[] result = new Locale[temp.size()];
453            return temp.toArray(result);
454        }

516         * Gets the full country name in the specified {@code Locale} for the country code

520         * @param locale
521         *            the {@code Locale} for which the display name is retrieved.
522         * @return a country name.
523         * @since Android 1.0
524         */
525        public String getDisplayCountry(Locale locale) {
526            if (countryCode.length() == 0) {
527                return countryCode;
528            }
529            try {
530                // First try the specified locale
531                ResourceBundle bundle = getBundle("Country", locale); //$NON-NLS-1$
532                // BEGIN android-changed
533                String result = bundle.getString(this.toString());
534                // END android-changed
535                if (result != null) {
536                    return result;
537                }
538                // Now use the default locale
539                if (locale != Locale.getDefault()) {
540                    bundle = getBundle("Country", Locale.getDefault()); //$NON-NLS-1$
541                }
542                return bundle.getString(countryCode);
543            } catch (MissingResourceException e) {
544                return countryCode;
545            }
546        }

604        /**
605         * Gets the full language, country, and variant names in the specified
606         * Locale for the codes of this {@code Locale}.
607         * 
608         * @param locale
609         *            the {@code Locale} for which the display name is retrieved.
610         * @return a {@code Locale} name.
611         * @since Android 1.0
612         */
613        public String getDisplayName(Locale locale) {
614            int count = 0;
615            StringBuffer buffer = new StringBuffer();
616            if (languageCode.length() > 0) {
617                buffer.append(getDisplayLanguage(locale));
618                count++;
619            }
620            if (countryCode.length() > 0) {
621                if (count == 1) {
622                    buffer.append(" ("); //$NON-NLS-1$
623                }
624                buffer.append(getDisplayCountry(locale));
625                count++;
626            }
627            if (variantCode.length() > 0) {
628                if (count == 1) {
629                    buffer.append(" ("); //$NON-NLS-1$
630                } else if (count == 2) {
631                    buffer.append(","); //$NON-NLS-1$
632                }
633                buffer.append(getDisplayVariant(locale));
634                count++;
635            }
636            if (count > 1) {
637                buffer.append(")"); //$NON-NLS-1$
638            }
639            return buffer.toString();
640        }

859        /**
860         * Returns the string representation of this {@code Locale}. It consists of the
861         * language followed by the country and at the end the variant. They are
862         * separated by underscores. If the language is missing the string begins
863         * with an underscore. If the country is missing there are 2 underscores
864         * between the language and the variant. the variant alone canot be defined
865         * without a language and/or a country (in this case this method would
866         * return the empty string).
867         * 
868         * Examples: "en", "en_US", "_US", "en__POSIX", "en_US_POSIX"
869         * 
870         * @return the string representation of this {@code Locale}.
871         * @since Android 1.0
872         */
873        @Override
874        public final String toString() {
875            StringBuilder result = new StringBuilder();
876            result.append(languageCode);
877            if (countryCode.length() > 0) {
878                result.append('_');
879                result.append(countryCode);
880            }
881            if (variantCode.length() > 0 && result.length() > 0 ) {
882                if (0 == countryCode.length()) {
883                    result.append("__"); //$NON-NLS-1$
884                } else {
885                    result.append('_');
886                }
887                result.append(variantCode);
888            }
889            return result.toString();
890        }
