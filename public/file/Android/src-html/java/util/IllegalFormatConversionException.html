017    package java.util;
018    
019    import java.io.Serializable;
020    
021    /**
022     * An {@code IllegalFormatConversionException} will be thrown when the parameter
023     * is incompatible with the corresponding format specifier.
024     * 
025     * @see java.lang.RuntimeException
026     * @since Android 1.0
027     */
028    public class IllegalFormatConversionException extends IllegalFormatException
029            implements Serializable {
030        private static final long serialVersionUID = 17000126L;
031    
032        private char c;
033    
034        private Class<?> arg;
035    
036        /**
037         * Constructs a new {@code IllegalFormatConversionException} with the class
038         * of the mismatched conversion and corresponding parameter.
039         * 
040         * @param c
041         *           the class of the mismatched conversion.
042         * @param arg
043         *           the corresponding parameter.
044         */
045        public IllegalFormatConversionException(char c, Class<?> arg) {
046            this.c = c;
047            if (arg == null) {
048                throw new NullPointerException();
049            }
050            this.arg = arg;
051        }
052    
053        /**
054         * Returns the class of the mismatched parameter.
055         * 
056         * @return the class of the mismatched parameter.
057         */
058        public Class<?> getArgumentClass() {
059            return arg;
060        }
061    
062        /**
063         * Returns the incompatible conversion.
064         * 
065         * @return the incompatible conversion.
066         */
067        public char getConversion() {
068            return c;
069        }
070    
071        /**
072         * Returns the message string of the IllegalFormatConversionException.
073         * 
074         * @return the message string of the IllegalFormatConversionException.
075         */
076        @Override
077        public String getMessage() {
078            StringBuilder buffer = new StringBuilder();
079            buffer.append(c);
080            buffer.append(" is incompatible with ");
081            buffer.append(arg.getName());
082            return buffer.toString();
083        }
084    
085    }
