025    /**
026     * This class is an abstract implementation of the {@code Map} interface. This
027     * implementation does not support adding. A subclass must implement the
028     * abstract method entrySet().
029     * 
030     * @since Android 1.0
031     */
032    public abstract class AbstractMap<K, V> implements Map<K, V> {
033    
034        // Lazily initialized key set.
035        Set<K> keySet;
036    
037        Collection<V> valuesCollection;
038    
039        /**
040         * Constructs a new instance of this {@code AbstractMap}.
041         * 
042         * @since Android 1.0
043         */
044        protected AbstractMap() {
045            super();
046        }
047    
048        /**
049         * Removes all elements from this map, leaving it empty.
050         * 
051         * @throws UnsupportedOperationException
052         *                if removing from this map is not supported.
053         * @see #isEmpty()
054         * @see #size()
055         * @since Android 1.0
056         */
057        public void clear() {
058            entrySet().clear();
059        }
060    
061        /**
062         * Returns whether this map contains the specified key.
063         * 
064         * @param key
065         *            the key to search for.
066         * @return {@code true} if this map contains the specified key,
067         *         {@code false} otherwise.
068         * @since Android 1.0
069         */
070        public boolean containsKey(Object key) {
071            Iterator<Map.Entry<K, V>> it = entrySet().iterator();
072            if (key != null) {
073                while (it.hasNext()) {
074                    if (key.equals(it.next().getKey())) {
075                        return true;
076                    }
077                }
078            } else {
079                while (it.hasNext()) {
080                    if (it.next().getKey() == null) {
081                        return true;
082                    }
083                }
084            }
085            return false;
086        }
087    
088        /**
089         * Returns whether this map contains the specified value.
090         * 
091         * @param value
092         *            the value to search for.
093         * @return {@code true} if this map contains the specified value,
094         *         {@code false} otherwise.
095         * @since Android 1.0
096         */
097        public boolean containsValue(Object value) {
098            Iterator<Map.Entry<K, V>> it = entrySet().iterator();
099            if (value != null) {
100                while (it.hasNext()) {
101                    if (value.equals(it.next().getValue())) {
102                        return true;
103                    }
104                }
105            } else {
106                while (it.hasNext()) {
107                    if (it.next().getValue() == null) {
108                        return true;
109                    }
110                }
111            }
112            return false;
113        }
114    
115        /**
116         * Returns a set containing all of the mappings in this map. Each mapping is
117         * an instance of {@link Map.Entry}. As the set is backed by this map,
118         * changes in one will be reflected in the other.
119         * 
120         * @return a set of the mappings.
121         * @since Android 1.0
122         */
123        public abstract Set<Map.Entry<K, V>> entrySet();
124    
125        /**
126         * Compares the specified object to this instance, and returns {@code true}
127         * if the specified object is a map and both maps contain the same mappings.
128         * 
129         * @param object
130         *            the object to compare with this object.
131         * @return boolean {@code true} if the object is the same as this object,
132         *         and {@code false} if it is different from this object.
133         * @see #hashCode()
134         * @see #entrySet()
135         * @since Android 1.0
136         */
137        @Override
138        public boolean equals(Object object) {
139            if (this == object) {
140                return true;
141            }
142            if (object instanceof Map) {
143                Map<?, ?> map = (Map<?, ?>) object;
144                if (size() != map.size()) {
145                    return false;
146                }
147    
148                // BEGIN android-changed
149                // copied from newer version of harmony
150                Iterator<Map.Entry<K, V>> it = entrySet().iterator();
151                while (it.hasNext()) {
152                    Entry<K, V> entry = it.next();
153                    K key = entry.getKey();
154                    V value = entry.getValue();
155                    Object obj = map.get(key);
156                    if( null != obj && (!obj.equals(value)) || null == obj && obj != value) {
157                        return false;
158                    }
159                }
160                // END android-changed
161                return true;
162            }
163            return false;
164        }
165    
166        /**
167         * Returns the value of the mapping with the specified key.
168         * 
169         * @param key
170         *            the key.
171         * @return the value of the mapping with the specified key, or {@code null}
172         *         if no mapping for the specified key is found.
173         * @since Android 1.0
174         */
175        public V get(Object key) {
176            Iterator<Map.Entry<K, V>> it = entrySet().iterator();
177            if (key != null) {
178                while (it.hasNext()) {
179                    Map.Entry<K, V> entry = it.next();
180                    if (key.equals(entry.getKey())) {
181                        return entry.getValue();
182                    }
183                }
184            } else {
185                while (it.hasNext()) {
186                    Map.Entry<K, V> entry = it.next();
187                    if (entry.getKey() == null) {
188                        return entry.getValue();
189                    }
190                }
191            }
192            return null;
193        }
194    
195        /**
196         * Returns the hash code for this object. Objects which are equal must
197         * return the same value for this method.
198         * 
199         * @return the hash code of this object.
200         * @see #equals(Object)
201         * @since Android 1.0
202         */
203        @Override
204        public int hashCode() {
205            int result = 0;
206            Iterator<Map.Entry<K, V>> it = entrySet().iterator();
207            while (it.hasNext()) {
208                result += it.next().hashCode();
209            }
210            return result;
211        }
212    
213        /**
214         * Returns whether this map is empty.
215         * 
216         * @return {@code true} if this map has no elements, {@code false}
217         *         otherwise.
218         * @see #size()
219         * @since Android 1.0
220         */
221        public boolean isEmpty() {
222            return size() == 0;
223        }
224    
225        /**
226         * Returns a set of the keys contained in this map. The set is backed by
227         * this map so changes to one are reflected by the other. The returned set
228         * does not support adding.
229         * 
230         * @return a set of the keys.
231         * @since Android 1.0
232         */
233        public Set<K> keySet() {
234            if (keySet == null) {
235                keySet = new AbstractSet<K>() {
236                    @Override
237                    public boolean contains(Object object) {
238                        return containsKey(object);
239                    }
240    
241                    @Override
242                    public int size() {
243                        return AbstractMap.this.size();
244                    }
245    
246                    @Override
247                    public Iterator<K> iterator() {
248                        return new Iterator<K>() {
249                            Iterator<Map.Entry<K, V>> setIterator = entrySet()
250                                    .iterator();
251    
252                            public boolean hasNext() {
253                                return setIterator.hasNext();
254                            }
255    
256                            public K next() {
257                                return setIterator.next().getKey();
258                            }
259    
260                            public void remove() {
261                                setIterator.remove();
262                            }
263                        };
264                    }
265                };
266            }
267            return keySet;
268        }
269    
270        /**
271         * Maps the specified key to the specified value.
272         * 
273         * @param key
274         *            the key.
275         * @param value
276         *            the value.
277         * @return the value of any previous mapping with the specified key or
278         *         {@code null} if there was no mapping.
279         * @throws UnsupportedOperationException
280         *                if adding to this map is not supported.
281         * @throws ClassCastException
282         *                if the class of the key or value is inappropriate for this
283         *                map.
284         * @throws IllegalArgumentException
285         *                if the key or value cannot be added to this map.
286         * @throws NullPointerException
287         *                if the key or value is {@code null} and this Map does not
288         *                support {@code null} keys or values.
289         * @since Android 1.0
290         */
291        public V put(K key, V value) {
292            throw new UnsupportedOperationException();
293        }
294    
295        /**
296         * Copies every mapping in the specified map to this map.
297         * 
298         * @param map
299         *            the map to copy mappings from.
300         * @throws UnsupportedOperationException
301         *                if adding to this map is not supported.
302         * @throws ClassCastException
303         *                if the class of a key or value is inappropriate for this
304         *                map.
305         * @throws IllegalArgumentException
306         *                if a key or value cannot be added to this map.
307         * @throws NullPointerException
308         *                if a key or value is {@code null} and this map does not
309         *                support {@code null} keys or values.
310         * @since Android 1.0
311         */
312        public void putAll(Map<? extends K, ? extends V> map) {
313            for (Map.Entry<? extends K, ? extends V> entry : map.entrySet()) {
314                put(entry.getKey(), entry.getValue());
315            }
316        }
317    
318        /**
319         * Removes a mapping with the specified key from this Map.
320         * 
321         * @param key
322         *            the key of the mapping to remove.
323         * @return the value of the removed mapping or {@code null} if no mapping
324         *         for the specified key was found.
325         * @throws UnsupportedOperationException
326         *                if removing from this map is not supported.
327         * @since Android 1.0
328         */
329        public V remove(Object key) {
330            Iterator<Map.Entry<K, V>> it = entrySet().iterator();
331            if (key != null) {
332                while (it.hasNext()) {
333                    Map.Entry<K, V> entry = it.next();
334                    if (key.equals(entry.getKey())) {
335                        it.remove();
336                        return entry.getValue();
337                    }
338                }
339            } else {
340                while (it.hasNext()) {
341                    Map.Entry<K, V> entry = it.next();
342                    if (entry.getKey() == null) {
343                        it.remove();
344                        return entry.getValue();
345                    }
346                }
347            }
348            return null;
349        }
350    
351        /**
352         * Returns the number of elements in this map.
353         * 
354         * @return the number of elements in this map.
355         * @since Android 1.0
356         */
357        public int size() {
358            return entrySet().size();
359        }
360    
361        /**
362         * Returns the string representation of this map.
363         * 
364         * @return the string representation of this map.
365         * @since Android 1.0
366         */
367        @Override
368        public String toString() {
369            if (isEmpty()) {
370                return "{}"; //$NON-NLS-1$
371            }
372    
373            StringBuilder buffer = new StringBuilder(size() * 28);
374            buffer.append('{');
375            Iterator<Map.Entry<K, V>> it = entrySet().iterator();
376            while (it.hasNext()) {
377                Map.Entry<K, V> entry = it.next();
378                Object key = entry.getKey();
379                if (key != this) {
380                    buffer.append(key);
381                } else {
382                    buffer.append("(this Map)"); //$NON-NLS-1$
383                }
384                buffer.append('=');
385                Object value = entry.getValue();
386                if (value != this) {
387                    buffer.append(value);
388                } else {
389                    buffer.append("(this Map)"); //$NON-NLS-1$
390                }
391                if (it.hasNext()) {
392                    buffer.append(", "); //$NON-NLS-1$
393                }
394            }
395            buffer.append('}');
396            return buffer.toString();
397        }

399        /**
400         * Returns a collection of the values contained in this map. The collection
401         * is backed by this map so changes to one are reflected by the other. The
402         * collection supports remove, removeAll, retainAll and clear operations,
403         * and it does not support add or addAll operations.
404         * <p>
405         * This method returns a collection which is the subclass of
406         * AbstractCollection. The iterator method of this subclass returns a
407         * "wrapper object" over the iterator of map's entrySet(). The {@code size}
408         * method wraps the map's size method and the {@code contains} method wraps
409         * the map's containsValue method.
410         * </p>
411         * <p>
412         * The collection is created when this method is called for the first time
413         * and returned in response to all subsequent calls. This method may return
414         * different collections when multiple concurrent calls occur to this
415         * method, since no synchronization is performed.
416         * </p>
417         * 
418         * @return a collection of the values contained in this map.
419         * @since Android 1.0
420         */
421        public Collection<V> values() {
422            if (valuesCollection == null) {
423                valuesCollection = new AbstractCollection<V>() {
424                    @Override
425                    public int size() {
426                        return AbstractMap.this.size();
427                    }
428    
429                    @Override
430                    public boolean contains(Object object) {
431                        return containsValue(object);
432                    }
433    
434                    @Override
435                    public Iterator<V> iterator() {
436                        return new Iterator<V>() {
437                            Iterator<Map.Entry<K, V>> setIterator = entrySet()
438                                    .iterator();
439    
440                            public boolean hasNext() {
441                                return setIterator.hasNext();
442                            }
443    
444                            public V next() {
445                                return setIterator.next().getValue();
446                            }
447    
448                            public void remove() {
449                                setIterator.remove();
450                            }
451                        };
452                    }
453                };
454            }
455            return valuesCollection;
456        }

458        /**
459         * Returns a new instance of the same class as this instance, whose slots
460         * have been filled in with the values of the slots of this instance.
461         * 
462         * @return a shallow copy of this object.
463         * @throws CloneNotSupportedException
464         *                if the receiver's class does not implement the interface
465         *                {@code Cloneable}.
466         * @since Android 1.0
467         */
468        @Override
469        @SuppressWarnings("unchecked")
470        protected Object clone() throws CloneNotSupportedException {
471            AbstractMap<K, V> result = (AbstractMap<K, V>) super.clone();
472            result.keySet = null;
473            result.valuesCollection = null;
474            return result;
475        }
