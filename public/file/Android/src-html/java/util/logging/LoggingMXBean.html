018    package java.util.logging;
019    
020    import java.util.List;
021    
022    /**
023     * {@code LoggingMXBean} is the management interface for the logging sub-system.
024     * <p>
025     * The ObjectName for identifying the {@code LoggingMXBean} in a bean server is
026     * {@link LogManager#LOGGING_MXBEAN_NAME}.
027     * </p>
028     * 
029     * @since Android 1.0
030     */
031    public interface LoggingMXBean {
032        /**
033         * Gets the string value of the logging level of a logger. An empty string
034         * is returned when the logger's level is defined by its parent. A
035         * {@code null} is returned if the specified logger does not exist.
036         * 
037         * @param loggerName
038         *            the name of the logger lookup.
039         * @return a {@code String} if the logger is found, otherwise {@code null}.
040         * @see Level#getName()
041         * @since Android 1.0
042         */
043        String getLoggerLevel(String loggerName);
044    
045        /**
046         * Gets a list of all currently registered logger names. This is performed
047         * using the {@link LogManager#getLoggerNames()}.
048         * 
049         * @return a list of logger names.
050         * @since Android 1.0
051         */
052        List<String> getLoggerNames();
053    
054        /**
055         * Gets the name of the parent logger of a logger. If the logger doesn't
056         * exist then {@code null} is returned. If the logger is the root logger,
057         * then an empty {@code String} is returned.
058         * 
059         * @param loggerName
060         *            the name of the logger to lookup.
061         * @return a {@code String} if the logger was found, otherwise {@code null}.
062         * @since Android 1.0
063         */
064        String getParentLoggerName(String loggerName);
065    
066        /**
067         * Sets the log level of a logger. LevelName set to {@code null} means the
068         * level is inherited from the nearest non-null ancestor.
069         * 
070         * @param loggerName
071         *            the name of the logger to set the level on, which must not be
072         *            {@code null}.
073         * @param levelName
074         *            the level to set on the logger, which may be {@code null}.
075         * @throws IllegalArgumentException
076         *             if {@code loggerName} is not a registered logger or if
077         *             {@code levelName} is not null and not valid.
078         * @throws SecurityException
079         *             if a security manager exists and the caller doesn't have
080         *             LoggingPermission("control").
081         * @see Level#parse(String)
082         * @since Android 1.0
083         */
084        void setLoggerLevel(String loggerName, String levelName);
085    }
