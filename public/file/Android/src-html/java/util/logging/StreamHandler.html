019    package java.util.logging;
020    
021    import java.io.OutputStream;
022    import java.io.OutputStreamWriter;
023    import java.io.UnsupportedEncodingException;
024    import java.io.Writer;
025    
026    import org.apache.harmony.logging.internal.nls.Messages;
027    
028    /**
029     * A {@code StreamHandler} object writes log messages to an output stream, that
030     * is, objects of the class {@link java.io.OutputStream}.
031     * <p>
032     * A {@code StreamHandler} object reads the following properties from the log
033     * manager to initialize itself:
034     * <ul>
035     * <li>java.util.logging.StreamHandler.level specifies the logging level,
036     * defaults to {@code Level.INFO} if this property is not found or has an
037     * invalid value.
038     * <li>java.util.logging.StreamHandler.filter specifies the name of the filter
039     * class to be associated with this handler, defaults to {@code null} if this
040     * property is not found or has an invalid value.
041     * <li>java.util.logging.StreamHandler.formatter specifies the name of the
042     * formatter class to be associated with this handler, defaults to
043     * {@code java.util.logging.SimpleFormatter} if this property is not found or
044     * has an invalid value.
045     * <li>java.util.logging.StreamHandler.encoding specifies the encoding this
046     * handler will use to encode log messages, defaults to {@code null} if this
047     * property is not found or has an invalid value.
048     * </ul>
049     * </p>
050     * <p>
051     * This class is not thread-safe.
052     * </p>
053     * 
054     * @since Android 1.0
055     */
056    public class StreamHandler extends Handler {
057        // the output stream this handler writes to
058        private OutputStream os;
059    
060        // the writer that writes to the output stream
061        private Writer writer;
062    
063        // the flag indicating whether the writer has been initialized
064        private boolean writerNotInitialized;
065    
066        /**
067         * Constructs a {@code StreamHandler} object. The new stream handler
068         * does not have an associated output stream.
069         * 
070         * @since Android 1.0
071         */
072        public StreamHandler() {
073            initProperties("INFO", null, "java.util.logging.SimpleFormatter",  //$NON-NLS-1$//$NON-NLS-2$
074                    null);
075            this.os = null;
076            this.writer = null;
077            this.writerNotInitialized = true;
078        }
079    
080        /**
081         * Constructs a {@code StreamHandler} object with the supplied output
082         * stream. Default properties are read.
083         * 
084         * @param os
085         *            the output stream this handler writes to.
086         */
087        StreamHandler(OutputStream os) {
088            this();
089            this.os = os;
090        }
091    
092        /**
093         * Constructs a {@code StreamHandler} object. The specified default values
094         * will be used if the corresponding properties are not found in the log
095         * manager's properties.
096         */
097        StreamHandler(String defaultLevel, String defaultFilter,
098                String defaultFormatter, String defaultEncoding) {
099            initProperties(defaultLevel, defaultFilter, defaultFormatter,
100                    defaultEncoding);
101            this.os = null;
102            this.writer = null;
103            this.writerNotInitialized = true;
104        }
105    
106        /**
107         * Constructs a {@code StreamHandler} object with the supplied output stream
108         * and formatter.
109         * 
110         * @param os
111         *            the output stream this handler writes to.
112         * @param formatter
113         *            the formatter this handler uses to format the output.
114         * @throws NullPointerException
115         *             if {@code os} or {@code formatter} is {@code null}.
116         * @since Android 1.0
117         */
118        public StreamHandler(OutputStream os, Formatter formatter) {
119            this();
120            if (os == null) {
121                // logging.2=The OutputStream parameter is null
122                throw new NullPointerException(Messages.getString("logging.2")); //$NON-NLS-1$
123            }
124            if (formatter == null) {
125                // logging.3=The Formatter parameter is null.
126                throw new NullPointerException(Messages.getString("logging.3")); //$NON-NLS-1$
127            }
128            this.os = os;
129            internalSetFormatter(formatter);
130        }
131    
132        // initialize the writer
133        private void initializeWritter() {
134            this.writerNotInitialized = false;
135            if (null == getEncoding()) {
136                this.writer = new OutputStreamWriter(this.os);
137            } else {
138                try {
139                    this.writer = new OutputStreamWriter(this.os, getEncoding());
140                } catch (UnsupportedEncodingException e) {
141                    /*
142                     * Should not happen because it's checked in
143                     * super.initProperties().
144                     */
145                }
146            }
147            write(getFormatter().getHead(this));
148        }
149    
150        // Write a string to the output stream.
151        private void write(String s) {
152            try {
153                this.writer.write(s);
154            } catch (Exception e) {
155                // logging.14=Exception occurred when writing to the output stream.
156                getErrorManager().error(Messages.getString("logging.14"), e, //$NON-NLS-1$
157                        ErrorManager.WRITE_FAILURE);
158            }
159        }
160    
161        /**
162         * Sets the output stream this handler writes to. Note it does nothing else.
163         * 
164         * @param newOs
165         *            the new output stream
166         */
167        void internalSetOutputStream(OutputStream newOs) {
168            this.os = newOs;
169        }
170    
171        
172        /**
173         * Sets the output stream this handler writes to. If there's an existing
174         * output stream, the tail string of the associated formatter will be
175         * written to it. Then it will be flushed, closed and replaced with 
176         * {@code os}.
177         * 
178         * @param os
179         *            the new output stream.
180         * @throws SecurityException
181         *             if a security manager determines that the caller does not
182         *             have the required permission.
183         * @throws NullPointerException
184         *             if {@code os} is {@code null}.
185         */
186        protected void setOutputStream(OutputStream os) {
187            if (null == os) {
188                throw new NullPointerException();
189            }
190            LogManager.getLogManager().checkAccess();
191            close(true);
192            this.writer = null;
193            this.os = os;
194            this.writerNotInitialized = true;
195        }
196    
197        /**
198         * Sets the character encoding used by this handler. A {@code null} value
199         * indicates that the default encoding should be used.
200         * 
201         * @param encoding
202         *            the character encoding to set.
203         * @throws SecurityException
204         *             if a security manager determines that the caller does not
205         *             have the required permission.
206         * @throws UnsupportedEncodingException
207         *             if the specified encoding is not supported by the runtime.
208         * @since Android 1.0
209         */
210        @Override
211        public void setEncoding(String encoding) throws SecurityException,
212                UnsupportedEncodingException {
213            //flush first before set new encoding
214            this.flush();
215            super.setEncoding(encoding);
216            // renew writer only if the writer exists
217            if (null != this.writer) {
218                if (null == getEncoding()) {
219                    this.writer = new OutputStreamWriter(this.os);
220                } else {
221                    try {
222                        this.writer = new OutputStreamWriter(this.os, getEncoding());
223                    } catch (UnsupportedEncodingException e) {
224                        /*
225                         * Should not happen because it's checked in
226                         * super.initProperties().
227                         */
228                        throw new AssertionError(e);
229                    }
230                }
231            }
232        }
233    
234        /**
235         * Closes this handler, but the underlying output stream is only closed if
236         * {@code closeStream} is {@code true}. Security is not checked.
237         * 
238         * @param closeStream
239         *            whether to close the underlying output stream.
240         */
241        void close(boolean closeStream) {
242            if (null != this.os) {
243                if (this.writerNotInitialized) {
244                    initializeWritter();
245                }
246                write(getFormatter().getTail(this));
247                try {
248                    this.writer.flush();
249                    if (closeStream) {
250                        this.writer.close();
251                        this.writer = null;
252                        this.os = null;
253                    }
254                } catch (Exception e) {
255                    // logging.15=Exception occurred when closing the output stream.
256                    getErrorManager().error(Messages.getString("logging.15"), e, //$NON-NLS-1$
257                            ErrorManager.CLOSE_FAILURE);
258                }
259            }
260        }
261    
262        /**
263         * Closes this handler. The tail string of the formatter associated with
264         * this handler is written out. A flush operation and a subsequent close
265         * operation is then performed upon the output stream. Client applications
266         * should not use a handler after closing it.
267         * 
268         * @throws SecurityException
269         *             if a security manager determines that the caller does not
270         *             have the required permission.
271         * @since Android 1.0
272         */
273        @Override
274        public void close() {
275            LogManager.getLogManager().checkAccess();
276            close(true);
277        }
278    
279        /**
280         * Flushes any buffered output.
281         * 
282         * @since Android 1.0
283         */
284        @Override
285        public void flush() {
286            if (null != this.os) {
287                try {
288                    if (null != this.writer) {
289                        this.writer.flush();
290                    } else {
291                        this.os.flush();
292                    }
293                } catch (Exception e) {
294                    // logging.16=Exception occurred while flushing the output stream.
295                    getErrorManager().error(Messages.getString("logging.16"), //$NON-NLS-1$
296                            e, ErrorManager.FLUSH_FAILURE);
297                }
298            }
299        }
300    
301        /**
302         * Accepts a logging request. The log record is formatted and written to the
303         * output stream if the following three conditions are met:
304         * <ul>
305         * <li>the supplied log record has at least the required logging level;
306         * <li>the supplied log record passes the filter associated with this
307         * handler, if any;
308         * <li>the output stream associated with this handler is not {@code null}.
309         * </ul>
310         * If it is the first time a log record is written out, the head string of
311         * the formatter associated with this handler is written out first.
312         * 
313         * @param record
314         *            the log record to be logged.
315         * @since Android 1.0
316         */
317        @Override
318        public synchronized void publish(LogRecord record) {
319            try {
320                if (this.isLoggable(record)) {
321                    if (this.writerNotInitialized) {
322                        initializeWritter();
323                    }
324                    String msg = null;
325                    try {
326                        msg = getFormatter().format(record);
327                    } catch (Exception e) {
328                        // logging.17=Exception occurred while formatting the log record.
329                        getErrorManager().error(Messages.getString("logging.17"), //$NON-NLS-1$
330                                e, ErrorManager.FORMAT_FAILURE);
331                    }
332                    write(msg);
333                }
334            } catch (Exception e) {
335                // logging.18=Exception occurred while logging the record.
336                getErrorManager().error(Messages.getString("logging.18"), e, //$NON-NLS-1$
337                        ErrorManager.GENERIC_FAILURE);
338            }
339        }
340    
341        /**
342         * Determines whether the supplied log record needs to be logged. The
343         * logging levels are checked as well as the filter. The output stream of
344         * this handler is also checked. If it is {@code null}, this method returns
345         * {@code false}.
346         * <p>
347         * Notice : Case of no output stream will return {@code false}.
348         * </p>
349         * 
350         * @param record
351         *            the log record to be checked.
352         * @return {@code true} if {@code record} needs to be logged, {@code false}
353         *         otherwise.
354         * @since Android 1.0
355         */
356        @Override
357        public boolean isLoggable(LogRecord record) {
358            if (null == record) {
359                return false;
360            }
361            if (null != this.os && super.isLoggable(record)) {
362                return true;
363            }
364            return false;
365        }
