029    /**
030     * A {@code LogRecord} object represents a logging request. It is passed between
031     * the logging framework and individual logging handlers. Client applications
032     * should not modify a {@code LogRecord} object that has been passed into the
033     * logging framework.
034     * <p>
035     * The {@code LogRecord} class will infer the source method name and source
036     * class name the first time they are accessed if the client application didn't
037     * specify them explicitly. This automatic inference is based on the analysis of
038     * the call stack and is not guaranteed to be precise. Client applications
039     * should force the initialization of these two fields by calling
040     * {@code getSourceClassName} or {@code getSourceMethodName} if they expect to
041     * use them after passing the {@code LogRecord} object to another thread or
042     * transmitting it over RMI.
043     * </p>
044     * 
045     * @since Android 1.0
046     */
047    public class LogRecord implements Serializable {

145        /**
146         * Constructs a {@code LogRecord} object using the supplied the logging
147         * level and message. The millis property is set to the current time. The
148         * sequence property is set to a new unique value, allocated in increasing
149         * order within the virtual machine. The thread ID is set to a unique value
150         * for the current thread. All other properties are set to {@code null}.
151         * 
152         * @param level
153         *            the logging level, may not be {@code null}.
154         * @param msg
155         *            the raw message.
156         * @throws NullPointerException
157         *             if {@code level} is {@code null}.
158         * @since Android 1.0
159         */
160        public LogRecord(Level level, String msg) {
161            if (null == level) {
162                // logging.4=The 'level' parameter is null.
163                throw new NullPointerException(Messages.getString("logging.4")); //$NON-NLS-1$
164            }
165            this.level = level;
166            this.message = msg;
167            this.millis = System.currentTimeMillis();
168    
169            synchronized (LogRecord.class) {
170                this.sequenceNumber = currentSequenceNumber++;
171                Integer id = currentThreadId.get();
172                if (null == id) {
173                    this.threadID = initThreadId;
174                    currentThreadId.set(Integer.valueOf(initThreadId++));
175                } else {
176                    this.threadID = id.intValue();
177                }
178            }
179    
180            this.sourceClassName = null;
181            this.sourceMethodName = null;
182            this.loggerName = null;
183            this.parameters = null;
184            this.resourceBundle = null;
185            this.resourceBundleName = null;
186            this.thrown = null;
187        }

383        /*
384         *  Init the sourceClass and sourceMethod fields.
385         */
386        private void initSource() {
387            if (!sourceInited) {
388                StackTraceElement[] elements = (new Throwable()).getStackTrace();
389                int i = 0;
390                String current = null;
391                FINDLOG: for (; i < elements.length; i++) {
392                    current = elements[i].getClassName();
393                    if (current.equals(Logger.class.getName())) {
394                        break FINDLOG;
395                    }
396                }
397                while(++i<elements.length && elements[i].getClassName().equals(current)) {
398                    // do nothing
399                }
400                if (i < elements.length) {
401                    this.sourceClassName = elements[i].getClassName();
402                    this.sourceMethodName = elements[i].getMethodName();
403                }
404                sourceInited = true;
405            }
406        }

492        /*
493         * Customized serialization.
494         */
495        private void writeObject(ObjectOutputStream out) throws IOException {
496            out.defaultWriteObject();
497            out.writeByte(MAJOR);
498            out.writeByte(MINOR);
499            if (null == parameters) {
500                out.writeInt(-1);
501            } else {
502                out.writeInt(parameters.length);
503                for (Object element : parameters) {
504                    out.writeObject(null == element ? null : element.toString());
505                }
506            }
507        }
508    
509        /*
510         * Customized deserialization.
511         */
512        private void readObject(ObjectInputStream in) throws IOException,
513                ClassNotFoundException {
514            in.defaultReadObject();
515            byte major = in.readByte();
516            byte minor = in.readByte();
517            //only check MAJOR version
518            if (major != MAJOR) {
519                // logging.5=Different version - {0}.{1}
520                throw new IOException(Messages.getString("logging.5", major, minor)); //$NON-NLS-1$ 
521            }
522            
523            int length = in.readInt();
524            if (length >= 0) {
525                parameters = new Object[length];
526                for (int i = 0; i < parameters.length; i++) {
527                    parameters[i] = in.readObject();
528                }
529            }
530            if (null != resourceBundleName) {
531                try {
532                    resourceBundle = Logger.loadResourceBundle(resourceBundleName);
533                } catch (MissingResourceException e) {
534                    // Cannot find the specified resource bundle
535                    resourceBundle = null;
536                }
537            }
538        }
