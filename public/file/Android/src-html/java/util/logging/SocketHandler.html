019    package java.util.logging;
020    
021    import java.net.Socket;
022    import java.io.BufferedOutputStream;
023    import java.io.IOException;
024    
025    import org.apache.harmony.logging.internal.nls.Messages;
026    
027    /**
028     * A handler that writes log messages to a socket connection.
029     * <p>
030     * This handler reads the following properties from the log manager to
031     * initialize itself:
032     * <ul>
033     * <li>java.util.logging.ConsoleHandler.level specifies the logging level,
034     * defaults to {@code Level.ALL} if this property is not found or has an invalid
035     * value.
036     * <li>java.util.logging.SocketHandler.filter specifies the name of the filter
037     * class to be associated with this handler, defaults to {@code null} if this
038     * property is not found or has an invalid value.
039     * <li>java.util.logging.SocketHandler.formatter specifies the name of the
040     * formatter class to be associated with this handler, defaults to
041     * {@code java.util.logging.XMLFormatter} if this property is not found or has
042     * an invalid value.
043     * <li>java.util.logging.SocketHandler.encoding specifies the encoding this
044     * handler will use to encode log messages, defaults to {@code null} if this
045     * property is not found or has an invalid value.
046     * <li>java.util.logging.SocketHandler.host specifies the name of the host that
047     * this handler should connect to. There's no default value for this property.
048     * <li>java.util.logging.SocketHandler.encoding specifies the port number that
049     * this handler should connect to. There's no default value for this property.
050     * </ul>
051     * </p>
052     * <p>
053     * This handler buffers the outgoing messages, but flushes each time a log
054     * record has been published.
055     * </p>
056     * <p>
057     * This class is not thread-safe.
058     * </p>
059     */
060    public class SocketHandler extends StreamHandler {
061        // default level
062        private static final String DEFAULT_LEVEL = "ALL"; //$NON-NLS-1$
063    
064        // default formatter
065        private static final String DEFAULT_FORMATTER = "java.util.logging.XMLFormatter"; //$NON-NLS-1$
066    
067        // the socket connection
068        private Socket socket;
069    
070        /**
071         * Constructs a {@code SocketHandler} object using the properties read by
072         * the log manager, including the host name and port number. Default
073         * formatting uses the XMLFormatter class and level is set to ALL.
074         * 
075         * @throws IOException
076         *             if failed to connect to the specified host and port.
077         * @throws IllegalArgumentException
078         *             if the host name or port number is illegal.
079         * @throws SecurityException
080         *             if a security manager determines that the caller does not
081         *             have the required permission to control this handler.
082         */
083        public SocketHandler() throws IOException {
084            super(DEFAULT_LEVEL, null, DEFAULT_FORMATTER, null);
085            initSocket(LogManager.getLogManager().getProperty(
086                    "java.util.logging.SocketHandler.host"), LogManager //$NON-NLS-1$
087                    .getLogManager().getProperty(
088                            "java.util.logging.SocketHandler.port")); //$NON-NLS-1$
089        }
090    
091        /**
092         * Constructs a {@code SocketHandler} object using the specified host name
093         * and port number together with other properties read by the log manager.
094         * Default formatting uses the XMLFormatter class and level is set to ALL.
095         * 
096         * @param host
097         *            the host name
098         * @param port
099         *            the port number
100         * @throws IOException
101         *             if failed to connect to the specified host and port.
102         * @throws IllegalArgumentException
103         *             if the host name or port number is illegal.
104         * @throws SecurityException
105         *             if a security manager determines that the caller does not
106         *             have the required permission to control this handler.
107         */
108        public SocketHandler(String host, int port) throws IOException {
109            super(DEFAULT_LEVEL, null, DEFAULT_FORMATTER, null);
110            initSocket(host, String.valueOf(port));
111        }
112    
113        // Initialize the socket connection and prepare the output stream
114        private void initSocket(String host, String port) throws IOException {
115            // check the validity of the host name
116            if (null == host || "".equals(host)) { //$NON-NLS-1$
117                // logging.C=Illegal host argument.
118                throw new IllegalArgumentException(Messages.getString("logging.C")); //$NON-NLS-1$
119            }
120            // check the validity of the port number
121            int p = 0;
122            try {
123                p = Integer.parseInt(port);
124            } catch (NumberFormatException e) {
125                // logging.D=Illegal port argument.
126                throw new IllegalArgumentException(Messages.getString("logging.D")); //$NON-NLS-1$
127            }
128            if (p <= 0) {
129                // logging.D=Illegal port argument.
130                throw new IllegalArgumentException(Messages.getString("logging.D")); //$NON-NLS-1$
131            }
132            // establish the network connection
133            try {
134                this.socket = new Socket(host, p);
135            } catch (IOException e) {
136                // logging.E=Failed to establish the network connection.
137                getErrorManager().error(Messages.getString("logging.E"), e, //$NON-NLS-1$
138                        ErrorManager.OPEN_FAILURE);
139                throw e;
140            }
141            // BEGIN android-modified
142            super.internalSetOutputStream(new BufferedOutputStream(this.socket
143                            .getOutputStream(), 8192));
144            // END android-modified
145        }
146    
147        /**
148         * Closes this handler. The network connection to the host is also closed.
149         * 
150         * @throws SecurityException
151         *             If a security manager determines that the caller does not
152         *             have the required permission to control this handler.
153         */
154        @Override
155        public void close() {
156            try {
157                super.close();
158                if (null != this.socket) {
159                    this.socket.close();
160                    this.socket = null;
161                }
162            } catch (Exception e) {
163                // logging.F=Exception occurred when closing the socket handler.
164                getErrorManager().error(Messages.getString("logging.F"), e, //$NON-NLS-1$
165                        ErrorManager.CLOSE_FAILURE);
166            }
167        }
168    
169        /**
170         * Logs a record if necessary. A flush operation will be done afterwards.
171         * 
172         * @param record
173         *            the log record to be logged.
174         */
175        @Override
176        public void publish(LogRecord record) {
177            super.publish(record);
178            super.flush();
179        }
180    
181    }
