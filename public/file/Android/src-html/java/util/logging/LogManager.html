047    /**
048     * {@code LogManager} is used to maintain configuration properties of the
049     * logging framework, and to manage a hierarchical namespace of all named
050     * {@code Logger} objects.
051     * <p>
052     *
053     * There is only one global {@code LogManager} instance in the
054     * application, which can be get by calling static method
055     * {@link #getLogManager()}. This instance is created and
056     * initialized during class initialization and cannot be changed.
057     * </p>
058     * <p>
059     * The {@code LogManager} class can be specified by
060     * java.util.logging.manager system property, if the property is unavailable or
061     * invalid, the default class {@link java.util.logging.LogManager} will
062     * be used.
063     * </p>
064     * <p>
065     * When initialization, {@code LogManager} read its configuration from a
066     * properties file, which by default is the "lib/logging.properties" in the JRE
067     * directory.
068     * </p>
069     * <p>
070     * However, two optional system properties can be used to customize the initial
071     * configuration process of {@code LogManager}.
072     * <ul>
073     * <li>"java.util.logging.config.class"</li>
074     * <li>"java.util.logging.config.file"</li>
075     * </ul>
076     * </p>
077     * <p>
078     * These two properties can be set in three ways, by the Preferences API, by the
079     * "java" command line property definitions, or by system property definitions
080     * passed to JNI_CreateJavaVM.
081     * </p>
082     * <p>
083     * The "java.util.logging.config.class" should specifies a class name. If it is
084     * set, this given class will be loaded and instantiated during
085     * {@code LogManager} initialization, so that this object's default
086     * constructor can read the initial configuration and define properties for
087     * {@code LogManager}.
088     * </p>
089     * <p>
090     * If "java.util.logging.config.class" property is not set, or it is invalid, or
091     * some exception is thrown during the instantiation, then the
092     * "java.util.logging.config.file" system property can be used to specify a
093     * properties file. The {@code LogManager} will read initial
094     * configuration from this file.
095     * </p>
096     * <p>
097     * If neither of these properties is defined, or some exception is thrown
098     * during these two properties using, the {@code LogManager} will read
099     * its initial configuration from default properties file, as described above.
100     * </p>
101     * <p>
102     * The global logging properties may include:
103     * <ul>
104     * <li>"handlers". This property's values should be a list of class names for
105     * handler classes separated by whitespace, these classes must be subclasses of
106     * {@code Handler} and each must have a default constructor, these
107     * classes will be loaded, instantiated and registered as handlers on the root
108     * {@code Logger} (the {@code Logger} named ""). These
109     * {@code Handler}s maybe initialized lazily.</li>
110     * <li>"config". The property defines a list of class names separated by
111     * whitespace. Each class must have a default constructor, in which it can
112     * update the logging configuration, such as levels, handlers, or filters for
113     * some logger, etc. These classes will be loaded and instantiated during
114     * {@code LogManager} configuration</li>
115     * </ul>
116     * </p>
117     * <p>
118     * This class, together with any handler and configuration classes associated
119     * with it, <b>must</b> be loaded from the system classpath when
120     * {@code LogManager} configuration occurs.
121     * </p>
122     * <p>
123     * Besides global properties, the properties for loggers and Handlers can be
124     * specified in the property files. The names of these properties will start
125     * with the complete dot separated names for the handlers or loggers.
126     * </p>
127     * <p>
128     * In the {@code LogManager}'s hierarchical namespace,
129     * {@code Loggers} are organized based on their dot separated names. For
130     * example, "x.y.z" is child of "x.y".
131     * </p>
132     * <p>
133     * Levels for {@code Loggers} can be defined by properties whose name end
134     * with ".level". Thus "alogger.level" defines a level for the logger named as
135     * "alogger" and for all its children in the naming hierarchy. Log levels
136     * properties are read and applied in the same order as they are specified in
137     * the property file. The root logger's level can be defined by the property
138     * named as ".level".
139     * </p>
140     * <p>
141     * All methods on this type can be taken as being thread safe.
142     * </p>
143     * 
144     */

234        static {
235            // init LogManager singleton instance
236            AccessController.doPrivileged(new PrivilegedAction<Object>() {
237                public Object run() {
238                    String className = System.getProperty(
239                            "java.util.logging.manager"); //$NON-NLS-1$
240                    
241                    if (null != className) {
242                        manager = (LogManager) getInstanceByClass(className);
243                    }
244                    if (null == manager) {
245                        manager = new LogManager();
246                    }
247    
248                    // read configuration
249                    try {
250                        manager.readConfiguration();
251                    } catch (Exception e) {
252                        e.printStackTrace();
253                    }
254    
255                    // if global logger has been initialized, set root as its parent
256                    Logger root = new Logger("", null); //$NON-NLS-1$
257                    root.setLevel(Level.INFO);
258                    Logger.global.setParent(root);
259                    
260                    manager.addLogger(root);
261                    manager.addLogger(Logger.global);
262                    return null;
263                }
264            });
265        }

267        /**
268         * Default constructor. This is not public because there should be only one
269         * {@code LogManager} instance, which can be get by
270         * {@code LogManager.getLogManager(}. This is protected so that
271         * application can subclass the object.
272         */
273        protected LogManager() {
274            loggers = new Hashtable<String, Logger>();
275            props = new Properties();
276            listeners = new PropertyChangeSupport(this);
277            // add shutdown hook to ensure that the associated resource will be
278            // freed when JVM exits
279            AccessController.doPrivileged(new PrivilegedAction<Void>() {
280                public Void run() {
281                    Runtime.getRuntime().addShutdownHook(new Thread() {
282                        @Override
283                        public void run() {
284                            reset();
285                        }
286                    });
287                    return null;
288                }
289            });
290        }

322        /**
323         * Add a given logger into the hierarchical namespace. The
324         * {@code Logger.addLogger()} factory methods call this method to add newly
325         * created Logger. This returns false if a logger with the given name has
326         * existed in the namespace
327         * <p>
328         * Note that the {@code LogManager} may only retain weak references to
329         * registered loggers. In order to prevent {@code Logger} objects from being
330         * unexpectedly garbage collected it is necessary for <i>applications</i>
331         * to maintain references to them.
332         * </p>
333         * 
334         * @param logger
335         *            the logger to be added.
336         * @return true if the given logger is added into the namespace
337         *         successfully, false if the given logger exists in the namespace.
338         */
339        public synchronized boolean addLogger(Logger logger) {
340            String name = logger.getName();
341            if (null != loggers.get(name)) {
342                return false;
343            }
344            addToFamilyTree(logger, name);
345            loggers.put(name, logger);
346            logger.setManager(this);
347            return true;
348        }
349    
350    
351        private void addToFamilyTree(Logger logger, String name) {
352            Logger parent = null;
353            // find parent
354            int lastSeparator;
355            String parentName = name;
356            while ((lastSeparator = parentName.lastIndexOf('.')) != -1) {
357                parentName = parentName.substring(0, lastSeparator);
358                parent = loggers.get(parentName);
359                if (parent != null) {
360                    logger.internalSetParent(parent);
361                    break;
362                } else if (getProperty(parentName+".level") != null || //$NON-NLS-1$
363                        getProperty(parentName+".handlers") != null) { //$NON-NLS-1$
364                    parent = Logger.getLogger(parentName);
365                    logger.internalSetParent(parent);
366                    break;
367                }
368            }
369            if (parent == null && null != (parent = loggers.get(""))) { //$NON-NLS-1$
370                logger.internalSetParent(parent);
371            }
372    
373            // find children
374            //TODO: performance can be improved here?
375            Collection<Logger> allLoggers = loggers.values();
376            for (Logger child : allLoggers) {
377                Logger oldParent = child.getParent();
378                if (parent == oldParent
379                        && (name.length() == 0 || child.getName().startsWith(
380                                name + '.'))) {
381                    child.setParent(logger);
382                    if (null != oldParent) {
383                        //-- remove from old parent as the parent has been changed
384                        oldParent.removeChild(child);
385                    }
386                }
387            }
388        }

430        /**
431         * Re-initialize the properties and configuration. The initialization
432         * process is same as the {@code LogManager} instantiation.
433         * <p>
434         * Notice : No {@code PropertyChangeEvent} are fired.
435         * </p>
436         * 
437         * @throws IOException
438         *             if any IO related problems happened.
439         * @throws SecurityException
440         *             if security manager exists and it determines that caller does
441         *             not have the required permissions to perform this action.
442         */
443        public void readConfiguration() throws IOException {
444            checkAccess();
445            // check config class
446            String configClassName = System.getProperty(
447                    "java.util.logging.config.class"); //$NON-NLS-1$
448            if (null == configClassName || null == getInstanceByClass(configClassName)) {
449                // if config class failed, check config file       
450                String configFile = System.getProperty(
451                        "java.util.logging.config.file"); //$NON-NLS-1$
452    
453                if (null == configFile) {
454                    // if cannot find configFile, use default logging.properties
455                    configFile = new StringBuilder().append(
456                            System.getProperty("java.home")).append(File.separator) //$NON-NLS-1$
457                            .append("lib").append(File.separator).append( //$NON-NLS-1$
458                            "logging.properties").toString(); //$NON-NLS-1$
459                }
460    
461                InputStream input = null;
462                try {
463                    // BEGIN android-removed
464                    // input = new BufferedInputStream(new FileInputStream(configFile));
465                    // END android-removed
466                    
467                    // BEGIN android-added
468                    try {
469                        input = new BufferedInputStream(
470                                new FileInputStream(configFile), 8192);
471                    } catch (Exception ex) {
472                        // consult fixed resource as a last resort
473                        input = new BufferedInputStream(
474                                getClass().getResourceAsStream(
475                                        "logging.properties"), 8192);
476                    }
477                    // END android-added
478                    readConfigurationImpl(input);
479                } finally {
480                    if (input != null) {
481                        try {
482                            input.close();
483                        } catch (Exception e) {// ignore
484                        }
485                    }
486                }
487            }
488        }

499        // use SystemClassLoader to load class from system classpath
500        static Object getInstanceByClass(final String className) {
501            try {
502                Class<?> clazz = ClassLoader.getSystemClassLoader().loadClass(
503                        className);
504                return clazz.newInstance();
505            } catch (Exception e) {
506                try {
507                    Class<?> clazz = Thread.currentThread()
508                            .getContextClassLoader().loadClass(className);
509                    return clazz.newInstance();
510                } catch (Exception innerE) {
511                    //logging.20=Loading class "{0}" failed
512                    System.err.println(Messages.getString(
513                            "logging.20", className)); //$NON-NLS-1$
514                    System.err.println(innerE);
515                    return null;
516                }
517            }
518    
519        }

521        // actual initialization process from a given input stream
522        private synchronized void readConfigurationImpl(InputStream ins)
523                throws IOException {
524            reset();
525            props.load(ins);
526            
527            // parse property "config" and apply setting
528            String configs = props.getProperty("config"); //$NON-NLS-1$
529            if (null != configs) {
530                StringTokenizer st = new StringTokenizer(configs, " "); //$NON-NLS-1$
531                while (st.hasMoreTokens()) {
532                    String configerName = st.nextToken();
533                    getInstanceByClass(configerName);
534                }
535            }
536            
537            // set levels for logger
538            Collection<Logger> allLoggers = loggers.values();
539            for(Logger logger : allLoggers){
540                String property = props.getProperty(
541                        logger.getName()+".level"); //$NON-NLS-1$
542                if(null != property){
543                    logger.setLevel(Level.parse(property));
544                }
545            }
546            listeners.firePropertyChange(null, null, null);
547        }

570        /**
571         * Reset configuration.
572         * <p>
573         * All handlers are closed and removed from any named loggers. All loggers'
574         * level is set to null, except the root logger's level is set to
575         * {@code Level.INFO}.
576         * </p>
577         * 
578         * @throws SecurityException
579         *             if security manager exists and it determines that caller does
580         *             not have the required permissions to perform this action.
581         */
582        public void reset() {
583            checkAccess();
584            props = new Properties();
585            Enumeration<String> names = getLoggerNames();
586            while(names.hasMoreElements()){
587                String name = names.nextElement();
588                Logger logger = getLogger(name);
589                if(logger != null){
590                    logger.reset();
591                }
592            }
593            Logger root = loggers.get(""); //$NON-NLS-1$
594            if (null != root) {
595                root.setLevel(Level.INFO);
596            }
597        }
