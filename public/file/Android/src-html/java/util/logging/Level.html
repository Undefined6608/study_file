034    /**
035     * {@code Level} objects are used to indicate the level of logging. There are a
036     * set of predefined logging levels, each associated with an integer value.
037     * Enabling a certain logging level also enables all logging levels with larger
038     * values.
039     * <p>
040     * The predefined levels in ascending order are FINEST, FINER, FINE, CONFIG,
041     * INFO, WARNING, SEVERE. There are two additional predefined levels, which are
042     * ALL and OFF. ALL indicates logging all messages, and OFF indicates logging no
043     * messages.
044     * </p>
045     * @since Android 1.0
046     */
047    public class Level implements Serializable {
048    
049        private static final long serialVersionUID = -8176160795706313070L;
050    
051        private static final List<Level> levels = new ArrayList<Level>(9);
052    
053        /**
054         * The OFF level provides no logging messages.
055         * 
056         * @since Android 1.0
057         */
058        public static final Level OFF = new Level("OFF", Integer.MAX_VALUE); //$NON-NLS-1$
059    
060        /**
061         * The SEVERE level provides severe failure messages.
062         * 
063         * @since Android 1.0
064         */
065        public static final Level SEVERE = new Level("SEVERE", 1000); //$NON-NLS-1$
066    
067        /**
068         * The WARNING level provides warnings.
069         * 
070         * @since Android 1.0
071         */
072        public static final Level WARNING = new Level("WARNING", 900); //$NON-NLS-1$
073    
074        /**
075         * The INFO level provides informative messages.
076         * 
077         * @since Android 1.0
078         */
079        public static final Level INFO = new Level("INFO", 800); //$NON-NLS-1$
080    
081        /**
082         * The CONFIG level provides static configuration messages.
083         * 
084         * @since Android 1.0
085         */
086        public static final Level CONFIG = new Level("CONFIG", 700); //$NON-NLS-1$
087    
088        /**
089         * The FINE level provides tracing messages.
090         * 
091         * @since Android 1.0
092         */
093        public static final Level FINE = new Level("FINE", 500); //$NON-NLS-1$
094    
095        /**
096         * The FINER level provides more detailed tracing messages.
097         * 
098         * @since Android 1.0
099         */
100        public static final Level FINER = new Level("FINER", 400); //$NON-NLS-1$
101    
102        /**
103         * The FINEST level provides highly detailed tracing messages.
104         * 
105         * @since Android 1.0
106         */
107        public static final Level FINEST = new Level("FINEST", 300); //$NON-NLS-1$
108    
109        /**
110         * The ALL level provides all logging messages.
111         * 
112         * @since Android 1.0
113         */
114        public static final Level ALL = new Level("ALL", Integer.MIN_VALUE); //$NON-NLS-1$

116        /**
117         * Parses a level name into a {@code Level} object.
118         * 
119         * @param name
120         *            the name of the desired {@code level}, which cannot be
121         *            {@code null}.
122         * @return the level with the specified name.
123         * @throws NullPointerException
124         *             if {@code name} is {@code null}.
125         * @throws IllegalArgumentException
126         *             if {@code name} is not valid.
127         * @since Android 1.0
128         */
129        public static Level parse(String name) throws IllegalArgumentException {
130            // BEGIN android-note
131            // final modifier removed and IAE added to get closer to the RI
132            // copied from newer version of harmony
133            // END android-note
134            if (name == null) {
135                // logging.1C=The 'name' parameter is null.
136                throw new NullPointerException(Messages.getString("logging.1C")); //$NON-NLS-1$
137            }
138    
139            boolean isNameAnInt;
140            int nameAsInt;
141            try {
142                nameAsInt = Integer.parseInt(name);
143                isNameAnInt = true;
144            } catch (NumberFormatException e) {
145                nameAsInt = 0;
146                isNameAnInt = false;
147            }
148    
149            synchronized (levels) {
150                for (Level level : levels) {
151                    if (name.equals(level.getName())) {
152                        return level;
153                    }
154                }
155    
156                if (isNameAnInt) {
157                    /*
158                     * Loop through levels a second time, so that the
159                     * returned instance will be passed on the order of construction.
160                     */
161                    for (Level level : levels) {
162                        if (nameAsInt == level.intValue()) {
163                            return level;
164                        }
165                    }
166                }
167            }
168    
169            if (!isNameAnInt) {
170                // logging.1D=Cannot parse this name: {0}
171                throw new IllegalArgumentException(Messages.getString("logging.1D", name)); //$NON-NLS-1$
172            }
173    
174            return new Level(name, nameAsInt);
175        }
176    
177        /**
178         * The name of this Level.
179         * 
180         * @serial
181         */
182        private final String name;
183    
184        /**
185         * The integer value indicating the level.
186         * 
187         * @serial
188         */
189        private final int value;
190    
191        /**
192         * The name of the resource bundle used to localize the level name.
193         * 
194         * @serial
195         */
196        private final String resourceBundleName;
197    
198        /**
199         * The resource bundle associated with this level, used to localize the
200         * level name.
201         */
202        private transient ResourceBundle rb;
203    
204        /**
205         * Constructs an instance of {@code Level} taking the supplied name and
206         * level value.
207         * 
208         * @param name
209         *            the name of the level.
210         * @param level
211         *            an integer value indicating the level.
212         * @throws NullPointerException
213         *             if {@code name} is {@code null}.
214         * @since Android 1.0             
215         */
216        protected Level(String name, int level) {
217            this(name, level, null);
218        }
219    
220        /**
221         * Constructs an instance of {@code Level} taking the supplied name, level
222         * value and resource bundle name.
223         * 
224         * @param name
225         *            the name of the level.
226         * @param level
227         *            an integer value indicating the level.
228         * @param resourceBundleName
229         *            the name of the resource bundle to use.
230         * @throws NullPointerException
231         *             if {@code name} is {@code null}.
232         * @since Android 1.0
233         */
234        protected Level(String name, int level, String resourceBundleName) {
235            if (name == null) {
236                // logging.1C=The 'name' parameter is null.
237                throw new NullPointerException(Messages.getString("logging.1C")); //$NON-NLS-1$
238            }
239            this.name = name;
240            this.value = level;
241            this.resourceBundleName = resourceBundleName;
242            if (resourceBundleName != null) {
243                try {
244                    rb = ResourceBundle.getBundle(resourceBundleName,
245                            // BEGIN android-changed
246                            Locale.getDefault(), VMStack.getCallingClassLoader());
247                            // BEGIN android-changed
248                } catch (MissingResourceException e) {
249                    rb = null;
250                }
251            }
252            synchronized (levels) {
253                levels.add(this);
254            }
255        }

285        }
286    
287        /**
288         * <p>
289         * Serialization helper method to maintain singletons and add any new
290         * levels.
291         * </p>
292         * 
293         * @return the resolved instance.
294         */
295        private Object readResolve() {
296            synchronized (levels) {
297                for (Level level : levels) {
298                    if (value != level.value) {
299                        continue;
300                    }
301                    if (!name.equals(name)) {
302                        continue;
303                    }
304                    if (resourceBundleName == level.resourceBundleName) {
305                        return level;
306                    } else if (resourceBundleName != null
307                            && resourceBundleName.equals(level.resourceBundleName)) {
308                        return level;
309                    }
310                }
311                // This is a new value, so add it.
312                levels.add(this);
313                return this;
314            }
315        }
316    
317        /**
318         * Serialization helper to setup transient resource bundle instance.
319         * 
320         * @param in
321         *            the input stream to read the instance data from.
322         * @throws IOException
323         *             if an IO error occurs.
324         * @throws ClassNotFoundException
325         *             if a class is not found.
326         */
327        private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
328            in.defaultReadObject();
329            if (resourceBundleName != null) {
330                try {
331                    rb = ResourceBundle.getBundle(resourceBundleName);
332                } catch (MissingResourceException e) {
333                    rb = null;
334                }
335            }
336        }
337    
338        /**
339         * Gets the localized name of this level. The default locale is used. If no
340         * resource bundle is associated with this level then the original level
341         * name is returned.
342         * 
343         * @return the localized name of this level.
344         * @since Android 1.0
345         */
346        public String getLocalizedName() {
347            if (rb == null) {
348                return name;
349            }
350    
351            try {
352                return rb.getString(name);
353            } catch (MissingResourceException e) {
354                return name;
355            }
356        }
357    
358        /**
359         * Compares two {@code Level} objects for equality. They are considered to
360         * be equal if they have the same level value.
361         * 
362         * @param o
363         *            the other object to compare this level to.
364         * @return {@code true} if this object equals to the supplied object,
365         *         {@code false} otherwise.
366         * @since Android 1.0
367         */
368        @Override
369        public boolean equals(Object o) {
370            if (this == o) {
371                return true;
372            }
373    
374            if (!(o instanceof Level)) {
375                return false;
376            }
377    
378            return ((Level) o).intValue() == this.value;
379        }

390        }
