018    package java.util.logging;
019    
020    import java.security.AccessController;
021    import java.security.PrivilegedExceptionAction;
022    
023    import org.apache.harmony.logging.internal.nls.Messages;
024    
025    
026    /**
027     * A {@code Handler} put the description of log events into a cycled memory
028     * buffer.
029     * <p>
030     * Mostly this {@code MemoryHandler} just puts the given {@code LogRecord} into
031     * the internal buffer and doesn't perform any formatting or any other process.
032     * When the buffer is full, the earliest buffered records will be discarded.
033     * </p>
034     * <p>
035     * Every {@code MemoryHandler} has a target handler, and push action can be
036     * triggered so that all buffered records will be output to the target handler
037     * and normally the latter will publish the records. After the push action, the
038     * buffer will be cleared.
039     * </p>
040     * <p>
041     * The push action can be triggered in three ways:
042     * <ul>
043     * <li>The push method is called explicitly</li>
044     * <li>When a new {@code LogRecord} is put into the internal buffer, and it has
045     * a level which is not less than the specified push level.</li>
046     * <li>A subclass extends this {@code MemoryHandler} and call push method
047     * implicitly according to some criteria.</li>
048     * </ul>
049     * </p>
050     * <p>
051     * {@code MemoryHandler} will read following {@code LogManager} properties for
052     * initialization, if given properties are not defined or has invalid values,
053     * default value will be used.
054     * <ul>
055     * <li>java.util.logging.MemoryHandler.level specifies the level for this
056     * {@code Handler}, defaults to {@code Level.ALL}.</li>
057     * <li>java.util.logging.MemoryHandler.filter specifies the {@code Filter}
058     * class name, defaults to no {@code Filter}.</li>
059     * <li>java.util.logging.MemoryHandler.size specifies the buffer size in number
060     * of {@code LogRecord}, defaults to 1000.</li>
061     * <li>java.util.logging.MemoryHandler.push specifies the push level, defaults
062     * to level.SEVERE.</li>
063     * <li>java.util.logging.MemoryHandler.target specifies the class of the target
064     * {@code Handler}, no default value, which means this property must be
065     * specified either by property setting or by constructor.</li>
066     * </ul>
067     * </p>
068     */
069    public class MemoryHandler extends Handler {
070    
071        //default maximum buffered number of LogRecord 
072        private static final int DEFAULT_SIZE = 1000;
073        //target handler
074        private Handler target;
075        
076        //buffer size
077        private int size = DEFAULT_SIZE;
078        
079        //push level
080        private Level push = Level.SEVERE;
081    
082        //LogManager instance for convenience
083        private final LogManager manager = LogManager.getLogManager();
084        
085        //buffer
086        private LogRecord[] buffer;
087        
088        //current position in buffer
089        private int cursor;
090        
091        /**
092         * Default constructor, construct and init a {@code MemoryHandler} using
093         * {@code LogManager} properties or default values.
094         * 
095         * @throws RuntimeException
096         *             if property value are invalid and no default value could be
097         *             used.
098         */
099        public MemoryHandler() {
100            super();
101            String className = this.getClass().getName();
102            //init target
103            final String targetName = manager.getProperty(className+".target"); //$NON-NLS-1$
104            try {
105                Class<?> targetClass = AccessController.doPrivileged(new PrivilegedExceptionAction<Class<?>>(){
106                    public Class<?> run() throws Exception{
107                        ClassLoader loader = Thread.currentThread().getContextClassLoader();
108                        if(loader == null){
109                            loader = ClassLoader.getSystemClassLoader();
110                        }
111                        return loader.loadClass(targetName);
112                    }
113                });
114                target = (Handler) targetClass.newInstance();
115            } catch (Exception e) {
116                // logging.10=Cannot load target handler:{0}
117                throw new RuntimeException(Messages.getString("logging.10", //$NON-NLS-1$
118                        targetName));
119            }
120            //init size
121            String sizeString = manager.getProperty(className+".size"); //$NON-NLS-1$
122            if (null != sizeString) {
123                try {
124                    size = Integer.parseInt(sizeString);
125                    if(size <= 0){
126                        size = DEFAULT_SIZE;
127                    }
128                } catch (Exception e) {
129                    printInvalidPropMessage(className+".size", sizeString, e); //$NON-NLS-1$
130                }
131            }
132            //init push level
133            String pushName = manager.getProperty(className+".push"); //$NON-NLS-1$
134            if (null != pushName) {
135                try {
136                    push = Level.parse(pushName);
137                } catch (Exception e) {
138                    printInvalidPropMessage(className+".push", pushName, e); //$NON-NLS-1$
139                }
140            }
141            //init other properties which are common for all Handler
142            initProperties("ALL", null, "java.util.logging.SimpleFormatter", null);  //$NON-NLS-1$//$NON-NLS-2$
143            buffer = new LogRecord[size];
144        }
145        
146        /**
147         * Construct and init a {@code MemoryHandler} using given target, size and
148         * push level, other properties using {@code LogManager} properties or
149         * default values.
150         * 
151         * @param target
152         *            the given {@code Handler} to output
153         * @param size
154         *            the maximum number of buffered {@code LogRecord}, greater than
155         *            zero
156         * @param pushLevel
157         *            the push level
158         * @throws IllegalArgumentException
159         *             if {@code size}<=0
160         * @throws RuntimeException
161         *             if property value are invalid and no default value could be
162         *             used.
163         */
164        public MemoryHandler(Handler target, int size, Level pushLevel) {
165            if (size <= 0) {
166                // logging.11=Size must be positive.
167                throw new IllegalArgumentException(Messages.getString("logging.11")); //$NON-NLS-1$
168            }
169            target.getLevel();
170            pushLevel.intValue();
171            this.target = target;
172            this.size = size;
173            this.push = pushLevel;
174            initProperties("ALL", null, "java.util.logging.SimpleFormatter", null);  //$NON-NLS-1$//$NON-NLS-2$
175            buffer = new LogRecord[size];
176        }
177        
178        /**
179         * Close this handler and target handler, free all associated resources.
180         * 
181         * @throws SecurityException
182         *             if security manager exists and it determines that caller does
183         *             not have the required permissions to control this handler.
184         */
185        @Override
186        public void close() {
187            manager.checkAccess();
188            target.close();
189            setLevel(Level.OFF);
190        }
191    
192        /**
193         * Call target handler to flush any buffered output. Note that this doesn't
194         * cause this {@code MemoryHandler} to push.
195         */
196        @Override
197        public void flush() {
198            target.flush();
199        }
200    
201        /**
202         * Put a given {@code LogRecord} into internal buffer. If given record is
203         * not loggable, just return. Otherwise it is stored in the buffer.
204         * Furthermore if the record's level is not less than the push level, the
205         * push action is triggered to output all the buffered records to the target
206         * handler, and the target handler will publish them.
207         * 
208         * @param record
209         *            the log record
210         */
211        @Override
212        public synchronized void publish(LogRecord record) {
213            if (!isLoggable(record)) {
214                return;
215            }
216            if (cursor >= size) {
217                cursor = 0;
218            }
219            buffer[cursor++] = record;
220            if (record.getLevel().intValue() >= push.intValue()) {
221                push();
222            }
223            return;
224        }
225    
226        /**
227         * Return the push level.
228         * 
229         * @return the push level
230         */
231        public Level getPushLevel() {
232            return push;
233        }
234    
235        /**
236         * <p>
237         * Check if given {@code LogRecord} would be put into this
238         * {@code MemoryHandler}'s internal buffer.
239         * </p>
240         * <p>
241         * The given {@code LogRecord} is loggable if and only if it has appropriate
242         * level and it pass any associated filter's check.
243         * </p>
244         * <p>
245         * Note that the push level is not used for this check.
246         * </p>
247         * 
248         * @param record
249         *            the given {@code LogRecord}
250         * @return the given {@code LogRecord} if it should be logged, {@code false}
251         *         if {@code LogRecord} is {@code null}.
252         */
253        @Override
254        public boolean isLoggable(LogRecord record) {
255            return super.isLoggable(record);
256        }
257    
258        /**
259         * Triggers a push action to output all buffered records to the target handler,
260         * and the target handler will publish them. Then the buffer is cleared.
261         */
262        public void push() {
263            for (int i = cursor; i < size; i++) {
264                if(null != buffer[i]) {
265                    target.publish(buffer[i]);
266                }
267                buffer[i] = null;
268            }
269            for (int i = 0; i < cursor; i++) {
270                if(null != buffer[i]) {
271                    target.publish(buffer[i]);
272                }
273                buffer[i] = null;
274            }
275            cursor = 0;
276        }
277    
278        /**
279         * Set the push level. The push level is used to check the push action 
280         * triggering. When a new {@code LogRecord} is put into the internal
281         * buffer and its level is not less than the push level, the push action 
282         * will be triggered. Note that set new push level won't trigger push action.
283         * 
284         * @param newLevel
285         *                 the new level to set.
286         * @throws SecurityException
287         *                 if security manager exists and it determines that caller 
288         *                 does not have the required permissions to control this handler.
289         */
290        public void setPushLevel(Level newLevel) {
291            manager.checkAccess();
292            newLevel.intValue();
293            this.push = newLevel;
294        }
295    }
