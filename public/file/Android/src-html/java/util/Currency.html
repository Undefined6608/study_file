018    package java.util;
019    
020    import java.io.Serializable;
021    
022    import org.apache.harmony.luni.util.Msg;
023    
024    /**
025     * This class represents a currency as identified in the ISO 4217 currency
026     * codes.
027     * 
028     * @since Android 1.0
029     */
030    public final class Currency implements Serializable {
031    
032        private static final long serialVersionUID = -158308464356906721L;
033    
034        private static Hashtable<String, Currency> codesToCurrencies = new Hashtable<String, Currency>();
035    
036        private String currencyCode;
037    
038        private static String currencyVars = "EURO, HK, PREEURO"; //$NON-NLS-1$
039    
040        private transient int defaultFractionDigits;
041    
042        /**
043         * @param currencyCode
044         */
045        private Currency(String currencyCode) {
046            this.currencyCode = currencyCode;
047        }
048    
049        /**
050         * Returns the {@code Currency} instance for this currency code.
051         * <p>
052         * 
053         * @param currencyCode
054         *            the currency code.
055         * @return the {@code Currency} instance for this currency code.
056         * 
057         * @throws IllegalArgumentException
058         *             if the currency code is not a supported ISO 4217 currency
059         *             code.
060         * @since Android 1.0
061         */
062        public static Currency getInstance(String currencyCode) {
063            Currency currency = codesToCurrencies.get(currencyCode);
064    
065            if (currency == null) {
066                ResourceBundle bundle = Locale.getBundle(
067                        "ISO4CurrenciesToDigits", Locale.getDefault()); //$NON-NLS-1$
068                currency = new Currency(currencyCode);
069    
070                String defaultFractionDigits = null;
071                try {
072                    defaultFractionDigits = bundle.getString(currencyCode);
073                } catch (MissingResourceException e) {
074                    throw new IllegalArgumentException(
075                            org.apache.harmony.luni.util.Msg.getString(
076                                    "K0322", currencyCode)); //$NON-NLS-1$
077                }
078                currency.defaultFractionDigits = Integer
079                        .parseInt(defaultFractionDigits);
080                codesToCurrencies.put(currencyCode, currency);
081            }
082    
083            return currency;
084        }
085    
086        /**
087         * Returns the {@code Currency} instance for this {@code Locale}'s country.
088         * 
089         * @param locale
090         *            the {@code Locale} of a country.
091         * @return the {@code Currency} used in the country defined by the locale parameter.
092         * 
093         * @throws IllegalArgumentException
094         *             if the locale's country is not a supported ISO 3166 Country.
095         * @since Android 1.0
096         */
097        public static Currency getInstance(Locale locale) {
098            String country = locale.getCountry();
099            String variant = locale.getVariant();
100            if (!variant.equals("") && currencyVars.indexOf(variant) > -1) { //$NON-NLS-1$
101                country = country + "_" + variant; //$NON-NLS-1$
102            }
103    
104            ResourceBundle bundle = Locale.getBundle(
105                    "ISO4Currencies", Locale.getDefault()); //$NON-NLS-1$
106            String currencyCode = null;
107            try {
108                currencyCode = bundle.getString(country);
109            } catch (MissingResourceException e) {
110                throw new IllegalArgumentException(Msg.getString(
111                        "K0323", locale.toString())); //$NON-NLS-1$
112            }
113    
114            if (currencyCode.equals("None")) { //$NON-NLS-1$
115                return null;
116            }
117    
118            return getInstance(currencyCode);
119        }
120    
121        /**
122         * Returns this {@code Currency}'s ISO 4217 currency code.
123         * 
124         * @return this {@code Currency}'s ISO 4217 currency code.
125         * @since Android 1.0
126         */
127        public String getCurrencyCode() {
128            return currencyCode;
129        }
130    
131        /**
132         * Returns the symbol for this currency in the default locale. For instance,
133         * if the default locale is the US, the symbol of the US dollar is "$". For
134         * other locales it may be "US$". If no symbol can be determined, the ISO
135         * 4217 currency code of the US dollar is returned.
136         * 
137         * @return the symbol for this {@code Currency} in the default {@code Locale}.
138         * @since Android 1.0
139         */
140        public String getSymbol() {
141            return getSymbol(Locale.getDefault());
142        }
143    
144        /**
145         * Returns the symbol for this currency in the given {@code Locale}.
146         * <p>
147         * If the locale doesn't have any countries (e.g.
148         * {@code Locale.JAPANESE, new Locale("en","")}), the the ISO
149         * 4217 currency code is returned.
150         * <p>
151         * First the locale's resource bundle is checked, if the locale has the same currency,
152         * the CurrencySymbol in this locale bundle is returned.
153         * <p>
154         * Then a currency bundle for this locale is searched.
155         * <p>
156         * If a currency bundle for this locale does not exist, or there is no
157         * symbol for this currency in this bundle, then the
158         * ISO 4217 currency code is returned.
159         * <p>
160         * 
161         * @param locale
162         *            the locale for which the currency symbol should be returned.
163         * @return the representation of this {@code Currency}'s symbol in the specified
164         *         locale.
165         * @since Android 1.0
166         */
167        public String getSymbol(Locale locale) {
168            if (locale.getCountry().equals("")) { //$NON-NLS-1$
169                return currencyCode;
170            }
171    
172            // check in the Locale bundle first, if the local has the same currency
173            ResourceBundle bundle = Locale.getBundle("Locale", locale); //$NON-NLS-1$
174            if (((String) bundle.getObject("IntCurrencySymbol")) //$NON-NLS-1$
175                    .equals(currencyCode)) {
176                return (String) bundle.getObject("CurrencySymbol"); //$NON-NLS-1$
177            }
178    
179            // search for a Currency bundle
180            bundle = null;
181            try {
182                bundle = Locale.getBundle("Currency", locale); //$NON-NLS-1$
183            } catch (MissingResourceException e) {
184                return currencyCode;
185            }
186    
187            // is the bundle found for a different country? (for instance the
188            // default locale's currency bundle)
189            if (!bundle.getLocale().getCountry().equals(locale.getCountry())) {
190                return currencyCode;
191            }
192    
193            // check if the currency bundle for this locale
194            // has an entry for this currency
195            String result = (String) bundle.handleGetObject(currencyCode);
196            if (result != null) {
197                return result;
198            }
199            return currencyCode;
200        }
201    
202        /**
203         * Returns the default number of fraction digits for this currency. For
204         * instance, the default number of fraction digits for the US dollar is 2.
205         * For the Japanese Yen the number is 0. In the case of pseudo-currencies,
206         * such as IMF Special Drawing Rights, -1 is returned.
207         * 
208         * @return the default number of fraction digits for this currency.
209         * @since Android 1.0
210         */
211        public int getDefaultFractionDigits() {
212            return defaultFractionDigits;
213        }
214    
215        /**
216         * Returns this currency's ISO 4217 currency code.
217         * 
218         * @return this currency's ISO 4217 currency code.
219         * @since Android 1.0
220         */
221        @Override
222        public String toString() {
223            return currencyCode;
224        }
225    
226        private Object readResolve() {
227            return getInstance(currencyCode);
228        }
229    }
