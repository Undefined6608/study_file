017    package java.util;
018    
019    /**
020     * Classes that handle custom formatting for the 's' specifier of {@code Formatter}
021     * should implement the {@code Formattable} interface. It gives basic control over
022     * formatting objects.
023     *  
024     * @see Formatter
025     * @since Android 1.0
026     */
027    
028    public interface Formattable {
029    
030        /**
031         * Formats the object using the specified {@code Formatter}.
032         * 
033         * @param formatter
034         *            the {@code Formatter} to use.
035         * @param flags
036         *            the flags applied to the output format, which is a bitmask
037         *            that is any combination of {@code FormattableFlags.LEFT_JUSTIFY},
038         *            {@code FormattableFlags.UPPERCASE}, and {@code FormattableFlags.ALTERNATE}. If
039         *            no such flag is set, the output is formatted by the default
040         *            formatting of the implementation.
041         * @param width
042         *            the minimum number of characters that should be written to the
043         *            output. If the length of the converted value is less than {@code width}
044         *            Additional space characters (' ') are added to the output if the
045         *            as needed to make up the difference. These spaces are added at the
046         *            beginning by default unless the flag
047         *            FormattableFlags.LEFT_JUSTIFY is set, which denotes that
048         *            padding should be added at the end. If width is -1, then
049         *            minimum length is not enforced.
050         * @param precision
051         *            the maximum number of characters that can be written to the
052         *            output. The length of the output is trimmed down to this size
053         *            before the width padding is applied. If the precision
054         *            is -1, then maximum length is not enforced.
055         * @throws IllegalFormatException
056         *             if any of the parameters is not supported.
057         * @since Android 1.0
058         */
059        void formatTo(Formatter formatter, int flags, int width, int precision)
060                throws IllegalFormatException;
061    }
