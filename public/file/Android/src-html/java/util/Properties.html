117        private void dumpString(StringBuilder buffer, String string, boolean key) {
118            int i = 0;
119            if (!key && i < string.length() && string.charAt(i) == ' ') {
120                buffer.append("\\ "); //$NON-NLS-1$
121                i++;
122            }
123    
124            for (; i < string.length(); i++) {
125                char ch = string.charAt(i);
126                switch (ch) {
127                case '\t':
128                    buffer.append("\\t"); //$NON-NLS-1$
129                    break;
130                case '\n':
131                    buffer.append("\\n"); //$NON-NLS-1$
132                    break;
133                case '\f':
134                    buffer.append("\\f"); //$NON-NLS-1$
135                    break;
136                case '\r':
137                    buffer.append("\\r"); //$NON-NLS-1$
138                    break;
139                default:
140                    if ("\\#!=:".indexOf(ch) >= 0 || (key && ch == ' ')) {
141                        buffer.append('\\');
142                    }
143                    if (ch >= ' ' && ch <= '~') {
144                        buffer.append(ch);
145                    } else {
146                        String hex = Integer.toHexString(ch);
147                        buffer.append("\\u"); //$NON-NLS-1$
148                        for (int j = 0; j < 4 - hex.length(); j++) {
149                            buffer.append("0"); //$NON-NLS-1$
150                        }
151                        buffer.append(hex);
152                    }
153                }
154            }
155        }

158         * Searches for the property with the specified name. If the property is not
159         * found, the default {@code Properties} are checked. If the property is not
160         * found in the default {@code Properties}, {@code null} is returned.
161         * 
162         * @param name
163         *            the name of the property to find.
164         * @return the named property value, or {@code null} if it can't be found.
165         * @since Android 1.0
166         */
167        public String getProperty(String name) {
168            Object result = super.get(name);
169            String property = result instanceof String ? (String) result : null;
170            if (property == null && defaults != null) {
171                property = defaults.getProperty(name);
172            }
173            return property;
174        }

177         * Searches for the property with the specified name. If the property is not
178         * found, it looks in the default {@code Properties}. If the property is not
179         * found in the default {@code Properties}, it returns the specified
180         * default.
181         * 
182         * @param name
183         *            the name of the property to find.
184         * @param defaultValue
185         *            the default value.
186         * @return the named property value.

191            String property = result instanceof String ? (String) result : null;

193                property = defaults.getProperty(name);
194            }
195            if (property == null) {
196                return defaultValue;
197            }
198            return property;

201        /**
202         * Lists the mappings in this {@code Properties} to the specified
203         * {@code PrintStream} in a
204         * human readable form.
205         * 
206         * @param out
207         *            the {@code PrintStream} to write the content to in human readable
208         *            form.
209         * @since Android 1.0
210         */
211        public void list(PrintStream out) {
212            if (out == null) {
213                throw new NullPointerException();
214            }
215            StringBuffer buffer = new StringBuffer(80);
216            Enumeration<?> keys = propertyNames();
217            while (keys.hasMoreElements()) {
218                String key = (String) keys.nextElement();
219                buffer.append(key);
220                buffer.append('=');
221                String property = (String) super.get(key);
222                Properties def = defaults;
223                while (property == null) {
224                    property = (String) def.get(key);
225                    def = def.defaults;
226                }
227                if (property.length() > 40) {
228                    buffer.append(property.substring(0, 37));
229                    buffer.append("..."); //$NON-NLS-1$
230                } else {
231                    buffer.append(property);
232                }
233                out.println(buffer.toString());
234                buffer.setLength(0);
235            }
236        }

275        /**
276         * Loads properties from the specified {@code InputStream}. The encoding is
277         * ISO8859-1. The {@code Properties} file is interpreted according to the
278         * following rules:
279         * <ul>
280         * <li>Empty lines are ignored.</li>
281         * <li>Lines starting with either a "#" or a "!" are comment lines and are
282         * ignored.</li>
283         * <li>A backslash at the end of the line escapes the following newline
284         * character ("\r", "\n", "\r\n"). If there's a whitespace after the
285         * backslash it will just escape that whitespace instead of concatenating
286         * the lines. This does not apply to comment lines.</li>
287         * <li>A property line consists of the key, the space between the key and
288         * the value, and the value. The key goes up to the first whitespace, "=" or
289         * ":" that is not escaped. The space between the key and the value contains
290         * either one whitespace, one "=" or one ":" and any number of additional
291         * whitespaces before and after that character. The value starts with the
292         * first character after the space between the key and the value.</li>
293         * <li>Following escape sequences are recognized: "\ ", "\\", "\r", "\n",
294         * "\!", "\#", "\t", "\b", "\f", and "&#92;uXXXX" (unicode character).</li>
295         * </ul>
296         * 
297         * @param in
298         *            the {@code InputStream}.
299         * @throws IOException
300         *             if error occurs during reading from the {@code InputStream}.
301         * @since Android 1.0
302         */
303        public synchronized void load(InputStream in) throws IOException {
304            int mode = NONE, unicode = 0, count = 0;
305            char nextChar, buf[] = new char[40];
306            int offset = 0, keyLength = -1;
307            boolean firstChar = true;
308            byte[] inbuf = new byte[256];
309            int inbufCount = 0, inbufPos = 0;
310    
311            while (true) {
312                if (inbufPos == inbufCount) {
313                    if ((inbufCount = in.read(inbuf)) == -1) {
314                        break;
315                    }
316                    inbufPos = 0;
317                }
318                nextChar = (char) (inbuf[inbufPos++] & 0xff);
319    
320                if (offset == buf.length) {
321                    char[] newBuf = new char[buf.length * 2];
322                    System.arraycopy(buf, 0, newBuf, 0, offset);
323                    buf = newBuf;
324                }
325                if (mode == UNICODE) {
326                    int digit = Character.digit(nextChar, 16);
327                    // BEGIN android-changed
328                    // copied from newer version of Harmony
329                    if (digit >= 0) {
330                        unicode = (unicode << 4) + digit;
331                        if (++count < 4) {
332                            continue;
333                        }
334                    } else if (count <= 4) {
335                        // luni.09=Invalid Unicode sequence: illegal character
336                        throw new IllegalArgumentException(Messages.getString("luni.09"));
337                    }
338                    // END android-changed
339                    mode = NONE;
340                    buf[offset++] = (char) unicode;
341                    if (nextChar != '\n') {
342                        continue;
343                    }
344                }
345                if (mode == SLASH) {
346                    mode = NONE;
347                    switch (nextChar) {
348                    case '\r':
349                        mode = CONTINUE; // Look for a following \n
350                        continue;
351                    case '\n':
352                        mode = IGNORE; // Ignore whitespace on the next line
353                        continue;
354                    case 'b':
355                        nextChar = '\b';
356                        break;
357                    case 'f':
358                        nextChar = '\f';
359                        break;
360                    case 'n':
361                        nextChar = '\n';
362                        break;
363                    case 'r':
364                        nextChar = '\r';
365                        break;
366                    case 't':
367                        nextChar = '\t';
368                        break;
369                    case 'u':
370                        mode = UNICODE;
371                        unicode = count = 0;
372                        continue;
373                    }
374                } else {
375                    switch (nextChar) {
376                    case '#':
377                    case '!':
378                        if (firstChar) {
379                            while (true) {
380                                if (inbufPos == inbufCount) {
381                                    if ((inbufCount = in.read(inbuf)) == -1) {
382                                        inbufPos = -1;
383                                        break;
384                                    }
385                                    inbufPos = 0;
386                                }
387                                nextChar = (char) inbuf[inbufPos++]; // & 0xff
388                                                                        // not
389                                                                        // required
390                                if (nextChar == '\r' || nextChar == '\n') {
391                                    break;
392                                }
393                            }
394                            continue;
395                        }
396                        break;
397                    case '\n':
398                        if (mode == CONTINUE) { // Part of a \r\n sequence
399                            mode = IGNORE; // Ignore whitespace on the next line
400                            continue;
401                        }
402                    // fall into the next case
403                    case '\r':
404                        mode = NONE;
405                        firstChar = true;
406                        if (offset > 0) {
407                            if (keyLength == -1) {
408                                keyLength = offset;
409                            }
410                            String temp = new String(buf, 0, offset);
411                            put(temp.substring(0, keyLength), temp
412                                    .substring(keyLength));
413                        }
414                        keyLength = -1;
415                        offset = 0;
416                        continue;
417                    case '\\':
418                        if (mode == KEY_DONE) {
419                            keyLength = offset;
420                        }
421                        mode = SLASH;
422                        continue;
423                    case ':':
424                    case '=':
425                        if (keyLength == -1) { // if parsing the key
426                            mode = NONE;
427                            keyLength = offset;
428                            continue;
429                        }
430                        break;
431                    }
432                    if (Character.isWhitespace(nextChar)) {
433                        if (mode == CONTINUE) {
434                            mode = IGNORE;
435                        }
436                        // if key length == 0 or value length == 0
437                        if (offset == 0 || offset == keyLength || mode == IGNORE) {
438                            continue;
439                        }
440                        if (keyLength == -1) { // if parsing the key
441                            mode = KEY_DONE;
442                            continue;
443                        }
444                    }
445                    if (mode == IGNORE || mode == CONTINUE) {
446                        mode = NONE;
447                    }
448                }
449                firstChar = false;
450                if (mode == KEY_DONE) {
451                    keyLength = offset;
452                    mode = NONE;
453                }
454                buf[offset++] = nextChar;
455            }
456            // BEGIN android-added
457            // copied from a newer version of Harmony
458            if (mode == UNICODE && count <= 4) {
459                // luni.08=Invalid Unicode sequence: expected format \\uxxxx
460                throw new IllegalArgumentException(Messages.getString("luni.08"));
461            }
462            // END android-added
463            if (keyLength >= 0) {
464                String temp = new String(buf, 0, offset);
465                put(temp.substring(0, keyLength), temp.substring(keyLength));
466            }
467        }

469        /**
470         * Returns all of the property names that this {@code Properties} object
471         * contains.
472         * 
473         * @return an {@code Enumeration} containing the names of all properties
474         *         that this {@code Properties} object contains.
475         * @since Android 1.0
476         */
477        public Enumeration<?> propertyNames() {
478            if (defaults == null) {
479                return keys();
480            }
481    
482            Hashtable<Object, Object> set = new Hashtable<Object, Object>(defaults.size() + size());
483            Enumeration<?> keys = defaults.propertyNames();
484            while (keys.hasMoreElements()) {
485                set.put(keys.nextElement(), set);
486            }
487            keys = keys();
488            while (keys.hasMoreElements()) {
489                set.put(keys.nextElement(), set);
490            }
491            return set.keys();
492        }

534        /**
535         * Stores the mappings in this {@code Properties} to the specified {@code
536         * OutputStream}, putting the specified comment at the beginning. The output
537         * from this method is suitable for being read by the
538         * {@link #load(InputStream)} method.
539         * 
540         * @param out the {@code OutputStream} to write to.
541         * @param comment the comment to put at the beginning.
542         * @throws IOException if an error occurs during the write to the {@code
543         *             OutputStream}.
544         * @exception ClassCastException when the key or value of a mapping is not a
545         *                {@code String}.
546         * @since Android 1.0
547         */
548        public synchronized void store(OutputStream out, String comment)
549                throws IOException {
550            if (lineSeparator == null) {
551                lineSeparator = AccessController
552                        .doPrivileged(new PriviAction<String>("line.separator")); //$NON-NLS-1$
553            }
554    
555            StringBuilder buffer = new StringBuilder(200);
556            OutputStreamWriter writer = new OutputStreamWriter(out, "ISO8859_1"); //$NON-NLS-1$
557            if (comment != null) {
558                writer.write("#"); //$NON-NLS-1$
559                writer.write(comment);
560                writer.write(lineSeparator); 
561            }
562            writer.write("#"); //$NON-NLS-1$
563            writer.write(new Date().toString());
564            writer.write(lineSeparator); 
565    
566            for (Map.Entry<Object, Object> entry : entrySet()) {
567                String key = (String) entry.getKey();
568                dumpString(buffer, key, true);
569                buffer.append('=');
570                dumpString(buffer, (String) entry.getValue(), false);
571                buffer.append(lineSeparator);
572                writer.write(buffer.toString());
573                buffer.setLength(0);
574            }
575            writer.flush();
576        }

578        /**
579         * Loads the properties from an {@code InputStream} containing the
580         * properties in XML form. The XML document must begin with (and conform to)
581         * following DOCTYPE:
582         * 
583         * <pre>
584         * &lt;!DOCTYPE properties SYSTEM &quot;http://java.sun.com/dtd/properties.dtd&quot;&gt;
585         * </pre>
586         * 
587         * Also the content of the XML data must satisfy the DTD but the xml is not
588         * validated against it. The DTD is not loaded from the SYSTEM ID. After
589         * this method returns the InputStream is not closed.
590         * 
591         * @param in the InputStream containing the XML document.
592         * @throws IOException in case an error occurs during a read operation.
593         * @throws InvalidPropertiesFormatException if the XML data is not a valid
594         *             properties file.
595         * @since Android 1.0
596         */
597        public synchronized void loadFromXML(InputStream in) 
598                throws IOException, InvalidPropertiesFormatException {
599            if (in == null) {
600                throw new NullPointerException();
601            }
602            
603            if (builder == null) {
604                DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
605                // BEGIN android-removed
606                // factory.setValidating(true);
607                // END android-removed
608                
609                try {
610                    builder = factory.newDocumentBuilder();
611                } catch (ParserConfigurationException e) {
612                    throw new Error(e);
613                }
614                
615                builder.setErrorHandler(new ErrorHandler() {
616                    public void warning(SAXParseException e) throws SAXException {
617                        throw e;
618                    }
619    
620                    public void error(SAXParseException e) throws SAXException {
621                        throw e;
622                    }
623    
624                    public void fatalError(SAXParseException e) throws SAXException {
625                        throw e;
626                    }
627                });
628                
629                builder.setEntityResolver(new EntityResolver() {
630                    public InputSource resolveEntity(String publicId, String systemId)
631                            throws SAXException, IOException {
632                        if (systemId.equals(PROP_DTD_NAME)) {
633                            InputSource result = new InputSource(new StringReader(
634                                    PROP_DTD));
635                            result.setSystemId(PROP_DTD_NAME);
636                            return result;
637                        }
638                        throw new SAXException(
639                                "Invalid DOCTYPE declaration: " + systemId);
640                    }
641                });
642            }
643            
644            try {
645                Document doc = builder.parse(in);
646                NodeList entries = doc.getElementsByTagName("entry"); 
647                if (entries == null) {
648                    return;
649                }
650                int entriesListLength = entries.getLength();
651                
652                for (int i = 0; i < entriesListLength; i++) {
653                    Element entry = (Element) entries.item(i);
654                    String key = entry.getAttribute("key");
655                    // BEGIN android-removed
656                    // String value = entry.getTextContent();
657                    // END android-removed
658                    // BEGIN android-added
659                    String value = getTextContent(entry);
660                    // END android-added
661                    
662                    /*
663                     * key != null & value != null
664                     * but key or(and) value can be empty String
665                     */
666                    put(key, value);
667                }
668            } catch (IOException e) {
669                throw e;
670            } catch (SAXException e) {
671                throw new InvalidPropertiesFormatException(e);
672            }
673        }

698        /**
699         * Writes all properties stored in this instance into the {@code
700         * OutputStream} in XML representation. The DOCTYPE is
701         * 
702         * <pre>
703         * &lt;!DOCTYPE properties SYSTEM &quot;http://java.sun.com/dtd/properties.dtd&quot;&gt;
704         * </pre>
705         * 
706         * If the comment is null, no comment is added to the output. The parameter
707         * {@code encoding} defines which encoding should be used. The {@code
708         * OutputStream} is not closed at the end.
709         * 
710         * @param os the {@code OutputStream} to write to.
711         * @param comment the comment to add. If null, no comment is added.
712         * @param encoding the code identifying the encoding that should be used to
713         *            write into the {@code OutputStream}.
714         * @throws IOException if an error occurs during writing to the output.
715         * @since Android 1.0
716         */
717        public synchronized void storeToXML(OutputStream os, String comment,
718                String encoding) throws IOException {
719    
720            if (os == null || encoding == null) {
721                throw new NullPointerException();
722            }
723            
724            /*
725             * We can write to XML file using encoding parameter but note that some
726             * aliases for encodings are not supported by the XML parser. Thus we
727             * have to know canonical name for encoding used to store data in XML
728             * since the XML parser must recognize encoding name used to store data.
729             */
730            
731            String encodingCanonicalName;
732            try {
733                encodingCanonicalName = Charset.forName(encoding).name();
734            } catch (IllegalCharsetNameException e) {
735                System.out.println("Warning: encoding name " + encoding
736                        + " is illegal, using UTF-8 as default encoding");
737                encodingCanonicalName = "UTF-8";
738            } catch (UnsupportedCharsetException e) {
739                System.out.println("Warning: encoding " + encoding
740                        + " is not supported, using UTF-8 as default encoding");
741                encodingCanonicalName = "UTF-8";
742            }
743    
744            PrintStream printStream = new PrintStream(os, false, encodingCanonicalName);
745            
746            printStream.print("<?xml version=\"1.0\" encoding=\"");
747            printStream.print(encodingCanonicalName);
748            printStream.println("\"?>");
749            
750            printStream.print("<!DOCTYPE properties SYSTEM \"");
751            printStream.print(PROP_DTD_NAME);
752            printStream.println("\">");
753            
754            printStream.println("<properties>");
755            
756            if (comment != null) {
757                printStream.print("<comment>");
758                printStream.print(substitutePredefinedEntries(comment));
759                printStream.println("</comment>");
760            }
761    
762            for (Map.Entry<Object, Object> entry : entrySet()) {
763                String keyValue = (String) entry.getKey();
764                String entryValue = (String) entry.getValue();
765                printStream.print("<entry key=\"");
766                printStream.print(substitutePredefinedEntries(keyValue));
767                printStream.print("\">");
768                printStream.print(substitutePredefinedEntries(entryValue));
769                printStream.println("</entry>");
770            }
771            printStream.println("</properties>");
772            printStream.flush();
773        }
774        
775        private String substitutePredefinedEntries(String s) {
776            
777            /*
778             * substitution for predefined character entities
779             * to use them safely in XML
780             */
781            return s.replaceAll("&", "&amp;")
782                .replaceAll("<", "&lt;")
783                .replaceAll(">", "&gt;")
784                .replaceAll("\u0027", "&apos;")
785                .replaceAll("\"", "&quot;");
786        }

788        // BEGIN android-added
789        private String getTextContent(Node node) {
790            String result = (node instanceof Text ? ((Text) node).getData() : "");
791    
792            Node child = node.getFirstChild();
793            while (child != null) {
794                result = result + getTextContent(child);
795                child = child.getNextSibling();
796            }
797    
798            return result;
799        }
