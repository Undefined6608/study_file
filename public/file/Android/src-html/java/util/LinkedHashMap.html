021    /**
022     * LinkedHashMap is a variant of HashMap. Its entries are kept in a
023     * doubly-linked list. The iteration order is, by default, the order in which
024     * keys were inserted. Reinserting an already existing key doesn't change the
025     * order. A key is existing if a call to {@code containsKey} would return true.
026     * <p>
027     * If the three argument constructor is used, and {@code order} is specified as
028     * {@code true}, the iteration will be in the order that entries were accessed.
029     * The access order gets affected by put(), get(), putAll() operations, but not
030     * by operations on the collection views.
031     * <p>
032     * Null elements are allowed, and all the optional map operations are supported.
033     * <p>
034     * <b>Note:</b> The implementation of {@code LinkedHashMap} is not synchronized.
035     * If one thread of several threads accessing an instance modifies the map
036     * structurally, access to the map needs to be synchronized. For
037     * insertion-ordered instances a structural modification is an operation that
038     * removes or adds an entry. Access-ordered instances also are structurally
039     * modified by put(), get() and putAll() since these methods change the order of
040     * the entries. Changes in the value of an entry are not structural changes.
041     * <p>
042     * The Iterator that can be created by calling the {@code iterator} method
043     * throws a {@code ConcurrentModificationException} if the map is structurally
044     * changed while an iterator is used to iterate over the elements. Only the
045     * {@code remove} method that is provided by the iterator allows for removal of
046     * elements during iteration. It is not possible to guarantee that this
047     * mechanism works in all cases of unsynchronized concurrent modification. It
048     * should only be used for debugging purposes.
049     * 
050     * @since Android 1.0
051     */
052    public class LinkedHashMap<K, V> extends HashMap<K, V> {

147        static final class LinkedHashIterator<E, KT, VT> extends HashMapIterator<E, KT, VT> {
148            LinkedHashIterator(MapEntry.Type<E, KT, VT> value, LinkedHashMap<KT, VT> hm) {
149                super(value, hm);
150                entry = hm.head;
151            }
152    
153            @Override
154            public boolean hasNext() {
155                return (entry != null);
156            }
157    
158            @Override
159            public E next() {
160                checkConcurrentMod();
161                if (!hasNext()) {
162                    throw new NoSuchElementException();
163                }
164                E result = type.get(entry);
165                lastEntry = entry;
166                entry = ((LinkedHashMapEntry<KT, VT>)entry).chainForward;
167                canRemove = true;
168                return result;
169            }
170    
171            @Override
172            public void remove() {
173                checkConcurrentMod();
174                if (!canRemove) {
175                    throw new IllegalStateException();
176                }
177    
178                canRemove = false;
179                associatedMap.modCount++;
180    
181                int index = (lastEntry.key == null)? 0 : (lastEntry.key.hashCode() & 0x7FFFFFFF) % associatedMap.elementData.length;
182                LinkedHashMapEntry<KT, VT> m = (LinkedHashMapEntry<KT, VT>) associatedMap.elementData[index];
183                if (m == lastEntry) {
184                    associatedMap.elementData[index] = lastEntry.next;
185                } else {
186                    while (m.next != null) {
187                        if (m.next == lastEntry) {
188                            break;
189                        }
190                        m = (LinkedHashMapEntry<KT, VT>) m.next;
191                    }
192                    // assert m.next == entry
193                    m.next = lastEntry.next;
194                }
195                LinkedHashMapEntry<KT, VT> lhme = (LinkedHashMapEntry<KT, VT>) lastEntry;
196                LinkedHashMapEntry<KT, VT> p = lhme.chainBackward;
197                LinkedHashMapEntry<KT, VT> n = lhme.chainForward;
198                LinkedHashMap<KT, VT> lhm = (LinkedHashMap<KT, VT>) associatedMap;
199                if (p != null) {
200                    p.chainForward = n;
201                    if (n != null) {
202                        n.chainBackward = p;
203                    } else {
204                        lhm.tail = p;
205                    }
206                } else {
207                    lhm.head = n;
208                    if (n != null) {
209                        n.chainBackward = null;
210                    } else {
211                        lhm.tail = null;
212                    }
213                }
214                associatedMap.elementCount--;
215                expectedModCount++;
216            }
217        }

234        static final class LinkedHashMapEntry<K, V> extends Entry<K, V> {
235            LinkedHashMapEntry<K, V> chainForward, chainBackward;
236    
237            LinkedHashMapEntry(K theKey, V theValue) {
238                super(theKey, theValue);
239                chainForward = null;
240                chainBackward = null;
241            }
242    
243            LinkedHashMapEntry(K theKey, int hash) {
244                super(theKey, hash);
245                chainForward = null;
246                chainBackward = null;
247            }
248    
249    
250            @Override
251            @SuppressWarnings("unchecked")
252            public Object clone() {
253                LinkedHashMapEntry<K, V> entry = (LinkedHashMapEntry<K, V>) super.clone();
254                entry.chainBackward = chainBackward;
255                entry.chainForward = chainForward;
256                LinkedHashMapEntry<K, V> lnext = (LinkedHashMapEntry<K, V>) entry.next;
257                if (lnext != null) {
258                    entry.next = (LinkedHashMapEntry<K, V>) lnext.clone();
259                }
260                return entry;
261            }
262        }

270        /**
271         * Returns the value of the mapping with the specified key.
272         * 
273         * @param key
274         *            the key.
275         * @return the value of the mapping with the specified key, or {@code null}
276         *         if no mapping for the specified key is found.
277         * @since Android 1.0
278         */
279        @Override
280        public V get(Object key) {
281            LinkedHashMapEntry<K, V> m;
282            if (key == null) {
283                m = (LinkedHashMapEntry<K, V>)findNullKeyEntry();
284            } else {
285                int hash = key.hashCode();
286                int index = (hash & 0x7FFFFFFF) % elementData.length;
287                m = (LinkedHashMapEntry<K, V>)findNonNullKeyEntry(key, index, hash);
288            }
289            if (m == null) {
290                return null;
291            }
292            if (accessOrder && tail != m) {
293                // BEGIN android-added
294                modCount++;
295                // END android-added
296                LinkedHashMapEntry<K, V> p = m.chainBackward;
297                LinkedHashMapEntry<K, V> n = m.chainForward;
298                n.chainBackward = p;
299                if (p != null) {
300                    p.chainForward = n;
301                } else {
302                    head = n;
303                }
304                m.chainForward = null;
305                m.chainBackward = tail;
306                tail.chainForward = m;
307                tail = m;
308            }
309            return m.value;
310        }

317            LinkedHashMapEntry<K, V> m = new LinkedHashMapEntry<K, V>(key, value);
318            m.next = elementData[index];
319            elementData[index] = m;
320            linkEntry(m);
321            return m;

356        V putImpl(K key, V value){
357            LinkedHashMapEntry<K, V> m;
358            if (elementCount == 0){
359                head = tail = null;
360            }
361            if (key == null) {
362                m = (LinkedHashMapEntry<K, V>)findNullKeyEntry();
363                if (m == null) {
364                    modCount++;
365                    // Check if we need to remove the oldest entry
366                    // The check includes accessOrder since an accessOrder LinkedHashMap
367                    // does not record
368                    // the oldest member in 'head'.
369                    if (++elementCount > threshold) {
370                        rehash();
371                    }
372                        m = (LinkedHashMapEntry<K, V>) createHashedEntry(null, 0, 0);
373                } else {
374                    linkEntry(m);
375                }
376            } else {
377                int hash = key.hashCode();
378                int index = (hash & 0x7FFFFFFF) % elementData.length;
379                m = (LinkedHashMapEntry<K, V>)findNonNullKeyEntry(key, index, hash);
380                if (m == null) {
381                    modCount++;
382                    if (++elementCount > threshold) {
383                        rehash();
384                        index = (hash & 0x7FFFFFFF) % elementData.length;
385                    }
386                    m = (LinkedHashMapEntry<K, V>) createHashedEntry(key, index, hash);
387                } else {
388                    linkEntry(m);
389                }
390            }
391    
392            V result = m.value;
393            m.value = value;
394            return result;
395        }

398        /*
399         * @param m
400         */
401        void linkEntry(LinkedHashMapEntry<K, V> m) {
402            if (tail == m) {
403                return;
404            }
405    
406            if (head == null) {
407                // Check if the map is empty
408                head = tail = m;
409                return;
410            }
411    
412            // we need to link the new entry into either the head or tail
413            // of the chain depending on if the LinkedHashMap is accessOrder or not
414            LinkedHashMapEntry<K, V> p = m.chainBackward;
415            LinkedHashMapEntry<K, V> n = m.chainForward;
416            if (p == null) {
417                if (n != null) {
418                    // The entry must be the head but not the tail
419                    if (accessOrder) {
420                        head = n;
421                        n.chainBackward = null;
422                        m.chainBackward = tail;
423                        m.chainForward = null;
424                        tail.chainForward = m;
425                        tail = m;
426                    }
427                } else {
428                    // This is a new entry
429                    m.chainBackward = tail;
430                    m.chainForward = null;
431                    tail.chainForward = m;
432                    tail = m;
433                }
434                return;
435            }
436    
437            if (n == null) {
438                // The entry must be the tail so we can't get here
439                return;
440            }
441    
442            // The entry is neither the head nor tail
443            if (accessOrder) {
444                p.chainForward = n;
445                n.chainBackward = p;
446                m.chainForward = null;
447                m.chainBackward = tail;
448                tail.chainForward = m;
449                tail = m;
450            }
451    
452        }

517         * Returns a collection of the values contained in this map. The collection
518         * is backed by this map so changes to one are reflected by the other. The

521         * <p>
522         * This method returns a collection which is the subclass of
523         * AbstractCollection. The iterator method of this subclass returns a
524         * "wrapper object" over the iterator of map's entrySet(). The size method

527         * </p>
528         * <p>
529         * The collection is created when this method is called for the first time
530         * and returned in response to all subsequent calls. This method may return
531         * different collections when multiple concurrent calls occur, since no
532         * synchronization is performed.
533         * </p>
534         *

536         * @since Android 1.0

550                    }

570        /**
571         * Removes the mapping with the specified key from this map.
572         * 
573         * @param key
574         *            the key of the mapping to remove.
575         * @return the value of the removed mapping or {@code null} if no mapping
576         *         for the specified key was found.
577         * @since Android 1.0
578         */
579        @Override
580        public V remove(Object key) {
581            LinkedHashMapEntry<K, V> m = (LinkedHashMapEntry<K, V>) removeEntry(key);
582            if (m == null) {
583                return null;
584            }
585            LinkedHashMapEntry<K, V> p = m.chainBackward;
586            LinkedHashMapEntry<K, V> n = m.chainForward;
587            if (p != null) {
588                p.chainForward = n;
589            } else {
590                head = n;
591            }
592            if (n != null) {
593                n.chainBackward = p;
594            } else {
595                tail = p;
596            }
597            return m.value;
598        }
