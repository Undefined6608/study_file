018    package java.util.zip;
019    
020    
021    import java.io.EOFException;
022    import java.io.IOException;
023    import java.io.InputStream;
024    
025    import org.apache.harmony.archive.internal.nls.Messages;
026    
027    /**
028     * The {@code GZIPInputStream} class is used to read data stored in the GZIP
029     * format, reading and decompressing GZIP data from the underlying stream into
030     * its buffer.
031     * 
032     * @since Android 1.0
033     */
034    public class GZIPInputStream extends java.util.zip.InflaterInputStream {
035    
036        /**
037         * The checksum algorithm used when handling uncompressed data.
038         * 
039         * @since Android 1.0
040         */
041        protected CRC32 crc = new CRC32();
042    
043        /**
044         * Indicates the end of the input stream.
045         * 
046         * @since Android 1.0
047         */
048        protected boolean eos = false;
049    
050        /**
051         * The magic header for the GZIP format.
052         * 
053         * @since Android 1.0
054         */
055        public final static int GZIP_MAGIC = 0x8b1f;
056    
057        private static final int FHCRC = 2;
058    
059        private static final int FEXTRA = 4;
060    
061        private static final int FNAME = 8;
062    
063        private static final int FCOMMENT = 16;
064    
065        /**
066         * Construct a {@code GZIPInputStream} to read from GZIP data from the
067         * underlying stream.
068         * 
069         * @param is
070         *            the {@code InputStream} to read data from.
071         * @throws IOException
072         *             if an {@code IOException} occurs.
073         * @since Android 1.0
074         */
075        public GZIPInputStream(InputStream is) throws IOException {
076            this(is, BUF_SIZE);
077        }
078    
079        /**
080         * Construct a {@code GZIPInputStream} to read from GZIP data from the
081         * underlying stream. Set the internal buffer size to {@code size}.
082         * 
083         * @param is
084         *            the {@code InputStream} to read data from.
085         * @param size
086         *            the internal read buffer size.
087         * @throws IOException
088         *             if an {@code IOException} occurs.
089         * @since Android 1.0
090         */
091        public GZIPInputStream(InputStream is, int size) throws IOException {
092            super(is, new Inflater(true), size);
093            byte[] header = new byte[10];
094            readFully(header, 0, header.length);
095            if (getShort(header, 0) != GZIP_MAGIC) {
096                throw new IOException(Messages.getString("archive.1F")); //$NON-NLS-1$;
097            }
098            int flags = header[3];
099            boolean hcrc = (flags & FHCRC) != 0;
100            if (hcrc) {
101                crc.update(header, 0, header.length);
102            }
103            if ((flags & FEXTRA) != 0) {
104                readFully(header, 0, 2);
105                if (hcrc) {
106                    crc.update(header, 0, 2);
107                }
108                int length = getShort(header, 0);
109                while (length > 0) {
110                    int max = length > buf.length ? buf.length : length;
111                    int result = in.read(buf, 0, max);
112                    if (result == -1) {
113                        throw new EOFException();
114                    }
115                    if (hcrc) {
116                        crc.update(buf, 0, result);
117                    }
118                    length -= result;
119                }
120            }
121            if ((flags & FNAME) != 0) {
122                readZeroTerminated(hcrc);
123            }
124            if ((flags & FCOMMENT) != 0) {
125                readZeroTerminated(hcrc);
126            }
127            if (hcrc) {
128                readFully(header, 0, 2);
129                int crc16 = getShort(header, 0);
130                if ((crc.getValue() & 0xffff) != crc16) {
131                    throw new IOException(Messages.getString("archive.20")); //$NON-NLS-1$
132                }
133                crc.reset();
134            }
135        }
136    
137        private long getLong(byte[] buffer, int off) {
138            long l = 0;
139            l |= (buffer[off] & 0xFF);
140            l |= (buffer[off + 1] & 0xFF) << 8;
141            l |= (buffer[off + 2] & 0xFF) << 16;
142            l |= ((long) (buffer[off + 3] & 0xFF)) << 24;
143            return l;
144        }
145    
146        private int getShort(byte[] buffer, int off) {
147            return (buffer[off] & 0xFF) | ((buffer[off + 1] & 0xFF) << 8);
148        }
149    
150        @Override
151        public int read(byte[] buffer, int off, int nbytes) throws IOException {
152            if (closed) {
153                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
154            }
155            if(eof){
156                return -1;
157            }
158            // avoid int overflow, check null buffer
159            if (off <= buffer.length && nbytes >= 0 && off >= 0
160                    && buffer.length - off >= nbytes) {
161                int val = super.read(buffer, off, nbytes);
162                if (val != -1) {
163                    crc.update(buffer, off, val);
164                } else if (!eos) {
165                    eos = true;
166                    // Get non-compressed bytes read by fill
167                    // BEGIN android-changed
168                    // copied from newer version of harmony
169                    int size = inf.getRemaining();
170                    final int trailerSize = 8; // crc (4 bytes) + total out (4
171                                                // bytes)
172                    byte[] b = new byte[trailerSize];
173                    int copySize = (size > trailerSize) ? trailerSize : size;
174    
175                    System.arraycopy(buf, len - size, b, 0, copySize);
176                    readFully(b, copySize, trailerSize - copySize);                
177                    // END android-changed
178                    if (getLong(b, 0) != crc.getValue()) {
179                        throw new IOException(Messages.getString("archive.20")); //$NON-NLS-1$
180                    }
181                    if ((int) getLong(b, 4) != inf.getTotalOut()) {
182                        throw new IOException(Messages.getString("archive.21")); //$NON-NLS-1$
183                    }
184                }
185                return val;
186            }
187            throw new ArrayIndexOutOfBoundsException();
188        }
189        
190        @Override
191        public void close() throws IOException {
192            eos = true;
193            super.close();
194        }
195    
196        private void readFully(byte[] buffer, int offset, int length)
197                throws IOException {
198            int result;
199            while (length > 0) {
200                result = in.read(buffer, offset, length);
201                if (result == -1) {
202                    throw new EOFException();
203                }
204                offset += result;
205                length -= result;
206            }
207        }
208    
209        private void readZeroTerminated(boolean hcrc) throws IOException {
210            int result;
211            while ((result = in.read()) > 0) {
212                if (hcrc) {
213                    crc.update(result);
214                }
215            }
216            if (result == -1) {
217                throw new EOFException();
218            }
219            // Add the zero
220            if (hcrc) {
221                crc.update(result);
222            }
223        }
224    }
