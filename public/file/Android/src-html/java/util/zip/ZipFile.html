034    /**
035     * This class provides random read access to a <i>ZIP-archive</i> file.
036     * <p>
037     * While {@code ZipInputStream} provides stream based read access to a
038     * <i>ZIP-archive</i>, this class implements more efficient (file based) access
039     * and makes use of the <i>central directory</i> within a <i>ZIP-archive</i>.
040     * </p>
041     * <p>
042     * Use {@code ZipOutputStream} if you want to create an archive.
043     * </p>
044     * <p>
045     * A temporary ZIP file can be marked for automatic deletion upon closing it.
046     * </p>
047     * 
048     * @see ZipEntry
049     * @see ZipOutputStream
050     * @since Android 1.0
051     */
052    public class ZipFile implements ZipConstants {
053    
054        String fileName;
055    
056        File fileToDeleteOnClose;
057    
058        /**
059         * Open zip file for read.
060         * 
061         * @since Android 1.0
062         */
063        public static final int OPEN_READ = 1;
064    
065        /**
066         * Delete zip file when closed.
067         * 
068         * @since Android 1.0
069         */
070        public static final int OPEN_DELETE = 4;
071    
072        /**
073         * Constructs a new {@code ZipFile} with the specified file.
074         * 
075         * @param file
076         *            the file to read from.
077         * @throws ZipException
078         *             if a ZIP error occurs.
079         * @throws IOException
080         *             if an {@code IOException} occurs.
081         * @since Android 1.0
082         */
083        public ZipFile(File file) throws ZipException, IOException {
084            this(file, OPEN_READ);
085        }
086    
087        /**
088         * Opens a file as <i>ZIP-archive</i>. "mode" must be {@code OPEN_READ} or
089         * {@code OPEN_DELETE} . The latter sets the "delete on exit" flag through a
090         * file.
091         * 
092         * @param file
093         *            the ZIP file to read.
094         * @param mode
095         *            the mode of the file open operation.
096         * @throws IOException
097         *             if an {@code IOException} occurs.
098         * @since Android 1.0
099         */
100        public ZipFile(File file, int mode) throws IOException {
101            if (mode == (OPEN_READ | OPEN_DELETE))
102                fileToDeleteOnClose = file; // file.deleteOnExit();
103            else if (mode != OPEN_READ)
104                throw new IllegalArgumentException("invalid mode");
105    
106            fileName = file.getPath();
107            mRaf = new RandomAccessFile(fileName, "r");
108    
109            mEntryList = new ArrayList<ZipEntry>();
110    
111            readCentralDir();
112    
113            /*
114             * No LinkedHashMap yet, so optimize lookup-by-name by creating
115             * a parallel data structure.
116             */
117            mFastLookup = new HashMap<String, ZipEntry>(mEntryList.size() * 2);
118            for (int i = 0; i < mEntryList.size(); i++) {
119                ZipEntry entry = mEntryList.get(i);
120    
121                mFastLookup.put(entry.getName(), entry);
122            }
123        }
124    
125        /**
126         * Opens a ZIP archived file.
127         * 
128         * @param name
129         *            the name of the ZIP file.
130         * @throws IOException
131         *             if an IOException occurs.
132         * @since Android 1.0
133         */
134        public ZipFile(String name) throws IOException {
135            this(new File(name), OPEN_READ);
136        }
137    
138        @Override
139        protected void finalize() throws IOException {
140            close();
141        }
142    
143        /**
144         * Closes this ZIP file.
145         * 
146         * @throws IOException
147         *             if an IOException occurs.
148         * @since Android 1.0
149         */
150        public void close() throws IOException {
151            RandomAccessFile raf = mRaf;
152    
153            if (raf != null) { // Only close initialized instances
154                synchronized(raf) {
155                    mRaf = null;
156                    raf.close();
157                }
158                if (fileToDeleteOnClose != null) {
159                    AccessController.doPrivileged(new PrivilegedAction<Object>() {
160                        public Object run() {
161                            new File(fileName).delete();
162                            return null;
163                        }
164                    });
165                    // fileToDeleteOnClose.delete();
166                    fileToDeleteOnClose = null;
167                }
168            }
169        }
170    
171        /**
172         * Returns an enumeration of the entries. The entries are listed in the
173         * order in which they appear in the ZIP archive.
174         * 
175         * @return the enumeration of the entries.
176         * @since Android 1.0
177         */
178        public Enumeration<? extends ZipEntry> entries() {
179            return new Enumeration<ZipEntry>() {
180                private int i = 0;
181    
182                public boolean hasMoreElements() {
183                    if (mRaf == null) throw new IllegalStateException("Zip File closed.");
184                    return i < mEntryList.size();
185                }
186    
187                public ZipEntry nextElement() {
188                    if (mRaf == null) throw new IllegalStateException("Zip File closed.");
189                    if (i >= mEntryList.size())
190                        throw new NoSuchElementException();
191                    return (ZipEntry) mEntryList.get(i++);
192                }
193            };
194        }
195    
196        /**
197         * Gets the ZIP entry with the specified name from this {@code ZipFile}.
198         * 
199         * @param entryName
200         *            the name of the entry in the ZIP file.
201         * @return a {@code ZipEntry} or {@code null} if the entry name does not
202         *         exist in the ZIP file.
203         * @since Android 1.0
204         */
205        public ZipEntry getEntry(String entryName) {
206            if (entryName != null) {
207                ZipEntry ze = mFastLookup.get(entryName);
208                if (ze == null) ze = mFastLookup.get(entryName + "/");
209                return ze;
210            }
211            throw new NullPointerException();
212        }
213    
214        /**
215         * Returns an input stream on the data of the specified {@code ZipEntry}.
216         * 
217         * @param entry
218         *            the ZipEntry.
219         * @return an input stream of the data contained in the {@code ZipEntry}.
220         * @throws IOException
221         *             if an {@code IOException} occurs.
222         * @since Android 1.0
223         */
224        public InputStream getInputStream(ZipEntry entry) throws IOException {
225            /*
226             * Make sure this ZipEntry is in this Zip file.  We run it through
227             * the name lookup.
228             */
229            entry = getEntry(entry.getName());
230            if (entry == null)
231                return null;
232    
233            /*
234             * Create a ZipInputStream at the right part of the file.
235             */
236            RandomAccessFile raf = mRaf;
237            if (raf != null) {
238                synchronized (raf) {
239                    // Unfortunately we don't know the entry data's start position.
240                    // All we have is the position of the entry's local header.
241                    // At position 28 we find the length of the extra data.
242                    // In some cases this length differs from the one coming in
243                    // the central header!!!
244                    RAFStream rafstrm = new RAFStream(raf, entry.mLocalHeaderRelOffset + 28);
245                    int localExtraLenOrWhatever = ler.readShortLE(rafstrm);
246                    // Now we need to skip the name
247                    // and this "extra" data or whatever it is:
248                    rafstrm.skip(entry.nameLen + localExtraLenOrWhatever);
249                    rafstrm.mLength = rafstrm.mOffset + entry.compressedSize;
250                    if (entry.compressionMethod == ZipEntry.DEFLATED) {
251                        return new InflaterInputStream(rafstrm, new Inflater(true));
252                    } else {
253                        return rafstrm;
254                    }
255                }
256            }
257            throw new IllegalStateException("Zip File closed");
258        }
259    
260        /**
261         * Gets the file name of this {@code ZipFile}.
262         * 
263         * @return the file name of this {@code ZipFile}.
264         * @since Android 1.0
265         */
266        public String getName() {
267            return fileName;
268        }

270        /**
271         * Returns the number of {@code ZipEntries} in this {@code ZipFile}.
272         * 
273         * @return the number of entries in this file.
274         * @since Android 1.0
275         */
276        public int size() {
277            return mEntryList.size();
278        }

280        /*
281         * Find the central directory and read the contents.
282         *
283         * The central directory can be followed by a variable-length comment
284         * field, so we have to scan through it backwards.  The comment is at
285         * most 64K, plus we have 18 bytes for the end-of-central-dir stuff
286         * itself, plus apparently sometimes people throw random junk on the end
287         * just for the fun of it.
288         *  
289         * This is all a little wobbly.  If the wrong value ends up in the EOCD
290         * area, we're hosed.  This appears to be the way that everbody handles
291         * it though, so we're in pretty good company if this fails.
292         */
293        private void readCentralDir() throws IOException {
294            long scanOffset, stopOffset;
295            long sig;
296    
297            /*
298             * Scan back, looking for the End Of Central Directory field.  If
299             * the archive doesn't have a comment, we'll hit it on the first
300             * try.
301             *
302             * No need to synchronize mRaf here -- we only do this when we
303             * first open the Zip file.
304             */
305            scanOffset = mRaf.length() - ENDHDR;
306            if (scanOffset < 0)
307                throw new ZipException("too short to be Zip");
308    
309            stopOffset = scanOffset - 65536;
310            if (stopOffset < 0)
311                stopOffset = 0;
312    
313            while (true) {
314                mRaf.seek(scanOffset);
315                if (ZipEntry.readIntLE(mRaf) == 101010256L)
316                    break;
317    
318                //System.out.println("not found at " + scanOffset);
319                scanOffset--;
320                if (scanOffset < stopOffset)
321                    throw new ZipException("EOCD not found; not a Zip archive?");
322            }
323    
324            /*
325             * Found it, read the EOCD.
326             *
327             * For performance we want to use buffered I/O when reading the
328             * file.  We wrap a buffered stream around the random-access file
329             * object.  If we just read from the RandomAccessFile we'll be
330             * doing a read() system call every time.
331             */
332            RAFStream rafs = new RAFStream(mRaf, mRaf.getFilePointer());
333            BufferedInputStream bin = new BufferedInputStream(rafs, ENDHDR);
334            int diskNumber, diskWithCentralDir, numEntries, totalNumEntries;
335            //long centralDirSize;
336            long centralDirOffset;
337            //int commentLen;
338    
339            diskNumber = ler.readShortLE(bin);
340            diskWithCentralDir = ler.readShortLE(bin);
341            numEntries = ler.readShortLE(bin);
342            totalNumEntries = ler.readShortLE(bin);
343            /*centralDirSize =*/ ler.readIntLE(bin);
344            centralDirOffset = ler.readIntLE(bin);
345            /*commentLen =*/ ler.readShortLE(bin);
346    
347            if (numEntries != totalNumEntries ||
348                diskNumber != 0 ||
349                diskWithCentralDir != 0)
350                throw new ZipException("spanned archives not supported");
351    
352            /*
353             * Seek to the first CDE and read all entries.
354             */
355            rafs = new RAFStream(mRaf, centralDirOffset);
356            bin = new BufferedInputStream(rafs, 4096);
357            for (int i = 0; i < numEntries; i++) {
358                ZipEntry newEntry;
359    
360                newEntry = new ZipEntry(ler, bin);
361                mEntryList.add(newEntry);
362            }
363        }
364    
365        /*
366         * Local data items.
367         */
368        private RandomAccessFile mRaf;
369    
370        ZipEntry.LittleEndianReader ler = new ZipEntry.LittleEndianReader();
371    
372        /*
373         * What we really want here is a LinkedHashMap, because we want fast
374         * lookups by name, but we want to preserve the ordering of the archive
375         * entries.  Unfortunately we don't yet have a LinkedHashMap
376         * implementation.
377         */
378        private ArrayList<ZipEntry> mEntryList;
379        private HashMap<String, ZipEntry> mFastLookup;
380    
381        /*
382         * Wrap a stream around a RandomAccessFile.  The RandomAccessFile
383         * is shared among all streams returned by getInputStream(), so we
384         * have to synchronize access to it.  (We can optimize this by
385         * adding buffering here to reduce collisions.)
386         *
387         * We could support mark/reset, but we don't currently need them.
388         */
389        static class RAFStream extends InputStream {
390            public RAFStream(RandomAccessFile raf, long pos) throws IOException {
391                mSharedRaf = raf;
392                mOffset = pos;
393                mLength = raf.length();
394            }
395    
396            @Override
397            public int available() throws IOException {
398                return (mOffset < mLength ? 1 : 0);
399            }
400    
401            public int read() throws IOException {
402                if (read(singleByteBuf, 0, 1) == 1) return singleByteBuf[0] & 0XFF;
403                else return -1;
404            }
405    
406            public int read(byte[] b, int off, int len) throws IOException {
407                int count;
408                synchronized (mSharedRaf) {
409                    mSharedRaf.seek(mOffset);
410                    if (mOffset + len > mLength) len = (int) (mLength - mOffset);
411                    count = mSharedRaf.read(b, off, len);
412                    if (count > 0) {
413                        mOffset += count;
414                    }
415                    else return -1;
416                }
417                return count;
418            }
419    
420            @Override
421            public long skip(long n) throws IOException {
422                if (mOffset + n > mLength)
423                    n = mLength - mOffset;
424                mOffset += n;
425                return n;
426            }
427    
428            RandomAccessFile mSharedRaf;
429            long mOffset;
430            long mLength;
431            private byte[] singleByteBuf = new byte[1];
432        }
