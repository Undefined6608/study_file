018    package java.util.zip;
019    
020    
021    import java.io.EOFException;
022    import java.io.FilterInputStream;
023    import java.io.IOException;
024    import java.io.InputStream;
025    
026    import org.apache.harmony.archive.internal.nls.Messages;
027    
028    /**
029     * This class provides an implementation of {@code FilterInputStream} that
030     * uncompresses data that was compressed using the <i>DEFLATE</i> algorithm
031     * (see <a href="http://www.gzip.org/algorithm.txt">specification</a>).
032     * Basically it wraps the {@code Inflater} class and takes care of the
033     * buffering.
034     * 
035     * @see Inflater
036     * @see DeflaterOutputStream
037     * @since Android 1.0
038     */
039    public class InflaterInputStream extends FilterInputStream {
040    
041        /**
042         * The inflater used for this stream.
043         * 
044         * @since Android 1.0
045         */
046        protected Inflater inf;
047    
048        /**
049         * The input buffer used for decompression.
050         * 
051         * @since Android 1.0
052         */
053        protected byte[] buf;
054    
055        /**
056         * The length of the buffer.
057         * 
058         * @since Android 1.0
059         */
060        protected int len;
061    
062        boolean closed;
063    
064        boolean eof;
065    
066        static final int BUF_SIZE = 512;
067    
068        // BEGIN android-added
069        int nativeEndBufSize = 0;
070        // END android-added
071    
072        /**
073         * This is the most basic constructor. You only need to pass the {@code
074         * InputStream} from which the compressed data is to be read from. Default
075         * settings for the {@code Inflater} and internal buffer are be used. In
076         * particular the Inflater expects a ZLIB header from the input stream.
077         * 
078         * @param is
079         *            the {@code InputStream} to read data from.
080         * @since Android 1.0
081         */
082        public InflaterInputStream(InputStream is) {
083            this(is, new Inflater(), BUF_SIZE);
084        }
085    
086        /**
087         * This constructor lets you pass a specifically initialized Inflater,
088         * for example one that expects no ZLIB header.
089         * 
090         * @param is
091         *            the {@code InputStream} to read data from.
092         * @param inf
093         *            the specific {@code Inflater} for uncompressing data.
094         * @since Android 1.0 
095         */
096        public InflaterInputStream(InputStream is, Inflater inf) {
097            this(is, inf, BUF_SIZE);
098        }
099    
100        /**
101         * This constructor lets you specify both the {@code Inflater} as well as
102         * the internal buffer size to be used.
103         * 
104         * @param is
105         *            the {@code InputStream} to read data from.
106         * @param inf
107         *            the specific {@code Inflater} for uncompressing data.
108         * @param bsize
109         *            the size to be used for the internal buffer.
110         * @since Android 1.0
111         */
112        public InflaterInputStream(InputStream is, Inflater inf, int bsize) {
113            super(is);
114            if (is == null || inf == null) {
115                throw new NullPointerException();
116            }
117            if (bsize <= 0) {
118                throw new IllegalArgumentException();
119            }
120            this.inf = inf;
121            // BEGIN android-changed
122            if (is instanceof ZipFile.RAFStream) {
123                nativeEndBufSize = bsize;
124            } else {
125                buf = new byte[bsize];
126            }
127            // END android-changed
128        }
129    
130        /**
131         * Reads a single byte of decompressed data.
132         * 
133         * @return the byte read.
134         * @throws IOException
135         *             if an error occurs reading the byte.
136         * @since Android 1.0
137         */
138        @Override
139        public int read() throws IOException {
140            byte[] b = new byte[1];
141            if (read(b, 0, 1) == -1) {
142                return -1;
143            }
144            return b[0] & 0xff;
145        }
146    
147        /**
148         * Reads up to {@code nbytes} of decompressed data and stores it in
149         * {@code buffer} starting at {@code off}.
150         * 
151         * @param buffer
152         *            the buffer to write data to.
153         * @param off
154         *            offset in buffer to start writing.
155         * @param nbytes
156         *            number of bytes to read.
157         * @return Number of uncompressed bytes read
158         * @throws IOException
159         *             if an IOException occurs.
160         * @since Android 1.0
161         */
162        @Override
163        public int read(byte[] buffer, int off, int nbytes) throws IOException {
164            /* archive.1E=Stream is closed */
165            if (closed) {
166                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
167            }
168    
169            if (null == buffer) {
170                throw new NullPointerException();
171            }
172    
173            if (off < 0 || nbytes < 0 || off + nbytes > buffer.length) {
174                throw new IndexOutOfBoundsException();
175            }
176    
177            if (nbytes == 0) {
178                return 0;
179            }
180    
181            if (inf.finished()) {
182                eof = true;
183                return -1;
184            }
185    
186            // avoid int overflow, check null buffer
187            if (off <= buffer.length && nbytes >= 0 && off >= 0
188                    && buffer.length - off >= nbytes) {
189                do {
190                    if (inf.needsInput()) {
191                        fill();
192                    }
193                    int result;
194                    try {
195                        result = inf.inflate(buffer, off, nbytes);
196                    } catch (DataFormatException e) {
197                        if (len == -1) {
198                            throw new EOFException();
199                        }
200                        throw (IOException)(new IOException().initCause(e));
201                    }
202                    if (result > 0) {
203                        return result;
204                    } else if (inf.finished()) {
205                        eof = true;
206                        return -1;
207                    } else if (inf.needsDictionary()) {
208                        return -1;
209                    } else if (len == -1) {
210                        throw new EOFException();
211                    // If result == 0, fill() and try again
212                    }
213                } while (true);
214            }
215            throw new ArrayIndexOutOfBoundsException();
216        }
217    
218        
219        /**
220         * Fills the input buffer with data to be decompressed.
221         * 
222         * @throws IOException
223         *             if an {@code IOException} occurs.
224         * @since Android 1.0
225         */
226        protected void fill() throws IOException {
227            if (closed) {
228                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
229            }
230            // BEGIN android-changed
231            if (nativeEndBufSize > 0) {
232                ZipFile.RAFStream is = (ZipFile.RAFStream)in;
233                synchronized (is.mSharedRaf) {
234                    long len = is.mLength - is.mOffset;
235                    if (len > nativeEndBufSize) len = nativeEndBufSize;
236                    int cnt = inf.setFileInput(is.mSharedRaf.getFD(), is.mOffset, (int)nativeEndBufSize);
237                    is.skip(cnt);
238                }
239            } else {
240                if ((len = in.read(buf)) > 0) {
241                    inf.setInput(buf, 0, len);
242                }
243            }
244            // END android-changed
245        }
246    
247        /**
248         * Skips up to n bytes of uncompressed data.
249         * 
250         * @param nbytes
251         *            the number of bytes to skip.
252         * @return the number of uncompressed bytes skipped.
253         * @throws IOException
254         *             if an error occurs skipping.
255         * @since Android 1.0
256         */
257        @Override
258        public long skip(long nbytes) throws IOException {
259            if (nbytes >= 0) {
260                long count = 0, rem = 0;
261                while (count < nbytes) {
262                    int x = read(buf, 0,
263                            (rem = nbytes - count) > buf.length ? buf.length
264                                    : (int) rem);
265                    if (x == -1) {
266                        eof = true;
267                        return count;
268                    }
269                    count += x;
270                }
271                return count;
272            }
273            throw new IllegalArgumentException();
274        }
275    
276        /**
277         * Returns whether data can be read from this stream.
278         * 
279         * @return 0 if this stream has been closed, 1 otherwise.
280         * @throws IOException
281         *             If an error occurs.
282         * @since Android 1.0
283         */
284        @Override
285        public int available() throws IOException {
286            if (closed) {
287                // archive.1E=Stream is closed
288                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
289            }
290            if (eof) {
291                return 0;
292            }
293            return 1;
294        }
295    
296        /**
297         * Closes the input stream.
298         * 
299         * @throws IOException
300         *             If an error occurs closing the input stream.
301         * @since Android 1.0
302         */
303        @Override
304        public void close() throws IOException {
305            if (!closed) {
306                inf.end();
307                closed = true;
308                eof = true;
309                super.close();
310            }
311        }
312        
313        /**
314         * This implementation overrides the super type implementation to do nothing
315         * at all.
316         * 
317         * @param readlimit
318         *            of no use.
319         * @since Android 1.0
320         */
321        @Override
322        @SuppressWarnings("unused")
323        public void mark(int readlimit) {
324            // do nothing
325        }
326    
327        /**
328         * Reset the position of the stream to the last marked position. This
329         * implementation overrides the supertype implementation and always throws
330         * an {@link IOException IOException} when called.
331         * 
332         * @throws IOException
333         *             if the method is called
334         * @since Android 1.0
335         */
336        @Override
337        public void reset() throws IOException{
338            throw new IOException();
339        }
340        
341        /**
342         * Returns whether the receiver implements {@code mark} semantics. This type
343         * does not support {@code mark()}, so always responds {@code false}.
344         * 
345         * @return false, always
346         * @since Android 1.0
347         */
348        @Override
349        public boolean markSupported() {
350            return false;
351        }
