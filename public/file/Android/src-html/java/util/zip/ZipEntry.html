350         * 
351         * @param value
352         *            the modification time as the number of milliseconds since Jan.
353         *            1, 1970.
354         * @since Android 1.0
355         */
356        public void setTime(long value) {
357            GregorianCalendar cal = new GregorianCalendar();
358            cal.setTime(new Date(value));
359            int year = cal.get(Calendar.YEAR);
360            if (year < 1980) {
361                modDate = 0x21;
362                time = 0;
363            } else {
364                modDate = cal.get(Calendar.DATE);
365                modDate = (cal.get(Calendar.MONTH) + 1 << 5) | modDate;
366                modDate = ((cal.get(Calendar.YEAR) - 1980) << 9) | modDate;
367                time = cal.get(Calendar.SECOND) >> 1;
368                time = (cal.get(Calendar.MINUTE) << 5) | time;
369                time = (cal.get(Calendar.HOUR_OF_DAY) << 11) | time;
370            }
371        }

460        /*
461         * Internal constructor.  Creates a new ZipEntry by reading the
462         * Central Directory Entry from "in", which must be positioned at
463         * the CDE signature.
464         *
465         * On exit, "in" will be positioned at the start of the next entry.
466         */
467        /*package*/ ZipEntry(LittleEndianReader ler, InputStream in) throws IOException {
468    
469            /*
470             * We're seeing performance issues when we call readShortLE and
471             * readIntLE, so we're going to read the entire header at once
472             * and then parse the results out without using any function calls.
473             * Uglier, but should be much faster.
474             * 
475             * Note that some lines look a bit different, because the corresponding
476             * fields or locals are long and so we need to do & 0xffffffffl to avoid
477             * problems induced by sign extension.
478             */
479    
480            byte[] hdrBuf = ler.hdrBuf;
481            myReadFully(in, hdrBuf);
482    
483            long sig = (hdrBuf[0] & 0xff) | ((hdrBuf[1] & 0xff) << 8) |
484                ((hdrBuf[2] & 0xff) << 16) | ((hdrBuf[3] << 24) & 0xffffffffL);
485            if (sig != CENSIG)
486                 throw new ZipException("Central Directory Entry not found");
487    
488            mVersionMadeBy = (hdrBuf[4] & 0xff) | ((hdrBuf[5] & 0xff) << 8);
489            mVersionToExtract = (hdrBuf[6] & 0xff) | ((hdrBuf[7] & 0xff) << 8);
490            mGPBitFlag = (hdrBuf[8] & 0xff) | ((hdrBuf[9] & 0xff) << 8);
491            compressionMethod = (hdrBuf[10] & 0xff) | ((hdrBuf[11] & 0xff) << 8);
492            time = (hdrBuf[12] & 0xff) | ((hdrBuf[13] & 0xff) << 8);
493            modDate = (hdrBuf[14] & 0xff) | ((hdrBuf[15] & 0xff) << 8);
494            crc = (hdrBuf[16] & 0xff) | ((hdrBuf[17] & 0xff) << 8)
495                    | ((hdrBuf[18] & 0xff) << 16)
496                    | ((hdrBuf[19] << 24) & 0xffffffffL);
497            compressedSize = (hdrBuf[20] & 0xff) | ((hdrBuf[21] & 0xff) << 8)
498                    | ((hdrBuf[22] & 0xff) << 16)
499                    | ((hdrBuf[23] << 24) & 0xffffffffL);
500            size = (hdrBuf[24] & 0xff) | ((hdrBuf[25] & 0xff) << 8)
501                    | ((hdrBuf[26] & 0xff) << 16)
502                    | ((hdrBuf[27] << 24) & 0xffffffffL);
503            nameLen = (hdrBuf[28] & 0xff) | ((hdrBuf[29] & 0xff) << 8);
504            extraLen = (hdrBuf[30] & 0xff) | ((hdrBuf[31] & 0xff) << 8);
505            commentLen = (hdrBuf[32] & 0xff) | ((hdrBuf[33] & 0xff) << 8);
506            mDiskNumberStart = (hdrBuf[34] & 0xff) | ((hdrBuf[35] & 0xff) << 8);
507            mInternalAttrs = (hdrBuf[36] & 0xff) | ((hdrBuf[37] & 0xff) << 8);
508            mExternalAttrs = (hdrBuf[38] & 0xff) | ((hdrBuf[39] & 0xff) << 8)
509                    | ((hdrBuf[40] & 0xff) << 16)
510                    | ((hdrBuf[41] << 24) & 0xffffffffL);
511            mLocalHeaderRelOffset = (hdrBuf[42] & 0xff) | ((hdrBuf[43] & 0xff) << 8)
512                    | ((hdrBuf[44] & 0xff) << 16)
513                    | ((hdrBuf[45] << 24) & 0xffffffffL);
514    
515            byte[] nameBytes = new byte[nameLen];
516            myReadFully(in, nameBytes);
517    
518            byte[] commentBytes = null;
519            if (commentLen > 0) {
520                commentBytes = new byte[commentLen];
521                myReadFully(in, commentBytes);
522            }
523    
524            if (extraLen > 0) {
525                extra = new byte[extraLen];
526                myReadFully(in, extra);
527            }
528    
529            try {
530                /*
531                 * The actual character set is "IBM Code Page 437".  As of
532                 * Sep 2006, the Zip spec (APPNOTE.TXT) supports UTF-8.  When
533                 * bit 11 of the GP flags field is set, the file name and
534                 * comment fields are UTF-8.
535                 *
536                 * TODO: add correct UTF-8 support.
537                 */
538                name = new String(nameBytes, "ISO-8859-1");
539                if (commentBytes != null)
540                    comment = new String(commentBytes, "ISO-8859-1");
541                else
542                    comment = null;
543            }
544            catch (UnsupportedEncodingException uee) {
545                throw new InternalError(uee.getMessage());
546            }
547        }
548        private void myReadFully(InputStream in, byte[] b) throws IOException {
549            int count;
550            int len = b.length;
551            int off = 0;
552        
553            while (len > 0) {
554                count = in.read(b, off, len);
555                if (count <= 0)
556                    throw new EOFException();
557                off += count;
558                len -= count;
559            }
560        }

585        /*
586         * Read a two-byte short in little-endian order.
587         *
588         * The DataInput interface, which RandomAccessFile implements, provides
589         * a readInt() function.  Unfortunately, it's defined to use big-endian.
590         */
591        /*package*/ static int readShortLE(RandomAccessFile raf) throws IOException
592        {
593            int b0, b1;
594    
595            b0 = raf.read();
596            b1 = raf.read();
597            if (b1 < 0)
598                throw new EOFException("in ZipEntry.readShortLE(RandomAccessFile)");
599            return b0 | (b1 << 8);
600        }
601    
602        /*
603         * Read a four-byte int in little-endian order.
604         */
605        /*package*/ static long readIntLE(RandomAccessFile raf) throws IOException
606        {
607            int b0, b1, b2, b3;
608    
609            b0 = raf.read();
610            b1 = raf.read();
611            b2 = raf.read();
612            b3 = raf.read();
613            if (b3 < 0)
614                throw new EOFException("in ZipEntry.readIntLE(RandomAccessFile)");
615            return b0 | (b1 << 8) | (b2 << 16) | (b3 << 24); // ATTENTION: DOES SIGN EXTENSION: IS THIS WANTED?
616        }

618        static class LittleEndianReader {
619            private byte[] b = new byte[4];
620            byte[] hdrBuf = new byte[CENHDR];
621    
622            /*
623             * Read a two-byte short in little-endian order.
624             */
625            int readShortLE(InputStream in) throws IOException {
626                if (in.read(b, 0, 2) == 2) {
627                    return (b[0] & 0XFF) | ((b[1] & 0XFF) << 8);
628                } else {
629                    throw new EOFException("in ZipEntry.readShortLE(InputStream)");
630                }
631            }
632    
633            /*
634             * Read a four-byte int in little-endian order.
635             */
636            long readIntLE(InputStream in) throws IOException {
637                if (in.read(b, 0, 4) == 4) {
638                    return (   ((b[0] & 0XFF))
639                             | ((b[1] & 0XFF) << 8)
640                             | ((b[2] & 0XFF) << 16)
641                             | ((b[3] & 0XFF) << 24))
642                           & 0XFFFFFFFFL; // Here for sure NO sign extension is wanted.
643                } else {
644                    throw new EOFException("in ZipEntry.readIntLE(InputStream)");
645                }
646            }
647        }

649        /*
650         * Write a two-byte short in little-endian order.
651         */
652        /*package*/ static void writeShortLE(OutputStream out, int val)
653            throws IOException
654        {
655            out.write(val & 0xff);
656            out.write((val >> 8) & 0xff);
657        }

659        /*
660         * Write a 4-byte int in little-endian order.  This takes a long because
661         * all of our 4-byte values are stored locally in longs.
662         */
663        /*package*/ static void writeIntLE(OutputStream out, long val)
664            throws IOException
665        {
666            if (val < 0)
667                throw new InternalError();
668            out.write((int) val & 0xff);
669            out.write(((int) val >> 8) & 0xff);
670            out.write(((int) val >> 16) & 0xff);
671            out.write(((int) val >> 24) & 0xff);
672        }

674        /*
675         * Write the Local File Header for this entry to the specified stream.
676         *
677         * Returns the #of bytes written.
678         */
679        /*package*/ int writeLFH(OutputStream out) throws IOException {
680            if (compressionMethod < 0 ||
681                time < 0 ||
682                modDate < 0 ||
683                crc < 0 ||
684                compressedSize < 0 ||
685                size < 0)
686                throw new InternalError();
687    
688            writeIntLE(out, LOCSIG);
689            writeShortLE(out, mVersionToExtract);
690            writeShortLE(out, mGPBitFlag);
691            writeShortLE(out, compressionMethod);
692            writeShortLE(out, time);
693            writeShortLE(out, modDate);
694            writeIntLE(out, crc);
695            writeIntLE(out, compressedSize);
696            writeIntLE(out, size);
697    
698            byte[] nameBytes;
699            try {
700                nameBytes = name.getBytes("ISO-8859-1");
