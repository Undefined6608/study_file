026    /**
027     * This class uncompresses data that was compressed using the <i>DEFLATE</i>
028     * algorithm (see <a href="http://www.gzip.org/algorithm.txt">specification</a>).
029     * <p>
030     * Basically this class is part of the API to the stream based ZLIB compression
031     * library and is used as such by {@code InflaterInputStream} and its
032     * descendants.
033     * </p>
034     * <p>
035     * The typical usage of a {@code Inflater} outside this package consists of a
036     * specific call to one of its constructors before being passed to an instance
037     * of {@code InflaterInputStream}.
038     * </p>
039     * 
040     * @see InflaterInputStream
041     * @see Deflater
042     * @since Android 1.0
043     */
044    public class Inflater {
045    
046        private boolean finished; // Set by the inflateImpl native
047    
048        private boolean needsDictionary; // Set by the inflateImpl native
049    
050        private long streamHandle = -1;
051    
052        int inRead;
053        
054        int inLength;
055    
056        // Fill in the JNI id caches
057        private static native void oneTimeInitialization();
058    
059        static {
060            oneTimeInitialization();
061        }
062        
063        private static final byte MAGIC_NUMBER = 120;
064        private boolean gotFirstByte = false;
065        private boolean pass_magic_number_check = true;
066        
067        /**
068         * Release any resources associated with this {@code Inflater}. Any unused
069         * input/output is discarded. This is also called by the finalize method.
070         * 
071         * @since Android 1.0
072         */
073        public synchronized void end() {
074            if (streamHandle != -1) {
075                endImpl(streamHandle);
076                inRead = 0;
077                inLength = 0;
078                streamHandle = -1;
079            }
080        }
081    
082        private native synchronized void endImpl(long handle);
083    
084        @Override
085        protected void finalize() {
086            end();
087        }
088    
089        /**
090         * Indicates if the {@code Inflater} has inflated the entire deflated
091         * stream. If deflated bytes remain and {@code needsInput()} returns {@code
092         * true} this method will return {@code false}. This method should be
093         * called after all deflated input is supplied to the {@code Inflater}.
094         * 
095         * @return {@code true} if all input has been inflated, {@code false}
096         *         otherwise.
097         * @since Android 1.0
098         */
099        public synchronized boolean finished() {
100            return finished;
101        }
102    
103        /**
104         * Returns the <i>Adler32</i> checksum of either all bytes inflated, or the
105         * checksum of the preset dictionary if one has been supplied.
106         * 
107         * @return The <i>Adler32</i> checksum associated with this
108         *         {@code Inflater}.
109         * @since Android 1.0 .
110         */
111        public synchronized int getAdler() {
112            if (streamHandle == -1) {
113                throw new IllegalStateException();
114            }
115            return getAdlerImpl(streamHandle);
116        }
117    
118        private native synchronized int getAdlerImpl(long handle);
119    
120        /**
121         * Returns the number of bytes of current input remaining to be read by the
122         * inflater.
123         * 
124         * @return the number of bytes of unread input.
125         * @since Android 1.0
126         */
127        public synchronized int getRemaining() {
128            return inLength - inRead;
129        }
130    
131        /**
132         * Returns total number of bytes of input read by the {@code Inflater}. The
133         * result value is limited by {@code Integer.MAX_VALUE}.
134         * 
135         * @return the total number of bytes read.
136         * @since Android 1.0
137         */
138        public synchronized int getTotalIn() {
139            if (streamHandle == -1) {
140                throw new IllegalStateException();
141            }
142            long totalIn = getTotalInImpl(streamHandle);
143            return (totalIn <= Integer.MAX_VALUE ? (int) totalIn
144                    : Integer.MAX_VALUE);
145        }
146    
147        private synchronized native long getTotalInImpl(long handle);
148    
149        /**
150         * Returns total number of bytes written to the output buffer by the {@code
151         * Inflater}. The result value is limited by {@code Integer.MAX_VALUE}.
152         * 
153         * @return the total bytes of output data written.
154         * @since Android 1.0
155         */
156        public synchronized int getTotalOut() {
157            if (streamHandle == -1) {
158                throw new IllegalStateException();
159            }
160            long totalOut = getTotalOutImpl(streamHandle);
161            return (totalOut <= Integer.MAX_VALUE ? (int) totalOut
162                    : Integer.MAX_VALUE);
163        }
164    
165        private native synchronized long getTotalOutImpl(long handle);
166    
167        /**
168         * Inflates bytes from current input and stores them in {@code buf}.
169         * 
170         * @param buf
171         *            the buffer where decompressed data bytes are written.
172         * @return the number of bytes inflated.
173         * @throws DataFormatException
174         *             if the underlying stream is corrupted or was not compressed
175         *             using a {@code Deflater}.
176         * @since Android 1.0
177         */
178        public int inflate(byte[] buf) throws DataFormatException {
179            return inflate(buf, 0, buf.length);
180        }
181    
182        /**
183         * Inflates up to n bytes from the current input and stores them in {@code
184         * buf} starting at {@code off}.
185         * 
186         * @param buf
187         *            the buffer to write inflated bytes to.
188         * @param off
189         *            the offset in buffer where to start writing decompressed data.
190         * @param nbytes
191         *            the number of inflated bytes to write to {@code buf}.
192         * @throws DataFormatException
193         *             if the underlying stream is corrupted or was not compressed
194         *             using a {@code Deflater}.
195         * @return the number of bytes inflated.
196         */
197        public synchronized int inflate(byte[] buf, int off, int nbytes)
198                throws DataFormatException {
199            // avoid int overflow, check null buf
200            if (off <= buf.length && nbytes >= 0 && off >= 0
201                    && buf.length - off >= nbytes) {
202                if (nbytes == 0)
203                    return 0;
204    
205                if (streamHandle == -1) {
206                    throw new IllegalStateException();
207                }
208                
209                if (!pass_magic_number_check) {
210                    throw new DataFormatException();
211                }
212    
213                if (needsInput()) {
214                    return 0;
215                }
216                
217                boolean neededDict = needsDictionary;
218                needsDictionary = false;
219                int result = inflateImpl(buf, off, nbytes, streamHandle);
220                if (needsDictionary && neededDict) {
221                    throw new DataFormatException(Messages.getString("archive.27")); //$NON-NLS-1$
222                }
223                return result;
224            }
225            throw new ArrayIndexOutOfBoundsException();
226        }
227    
228        private native synchronized int inflateImpl(byte[] buf, int off,
229                int nbytes, long handle);
230    
231        /**
232         * This constructor creates an inflater that expects a header from the input
233         * stream. Use {@code Inflater(boolean)} if the input comes without a ZLIB
234         * header.
235         * 
236         * @since Android 1.0
237         * @since Android 1.0
238         */
239        public Inflater() {
240            this(false);
241        }
242    
243        /**
244         * This constructor allows to create an inflater that expects no header from
245         * the input stream.
246         * 
247         * @param noHeader
248         *            {@code true} indicates that no ZLIB header comes with the
249         *            input.
250         * @since Android 1.0
251         */
252        public Inflater(boolean noHeader) {
253            streamHandle = createStream(noHeader);
254        }
255    
256        /**
257         * Indicates whether the input bytes were compressed with a preset
258         * dictionary. This method should be called prior to {@code inflate()} to
259         * determine whether a dictionary is required. If so {@code setDictionary()}
260         * should be called with the appropriate dictionary prior to calling {@code
261         * inflate()}.
262         * 
263         * @return {@code true} if a preset dictionary is required for inflation.
264         * @see #setDictionary(byte[])
265         * @see #setDictionary(byte[], int, int)
266         * @since Android 1.0
267         */
268        public synchronized boolean needsDictionary() {
269            return needsDictionary;
270        }
271    
272        /**
273         * Indicates that input has to be passed to the inflater.
274         * 
275         * @return {@code true} if {@code setInput} has to be called before
276         *         inflation can proceed.
277         * @see #setInput(byte[])
278         * @since Android 1.0
279         */
280        public synchronized boolean needsInput() {
281            return inRead == inLength;
282        }
283    
284        /**
285         * Resets the {@code Inflater}. Should be called prior to inflating a new
286         * set of data.
287         * 
288         * @since Android 1.0
289         */
290        public synchronized void reset() {
291            if (streamHandle == -1) {
292                throw new NullPointerException();
293            }
294            finished = false;
295            needsDictionary = false;
296            inLength = inRead = 0;
297            resetImpl(streamHandle);
298        }
299    
300        private native synchronized void resetImpl(long handle);
301    
302        /**
303         * Sets the preset dictionary to be used for inflation to {@code buf}.
304         * {@code needsDictionary()} can be called to determine whether the current
305         * input was deflated using a preset dictionary.
306         * 
307         * @param buf
308         *            The buffer containing the dictionary bytes.
309         * @see #needsDictionary
310         * @since Android 1.0
311         */
312        public synchronized void setDictionary(byte[] buf) {
313            setDictionary(buf, 0, buf.length);
314        }
315    
316        /**
317         * Like {@code setDictionary(byte[])}, allowing to define a specific region
318         * inside {@code buf} to be used as a dictionary.
319         * 
320         * @param buf
321         *            the buffer containing the dictionary data bytes.
322         * @param off
323         *            the offset of the data.
324         * @param nbytes
325         *            the length of the data.
326         * @see #needsDictionary
327         * @since Android 1.0
328         */
329        public synchronized void setDictionary(byte[] buf, int off, int nbytes) {
330            if (streamHandle == -1) {
331                throw new IllegalStateException();
332            }
333            // avoid int overflow, check null buf
334            if (off <= buf.length && nbytes >= 0 && off >= 0
335                    && buf.length - off >= nbytes) {
336                setDictionaryImpl(buf, off, nbytes, streamHandle);
337            } else {
338                throw new ArrayIndexOutOfBoundsException();
339            }
340        }
341    
342        private native synchronized void setDictionaryImpl(byte[] buf, int off,
343                int nbytes, long handle);
344    
345        /**
346         * Sets the current input to to be decrompressed. This method should only be
347         * called if {@code needsInput()} returns {@code true}.
348         * 
349         * @param buf
350         *            the input buffer.
351         * @see #needsInput
352         * @since Android 1.0
353         */
354        public synchronized void setInput(byte[] buf) {
355            setInput(buf, 0, buf.length);
356        }
357    
358        /**
359         * Sets the current input to the region of the input buffer starting at
360         * {@code off} and ending at {@code nbytes - 1} where data is written after
361         * decompression. This method should only be called if {@code needsInput()}
362         * returns {@code true}.
363         * 
364         * @param buf
365         *            the input buffer.
366         * @param off
367         *            the offset to read from the input buffer.
368         * @param nbytes
369         *            the number of bytes to read.
370         * @see #needsInput
371         * @since Android 1.0
372         */
373        public synchronized void setInput(byte[] buf, int off, int nbytes) {
374            if (streamHandle == -1) {
375                throw new IllegalStateException();
376            }
377            // avoid int overflow, check null buf
378            if (off <= buf.length && nbytes >= 0 && off >= 0
379                    && buf.length - off >= nbytes) {
380                inRead = 0;
381                inLength = nbytes;
382                setInputImpl(buf, off, nbytes, streamHandle);
383            } else {
384                throw new ArrayIndexOutOfBoundsException();
385            }
386            // BEGIN android-note
387            // Note:  pass_magic_number_check is set to false when setInput is
388            //        called for the first time and for a single byte.
389            //        Since setInput is called only by InflaterInputStream.fill
390            //        with an arbitrary byte len this check seems quite useless.
391            // FIXME: We should find out whether the first byte has to be the magic
392            //        number in all cases and correct the check as well as place it
393            //        in setFileInput accordingly.
394            //        And at a first glance it doesn't look like the first byte has
395            //        to be 120.
396            // END android-note
397            if(!gotFirstByte && nbytes>0)
398            {
399               pass_magic_number_check = (buf[off] == MAGIC_NUMBER || nbytes > 1);           
400               gotFirstByte = true;
401            }
402        }
403    
404        // BEGIN android-added
405        /**
406         * Sets the current input to the region within a file starting at {@code
407         * off} and ending at {@code nbytes - 1}. This method should only be called
408         * if {@code needsInput()} returns {@code true}.
409         * 
410         * @param file
411         *            the input file.
412         * @param off
413         *            the offset to read from in buffer.
414         * @param nbytes
415         *            the number of bytes to read.
416         * @see #needsInput
417         * @since Android 1.0
418         */
419        synchronized int setFileInput(FileDescriptor fd, long off, int nbytes) {
420            if (streamHandle == -1) {
421                throw new IllegalStateException();
422            }
423            inRead = 0;
424            inLength = setFileInputImpl(fd, off, nbytes, streamHandle);
425            return inLength;
426        }

429        /**
430         * Returns the total number of bytes read by the {@code Inflater}. This
431         * method performs the same as {@code getTotalIn()} except that it returns a
432         * {@code long} value instead of an integer.
433         * 
434         * @return the total number of bytes read.
435         * @since Android 1.0
436         */
437        public synchronized long getBytesRead() {
438            // Throw NPE here
439            if (streamHandle == -1) {
440                throw new NullPointerException();
441            }
442            return getTotalInImpl(streamHandle);
443        }

445        /**
446         * Returns a the total number of bytes read by the {@code Inflater}. This
447         * method performs the same as {@code getTotalOut} except it returns a
448         * {@code long} value instead of an integer.
449         * 
450         * @return the total bytes written to the output buffer.
451         * @since Android 1.0
452         */
453        public synchronized long getBytesWritten() {
454            // Throw NPE here
455            if (streamHandle == -1) {
456                throw new NullPointerException();
457            }
458            return getTotalOutImpl(streamHandle);
459        }
