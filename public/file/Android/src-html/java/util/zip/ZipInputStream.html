031    /**
032     * This class provides an implementation of {@code FilterInputStream} that
033     * uncompresses data from a <i>ZIP-archive</i> input stream.
034     * <p>
035     * A <i>ZIP-archive</i> is a collection of compressed (or uncompressed) files -
036     * the so called ZIP entries. Therefore when reading from a {@code
037     * ZipInputStream} first the entry's attributes will be retrieved with {@code
038     * getNextEntry} before its data is read.
039     * </p>
040     * <p>
041     * While {@code InflaterInputStream} can read a compressed <i>ZIP-archive</i>
042     * entry, this extension can read uncompressed entries as well.
043     * </p>
044     * <p>
045     * Use {@code ZipFile} if you can access the archive as a file directly.
046     * </p>
047     * 
048     * @see ZipEntry
049     * @see ZipFile
050     * @since Android 1.0
051     */
052    public class ZipInputStream extends InflaterInputStream implements ZipConstants {
053        static final int DEFLATED = 8;
054    
055        static final int STORED = 0;
056    
057        static final int ZIPDataDescriptorFlag = 8;
058    
059        static final int ZIPLocalHeaderVersionNeeded = 20;
060    
061        // BEGI android-removed
062        // private boolean zipClosed = false;
063        // END android-removed
064    
065        private boolean entriesEnd = false;
066    
067        private boolean hasDD = false;
068    
069        private int entryIn = 0;
070    
071        private int inRead, lastRead = 0;
072    
073        ZipEntry currentEntry;
074    
075        private final byte[] hdrBuf = new byte[LOCHDR - LOCVER];
076    
077        private final CRC32 crc = new CRC32();
078    
079        private byte[] nameBuf = new byte[256];
080    
081        private char[] charBuf = new char[256];
082    
083        /**
084         * Constructs a new {@code ZipInputStream} from the specified input stream.
085         * 
086         * @param stream
087         *            the input stream to representing a ZIP archive.
088         * @since Android 1.0
089         */
090        public ZipInputStream(InputStream stream) {
091            super(new PushbackInputStream(stream, BUF_SIZE), new Inflater(true));
092            if (stream == null) {
093                throw new NullPointerException();
094            }
095        }

097        /**
098         * Closes this {@code ZipInputStream}.
099         * 
100         * @throws IOException
101         *             if an {@code IOException} occurs.
102         * @since Android 1.0
103         */
104        @Override
105        public void close() throws IOException {
106            // BEGIN android-changed
107            if (closed != true) {
108                closeEntry(); // Close the current entry
109                super.close();
110            }
111            // END android-changed
112        }
113    
114        /**
115         * Closes the current ZIP entry and positions to read the next entry.
116         * 
117         * @throws IOException
118         *             if an {@code IOException} occurs.
119         * @since Android 1.0
120         */
121        public void closeEntry() throws IOException {
122            // BEGIN android-changed
123            if (closed) {
124                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
125            }
126            // END android-changed
127            if (currentEntry == null) {
128                return;
129            }
130            if (currentEntry instanceof java.util.jar.JarEntry) {
131                Attributes temp = ((JarEntry) currentEntry).getAttributes();
132                if (temp != null && temp.containsKey("hidden")) { //$NON-NLS-1$
133                    return;
134                }
135            }
136            // Ensure all entry bytes are read
137            skip(Long.MAX_VALUE);
138            int inB, out;
139            if (currentEntry.compressionMethod == DEFLATED) {
140                inB = inf.getTotalIn();
141                out = inf.getTotalOut();
142            } else {
143                inB = inRead;
144                out = inRead;
145            }
146            int diff = 0;
147            // Pushback any required bytes
148            if ((diff = entryIn - inB) != 0) {
149                ((PushbackInputStream) in).unread(buf, len - diff, diff);
150            }
151    
152            if (hasDD) {
153                in.read(hdrBuf, 0, EXTHDR);
154                if (getLong(hdrBuf, 0) != EXTSIG) {
155                    throw new ZipException(Messages.getString("archive.1F")); //$NON-NLS-1$
156                }
157                currentEntry.crc = getLong(hdrBuf, EXTCRC);
158                currentEntry.compressedSize = getLong(hdrBuf, EXTSIZ);
159                currentEntry.size = getLong(hdrBuf, EXTLEN);
160            }
161            if (currentEntry.crc != crc.getValue()) {
162                throw new ZipException(Messages.getString("archive.20")); //$NON-NLS-1$
163            }
164            if (currentEntry.compressedSize != inB || currentEntry.size != out) {
165                throw new ZipException(Messages.getString("archive.21")); //$NON-NLS-1$
166            }
167    
168            inf.reset();
169            lastRead = inRead = entryIn = len = 0;
170            crc.reset();
171            currentEntry = null;
172        }
173    
174        /**
175         * Reads the next entry from this {@code ZipInputStream}.
176         * 
177         * @return the next {@code ZipEntry} contained in the input stream.
178         * @throws IOException
179         *             if the stream is not positioned at the beginning of an entry
180         *             or if an other {@code IOException} occurs.
181         * @see ZipEntry
182         * @since Android 1.0
183         */
184        public ZipEntry getNextEntry() throws IOException {
185            if (currentEntry != null) {
186                closeEntry();
187            }
188            if (entriesEnd) {
189                return null;
190            }
191    
192            int x = 0, count = 0;
193            while (count != 4) {
194                count += x = in.read(hdrBuf, count, 4 - count);
195                if (x == -1) {
196                    return null;
197                }
198            }
199            long hdr = getLong(hdrBuf, 0);
200            if (hdr == CENSIG) {
201                entriesEnd = true;
202                return null;
203            }
204            if (hdr != LOCSIG) {
205                return null;
206            }
207    
208            // Read the local header
209            count = 0;
210            while (count != (LOCHDR - LOCVER)) {
211                count += x = in.read(hdrBuf, count, (LOCHDR - LOCVER) - count);
212                if (x == -1) {
213                    throw new EOFException();
214                }
215            }
216            int version = getShort(hdrBuf, 0) & 0xff;
217            if (version > ZIPLocalHeaderVersionNeeded) {
218                throw new ZipException(Messages.getString("archive.22")); //$NON-NLS-1$
219            }
220            int flags = getShort(hdrBuf, LOCFLG - LOCVER);
221            hasDD = ((flags & ZIPDataDescriptorFlag) == ZIPDataDescriptorFlag);
222            int cetime = getShort(hdrBuf, LOCTIM - LOCVER);
223            int cemodDate = getShort(hdrBuf, LOCTIM - LOCVER + 2);
224            int cecompressionMethod = getShort(hdrBuf, LOCHOW - LOCVER);
225            long cecrc = 0, cecompressedSize = 0, cesize = -1;
226            if (!hasDD) {
227                cecrc = getLong(hdrBuf, LOCCRC - LOCVER);
228                cecompressedSize = getLong(hdrBuf, LOCSIZ - LOCVER);
229                cesize = getLong(hdrBuf, LOCLEN - LOCVER);
230            }
231            int flen = getShort(hdrBuf, LOCNAM - LOCVER);
232            if (flen == 0) {
233                throw new ZipException(Messages.getString("archive.23")); //$NON-NLS-1$
234            }
235            int elen = getShort(hdrBuf, LOCEXT - LOCVER);
236    
237            count = 0;
238            if (flen > nameBuf.length) {
239                nameBuf = new byte[flen];
240                charBuf = new char[flen];
241            }
242            while (count != flen) {
243                count += x = in.read(nameBuf, count, flen - count);
244                if (x == -1) {
245                    throw new EOFException();
246                }
247            }
248            currentEntry = createZipEntry(Util.convertUTF8WithBuf(nameBuf, charBuf,
249                    0, flen));
250            currentEntry.time = cetime;
251            currentEntry.modDate = cemodDate;
252            currentEntry.setMethod(cecompressionMethod);
253            if (cesize != -1) {
254                currentEntry.setCrc(cecrc);
255                currentEntry.setSize(cesize);
256                currentEntry.setCompressedSize(cecompressedSize);
257            }
258            if (elen > 0) {
259                count = 0;
260                byte[] e = new byte[elen];
261                while (count != elen) {
262                    count += x = in.read(e, count, elen - count);
263                    if (x == -1) {
264                        throw new EOFException();
265                    }
266                }
267                currentEntry.setExtra(e);
268            }
269            // BEGIN android-added
270            eof = false;
271            // END android-added
272            return currentEntry;
273        }

275        /* Read 4 bytes from the buffer and store it as an int */
276    
277        @Override
278        public int read(byte[] buffer, int start, int length) throws IOException {
279            // BEGIN android-changed
280            if (closed) {
281                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
282            }
283            // END android-changed
284            if (inf.finished() || currentEntry == null) {
285                return -1;
286            }
287            // avoid int overflow, check null buffer
288            if (start <= buffer.length && length >= 0 && start >= 0
289                    && buffer.length - start >= length) {
290                if (currentEntry.compressionMethod == STORED) {
291                    int csize = (int) currentEntry.size;
292                    if (inRead >= csize) {
293                        // BEGIN android-added
294                        eof = true;
295                        // END android-added
296                        return -1;
297                    }
298                    if (lastRead >= len) {
299                        lastRead = 0;
300                        if ((len = in.read(buf)) == -1) {
301                            // BEGIN android-added
302                            eof = true;
303                            // END android-added
304                            return -1;
305                        }
306                        entryIn += len;
307                    }
308                    // BEGIN android-changed
309                    int toRead = length > (len - lastRead) ? len - lastRead : length;
310                    // END android-changed
311                    if ((csize - inRead) < toRead) {
312                        toRead = csize - inRead;
313                    }
314                    System.arraycopy(buf, lastRead, buffer, start, toRead);
315                    lastRead += toRead;
316                    inRead += toRead;
317                    crc.update(buffer, start, toRead);
318                    return toRead;
319                }
320                if (inf.needsInput()) {
321                    fill();
322                    if (len > 0) {
323                        entryIn += len;
324                    }
325                }
326                int read = 0;
327                try {
328                    read = inf.inflate(buffer, start, length);
329                } catch (DataFormatException e) {
330                    throw new ZipException(e.getMessage());
331                }
332                if (read == 0 && inf.finished()) {
333                    return -1;
334                }
335                crc.update(buffer, start, read);
336                return read;
337            }
338            throw new ArrayIndexOutOfBoundsException();
339        }

341        /**
342         * Skips up to the specified number of bytes in the current ZIP entry.
343         * 
344         * @param value
345         *            the number of bytes to skip.
346         * @return the number of bytes skipped.
347         * @throws IOException
348         *             if an {@code IOException} occurs.
349         * @since Android 1.0
350         */
351        @Override
352        public long skip(long value) throws IOException {
353            if (value >= 0) {
354                long skipped = 0;
355                byte[] b = new byte[1024];
356                while (skipped != value) {
357                    long rem = value - skipped;
358                    int x = read(b, 0, (int) (b.length > rem ? rem : b.length));
359                    if (x == -1) {
360                        return skipped;
361                    }
362                    skipped += x;
363                }
364                return skipped;
365            }
366            throw new IllegalArgumentException();
367    }

369        /**
370         * Returns 0 if the {@code EOF} has been reached, otherwise returns 1.
371         * 
372         * @return 0 after {@code EOF} of current entry, 1 otherwise.
373         * @throws IOException
374         *             if an IOException occurs.
375         * @since Android 1.0
376         */
377        @Override
378        public int available() throws IOException {
379            // BEGIN android-changed
380            if (closed) {
381                throw new IOException(Messages.getString("archive.1E")); //$NON-NLS-1$
382            }
383            if (currentEntry == null) {
384                return 1;
385            }
386            return super.available();
387            // END android-changed
388        }

390        /**
391         * creates a {@link ZipEntry } with the given name.
392         * 
393         * @param name
394         *            the name of the entry.
395         * @return the created {@code ZipEntry}.
396         * @since Android 1.0
397         */
398        protected ZipEntry createZipEntry(String name) {
399            return new ZipEntry(name);
400        }
401    
402        private int getShort(byte[] buffer, int off) {
403            return (buffer[off] & 0xFF) | ((buffer[off + 1] & 0xFF) << 8);
404        }
405    
406        private long getLong(byte[] buffer, int off) {
407            long l = 0;
408            l |= (buffer[off] & 0xFF);
409            l |= (buffer[off + 1] & 0xFF) << 8;
410            l |= (buffer[off + 2] & 0xFF) << 16;
411            l |= ((long) (buffer[off + 3] & 0xFF)) << 24;
412            return l;
413        }
