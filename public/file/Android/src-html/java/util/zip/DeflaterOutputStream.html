018    package java.util.zip;
019    
020    
021    import java.io.FilterOutputStream;
022    import java.io.IOException;
023    import java.io.OutputStream;
024    
025    import org.apache.harmony.archive.internal.nls.Messages;
026    
027    /**
028     * This class provides an implementation of {@code FilterOutputStream} that
029     * compresses data using the <i>DEFLATE</i> algorithm. Basically it wraps the
030     * {@code Deflater} class and takes care of the buffering.
031     * 
032     * @see Deflater
033     * @since Android 1.0
034     */
035    public class DeflaterOutputStream extends FilterOutputStream {
036        static final int BUF_SIZE = 512;
037    
038        /**
039         * The buffer for the data to be written to.
040         * 
041         * @since Android 1.0
042         */
043        protected byte[] buf;
044    
045        /**
046         * The deflater used.
047         * 
048         * @since Android 1.0
049         */
050        protected Deflater def;
051    
052        boolean done = false;
053    
054        /**
055         * This constructor lets you pass the {@code Deflater} specifying the
056         * compression algorithm.
057         * 
058         * @param os
059         *            is the {@code OutputStream} where to write the compressed data
060         *            to.
061         * @param def
062         *            is the specific {@code Deflater} that is used to compress
063         *            data.
064         * @since Android 1.0
065         */
066        public DeflaterOutputStream(OutputStream os, Deflater def) {
067            this(os, def, BUF_SIZE);
068        }
069    
070        /**
071         * This is the most basic constructor. You only need to pass the {@code
072         * OutputStream} to which the compressed data shall be written to. The
073         * default settings for the {@code Deflater} and internal buffer are used.
074         * In particular the {@code Deflater} produces a ZLIB header in the output
075         * stream.
076         * 
077         * @param os
078         *            is the OutputStream where to write the compressed data to.
079         * @since Android 1.0
080         */
081        public DeflaterOutputStream(OutputStream os) {
082            this(os, new Deflater());
083        }
084    
085        /**
086         * This constructor lets you specify both the compression algorithm as well
087         * as the internal buffer size to be used.
088         * 
089         * @param os
090         *            is the {@code OutputStream} where to write the compressed data
091         *            to.
092         * @param def
093         *            is the specific {@code Deflater} that will be used to compress
094         *            data.
095         * @param bsize
096         *            is the size to be used for the internal buffer.
097         * @since Android 1.0
098         */
099        public DeflaterOutputStream(OutputStream os, Deflater def, int bsize) {
100            super(os);
101            if (os == null || def == null) {
102                throw new NullPointerException();
103            }
104            if (bsize <= 0) {
105                throw new IllegalArgumentException();
106            }
107            this.def = def;
108            buf = new byte[bsize];
109        }
110    
111        /**
112         * Compress the data in the input buffer and write it to the underlying
113         * stream.
114         * 
115         * @throws IOException
116         *             If an error occurs during deflation.
117         * @since Android 1.0
118         */
119        protected void deflate() throws IOException {
120            int x = 0;
121            do {
122                x = def.deflate(buf);
123                out.write(buf, 0, x);
124            } while (!def.needsInput());
125        }
126    
127        /**
128         * Writes any unwritten compressed data to the underlying stream, the closes
129         * all underlying streams. This stream can no longer be used after close()
130         * has been called.
131         * 
132         * @throws IOException
133         *             If an error occurs while closing the data compression
134         *             process.
135         * @since Android 1.0
136         */
137        @Override
138        public void close() throws IOException {
139            if (!def.finished()) {
140                finish();
141            }
142            def.end();
143            out.close();
144        }
145    
146        /**
147         * Writes any unwritten data to the underlying stream. Does not close the
148         * stream.
149         * 
150         * @throws IOException
151         *             If an error occurs.
152         * @since Android 1.0
153         */
154        public void finish() throws IOException {
155            if (done) {
156                return;
157            }
158            def.finish();
159            int x = 0;
160            while (!def.finished()) {
161                if (def.needsInput()) {
162                    def.setInput(buf, 0, 0);
163                }
164                x = def.deflate(buf);
165                out.write(buf, 0, x);
166            }
167            done = true;
168        }
169    
170        @Override
171        public void write(int i) throws IOException {
172            byte[] b = new byte[1];
173            b[0] = (byte) i;
174            write(b, 0, 1);
175        }
176    
177        /**
178         * Compresses {@code nbytes} of data from {@code buf} starting at
179         * {@code off} and writes it to the underlying stream.
180         * 
181         * @param buffer
182         *            the buffer of data to compress.
183         * @param off
184         *            offset in buffer to extract data from.
185         * @param nbytes
186         *            the number of bytes of data to read from the buffer.
187         * @throws IOException
188         *             If an error occurs during writing.
189         * @since Android 1.0
190         */
191        @Override
192        public void write(byte[] buffer, int off, int nbytes) throws IOException {
193            if (done) {
194                throw new IOException(Messages.getString("archive.26")); //$NON-NLS-1$
195            }
196            // avoid int overflow, check null buf
197            if (off <= buffer.length && nbytes >= 0 && off >= 0
198                    && buffer.length - off >= nbytes) {
199                if (!def.needsInput()) {
200                    throw new IOException();
201                }
202                def.setInput(buffer, off, nbytes);
203                deflate();
204            } else {
205                throw new ArrayIndexOutOfBoundsException();
206            }
207        }
208    }
