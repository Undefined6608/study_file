031    /**
032     * {@code ResourceBundle} is an abstract class which is the superclass of classes which
033     * provide {@code Locale}-specific resources. A bundle contains a number of named
034     * resources, where the names are {@code Strings}. A bundle may have a parent bundle,
035     * and when a resource is not found in a bundle, the parent bundle is searched for
036     * the resource. If the fallback mechanism reaches the base bundle and still
037     * can't find the resource it throws a {@code MissingResourceException}.
038     * 
039     * <ul>
040     * <li>All bundles for the same group of resources share a common base bundle.
041     * This base bundle acts as the root and is the last fallback in case none of
042     * its children was able to respond to a request.</li>
043     * <li>The first level contains changes between different languages. Only the
044     * differences between a language and the language of the base bundle need to be
045     * handled by a language-specific {@code ResourceBundle}.</li>
046     * <li>The second level contains changes between different countries that use
047     * the same language. Only the differences between a country and the country of
048     * the language bundle need to be handled by a country-specific {@code ResourceBundle}.
049     * </li>
050     * <li>The third level contains changes that don't have a geographic reason
051     * (e.g. changes that where made at some point in time like {@code PREEURO} where the
052     * currency of come countries changed. The country bundle would return the
053     * current currency (Euro) and the {@code PREEURO} variant bundle would return the old
054     * currency (e.g. DM for Germany).</li>
055     * </ul>
056     * 
057     * <strong>Examples</strong>
058     * <ul>
059     * <li>BaseName (base bundle)
060     * <li>BaseName_de (german language bundle)
061     * <li>BaseName_fr (french language bundle)
062     * <li>BaseName_de_DE (bundle with Germany specific resources in german)
063     * <li>BaseName_de_CH (bundle with Switzerland specific resources in german)
064     * <li>BaseName_fr_CH (bundle with Switzerland specific resources in french)
065     * <li>BaseName_de_DE_PREEURO (bundle with Germany specific resources in german of
066     * the time before the Euro)
067     * <li>BaseName_fr_FR_PREEURO (bundle with France specific resources in french of
068     * the time before the Euro)
069     * </ul>
070     * 
071     * It's also possible to create variants for languages or countries. This can be
072     * done by just skipping the country or language abbreviation:
073     * BaseName_us__POSIX or BaseName__DE_PREEURO. But it's not allowed to
074     * circumvent both language and country: BaseName___VARIANT is illegal.
075     * 
076     * @see Properties
077     * @see PropertyResourceBundle
078     * @see ListResourceBundle
079     * @since Android 1.0
080     */
081    public abstract class ResourceBundle {

164        /**
165         * Finds the named resource bundle for the specified {@code Locale} and {@code ClassLoader}.
166         * 
167         * The passed base name and {@code Locale} are used to create resource bundle names.
168         * The first name is created by concatenating the base name with the result
169         * of {@link Locale#toString()}. From this name all parent bundle names are
170         * derived. Then the same thing is done for the default {@code Locale}. This results
171         * in a list of possible bundle names.
172         * 
173         * <strong>Example</strong> For the basename "BaseName", the {@code Locale} of the
174         * German part of Switzerland (de_CH) and the default {@code Locale} en_US the list
175         * would look something like this:
176         * 
177         * <ol>
178         * <li>BaseName_de_CH</li>
179         * <li>BaseName_de</li>
180         * <li>Basename_en_US</li>
181         * <li>Basename_en</li>
182         * <li>BaseName</li>
183         * </ol>
184         * 
185         * This list also shows the order in which the bundles will be searched for a requested
186         * resource in the German part of Switzerland (de_CH).
187         * 
188         * As a first step, this method tries to instantiate 
189         * a {@code ResourceBundle} with the names provided.
190         * If such a class can be instantiated and initialized, it is returned and
191         * all the parent bundles are instantiated too. If no such class can be
192         * found this method tries to load a {@code .properties} file with the names by
193         * replacing dots in the base name with a slash and by appending
194         * "{@code .properties}" at the end of the string. If such a resource can be found
195         * by calling {@link ClassLoader#getResource(String)} it is used to
196         * initialize a {@link PropertyResourceBundle}. If this succeeds, it will
197         * also load the parents of this {@code ResourceBundle}.
198         * 
199         * For compatibility with older code, the bundle name isn't required to be
200         * a fully qualified class name. It's also possible to directly pass
201         * the path to a properties file (without a file extension).
202         * 
203         * @param bundleName
204         *            the name of the {@code ResourceBundle}.
205         * @param locale
206         *            the {@code Locale}.
207         * @param loader
208         *            the {@code ClassLoader} to use.
209         * @return the requested {@code ResourceBundle}.
210         * @exception MissingResourceException
211         *                if the {@code ResourceBundle} cannot be found.
212         * @since Android 1.0
213         */
214        public static ResourceBundle getBundle(String bundleName, Locale locale,
215                ClassLoader loader) throws MissingResourceException {
216            if (loader == null) {
217                throw new NullPointerException();
218            }
219            // BEGIN android-changed
220            return getBundleImpl(bundleName, locale, loader);
221            // END android-changed
222        }

224        private static ResourceBundle getBundleImpl(String bundleName,
225                Locale locale, ClassLoader loader) throws MissingResourceException {
226            if (bundleName != null) {
227                ResourceBundle bundle;
228                // BEGIN android-added
229                if (!defaultLocale.equals(Locale.getDefault())) {
230                    cache.clear();
231                    defaultLocale = Locale.getDefault();
232                }
233                // END android-added
234                if (!locale.equals(Locale.getDefault())) {
235                    String localeName = locale.toString();
236                    if (localeName.length() > 0) {
237                        localeName = "_" + localeName; //$NON-NLS-1$
238                    }
239                    if ((bundle = handleGetBundle(bundleName, localeName, false,
240                            loader)) != null) {
241                        return bundle;
242                    }
243                }
244                String localeName = Locale.getDefault().toString();
245                if (localeName.length() > 0) {
246                    localeName = "_" + localeName; //$NON-NLS-1$
247                }
248                if ((bundle = handleGetBundle(bundleName, localeName, true, loader)) != null) {
249                    return bundle;
250                }
251                throw new MissingResourceException(null, bundleName + '_' + locale,
252                        ""); //$NON-NLS-1$
253            }
254            throw new NullPointerException();
255        }

277        /**
278         * Returns the named resource from this {@code ResourceBundle}. If the resource
279         * cannot be found in this bundle, it falls back to the parent bundle (if
280         * it's not null) by calling the {@link #handleGetObject} method. If the resource still
281         * can't be found it throws a {@code MissingResourceException}.
282         * 
283         * @param key
284         *            the name of the resource.
285         * @return the resource object.
286         * @exception MissingResourceException
287         *                if the resource is not found.
288         * @since Android 1.0
289         */
290        public final Object getObject(String key) {
291            ResourceBundle last, theParent = this;
292            do {
293                Object result = theParent.handleGetObject(key);
294                if (result != null) {
295                    return result;
296                }
297                last = theParent;
298                theParent = theParent.parent;
299            } while (theParent != null);
300            throw new MissingResourceException(null, last.getClass().getName(), key);
301        }

337        private static ResourceBundle handleGetBundle(String base, String locale,
338                boolean loadBase, final ClassLoader loader) {
339            ResourceBundle bundle = null;
340            String bundleName = base + locale;
341            Object cacheKey = loader != null ? (Object) loader : (Object) "null"; //$NON-NLS-1$
342            Hashtable<String, ResourceBundle> loaderCache;
343            synchronized (cache) {
344                loaderCache = cache.get(cacheKey);
345                if (loaderCache == null) {
346                    loaderCache = new Hashtable<String, ResourceBundle>(13);
347                    cache.put(cacheKey, loaderCache);
348                }
349            }
350            ResourceBundle result = loaderCache.get(bundleName);
351            if (result != null) {
352                if (result == MISSINGBASE) {
353                    return null;
354                }
355                if (result == MISSING) {
356                    if (!loadBase) {
357                        return null;
358                    }
359                    String extension = strip(locale);
360                    if (extension == null) {
361                        return null;
362                    }
363                    return handleGetBundle(base, extension, loadBase, loader);
364                }
365                return result;
366            }
367    
368            try {
369                // BEGIN android-changed
370                /*
371                 * Intercept loading of ResourceBundles that contain Harmony
372                 * I18N data. Deliver our special, ICU-based bundles in this case.
373                 * All other ResourceBundles use the ordinary mechanism, so user
374                 * code behaves as it should.
375                 */
376                if(bundleName.startsWith("org.apache.harmony.luni.internal.locale.")) {
377                    String icuBundleName = bundleName.substring(40);
378                    String icuLocale = (locale.length() > 0 ? locale.substring(1) : locale);
379                    // we know that Resources will deliver an assignable class
380                    bundle = Resources.getInstance(icuBundleName, icuLocale);
381                } else {
382                    Class<?> bundleClass = Class.forName(bundleName, true, loader);
383                    if (ResourceBundle.class.isAssignableFrom(bundleClass)) {
384                        bundle = (ResourceBundle) bundleClass.newInstance();
385                    }
386                }
387                // END android-changed
388            } catch (LinkageError e) {
389            } catch (Exception e) {
390            }
391    
392            // BEGIN android-added
393            // copied from newer version of Harmony
394            if (bundle != null) {
395                bundle.setLocale(locale);
396            }
397            // END android-added
398            if (bundle == null) {
399                final String fileName = bundleName.replace('.', '/');
400                InputStream stream = AccessController
401                        .doPrivileged(new PrivilegedAction<InputStream>() {
402                            public InputStream run() {
403                                return loader == null ? ClassLoader
404                                        .getSystemResourceAsStream(fileName
405                                                + ".properties") : loader //$NON-NLS-1$
406                                        .getResourceAsStream(fileName
407                                                + ".properties"); //$NON-NLS-1$
408                            }
409                        });
410                if (stream != null) {
411                    try {
412                        try {
413                            bundle = new PropertyResourceBundle(stream);
414                        } finally {
415                            stream.close();
416                        }
417                        bundle.setLocale(locale);
418                    } catch (IOException e) {
419                    }
420                }
421            }
422    
423            String extension = strip(locale);
424            if (bundle != null) {
425                if (extension != null) {
426                    ResourceBundle parent = handleGetBundle(base, extension, true,
427                            loader);
428                    if (parent != null) {
429                        bundle.setParent(parent);
430                    }
431                }
432                loaderCache.put(bundleName, bundle);
433                return bundle;
434            }
435    
436            if (extension != null && (loadBase || extension.length() > 0)) {
437                bundle = handleGetBundle(base, extension, loadBase, loader);
438                if (bundle != null) {
439                    loaderCache.put(bundleName, bundle);
440                    return bundle;
441                }
442            }
443            loaderCache.put(bundleName, loadBase ? MISSINGBASE : MISSING);
444            return null;
445        }

470        private static String strip(String name) {
471            int index = name.lastIndexOf('_');
472            if (index != -1) {
473                return name.substring(0, index);
474            }
475            return null;
476        }

478        private void setLocale(String name) {
479            String language = "", country = "", variant = ""; //$NON-NLS-1$//$NON-NLS-2$ //$NON-NLS-3$
480            if (name.length() > 1) {
481                int nextIndex = name.indexOf('_', 1);
482                if (nextIndex == -1) {
483                    nextIndex = name.length();
484                }
485                language = name.substring(1, nextIndex);
486                if (nextIndex + 1 < name.length()) {
487                    int index = nextIndex;
488                    nextIndex = name.indexOf('_', nextIndex + 1);
489                    if (nextIndex == -1) {
490                        nextIndex = name.length();
491                    }
492                    country = name.substring(index + 1, nextIndex);
493                    if (nextIndex + 1 < name.length()) {
494                        variant = name.substring(nextIndex + 1, name.length());
495                    }
496                }
497            }
498            locale = new Locale(language, country, variant);
499        }
