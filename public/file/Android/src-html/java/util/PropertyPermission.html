018    package java.util;
019    
020    import java.io.IOException;
021    import java.io.ObjectInputStream;
022    import java.io.ObjectOutputStream;
023    import java.io.ObjectStreamField;
024    import java.security.BasicPermission;
025    import java.security.Permission;
026    import java.security.PermissionCollection;
027    
028    import org.apache.harmony.luni.util.Util;
029    
030    /**
031     * {@code PropertyPermission} objects represent a permission to access system
032     * properties.
033     * <p>
034     * A permission is one of the possible permission strings like "user.name" or
035     * "java.version". It's also possible to use a wildcard to define the permission
036     * to several properties at once. For example "user.*" will define the
037     * permission for "user.home", "user.name", "user.dir", ... "*" defines the
038     * permission for all available properties.
039     * <p>
040     * There are two possible permission action types: read and write. Possible
041     * actions are "read", "write", or "read,write"/"write,read".
042     *  
043     * @since Android 1.0
044     */
045    public final class PropertyPermission extends BasicPermission {
046        private static final long serialVersionUID = 885438825399942851L;
047    
048        transient private boolean read, write;
049    
050        /**
051         * Constructs a new instance of this class.
052         * 
053         * @param name
054         *            the (possibly wildcarded) name of the property.
055         * @param actions
056         *            the actions which are applicable to it. Possible actions are
057         *            "read", "write", or "read,write"/"write,read". Anything else
058         *            will result in an {@code IllegalArgumentException}.
059         * @since Android 1.0
060         */
061        public PropertyPermission(String name, String actions) {
062            super(name);
063            decodeActions(actions);
064        }
065    
066        private void decodeActions(String actions) {
067            StringTokenizer tokenizer = new StringTokenizer(Util.toASCIILowerCase(actions),
068                    " \t\n\r,"); //$NON-NLS-1$
069            while (tokenizer.hasMoreTokens()) {
070                String token = tokenizer.nextToken();
071                if (token.equals("read")) { //$NON-NLS-1$
072                    read = true;
073                } else if (token.equals("write")) { //$NON-NLS-1$
074                    write = true;
075                } else {
076                    throw new IllegalArgumentException();
077                }
078            }
079            if (!read && !write) {
080                throw new IllegalArgumentException();
081            }
082        }
083    
084        /**
085         * Compares the argument to the receiver, and returns true if they represent
086         * the <em>same</em> object using a class specific comparison. In this
087         * case, the receiver must be a {@code PropertyPermission} for the same 
088         * property as the argument, and must have the same actions.
089         * If {@code o} is a permission that is not a {@code PropertyPermission},
090         * this method may throw a {@code ClassCastException}.     
091         * 
092         * @param o
093         *            the {@code Object} to compare with this {@code Object}.
094         * @return {@code true} if the {@code Object} is the same as this {@code Object},
095         *         {@code false} if it is different from this {@code Object}.
096         * @see #hashCode
097         * @since Android 1.0
098         */
099        @Override
100        public boolean equals(Object o) {
101            if (super.equals(o)) {
102                PropertyPermission pp = (PropertyPermission) o;
103                return read == pp.read && write == pp.write;
104            }
105            return false;
106        }
107    
108        /**
109         * Returns the actions associated with the receiver. The result will be
110         * either "read", "write", or "read,write".
111         * 
112         * @return the actions associated with the receiver.
113         * @since Android 1.0
114         */
115        @Override
116        public String getActions() {
117            return read ? (write ? "read,write" : "read") : "write"; //$NON-NLS-1$ //$NON-NLS-2$//$NON-NLS-3$
118        }
119    
120        /**
121         * Returns an integer hash code for the receiver. Any two objects which
122         * return {@code true} when passed to {@code equals} must return the same
123         * value for this method.
124         * 
125         * @return the receiver's hash.
126         * @see #equals
127         * @since Android 1.0
128         */
129        @Override
130        public int hashCode() {
131            return super.hashCode();
132        }
133    
134        /**
135         * Indicates whether the argument permission is implied by the receiver.
136         * 
137         * @return boolean {@code true} if the argument permission is implied by the
138         *         receiver, and {@code false} if it is not.
139         * @param permission
140         *            the permission to check.
141         * @since Android 1.0
142         */
143        @Override
144        public boolean implies(Permission permission) {
145            if (super.implies(permission)) {
146                PropertyPermission pp = (PropertyPermission) permission;
147                return (read || !pp.read) && (write || !pp.write);
148            }
149            return false;
150        }
151    
152        /**
153         * Returns a new {@code PermissionCollection} for holding permissions of this class.
154         * Returns {@code null} if any {@code PermissionCollection} can be used.
155         * 
156         * @return a new {@code PermissionCollection} or {@code null}.
157         * @see java.security.PermissionCollection
158         * @since Android 1.0
159         */
160        @Override
161        public PermissionCollection newPermissionCollection() {
162            return new PropertyPermissionCollection();
163        }
164    
165        private static final ObjectStreamField[] serialPersistentFields = { new ObjectStreamField(
166                "actions", String.class) }; //$NON-NLS-1$
167    
168        private void writeObject(ObjectOutputStream stream) throws IOException {
169            ObjectOutputStream.PutField fields = stream.putFields();
170            fields.put("actions", getActions()); //$NON-NLS-1$
171            stream.writeFields();
172        }
173    
174        private void readObject(ObjectInputStream stream) throws IOException,
175                ClassNotFoundException {
176            ObjectInputStream.GetField fields = stream.readFields();
177            String actions = (String) fields.get("actions", ""); //$NON-NLS-1$ //$NON-NLS-2$
178            decodeActions(actions);
179        }
180    }
