149        /**
150         * Constructs a new {@code AbstractPreferences} instance using the given parent node
151         * and node name.
152         * 
153         * @param parent
154         *            the parent node of the new node or {@code null} to indicate
155         *            that the new node is a root node.
156         * @param name
157         *            the name of the new node or an empty string to indicate that
158         *            this node is called "root".
159         * @throws IllegalArgumentException
160         *             if the name contains a slash character or is empty if {@code
161         *             parent} is not {@code null}.
162         * @since Android 1.0
163         */
164        protected AbstractPreferences(AbstractPreferences parent, String name) {
165            if ((null == parent ^ name.length() == 0) || name.indexOf("/") >= 0) { //$NON-NLS-1$
166                throw new IllegalArgumentException();
167            }
168            root = null == parent ? this : parent.root;
169            nodeChangeListeners = new LinkedList<EventListener>();
170            preferenceChangeListeners = new LinkedList<EventListener>();
171            isRemoved = false;
172            cachedNode = new HashMap<String, AbstractPreferences>();
173            nodeName = name;
174            parentPref = parent;
175            lock = new Lock();
176            userNode = root.userNode;
177        }

195         * Returns the child node with the specified name or {@code null} if it
196         * doesn't exist. Implementers can assume that the name supplied to this method 
197         * will be a valid node name string (conforming to the node naming format) and 
198         * will not correspond to a node that has been cached or removed.
199         * 
200         * @param name
201         *            the name of the desired child node.
202         * @return the child node with the given name or {@code null} if it doesn't
203         *         exist.
204         * @throws BackingStoreException
205         *             if the backing store is unavailable or causes an operation
206         *             failure.
207         * @since Android 1.0
208         */
209        protected AbstractPreferences getChild(String name)
210                throws BackingStoreException {
211            synchronized (lock) {
212                checkState();
213                AbstractPreferences result = null;
214                String[] childrenNames = childrenNames();
215                for (int i = 0; i < childrenNames.length; i++) {
216                    if (childrenNames[i].equals(name)) {
217                        result = childSpi(name);
218                        break;
219                    }
220                }
221                return result;
222            }

427            synchronized (lock) {
428                flushSpi();

438        public String get(String key, String deflt) {
439            if (key == null) {
440                throw new NullPointerException();
441            }
442            String result;
443            synchronized (lock) {
444                checkState();
445                try {
446                    result = getSpi(key);

448                    result = null;
449                }
450            }
451            return (result == null ? deflt : result);

455        public boolean getBoolean(String key, boolean deflt) {
456            String result = get(key, null);
457            if (result == null) {
458                return deflt;
459            } else if (result.equalsIgnoreCase("true")) { //$NON-NLS-1$
460                return true;
461            } else if (result.equalsIgnoreCase("false")) { //$NON-NLS-1$
462                return false;
463            } else {
464                return deflt;

468        @Override
469        public byte[] getByteArray(String key, byte[] deflt) {
470            String svalue = get(key, null);
471            if (svalue == null) {
472                return deflt;
473            }
474            if (svalue.length() == 0) { 
475                return new byte[0];
476            }
477            byte[] dres;
478            try {
479                byte[] bavalue = svalue.getBytes("US-ASCII"); //$NON-NLS-1$
480                if (bavalue.length % 4 != 0) {
481                    return deflt;
482                }
483                dres = Base64.decode(bavalue);
484            } catch (Exception e) {
485                dres = deflt;
486            }
487            return dres;
488        }

568        @Override
569        public Preferences node(String name) {
570            AbstractPreferences startNode = null;
571            synchronized (lock) {
572                checkState();
573                validateName(name);
574                if ("".equals(name)) { //$NON-NLS-1$
575                    return this;
576                } else if ("/".equals(name)) { //$NON-NLS-1$
577                    return root;
578                }
579                if (name.startsWith("/")) { //$NON-NLS-1$
580                    startNode = root;
581                    name = name.substring(1);
582                } else {
583                    startNode = this;
584                }
585            }
586            Preferences result = null;
587            try {
588                result = startNode.nodeImpl(name, true);
589            } catch (BackingStoreException e) {
590                //should not happen
591            }
592            return result;
593        }

595        private void validateName(String name) {
596            if (name.endsWith("/") && name.length() > 1) { //$NON-NLS-1$
597                // prefs.6=Name cannot end with '/'\!
598                throw new IllegalArgumentException(Messages.getString("prefs.6"));  //$NON-NLS-1$
599            }
600            if (name.indexOf("//") >= 0) { //$NON-NLS-1$
601                // prefs.7=Name cannot contains consecutive '/'\!
602                throw new IllegalArgumentException(
603                        Messages.getString("prefs.7"));  //$NON-NLS-1$
604            }
605        }

607        private AbstractPreferences nodeImpl(String path, boolean createNew)
608                throws BackingStoreException {
609            StringTokenizer st = new StringTokenizer(path, "/"); //$NON-NLS-1$
610            AbstractPreferences currentNode = this;
611            AbstractPreferences temp = null;
612            while (st.hasMoreTokens() && null != currentNode) {
613                String name = st.nextToken();
614                synchronized (currentNode.lock) {
615                    temp = currentNode.cachedNode.get(name);
616                    if (temp == null) {
617                        temp = getNodeFromBackend(createNew, currentNode, name);
618                    }
619                }
620    
621                currentNode = temp;
622            }
623            return currentNode;
624        }
625    
626        private AbstractPreferences getNodeFromBackend(boolean createNew,
627                AbstractPreferences currentNode, String name)
628                throws BackingStoreException {
629            AbstractPreferences temp;
630            if (name.length() > MAX_NAME_LENGTH) {
631                // prefs.8=Name length is too long: {0}
632                throw new IllegalArgumentException(Messages.getString("prefs.8",  //$NON-NLS-1$
633                        name));
634            }
635            if (createNew) {
636                temp = currentNode.childSpi(name);
637                currentNode.cachedNode.put(name, temp);
638                if (temp.newNode && currentNode.nodeChangeListeners.size() > 0) {
639                    currentNode.notifyChildAdded(temp);
640                }
641            } else {
642                temp = currentNode.getChild(name);
643            }
644            return temp;
645        }

647        @Override
648        public boolean nodeExists(String name) throws BackingStoreException {
649            AbstractPreferences startNode = null;
650            synchronized (lock) {
651                if (isRemoved()) {
652                    if ("".equals(name)) { //$NON-NLS-1$
653                        return false;
654                    }
655                    // prefs.9=This node has been removed\!
656                    throw new IllegalStateException(Messages.getString("prefs.9"));  //$NON-NLS-1$
657                }
658                validateName(name);
659                if ("".equals(name) || "/".equals(name)) { //$NON-NLS-1$ //$NON-NLS-2$
660                    return true;
661                }
662                if (name.startsWith("/")) { //$NON-NLS-1$
663                    startNode = root;
664                    name = name.substring(1);
665                } else {
666                    startNode = this;
667                }
668            }
669            try {
670                Preferences result = startNode.nodeImpl(name, false);
671                return null == result ? false : true;
672            } catch(IllegalArgumentException e) {
673                return false;
674            }
675        }

693                throw new NullPointerException();
694            }

696                throw new IllegalArgumentException();
697            }
698            synchronized (lock) {

701            }
702            notifyPreferenceChange(key, value);

753        @Override
754        public void removeNode() throws BackingStoreException {
755            if (root == this) {
756                // prefs.A=Cannot remove root node\!
757                throw new UnsupportedOperationException(Messages.getString("prefs.A"));  //$NON-NLS-1$
758            }
759            synchronized (parentPref.lock) {
760                removeNodeImpl();
761            }
762        }

764        private void removeNodeImpl() throws BackingStoreException {
765            synchronized (lock) {
766                checkState();
767                String[] childrenNames = childrenNamesSpi();
768                for (int i = 0; i < childrenNames.length; i++) {
769                    if (null == cachedNode.get(childrenNames[i])) {
770                        AbstractPreferences child = childSpi(childrenNames[i]);
771                        cachedNode.put(childrenNames[i], child);
772                    }
773                }
774                AbstractPreferences[] children = cachedNode
775                        .values().toArray(new AbstractPreferences[0]);
776                for (int i = 0; i < children.length; i++) {
777                    children[i].removeNodeImpl();
778                }
779                removeNodeSpi();
780                isRemoved = true;
781                parentPref.cachedNode.remove(nodeName);
782            }
783            if (parentPref.nodeChangeListeners.size() > 0) {
784                parentPref.notifyChildRemoved(this);
785            }
786        }

881        private static class EventDispatcher extends Thread {
882            EventDispatcher(String name){
883                super(name);
884            }
885            
886            @Override
887            public void run() {
888                while (true) {
889                    EventObject event = null;
890                    try {
891                        event = getEventObject();
892                    } catch (InterruptedException e) {
893                        e.printStackTrace();
894                        continue;
895                    }
896                    AbstractPreferences pref = (AbstractPreferences) event
897                            .getSource();
898                    if (event instanceof NodeAddEvent) {
899                        dispatchNodeAdd((NodeChangeEvent) event,
900                                pref.nodeChangeListeners);
901                    } else if (event instanceof NodeRemoveEvent) {
902                        dispatchNodeRemove((NodeChangeEvent) event,
903                                pref.nodeChangeListeners);
904                    } else if (event instanceof PreferenceChangeEvent) {
905                        dispatchPrefChange((PreferenceChangeEvent) event,
906                                pref.preferenceChangeListeners);
907                    }
908                }
909            }
910    
911            private EventObject getEventObject() throws InterruptedException {
912                synchronized (events) {
913                    if (events.isEmpty()) {
914                        events.wait();
915                    }
916                    EventObject event = events.get(0);
917                    events.remove(0);
918                    return event;
919                }
920            }
921    
922            private void dispatchPrefChange(PreferenceChangeEvent event,
923                    List<EventListener> preferenceChangeListeners) {
924                synchronized (preferenceChangeListeners) {
925                    Iterator<EventListener> i = preferenceChangeListeners.iterator();
926                    while (i.hasNext()) {
927                        PreferenceChangeListener pcl = (PreferenceChangeListener) i
928                                .next();
929                        pcl.preferenceChange(event);
930                    }
931                }
932            }
933    
934            private void dispatchNodeRemove(NodeChangeEvent event,
935                    List<EventListener> nodeChangeListeners) {
936                synchronized (nodeChangeListeners) {
937                    Iterator<EventListener> i = nodeChangeListeners.iterator();
938                    while (i.hasNext()) {
939                        NodeChangeListener ncl = (NodeChangeListener) i.next();
940                        ncl.childRemoved(event);
941                    }
942                }
943            }
944    
945            private void dispatchNodeAdd(NodeChangeEvent event,
946                    List<EventListener> nodeChangeListeners) {
947                synchronized (nodeChangeListeners) {
948                    Iterator<EventListener> i = nodeChangeListeners.iterator();
949                    while (i.hasNext()) {
950                        NodeChangeListener ncl = (NodeChangeListener) i.next();
951                        ncl.childAdded(event);
952                    }
953                }
954            }
955        }
