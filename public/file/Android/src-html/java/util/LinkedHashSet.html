018    package java.util;
019    
020    
021    import java.io.Serializable;
022    
023    /**
024     * LinkedHashSet is a variant of HashSet. Its entries are kept in a
025     * doubly-linked list. The iteration order is the order in which entries were
026     * inserted.
027     * <p>
028     * Null elements are allowed, and all the optional Set operations are supported.
029     * <p>
030     * Like HashSet, LinkedHashSet is not thread safe, so access by multiple threads
031     * must be synchronized by an external mechanism such as
032     * {@link Collections#synchronizedSet(Set)}.
033     * 
034     * @since Android 1.0
035     */
036    public class LinkedHashSet<E> extends HashSet<E> implements Set<E>, Cloneable,
037            Serializable {
038        
039        private static final long serialVersionUID = -2851667679971038690L;
040    
041        /**
042         * Constructs a new empty instance of {@code LinkedHashSet}.
043         * 
044         * @since Android 1.0
045         */
046        public LinkedHashSet() {
047            super(new LinkedHashMap<E, HashSet<E>>());
048        }
049    
050        /**
051         * Constructs a new instance of {@code LinkedHashSet} with the specified
052         * capacity.
053         * 
054         * @param capacity
055         *            the initial capacity of this {@code LinkedHashSet}.
056         * @since Android 1.0
057         */
058        public LinkedHashSet(int capacity) {
059            super(new LinkedHashMap<E, HashSet<E>>(capacity));
060        }
061    
062        /**
063         * Constructs a new instance of {@code LinkedHashSet} with the specified
064         * capacity and load factor.
065         * 
066         * @param capacity
067         *            the initial capacity.
068         * @param loadFactor
069         *            the initial load factor.
070         * @since Android 1.0
071         */
072        public LinkedHashSet(int capacity, float loadFactor) {
073            super(new LinkedHashMap<E, HashSet<E>>(capacity, loadFactor));
074        }
075    
076        /**
077         * Constructs a new instance of {@code LinkedHashSet} containing the unique
078         * elements in the specified collection.
079         * 
080         * @param collection
081         *            the collection of elements to add.
082         * @since Android 1.0
083         */
084        public LinkedHashSet(Collection<? extends E> collection) {
085            super(new LinkedHashMap<E, HashSet<E>>(collection.size() < 6 ? 11
086                    : collection.size() * 2));
087            for (E e : collection) {
088                add(e);
089            }
090        }
091    
092        /* overrides method in HashMap */
093        @Override
094        HashMap<E, HashSet<E>> createBackingMap(int capacity, float loadFactor) {
095            return new LinkedHashMap<E, HashSet<E>>(capacity, loadFactor);
096        }
097    }
