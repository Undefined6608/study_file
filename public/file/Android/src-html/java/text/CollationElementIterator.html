018    package java.text;
019    
020    // BEGIN android-note
021    // The icu implementation used was changed from icu4j to icu4jni.
022    // END android-note
023    
024    /**
025     * Created by a {@code RuleBasedCollator} to iterate through a string. The
026     * result of each iteration is a 32-bit collation element that defines the
027     * ordering priority of the next character or sequence of characters in the
028     * source string.
029     * <p>
030     * For illustration, consider the following in Spanish:
031     * </p>
032     * <p>
033     * "ca": the first collation element is collation_element('c') and second
034     * collation element is collation_element('a').
035     * </p>
036     * <p>
037     * Since "ch" in Spanish sorts as one entity, the example below returns one
038     * collation element for the two characters 'c' and 'h':
039     * </p>
040     * <p>
041     * "cha": the first collation element is collation_element('ch') and the second
042     * one is collation_element('a').
043     * </p>
044     * <p>
045     * In German, since the character '&#92;u0086' is a composed character of 'a'
046     * and 'e', the iterator returns two collation elements for the single character
047     * '&#92;u0086':
048     * </p>
049     * <p>
050     * "&#92;u0086b": the first collation element is collation_element('a'), the
051     * second one is collation_element('e'), and the third collation element is
052     * collation_element('b').
053     * </p>
054     * 
055     * @since Android 1.0
056     */
057    public final class CollationElementIterator {
058    
059        /**
060         * This constant is returned by the iterator in the methods
061         * {@code next()} and {@code previous()} when the end or the
062         * beginning of the source string has been reached, and there are no more
063         * valid collation elements to return.
064         * 
065         * @since Android 1.0
066         */
067        public static final int NULLORDER = -1;
068    
069        private com.ibm.icu4jni.text.CollationElementIterator icuIterator;
070    
071        CollationElementIterator(com.ibm.icu4jni.text.CollationElementIterator iterator) {
072            this.icuIterator = iterator;
073        }
074    
075        /**
076         * Obtains the maximum length of any expansion sequence that ends with the
077         * specified collation element. Returns {@code 1} if there is no expansion
078         * with this collation element as the last element.
079         * 
080         * @param order
081         *            a collation element that has been previously obtained from a
082         *            call to either the {@link #next()} or {@link #previous()}
083         *            method.
084         * @return the maximum length of any expansion sequence ending with the
085         *         specified collation element.
086         * @since Android 1.0
087         */
088        public int getMaxExpansion(int order) {
089            return this.icuIterator.getMaxExpansion(order);
090        }
091    
092        /**
093         * Obtains the character offset in the source string corresponding to the
094         * next collation element. This value could be any of:
095         * <ul>
096         * <li>The index of the first character in the source string that matches
097         * the value of the next collation element. This means that if
098         * {@code setOffset(offset)} sets the index in the middle of a contraction,
099         * {@code getOffset()} returns the index of the first character in the
100         * contraction, which may not be equal to the original offset that was set.
101         * Hence calling {@code getOffset()} immediately after
102         * {@code setOffset(offset)} does not guarantee that the original offset set
103         * will be returned.</li>
104         * <li>If normalization is on, the index of the immediate subsequent
105         * character, or composite character with the first character, having a
106         * combining class of 0.</li>
107         * <li>The length of the source string, if iteration has reached the end.
108         * </li>
109         * </ul>
110         * 
111         * @return The position of the collation element in the source string that
112         *         will be returned by the next invocation of the {@link #next()}
113         *         method.
114         * @since Android 1.0
115         */
116        public int getOffset() {
117            return this.icuIterator.getOffset();
118        }
119    
120        /**
121         * Obtains the next collation element in the source string.
122         * 
123         * @return the next collation element or {@code NULLORDER} if the end
124         *         of the iteration has been reached.
125         * @since Android 1.0
126         */
127        public int next() {
128            return this.icuIterator.next();
129        }
130    
131        /**
132         * Obtains the previous collation element in the source string.
133         * 
134         * @return the previous collation element, or {@code NULLORDER} when
135         *         the start of the iteration has been reached.
136         * @since Android 1.0
137         */
138        public int previous() {
139            return this.icuIterator.previous();
140        }
141    
142        /**
143         * Obtains the primary order of the specified collation element, i.e. the
144         * first 16 bits. This value is unsigned.
145         * 
146         * @param order
147         *            the element of the collation.
148         * @return the element's 16 bit primary order.
149         * @since Android 1.0
150         */
151        public static final int primaryOrder(int order) {
152            return com.ibm.icu4jni.text.CollationElementIterator.primaryOrder(order);
153        }
154    
155        /**
156         * Repositions the cursor to point at the first element of the current
157         * string. The next call to {@link #next()} or {@link #previous()} will
158         * return the first and last collation element in the string, respectively.
159         * <p>
160         * If the {@code RuleBasedCollator} used by this iterator has had its
161         * attributes changed, calling {@code reset()} reinitializes the iterator to
162         * use the new attributes.
163         * </p>
164         * 
165         * @since Android 1.0
166         */
167        public void reset() {
168            this.icuIterator.reset();
169        }
170    
171        /**
172         * Obtains the secondary order of the specified collation element, i.e. the
173         * 16th to 23th bits, inclusive. This value is unsigned.
174         * 
175         * @param order
176         *            the element of the collator.
177         * @return the 8 bit secondary order of the element.
178         * @since Android 1.0
179         */
180        public static final short secondaryOrder(int order) {
181            return (short) com.ibm.icu4jni.text.CollationElementIterator
182                    .secondaryOrder(order);
183        }
184    
185        /**
186         * Points the iterator at the collation element associated with the
187         * character in the source string which is found at the supplied offset.
188         * After this call completes, an invocation of the {@link #next()} method
189         * will return this collation element.
190         * <p>
191         * If {@code newOffset} corresponds to a character which is part of a
192         * sequence that maps to a single collation element then the iterator is
193         * adjusted to the start of that sequence. As a result of this, any
194         * subsequent call made to {@code getOffset()} may not return the same value
195         * set by this method.
196         * </p>
197         * <p>
198         * If the decomposition mode is on, and offset is in the middle of a
199         * decomposable range of source text, the iterator may not return a correct
200         * result for the next forwards or backwards iteration. The user must ensure
201         * that the offset is not in the middle of a decomposable range.
202         * </p>
203         * 
204         * @param newOffset
205         *            the character offset into the original source string to set.
206         *            Note that this is not an offset into the corresponding
207         *            sequence of collation elements.
208         * @since Android 1.0
209         */
210        public void setOffset(int newOffset) {
211            this.icuIterator.setOffset(newOffset);
212        }
213    
214        /**
215         * Sets a new source string iterator for iteration, and resets the offset to
216         * the beginning of the text.
217         * 
218         * @param source
219         *            the new source string iterator for iteration.
220         * @since Android 1.0
221         */
222        public void setText(CharacterIterator source) {
223            this.icuIterator.setText(source);
224        }
225    
226        /**
227         * Sets a new source string for iteration, and resets the offset to the
228         * beginning of the text.
229         * 
230         * @param source
231         *            the new source string for iteration.
232         * @since Android 1.0
233         */
234        public void setText(String source) {
235            this.icuIterator.setText(source);
236        }
237    
238        /**
239         * Obtains the tertiary order of the specified collation element, i.e. the
240         * last 8 bits. This value is unsigned.
241         * 
242         * @param order
243         *            the element of the collation.
244         * @return the 8 bit tertiary order of the element.
245         * @since Android 1.0
246         */
247        public static final short tertiaryOrder(int order) {
248            return (short) com.ibm.icu4jni.text.CollationElementIterator
249                    .tertiaryOrder(order);
250        }
251    }
