040    /**
041     * Performs locale-sensitive string comparison. A concrete subclass,
042     * {@link RuleBasedCollator}, allows customization of the collation ordering by
043     * the use of rule sets.
044     * <p>
045     * Following the <a href=http://www.unicode.org>Unicode Consortium</a>'s
046     * specifications for the <a
047     * href="http://www.unicode.org/unicode/reports/tr10/"> Unicode Collation
048     * Algorithm (UCA)</a>, there are 4 different levels of strength used in
049     * comparisons:
050     * </p>
051     * <ul>
052     * <li>PRIMARY strength: Typically, this is used to denote differences between
053     * base characters (for example, "a" &lt; "b"). It is the strongest difference.
054     * For example, dictionaries are divided into different sections by base
055     * character.
056     * <li>SECONDARY strength: Accents in the characters are considered secondary
057     * differences (for example, "as" &lt; "&agrave;s" &lt; "at"). Other differences
058     * between letters can also be considered secondary differences, depending on
059     * the language. A secondary difference is ignored when there is a primary
060     * difference anywhere in the strings.
061     * <li>TERTIARY strength: Upper and lower case differences in characters are
062     * distinguished at tertiary strength (for example, "ao" &lt; "Ao" &lt;
063     * "a&ograve;"). In addition, a variant of a letter differs from the base form
064     * on the tertiary strength (such as "A" and "&#9398;"). Another example is the
065     * difference between large and small Kana. A tertiary difference is ignored
066     * when there is a primary or secondary difference anywhere in the strings.
067     * <li>IDENTICAL strength: When all other strengths are equal, the IDENTICAL
068     * strength is used as a tiebreaker. The Unicode code point values of the NFD
069     * form of each string are compared, just in case there is no difference. For
070     * example, Hebrew cantellation marks are only distinguished at this strength.
071     * This strength should be used sparingly, as only code point value differences
072     * between two strings are an extremely rare occurrence. Using this strength
073     * substantially decreases the performance for both comparison and collation key
074     * generation APIs. This strength also increases the size of the collation key.
075     * </ul>
076     * <p>
077     * This {@code Collator} deals only with two decomposition modes, the canonical
078     * decomposition mode and one that does not use any decomposition. The
079     * compatibility decomposition mode
080     * {@code java.text.Collator.FULL_DECOMPOSITION} is not supported here. If the
081     * canonical decomposition mode is set, {@code Collator} handles un-normalized
082     * text properly, producing the same results as if the text were normalized in
083     * NFD. If canonical decomposition is turned off, it is the user's
084     * responsibility to ensure that all text is already in the appropriate form
085     * before performing a comparison or before getting a {@link CollationKey}.
086     * </p>
087     * <p>
088     * <em>Examples:</em>
089     * </p>
090     * <blockquote>
091     * 
092     * <pre>
093     * // Get the Collator for US English and set its strength to PRIMARY
094     * Collator usCollator = Collator.getInstance(Locale.US);
095     * usCollator.setStrength(Collator.PRIMARY);
096     * if (usCollator.compare(&quot;abc&quot;, &quot;ABC&quot;) == 0) {
097     *     System.out.println(&quot;Strings are equivalent&quot;);
098     * }
099     * </pre>
100     * 
101     * </blockquote>
102     * <p>
103     * The following example shows how to compare two strings using the collator for
104     * the default locale.
105     * </p>
106     * <blockquote>
107     * 
108     * <pre>
109     * // Compare two strings in the default locale
110     * Collator myCollator = Collator.getInstance();
111     * myCollator.setDecomposition(Collator.NO_DECOMPOSITION);
112     * if (myCollator.compare(&quot;\u00e0\u0325&quot;, &quot;a\u0325\u0300&quot;) != 0) {
113     *     System.out.println(&quot;\u00e0\u0325 is not equal to a\u0325\u0300 without decomposition&quot;);
114     *     myCollator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
115     *     if (myCollator.compare(&quot;\u00e0\u0325&quot;, &quot;a\u0325\u0300&quot;) != 0) {
116     *         System.out.println(&quot;Error: \u00e0\u0325 should be equal to a\u0325\u0300 with decomposition&quot;);
117     *     } else {
118     *         System.out.println(&quot;\u00e0\u0325 is equal to a\u0325\u0300 with decomposition&quot;);
119     *     }
120     * } else {
121     *     System.out.println(&quot;Error: \u00e0\u0325 should be not equal to a\u0325\u0300 without decomposition&quot;);
122     * }
123     * </pre>
124     * 
125     * </blockquote>
126     * 
127     * @see RuleBasedCollator
128     * @see CollationKey
129     * @since Android 1.0
130     */
131    public abstract class Collator implements Comparator<Object>, Cloneable {

133        static final int EQUAL = 0;
134    
135        static final int GREATER = 1;
136    
137        static final int LESS = -1;

139        /**
140         * Constant used to specify the decomposition rule.
141         * 
142         * @since Android 1.0

144        public static final int NO_DECOMPOSITION = 0;
145    
146        /**
147         * Constant used to specify the decomposition rule.
148         * 
149         * @since Android 1.0
150         */
151        public static final int CANONICAL_DECOMPOSITION = 1;
152    
153        /**
154         * Constant used to specify the decomposition rule. This value for
155         * decomposition is not supported.
156         * 
157         * @since Android 1.0
158         */
159        public static final int FULL_DECOMPOSITION = 2;
160    
161        /**
162         * Constant used to specify the collation strength.
163         * 
164         * @since Android 1.0
165         */
166        public static final int PRIMARY = 0;
167    
168        /**
169         * Constant used to specify the collation strength.
170         * 
171         * @since Android 1.0
172         */
173        public static final int SECONDARY = 1;
174    
175        /**
176         * Constant used to specify the collation strength.
177         * 
178         * @since Android 1.0
179         */
180        public static final int TERTIARY = 2;
181    
182        /**
183         * Constant used to specify the collation strength.
184         * 
185         * @since Android 1.0
186         */
187        public static final int IDENTICAL = 3;

189        private static int CACHE_SIZE;
190    
191        static {
192            // CACHE_SIZE includes key and value, so needs to be double
193            String cacheSize = AccessController
194                    .doPrivileged(new PriviAction<String>("collator.cache")); //$NON-NLS-1$
195            if (cacheSize != null) {
196                try {
197                    CACHE_SIZE = Integer.parseInt(cacheSize);
198                } catch (NumberFormatException e) {
199                    CACHE_SIZE = 6;
200                }
201            } else {
202                CACHE_SIZE = 6;
203            }
204        }
205    
206        private static Vector<Collator> cache = new Vector<Collator>(CACHE_SIZE);
207    
208        // Wrapper class of ICU4JNI Collator
209        com.ibm.icu4jni.text.Collator icuColl;
210    
211        Collator(com.ibm.icu4jni.text.Collator wrapper) {
212            this.icuColl = wrapper;
213        }
214    
215        /**
216         * Constructs a new {@code Collator} instance.
217         * 
218         * @since Android 1.0
219         */
220        protected Collator() {
221            super();
222            // BEGIN android-added
223            icuColl = com.ibm.icu4jni.text.Collator.getInstance(Locale.getDefault());
224            // END android-added
225        }
226    
227        /**
228         * Returns a new collator with the same decomposition mode and
229         * strength value as this collator.
230         * 
231         * @return a shallow copy of this collator.
232         * @see java.lang.Cloneable
233         * @since Android 1.0
234         */
235        @Override
236        public Object clone() {
237            try {
238                Collator clone = (Collator) super.clone();
239                clone.icuColl = (com.ibm.icu4jni.text.Collator) this.icuColl.clone();
240                return clone;
241            } catch (CloneNotSupportedException e) {
242                return null;
243            }
244        }
245    
246        /**
247         * Compares two objects to determine their relative order. The objects must
248         * be strings.
249         * 
250         * @param object1
251         *            the first string to compare.
252         * @param object2
253         *            the second string to compare.
254         * @return a negative value if {@code object1} is less than {@code object2},
255         *         0 if they are equal, and a positive value if {@code object1} is
256         *         greater than {@code object2}.
257         * @exception ClassCastException
258         *                if {@code object1} or {@code object2} is not a
259         *                {@code String}.
260         * @since Android 1.0
261         */
262        public int compare(Object object1, Object object2) {
263            return compare((String) object1, (String) object2);
264        }
265    
266        /**
267         * Compares two strings to determine their relative order.
268         * 
269         * @param string1
270         *            the first string to compare.
271         * @param string2
272         *            the second string to compare.
273         * @return a negative value if {@code string1} is less than {@code string2},
274         *         0 if they are equal and a positive value if {@code string1} is
275         *         greater than {@code string2}.
276         * @since Android 1.0
277         */
278        public abstract int compare(String string1, String string2);
279    
280        /**
281         * Compares this collator with the specified object and indicates if they
282         * are equal.
283         * 
284         * @param object
285         *            the object to compare with this object.
286         * @return {@code true} if {@code object} is a {@code Collator} object and
287         *         it has the same strength and decomposition values as this
288         *         collator; {@code false} otherwise.
289         * @see #hashCode
290         * @since Android 1.0
291         */
292        @Override
293        public boolean equals(Object object) {
294            if (!(object instanceof Collator)) {
295                return false;
296            }
297            Collator collator = (Collator) object;
298            return this.icuColl == null ? collator.icuColl == null : this.icuColl
299                    .equals(collator.icuColl);
300        }
301    
302        /**
303         * Compares two strings using the collation rules to determine if they are
304         * equal.
305         * 
306         * @param string1
307         *            the first string to compare.
308         * @param string2
309         *            the second string to compare.
310         * @return {@code true} if {@code string1} and {@code string2} are equal
311         *         using the collation rules, false otherwise.
312         * @since Android 1.0
313         */
314        public boolean equals(String string1, String string2) {
315            return compare(string1, string2) == 0;
316        }

338        public abstract CollationKey getCollationKey(String string);

340        /**
341         * Returns the decomposition rule for this collator.
342         * 
343         * @return the decomposition rule, either {@code NO_DECOMPOSITION} or

345         *         not supported.

348        public int getDecomposition() {
349            return decompositionMode_ICU_Java(this.icuColl.getDecomposition());

352        /**
353         * Returns a {@code Collator} instance which is appropriate for the default
354         * {@code Locale}.
355         * 
356         * @return the collator for the default locale.
357         * @since Android 1.0
358         */
359        public static Collator getInstance() {
360            return getInstance(Locale.getDefault());
361        }
362    
363        /**
364         * Returns a {@code Collator} instance which is appropriate for the
365         * specified {@code Locale}.
366         * 
367         * @param locale
368         *            the locale.
369         * @return the collator for {@code locale}.
370         * @since Android 1.0
371         */
372        public static Collator getInstance(Locale locale) {
373            String key = locale.toString();
374            for (int i = cache.size() - 1; i >= 0; i -= 2) {
375                if (cache.elementAt(i).equals(key)) {
376                    return (Collator) (cache.elementAt(i - 1)).clone();
377                }
378            }
379    
380            return new RuleBasedCollator(com.ibm.icu4jni.text.Collator
381                    .getInstance(locale));
382        }
383    
384        /**
385         * Returns the strength value for this collator.
386         * 
387         * @return the strength value, either PRIMARY, SECONDARY, TERTIARY or
388         *         IDENTICAL.
389         * @since Android 1.0
390         */
391        public int getStrength() {
392            return strength_ICU_Java(this.icuColl.getStrength());
393        }

396         * Returns an integer hash code for this collator.

407        /**
408         * Sets the decomposition rule for this collator.
409         * 
410         * @param value
411         *            the decomposition rule, either {@code NO_DECOMPOSITION} or
412         *            {@code CANONICAL_DECOMPOSITION}. {@code FULL_DECOMPOSITION}
413         *            is not supported.
414         * @exception IllegalArgumentException
415         *                if the provided decomposition rule is not valid. This 
416         *                includes {@code FULL_DECOMPOSITION}.
417         * @since Android 1.0

419        public void setDecomposition(int value) {
420            this.icuColl.setDecomposition(decompositionMode_Java_ICU(value));
421        }
422    
423        /**
424         * Sets the strength value for this collator.
425         * 
426         * @param value
427         *            the strength value, either PRIMARY, SECONDARY, TERTIARY, or
428         *            IDENTICAL.
429         * 
430         * @exception IllegalArgumentException
431         *                if the provided strength value is not valid.
432         * @since Android 1.0
433         */
434        public void setStrength(int value) {
435            this.icuColl.setStrength(strength_Java_ICU(value));
436        }
437    
438        private int decompositionMode_Java_ICU(int mode) {
439            int icuDecomp = mode;
440            switch (mode) {
441                case Collator.CANONICAL_DECOMPOSITION:
442                    icuDecomp = com.ibm.icu4jni.text.Collator.CANONICAL_DECOMPOSITION;
443                    break;
444                case Collator.NO_DECOMPOSITION:
445                    icuDecomp = com.ibm.icu4jni.text.Collator.NO_DECOMPOSITION;
446                    break;
447            }
448            return icuDecomp;
449        }
450    
451        private int decompositionMode_ICU_Java(int mode) {
452            int javaMode = mode;
453            switch (mode) {
454                case com.ibm.icu4jni.text.Collator.NO_DECOMPOSITION:
455                    javaMode = Collator.NO_DECOMPOSITION;
456                    break;
457                case com.ibm.icu4jni.text.Collator.CANONICAL_DECOMPOSITION:
458                    javaMode = Collator.CANONICAL_DECOMPOSITION;
459                    break;
460            }
461            return javaMode;
462        }
463    
464        private int strength_Java_ICU(int value) {
465            int icuValue = value;
466            switch (value) {
467                case Collator.PRIMARY:
468                    icuValue = com.ibm.icu4jni.text.Collator.PRIMARY;
469                    break;
470                case Collator.SECONDARY:
471                    icuValue = com.ibm.icu4jni.text.Collator.SECONDARY;
472                    break;
473                case Collator.TERTIARY:
474                    icuValue = com.ibm.icu4jni.text.Collator.TERTIARY;
475                    break;
476                case Collator.IDENTICAL:
477                    icuValue = com.ibm.icu4jni.text.Collator.IDENTICAL;
478                    break;
479            }
480            return icuValue;
481    
482        }
483    
484        private int strength_ICU_Java(int value) {
485            int javaValue = value;
486            switch (value) {
487                case com.ibm.icu4jni.text.Collator.PRIMARY:
488                    javaValue = Collator.PRIMARY;
489                    break;
490                case com.ibm.icu4jni.text.Collator.SECONDARY:
491                    javaValue = Collator.SECONDARY;
492                    break;
493                case com.ibm.icu4jni.text.Collator.TERTIARY:
494                    javaValue = Collator.TERTIARY;
495                    break;
496                case com.ibm.icu4jni.text.Collator.IDENTICAL:
497                    javaValue = Collator.IDENTICAL;
498                    break;
499            }
500            return javaValue;
501        }
