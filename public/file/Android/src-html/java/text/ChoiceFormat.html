037    /**
038     * Returns a fixed string based on a numeric value. The class can be used in
039     * conjunction with the {@link MessageFormat} class to handle plurals in
040     * messages. {@code ChoiceFormat} enables users to attach a format to a range of
041     * numbers. The choice is specified with an ascending list of doubles, where
042     * each item specifies a half-open interval up to the next item as in the
043     * following: X matches j if and only if {@code limit[j] <= X < limit[j+1]}.
044     * <p>
045     * If there is no match, then either the first or last index is used. The first
046     * or last index is used depending on whether the number is too low or too high.
047     * The length of the format array must be the same as the length of the limits
048     * array.
049     * </p>
050     * <h5>Examples:</h5>
051     * <blockquote>
052     * 
053     * <pre>
054     * double[] limits = {1, 2, 3, 4, 5, 6, 7};
055     * String[] fmts = {"Sun", "Mon", "Tue", "Wed", "Thur", "Fri", "Sat"};
056     * 
057     * double[] limits2 = {0, 1, ChoiceFormat.nextDouble(1)};
058     * String[] fmts2 = {"no files", "one file", "many files"};
059     * </pre>
060     * </blockquote>
061     * <p> 
062     * ChoiceFormat.nextDouble(double) allows to get the double following the one
063     * passed to the method. This is used to create half open intervals.
064     * </p>
065     * <p>
066     * {@code ChoiceFormat} objects also may be converted to and from patterns.
067     * The conversion can be done programmatically, as in the example above, or
068     * by using a pattern like the following:
069     * </p>
070     * <blockquote>
071     * 
072     * <pre>
073     * "1#Sun|2#Mon|3#Tue|4#Wed|5#Thur|6#Fri|7#Sat"
074     * "0#are no files|1#is one file|1&lt;are many files"
075     * </pre>
076     * 
077     * </blockquote>
078     * <p>
079     * where:
080     * </p>
081     * <ul>
082     * <li><number>"#"</number> specifies an inclusive limit value;</li>
083     * <li><number>"<"</number> specifies an exclusive limit value.</li>
084     * </ul>
085     * 
086     * @since Android 1.0
087     */
088    public class ChoiceFormat extends NumberFormat {
089    
090        private static final long serialVersionUID = 1795184449645032964L;
091    
092        private double[] choiceLimits;
093    
094        private String[] choiceFormats;
095    
096        /**
097         * Constructs a new {@code ChoiceFormat} with the specified double values
098         * and associated strings. When calling
099         * {@link #format(double, StringBuffer, FieldPosition) format} with a double
100         * value {@code d}, then the element {@code i} in {@code formats} is
101         * selected where {@code i} fulfills {@code limits[i] <= d < limits[i+1]}.
102         * <p>
103         * The length of the {@code limits} and {@code formats} arrays must be the
104         * same.
105         * </p>
106         * 
107         * @param limits
108         *            an array of doubles in ascending order. The lowest and highest
109         *            possible values are negative and positive infinity.
110         * @param formats
111         *            the strings associated with the ranges defined through {@code
112         *            limits}. The lower bound of the associated range is at the
113         *            same index as the string.
114         * @since Android 1.0
115         */
116        public ChoiceFormat(double[] limits, String[] formats) {
117            setChoices(limits, formats);
118        }
119    
120        /**
121         * Constructs a new {@code ChoiceFormat} with the strings and limits parsed
122         * from the specified pattern.
123         * 
124         * @param template
125         *            the pattern of strings and ranges.
126         * 
127         * @exception IllegalArgumentException
128         *                if an error occurs while parsing the pattern.
129         * @since Android 1.0
130         */
131        public ChoiceFormat(String template) {
132            applyPattern(template);
133        }
134    
135        /**
136         * Parses the pattern to determine new strings and ranges for this
137         * {@code ChoiceFormat}.
138         * 
139         * @param template
140         *            the pattern of strings and ranges.
141         * 
142         * @exception IllegalArgumentException
143         *                if an error occurs while parsing the pattern.
144         * @since Android 1.0
145         */
146        public void applyPattern(String template) {
147            double[] limits = new double[5];
148            List<String> formats = new ArrayList<String>();
149            int length = template.length(), limitCount = 0, index = 0;
150            StringBuffer buffer = new StringBuffer();
151            NumberFormat format = NumberFormat.getInstance(Locale.US);
152            ParsePosition position = new ParsePosition(0);
153            while (true) {
154                index = skipWhitespace(template, index);
155                if (index >= length) {
156                    if (limitCount == limits.length) {
157                        choiceLimits = limits;
158                    } else {
159                        choiceLimits = new double[limitCount];
160                        System.arraycopy(limits, 0, choiceLimits, 0, limitCount);
161                    }
162                    choiceFormats = new String[formats.size()];
163                    for (int i = 0; i < formats.size(); i++) {
164                        choiceFormats[i] = formats.get(i);
165                    }
166                    return;
167                }
168    
169                position.setIndex(index);
170                Number value = format.parse(template, position);
171                index = skipWhitespace(template, position.getIndex());
172                if (position.getErrorIndex() != -1 || index >= length) {
173                    // Fix Harmony 540
174                    choiceLimits = new double[0];
175                    choiceFormats = new String[0];
176                    return;
177                }
178                char ch = template.charAt(index++);
179                if (limitCount == limits.length) {
180                    double[] newLimits = new double[limitCount * 2];
181                    System.arraycopy(limits, 0, newLimits, 0, limitCount);
182                    limits = newLimits;
183                }
184                double next;
185                switch (ch) {
186                    case '#':
187                    case '\u2264':
188                        next = value.doubleValue();
189                        break;
190                    case '<':
191                        next = nextDouble(value.doubleValue());
192                        break;
193                    default:
194                        throw new IllegalArgumentException();
195                }
196                if (limitCount > 0 && next <= limits[limitCount - 1]) {
197                    throw new IllegalArgumentException();
198                }
199                buffer.setLength(0);
200                position.setIndex(index);
201                upTo(template, position, buffer, '|');
202                index = position.getIndex();
203                limits[limitCount++] = next;
204                formats.add(buffer.toString());
205            }
206        }

208        /**
209         * Returns a new instance of {@code ChoiceFormat} with the same ranges and
210         * strings as this {@code ChoiceFormat}.
211         * 
212         * @return a shallow copy of this {@code ChoiceFormat}.
213         * 
214         * @see java.lang.Cloneable
215         * @since Android 1.0
216         */
217        @Override
218        public Object clone() {
219            ChoiceFormat clone = (ChoiceFormat) super.clone();
220            clone.choiceLimits = choiceLimits.clone();
221            clone.choiceFormats = choiceFormats.clone();
222            return clone;
223        }

225        /**
226         * Compares the specified object with this {@code ChoiceFormat}. The object
227         * must be an instance of {@code ChoiceFormat} and have the same limits and
228         * formats to be equal to this instance.
229         * 
230         * @param object
231         *            the object to compare with this instance.
232         * @return {@code true} if the specified object is equal to this instance;
233         *         {@code false} otherwise.
234         * @see #hashCode
235         * @since Android 1.0
236         */
237        @Override
238        public boolean equals(Object object) {
239            if (this == object) {
240                return true;
241            }
242            if (!(object instanceof ChoiceFormat)) {
243                return false;
244            }
245            ChoiceFormat choice = (ChoiceFormat) object;
246            return Arrays.equals(choiceLimits, choice.choiceLimits)
247                    && Arrays.equals(choiceFormats, choice.choiceFormats);
248        }

250        /**
251         * Appends the string associated with the range in which the specified
252         * double value fits to the specified string buffer.
253         * 
254         * @param value
255         *            the double to format.
256         * @param buffer
257         *            the target string buffer to append the formatted value to.
258         * @param field
259         *            a {@code FieldPosition} which is ignored.
260         * @return the string buffer.
261         * @since Android 1.0
262         */
263        @Override
264        public StringBuffer format(double value, StringBuffer buffer,
265                FieldPosition field) {
266            for (int i = choiceLimits.length - 1; i >= 0; i--) {
267                if (choiceLimits[i] <= value) {
268                    return buffer.append(choiceFormats[i]);
269                }
270            }
271            return choiceFormats.length == 0 ? buffer : buffer
272                    .append(choiceFormats[0]);
273        }

276         * Appends the string associated with the range in which the specified long
277         * value fits to the specified string buffer.
278         * 
279         * @param value
280         *            the long to format.
281         * @param buffer
282         *            the target string buffer to append the formatted value to.
283         * @param field
284         *            a {@code FieldPosition} which is ignored.
285         * @return the string buffer.
286         * @since Android 1.0
287         */
288        @Override
289        public StringBuffer format(long value, StringBuffer buffer,
290                FieldPosition field) {
291            return format((double) value, buffer, field);

316        /**
317         * Returns an integer hash code for the receiver. Objects which are equal
318         * return the same value for this method.
319         * 
320         * @return the receiver's hash.
321         * 
322         * @see #equals
323         * @since Android 1.0
324         */
325        @Override
326        public int hashCode() {
327            int hashCode = 0;
328            for (int i = 0; i < choiceLimits.length; i++) {
329                long v = Double.doubleToLongBits(choiceLimits[i]);
330                hashCode += (int) (v ^ (v >>> 32)) + choiceFormats[i].hashCode();
331            }
332            return hashCode;
333        }
334    
335        /**
336         * Returns the double value which is closest to the specified double but
337         * larger.
338         * 
339         * @param value
340         *            a double value.
341         * @return the next larger double value.
342         * @since Android 1.0
343         */
344        public static final double nextDouble(double value) {
345            if (value == Double.POSITIVE_INFINITY) {
346                return value;
347            }
348            long bits;
349            // Handle -0.0
350            if (value == 0) {
351                bits = 0;
352            } else {
353                bits = Double.doubleToLongBits(value);
354            }
355            return Double.longBitsToDouble(value < 0 ? bits - 1 : bits + 1);
356        }
357    
358        /**
359         * Returns the double value which is closest to the specified double but
360         * either larger or smaller as specified.
361         * 
362         * @param value
363         *            a double value.
364         * @param increment
365         *            {@code true} to get the next larger value, {@code false} to
366         *            get the previous smaller value.
367         * @return the next larger or smaller double value.
368         * @since Android 1.0
369         */
370        public static double nextDouble(double value, boolean increment) {
371            return increment ? nextDouble(value) : previousDouble(value);
372        }
373    
374        /**
375         * Parses a double from the specified string starting at the index specified
376         * by {@code position}. The string is compared to the strings of this
377         * {@code ChoiceFormat} and if a match occurs then the lower bound of the
378         * corresponding range in the limits array is returned. If the string is
379         * successfully parsed then the index of the {@code ParsePosition} passed to
380         * this method is updated to the index following the parsed text.
381         * 
382         * @param string
383         *            the source string to parse.
384         * @param position
385         *            input/output parameter, specifies the start index in {@code string}
386         *            from where to start parsing. See the <em>Returns</em> section for
387         *            a description of the output values.
388         * @return if one of the format strings of this {@code ChoiceFormat} instance
389         * is found in {@code string} starting at the index specified by {@code position.getIndex()} then
390         * <ul>
391         * <li>the index in {@code position} is set to the index following the parsed text;
392         * <li>the {@link java.lang.Double Double} corresponding to the format string is returned.</li>
393         * </ul>
394         * <p>
395         * If none of the format strings is found in {@code string} then
396         * <ul> 
397         * <li>the error index in {@code position} is set to the current index in {@code position};</li>
398         * <li> {@link java.lang.Double#NaN Double.NaN} is returned.
399         * </ul>
400         * @since Android 1.0
401         */
402        @Override
403        public Number parse(String string, ParsePosition position) {
404            int offset = position.getIndex();
405            for (int i = 0; i < choiceFormats.length; i++) {
406                if (string.startsWith(choiceFormats[i], offset)) {
407                    position.setIndex(offset + choiceFormats[i].length());
408                    return new Double(choiceLimits[i]);
409                }
410            }
411            position.setErrorIndex(offset);
412            return new Double(Double.NaN);
413        }
414    
415        /**
416         * Returns the double value which is closest to the specified double but
417         * smaller.
418         * 
419         * @param value
420         *            a double value.
421         * @return the next smaller double value.
422         * @since Android 1.0
423         */
424        public static final double previousDouble(double value) {
425            if (value == Double.NEGATIVE_INFINITY) {
426                return value;
427            }
428            long bits;
429            // Handle 0.0
430            if (value == 0) {
431                bits = 0x8000000000000000L;
432            } else {
433                bits = Double.doubleToLongBits(value);
434            }
435            return Double.longBitsToDouble(value <= 0 ? bits + 1 : bits - 1);
436        }

438        /**
439         * Sets the double values and associated strings of this ChoiceFormat. When
440         * calling {@link #format(double, StringBuffer, FieldPosition) format} with
441         * a double value {@code d}, then the element {@code i} in {@code formats}
442         * is selected where {@code i} fulfills 
443         * {@code limits[i] <= d < limits[i+1]}.
444         * <p>
445         * The length of the {@code limits} and {@code formats} arrays must be the
446         * same.
447         * </p>
448         * 
449         * @param limits
450         *            an array of doubles in ascending order. The lowest and highest
451         *            possible values are negative and positive infinity.
452         * @param formats
453         *            the strings associated with the ranges defined through {@code
454         *            limits}. The lower bound of the associated range is at the
455         *            same index as the string.
456         * @since Android 1.0
457         */
458        public void setChoices(double[] limits, String[] formats) {
459            if (limits.length != formats.length) {
460                throw new IllegalArgumentException();
461            }
462            choiceLimits = limits;
463            choiceFormats = formats;
464        }
465    
466        private int skipWhitespace(String string, int index) {
467            int length = string.length();
468            while (index < length && Character.isWhitespace(string.charAt(index))) {
469                index++;
470            }
471            return index;
472        }
473    
474        /**
475         * Returns the pattern of this {@code ChoiceFormat} which specifies the
476         * ranges and their associated strings.
477         * 
478         * @return the pattern.
479         * @since Android 1.0
480         */
481        public String toPattern() {
482            StringBuffer buffer = new StringBuffer();
483            for (int i = 0; i < choiceLimits.length; i++) {
484                if (i != 0) {
485                    buffer.append('|');
486                }
487                String previous = String.valueOf(previousDouble(choiceLimits[i]));
488                String limit = String.valueOf(choiceLimits[i]);
489                if (previous.length() < limit.length()) {
490                    buffer.append(previous);
491                    buffer.append('<');
492                } else {
493                    buffer.append(limit);
494                    buffer.append('#');
495                }
496                boolean quote = (choiceFormats[i].indexOf('|') != -1);
497                if (quote) {
498                    buffer.append('\'');
499                }
500                buffer.append(choiceFormats[i]);
501                if (quote) {
502                    buffer.append('\'');
503                }
504            }
505            return buffer.toString();
506        }
