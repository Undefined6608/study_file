048    /**
049     * A concrete class for formatting and parsing dates in a locale-sensitive
050     * manner. It allows for formatting (date to text), parsing (text to date) and
051     * normalization.
052     * <p>
053     * {@code SimpleDateFormat} allows you to start by choosing any user-defined
054     * patterns for date-time formatting. However, you are encouraged to create a
055     * date-time formatter with either {@code getTimeInstance}, {@code
056     * getDateInstance}, or {@code getDateTimeInstance} in {@code DateFormat}. Each
057     * of these class methods can return a date/time formatter initialized with a
058     * default format pattern. You may modify the format pattern using the {@code
059     * applyPattern} methods as desired. For more information on using these
060     * methods, see {@link DateFormat}.
061     * </p>
062     * <h4>Time Format Syntax</h4>
063     * <p>
064     * To specify the time format, use a <em>time pattern</em> string. In this
065     * pattern, all ASCII letters are reserved as pattern letters, which are defined
066     * as follows:

068     * <table border=0 cellspacing=3 cellpadding=0>
069     * <tr bgcolor="#ccccff">

071     * <th>Meaning</th>

073     * <th>Example</th>
074     * </tr>

210     * <td>(Literal)</td>
211     * <td>'o''clock'</td>
212     * </tr>
213     * </table>
214     * <p>
215     * The count of pattern letters determines the format:
216     * <p>
217     * <strong>(Text)</strong>: 4 or more pattern letters &rarr; use the full form,
218     * less than 4 pattern letters &rarr; use a short or abbreviated form if one
219     * exists.
220     * </p>
221     * <p>
222     * <strong>(Number)</strong>: the minimum number of digits. Shorter numbers are
223     * zero-padded to this amount. Year is handled specially; that is, if the count
224     * of 'y' is 2, the year will be truncated to 2 digits. (if "yyyy" produces
225     * "1997", "yy" produces "97".) Unlike other fields, fractional seconds are
226     * padded on the right with zero.
227     * <p>
228     * </p>
229     * <strong>(Text & Number)</strong>: 3 or over, use text, otherwise use number.
230     * <p>
231     * Any characters in the pattern that are not in the ranges of ['a'..'z'] and
232     * ['A'..'Z'] will be treated as quoted text. For instance, characters like ':',
233     * '.', ' ', '#' and '@' will appear in the resulting time text even they are
234     * not embraced within single quotes.
235     * </p>
236     * <p>
237     * A pattern containing any invalid pattern letter will result in an exception
238     * thrown during formatting or parsing.
239     * </p>
240     * <h4>Examples Using the US Locale</h4> <blockquote>
241     * 
242     * <pre>
243     * Format Pattern                       Result
244     * --------------                       -------
245     * "yyyy.MM.dd G 'at' HH:mm:ss vvvv" &rarr;  1996.07.10 AD at 15:08:56 Pacific Time
246     * "EEE, MMM d, ''yy"                &rarr;  Wed, July 10, '96
247     * "h:mm a"                          &rarr;  12:08 PM
248     * "hh 'o''clock' a, zzzz"           &rarr;  12 o'clock PM, Pacific Daylight Time
249     * "K:mm a, vvv"                     &rarr;  0:00 PM, PT
250     * "yyyyy.MMMMM.dd GGG hh:mm aaa"    &rarr;  01996.July.10 AD 12:08 PM
251     * </pre>
252     * 
253     * </blockquote> <h4>Code Sample:</h4> <blockquote>
254     * 
255     * <pre>
256     * SimpleTimeZone pdt = new SimpleTimeZone(-8 * 60 * 60 * 1000, "PST");
257     * pdt.setStartRule(Calendar.APRIL, 1, Calendar.SUNDAY, 2 * 60 * 60 * 1000);
258     * pdt.setEndRule(Calendar.OCTOBER, -1, Calendar.SUNDAY, 2 * 60 * 60 * 1000);
259     * 
260     * // Format the current time.
261     * SimpleDateFormat formatter = new SimpleDateFormat(
262     *         "yyyy.MM.dd G 'at' hh:mm:ss a zzz");
263     * Date currentTime_1 = new Date();
264     * String dateString = formatter.format(currentTime_1);
265     * 
266     * // Parse the previous string back into a Date.
267     * ParsePosition pos = new ParsePosition(0);
268     * Date currentTime_2 = formatter.parse(dateString, pos);
269     * </pre>
270     * 
271     * </blockquote>
272     * <p>
273     * In the example, the time value {@code currentTime_2} obtained from parsing
274     * will be equal to {@code currentTime_1}. However, they may not be equal if the
275     * am/pm marker 'a' is left out from the format pattern while the
276     * "hour in am/pm" pattern symbol is used. This information loss can happen when
277     * formatting the time in PM.
278     * </p>
279     * <p>
280     * When parsing a date string using the abbreviated year pattern ("yy"), {@code
281     * SimpleDateFormat} must interpret the abbreviated year relative to some
282     * century. It does this by adjusting dates to be within 80 years before and 20
283     * years after the time the {@code SimpleDateFormat} instance is created. For
284     * example, using a pattern of "MM/dd/yy" and a {@code SimpleDateFormat}
285     * instance created on Jan 1, 1997, the string "01/11/12" would be interpreted
286     * as Jan 11, 2012 while the string "05/04/64" would be interpreted as May 4,
287     * 1964. During parsing, only strings consisting of exactly two digits, as
288     * defined by {@link java.lang.Character#isDigit(char)}, will be parsed into the
289     * default century. Any other numeric string, such as a one digit string, a
290     * three or more digit string, or a two digit string that isn't all digits (for
291     * example, "-1"), is interpreted literally. So "01/02/3" or "01/02/003" are
292     * parsed, using the same pattern, as Jan 2, 3 AD. Likewise, "01/02/-3" is
293     * parsed as Jan 2, 4 BC.
294     * </p>
295     * <p>
296     * If the year pattern does not have exactly two 'y' characters, the year is
297     * interpreted literally, regardless of the number of digits. So using the
298     * pattern "MM/dd/yyyy", "01/11/12" parses to Jan 11, 12 A.D.
299     * </p>
300     * <p>
301     * When numeric fields are adjacent directly, with no intervening delimiter
302     * characters, they constitute a run of adjacent numeric fields. Such runs are
303     * parsed specially. For example, the format "HHmmss" parses the input text
304     * "123456" to 12:34:56, parses the input text "12345" to 1:23:45, and fails to
305     * parse "1234". In other words, the leftmost field of the run is flexible,
306     * while the others keep a fixed width. If the parse fails anywhere in the run,
307     * then the leftmost field is shortened by one character, and the entire run is
308     * parsed again. This is repeated until either the parse succeeds or the
309     * leftmost field is one character in length. If the parse still fails at that
310     * point, the parse of the run fails.
311     * </p>
312     * <p>
313     * For time zones that have no names, use the strings "GMT+hours:minutes" or
314     * "GMT-hours:minutes".
315     * </p>
316     * <p>
317     * The calendar defines the first day of the week, the first week of the year,
318     * whether hours are zero based or not (0 vs. 12 or 24) and the time zone. There
319     * is one common decimal format to handle all the numbers; the digit count is
320     * handled programmatically according to the pattern.
321     * <h4>Synchronization</h4> Date formats are not synchronized. It is recommended
322     * to create separate format instances for each thread. If multiple threads
323     * access a format concurrently, it must be synchronized externally.
324     * 
325     * @see Calendar
326     * @see GregorianCalendar
327     * @see TimeZone
328     * @see DateFormat
329     * @see DateFormatSymbols
330     * @see DecimalFormat
331     * @since Android 1.0
332     */

415        private SimpleDateFormat(Locale locale) {
416            numberFormat = NumberFormat.getInstance(locale);
417            numberFormat.setParseIntegerOnly(true);
418            numberFormat.setGroupingUsed(false);
419            calendar = new GregorianCalendar(locale);
420            calendar.add(Calendar.YEAR, -80);
421            creationYear = calendar.get(Calendar.YEAR);
422            defaultCenturyStart = calendar.getTime();
423        }

425        private void append(StringBuffer buffer, FieldPosition position,
426                Vector<FieldPosition> fields, char format, int count) {
427            int field = -1;
428            int index = patternChars.indexOf(format);
429            if (index == -1) {
430                // text.03=Unknown pattern character - '{0}'
431                throw new IllegalArgumentException(Messages.getString(
432                        "text.03", format)); //$NON-NLS-1$
433            }
434    
435            int beginPosition = buffer.length();
436            Field dateFormatField = null;
437    
438            switch (index) {
439                case ERA_FIELD:
440                    dateFormatField = Field.ERA;
441                    buffer.append(formatData.eras[calendar.get(Calendar.ERA)]);
442                    break;
443                case YEAR_FIELD:
444                    dateFormatField = Field.YEAR;
445                    int year = calendar.get(Calendar.YEAR);
446                    if (count < 4) {
447                        appendNumber(buffer, 2, year %= 100);
448                    } else {
449                        appendNumber(buffer, count, year);
450                    }
451                    break;
452                case MONTH_FIELD:
453                    dateFormatField = Field.MONTH;
454                    int month = calendar.get(Calendar.MONTH);
455                    if (count <= 2) {
456                        appendNumber(buffer, count, month + 1);
457                    } else if (count == 3) {
458                        buffer.append(formatData.shortMonths[month]);
459                    } else {
460                        buffer.append(formatData.months[month]);
461                    }
462                    break;
463                case DATE_FIELD:
464                    dateFormatField = Field.DAY_OF_MONTH;
465                    field = Calendar.DATE;
466                    break;
467                case HOUR_OF_DAY1_FIELD: // k
468                    dateFormatField = Field.HOUR_OF_DAY1;
469                    int hour = calendar.get(Calendar.HOUR_OF_DAY);
470                    appendNumber(buffer, count, hour == 0 ? 24 : hour);
471                    break;
472                case HOUR_OF_DAY0_FIELD: // H
473                    dateFormatField = Field.HOUR_OF_DAY0;
474                    field = Calendar.HOUR_OF_DAY;
475                    break;
476                case MINUTE_FIELD:
477                    dateFormatField = Field.MINUTE;
478                    field = Calendar.MINUTE;
479                    break;
480                case SECOND_FIELD:
481                    dateFormatField = Field.SECOND;
482                    field = Calendar.SECOND;
483                    break;
484                case MILLISECOND_FIELD:
485                    dateFormatField = Field.MILLISECOND;
486                    int value = calendar.get(Calendar.MILLISECOND);
487                    appendNumber(buffer, count, value);
488                    break;
489                case DAY_OF_WEEK_FIELD:
490                    dateFormatField = Field.DAY_OF_WEEK;
491                    int day = calendar.get(Calendar.DAY_OF_WEEK);
492                    if (count < 4) {
493                        buffer.append(formatData.shortWeekdays[day]);
494                    } else {
495                        buffer.append(formatData.weekdays[day]);
496                    }
497                    break;
498                case DAY_OF_YEAR_FIELD:
499                    dateFormatField = Field.DAY_OF_YEAR;
500                    field = Calendar.DAY_OF_YEAR;
501                    break;
502                case DAY_OF_WEEK_IN_MONTH_FIELD:
503                    dateFormatField = Field.DAY_OF_WEEK_IN_MONTH;
504                    field = Calendar.DAY_OF_WEEK_IN_MONTH;
505                    break;
506                case WEEK_OF_YEAR_FIELD:
507                    dateFormatField = Field.WEEK_OF_YEAR;
508                    field = Calendar.WEEK_OF_YEAR;
509                    break;
510                case WEEK_OF_MONTH_FIELD:
511                    dateFormatField = Field.WEEK_OF_MONTH;
512                    field = Calendar.WEEK_OF_MONTH;
513                    break;
514                case AM_PM_FIELD:
515                    dateFormatField = Field.AM_PM;
516                    buffer.append(formatData.ampms[calendar.get(Calendar.AM_PM)]);
517                    break;
518                case HOUR1_FIELD: // h
519                    dateFormatField = Field.HOUR1;
520                    hour = calendar.get(Calendar.HOUR);
521                    appendNumber(buffer, count, hour == 0 ? 12 : hour);
522                    break;
523                case HOUR0_FIELD: // K
524                    dateFormatField = Field.HOUR0;
525                    field = Calendar.HOUR;
526                    break;
527                case TIMEZONE_FIELD: // z
528                    dateFormatField = Field.TIME_ZONE;
529                    appendTimeZone(buffer, count, true);
530                    break;
531                case (TIMEZONE_FIELD + 1): // Z
532                    dateFormatField = Field.TIME_ZONE;
533                    appendTimeZone(buffer, count, false);
534                    break;
535            }
536            if (field != -1) {
537                appendNumber(buffer, count, calendar.get(field));
538            }
539    
540            if (fields != null) {
541                position = new FieldPosition(dateFormatField);
542                position.setBeginIndex(beginPosition);
543                position.setEndIndex(buffer.length());
544                fields.add(position);
545            } else {
546                // Set to the first occurrence
547                if ((position.getFieldAttribute() == dateFormatField || (position
548                        .getFieldAttribute() == null && position.getField() == index))
549                        && position.getEndIndex() == 0) {
550                    position.setBeginIndex(beginPosition);
551                    position.setEndIndex(buffer.length());
552                }
553            }
554        }

556        private void appendTimeZone(StringBuffer buffer, int count,
557                boolean generalTimezone) {
558            // cannot call TimeZone.getDisplayName() because it would not use
559            // the DateFormatSymbols of this SimpleDateFormat
560    
561            if (generalTimezone) {
562                String id = calendar.getTimeZone().getID();
563                // BEGIN android-changed
564                String[][] zones = formatData.internalZoneStrings();
565                // END android-changed
566                String[] zone = null;
567                for (String[] element : zones) {
568                    if (id.equals(element[0])) {
569                        zone = element;
570                        break;
571                    }
572                }
573                if (zone == null) {
574                    int offset = calendar.get(Calendar.ZONE_OFFSET)
575                            + calendar.get(Calendar.DST_OFFSET);
576                    char sign = '+';
577                    if (offset < 0) {
578                        sign = '-';
579                        offset = -offset;
580                    }
581                    buffer.append("GMT"); //$NON-NLS-1$
582                    buffer.append(sign);
583                    appendNumber(buffer, 2, offset / 3600000);
584                    buffer.append(':');
585                    appendNumber(buffer, 2, (offset % 3600000) / 60000);
586                } else {
587                    int daylight = calendar.get(Calendar.DST_OFFSET) == 0 ? 0 : 2;
588                    if (count < 4) {
589                        buffer.append(zone[2 + daylight]);
590                    } else {
591                        buffer.append(zone[1 + daylight]);
592                    }
593                }
594            } else {
595                int offset = calendar.get(Calendar.ZONE_OFFSET)
596                        + calendar.get(Calendar.DST_OFFSET);
597                char sign = '+';
598                if (offset < 0) {
599                    sign = '-';
600                    offset = -offset;
601                }
602                buffer.append(sign);
603                appendNumber(buffer, 2, offset / 3600000);
604                appendNumber(buffer, 2, (offset % 3600000) / 60000);
605            }
606        }
607    
608        private void appendNumber(StringBuffer buffer, int count, int value) {
609            int minimumIntegerDigits = numberFormat.getMinimumIntegerDigits();
610            numberFormat.setMinimumIntegerDigits(count);
611            numberFormat.format(new Integer(value), buffer, new FieldPosition(0));
612            numberFormat.setMinimumIntegerDigits(minimumIntegerDigits);
613        }

795        /**
796         * Validates the pattern.
797         * 
798         * @param template
799         *            the pattern to validate.
800         * 
801         * @throws NullPointerException
802         *             if the pattern is null
803         * @throws IllegalArgumentException
804         *             if the pattern is invalid
805         */
806        private void validatePattern(String template) {
807            boolean quote = false;
808            int next, last = -1, count = 0;
809    
810            final int patternLength = template.length();
811            for (int i = 0; i < patternLength; i++) {
812                next = (template.charAt(i));
813                if (next == '\'') {
814                    if (count > 0) {
815                        validateFormat((char) last);
816                        count = 0;
817                    }
818                    if (last == next) {
819                        last = -1;
820                    } else {
821                        last = next;
822                    }
823                    quote = !quote;
824                    continue;
825                }
826                if (!quote
827                        && (last == next || (next >= 'a' && next <= 'z') || (next >= 'A' && next <= 'Z'))) {
828                    if (last == next) {
829                        count++;
830                    } else {
831                        if (count > 0) {
832                            validateFormat((char) last);
833                        }
834                        last = next;
835                        count = 1;
836                    }
837                } else {
838                    if (count > 0) {
839                        validateFormat((char) last);
840                        count = 0;
841                    }
842                    last = -1;
843                }
844            }
845            if (count > 0) {
846                validateFormat((char) last);
847            }
848    
849            if (quote) {
850                // text.04=Unterminated quote {0}
851                throw new IllegalArgumentException(Messages.getString("text.04")); //$NON-NLS-1$
852            }
853    
854        }

856        /**
857         * Formats the date.
858         * <p>
859         * If the FieldPosition {@code field} is not null, and the field
860         * specified by this FieldPosition is formatted, set the begin and end index
861         * of the formatted field in the FieldPosition.
862         * <p>
863         * If the Vector {@code fields} is not null, find fields of this
864         * date, set FieldPositions with these fields, and add them to the fields
865         * vector.
866         * 
867         * @param date
868         *            Date to Format
869         * @param buffer
870         *            StringBuffer to store the resulting formatted String
871         * @param field
872         *            FieldPosition to set begin and end index of the field
873         *            specified, if it is part of the format for this date
874         * @param fields
875         *            Vector used to store the FieldPositions for each field in this
876         *            date
877         * 
878         * @return the formatted Date
879         * 
880         * @exception IllegalArgumentException
881         *                when the object cannot be formatted by this Format
882         */
883        private StringBuffer formatImpl(Date date, StringBuffer buffer,
884                FieldPosition field, Vector<FieldPosition> fields) {
885    
886            boolean quote = false;
887            int next, last = -1, count = 0;
888            calendar.setTime(date);
889            if (field != null) {
890                field.clear();
891            }
892    
893            final int patternLength = pattern.length();
894            for (int i = 0; i < patternLength; i++) {
895                next = (pattern.charAt(i));
896                if (next == '\'') {
897                    if (count > 0) {
898                        append(buffer, field, fields, (char) last, count);
899                        count = 0;
900                    }
901                    if (last == next) {
902                        buffer.append('\'');
903                        last = -1;
904                    } else {
905                        last = next;
906                    }
907                    quote = !quote;
908                    continue;
909                }
910                if (!quote
911                        && (last == next || (next >= 'a' && next <= 'z') || (next >= 'A' && next <= 'Z'))) {
912                    if (last == next) {
913                        count++;
914                    } else {
915                        if (count > 0) {
916                            append(buffer, field, fields, (char) last, count);
917                        }
918                        last = next;
919                        count = 1;
920                    }
921                } else {
922                    if (count > 0) {
923                        append(buffer, field, fields, (char) last, count);
924                        count = 0;
925                    }
926                    last = -1;
927                    buffer.append((char) next);
928                }
929            }
930            if (count > 0) {
931                append(buffer, field, fields, (char) last, count);
932            }
933            return buffer;
934        }

964        private int parse(String string, int offset, char format, int count) {
965            int index = patternChars.indexOf(format);
966            if (index == -1) {
967                // text.03=Unknown pattern character - '{0}'
968                throw new IllegalArgumentException(Messages.getString(
969                        "text.03", format)); //$NON-NLS-1$
970            }
971            int field = -1;
972            int absolute = 0;
973            if (count < 0) {
974                count = -count;
975                absolute = count;
976            }
977            switch (index) {
978                case ERA_FIELD:
979                    return parseText(string, offset, formatData.eras, Calendar.ERA);
980                case YEAR_FIELD:
981                    if (count >= 3) {
982                        field = Calendar.YEAR;
983                    } else {
984                        ParsePosition position = new ParsePosition(offset);
985                        Number result = parseNumber(absolute, string, position);
986                        if (result == null) {
987                            return -position.getErrorIndex() - 1;
988                        }
989                        int year = result.intValue();
990                        // A two digit year must be exactly two digits, i.e. 01
991                        if ((position.getIndex() - offset) == 2 && year >= 0) {
992                            year += creationYear / 100 * 100;
993                            if (year < creationYear) {
994                                year += 100;
995                            }
996                        }
997                        calendar.set(Calendar.YEAR, year);
998                        return position.getIndex();
999                    }
1000                    break;
1001                case MONTH_FIELD:
1002                    if (count <= 2) {
1003                        return parseNumber(absolute, string, offset,
1004                                Calendar.MONTH, -1);
1005                    }
1006                    index = parseText(string, offset, formatData.months,
1007                            Calendar.MONTH);
1008                    if (index < 0) {
1009                        return parseText(string, offset, formatData.shortMonths,
1010                                Calendar.MONTH);
1011                    }
1012                    return index;
1013                case DATE_FIELD:
1014                    field = Calendar.DATE;
1015                    break;
1016                case HOUR_OF_DAY1_FIELD:
1017                    ParsePosition position = new ParsePosition(offset);
1018                    Number result = parseNumber(absolute, string, position);
1019                    if (result == null) {
1020                        return -position.getErrorIndex() - 1;
1021                    }
1022                    int hour = result.intValue();
1023                    if (hour == 24) {
1024                        hour = 0;
1025                    }
1026                    calendar.set(Calendar.HOUR_OF_DAY, hour);
1027                    return position.getIndex();
1028                case HOUR_OF_DAY0_FIELD:
1029                    field = Calendar.HOUR_OF_DAY;
1030                    break;
1031                case MINUTE_FIELD:
1032                    field = Calendar.MINUTE;
1033                    break;
1034                case SECOND_FIELD:
1035                    field = Calendar.SECOND;
1036                    break;
1037                case MILLISECOND_FIELD:
1038                    field = Calendar.MILLISECOND;
1039                    break;
1040                case DAY_OF_WEEK_FIELD:
1041                    index = parseText(string, offset, formatData.weekdays,
1042                            Calendar.DAY_OF_WEEK);
1043                    if (index < 0) {
1044                        return parseText(string, offset, formatData.shortWeekdays,
1045                                Calendar.DAY_OF_WEEK);
1046                    }
1047                    return index;
1048                case DAY_OF_YEAR_FIELD:
1049                    field = Calendar.DAY_OF_YEAR;
1050                    break;
1051                case DAY_OF_WEEK_IN_MONTH_FIELD:
1052                    field = Calendar.DAY_OF_WEEK_IN_MONTH;
1053                    break;
1054                case WEEK_OF_YEAR_FIELD:
1055                    field = Calendar.WEEK_OF_YEAR;
1056                    break;
1057                case WEEK_OF_MONTH_FIELD:
1058                    field = Calendar.WEEK_OF_MONTH;
1059                    break;
1060                case AM_PM_FIELD:
1061                    return parseText(string, offset, formatData.ampms,
1062                            Calendar.AM_PM);
1063                case HOUR1_FIELD:
1064                    position = new ParsePosition(offset);
1065                    result = parseNumber(absolute, string, position);
1066                    if (result == null) {
1067                        return -position.getErrorIndex() - 1;
1068                    }
1069                    hour = result.intValue();
1070                    if (hour == 12) {
1071                        hour = 0;
1072                    }
1073                    calendar.set(Calendar.HOUR, hour);
1074                    return position.getIndex();
1075                case HOUR0_FIELD:
1076                    field = Calendar.HOUR;
1077                    break;
1078                case TIMEZONE_FIELD:
1079                    return parseTimeZone(string, offset);
1080                case (TIMEZONE_FIELD + 1):
1081                    return parseTimeZone(string, offset);
1082            }
1083            if (field != -1) {
1084                return parseNumber(absolute, string, offset, field, 0);
1085            }
1086            return offset;
1087        }

1089        /**
1090         * Parses a date from the specified string starting at the index specified
1091         * by {@code position}. If the string is successfully parsed then the index
1092         * of the {@code ParsePosition} is updated to the index following the parsed
1093         * text. On error, the index is unchanged and the error index of {@code
1094         * ParsePosition} is set to the index where the error occurred.
1095         * 
1096         * @param string
1097         *            the string to parse using the pattern of this simple date
1098         *            format.
1099         * @param position
1100         *            input/output parameter, specifies the start index in {@code
1101         *            string} from where to start parsing. If parsing is successful,
1102         *            it is updated with the index following the parsed text; on
1103         *            error, the index is unchanged and the error index is set to
1104         *            the index where the error occurred.
1105         * @return the date resulting from the parse, or {@code null} if there is an
1106         *         error.
1107         * @throws IllegalArgumentException
1108         *             if there are invalid characters in the pattern.
1109         * @since Android 1.0
1110         */
1111        @Override
1112        public Date parse(String string, ParsePosition position) {
1113            boolean quote = false;
1114            int next, last = -1, count = 0, offset = position.getIndex();
1115            int length = string.length();
1116            calendar.clear();
1117            TimeZone zone = calendar.getTimeZone();
1118            final int patternLength = pattern.length();
1119            for (int i = 0; i < patternLength; i++) {
1120                next = pattern.charAt(i);
1121                if (next == '\'') {
1122                    if (count > 0) {
1123                        if ((offset = parse(string, offset, (char) last, count)) < 0) {
1124                            return error(position, -offset - 1, zone);
1125                        }
1126                        count = 0;
1127                    }
1128                    if (last == next) {
1129                        if (offset >= length || string.charAt(offset) != '\'') {
1130                            return error(position, offset, zone);
1131                        }
1132                        offset++;
1133                        last = -1;
1134                    } else {
1135                        last = next;
1136                    }
1137                    quote = !quote;
1138                    continue;
1139                }
1140                if (!quote
1141                        && (last == next || (next >= 'a' && next <= 'z') || (next >= 'A' && next <= 'Z'))) {
1142                    if (last == next) {
1143                        count++;
1144                    } else {
1145                        if (count > 0) {
1146                            if ((offset = parse(string, offset, (char) last, -count)) < 0) {
1147                                return error(position, -offset - 1, zone);
1148                            }
1149                        }
1150                        last = next;
1151                        count = 1;
1152                    }
1153                } else {
1154                    if (count > 0) {
1155                        if ((offset = parse(string, offset, (char) last, count)) < 0) {
1156                            return error(position, -offset - 1, zone);
1157                        }
1158                        count = 0;
1159                    }
1160                    last = -1;
1161                    if (offset >= length || string.charAt(offset) != next) {
1162                        return error(position, offset, zone);
1163                    }
1164                    offset++;
1165                }
1166            }
1167            if (count > 0) {
1168                if ((offset = parse(string, offset, (char) last, count)) < 0) {
1169                    return error(position, -offset - 1, zone);
1170                }
1171            }
1172            Date date;
1173            try {
1174                date = calendar.getTime();
1175            } catch (IllegalArgumentException e) {
1176                return error(position, offset, zone);
1177            }
1178            position.setIndex(offset);
1179            calendar.setTimeZone(zone);
1180            return date;
1181        }

1183        private Number parseNumber(int max, String string, ParsePosition position) {
1184            int digit, length = string.length(), result = 0;
1185            int index = position.getIndex();
1186            if (max > 0 && max < length - index) {
1187                length = index + max;
1188            }
1189            while (index < length
1190                    && (string.charAt(index) == ' ' || string.charAt(index) == '\t')) {
1191                index++;
1192            }
1193            if (max == 0) {
1194                position.setIndex(index);
1195                return numberFormat.parse(string, position);
1196            }
1197    
1198            while (index < length
1199                    && (digit = Character.digit(string.charAt(index), 10)) != -1) {
1200                index++;
1201                result = result * 10 + digit;
1202            }
1203            if (index == position.getIndex()) {
1204                position.setErrorIndex(index);
1205                return null;
1206            }
1207            position.setIndex(index);
1208            return new Integer(result);
1209        }

1211        private int parseNumber(int max, String string, int offset, int field,
1212                int skew) {
1213            ParsePosition position = new ParsePosition(offset);
1214            Number result = parseNumber(max, string, position);
1215            if (result == null) {
1216                return -position.getErrorIndex() - 1;
1217            }
1218            calendar.set(field, result.intValue() + skew);
1219            return position.getIndex();
1220        }

1222        private int parseText(String string, int offset, String[] text, int field) {
1223            int found = -1;
1224            for (int i = 0; i < text.length; i++) {
1225                if (text[i].length() == 0) {
1226                    continue;
1227                }
1228                if (string
1229                        .regionMatches(true, offset, text[i], 0, text[i].length())) {
1230                    // Search for the longest match, in case some fields are subsets
1231                    if (found == -1 || text[i].length() > text[found].length()) {
1232                        found = i;
1233                    }
1234                }
1235            }
1236            if (found != -1) {
1237                calendar.set(field, found);
1238                return offset + text[found].length();
1239            }
1240            return -offset - 1;
1241        }

1243        private int parseTimeZone(String string, int offset) {
1244            // BEGIN android-changed
1245            String[][] zones = formatData.internalZoneStrings();
1246            // END android-changed
1247            boolean foundGMT = string.regionMatches(offset, "GMT", 0, 3); //$NON-NLS-1$
1248            if (foundGMT) {
1249                offset += 3;
1250            }
1251            char sign;
1252            if (offset < string.length()
1253                    && ((sign = string.charAt(offset)) == '+' || sign == '-')) {
1254                ParsePosition position = new ParsePosition(offset + 1);
1255                Number result = numberFormat.parse(string, position);
1256                if (result == null) {
1257                    return -position.getErrorIndex() - 1;
1258                }
1259                int hour = result.intValue();
1260                int raw = hour * 3600000;
1261                int index = position.getIndex();
1262                if (index < string.length() && string.charAt(index) == ':') {
1263                    position.setIndex(index + 1);
1264                    result = numberFormat.parse(string, position);
1265                    if (result == null) {
1266                        return -position.getErrorIndex() - 1;
1267                    }
1268                    int minute = result.intValue();
1269                    raw += minute * 60000;
1270                } else if (hour >= 24) {
1271                    raw = (hour / 100 * 3600000) + (hour % 100 * 60000);
1272                }
1273                if (sign == '-') {
1274                    raw = -raw;
1275                }
1276                calendar.setTimeZone(new SimpleTimeZone(raw, "")); //$NON-NLS-1$
1277                return position.getIndex();
1278            }
1279            if (foundGMT) {
1280                calendar.setTimeZone(TimeZone.getTimeZone("GMT")); //$NON-NLS-1$
1281                return offset;
1282            }
1283            for (String[] element : zones) {
1284                for (int j = 1; j < 5; j++) {
1285                    if (string.regionMatches(true, offset, element[j], 0,
1286                            element[j].length())) {
1287                        TimeZone zone = TimeZone.getTimeZone(element[0]);
1288                        if (zone == null) {
1289                            return -offset - 1;
1290                        }
1291                        int raw = zone.getRawOffset();
1292                        if (j >= 3 && zone.useDaylightTime()) {
1293                            raw += 3600000;
1294                        }
1295                        calendar.setTimeZone(new SimpleTimeZone(raw, "")); //$NON-NLS-1$
1296                        return offset + element[j].length();
1297                    }
1298                }
1299            }
1300            return -offset - 1;
1301        }
