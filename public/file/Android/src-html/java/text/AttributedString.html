033    /**
034     * Holds a string with attributes describing the characters of
035     * this string.
036     * 
037     * @since Android 1.0
038     */
039    public class AttributedString {
040    
041        String text;
042    
043        Map<AttributedCharacterIterator.Attribute, List<Range>> attributeMap;
044    
045        static class Range {
046            int start;
047    
048            int end;
049    
050            Object value;
051    
052            Range(int s, int e, Object v) {
053                start = s;
054                end = e;
055                value = v;
056            }
057        }

059        static class AttributedIterator implements AttributedCharacterIterator {
060    
061            private int begin, end, offset;
062    
063            private AttributedString attrString;
064    
065            private HashSet<Attribute> attributesAllowed;
066    
067            AttributedIterator(AttributedString attrString) {
068                this.attrString = attrString;
069                begin = 0;
070                end = attrString.text.length();
071                offset = 0;
072            }
073    
074            AttributedIterator(AttributedString attrString,
075                    AttributedCharacterIterator.Attribute[] attributes, int begin,
076                    int end) {
077                if (begin < 0 || end > attrString.text.length() || begin > end) {
078                    throw new IllegalArgumentException();
079                }
080                this.begin = begin;
081                this.end = end;
082                offset = begin;
083                this.attrString = attrString;
084                if (attributes != null) {
085                    HashSet<Attribute> set = new HashSet<Attribute>(
086                            (attributes.length * 4 / 3) + 1);
087                    for (int i = attributes.length; --i >= 0;) {
088                        set.add(attributes[i]);
089                    }
090                    attributesAllowed = set;
091                }
092            }

094            /**
095             * Returns a new {@code AttributedIterator} with the same source string,
096             * begin, end, and current index as this attributed iterator.
097             * 
098             * @return a shallow copy of this attributed iterator.
099             * @see java.lang.Cloneable
100             */
101            @Override
102            @SuppressWarnings("unchecked")
103            public Object clone() {
104                try {
105                    AttributedIterator clone = (AttributedIterator) super.clone();
106                    if (attributesAllowed != null) {
107                        clone.attributesAllowed = (HashSet<Attribute>) attributesAllowed
108                                .clone();
109                    }
110                    return clone;
111                } catch (CloneNotSupportedException e) {
112                    return null;
113                }
114            }

116            public char current() {
117                if (offset == end) {
118                    return DONE;
119                }
120                return attrString.text.charAt(offset);
121            }

123            public char first() {
124                if (begin == end) {
125                    return DONE;
126                }
127                offset = begin;
128                return attrString.text.charAt(offset);
129            }

158            private boolean inRange(Range range) {
159                if (!(range.value instanceof Annotation)) {
160                    return true;
161                }
162                return range.start >= begin && range.start < end
163                        && range.end > begin && range.end <= end;
164            }
165    
166            private boolean inRange(List<Range> ranges) {
167                Iterator<Range> it = ranges.iterator();
168                while (it.hasNext()) {
169                    Range range = it.next();
170                    if (range.start >= begin && range.start < end) {
171                        return !(range.value instanceof Annotation)
172                                || (range.end > begin && range.end <= end);
173                    } else if (range.end > begin && range.end <= end) {
174                        return !(range.value instanceof Annotation)
175                                || (range.start >= begin && range.start < end);
176                    }
177                }
178                return false;
179            }
180    
181            /**
182             * Returns a set of attributes present in the {@code AttributedString}.
183             * An empty set returned indicates that no attributes where defined
184             * 
185             * @return a set of attribute keys that may be empty.
186             */
187            public Set<AttributedIterator.Attribute> getAllAttributeKeys() {
188                if (begin == 0 && end == attrString.text.length()
189                        && attributesAllowed == null) {
190                    return attrString.attributeMap.keySet();
191                }
192    
193                Set<AttributedIterator.Attribute> result = new HashSet<Attribute>(
194                        (attrString.attributeMap.size() * 4 / 3) + 1);
195                Iterator<Map.Entry<Attribute, List<Range>>> it = attrString.attributeMap
196                        .entrySet().iterator();
197                while (it.hasNext()) {
198                    Map.Entry<Attribute, List<Range>> entry = it.next();
199                    if (attributesAllowed == null
200                            || attributesAllowed.contains(entry.getKey())) {
201                        List<Range> ranges = entry.getValue();
202                        if (inRange(ranges)) {
203                            result.add(entry.getKey());
204                        }
205                    }
206                }
207                return result;
208            }
209    
210            private Object currentValue(List<Range> ranges) {
211                Iterator<Range> it = ranges.iterator();
212                while (it.hasNext()) {
213                    Range range = it.next();
214                    if (offset >= range.start && offset < range.end) {
215                        return inRange(range) ? range.value : null;
216                    }
217                }
218                return null;
219            }
220    
221            public Object getAttribute(
222                    AttributedCharacterIterator.Attribute attribute) {
223                if (attributesAllowed != null
224                        && !attributesAllowed.contains(attribute)) {
225                    return null;
226                }
227                ArrayList<Range> ranges = (ArrayList<Range>) attrString.attributeMap
228                        .get(attribute);
229                if (ranges == null) {
230                    return null;
231                }
232                return currentValue(ranges);
233            }
234    
235            public Map<Attribute, Object> getAttributes() {
236                Map<Attribute, Object> result = new HashMap<Attribute, Object>(
237                        (attrString.attributeMap.size() * 4 / 3) + 1);
238                Iterator<Map.Entry<Attribute, List<Range>>> it = attrString.attributeMap
239                        .entrySet().iterator();
240                while (it.hasNext()) {
241                    Map.Entry<Attribute, List<Range>> entry = it.next();
242                    if (attributesAllowed == null
243                            || attributesAllowed.contains(entry.getKey())) {
244                        Object value = currentValue(entry.getValue());
245                        if (value != null) {
246                            result.put(entry.getKey(), value);
247                        }
248                    }
249                }
250                return result;
251            }
252    
253            public int getRunLimit() {
254                return getRunLimit(getAllAttributeKeys());
255            }
256    
257            private int runLimit(List<Range> ranges) {
258                int result = end;
259                ListIterator<Range> it = ranges.listIterator(ranges.size());
260                while (it.hasPrevious()) {
261                    Range range = it.previous();
262                    if (range.end <= begin) {
263                        break;
264                    }
265                    if (offset >= range.start && offset < range.end) {
266                        return inRange(range) ? range.end : result;
267                    } else if (offset >= range.end) {
268                        break;
269                    }
270                    result = range.start;
271                }
272                return result;
273            }
274    
275            public int getRunLimit(AttributedCharacterIterator.Attribute attribute) {
276                if (attributesAllowed != null
277                        && !attributesAllowed.contains(attribute)) {
278                    return end;
279                }
280                ArrayList<Range> ranges = (ArrayList<Range>) attrString.attributeMap
281                        .get(attribute);
282                if (ranges == null) {
283                    return end;
284                }
285                return runLimit(ranges);
286            }
287    
288            public int getRunLimit(Set<? extends Attribute> attributes) {
289                int limit = end;
290                Iterator<? extends Attribute> it = attributes.iterator();
291                while (it.hasNext()) {
292                    AttributedCharacterIterator.Attribute attribute = it.next();
293                    int newLimit = getRunLimit(attribute);
294                    if (newLimit < limit) {
295                        limit = newLimit;
296                    }
297                }
298                return limit;
299            }
300    
301            public int getRunStart() {
302                return getRunStart(getAllAttributeKeys());
303            }
304    
305            private int runStart(List<Range> ranges) {
306                int result = begin;
307                Iterator<Range> it = ranges.iterator();
308                while (it.hasNext()) {
309                    Range range = it.next();
310                    if (range.start >= end) {
311                        break;
312                    }
313                    if (offset >= range.start && offset < range.end) {
314                        return inRange(range) ? range.start : result;
315                    } else if (offset < range.start) {
316                        break;
317                    }
318                    result = range.end;
319                }
320                return result;
321            }
322    
323            public int getRunStart(AttributedCharacterIterator.Attribute attribute) {
324                if (attributesAllowed != null
325                        && !attributesAllowed.contains(attribute)) {
326                    return begin;
327                }
328                ArrayList<Range> ranges = (ArrayList<Range>) attrString.attributeMap
329                        .get(attribute);
330                if (ranges == null) {
331                    return begin;
332                }
333                return runStart(ranges);
334            }
335    
336            public int getRunStart(Set<? extends Attribute> attributes) {
337                int start = begin;
338                Iterator<? extends Attribute> it = attributes.iterator();
339                while (it.hasNext()) {
340                    AttributedCharacterIterator.Attribute attribute = it.next();
341                    int newStart = getRunStart(attribute);
342                    if (newStart > start) {
343                        start = newStart;
344                    }
345                }
346                return start;
347            }
348    
349            public char last() {
350                if (begin == end) {
351                    return DONE;
352                }
353                offset = end - 1;
354                return attrString.text.charAt(offset);
355            }
356    
357            public char next() {
358                if (offset >= (end - 1)) {
359                    offset = end;
360                    return DONE;
361                }
362                return attrString.text.charAt(++offset);
363            }
364    
365            public char previous() {
366                if (offset == begin) {
367                    return DONE;
368                }
369                return attrString.text.charAt(--offset);
370            }
371    
372            public char setIndex(int location) {
373                if (location < begin || location > end) {
374                    throw new IllegalArgumentException();
375                }
376                offset = location;
377                if (offset == end) {
378                    return DONE;
379                }
380                return attrString.text.charAt(offset);
381            }
382        }

384        /**
385         * Constructs an {@code AttributedString} from an {@code
386         * AttributedCharacterIterator}, which represents attributed text.
387         * 
388         * @param iterator
389         *            the {@code AttributedCharacterIterator} that contains the text
390         *            for this attributed string.
391         * @since Android 1.0
392         */
393        public AttributedString(AttributedCharacterIterator iterator) {
394            if (iterator.getBeginIndex() > iterator.getEndIndex()) {
395                // text.0A=Invalid substring range
396                throw new IllegalArgumentException(Messages.getString("text.0A")); //$NON-NLS-1$
397            }
398            StringBuffer buffer = new StringBuffer();
399            for (int i = iterator.getBeginIndex(); i < iterator.getEndIndex(); i++) {
400                buffer.append(iterator.current());
401                iterator.next();
402            }
403            text = buffer.toString();
404            Set<AttributedCharacterIterator.Attribute> attributes = iterator
405                    .getAllAttributeKeys();
406            if (attributes == null) {
407                return;
408            }
409            attributeMap = new HashMap<Attribute, List<Range>>(
410                    (attributes.size() * 4 / 3) + 1);
411    
412            Iterator<Attribute> it = attributes.iterator();
413            while (it.hasNext()) {
414                AttributedCharacterIterator.Attribute attribute = it.next();
415                iterator.setIndex(0);
416                while (iterator.current() != CharacterIterator.DONE) {
417                    int start = iterator.getRunStart(attribute);
418                    int limit = iterator.getRunLimit(attribute);
419                    Object value = iterator.getAttribute(attribute);
420                    if (value != null) {
421                        addAttribute(attribute, value, start, limit);
422                    }
423                    iterator.setIndex(limit);
424                }
425            }
426        }
427    
428        private AttributedString(AttributedCharacterIterator iterator, int start,
429                int end, Set<Attribute> attributes) {
430            if (start < iterator.getBeginIndex() || end > iterator.getEndIndex()
431                    || start > end) {
432                throw new IllegalArgumentException();
433            }
434    
435            if (attributes == null) {
436                return;
437            }
438    
439            StringBuffer buffer = new StringBuffer();
440            iterator.setIndex(start);
441            while (iterator.getIndex() < end) {
442                buffer.append(iterator.current());
443                iterator.next();
444            }
445            text = buffer.toString();
446            attributeMap = new HashMap<Attribute, List<Range>>(
447                    (attributes.size() * 4 / 3) + 1);
448    
449            Iterator<Attribute> it = attributes.iterator();
450            while (it.hasNext()) {
451                AttributedCharacterIterator.Attribute attribute = it.next();
452                iterator.setIndex(start);
453                while (iterator.getIndex() < end) {
454                    Object value = iterator.getAttribute(attribute);
455                    int runStart = iterator.getRunStart(attribute);
456                    int limit = iterator.getRunLimit(attribute);
457                    if ((value instanceof Annotation && runStart >= start && limit <= end)
458                            || (value != null && !(value instanceof Annotation))) {
459                        addAttribute(attribute, value, (runStart < start ? start
460                                : runStart)
461                                - start, (limit > end ? end : limit) - start);
462                    }
463                    iterator.setIndex(limit);
464                }
465            }
466        }

510         *             if {@code start} is less than first index of

529         * Creates an {@code AttributedString} from the given text.
530         * 
531         * @param value
532         *            the text to take as base for this attributed string.
533         * @since Android 1.0
534         */
535        public AttributedString(String value) {
536            if (value == null) {
537                throw new NullPointerException();
538            }
539            text = value;
540            attributeMap = new HashMap<Attribute, List<Range>>(11);
541        }

543        /**
544         * Creates an {@code AttributedString} from the given text and the
545         * attributes. The whole text has the given attributes applied.
546         * 
547         * @param value
548         *            the text to take as base for this attributed string.
549         * @param attributes
550         *            the attributes that the text is associated with.
551         * @throws IllegalArgumentException
552         *             if the length of {@code value} is 0 but the size of {@code
553         *             attributes} is greater than 0.
554         * @throws NullPointerException
555         *             if {@code value} is {@code null}.
556         * @since Android 1.0
557         */
558        public AttributedString(String value,
559                Map<? extends AttributedCharacterIterator.Attribute, ?> attributes) {
560            if (value == null) {
561                throw new NullPointerException();
562            }
563            if (value.length() == 0 && !attributes.isEmpty()) {
564                // text.0B=Cannot add attributes to empty string
565                throw new IllegalArgumentException(Messages.getString("text.0B")); //$NON-NLS-1$
566            }
567            text = value;
568            attributeMap = new HashMap<Attribute, List<Range>>(
569                    (attributes.size() * 4 / 3) + 1);
570            Iterator<?> it = attributes.entrySet().iterator();
571            while (it.hasNext()) {
572                Map.Entry<?, ?> entry = (Map.Entry<?, ?>) it.next();
573                ArrayList<Range> ranges = new ArrayList<Range>(1);
574                ranges.add(new Range(0, text.length(), entry.getValue()));
575                attributeMap.put((AttributedCharacterIterator.Attribute) entry
576                        .getKey(), ranges);
577            }
578        }

580        /**
581         * Applies a given attribute to this string.
582         * 
583         * @param attribute
584         *            the attribute that will be applied to this string.
585         * @param value
586         *            the value of the attribute that will be applied to this
587         *            string.
588         * @throws IllegalArgumentException
589         *             if the length of this attributed string is 0.
590         * @throws NullPointerException
591         *             if {@code attribute} is {@code null}.
592         * @since Android 1.0
593         */
594        public void addAttribute(AttributedCharacterIterator.Attribute attribute,
595                Object value) {
596            if (null == attribute) {
597                throw new NullPointerException();
598            }
599            if (text.length() == 0) {
600                throw new IllegalArgumentException();
601            }
602    
603            List<Range> ranges = attributeMap.get(attribute);
604            if (ranges == null) {
605                ranges = new ArrayList<Range>(1);
606                attributeMap.put(attribute, ranges);
607            } else {
608                ranges.clear();
609            }
610            ranges.add(new Range(0, text.length(), value));
611        }

613        /**
614         * Applies a given attribute to the given range of this string.
615         * 
616         * @param attribute
617         *            the attribute that will be applied to this string.
618         * @param value
619         *            the value of the attribute that will be applied to this
620         *            string.
621         * @param start
622         *            the start of the range where the attribute will be applied.
623         * @param end
624         *            the end of the range where the attribute will be applied.
625         * @throws IllegalArgumentException
626         *             if {@code start < 0}, {@code end} is greater than the length
627         *             of this string, or if {@code start >= end}.
628         * @throws NullPointerException
629         *             if {@code attribute} is {@code null}.
630         * @since Android 1.0
631         */
632        public void addAttribute(AttributedCharacterIterator.Attribute attribute,
633                Object value, int start, int end) {
634            if (null == attribute) {
635                throw new NullPointerException();
636            }
637            if (start < 0 || end > text.length() || start >= end) {
638                throw new IllegalArgumentException();
639            }
640    
641            if (value == null) {
642                return;
643            }
644    
645            List<Range> ranges = attributeMap.get(attribute);
646            if (ranges == null) {
647                ranges = new ArrayList<Range>(1);
648                ranges.add(new Range(start, end, value));
649                attributeMap.put(attribute, ranges);
650                return;
651            }
652            ListIterator<Range> it = ranges.listIterator();
653            // BEGIN android-changed
654            // copied from a newer version of harmony
655            // value can't be null
656            while (it.hasNext()) {
657                Range range = it.next();
658                if (end <= range.start) {
659                    it.previous();
660                    break;
661                } else if (start < range.end
662                        || (start == range.end && value.equals(range.value))) {
663                    Range r1 = null, r3;
664                    it.remove();
665                    r1 = new Range(range.start, start, range.value);
666                    r3 = new Range(end, range.end, range.value);
667    
668                    while (end > range.end && it.hasNext()) {
669                        range = it.next();
670                        if (end <= range.end) {
671                            if (end > range.start
672                                    || (end == range.start && value.equals(range.value))) {
673                                it.remove();
674                                r3 = new Range(end, range.end, range.value);
675                                break;
676                            }
677                        } else {
678                            it.remove();
679                        }
680                    }
681    
682                    if (value.equals(r1.value)) {
683                        if (value.equals(r3.value)) {
684                            it.add(new Range(r1.start < start ? r1.start : start,
685                                    r3.end > end ? r3.end : end, r1.value));
686                        } else {
687                            it.add(new Range(r1.start < start ? r1.start : start,
688                                    end, r1.value));
689                            if (r3.start < r3.end) {
690                                it.add(r3);
691                            }
692                        }
693                    } else {
694                        if (value.equals(r3.value)) {
695                            if (r1.start < r1.end) {
696                                it.add(r1);
697                            }
698                            it.add(new Range(start, r3.end > end ? r3.end : end,
699                                    r3.value));
700                        } else {
701                            if (r1.start < r1.end) {
702                                it.add(r1);
703                            }
704                            it.add(new Range(start, end, value));
705                            if (r3.start < r3.end) {
706                                it.add(r3);
707                            }
708                        }
709                    }
710                    return;
711                }
712            }
713            // END android-changed
714            it.add(new Range(start, end, value));
715        }

717        /**
718         * Applies a given set of attributes to the given range of the string.
719         * 
720         * @param attributes
721         *            the set of attributes that will be applied to this string.
722         * @param start
723         *            the start of the range where the attribute will be applied.
724         * @param end
725         *            the end of the range where the attribute will be applied.
726         * @throws IllegalArgumentException
727         *             if {@code start < 0}, {@code end} is greater than the length
728         *             of this string, or if {@code start >= end}.
729         * @since Android 1.0
730         */
731        public void addAttributes(
732                Map<? extends AttributedCharacterIterator.Attribute, ?> attributes,
733                int start, int end) {
734            Iterator<?> it = attributes.entrySet().iterator();
735            while (it.hasNext()) {
736                Map.Entry<?, ?> entry = (Map.Entry<?, ?>) it.next();
737                addAttribute(
738                        (AttributedCharacterIterator.Attribute) entry.getKey(),
739                        entry.getValue(), start, end);
740            }
741        }
