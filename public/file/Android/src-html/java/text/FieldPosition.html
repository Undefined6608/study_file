018    package java.text;
019    
020    /**
021     * Identifies fields in formatted strings. If a {@code FieldPosition} is passed
022     * to the format method with such a parameter, then the indices will be set to
023     * the start and end indices of the field in the formatted string.
024     * <p>
025     * A {@code FieldPosition} can be created by using the integer constants in the
026     * various format classes (for example {@code NumberFormat.INTEGER_FIELD}) or
027     * one of the fields of type {@code Format.Field}.
028     * </p>
029     * <p>
030     * If more than one field information is needed, the method
031     * {@link NumberFormat#formatToCharacterIterator(Object)} should be used.
032     * </p>
033     * 
034     * @since Android 1.0
035     */
036    public class FieldPosition {
037    
038        private int myField, beginIndex, endIndex;
039    
040        private Format.Field myAttribute;
041    
042        /**
043         * Constructs a new {@code FieldPosition} for the specified field.
044         * 
045         * @param field
046         *            the field to identify.
047         * @since Android 1.0
048         */
049        public FieldPosition(int field) {
050            myField = field;
051        }
052    
053        /**
054         * Constructs a new {@code FieldPosition} for the specified {@code Field}
055         * attribute.
056         * 
057         * @param attribute
058         *            the field attribute to identify.
059         * @since Android 1.0
060         */
061        public FieldPosition(Format.Field attribute) {
062            myAttribute = attribute;
063            myField = -1;
064        }
065    
066        /**
067         * Constructs a new {@code FieldPosition} for the specified {@code Field}
068         * attribute and field id.
069         * 
070         * @param attribute
071         *            the field attribute to identify.
072         * @param field
073         *            the field to identify.
074         * @since Android 1.0
075         */
076        public FieldPosition(Format.Field attribute, int field) {
077            myAttribute = attribute;
078            myField = field;
079        }
080    
081        void clear() {
082            beginIndex = endIndex = 0;
083        }
084    
085        /**
086         * Compares the specified object to this field position and indicates if
087         * they are equal. In order to be equal, {@code object} must be an instance
088         * of {@code FieldPosition} with the same field, begin index and end index.
089         * 
090         * @param object
091         *            the object to compare with this object.
092         * @return {@code true} if the specified object is equal to this field
093         *         position; {@code false} otherwise.
094         * @see #hashCode
095         * @since Android 1.0
096         */
097        @Override
098        public boolean equals(Object object) {
099            if (!(object instanceof FieldPosition)) {
100                return false;
101            }
102            FieldPosition pos = (FieldPosition) object;
103            return myField == pos.myField && myAttribute == pos.myAttribute
104                    && beginIndex == pos.beginIndex && endIndex == pos.endIndex;
105        }
106    
107        /**
108         * Returns the index of the beginning of the field.
109         * 
110         * @return the first index of the field.
111         * @since Android 1.0
112         */
113        public int getBeginIndex() {
114            return beginIndex;
115        }
116    
117        /**
118         * Returns the index one past the end of the field.
119         * 
120         * @return one past the index of the last character in the field.
121         * @since Android 1.0
122         */
123        public int getEndIndex() {
124            return endIndex;
125        }
126    
127        /**
128         * Returns the field which is being identified.
129         * 
130         * @return the field constant.
131         * @since Android 1.0
132         */
133        public int getField() {
134            return myField;
135        }
136    
137        /**
138         * Returns the attribute which is being identified.
139         * 
140         * @return the field.
141         * @since Android 1.0
142         */
143        public Format.Field getFieldAttribute() {
144            return myAttribute;
145        }
146    
147        @Override
148        public int hashCode() {
149            int attributeHash = (myAttribute == null) ? 0 : myAttribute.hashCode();
150            return attributeHash + myField * 10 + beginIndex * 100 + endIndex;
151        }
152    
153        /**
154         * Sets the index of the beginning of the field.
155         * 
156         * @param index
157         *            the index of the first character in the field.
158         * @since Android 1.0
159         */
160        public void setBeginIndex(int index) {
161            beginIndex = index;
162        }
163    
164        /**
165         * Sets the index of the end of the field.
166         * 
167         * @param index
168         *            one past the index of the last character in the field.
169         * @since Android 1.0
170         */
171        public void setEndIndex(int index) {
172            endIndex = index;
173        }
174    
175        /**
176         * Returns the string representation of this field position.
177         * 
178         * @return the string representation of this field position.
179         * @since Android 1.0
180         */
181        @Override
182        public String toString() {
183            return getClass().getName() + "[attribute=" + myAttribute + ", field=" //$NON-NLS-1$ //$NON-NLS-2$
184                    + myField + ", beginIndex=" + beginIndex + ", endIndex=" //$NON-NLS-1$ //$NON-NLS-2$
185                    + endIndex + "]"; //$NON-NLS-1$
186        }
187    }
