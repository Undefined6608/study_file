040    /**
041     * The base class for all formats.
042     * <p>
043     * This is an abstract base class which specifies the protocol for classes which
044     * convert other objects or values, such as numeric values and dates, and their
045     * string representations. In some cases these representations may be localized
046     * or contain localized characters or strings. For example, a numeric formatter
047     * such as {@code DecimalFormat} may convert a numeric value such as 12345 to
048     * the string "$12,345". It may also parse the string back into a numeric value.
049     * A date and time formatter like {@code SimpleDateFormat} may represent a
050     * specific date, encoded numerically, as a string such as "Wednesday, February
051     * 26, 1997 AD".
052     * </p>
053     * <p>
054     * Many of the concrete subclasses of {@code Format} employ the notion of a
055     * pattern. A pattern is a string representation of the rules which govern the
056     * conversion between values and strings. For example, a {@code DecimalFormat}
057     * object may be associated with the pattern "$#,##0.00;($#,##0.00)", which is a
058     * common US English format for currency values, yielding strings such as
059     * "$1,234.45" for 1234.45, and "($987.65)" for -987.6543. The specific syntax
060     * of a pattern is defined by each subclass. Even though many subclasses use
061     * patterns, the notion of a pattern is not inherent to {@code Format} classes
062     * in general, and is not part of the explicit base class protocol.
063     * </p>
064     * <p>
065     * Two complex formatting classes are worth mentioning: {@code MessageFormat}
066     * and {@code ChoiceFormat}. {@code ChoiceFormat} is a subclass of
067     * {@code NumberFormat} which allows the user to format different number ranges
068     * as strings. For instance, 0 may be represented as "no files", 1 as "one
069     * file", and any number greater than 1 as "many files". {@code MessageFormat}
070     * is a formatter which utilizes other {@code Format} objects to format a string
071     * containing multiple values. For instance, a {@code MessageFormat} object
072     * might produce the string "There are no files on the disk MyDisk on February
073     * 27, 1997." given the arguments 0, "MyDisk", and the date value of 2/27/97.
074     * See the {@link ChoiceFormat} and {@link MessageFormat} descriptions for
075     * further information.
076     * </p>
077     * 
078     * @since Android 1.0
079     */
080    public abstract class Format implements Serializable, Cloneable {

120        String convertPattern(String template, String fromChars, String toChars,
121                boolean check) {
122            if (!check && fromChars.equals(toChars)) {
123                return template;
124            }
125            boolean quote = false;
126            StringBuilder output = new StringBuilder();
127            int length = template.length();
128            for (int i = 0; i < length; i++) {
129                int index;
130                char next = template.charAt(i);
131                if (next == '\'') {
132                    quote = !quote;
133                }
134                if (!quote && (index = fromChars.indexOf(next)) != -1) {
135                    output.append(toChars.charAt(index));
136                } else if (check
137                        && !quote
138                        && ((next >= 'a' && next <= 'z') || (next >= 'A' && next <= 'Z'))) {
139                    // text.05=Invalid pattern char {0} in {1}
140                    throw new IllegalArgumentException(Messages.getString(
141                            "text.05", String.valueOf(next), template)); //$NON-NLS-1$
142                } else {
143                    output.append(next);
144                }
145            }
146            if (quote) {
147                // text.04=Unterminated quote
148                throw new IllegalArgumentException(Messages.getString("text.04")); //$NON-NLS-1$
149            }
150            return output.toString();
151        }

214        /**
215         * Parses the specified string using the rules of this format.
216         * 
217         * @param string
218         *            the string to parse.
219         * @return the object resulting from the parse.
220         * @exception ParseException
221         *                if an error occurs during parsing.
222         * @since Android 1.0

224        public Object parseObject(String string) throws ParseException {
225            ParsePosition position = new ParsePosition(0);
226            Object result = parseObject(string, position);
227            if (position.getErrorIndex() != -1 || position.getIndex() == 0) {
228                throw new ParseException(null, position.getErrorIndex());
229            }
230            return result;
231        }

234         * Parses the specified string starting at the index specified by

236         * the {@code ParsePosition} is updated to the index following the parsed
237         * text. On error, the index is unchanged and the error index of
238         * {@code ParsePosition} is set to the index where the error occurred.

254        static boolean upTo(String string, ParsePosition position,
255                StringBuffer buffer, char stop) {
256            int index = position.getIndex(), length = string.length();
257            boolean lastQuote = false, quote = false;
258            while (index < length) {
259                char ch = string.charAt(index++);
260                if (ch == '\'') {
261                    if (lastQuote) {
262                        buffer.append('\'');
263                    }
264                    quote = !quote;
265                    lastQuote = true;
266                } else if (ch == stop && !quote) {
267                    position.setIndex(index);
268                    return true;
269                } else {
270                    lastQuote = false;
271                    buffer.append(ch);
272                }
273            }
274            position.setIndex(index);
275            return false;
276        }

278        static boolean upToWithQuotes(String string, ParsePosition position,
279                StringBuffer buffer, char stop, char start) {
280            int index = position.getIndex(), length = string.length(), count = 1;
281            boolean quote = false;
282            while (index < length) {
283                char ch = string.charAt(index++);
284                if (ch == '\'') {
285                    quote = !quote;
286                }
287                if (!quote) {
288                    if (ch == stop) {
289                        count--;
290                    }
291                    if (count == 0) {
292                        position.setIndex(index);
293                        return true;
294                    }
295                    if (ch == start) {
296                        count++;
297                    }
298                }
299                buffer.append(ch);
300            }
301            // text.07=Unmatched braces in the pattern
302            throw new IllegalArgumentException(Messages.getString("text.07")); //$NON-NLS-1$
303        }
