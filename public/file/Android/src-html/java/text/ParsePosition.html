018    package java.text;
019    
020    /**
021     * Tracks the current position in a parsed string. In case of an error the error
022     * index can be set to the position where the error occurred without having to
023     * change the parse position.
024     * 
025     * @since Android 1.0
026     */
027    public class ParsePosition {
028    
029        private int currentPosition, errorIndex = -1;
030    
031        /**
032         * Constructs a new {@code ParsePosition} with the specified index.
033         * 
034         * @param index
035         *            the index to begin parsing.
036         * @since Android 1.0
037         */
038        public ParsePosition(int index) {
039            currentPosition = index;
040        }
041    
042        /**
043         * Compares the specified object to this {@code ParsePosition} and indicates
044         * if they are equal. In order to be equal, {@code object} must be an
045         * instance of {@code ParsePosition} and it must have the same index and
046         * error index.
047         * 
048         * @param object
049         *            the object to compare with this object.
050         * @return {@code true} if the specified object is equal to this
051         *         {@code ParsePosition}; {@code false} otherwise.
052         * @see #hashCode
053         * @since Android 1.0
054         */
055        @Override
056        public boolean equals(Object object) {
057            if (!(object instanceof ParsePosition)) {
058                return false;
059            }
060            ParsePosition pos = (ParsePosition) object;
061            return currentPosition == pos.currentPosition
062                    && errorIndex == pos.errorIndex;
063        }
064    
065        /**
066         * Returns the index at which the parse could not continue.
067         * 
068         * @return the index of the parse error or -1 if there is no error.
069         * @since Android 1.0
070         */
071        public int getErrorIndex() {
072            return errorIndex;
073        }
074    
075        /**
076         * Returns the current parse position.
077         * 
078         * @return the current position.
079         * @since Android 1.0
080         */
081        public int getIndex() {
082            return currentPosition;
083        }
084    
085        @Override
086        public int hashCode() {
087            return currentPosition + errorIndex;
088        }
089    
090        /**
091         * Sets the index at which the parse could not continue.
092         * 
093         * @param index
094         *            the index of the parse error.
095         * @since Android 1.0
096         */
097        public void setErrorIndex(int index) {
098            errorIndex = index;
099        }
100    
101        /**
102         * Sets the current parse position.
103         * 
104         * @param index
105         *            the current parse position.
106         * @since Android 1.0
107         */
108        public void setIndex(int index) {
109            currentPosition = index;
110        }
111    
112        /**
113         * Returns the string representation of this parse position.
114         * 
115         * @return the string representation of this parse position.
116         * @since Android 1.0
117         */
118        @Override
119        public String toString() {
120            return getClass().getName() + "[index=" + currentPosition //$NON-NLS-1$
121                    + ", errorIndex=" + errorIndex + "]"; //$NON-NLS-1$ //$NON-NLS-2$
122        }
123    }
