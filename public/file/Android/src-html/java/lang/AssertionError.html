018    package java.lang;
019    
020    /**
021     * Thrown when an assertion has failed.
022     * 
023     * @since Android 1.0
024     */
025    public class AssertionError extends Error {
026    
027        private static final long serialVersionUID = -5013299493970297370L;
028    
029        /**
030         * Constructs a new {@code AssertionError} with no message.
031         * 
032         * @since Android 1.0
033         */
034        public AssertionError() {
035            super();
036        }
037    
038        /**
039         * Constructs a new {@code AssertionError} with a message based on calling
040         * {@link String#valueOf(Object)} with the specified object. If the object
041         * is an instance of {@link Throwable}, then it also becomes the cause of
042         * this error.
043         * 
044         * @param detailMessage
045         *            the object to be converted into the detail message and
046         *            optionally the cause.
047         * @since Android 1.0
048         */
049        public AssertionError(Object detailMessage) {
050            super(String.valueOf(detailMessage),
051                    (detailMessage instanceof Throwable ? (Throwable) detailMessage
052                            : null));
053        }
054    
055        /**
056         * Constructs a new {@code AssertionError} with a message based on calling
057         * {@link String#valueOf(boolean)} with the specified boolean value.
058         * 
059         * @param detailMessage
060         *            the value to be converted into the message.
061         * @since Android 1.0
062         */
063        public AssertionError(boolean detailMessage) {
064            this(String.valueOf(detailMessage));
065        }
066    
067        /**
068         * Constructs a new {@code AssertionError} with a message based on calling
069         * {@link String#valueOf(char)} with the specified character value.
070         * 
071         * @param detailMessage
072         *            the value to be converted into the message.
073         * @since Android 1.0
074         */
075        public AssertionError(char detailMessage) {
076            this(String.valueOf(detailMessage));
077        }
078    
079        /**
080         * Constructs a new {@code AssertionError} with a message based on calling
081         * {@link String#valueOf(int)} with the specified integer value.
082         * 
083         * @param detailMessage
084         *            the value to be converted into the message.
085         * @since Android 1.0
086         */
087        public AssertionError(int detailMessage) {
088            this(Integer.toString(detailMessage));
089        }
090    
091        /**
092         * Constructs a new {@code AssertionError} with a message based on calling
093         * {@link String#valueOf(long)} with the specified long value.
094         * 
095         * @param detailMessage
096         *            the value to be converted into the message.
097         * @since Android 1.0
098         */
099        public AssertionError(long detailMessage) {
100            this(Long.toString(detailMessage));
101        }
102    
103        /**
104         * Constructs a new {@code AssertionError} with a message based on calling
105         * {@link String#valueOf(float)} with the specified float value.
106         * 
107         * @param detailMessage
108         *            the value to be converted into the message.
109         * @since Android 1.0
110         */
111        public AssertionError(float detailMessage) {
112            this(Float.toString(detailMessage));
113        }
114    
115        /**
116         * Constructs a new {@code AssertionError} with a message based on calling
117         * {@link String#valueOf(double)} with the specified double value.
118         * 
119         * @param detailMessage
120         *            the value to be converted into the message.
121         * @since Android 1.0
122         */
123        public AssertionError(double detailMessage) {
124            this(Double.toString(detailMessage));
125        }
126    }
