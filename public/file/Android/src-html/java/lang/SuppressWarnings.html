017    package java.lang;
018    
019    import static java.lang.annotation.ElementType.CONSTRUCTOR;
020    import static java.lang.annotation.ElementType.FIELD;
021    import static java.lang.annotation.ElementType.LOCAL_VARIABLE;
022    import static java.lang.annotation.ElementType.METHOD;
023    import static java.lang.annotation.ElementType.PARAMETER;
024    import static java.lang.annotation.ElementType.TYPE;
025    import static java.lang.annotation.RetentionPolicy.SOURCE;
026    
027    import java.lang.annotation.Retention;
028    import java.lang.annotation.Target;
029    
030    /**
031     * Annotation type used to indicate that the compiler should not issue the
032     * specified warnings for the marked program element. Warnings are not only
033     * suppressed for the annotated element but also for all program elements
034     * contained in that element.
035     * <p>
036     * It is recommended that programmers always use this annotation on the most
037     * deeply nested element where it is actually needed.
038     * </p>
039     * 
040     * @since Android 1.0
041     */
042    @Target(value={TYPE,FIELD,METHOD,PARAMETER,CONSTRUCTOR,LOCAL_VARIABLE})
043    @Retention(value=SOURCE)
044    public @interface SuppressWarnings
045    {
046        /**
047         * The list of warnings a compiler should not issue.
048         * 
049         * @since Android 1.0
050         */
051        String[] value();
052    }
