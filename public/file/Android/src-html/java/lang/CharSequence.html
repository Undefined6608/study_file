018    package java.lang;
019    
020    
021    /**
022     * This interface represents an ordered set of characters and defines the
023     * methods to probe them.
024     * 
025     * @since Android 1.0
026     */
027    public interface CharSequence {
028    
029        /**
030         * Returns the number of characters in this sequence.
031         * 
032         * @return the number of characters.
033         * @since Android 1.0
034         */
035        public int length();
036    
037        /**
038         * Returns the character at the specified index, with the first character
039         * having index zero.
040         * 
041         * @param index
042         *            the index of the character to return.
043         * @return the requested character.
044         * @throws IndexOutOfBoundsException
045         *             if {@code index < 0} or {@code index} is greater than the
046         *             length of this sequence.
047         * @since Android 1.0
048         */
049        public char charAt(int index);
050    
051        /**
052         * Returns a {@code CharSequence} from the {@code start} index (inclusive)
053         * to the {@code end} index (exclusive) of this sequence.
054         * 
055         * @param start
056         *            the start offset of the sub-sequence. It is inclusive, that
057         *            is, the index of the first character that is included in the
058         *            sub-sequence.
059         * @param end
060         *            the end offset of the sub-sequence. It is exclusive, that is,
061         *            the index of the first character after those that are included
062         *            in the sub-sequence
063         * @return the requested sub-sequence.
064         * @throws IndexOutOfBoundsException
065         *             if {@code start < 0}, {@code end < 0}, {@code start > end},
066         *             or if {@code start} or {@code end} are greater than the
067         *             length of this sequence.
068         * @since Android 1.0
069         */
070        public CharSequence subSequence(int start, int end);
071    
072        /**
073         * Returns a string with the same characters in the same order as in this
074         * sequence.
075         * 
076         * @return a string based on this sequence.
077         * @since Android 1.0
078         */
079        public String toString();
080    }
