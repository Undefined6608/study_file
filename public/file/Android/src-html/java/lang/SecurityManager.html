043    /**
044     * Provides security verification facilities for applications. {@code
045     * SecurityManager} contains a set of {@code checkXXX} methods which determine
046     * if it is safe to perform a specific operation such as establishing network
047     * connections, modifying files, and many more. In general, these methods simply
048     * return if they allow the application to perform the operation; if an
049     * operation is not allowed, then they throw a {@link SecurityException}. The
050     * only exception is {@link #checkTopLevelWindow(Object)}, which returns a
051     * boolean to indicate permission.
052     * 
053     * @since Android 1.0
054     */
055    public class SecurityManager {

073        /**
074         * Constructs a new {@code SecurityManager} instance.
075         * <p>
076         * The {@code RuntimePermission("createSecurityManager")} is checked if a
077         * security manager is installed.
078         * </p>
079         * 
080         * @since Android 1.0
081         */
082        public SecurityManager() {
083            SecurityManager security = System.getSecurityManager();
084            if (security != null) {
085                security
086                        .checkPermission(RuntimePermission.permissionToCreateSecurityManager);
087            }
088            Class<?> type = Security.class; // initialize Security properties
089            if (type == null) {
090                throw new AssertionError();
091            }
092        }

307        /**
308         * Checks whether the calling thread is allowed to access members. The
309         * default is to allow access to public members (that is, {@code
310         * java.lang.reflect.Member.PUBLIC}) and to classes loaded by the same
311         * loader as the original caller (that is, the method that called the
312         * reflect API). Due to the nature of the check, overriding implementations
313         * cannot call {@code super.checkMemberAccess()} since the stack would no
314         * longer be of the expected shape.
315         * 
316         * @param cls
317         *            the class of which members are accessed.
318         * @param type
319         *            the access type, either {@code
320         *            java.lang.reflect.Member.PUBLIC} or {@code
321         *            java.lang.reflect.Member.DECLARED}.
322         * @throws SecurityException
323         *             if the calling thread is not allowed to access members of
324         *             {@code cls}.
325         * @since Android 1.0
326         */
327        public void checkMemberAccess(Class<?> cls, int type) {
328            if (cls == null) {
329                throw new NullPointerException();
330            }
331            if (type == Member.PUBLIC) {
332                return;
333            }
334            //
335            // Need to compare the classloaders.
336            // Stack shape is
337            // <user code> <- want this class
338            // Class.getDeclared*();
339            // Class.checkMemberAccess();
340            // SecurityManager.checkMemberAccess(); <- current frame
341            //
342            // Use getClassLoaderImpl() since getClassLoader()
343            // returns null for the bootstrap class loader.
344            if (ClassLoader.getStackClassLoader(3) == cls.getClassLoaderImpl()) {
345                return;
346            }
347    
348            // Forward off to the permission mechanism.
349            checkPermission(new RuntimePermission("accessDeclaredMembers")); //$NON-NLS-1$
350        }

429        /**
430         * Returns true if the package name is restricted by the specified security
431         * property.
432         */
433        private static boolean checkPackageProperty(final String property,
434                final String pkg) {
435            String list = AccessController.doPrivileged(PriviAction
436                    .getSecurityProperty(property));
437            if (list != null) {
438                int plen = pkg.length();
439                StringTokenizer tokenizer = new StringTokenizer(list, ", "); //$NON-NLS-1$
440                while (tokenizer.hasMoreTokens()) {
441                    String token = tokenizer.nextToken();
442                    int tlen = token.length();
443                    if (plen > tlen
444                            && pkg.startsWith(token)
445                            && (token.charAt(tlen - 1) == '.' || pkg.charAt(tlen) == '.')) {
446                        return true;
447                    } else if (plen == tlen && token.startsWith(pkg)) {
448                        return true;
449                    } else if (plen + 1 == tlen && token.startsWith(pkg)
450                            && token.charAt(tlen - 1) == '.') {
451                        return true;
452                    }
453                }
454            }
455    
456            return false;
457        }

564         * Checks whether the calling thread is trusted to show the specified top
565         * level window.
566         * 
567         * @param window
568         *            the window to show.
569         * @return {@code true} if the calling thread is allowed to show {@code

571         * @throws NullPointerException

573         * @since Android 1.0
574         */
575        public boolean checkTopLevelWindow(Object window) {
576            if (window == null) {
577                throw new NullPointerException();
578            }
579            try {
580                Class<?> awtPermission = Class.forName("java.awt.AWTPermission"); //$NON-NLS-1$
581                Constructor<?> constructor = awtPermission
582                        .getConstructor(String.class);
583                Object perm = constructor
584                        .newInstance("showWindowWithoutWarningBanner"); //$NON-NLS-1$
585                checkPermission((Permission) perm);
586            } catch (ClassNotFoundException e) {
587            } catch (NoSuchMethodException e) {
588            } catch (InstantiationException e) {
589            } catch (IllegalAccessException e) {
590            } catch (InvocationTargetException e) {
591            } catch (SecurityException e) {
592                return false;
593            }
594            return true;
595        }

729

904        /**
905         * Checks whether the calling thread is allowed to access the resource being
906         * guarded by the specified permission object.
907         * 
908         * @param permission
909         *            the permission to check.
910         * @throws SecurityException
911         *             if the requested {@code permission} is denied according to
912         *             the current security policy.
913         * @since Android 1.0
914         */
915        public void checkPermission(Permission permission) {
916            try {
917                inCheck = true;
918                AccessController.checkPermission(permission);
919            } finally {
920                inCheck = false;
921            }
922        }

924        /**
925         * Checks whether the specified security context is allowed to access the
926         * resource being guarded by the specified permission object.
927         * 
928         * @param permission
929         *            the permission to check.
930         * @param context
931         *            the security context for which to check permission.
932         * @throws SecurityException
933         *             if {@code context} is not an instance of {@code
934         *             AccessControlContext} or if the requested {@code permission}
935         *             is denied for {@code context} according to the current
936         *             security policy.
937         * @since Android 1.0
938         */
939        public void checkPermission(Permission permission, Object context) {
940            try {
941                inCheck = true;
942                // Must be an AccessControlContext. If we don't check
943                // this, then applications could pass in an arbitrary
944                // object which circumvents the security check.
945                if (context instanceof AccessControlContext) {
946                    ((AccessControlContext) context).checkPermission(permission);
947                } else {
948                    throw new SecurityException();
949                }
950            } finally {
951                inCheck = false;
952            }
953        }
