024    /**
025     * Class StrictMath provides basic math constants and operations such as
026     * trigonometric functions, hyperbolic functions, exponential, logarithms, etc.
027     * <p>
028     * In contrast to class {@link Math}, the methods in this class return exactly
029     * the same results on all platforms. Algorithms based on these methods thus
030     * behave the same (e.g. regarding numerical convergence) on all platforms,
031     * complying with the slogan "write once, run everywhere". On the other side,
032     * the implementation of class StrictMath may be less efficient than that of
033     * class Math, as class StrictMath cannot utilize platform specific features
034     * such as an extended precision math co-processors.
035     * <p>
036     * The methods in this class are specified using the "Freely Distributable Math
037     * Library" (fdlibm), version 5.3.
038     * <p>
039     * <a href="http://www.netlib.org/fdlibm/">http://www.netlib.org/fdlibm/</a>
040     * 
041     * @since Android 1.0
042     */

068        /**
069         * Returns the absolute value of the argument.
070         * <p>
071         * Special cases:
072         * <ul>
073         * <li>{@code abs(-0.0) = +0.0}</li>
074         * <li>{@code abs(+infinity) = +infinity}</li>
075         * <li>{@code abs(-infinity) = +infinity}</li>
076         * <li>{@code abs(NaN) = NaN}</li>
077         * </ul>
078         * </p>
079         * 
080         * @param d
081         *            the value whose absolute value has to be computed.
082         * @return the absolute value of the argument.
083         * @since Android 1.0
084         */
085        public static double abs(double d) {
086            long bits = Double.doubleToLongBits(d);
087            bits &= 0x7fffffffffffffffL;
088            return Double.longBitsToDouble(bits);
089        }

522        /**
523         * Returns the most positive (closest to positive infinity) of the two
524         * arguments.
525         * <p>
526         * Special cases:
527         * <ul>
528         * <li>{@code max(NaN, (anything)) = NaN}</li>
529         * <li>{@code max((anything), NaN) = NaN}</li>
530         * <li>{@code max(+0.0, -0.0) = +0.0}</li>
531         * <li>{@code max(-0.0, +0.0) = +0.0}</li>
532         * </ul>
533         * </p>
534         * 
535         * @param d1
536         *            the first argument.
537         * @param d2
538         *            the second argument.
539         * @return the larger of {@code d1} and {@code d2}.
540         * @since Android 1.0
541         */
542        public static double max(double d1, double d2) {
543            if (d1 > d2)
544                return d1;
545            if (d1 < d2)
546                return d2;
547            /* if either arg is NaN, return NaN */
548            if (d1 != d2)
549                return Double.NaN;
550            /* max( +0.0,-0.0) == +0.0 */
551            if (d1 == 0.0
552                    && ((Double.doubleToLongBits(d1) & Double.doubleToLongBits(d2)) & 0x8000000000000000L) == 0)
553                return 0.0;
554            return d1;
555        }

645            if (d1 < d2)

652                    && ((Double.doubleToLongBits(d1) | Double.doubleToLongBits(d2)) & 0x8000000000000000l) != 0)
653                return 0.0 * (-1.0);
654            return d1;
