018    package java.lang;
019    
020    /**
021     * Placeholder class for environments which explicitly manage the action of a
022     * <em>Just In Time (JIT)</em> compiler. This class is usually implemented by
023     * the virtual machine vendor. The Android reference implementation does not
024     * (yet) contain such a JIT compiler, though other implementations may choose to
025     * provide one.
026     * 
027     * @since Android 1.0
028     */
029    public final class Compiler {
030    
031        /**
032         * Prevent this class from being instantiated.
033         */
034        private Compiler(){
035            //do nothing
036        }
037    
038        /**
039         * Executes an operation according to the specified command object. This
040         * method is the low-level interface to the JIT compiler. It may return any
041         * object or {@code null} if no JIT compiler is available.
042         * 
043         * @param cmd
044         *            the command object for the JIT compiler.
045         * @return the result of executing command or {@code null}.
046         * @since Android 1.0
047         */
048        public static Object command(Object cmd) {
049            return null;
050        }
051    
052        /**
053         * Compiles the specified class using the JIT compiler and indicates if
054         * compilation has been successful.
055         * 
056         * @param classToCompile
057         *            java.lang.Class the class to JIT compile
058         * @return {@code true} if the compilation has been successful;
059         *         {@code false} if it has failed or if there is no JIT compiler
060         *         available.
061         * @since Android 1.0
062         */
063        public static boolean compileClass(Class<?> classToCompile) {
064            return false;
065        }
066    
067        /**
068         * Compiles all classes whose name matches the specified name using the JIT
069         * compiler and indicates if compilation has been successful.
070         * 
071         * @param nameRoot
072         *            the string to match class names with.
073         * @return {@code true} if the compilation has been successful;
074         *         {@code false} if it has failed or if there is no JIT compiler
075         *         available.
076         * @since Android 1.0
077         */
078        public static boolean compileClasses(String nameRoot) {
079            return false;
080        }
081    
082        /**
083         * Disables the JIT compiler.
084         * 
085         * @since Android 1.0
086         */
087        public static void disable() {
088            return;
089        }
090    
091        /**
092         * Enables the JIT compiler.
093         * 
094         * @since Android 1.0
095         */
096        public static void enable() {
097            return;
098        }
099    
100    }
