170        /**
171         * Executes the specified command and its arguments in a separate native
172         * process. The new process uses the environment provided in {@code envp}
173         * and the working directory specified by {@code directory}.
174         * 
175         * @param progArray
176         *            the array containing the program to execute as well as any
177         *            arguments to the program.
178         * @param envp
179         *            the array containing the environment to start the new process
180         *            in.
181         * @param directory
182         *            the directory in which to execute the program. If {@code null},
183         *            execute if in the same directory as the parent process.
184         * @return the new {@code Process} object that represents the native
185         *         process.
186         * @throws IOException
187         *             if the requested program can not be executed.
188         * @throws SecurityException
189         *             if the current {@code SecurityManager} disallows program
190         *             execution.
191         * @see SecurityManager#checkExec
192         * @since Android 1.0
193         */    
194        public Process exec(String[] progArray, String[] envp, File directory)
195                throws java.io.IOException {
196            
197            // Sanity checks
198            if (progArray == null) {
199                throw new NullPointerException();
200            } else if (progArray.length == 0) {
201                throw new IndexOutOfBoundsException();
202            } else {
203                for (int i = 0; i < progArray.length; i++) {
204                    if (progArray[i] == null) {
205                        throw new NullPointerException();
206                    }
207                }
208            }
209            
210            if (envp != null) {
211                for (int i = 0; i < envp.length; i++) {
212                    if (envp[i] == null) {
213                        throw new NullPointerException();
214                    }
215                }
216            }
217            
218            // Security checks
219            SecurityManager smgr = System.getSecurityManager();
220            if (smgr != null) {
221                smgr.checkExec(progArray[0]);
222            }
223            
224            // Delegate the execution
225            return ProcessManager.getInstance().exec(progArray, envp, directory);
226        }

273        /**
274         * Executes the specified program in a separate native process. The new
275         * process uses the environment provided in {@code envp} and the working
276         * directory specified by {@code directory}.
277         * 
278         * @param prog
279         *            the name of the program to execute.
280         * @param envp
281         *            the array containing the environment to start the new process
282         *            in.
283         * @param directory
284         *            the directory in which to execute the program. If {@code null},
285         *            execute if in the same directory as the parent process.
286         * @return the new {@code Process} object that represents the native
287         *         process.
288         * @throws IOException
289         *             if the requested program can not be executed.
290         * @throws SecurityException
291         *             if the current {@code SecurityManager} disallows program
292         *             execution.
293         * @see SecurityManager#checkExec
294         * @since Android 1.0
295         */
296        public Process exec(String prog, String[] envp, File directory) throws java.io.IOException {
297            // Sanity checks
298            if (prog == null) {
299                throw new NullPointerException();
300            } else if (prog.length() == 0) {
301                throw new IllegalArgumentException();
302            }
303            
304            // Break down into tokens, as described in Java docs
305            StringTokenizer tokenizer = new StringTokenizer(prog);
306            int length = tokenizer.countTokens();
307            String[] progArray = new String[length];
308            for (int i = 0; i < length; i++) {
309                progArray[i] = tokenizer.nextToken();
310            }
311            
312            // Delegate
313            return exec(progArray, envp, directory);
314        }

316        /**
317         * Causes the virtual machine to stop running and the program to exit. If
318         * {@link #runFinalizersOnExit(boolean)} has been previously invoked with a
319         * {@code true} argument, then all all objects will be properly
320         * garbage-collected and finalized first.
321         * 
322         * @param code
323         *            the return code. By convention, non-zero return codes indicate
324         *            abnormal terminations.
325         * @throws SecurityException
326         *             if the current {@code SecurityManager} does not allow the
327         *             running thread to terminate the virtual machine.
328         * @see SecurityManager#checkExit
329         * @since Android 1.0
330         */
331        public void exit(int code) {
332            // Security checks
333            SecurityManager smgr = System.getSecurityManager();
334            if (smgr != null) {
335                smgr.checkExit(code);
336            }
337    
338            // Make sure we don't try this several times
339            synchronized(this) {
340                if (!shuttingDown) {
341                    shuttingDown = true;
342    
343                    Thread[] hooks;
344                    synchronized (shutdownHooks) {
345                        // create a copy of the hooks
346                        hooks = new Thread[shutdownHooks.size()];
347                        shutdownHooks.toArray(hooks);
348                    }
349    
350                    // Start all shutdown hooks concurrently
351                    for (int i = 0; i < hooks.length; i++) {
352                        hooks[i].start();
353                    }
354    
355                    // Wait for all shutdown hooks to finish
356                    for (Thread hook : hooks) {
357                        try {
358                            hook.join();
359                        } catch (InterruptedException ex) {
360                            // Ignore, since we are at VM shutdown.
361                        }
362                    }
363    
364                    // Ensure finalization on exit, if requested
365                    if (finalizeOnExit) {
366                        runFinalization(true);
367                    }
368    
369                    // Get out of here finally...
370                    nativeExit(code, true);
371                }
372            }
373        }

467        /*
468         * Loads and links a library without security checks.
469         */
470        void loadLibrary(String libname, ClassLoader loader) {
471            String filename;
472            int i;
473    
474            if (loader != null) {
475                filename = loader.findLibrary(libname);
476                if (filename != null && nativeLoad(filename, loader))
477                    return;
478                // else fall through to exception
479            } else {
480                filename = System.mapLibraryName(libname);
481                for (i = 0; i < mLibPaths.length; i++) {
482                    if (false)
483                        System.out.println("Trying " + mLibPaths[i] + filename);
484                    if (nativeLoad(mLibPaths[i] + filename, loader))
485                        return;
486                }
487            }
488    
489            throw new UnsatisfiedLinkError("Library " + libname + " not found");
490        }

619        /**
620         * Registers a virtual-machine shutdown hook. A shutdown hook is a
621         * {@code Thread} that is ready to run, but has not yet been started. All
622         * registered shutdown hooks will be executed once the virtual machine shuts
623         * down properly. A proper shutdown happens when either the
624         * {@link #exit(int)} method is called or the surrounding system decides to
625         * terminate the application, for example in response to a {@code CTRL-C} or
626         * a system-wide shutdown. A termination of the virtual machine due to the
627         * {@link #halt(int)} method, an {@link Error} or a {@code SIGKILL}, in
628         * contrast, is not considered a proper shutdown. In these cases the
629         * shutdown hooks will not be run.
630         * <p>
631         * Shutdown hooks are run concurrently and in an unspecified order. Hooks
632         * failing due to an unhandled exception are not a problem, but the stack
633         * trace might be printed to the console. Once initiated, the whole shutdown
634         * process can only be terminated by calling {@code halt()}.
635         * <p>
636         * If {@link #runFinalizersOnExit(boolean)} has been called with a {@code
637         * true} argument, garbage collection and finalization will take place after
638         * all hooks are either finished or have failed. Then the virtual machine
639         * terminates.
640         * <p>
641         * It is recommended that shutdown hooks do not do any time-consuming
642         * activities, in order to not hold up the shutdown process longer than
643         * necessary.
644         * 
645         * @param hook
646         *            the shutdown hook to register.
647         * @throws IllegalArgumentException
648         *             if the hook has already been started or if it has already
649         *             been registered.
650         * @throws IllegalStateException
651         *             if the virtual machine is already shutting down.
652         * @throws SecurityException
653         *             if a SecurityManager is registered and the calling code
654         *             doesn't have the RuntimePermission("shutdownHooks").
655         */
656        public void addShutdownHook(Thread hook) {
657            // Sanity checks
658            if (hook == null) {
659                throw new NullPointerException("Hook may not be null.");
660            }
661    
662            if (shuttingDown) {
663                throw new IllegalStateException("VM already shutting down");
664            }
665    
666            if (hook.hasBeenStarted) {
667                throw new IllegalArgumentException("Hook has already been started");
668            }
669    
670            SecurityManager sm = System.getSecurityManager();
671            if (sm != null) {
672                sm.checkPermission(new RuntimePermission("shutdownHooks"));
673            }
674            
675            synchronized (shutdownHooks) {
676                if (shutdownHooks.contains(hook)) {
677                    throw new IllegalArgumentException("Hook already registered.");
678                }
679        
680                shutdownHooks.add(hook);
681            }
682        }

687         * @param hook
688         *            the shutdown hook to remove.
689         * @return {@code true} if the hook has been removed successfully; {@code
690         *         false} otherwise.
691         * @throws IllegalStateException
692         *             if the virtual machine is already shutting down.
693         * @throws SecurityException
694         *             if a SecurityManager is registered and the calling code
695         *             doesn't have the RuntimePermission("shutdownHooks").

697        public boolean removeShutdownHook(Thread hook) {
698            // Sanity checks
699            if (hook == null) {
700                throw new NullPointerException("Hook may not be null.");

767    /*
768     * Internal helper class for creating a localized InputStream. A reader
769     * wrapped in an InputStream.
770     */
771    class ReaderInputStream extends InputStream {
772    
773        private Reader reader;
774        
775        private Writer writer;
776    
777        ByteArrayOutputStream out = new ByteArrayOutputStream(256);
778        
779        private byte[] bytes;
780        
781        private int nextByte;
782        
783        private int numBytes;
784        
785        String encoding = System.getProperty("file.encoding", "UTF-8");
786        
787        public ReaderInputStream(InputStream stream) {
788            try {
789                reader = new InputStreamReader(stream, "UTF-8");
790                writer = new OutputStreamWriter(out, encoding);
791            } catch (UnsupportedEncodingException e) {
792                // Should never happen, since UTF-8 and platform encoding must be
793                // supported.
794                throw new RuntimeException(e);
795            }
796        }
797    
798        @Override
799        public int read() throws IOException {
800            if (nextByte >= numBytes) {
801                readBuffer();
802            }
803    
804            return (numBytes < 0) ? -1 : bytes[nextByte++];
805        }
806    
807        private void readBuffer() throws IOException {
808            char[] chars = new char[128];
809            int read = reader.read(chars);
810            if (read < 0) {
811                numBytes = read;
812                return;
813            }
814    
815            writer.write(chars, 0, read);
816            writer.flush();
817            bytes = out.toByteArray();
818            numBytes = bytes.length;
819            nextByte = 0;
820        }        
821        
822    }

824    /*
825     * Internal helper class for creating a localized OutputStream. A writer
826     * wrapped in an OutputStream. Bytes are written to characters in big-endian
827     * fashion.
828     */
829    class WriterOutputStream extends OutputStream {
830    
831        private Reader reader;
832    
833        private Writer writer;
834    
835        private PipedOutputStream out;
836    
837        private PipedInputStream pipe;
838    
839        private int numBytes;
840    
841        private String enc = System.getProperty("file.encoding", "UTF-8");
842    
843        public WriterOutputStream(OutputStream stream) {
844            try {
845                // sink
846                this.writer = new OutputStreamWriter(stream, enc);
847    
848                // transcriber
849                out = new PipedOutputStream();
850                pipe = new PipedInputStream(out);
851                this.reader = new InputStreamReader(pipe, "UTF-8");
852    
853            } catch (UnsupportedEncodingException e) {
854                // Should never happen, since platform encoding must be supported.
855                throw new RuntimeException(e);
856            } catch (IOException e) {
857                throw new RuntimeException(e);
858            }
859        }
860    
861        @Override
862        public void write(int b) throws IOException {
863            out.write(b);
864            if( ++numBytes > 256) {
865                flush();
866                numBytes = 0;
867            }
868        }
869    
870        @Override
871        public void flush() throws IOException {
872            out.flush();
873            char[] chars = new char[128];
874            if (pipe.available() > 0) {
875                int read = reader.read(chars);
876                if (read > 0) {
877                    writer.write(chars, 0, read);
878                }
879            }
880            writer.flush();
881        }
882    
883        @Override
884        public void close() throws IOException {
885            out.close();
886            flush();
887            writer.close();
888        }
889    }
