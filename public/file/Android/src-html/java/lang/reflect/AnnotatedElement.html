018    package java.lang.reflect;
019    
020    import java.lang.annotation.Annotation;
021    
022    /**
023     * This interface provides reflective access to annotation information.
024     * 
025     * @since Android 1.0
026     */
027    public interface AnnotatedElement {
028    
029        /**
030         * Returns, for this element, the annotation with the specified type, or
031         * {@code null} if no annotation with the specified type is present
032         * (including inherited annotations).
033         * 
034         * @param annotationType
035         *            the type of the annotation to search for
036         *            
037         * @return the annotation with the specified type or {@code null}
038         * 
039         * @throws NullPointerException
040         *             if {@code annotationType} is {@code null}
041         * 
042         * @since Android 1.0
043         */
044        <T extends Annotation> T getAnnotation(Class<T> annotationType);
045    
046        /**
047         * Returns, for this element, an array containing all annotations (including
048         * inherited annotations). If there are no annotations present, this method
049         * returns a zero length array.
050         * 
051         * @return an array of all annotations for this element
052         * 
053         * @since Android 1.0
054         */
055        Annotation[] getAnnotations();
056    
057        /**
058         * Returns, for this element, all annotations that are explicitly declared
059         * (not inherited). If there are no declared annotations present, this
060         * method returns a zero length array.
061         * 
062         * @return an array of annotations declared for this element
063         * 
064         * @since Android 1.0
065         */
066        Annotation[] getDeclaredAnnotations();
067    
068        /**
069         * Indicates whether or not this element has an annotation with the
070         * specified annotation type (including inherited annotations).
071         * 
072         * @param annotationType
073         *            the type of the annotation to search for
074         *            
075         * @return {@code true} if the annotation exists, {@code false} otherwise
076         * 
077         * @throws NullPointerException
078         *             if {@code annotationType} is {@code null}
079         * 
080         * @since Android 1.0
081         */
082        boolean isAnnotationPresent(Class<? extends Annotation> annotationType);
083    }
