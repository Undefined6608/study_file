018    package java.lang.annotation;
019    
020    /**
021     * Defines the interface implemented by all annotations. Note that the interface
022     * itself is <i>not</i> an annotation, and neither is an interface that simply
023     * extends this one. Only the compiler is able to create proper annotation
024     * types.
025     * 
026     * @since Android 1.0
027     */
028    public interface Annotation {
029    
030        /**
031         * Returns the type of this annotation.
032         * 
033         * @return A {@code Class} instance representing the annotation type.
034         * 
035         * @since Android 1.0
036         */
037        Class<? extends Annotation> annotationType();
038    
039        /**
040         * Determines whether or not this annotation is equivalent to the annotation
041         * passed. This is determined according to the following rules:
042         * 
043         * <ul>
044         *     <li>
045         *         Two annotations {@code x} and {@code y} are equal if and only if
046         *         they are members of the same annotation type and all the member
047         *         values of {@code x} are equal to the corresponding member values
048         *         of {@code y}.
049         *     </li>
050         *     <li>
051         *         The equality of primitive member values {@code x} and {@code y}
052         *         is determined (in a way similar to) using the corresponding
053         *         wrapper classes. For example,
054         *         {@code Integer.valueOf(x).equals(Integer.valueOf(y)} is used for
055         *         {@code int} values. Note: The behavior is identical to the
056         *         {@code ==} operator for all but the floating point type, so the
057         *         implementation may as well use {@code ==} in these cases for
058         *         performance reasons. Only for the {@code float} and {@code double}
059         *         types the result will be slightly different: {@code NaN} is equal
060         *         to {@code NaN}, and {@code -0.0} is equal to {@code 0.0}, both of
061         *         which is normally not the case.
062         *     </li>
063         *     <li>
064         *         The equality of two array member values {@code x} and {@code y}
065         *         is determined using the corresponding {@code equals(x, y)}
066         *         helper function in {@link java.util.Arrays}.
067         *     </li>
068         *     <li>
069         *         The hash code for all other member values is determined by simply
070         *         calling their {@code equals()} method.
071         *     </li>
072         * </ul>
073         * 
074         * @param obj
075         *            The object to compare to.
076         * 
077         * @return {@code true} if {@code obj} is equal to this annotation, 
078         * {@code false} otherwise.
079         *         
080         * @since Android 1.0
081         */
082        boolean equals(Object obj);
083    
084        /**
085         * Returns the hash code of this annotation. The hash code is determined
086         * according to the following rules:
087         * 
088         * <ul>
089         *     <li>
090         *         The hash code of an annotation is the sum of the hash codes of
091         *         its annotation members.
092         *     </li>
093         *     <li>
094         *         The hash code of an annotation member is calculated as {@code
095         *         (0x7f * n.hashCode()) ^ v.hashCode())}, where {@code n} is the
096         *         name of the member (as a {@code String}) and {@code v} its value.
097         *     </li>
098         *     <li>
099         *         The hash code for a primitive member value is determined using
100         *         the corresponding wrapper type. For example, {@code
101         *         Integer.valueOf(v).hashCode()} is used for an {@code int} value
102         *         {@code v}.
103         *     </li>
104         *     <li>
105         *         The hash code for an array member value {@code v} is determined
106         *         using the corresponding {@code hashCode(v)} helper function in
107         *         {@link java.util.Arrays}.
108         *     </li>
109         *     <li>
110         *         The hash code for all other member values is determined by simply
111         *         calling their {@code hashCode} method.
112         *     </li>
113         * </ul>
114         * 
115         * @return the hash code.
116         * 
117         * @since Android 1.0
118         */
119        int hashCode();
120    
121        /**
122         * Returns a {@code String} representation of this annotation. It is not
123         * strictly defined what the representation has to look like, but it usually
124         * consists of the name of the annotation, preceded by a "@". If the
125         * annotation contains field members, their names and values are also
126         * included in the result. 
127         * 
128         * @return the {@code String} that represents this annotation.
129         * 
130         * @since Android 1.0
131         */
132        String toString();
133    }
