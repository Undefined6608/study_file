020    /**
021     * The wrapper for the primitive type {@code float}.
022     * 
023     * @see java.lang.Number
024     * @since Android 1.0

127        /**
128         * Compares this object to the specified float object to determine their
129         * relative order. There are two special cases:
130         * <ul>
131         * <li>{@code Float.NaN} is equal to {@code Float.NaN} and it is greater
132         * than any other float value, including {@code Float.POSITIVE_INFINITY};</li>
133         * <li>+0.0f is greater than -0.0f</li>
134         * </ul>
135         * 
136         * @param object
137         *            the float object to compare this object to.
138         * @return a negative value if the value of this float is less than the
139         *         value of {@code object}; 0 if the value of this float and the
140         *         value of {@code object} are equal; a positive value if the value
141         *         of this float is greater than the value of {@code object}.
142         * @see java.lang.Comparable
143         * @since Android 1.0
144         */
145        public int compareTo(Float object) {
146            int f1, f2;
147            int NaNbits = Float.floatToIntBits(Float.NaN);
148            if ((f1 = Float.floatToIntBits(value)) == NaNbits) {
149                if (Float.floatToIntBits(object.value) == NaNbits) {
150                    return 0;
151                }
152                return 1;
153            }
154            if ((f2 = Float.floatToIntBits(object.value)) == NaNbits) {
155                return -1;
156            }
157            if (value == object.value) {
158                if (f1 == f2) {
159                    return 0;
160                }
161                // check for -0
162                return f1 > f2 ? 1 : -1;
163            }
164            return value > object.value ? 1 : -1;
165        }

177        /**
178         * Compares this instance with the specified object and indicates if they
179         * are equal. In order to be equal, {@code object} must be an instance of
180         * {@code Float} and have the same float value as this object.
181         * 
182         * @param object
183         *            the object to compare this float with.
184         * @return {@code true} if the specified object is equal to this
185         *         {@code Float}; {@code false} otherwise.
186         * @since Android 1.0
187         */
188        @Override
189        public boolean equals(Object object) {
190            return (object == this)
191                    || (object instanceof Float)
192                    && (floatToIntBits(this.value) == floatToIntBits(((Float) object).value));
193        }

371        /**
372         * Compares the two specified float values. There are two special cases:
373         * <ul>
374         * <li>{@code Float.NaN} is equal to {@code Float.NaN} and it is greater
375         * than any other float value, including {@code Float.POSITIVE_INFINITY};</li>
376         * <li>+0.0f is greater than -0.0f</li>
377         * </ul>
378         * 
379         * @param float1
380         *            the first value to compare.
381         * @param float2
382         *            the second value to compare.
383         * @return a negative value if {@code float1} is less than {@code float2};
384         *         0 if {@code float1} and {@code float2} are equal; a positive
385         *         value if {@code float1} is greater than {@code float2}.
386         * @since Android 1.0
387         */
388        public static int compare(float float1, float float2) {
389            int f1, f2;
390            int NaNbits = Float.floatToIntBits(Float.NaN);
391            if ((f1 = Float.floatToIntBits(float1)) == NaNbits) {
392                if (Float.floatToIntBits(float2) == NaNbits) {
393                    return 0;
394                }
395                return 1;
396            }
397            if ((f2 = Float.floatToIntBits(float2)) == NaNbits) {
398                return -1;
399            }
400            if (float1 == float2) {
401                if (f1 == f2) {
402                    return 0;
403                }
404                // check for -0
405                return f1 > f2 ? 1 : -1;
406            }
407            return float1 > float2 ? 1 : -1;
408        }

422        /**
423         * Converts the specified float into its hexadecimal string representation.
424         * 
425         * @param f
426         *            the float to convert.
427         * @return the hexadecimal string representation of {@code f}.
428         * @since Android 1.0
429         */
430        public static String toHexString(float f) {
431            /*
432             * Reference: http://en.wikipedia.org/wiki/IEEE_754
433             */
434            if (f != f) {
435                return "NaN"; //$NON-NLS-1$
436            }
437            if (f == POSITIVE_INFINITY) {
438                return "Infinity"; //$NON-NLS-1$
439            }
440            if (f == NEGATIVE_INFINITY) {
441                return "-Infinity"; //$NON-NLS-1$
442            }
443    
444            int bitValue = floatToIntBits(f);
445    
446            boolean negative = (bitValue & 0x80000000) != 0;
447            // mask exponent bits and shift down
448            int exponent = (bitValue & 0x7f800000) >>> 23;
449            // mask significand bits and shift up
450            // significand is 23-bits, so we shift to treat it like 24-bits
451            int significand = (bitValue & 0x007FFFFF) << 1;
452    
453            if (exponent == 0 && significand == 0) {
454                return (negative ? "-0x0.0p0" : "0x0.0p0"); //$NON-NLS-1$ //$NON-NLS-2$
455            }
456    
457            StringBuilder hexString = new StringBuilder(10);
458            if (negative) {
459                hexString.append("-0x"); //$NON-NLS-1$
460            } else {
461                hexString.append("0x"); //$NON-NLS-1$
462            }
463    
464            if (exponent == 0) { // denormal (subnormal) value
465                hexString.append("0."); //$NON-NLS-1$
466                // significand is 23-bits, so there can be 6 hex digits
467                int fractionDigits = 6;
468                // remove trailing hex zeros, so Integer.toHexString() won't print
469                // them
470                while ((significand != 0) && ((significand & 0xF) == 0)) {
471                    significand >>>= 4;
472                    fractionDigits--;
473                }
474                // this assumes Integer.toHexString() returns lowercase characters
475                String hexSignificand = Integer.toHexString(significand);
476    
477                // if there are digits left, then insert some '0' chars first
478                if (significand != 0 && fractionDigits > hexSignificand.length()) {
479                    int digitDiff = fractionDigits - hexSignificand.length();
480                    while (digitDiff-- != 0) {
481                        hexString.append('0');
482                    }
483                }
484                hexString.append(hexSignificand);
485                hexString.append("p-126"); //$NON-NLS-1$
486            } else { // normal value
487                hexString.append("1."); //$NON-NLS-1$
488                // significand is 23-bits, so there can be 6 hex digits
489                int fractionDigits = 6;
490                // remove trailing hex zeros, so Integer.toHexString() won't print
491                // them
492                while ((significand != 0) && ((significand & 0xF) == 0)) {
493                    significand >>>= 4;
494                    fractionDigits--;
495                }
496                // this assumes Integer.toHexString() returns lowercase characters
497                String hexSignificand = Integer.toHexString(significand);
498    
499                // if there are digits left, then insert some '0' chars first
500                if (significand != 0 && fractionDigits > hexSignificand.length()) {
501                    int digitDiff = fractionDigits - hexSignificand.length();
502                    while (digitDiff-- != 0) {
503                        hexString.append('0');
504                    }
505                }
506                hexString.append(hexSignificand);
507                hexString.append('p');
508                // remove exponent's 'bias' and convert to a string
509                hexString.append(Integer.toString(exponent - 127));
510            }
511            return hexString.toString();
512        }
