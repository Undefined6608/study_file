018    package java.lang;
019    
020    import java.lang.reflect.Method;
021    
022    /**
023     * Placeholder class for the Java keyword {@code void}.
024     * 
025     * @since Android 1.0
026     */
027    public final class Void extends Object {
028        
029        /**
030         * The {@link Class} object that represents the primitive type {@code void}.
031         * 
032         * @since Android 1.0
033         */
034        public static final Class<Void> TYPE = lookupType();
035    
036        // Note: This can't be set to "void.class", since *that* is
037        // defined to be "java.lang.Void.TYPE";
038    
039        @SuppressWarnings("unchecked")
040        private static Class<Void> lookupType() {
041            Class<?> voidType = null;
042            try {
043                Method method = Runnable.class.getMethod("run", new Class[0]); //$NON-NLS-1$
044                voidType = method.getReturnType();
045            } catch (Exception e) {
046                throw new RuntimeException(e);
047            }
048            return (Class<Void>) voidType;
049        }
050    
051        private Void() {
052        }
053    }
