018    package java.lang;
019    
020    
021    import java.io.InputStream;
022    import java.io.OutputStream;
023    
024    /**
025     * Represents an external process. Enables writing to, reading from, destroying,
026     * and waiting for the external process, as well as querying its exit value.
027     * 
028     * @see Runtime#exec
029     * @see ProcessBuilder#start()
030     * 
031     * @since Android 1.0
032     */
033    public abstract class Process {
034    
035        /**
036         * Terminates this process and closes any associated streams.
037         * 
038         * @since Android 1.0
039         */
040        abstract public void destroy();
041    
042        /**
043         * Returns the exit value of the native process represented by this object.
044         * It is available only when the native process has terminated.
045         * 
046         * @return the exit value of this process.
047         * @throws IllegalThreadStateException
048         *             if this process has not terminated.
049         * @since Android 1.0
050         */
051        abstract public int exitValue();
052    
053        /**
054         * Returns an input stream that is connected to the error stream
055         * <em>(stderr)</em> of the native process represented by this object.
056         * 
057         * @return the input stream to read from the error stream associated with
058         *         the native process.
059         * @since Android 1.0
060         */
061        abstract public InputStream getErrorStream();
062    
063        /**
064         * Returns an input stream that is connected to the standard output stream
065         * <em>(stdout)</em> of the native process represented by this object.
066         * 
067         * @return the input stream to read from the output stream associated with
068         *         the native process.
069         * @since Android 1.0
070         */
071        abstract public InputStream getInputStream();
072    
073        /**
074         * Returns an output stream that is connected to the standard input stream
075         * <em>(stdin)</em> of the native process represented by this object.
076         * 
077         * @return the output stream to write to the input stream associated with
078         *         the native process.
079         * @since Android 1.0
080         */
081        abstract public OutputStream getOutputStream();
082    
083        /**
084         * Causes the calling thread to wait for the native process associated with
085         * this object to finish executing.
086         * 
087         * @return the exit value of the native process being waited on.
088         * @throws InterruptedException
089         *             if the calling thread is interrupted.
090         * @since Android 1.0
091         */
092        abstract public int waitFor() throws InterruptedException;
093    }
