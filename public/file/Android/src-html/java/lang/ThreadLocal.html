017    package java.lang;
018    
019    import java.lang.ref.WeakReference;
020    import java.lang.ref.Reference;
021    import java.util.concurrent.atomic.AtomicInteger;
022    
023    /**
024     * Implements a thread-local storage, that is, a variable for which each thread
025     * has its own value. All threads share the same {@code ThreadLocal} object,
026     * but each sees a different value when accessing it, and changes made by one
027     * thread do not affect the other threads. The implementation supports
028     * {@code null} values.
029     * 
030     * @see java.lang.Thread
031     * @author Bob Lee
032     * @since Android 1.0
033     */
034    public class ThreadLocal<T> {
035    
036        /* Thanks to Josh Bloch and Doug Lea for code reviews and impl advice. */
037    
038        /**
039         * Creates a new thread-local variable.
040         */
041        public ThreadLocal() {}
042    
043        /**
044         * Returns the value of this variable for the current thread. If an entry
045         * doesn't yet exist for this variable on this thread, this method will
046         * create an entry, populating the value with the result of
047         * {@link #initialValue()}.
048         * 
049         * @return the current value of the variable for the calling thread.
050         */
051        @SuppressWarnings("unchecked")
052        public T get() {
053            // Optimized for the fast path.
054            Thread currentThread = Thread.currentThread();
055            Values values = values(currentThread);
056            if (values != null) {
057                Object[] table = values.table;
058                int index = hash & values.mask;
059                if (this.reference == table[index]) {
060                    return (T) table[index + 1];
061                }
062            } else {
063                values = initializeValues(currentThread);
064            }
065    
066            return (T) values.getAfterMiss(this);
067        }
068    
069        /**
070         * Provides the initial value of this variable for the current thread.
071         * The default implementation returns {@code null}.
072         * 
073         * @return the initial value of the variable.
074         */
075        protected T initialValue() {
076            return null;
077        }
078    
079        /**
080         * Sets the value of this variable for the current thread. If set to
081         * {@code null}, the value will be set to null and the underlying entry will
082         * still be present.
083         * 
084         * @param value the new value of the variable for the caller thread.
085         */
086        public void set(T value) {
087            Thread currentThread = Thread.currentThread();
088            Values values = values(currentThread);
089            if (values == null) {
090                values = initializeValues(currentThread);
091            }
092            values.put(this, value);
093        }
094    
095        /**
096         * Removes the entry for this variable in the current thread. If this call
097         * is followed by a {@link #get()} before a {@link #set},
098         * {@code #get()} will call {@link #initialValue()} and create a new
099         * entry with the resulting value.
100         */
101        public void remove() {
102            Thread currentThread = Thread.currentThread();
103            Values values = values(currentThread);
104            if (values != null) {
105                values.remove(this);
106            }
107        }
108    
109        /**
110         * Creates Values instance for this thread and variable type.
111         */
112        Values initializeValues(Thread current) {
113            return current.localValues = new Values();
114        }
115    
116        /**
117         * Gets Values instance for this thread and variable type.
118         */
119        Values values(Thread current) {
120            return current.localValues;
121        }
122    
123        /** Weak reference to this thread local instance. */
124        private final Reference<ThreadLocal<T>> reference
125                = new WeakReference<ThreadLocal<T>>(this);
126    
127        /** Hash counter. */
128        private static AtomicInteger hashCounter = new AtomicInteger(0);
129    
130        /**
131         * Internal hash. We deliberately don't bother with #hashCode().
132         * Hashes must be even. This ensures that the result of
133         * (hash & (table.length - 1)) points to a key and not a value.
134         *
135         * We increment by Doug Lea's Magic Number(TM) (*2 since keys are in
136         * every other bucket) to help prevent clustering.
137         */
138        private final int hash = hashCounter.getAndAdd(0x61c88647 << 1);
139    
140        /**
141         * Per-thread map of ThreadLocal instances to values.
142         */
143        static class Values {
144    
145            /**
146             * Size must always be a power of 2.
147             */
148            private static final int INITIAL_SIZE = 16;
149    
150            /**
151             * Placeholder for deleted entries.
152             */
153            private static final Object TOMBSTONE = new Object();
154    
155            /**
156             * Map entries. Contains alternating keys (ThreadLocal) and values.
157             * The length is always a power of 2.
158             */
159            private Object[] table;
160    
161            /** Used to turn hashes into indices. */
162            private int mask;
163    
164            /** Number of live entries. */
165            private int size;
166    
167            /** Number of tombstones. */
168            private int tombstones;
169    
170            /** Maximum number of live entries and tombstones. */
171            private int maximumLoad;
172    
173            /** Points to the next cell to clean up. */
174            private int clean;
175    
176            /**
177             * Constructs a new, empty instance.
178             */
179            Values() {
180                initializeTable(INITIAL_SIZE);
181                this.size = 0;
182                this.tombstones = 0;
183            }
184    
185            /**
186             * Used for InheritableThreadLocals.
187             */
188            Values(Values fromParent) {
189                this.table = fromParent.table.clone();
190                this.mask = fromParent.mask;
191                this.size = fromParent.size;
192                this.tombstones = fromParent.tombstones;
193                this.maximumLoad = fromParent.maximumLoad;
194                this.clean = fromParent.clean;
195                inheritValues(fromParent);
196            }
197    
198            /**
199             * Inherits values from a parent thread.
200             */
201            @SuppressWarnings({"unchecked"})
202            private void inheritValues(Values fromParent) {
203                // Transfer values from parent to child thread.
204                Object[] table = this.table;
205                for (int i = table.length - 2; i >= 0; i -= 2) {
206                    Object k = table[i];
207    
208                    if (k == null || k == TOMBSTONE) {
209                        // Skip this entry.
210                        continue;
211                    }
212    
213                    // The table can only contain null, tombstones and references.
214                    Reference<InheritableThreadLocal<?>> reference
215                            = (Reference<InheritableThreadLocal<?>>) k;
216                    // Raw type enables us to pass in an Object below.
217                    InheritableThreadLocal key = reference.get();
218                    if (key != null) {
219                        // Replace value with filtered value.
220                        // We should just let exceptions bubble out and tank
221                        // the thread creation
222                        table[i + 1] = key.childValue(fromParent.table[i + 1]);
223                    } else {
224                        // The key was reclaimed.
225                        table[i] = TOMBSTONE;
226                        table[i + 1] = null;
227                        fromParent.table[i] = TOMBSTONE;
228                        fromParent.table[i + 1] = null;
229    
230                        tombstones++;
231                        fromParent.tombstones++;
232    
233                        size--;
234                        fromParent.size--;
235                    }
236                }
237            }
238    
239            /**
240             * Creates a new, empty table with the given capacity.
241             */
242            private void initializeTable(int capacity) {
243                this.table = new Object[capacity << 1];
244                this.mask = table.length - 1;
245                this.clean = 0;
246                this.maximumLoad = capacity * 2 / 3; // 2/3
247            }
248    
249            /**
250             * Cleans up after garbage-collected thread locals.
251             */
252            private void cleanUp() {
253                if (rehash()) {
254                    // If we rehashed, we needn't clean up (clean up happens as
255                    // a side effect).
256                    return;
257                }
258    
259                if (size == 0) {
260                    // No live entries == nothing to clean.
261                    return;
262                }
263    
264                // Clean log(table.length) entries picking up where we left off
265                // last time.
266                int index = clean;
267                Object[] table = this.table;
268                for (int counter = table.length; counter > 0; counter >>= 1,
269                        index = next(index)) {
270                    Object k = table[index];
271    
272                    if (k == TOMBSTONE || k == null) {
273                        continue; // on to next entry
274                    }
275    
276                    // The table can only contain null, tombstones and references.
277                    @SuppressWarnings("unchecked")
278                    Reference<ThreadLocal<?>> reference
279                            = (Reference<ThreadLocal<?>>) k;
280                    if (reference.get() == null) {
281                        // This thread local was reclaimed by the garbage collector.
282                        table[index] = TOMBSTONE;
283                        table[index + 1] = null;
284                        tombstones++;
285                        size--;
286                    }
287                }
288    
289                // Point cursor to next index.
290                clean = index;
291            }
292    
293            /**
294             * Rehashes the table, expanding or contracting it as necessary.
295             * Gets rid of tombstones. Returns true if a rehash occurred.
296             * We must rehash every time we fill a null slot; we depend on the
297             * presence of null slots to end searches (otherwise, we'll infinitely
298             * loop).
299             */
300            private boolean rehash() {
301                if (tombstones + size < maximumLoad) {
302                    return false;
303                }
304    
305                int capacity = table.length >> 1;
306    
307                // Default to the same capacity. This will create a table of the
308                // same size and move over the live entries, analogous to a
309                // garbage collection. This should only happen if you churn a
310                // bunch of thread local garbage (removing and reinserting
311                // the same thread locals over and over will overwrite tombstones
312                // and not fill up the table).
313                int newCapacity = capacity;
314    
315                if (size > (capacity >> 1)) {
316                    // More than 1/2 filled w/ live entries.
317                    // Double size.
318                    newCapacity = capacity << 1;
319                }
320    
321                Object[] oldTable = this.table;
322    
323                // Allocate new table.
324                initializeTable(newCapacity);
325    
326                // We won't have any tombstones after this.
327                this.tombstones = 0;
328    
329                // If we have no live entries, we can quit here.
330                if (size == 0) {
331                    return true;
332                }
333    
334                // Move over entries.
335                for (int i = oldTable.length - 2; i >= 0; i -= 2) {
336                    Object k = oldTable[i];
337                    if (k == null || k == TOMBSTONE) {
338                        // Skip this entry.
339                        continue;
340                    }
341    
342                    // The table can only contain null, tombstones and references.
343                    @SuppressWarnings("unchecked")
344                    Reference<ThreadLocal<?>> reference
345                            = (Reference<ThreadLocal<?>>) k;
346                    ThreadLocal<?> key = reference.get();
347                    if (key != null) {
348                        // Entry is still live. Move it over.
349                        add(key, oldTable[i + 1]);
350                    } else {
351                        // The key was reclaimed.
352                        size--;
353                    }
354                }
355    
356                return true;
357            }
358    
359            /**
360             * Adds an entry during rehashing. Compared to put(), this method
361             * doesn't have to clean up, check for existing entries, account for
362             * tombstones, etc.
363             */
364            void add(ThreadLocal<?> key, Object value) {
365                for (int index = key.hash & mask;; index = next(index)) {
366                    Object k = table[index];
367                    if (k == null) {
368                        table[index] = key.reference;
369                        table[index + 1] = value;
370                        return;
371                    }
372                }
373            }
374    
375            /**
376             * Sets entry for given ThreadLocal to given value, creating an
377             * entry if necessary.
378             */
379            void put(ThreadLocal<?> key, Object value) {
380                cleanUp();
381    
382                // Keep track of first tombstone. That's where we want to go back
383                // and add an entry if necessary.
384                int firstTombstone = -1;
385    
386                for (int index = key.hash & mask;; index = next(index)) {
387                    Object k = table[index];
388    
389                    if (k == key.reference) {
390                        // Replace existing entry.
391                        table[index + 1] = value;
392                        return;
393                    }
394    
395                    if (k == null) {
396                        if (firstTombstone == -1) {
397                            // Fill in null slot.
398                            table[index] = key.reference;
399                            table[index + 1] = value;
400                            size++;
401                            return;
402                        }
403    
404                        // Go back and replace first tombstone.
405                        table[firstTombstone] = key.reference;
406                        table[firstTombstone + 1] = value;
407                        tombstones--;
408                        size++;
409                        return;
410                    }
411    
412                    // Remember first tombstone.
413                    if (firstTombstone == -1 && k == TOMBSTONE) {
414                        firstTombstone = index;
415                    }
416                }
417            }
418    
419            /**
420             * Gets value for given ThreadLocal after not finding it in the first
421             * slot.
422             */
423            Object getAfterMiss(ThreadLocal<?> key) {
424                Object[] table = this.table;
425                int index = key.hash & mask;
426    
427                // If the first slot is empty, the search is over.
428                if (table[index] == null) {
429                    Object value = key.initialValue();
430    
431                    // If the table is still the same and the slot is still empty...
432                    if (this.table == table && table[index] == null) {
433                        table[index] = key.reference;
434                        table[index + 1] = value;
435                        size++;
436    
437                        cleanUp();
438                        return value;
439                    }
440    
441                    // The table changed during initialValue().
442                    put(key, value);
443                    return value;
444                }
445    
446                // Keep track of first tombstone. That's where we want to go back
447                // and add an entry if necessary.
448                int firstTombstone = -1;
449    
450                // Continue search.
451                for (index = next(index);; index = next(index)) {
452                    Object reference = table[index];
453                    if (reference == key.reference) {
454                        return table[index + 1];
455                    }
456    
457                    // If no entry was found...
458                    if (reference == null) {
459                        Object value = key.initialValue();
460    
461                        // If the table is still the same...
462                        if (this.table == table) {
463                            // If we passed a tombstone and that slot still
464                            // contains a tombstone...
465                            if (firstTombstone > -1
466                                    && table[firstTombstone] == TOMBSTONE) {
467                                table[firstTombstone] = key.reference;
468                                table[firstTombstone + 1] = value;
469                                tombstones--;
470                                size++;
471    
472                                // No need to clean up here. We aren't filling
473                                // in a null slot.
474                                return value;
475                            }
476    
477                            // If this slot is still empty...
478                            if (table[index] == null) {
479                                table[index] = key.reference;
480                                table[index + 1] = value;
481                                size++;
482    
483                                cleanUp();
484                                return value;
485                            }
486                        }
487    
488                        // The table changed during initialValue().
489                        put(key, value);
490                        return value;
491                    }
492    
493                    if (firstTombstone == -1 && reference == TOMBSTONE) {
494                        // Keep track of this tombstone so we can overwrite it.
495                        firstTombstone = index;
496                    }
497                }
498            }
499    
500            /**
501             * Removes entry for the given ThreadLocal.
502             */
503            void remove(ThreadLocal<?> key) {
504                cleanUp();
505    
506                for (int index = key.hash & mask;; index = next(index)) {
507                    Object reference = table[index];
508    
509                    if (reference == key.reference) {
510                        // Success!
511                        table[index] = TOMBSTONE;
512                        table[index + 1] = null;
513                        tombstones++;
514                        size--;
515                        return;
516                    }
517    
518                    if (reference == null) {
519                        // No entry found.
520                        return;
521                    }
522                }
523            }
524    
525            /**
526             * Gets the next index. If we're at the end of the table, we wrap back
527             * around to 0.
528             */
529            private int next(int index) {
530                return (index + 2) & mask;
531            }
532        }
