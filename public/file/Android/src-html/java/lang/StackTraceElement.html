018    package java.lang;
019    
020    import java.io.Serializable;
021    
022    /**
023     * A representation of a single stack frame. Arrays of {@code StackTraceElement}
024     * are stored in {@link Throwable} objects to represent the whole state of the
025     * call stack at the time a {@code Throwable} gets thrown.
026     * 
027     * @see Throwable#getStackTrace()
028     * @since Android 1.0
029     */
030    public final class StackTraceElement implements Serializable {
031    
032        private static final long serialVersionUID = 6992337162326171013L;
033    
034        // BEGIN android-added
035        private static final int NATIVE_LINE_NUMBER = -2;
036        // END android-added
037        
038        String declaringClass;
039    
040        String methodName;
041    
042        String fileName;
043    
044        int lineNumber;
045    
046        /**
047         * Constructs a new {@code StackTraceElement} for a specified execution
048         * point.
049         * 
050         * @param cls
051         *            the fully qualified name of the class where execution is at.
052         * @param method
053         *            the name of the method where execution is at.
054         * @param file
055         *            The name of the file where execution is at or {@code null}.
056         * @param line
057         *            the line of the file where execution is at, a negative number
058         *            if unknown or {@code -2} if the execution is in a native
059         *            method.
060         * @throws NullPointerException
061         *             if {@code cls} or {@code method} is {@code null}.
062         * @since Android 1.0
063         */
064        public StackTraceElement(String cls, String method, String file, int line) {
065            super();
066            if (cls == null || method == null) {
067                throw new NullPointerException();
068            }
069            declaringClass = cls;
070            methodName = method;
071            fileName = file;
072            lineNumber = line;
073        }
074    
075        /**
076         * <p>
077         * Private, nullary constructor for VM use only.
078         * </p>
079         */
080        private StackTraceElement() {
081            super();
082        }
083    
084        /**
085         * Compares this instance with the specified object and indicates if they
086         * are equal. In order to be equal, the following conditions must be
087         * fulfilled:
088         * <ul>
089         * <li>{@code obj} must be a stack trace element,</li>
090         * <li>the method names of this stack trace element and of {@code obj} must
091         * not be {@code null},</li>
092         * <li>the class, method and file names as well as the line number of this
093         * stack trace element and of {@code obj} must be equal.</li>
094         * </ul>
095         * 
096         * @param obj
097         *            the object to compare this instance with.
098         * @return {@code true} if the specified object is equal to this
099         *         {@code StackTraceElement}; {@code false} otherwise.
100         * @see #hashCode
101         * @since Android 1.0
102         */
103        @Override
104        public boolean equals(Object obj) {
105            if (!(obj instanceof StackTraceElement)) {
106                return false;
107            }
108            StackTraceElement castObj = (StackTraceElement) obj;
109    
110            /*
111             * Unknown methods are never equal to anything (not strictly to spec,
112             * but spec does not allow null method/class names)
113             */
114            if ((methodName == null) || (castObj.methodName == null)) {
115                return false;
116            }
117    
118            if (!getMethodName().equals(castObj.getMethodName())) {
119                return false;
120            }
121            if (!getClassName().equals(castObj.getClassName())) {
122                return false;
123            }
124            String localFileName = getFileName();
125            if (localFileName == null) {
126                if (castObj.getFileName() != null) {
127                    return false;
128                }
129            } else {
130                if (!localFileName.equals(castObj.getFileName())) {
131                    return false;
132                }
133            }
134            if (getLineNumber() != castObj.getLineNumber()) {
135                return false;
136            }
137    
138            return true;
139        }
140    
141        /**
142         * Returns the fully qualified name of the class belonging to this
143         * {@code StackTraceElement}.
144         * 
145         * @return the fully qualified type name of the class
146         * @since Android 1.0
147         */
148        public String getClassName() {
149            return (declaringClass == null) ? "<unknown class>" : declaringClass;
150        }
151    
152        /**
153         * Returns the name of the Java source file containing class belonging to
154         * this {@code StackTraceElement}.
155         * 
156         * @return the name of the file, or {@code null} if this information is not
157         *         available.
158         * @since Android 1.0
159         */
160        public String getFileName() {
161            return fileName;
162        }
163    
164        /**
165         * Returns the line number in the source for the class belonging to this
166         * {@code StackTraceElement}.
167         * 
168         * @return the line number, or a negative number if this information is not
169         *         available.
170         * @since Android 1.0
171         */
172        public int getLineNumber() {
173            return lineNumber;
174        }
175    
176        /**
177         * Returns the name of the method belonging to this {@code
178         * StackTraceElement}.
179         * 
180         * @return the name of the method, or "<unknown method>" if this information
181         *         is not available.
182         * @since Android 1.0
183         */
184        public String getMethodName() {
185            return (methodName == null) ? "<unknown method>" : methodName;
186        }
187    
188        @Override
189        public int hashCode() {
190            /*
191             * Either both methodName and declaringClass are null, or neither are
192             * null.
193             */
194            if (methodName == null) {
195                // all unknown methods hash the same
196                return 0;
197            }
198            // declaringClass never null if methodName is non-null
199            return methodName.hashCode() ^ declaringClass.hashCode();
200        }
201    
202        /**
203         * Indicates if the method name returned by {@link #getMethodName()} is
204         * implemented as a native method.
205         * 
206         * @return {@code true} if the method in which this stack trace element is
207         *         executing is a native method; {@code false} otherwise.
208         * @since Android 1.0
209         */
210        public boolean isNativeMethod() {
211            // BEGIN android-changed
212            return lineNumber == NATIVE_LINE_NUMBER;
213            // END android-changed
214        }
215    
216        @Override
217        public String toString() {
218            StringBuilder buf = new StringBuilder(80);
219    
220            buf.append(getClassName());
221            buf.append('.');
222            buf.append(getMethodName());
223    
224            if (isNativeMethod()) {
225                buf.append("(Native Method)");
226            } else {
227                String fName = getFileName();
228    
229                if (fName == null) {
230                    buf.append("(Unknown Source)");
231                } else {
232                    int lineNum = getLineNumber();
233    
234                    buf.append('(');
235                    buf.append(fName);
236                    if (lineNum >= 0) {
237                        buf.append(':');
238                        buf.append(lineNum);
239                    }
240                    buf.append(')');
241                }
242            }
243            return buf.toString();
244        }
245    }
