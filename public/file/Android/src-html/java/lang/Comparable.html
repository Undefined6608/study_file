018    package java.lang;
019    
020    /**
021     * This interface should be implemented by all classes that wish to define a
022     * <em>natural order</em> of their instances.
023     * {@link java.util.Collections#sort} and {@code java.util.Arrays#sort} can then
024     * be used to automatically sort lists of classes that implement this interface.
025     * <p>
026     * The order rule must be both transitive (if {@code x.compareTo(y) < 0} and
027     * {@code y.compareTo(z) < 0}, then {@code x.compareTo(z) < 0} must hold) and
028     * invertible (the sign of the result of x.compareTo(y) must be equal to the
029     * negation of the sign of the result of y.compareTo(x) for all combinations of
030     * x and y).
031     * </p>
032     * <p>
033     * In addition, it is recommended (but not required) that if and only if the
034     * result of x.compareTo(y) is zero, then the result of x.equals(y) should be
035     * {@code true}.
036     * </p>
037     * 
038     * @since Android 1.0
039     */
040    public interface Comparable<T> {
041        
042        /**
043         * Compares this object to the specified object to determine their relative
044         * order.
045         * 
046         * @param another
047         *            the object to compare to this instance.
048         * @return a negative integer if this instance is less than {@code another};
049         *         a positive integer if this instance is greater than
050         *         {@code another}; 0 if this instance has the same order as
051         *         {@code another}.
052         * @throws ClassCastException
053         *             if {@code another} cannot be converted into something
054         *             comparable to {@code this} instance.
055         * @since Android 1.0
056         */
057        int compareTo(T another);
058    }
