033    package java.lang.ref;
034    
035    /**
036     * Implements a weak reference, which is the middle of the three types of
037     * references. Once the garbage collector decides that an object {@code obj} is
038     * is weakly-reachable, the following
039     * happens:
040     * 
041     * <ul>
042     *   <li>
043     *     A set {@code ref} of references is determined. {@code ref} contains the
044     *     following elements:
045     *     <ul>
046     *       <li>
047     *         All weak references pointing to {@code obj}.
048     *       </li>
049     *       <li>
050     *         All weak references pointing to objects from which {@code obj} is
051     *         either strongly or softly reachable.
052     *       </li>
053     *     </ul>
054     *   </li>
055     *   <li>
056     *     All references in {@code ref} are atomically cleared.
057     *   </li>
058     *   <li>
059     *     All objects formerly being referenced by {@code ref} become eligible for
060     *     finalization.
061     *   </li>
062     *   <li>
063     *     At some future point, all references in {@code ref} will be enqueued
064     *     with their corresponding reference queues, if any.
065     *   </li>
066     * </ul>
067     * 
068     * Weak references are useful for mappings that should have their entries
069     * removed automatically once they are not referenced any more (from outside).
070     * The difference between a {@code SoftReference} and a {@code WeakReference} is
071     * the point of time at which the decision is made to clear and enqueue the
072     * reference:
073     * 
074     * <ul>
075     *   <li>
076     *     A {@code SoftReference} should be cleared and enqueued <em>as late as
077     *     possible</em>, that is, in case the VM is in danger of running out of
078     *     memory.
079     *   </li>
080     *   <li>
081     *     A {@code WeakReference} may be cleared and enqueued as soon as is
082     *     known to be weakly-referenced.  
083     *   </li>
084     * </ul>
085     * 
086     * @since Android 1.0
087     */
088    public class WeakReference<T> extends Reference<T> {
089    
090        /**
091         * Constructs a new weak reference to the given referent. The newly created
092         * reference is not registered with any reference queue.
093         * 
094         * @param r the referent to track
095         * 
096         * @since Android 1.0
097         */
098        public WeakReference(T r) {
099            super();
100            referent = r;
101        }
102    
103        /**
104         * Constructs a new weak reference to the given referent. The newly created
105         * reference is registered with the given reference queue.
106         * 
107         * @param r the referent to track
108         * @param q the queue to register to the reference object with. A null value
109         *          results in a weak reference that is not associated with any
110         *          queue.
111         * 
112         * @since Android 1.0
113         */
114        public WeakReference(T r, ReferenceQueue<? super T> q) {
115            super();
116            referent = r;
117            queue = q;
118        }
119    }
