021    package java.lang.ref;
022    
023    /**
024     * The {@code ReferenceQueue} is the container on which reference objects are
025     * enqueued when the garbage collector detects the reachability type specified
026     * for the referent.
027     * 
028     * @since Android 1.0
029     */
030    public class ReferenceQueue<T> {
031        
032        private Reference<? extends T> head;
033    
034        /**
035         * Constructs a new instance of this class.
036         * 
037         * @since Android 1.0
038         */
039        public ReferenceQueue() {
040            super();
041        }
042    
043        /**
044         * Returns the next available reference from the queue, removing it in the
045         * process. Does not wait for a reference to become available.
046         * 
047         * @return the next available reference, or {@code null} if no reference is
048         *         immediately available
049         *            
050         * @since Android 1.0
051         */
052        @SuppressWarnings("unchecked")
053        public synchronized Reference<? extends T> poll() {
054            if (head == null) {
055                return null;
056            }
057    
058            Reference<? extends T> ret;
059    
060            ret = head;
061    
062            if (head == head.queueNext) {
063                head = null;
064            } else {
065                head = head.queueNext;
066            }
067    
068            ret.queueNext = null;
069    
070            return ret;
071        }
072    
073        /**
074         * Returns the next available reference from the queue, removing it in the
075         * process. Waits indefinitely for a reference to become available.
076         * 
077         * @return the next available reference
078         * 
079         * @throws InterruptedException
080         *             if the blocking call was interrupted for some reason  
081         *            
082         * @since Android 1.0
083         */
084        public Reference<? extends T> remove() throws InterruptedException {
085            return remove(0L);
086        }
087        
088        /**
089         * Returns the next available reference from the queue, removing it in the
090         * process. Waits for a reference to become available or the given timeout
091         * period to elapse, whichever happens first.
092         * 
093         * @param timeout
094         *            maximum time (in ms) to spend waiting for a reference object
095         *            to become available. A value of zero results in the method
096         *            waiting indefinitely.
097         *            
098         * @return the next available reference, or {@code null} if no reference
099         *         becomes available within the timeout period
100         * 
101         * @throws IllegalArgumentException
102         *             if the wait period is negative.
103         * @throws InterruptedException
104         *             if the blocking call was interrupted for some reason  
105         *            
106         * @since Android 1.0
107         */
108        public synchronized Reference<? extends T> remove(long timeout) throws IllegalArgumentException,
109                InterruptedException {
110            if (timeout < 0) {
111                throw new IllegalArgumentException();
112            }
113    
114            if (timeout == 0L) {
115                while (head == null) {
116                    wait(0L);
117                }
118            } else {
119                long now = System.currentTimeMillis();
120                long wakeupTime = now + timeout + 1L;
121                while (head == null && now < wakeupTime) {
122                    wait(wakeupTime - now);
123                    now = System.currentTimeMillis();
124                }
125            }
126    
127            return poll();
128        }
129    
130        /**
131         * Enqueues the reference object on the receiver.
132         * 
133         * @param toQueue
134         *            reference object to be enqueued.
135         *            
136         * @since Android 1.0
137         */
138        synchronized void enqueue(Reference<? extends T> toQueue) {
139            if (head == null) {
140                toQueue.queueNext = toQueue;
141            } else {
142                toQueue.queueNext = head;
143            }
144            head = toQueue;
145            notify();
146        }
147    }
