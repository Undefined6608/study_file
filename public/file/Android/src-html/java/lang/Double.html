020    /**
021     * The wrapper for the primitive type {@code double}.
022     * 
023     * @see java.lang.Number
024     * @since Android 1.0
025     */
026    public final class Double extends Number implements Comparable<Double> {
027    
028        private static final long serialVersionUID = -9172774392245257468L;
029    
030        /**
031         * The value which the receiver represents.
032         */
033        private final double value;
034    
035        /**
036         * Constant for the maximum {@code double} value, (2 - 2<sup>-52</sup>) *
037         * 2<sup>1023</sup>.
038         * 
039         * @since Android 1.0
040         */
041        public static final double MAX_VALUE = 1.79769313486231570e+308;
042    
043        /**
044         * Constant for the minimum {@code double} value, 2<sup>-1074</sup>.
045         * 
046         * @since Android 1.0
047         */
048        public static final double MIN_VALUE = 5e-324;

052        /**
053         * Constant for the Not-a-Number (NaN) value of the {@code double} type.
054         * 
055         * @since Android 1.0

057        public static final double NaN = 0.0 / 0.0;

059        /**
060         * Constant for the Positive Infinity value of the {@code double} type.
061         *

063         */
064        public static final double POSITIVE_INFINITY = 1.0 / 0.0;

066        /**
067         * Constant for the Negative Infinity value of the {@code double} type.
068         * 
069         * @since Android 1.0

071        public static final double NEGATIVE_INFINITY = -1.0 / 0.0;

073        /**
074         * The {@link Class} object that represents the primitive type {@code
075         * double}.
076         * 
077         * @since Android 1.0

081                .getClass().getComponentType();

086        /**
087         * Constant for the number of bits needed to represent a {@code double} in
088         * two's complement form.
089         *

092        public static final int SIZE = 64;

094        /**
095         * Constructs a new {@code Double} with the specified primitive double
096         * value.
097         * 
098         * @param value
099         *            the primitive double value to store in the new instance.
100         * @since Android 1.0
101         */
102        public Double(double value) {
103            this.value = value;
104        }
105    
106        /**
107         * Constructs a new {@code Double} from the specified string.
108         * 
109         * @param string
110         *            the string representation of a double value.
111         * @throws NumberFormatException
112         *             if {@code string} can not be decoded into a double value.
113         * @see #parseDouble(String)
114         * @since Android 1.0
115         */
116        public Double(String string) throws NumberFormatException {
117            this(parseDouble(string));
118        }
119    
120        /**
121         * Compares this object to the specified double object to determine their
122         * relative order. There are two special cases:
123         * <ul>
124         * <li>{@code Double.NaN} is equal to {@code Double.NaN} and it is greater
125         * than any other double value, including {@code Double.POSITIVE_INFINITY};</li>
126         * <li>+0.0d is greater than -0.0d</li>
127         * </ul>
128         * 
129         * @param object
130         *            the double object to compare this object to.
131         * @return a negative value if the value of this double is less than the
132         *         value of {@code object}; 0 if the value of this double and the
133         *         value of {@code object} are equal; a positive value if the value
134         *         of this double is greater than the value of {@code object}.
135         * @see java.lang.Comparable
136         * @since Android 1.0
137         */
138        public int compareTo(Double object) {
139            long d1, d2;
140            long NaNbits = Double.doubleToLongBits(Double.NaN);
141            if ((d1 = Double.doubleToLongBits(value)) == NaNbits) {
142                if (Double.doubleToLongBits(object.value) == NaNbits) {
143                    return 0;
144                }
145                return 1;
146            }
147            if ((d2 = Double.doubleToLongBits(object.value)) == NaNbits) {
148                return -1;
149            }
150            if (value == object.value) {
151                if (d1 == d2) {
152                    return 0;
153                }
154                // check for -0
155                return d1 > d2 ? 1 : -1;
156            }
157            return value > object.value ? 1 : -1;
158        }

165        /**
166         * Converts the specified double value to a binary representation conforming
167         * to the IEEE 754 floating-point double precision bit layout. All
168         * <em>Not-a-Number (NaN)</em> values are converted to a single NaN
169         * representation ({@code 0x7ff8000000000000L}).
170         * 
171         * @param value
172         *            the double value to convert.
173         * @return the IEEE 754 floating-point double precision representation of
174         *         {@code value}.
175         * @see #doubleToRawLongBits(double)
176         * @see #longBitsToDouble(long)
177         * @since Android 1.0
178         */
179        public static native long doubleToLongBits(double value);
180    
181        /**
182         * Converts the specified double value to a binary representation conforming
183         * to the IEEE 754 floating-point double precision bit layout.
184         * <em>Not-a-Number (NaN)</em> values are preserved.
185         * 
186         * @param value
187         *            the double value to convert.
188         * @return the IEEE 754 floating-point double precision representation of
189         *         {@code value}.
190         * @see #doubleToLongBits(double)
191         * @see #longBitsToDouble(long)
192         * @since Android 1.0
193         */
194        public static native long doubleToRawLongBits(double value);

197         * Gets the primitive value of this double.
198         * 
199         * @return this object's primitive value.
200         * @since Android 1.0

207        /**
208         * Compares this object with the specified object and indicates if they are
209         * equal. In order to be equal, {@code object} must be an instance of
210         * {@code Double} and the bit pattern of its double value is the same as
211         * this object's.
212         * 
213         * @param object
214         *            the object to compare this double with.
215         * @return {@code true} if the specified object is equal to this
216         *         {@code Double}; {@code false} otherwise.
217         * @since Android 1.0
218         */
219        @Override
220        public boolean equals(Object object) {
221            return (object == this)
222                    || (object instanceof Double)
223                    && (doubleToLongBits(this.value) == doubleToLongBits(((Double) object).value));
224        }

367        /**
368         * Compares the two specified double values. There are two special cases:
369         * <ul>
370         * <li>{@code Double.NaN} is equal to {@code Double.NaN} and it is greater
371         * than any other double value, including {@code Double.POSITIVE_INFINITY};</li>
372         * <li>+0.0d is greater than -0.0d</li>
373         * </ul>
374         * 
375         * @param double1
376         *            the first value to compare.
377         * @param double2
378         *            the second value to compare.
379         * @return a negative value if {@code double1} is less than {@code double2};
380         *         0 if {@code double1} and {@code double2} are equal; a positive
381         *         value if {@code double1} is greater than {@code double2}.
382         * @since Android 1.0         
383         */
384        public static int compare(double double1, double double2) {
385            long d1, d2;
386            long NaNbits = Double.doubleToLongBits(Double.NaN);
387            if ((d1 = Double.doubleToLongBits(double1)) == NaNbits) {
388                if (Double.doubleToLongBits(double2) == NaNbits) {
389                    return 0;
390                }
391                return 1;
392            }
393            if ((d2 = Double.doubleToLongBits(double2)) == NaNbits) {
394                return -1;
395            }
396            if (double1 == double2) {
397                if (d1 == d2) {
398                    return 0;
399                }
400                // check for -0
401                return d1 > d2 ? 1 : -1;
402            }
403            return double1 > double2 ? 1 : -1;
404        }

418        /**
419         * Converts the specified double into its hexadecimal string representation.
420         * 
421         * @param d
422         *            the double to convert.
423         * @return the hexadecimal string representation of {@code d}.
424         * @since Android 1.0
425         */
426        public static String toHexString(double d) {
427            /*
428             * Reference: http://en.wikipedia.org/wiki/IEEE_754
429             */
430            if (d != d) {
431                return "NaN"; //$NON-NLS-1$
432            }
433            if (d == POSITIVE_INFINITY) {
434                return "Infinity"; //$NON-NLS-1$
435            }
436            if (d == NEGATIVE_INFINITY) {
437                return "-Infinity"; //$NON-NLS-1$
438            }
439    
440            long bitValue = doubleToLongBits(d);
441    
442            boolean negative = (bitValue & 0x8000000000000000L) != 0;
443            // mask exponent bits and shift down
444            long exponent = (bitValue & 0x7FF0000000000000L) >>> 52;
445            // mask significand bits and shift up
446            long significand = bitValue & 0x000FFFFFFFFFFFFFL;
447    
448            if (exponent == 0 && significand == 0) {
449                return (negative ? "-0x0.0p0" : "0x0.0p0"); //$NON-NLS-1$ //$NON-NLS-2$
450            }
451    
452            StringBuilder hexString = new StringBuilder(10);
453            if (negative) {
454                hexString.append("-0x"); //$NON-NLS-1$
455            } else {
456                hexString.append("0x"); //$NON-NLS-1$
457            }
458    
459            if (exponent == 0) { // denormal (subnormal) value
460                hexString.append("0."); //$NON-NLS-1$
461                // significand is 52-bits, so there can be 13 hex digits
462                int fractionDigits = 13;
463                // remove trailing hex zeros, so Integer.toHexString() won't print
464                // them
465                while ((significand != 0) && ((significand & 0xF) == 0)) {
466                    significand >>>= 4;
467                    fractionDigits--;
468                }
469                // this assumes Integer.toHexString() returns lowercase characters
470                String hexSignificand = Long.toHexString(significand);
471    
472                // if there are digits left, then insert some '0' chars first
473                if (significand != 0 && fractionDigits > hexSignificand.length()) {
474                    int digitDiff = fractionDigits - hexSignificand.length();
475                    while (digitDiff-- != 0) {
476                        hexString.append('0');
477                    }
478                }
479                hexString.append(hexSignificand);
480                hexString.append("p-1022"); //$NON-NLS-1$
481            } else { // normal value
482                hexString.append("1."); //$NON-NLS-1$
483                // significand is 52-bits, so there can be 13 hex digits
484                int fractionDigits = 13;
485                // remove trailing hex zeros, so Integer.toHexString() won't print
486                // them
487                while ((significand != 0) && ((significand & 0xF) == 0)) {
488                    significand >>>= 4;
489                    fractionDigits--;
490                }
491                // this assumes Integer.toHexString() returns lowercase characters
492                String hexSignificand = Long.toHexString(significand);
493    
494                // if there are digits left, then insert some '0' chars first
495                if (significand != 0 && fractionDigits > hexSignificand.length()) {
496                    int digitDiff = fractionDigits - hexSignificand.length();
497                    while (digitDiff-- != 0) {
498                        hexString.append('0');
499                    }
500                }
501    
502                hexString.append(hexSignificand);
503                hexString.append('p');
504                // remove exponent's 'bias' and convert to a string
505                hexString.append(Long.toString(exponent - 1023));
506            }
507            return hexString.toString();
508        }
