051        /**
052         * Returns the absolute value of the argument.
053         * <p>
054         * Special cases:
055         * <ul>
056         * <li>{@code abs(-0.0) = +0.0}</li>
057         * <li>{@code abs(+infinity) = +infinity}</li>
058         * <li>{@code abs(-infinity) = +infinity}</li>
059         * <li>{@code abs(NaN) = NaN}</li>
060         * </ul>
061         * </p>
062         * 
063         * @param d
064         *            the value whose absolute value has to be computed.
065         * @return the absolute value of the argument.
066         * @since Android 1.0
067         */
068        public static double abs(double d) {
069            long bits = Double.doubleToLongBits(d);
070            bits &= 0x7fffffffffffffffL;
071            return Double.longBitsToDouble(bits);
072        }

520        /**
521         * Returns the most positive (closest to positive infinity) of the two
522         * arguments.
523         * <p>
524         * Special cases:
525         * <ul>
526         * <li>{@code max(NaN, (anything)) = NaN}</li>
527         * <li>{@code max((anything), NaN) = NaN}</li>
528         * <li>{@code max(+0.0, -0.0) = +0.0}</li>
529         * <li>{@code max(-0.0, +0.0) = +0.0}</li>
530         * </ul>
531         * </p>
532         * 
533         * @param d1
534         *            the first argument.
535         * @param d2
536         *            the second argument.
537         * @return the larger of {@code d1} and {@code d2}.
538         * @since Android 1.0
539         */
540        public static double max(double d1, double d2) {
541            if (d1 > d2)
542                return d1;
543            if (d1 < d2)
544                return d2;
545            /* if either arg is NaN, return NaN */
546            if (d1 != d2)
547                return Double.NaN;
548            /* max( +0.0,-0.0) == +0.0 */
549            if (d1 == 0.0
550                    && ((Double.doubleToLongBits(d1) & Double.doubleToLongBits(d2)) & 0x8000000000000000L) == 0)
551                return 0.0;
552            return d1;
553        }

566         * </p>
567         * 
568         * @param f1

585                    && ((Float.floatToIntBits(f1) & Float.floatToIntBits(f2)) & 0x80000000) == 0)

641            if (d1 > d2)
642                return d2;
643            if (d1 < d2)
644                return d1;
645            /* if either arg is NaN, return NaN */
646            if (d1 != d2)
647                return Double.NaN;
648            /* min( +0.0,-0.0) == -0.0 */
649            if (d1 == 0.0
650                    && ((Double.doubleToLongBits(d1) | Double.doubleToLongBits(d2)) & 0x8000000000000000l) != 0)
651                return 0.0 * (-1.0);
652            return d1;
653        }
