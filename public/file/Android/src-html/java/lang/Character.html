032    /**
033     * The wrapper for the primitive type {@code char}. This class also provides a
034     * number of utility methods for working with characters.
035     * <p>
036     * Character data is based upon the Unicode Standard, 4.0. The Unicode
037     * specification, character tables and other information are available at <a
038     * href="http://www.unicode.org/">http://www.unicode.org/</a>.
039     * </p>
040     * <p>
041     * Unicode characters are referred to as <i>code points</i>. The range of valid
042     * code points is U+0000 to U+10FFFF. The <i>Basic Multilingual Plane (BMP)</i>
043     * is the code point range U+0000 to U+FFFF. Characters above the BMP are
044     * referred to as <i>Supplementary Characters</i>. On the Java platform, UTF-16
045     * encoding and {@code char} pairs are used to represent code points in the
046     * supplementary range. A pair of {@code char} values that represent a
047     * supplementary character are made up of a <i>high surrogate</i> with a value
048     * range of 0xD800 to 0xDBFF and a <i>low surrogate</i> with a value range of
049     * 0xDC00 to 0xDFFF.
050     * </p>
051     * <p>
052     * On the Java platform a {@code char} value represents either a single BMP code
053     * point or a UTF-16 unit that's part of a surrogate pair. The {@code int} type
054     * is used to represent all Unicode code points.
055     * </p>
056     * 
057     * @since Android 1.0
058     */

566         * 
567         * @since Android 1.0

570            String name;
571    
572            /**
573             * Constructs a new {@code Subset}.
574             * 
575             * @param string
576             *            this subset's name.
577             * @since Android 1.0
578             */
579            protected Subset(String string) {
580                if (string == null) {
581                    throw new NullPointerException();
582                }
583                name = string;
584            }
585    
586            /**
587             * Compares this character subset with the specified object. Uses
588             * {@link java.lang.Object#equals(Object)} to do the comparison.
589             * 
590             * @param object
591             *            the object to compare this character subset with.
592             * @return {@code true} if {@code object} is this subset, that is, if
593             *         {@code object == this}; {@code false} otherwise.
594             * @since Android 1.0
595             */
596            @Override
597            public final boolean equals(Object object) {
598                return super.equals(object);
599            }
600    
601            /**
602             * Returns the integer hash code for this character subset.
603             * 
604             * @return this subset's hash code, which is the hash code computed by
605             *         {@link java.lang.Object#hashCode()}.
606             * @since Android 1.0
607             */
608            @Override
609            public final int hashCode() {
610                return super.hashCode();
611            }
612    
613            /**
614             * Returns the string representation of this subset.
615             * 
616             * @return this subset's name.
617             * @since Android 1.0
618             */
619            @Override
620            public final String toString() {
621                return name;
622            }
623        }

1963         * @throws IllegalArgumentException
1964         *             if {@code codePoint} is not a valid Unicode code point.
1965         * @throws IndexOutOfBoundsException

1967         *             {@code dst.length} or equals {@code dst.length - 1} when
1968         *             {@code codePoint} is a
1969         *             {@link #isSupplementaryCodePoint(int) supplementary code point}.
1970         * @since Android 1.0
1971         */
1972        public static int toChars(int codePoint, char[] dst, int dstIndex) {
1973            if (!isValidCodePoint(codePoint)) {
1974                throw new IllegalArgumentException();
1975            }
1976            if (dst == null) {
1977                throw new NullPointerException();
1978            }
1979            if (dstIndex < 0 || dstIndex >= dst.length) {
1980                throw new IndexOutOfBoundsException();
1981            }
1982    
1983            if (isSupplementaryCodePoint(codePoint)) {
1984                if (dstIndex == dst.length - 1) {
1985                    throw new IndexOutOfBoundsException();
1986                }
1987                // See RFC 2781, Section 2.1
