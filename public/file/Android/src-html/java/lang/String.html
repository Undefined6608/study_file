295        /**
296         * Converts the byte array to a string using the specified encoding.
297         * 
298         * @param data
299         *            the byte array to convert to a string.
300         * @param start
301         *            the starting offset in the byte array.
302         * @param length
303         *            the number of bytes to convert.
304         * @param encoding
305         *            the encoding.
306         * @throws IndexOutOfBoundsException
307         *             if {@code length < 0, start < 0} or {@code start + length >
308         *             data.length}.
309         * @throws UnsupportedEncodingException
310         *             if {@code encoding} is not supported.
311         * @since Android 1.0
312         */
313        public String(byte[] data, int start, int length, final String encoding)
314                throws UnsupportedEncodingException {
315            if (encoding == null) {
316                throw new NullPointerException();
317            }
318            // start + length could overflow, start/length maybe MaxInt
319            if (start >= 0 && 0 <= length && length <= data.length - start) {
320                offset = 0;
321                // BEGIN android-added
322                // Special-case ISO-88589-1 and UTF 8 decoding
323                if (encoding.equalsIgnoreCase("ISO-8859-1") ||
324                    encoding.equalsIgnoreCase("ISO8859_1")) {
325                    value = new char[length];
326                    count = length;
327                    for (int i = 0; i < count; i++) {
328                        value[i] = (char) (data[start++] & 0xff);
329                    }
330                    return;
331                } else if ("utf8".equals(encoding) ||
332                           "utf-8".equals(encoding) ||
333                           "UTF8".equals(encoding) ||
334                           "UTF-8".equals(encoding)) {
335                    // We inline UTF8 decoding for speed and because a
336                    // non-constructor can't write directly to the final
337                    // members 'value' or 'count'.
338                    byte[] d = data;
339                    char[] v = new char[length];
340    
341                    int idx = start, last = start + length, s = 0;
342                    outer:
343                    while (idx < last) {
344                        byte b0 = d[idx++];
345                        if ((b0 & 0x80) == 0) {
346                            // 0xxxxxxx
347                            // Range:  U-00000000 - U-0000007F
348                            int val = b0 & 0xff;
349                            v[s++] = (char) val;
350                        } else if (((b0 & 0xe0) == 0xc0) ||
351                                   ((b0 & 0xf0) == 0xe0) ||
352                                   ((b0 & 0xf8) == 0xf0) ||
353                                   ((b0 & 0xfc) == 0xf8) ||
354                                   ((b0 & 0xfe) == 0xfc)) {
355                            int utfCount = 1;
356                            if ((b0 & 0xf0) == 0xe0) utfCount = 2;
357                            else if ((b0 & 0xf8) == 0xf0) utfCount = 3;
358                            else if ((b0 & 0xfc) == 0xf8) utfCount = 4;
359                            else if ((b0 & 0xfe) == 0xfc) utfCount = 5;
360    
361                            // 110xxxxx (10xxxxxx)+
362                            // Range:  U-00000080 - U-000007FF (count == 1)
363                            // Range:  U-00000800 - U-0000FFFF (count == 2)
364                            // Range:  U-00010000 - U-001FFFFF (count == 3)
365                            // Range:  U-00200000 - U-03FFFFFF (count == 4)
366                            // Range:  U-04000000 - U-7FFFFFFF (count == 5)
367    
368                            if (idx + utfCount > last) {
369                                v[s++] = REPLACEMENT_CHAR;
370                                break;
371                            }
372    
373                            // Extract usable bits from b0
374                            int val = b0 & (0x1f >> (utfCount - 1));
375                            for (int i = 0; i < utfCount; i++) {
376                                byte b = d[idx++];
377                                if ((b & 0xC0) != 0x80) {
378                                    v[s++] = REPLACEMENT_CHAR;
379                                    idx--; // Put the input char back
380                                    continue outer;
381                                }
382                                // Push new bits in from the right side
383                                val <<= 6;
384                                val |= b & 0x3f;
385                            }
386    
387                            // Note: Java allows overlong char
388                            // specifications To disallow, check that val
389                            // is greater than or equal to the minimum
390                            // value for each count:
391                            //
392                            // count    min value
393                            // -----   ----------
394                            //   1           0x80
395                            //   2          0x800
396                            //   3        0x10000
397                            //   4       0x200000
398                            //   5      0x4000000
399    
400                            // Allow surrogate values (0xD800 - 0xDFFF) to
401                            // be specified using 3-byte UTF values only
402                            if ((utfCount != 2) &&
403                                (val >= 0xD800) && (val <= 0xDFFF)) {
404                                v[s++] = REPLACEMENT_CHAR;
405                                continue;
406                            }
407    
408                            // Reject chars greater than the Unicode
409                            // maximum of U+10FFFF
410                            if (val > 0x10FFFF) {
411                                v[s++] = REPLACEMENT_CHAR;
412                                continue;
413                            }
414    
415                            // Encode chars from U+10000 up as surrogate pairs
416                            if (val < 0x10000) {
417                                v[s++] = (char) val;
418                            } else {
419                                int x = val & 0xffff;
420                                int u = (val >> 16) & 0x1f;
421                                int w = (u - 1) & 0xffff;
422                                int hi = 0xd800 | (w << 6) | (x >> 10);
423                                int lo = 0xdc00 | (x & 0x3ff);
424                                v[s++] = (char) hi;
425                                v[s++] = (char) lo;
426                            }
427                        } else {
428                            // Illegal values 0x8*, 0x9*, 0xa*, 0xb*, 0xfd-0xff
429                            v[s++] = REPLACEMENT_CHAR;
430                        }
431                    }
432    
433                    // Reallocate the array to fit the contents
434                    count = s;
435                    value = new char[s];
436                    System.arraycopy(v, 0, value, 0, s);
437                    return;
438                }
439                // END android-added
440                Charset charset = getCharset(encoding);
441    
442                int result;
443                CharBuffer cb;
444                try {
445                    cb = charset.decode(ByteBuffer.wrap(data, start, length));
446                } catch (Exception e) {
447                    // do nothing. according to spec: 
448                    // behavior is unspecified for invalid array
449                    cb = CharBuffer.wrap("\u003f".toCharArray()); //$NON-NLS-1$
450                }
451                if ((result = cb.length()) > 0) {
452                    value = cb.array();
453                    count = result;
454                } else {
455                    count = 0;
456                    value = new char[0];
457                }
458            } else {
459                throw new StringIndexOutOfBoundsException();
460            }
461        }

564         * @param offset
565         *            the inclusive index into {@code codePoints} to begin

573         *             if {@code offset} or {@code count} are not within the bounds

578            super();
579            if (codePoints == null) {
580                throw new NullPointerException();
581            }
582            if (offset < 0 || count < 0
583                    || (long) offset + (long) count > codePoints.length) {

587            this.value = new char[count * 2];
588            int end = offset + count;
589            int c = 0;
590            for (int i = offset; i < end; i++) {
591                c += Character.toChars(codePoints[i], this.value, c);
592            }
593            this.count = c;

648        /**
649         * Compares the specified string to this string using the Unicode values of
650         * the characters. Returns 0 if the strings contain the same characters in
651         * the same order. Returns a negative integer if the first non-equal
652         * character in this string has a Unicode value which is less than the
653         * Unicode value of the character at the same position in the specified
654         * string, or if this string is a prefix of the specified string. Returns a
655         * positive integer if the first non-equal character in this string has a
656         * Unicode value which is greater than the Unicode value of the character at
657         * the same position in the specified string, or if the specified string is
658         * a prefix of this string.
659         * 
660         * @param string
661         *            the string to compare.
662         * @return 0 if the strings are equal, a negative integer if this string is
663         *         before the specified string, or a positive integer if this string
664         *         is after the specified string.
665         * @since Android 1.0
666         */
667        public int compareTo(String string) {
668            // Code adapted from K&R, pg 101
669            int o1 = offset, o2 = string.offset, result;
670            int end = offset + (count < string.count ? count : string.count);
671            char[] target = string.value;
672            while (o1 < end) {
673                if ((result = value[o1++] - target[o2++]) != 0) {
674                    return result;
675                }
676            }
677            return count - string.count;
678        }

680        /**
681         * Compares the specified string to this string using the Unicode values of
682         * the characters, ignoring case differences. Returns 0 if the strings
683         * contain the same characters in the same order. Returns a negative integer
684         * if the first non-equal character in this string has a Unicode value which
685         * is less than the Unicode value of the character at the same position in
686         * the specified string, or if this string is a prefix of the specified
687         * string. Returns a positive integer if the first non-equal character in
688         * this string has a Unicode value which is greater than the Unicode value
689         * of the character at the same position in the specified string, or if the
690         * specified string is a prefix of this string.
691         * 
692         * @param string
693         *            the string to compare.
694         * @return 0 if the strings are equal, a negative integer if this string is
695         *         before the specified string, or a positive integer if this string
696         *         is after the specified string.
697         * @since Android 1.0
698         */
699        public int compareToIgnoreCase(String string) {
700            int o1 = offset, o2 = string.offset, result;
701            int end = offset + (count < string.count ? count : string.count);
702            char c1, c2;
703            char[] target = string.value;
704            while (o1 < end) {
705                if ((c1 = value[o1++]) == (c2 = target[o2++])) {
706                    continue;
707                }
708                c1 = Character.toLowerCase(Character.toUpperCase(c1));
709                c2 = Character.toLowerCase(Character.toUpperCase(c2));

2250        /*
2251         * An implementation of a String.indexOf that is supposed to perform
2252         * substantially better than the default algorithm if the "needle" (the
2253         * subString being searched for) is a constant string.
2254         *
2255         * For example, a JIT, upon encoutering a call to String.indexOf(String),
2256         * where the needle is a constant string, may compute the values cache, md2
2257         * and lastChar, and change the call to the following method.
2258         */
2259        @SuppressWarnings("unused")
2260        private static int indexOf(String haystackString, String needleString,
2261                int cache, int md2, char lastChar) {
2262            char[] haystack = haystackString.value;
2263            int haystackOffset = haystackString.offset;
2264            int haystackLength = haystackString.count;
2265            char[] needle = needleString.value;
2266            int needleOffset = needleString.offset;
2267            int needleLength = needleString.count;
2268            int needleLengthMinus1 = needleLength - 1;
2269            int haystackEnd = haystackOffset + haystackLength;
2270            outer_loop: for (int i = haystackOffset + needleLengthMinus1; i < haystackEnd;) {
2271                if (lastChar == haystack[i]) {
2272                    for (int j = 0; j < needleLengthMinus1; ++j) {
2273                        if (needle[j + needleOffset] != haystack[i + j
2274                                - needleLengthMinus1]) {
2275                            int skip = 1;
2276                            if ((cache & (1 << haystack[i])) == 0) {
2277                                skip += j;
2278                            }
2279                            i += Math.max(md2, skip);
2280                            continue outer_loop;
2281                        }
2282                    }
2283                    return i - needleLengthMinus1 - haystackOffset;
2284                }
2285    
2286                if ((cache & (1 << haystack[i])) == 0) {
2287                    i += needleLengthMinus1;
2288                }
2289                i++;
2290            }
2291            return -1;
2292        }
