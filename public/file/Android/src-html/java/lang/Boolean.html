018    package java.lang;
019    
020    import java.io.Serializable;
021    
022    /**
023     * The wrapper for the primitive type {@code boolean}.
024     * 
025     * @since Android 1.0
026     */
027    public final class Boolean implements Serializable, Comparable<Boolean> {
028    
029        private static final long serialVersionUID = -3665804199014368530L;
030    
031        /**
032         * The boolean value of the receiver.
033         */
034        private final boolean value;
035    
036        /**
037         * The {@link Class} object that represents the primitive type {@code
038         * boolean}.
039         * 
040         * @since Android 1.0
041         */
042        @SuppressWarnings("unchecked")
043        public static final Class<Boolean> TYPE = (Class<Boolean>) new boolean[0]
044                .getClass().getComponentType();
045    
046        // Note: This can't be set to "boolean.class", since *that* is
047        // defined to be "java.lang.Boolean.TYPE";
048    
049        /**
050         * The {@code Boolean} object that represents the primitive value
051         * {@code true}.
052         * 
053         * @since Android 1.0
054         */
055        public static final Boolean TRUE = new Boolean(true);
056    
057        /**
058         * The {@code Boolean} object that represents the primitive value
059         * {@code false}.
060         * 
061         * @since Android 1.0
062         */
063        public static final Boolean FALSE = new Boolean(false);
064    
065        /**
066         * Constructs a new {@code Boolean} with its boolean value specified by
067         * {@code string}. If {@code string} is not {@code null} and is equal to
068         * "true" using a non-case sensitive comparison, the result will be a
069         * Boolean representing the primitive value {@code true}, otherwise it will
070         * be a Boolean representing the primitive value {@code false}.
071         * 
072         * @param string
073         *            the string representing a boolean value.
074         * @since Android 1.0
075         */
076        public Boolean(String string) {
077            this(parseBoolean(string));
078        }
079    
080        /**
081         * Constructs a new {@code Boolean} with the specified primitive boolean
082         * value.
083         * 
084         * @param value
085         *            the primitive boolean value, {@code true} or {@code false}.
086         * @since Android 1.0
087         */
088        public Boolean(boolean value) {
089            this.value = value;
090        }
091    
092        /**
093         * Gets the primitive value of this boolean, either {@code true} or
094         * {@code false}.
095         * 
096         * @return this object's primitive value, {@code true} or {@code false}.
097         * @since Android 1.0
098         */
099        public boolean booleanValue() {
100            return value;
101        }
102    
103        /**
104         * Compares this instance with the specified object and indicates if they
105         * are equal. In order to be equal, {@code o} must be an instance of
106         * {@code Boolean} and have the same boolean value as this object.
107         * 
108         * @param o
109         *            the object to compare this boolean with.
110         * @return {@code true} if the specified object is equal to this
111         *         {@code Boolean}; {@code false} otherwise.
112         * @since Android 1.0
113         */
114        @Override
115        public boolean equals(Object o) {
116            return (o == this)
117                    || ((o instanceof Boolean) && (value == ((Boolean) o).value));
118        }
119    
120        /**
121         * Compares this object to the specified boolean object to determine their
122         * relative order.
123         * 
124         * @param that
125         *            the boolean object to compare this object to.
126         * @return 0 if the value of this boolean and the value of {@code that} are
127         *         equal; a positive value if the value of this boolean is
128         *         {@code true} and the value of {@code that} is {@code false}; a
129         *         negative value if the value if this boolean is {@code false} and
130         *         the value of {@code that} is {@code true}.
131         * @see java.lang.Comparable
132         * @since Android 1.0
133         */
134        public int compareTo(Boolean that) {
135            if (that == null) {
136                throw new NullPointerException();
137            }
138    
139            if (this.value == that.value) {
140                return 0;
141            }
142    
143            return this.value ? 1 : -1;
144        }
145    
146        /**
147         * Returns an integer hash code for this boolean.
148         * 
149         * @return this boolean's hash code, which is {@code 1231} for {@code true}
150         *         values and {@code 1237} for {@code false} values.
151         * @since Android 1.0
152         */
153        @Override
154        public int hashCode() {
155            return value ? 1231 : 1237;
156        }
157    
158        /**
159         * Returns a string containing a concise, human-readable description of this
160         * boolean.
161         * 
162         * @return "true" if the value of this boolean is {@code true}, "false"
163         *         otherwise.
164         * @since Android 1.0
165         */
166        @Override
167        public String toString() {
168            return String.valueOf(value);
169        }
170    
171        /**
172         * Returns the {@code boolean} value of the system property identified by
173         * {@code string}.
174         * 
175         * @param string
176         *            the name of the requested system property.
177         * @return {@code true} if the system property named by {@code string}
178         *         exists and it is equal to "true" using case insensitive
179         *         comparison, {@code false} otherwise.
180         * @see System#getProperty(String)
181         * @since Android 1.0
182         */
183        public static boolean getBoolean(String string) {
184            if (string == null || string.length() == 0) {
185                return false;
186            }
187            return (parseBoolean(System.getProperty(string)));
188        }
189    
190        /**
191         * Parses the specified string as a {@code boolean}.
192         * 
193         * @param s
194         *            the string representation of a boolean value.
195         * @return {@code true} if {@code s} is not {@code null} and is equal to
196         *         {@code "true"} using case insensitive comparison, {@code false}
197         *         otherwise.
198         * @since Android 1.0
199         */
200        public static boolean parseBoolean(String s) {
201            return "true".equalsIgnoreCase(s); //$NON-NLS-1$
202        }
203    
204        /**
205         * Converts the specified boolean to its string representation. 
206         * 
207         * @param value
208         *            the boolean to convert.
209         * @return "true" if {@code value} is {@code true}, "false" otherwise.
210         * @since Android 1.0
211         */
212        public static String toString(boolean value) {
213            return String.valueOf(value);
214        }
215    
216        /**
217         * Parses the specified string as a boolean value.
218         * 
219         * @param string
220         *            the string representation of a boolean value.
221         * @return {@code Boolean.TRUE} if {@code string} is equal to "true" using
222         *         case insensitive comparison, {@code Boolean.FALSE} otherwise.
223         * @see #parseBoolean(String)
224         * @since Android 1.0
225         */
226        public static Boolean valueOf(String string) {
227            return parseBoolean(string) ? Boolean.TRUE : Boolean.FALSE;
228        }
229    
230        /**
231         * Returns a {@code Boolean} instance for the specified boolean value.
232         * <p>
233         * If it is not necessary to get a new {@code Boolean} instance, it is
234         * recommended to use this method instead of the constructor, since it
235         * returns its static instances, which results in better performance.
236         * </p>
237         * 
238         * @param b
239         *            the boolean to convert to a {@code Boolean}.
240         * @return {@code Boolean.TRUE} if {@code b} is equal to {@code true},
241         *         {@code Boolean.FALSE} otherwise.
242         * @since Android 1.0
243         */
244        public static Boolean valueOf(boolean b) {
245            return b ? Boolean.TRUE : Boolean.FALSE;
246        }
247    }
