017    package java.nio;
018    
019    
020    /**
021     * A buffer is a list of elements of a specific primitive type.
022     * <p>
023     * A buffer can be described by the following properties:
024     * <ul>
025     * <li>Capacity: the number of elements a buffer can hold. Capacity may not be
026     * negative and never changes.</li>
027     * <li>Position: a cursor of this buffer. Elements are read or written at the
028     * position if you do not specify an index explicitly. Position may not be
029     * negative and not greater than the limit.</li>
030     * <li>Limit: controls the scope of accessible elements. You can only read or
031     * write elements from index zero to <code>limit - 1</code>. Accessing
032     * elements out of the scope will cause an exception. Limit may not be negative
033     * and not greater than capacity.</li>
034     * <li>Mark: used to remember the current position, so that you can reset the
035     * position later. Mark may not be negative and no greater than position.</li>
036     * <li>A buffer can be read-only or read-write. Trying to modify the elements
037     * of a read-only buffer will cause a <code>ReadOnlyBufferException</code>,
038     * while changing the position, limit and mark of a read-only buffer is OK.</li>
039     * <li>A buffer can be direct or indirect. A direct buffer will try its best to
040     * take advantage of native memory APIs and it may not stay in the Java heap,
041     * thus it is not affected by garbage collection.</li>
042     * </ul>
043     * </p>
044     * <p>
045     * Buffers are not thread-safe. If concurrent access to a buffer instance is
046     * required, then the callers are responsible to take care of the
047     * synchronization issues.
048     * </p>
049     * 
050     * @since Android 1.0
051     */
052    public abstract class Buffer {
053    
054        /**
055         * <code>UNSET_MARK</code> means the mark has not been set.
056         */
057        final static int UNSET_MARK = -1;
058    
059        /**
060         * The capacity of this buffer, which never change.
061         */
062        final int capacity;
063    
064        /**
065         * <code>limit - 1</code> is the last element that can be read or written.
066         * Limit must be no less than zero and no greater than <code>capacity</code>.
067         */
068        int limit;
069    
070        /**
071         * Mark is where position will be set when <code>reset()</code> is called.
072         * Mark is not set by default. Mark is always no less than zero and no
073         * greater than <code>position</code>.
074         */
075        int mark = UNSET_MARK;
076    
077        /**
078         * The current position of this buffer. Position is always no less than zero
079         * and no greater than <code>limit</code>.
080         */
081        int position = 0;
082    
083        // BEGIN android-added
084        /**
085         * The log base 2 of the element size of this buffer.  Each typed subclass
086         * (ByteBuffer, CharBuffer, etc.) is responsible for initializing this
087         * value.  The value is used by native code to avoid the need for costly
088         * 'instanceof' tests.
089         *
090         */
091        int _elementSizeShift;
092    
093        /**
094         * Returns the array associated with this buffer, or null if none exists.
095         * Each typed subclass (ByteBuffer, CharBuffer, etc.) overrides this method
096         * to call its array() method with appropriate checks.
097         *
098         * @return a primitive array or null
099         */
100        Object _array() {
101            return null;
102        }
103    
104        /**
105         * Returns the offset into the backing array, if one exists, otherwise 0.
106         * Each typed subclass (ByteBuffer, CharBuffer, etc.) overrides this method
107         * to call its arrayOffset() method with appropriate checks.
108         *
109         * @return the array offset, or 0
110         */
111        int _arrayOffset() {
112            return 0;
113        }
114        // END android-added
115    
116        /**
117         * Construct a buffer with the specified capacity.
118         *
119         * @param capacity
120         *            the capacity of this buffer.
121         */
122        Buffer(int capacity) {
123            super();
124            if (capacity < 0) {
125                throw new IllegalArgumentException();
126            }
127            this.capacity = this.limit = capacity;
128        }
129    
130        /**
131         * Returns the capacity of this buffer.
132         * 
133         * @return the number of elements that are contained in this buffer.
134         * @since Android 1.0
135         */
136        public final int capacity() {
137            return capacity;
138        }
139    
140        /**
141         * Clears this buffer.
142         * <p>
143         * While the content of this buffer is not changed, the following internal
144         * changes take place: the current position is reset back to the start of
145         * the buffer, the value of the buffer limit is made equal to the capacity
146         * and mark is cleared.
147         * </p>
148         * 
149         * @return this buffer.
150         * @since Android 1.0
151         */
152        public final Buffer clear() {
153            position = 0;
154            mark = UNSET_MARK;
155            limit = capacity;
156            return this;
157        }
158    
159        /**
160         * Flips this buffer.
161         * <p>
162         * The limit is set to the current position, then the position is set to
163         * zero, and the mark is cleared.
164         * </p>
165         * <p>
166         * The content of this buffer is not changed.
167         * </p>
168         * 
169         * @return this buffer.
170         * @since Android 1.0
171         */
172        public final Buffer flip() {
173            limit = position;
174            position = 0;
175            mark = UNSET_MARK;
176            return this;
177        }
178    
179        /**
180         * Indicates if there are elements remaining in this buffer, that is if
181         * {@code position < limit}.
182         * 
183         * @return {@code true} if there are elements remaining in this buffer,
184         *         {@code false} otherwise.
185         * @since Android 1.0
186         */
187        public final boolean hasRemaining() {
188            return position < limit;
189        }
190    
191        /**
192         * Indicates whether this buffer is read-only.
193         * 
194         * @return {@code true} if this buffer is read-only, {@code false}
195         *         otherwise.
196         * @since Android 1.0
197         */
198        public abstract boolean isReadOnly();
199    
200        /**
201         * Returns the limit of this buffer.
202         * 
203         * @return the limit of this buffer.
204         * @since Android 1.0
205         */
206        public final int limit() {
207            return limit;
208        }
209    
210        /**
211         * Sets the limit of this buffer.
212         * <p>
213         * If the current position in the buffer is in excess of
214         * <code>newLimit</code> then, on returning from this call, it will have
215         * been adjusted to be equivalent to <code>newLimit</code>. If the mark
216         * is set and is greater than the new limit, then it is cleared.
217         * </p>
218         * 
219         * @param newLimit
220         *            the new limit, must not be negative and not greater than
221         *            capacity.
222         * @return this buffer.
223         * @exception IllegalArgumentException
224         *                if <code>newLimit</code> is invalid.
225         * @since Android 1.0
226         */
227        public final Buffer limit(int newLimit) {
228            if (newLimit < 0 || newLimit > capacity) {
229                throw new IllegalArgumentException();
230            }
231    
232            limit = newLimit;
233            if (position > newLimit) {
234                position = newLimit;
235            }
236            if ((mark != UNSET_MARK) && (mark > newLimit)) {
237                mark = UNSET_MARK;
238            }
239            return this;
240        }
241    
242        /**
243         * Marks the current position, so that the position may return to this point
244         * later by calling <code>reset()</code>.
245         * 
246         * @return this buffer.
247         * @since Android 1.0
248         */
249        public final Buffer mark() {
250            mark = position;
251            return this;
252        }
253    
254        /**
255         * Returns the position of this buffer.
256         * 
257         * @return the value of this buffer's current position.
258         * @since Android 1.0
259         */
260        public final int position() {
261            return position;
262        }
263    
264        /**
265         * Sets the position of this buffer.
266         * <p>
267         * If the mark is set and it is greater than the new position, then it is
268         * cleared.
269         * </p>
270         * 
271         * @param newPosition
272         *            the new position, must be not negative and not greater than
273         *            limit.
274         * @return this buffer.
275         * @exception IllegalArgumentException
276         *                if <code>newPosition</code> is invalid.
277         * @since Android 1.0
278         */
279        public final Buffer position(int newPosition) {
280            if (newPosition < 0 || newPosition > limit) {
281                throw new IllegalArgumentException();
282            }
283    
284            position = newPosition;
285            if ((mark != UNSET_MARK) && (mark > position)) {
286                mark = UNSET_MARK;
287            }
288            return this;
289        }
290    
291        /**
292         * Returns the number of remaining elements in this buffer, that is
293         * {@code limit - position}.
294         * 
295         * @return the number of remaining elements in this buffer.
296         * @since Android 1.0
297         */
298        public final int remaining() {
299            return limit - position;
300        }
301    
302        /**
303         * Resets the position of this buffer to the <code>mark</code>.
304         * 
305         * @return this buffer.
306         * @exception InvalidMarkException
307         *                if the mark is not set.
308         * @since Android 1.0
309         */
310        public final Buffer reset() {
311            if (mark == UNSET_MARK) {
312                throw new InvalidMarkException();
313            }
314            position = mark;
315            return this;
316        }
317    
318        /**
319         * Rewinds this buffer.
320         * <p>
321         * The position is set to zero, and the mark is cleared. The content of this
322         * buffer is not changed.
323         * </p>
324         * 
325         * @return this buffer.
326         * @since Android 1.0
327         */
328        public final Buffer rewind() {
329            position = 0;
330            mark = UNSET_MARK;
331            return this;
332        }
333    }
