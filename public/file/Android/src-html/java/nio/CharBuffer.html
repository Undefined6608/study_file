022    /**
023     * A buffer of chars.
024     * <p>
025     * A char buffer can be created in either one of the following ways:
026     * </p>
027     * <ul>
028     * <li>{@link #allocate(int) Allocate} a new char array and create a buffer
029     * based on it;</li>
030     * <li>{@link #wrap(char[]) Wrap} an existing char array to create a new
031     * buffer;</li>
032     * <li>{@link #wrap(CharSequence) Wrap} an existing char sequence to create a
033     * new buffer;</li>
034     * <li>Use {@link java.nio.ByteBuffer#asCharBuffer() ByteBuffer.asCharBuffer}
035     * to create a char buffer based on a byte buffer.</li>
036     * </ul>
037     * 
038     * @since Android 1.0
039     */
040    public abstract class CharBuffer extends Buffer implements Comparable<CharBuffer>,
041            CharSequence, Appendable, Readable {
042    
043        /**
044         * Creates a char buffer based on a newly allocated char array.
045         * 
046         * @param capacity
047         *            the capacity of the new buffer.
048         * @return the created char buffer.
049         * @throws IllegalArgumentException
050         *             if {@code capacity} is less than zero.
051         * @since Android 1.0
052         */
053        public static CharBuffer allocate(int capacity) {
054            if (capacity < 0) {
055                throw new IllegalArgumentException();
056            }
057            return BufferFactory.newCharBuffer(capacity);
058        }

060        /**
061         * Creates a new char buffer by wrapping the given char array.
062         * <p>
063         * Calling this method has the same effect as
064         * {@code wrap(array, 0, array.length)}.
065         * </p>
066         * 
067         * @param array
068         *            the char array which the new buffer will be based on.
069         * @return the created char buffer.
070         * @since Android 1.0
071         */
072        public static CharBuffer wrap(char[] array) {
073            return wrap(array, 0, array.length);
074        }

076        /**
077         * Creates a new char buffer by wrapping the given char array.
078         * <p>
079         * The new buffer's position will be {@code start}, limit will be
080         * {@code start + len}, capacity will be the length of the array.
081         * </p>
082         * 
083         * @param array
084         *            the char array which the new buffer will be based on.
085         * @param start
086         *            the start index, must not be negative and not greater than
087         *            {@code array.length}.
088         * @param len
089         *            the length, must not be negative and not greater than
090         *            {@code array.length - start}.
091         * @return the created char buffer.
092         * @exception IndexOutOfBoundsException
093         *                if either {@code start} or {@code len} is invalid.
094         * @since Android 1.0
095         */
096        public static CharBuffer wrap(char[] array, int start, int len) {
097            int length = array.length;
098            if ((start < 0) || (len < 0)
099                    || (long) start + (long) len > length) {
100                throw new IndexOutOfBoundsException();
101            }
102    
103            CharBuffer buf = BufferFactory.newCharBuffer(array);
104            buf.position = start;
105            buf.limit = start + len;
106    
107            return buf;
108        }

114         * {@code wrap(chseq, 0, chseq.length())}.
115         * </p>
116         * 
117         * @param chseq
118         *            the char sequence which the new buffer will be based on.
119         * @return the created char buffer.
120         * @since Android 1.0
121         */
122        public static CharBuffer wrap(CharSequence chseq) {
123            return BufferFactory.newCharBuffer(chseq);
124        }

126        /**
127         * Creates a new char buffer by wrapping the given char sequence.
128         * <p>
129         * The new buffer's position will be {@code start}, limit will be
130         * {@code end}, capacity will be the length of the char sequence. The new
131         * buffer is read-only.
132         * </p>
133         * 
134         * @param chseq
135         *            the char sequence which the new buffer will be based on.
136         * @param start
137         *            the start index, must not be negative and not greater than
138         *            {@code chseq.length()}.
139         * @param end
140         *            the end index, must be no less than {@code start} and no
141         *            greater than {@code chseq.length()}.
142         * @return the created char buffer.
143         * @exception IndexOutOfBoundsException
144         *                if either {@code start} or {@code end} is invalid.
145         * @since Android 1.0
146         */
147        public static CharBuffer wrap(CharSequence chseq, int start, int end) {
148            if (chseq == null) {
149                throw new NullPointerException();
150            }
151            if (start < 0 || end < start || end > chseq.length()) {
152                throw new IndexOutOfBoundsException();
153            }
154            
155            CharBuffer result = BufferFactory.newCharBuffer(chseq);
156            result.position = start;
157            result.limit = end;
158            return result;
159        }

245         * Returns the character located at the specified index in the buffer. The

280        /**
281         * Compare the remaining chars of this buffer to another char
282         * buffer's remaining chars.
283         * 
284         * @param otherBuffer
285         *            another char buffer.
286         * @return a negative value if this is less than {@code otherBuffer}; 0 if
287         *         this equals to {@code otherBuffer}; a positive value if this is
288         *         greater than {@code otherBuffer}.
289         * @exception ClassCastException
290         *                if {@code otherBuffer} is not a char buffer.
291         * @since Android 1.0
292         */
293        public int compareTo(CharBuffer otherBuffer) {
294            int compareRemaining = (remaining() < otherBuffer.remaining()) ? remaining()
295                    : otherBuffer.remaining();
296            int thisPos = position;
297            int otherPos = otherBuffer.position;
298            char thisByte, otherByte;
299            while (compareRemaining > 0) {
300                thisByte = get(thisPos);
301                otherByte = otherBuffer.get(otherPos);
302                if (thisByte != otherByte) {
303                    return thisByte < otherByte ? -1 : 1;
304                }
305                thisPos++;
306                otherPos++;
307                compareRemaining--;
308            }
309            return remaining() - otherBuffer.remaining();
310        }

317         * byte order are the same as this buffer's, too.

319         * <p>
320         * The new buffer shares its content with this buffer, which means either

330        /**
331         * Checks whether this char buffer is equal to another object.
332         * <p>
333         * If {@code other} is not a char buffer then {@code false} is returned. Two
334         * char buffers are equal if and only if their remaining chars are exactly
335         * the same. Position, limit, capacity and mark are not considered.
336         * </p>
337         * 
338         * @param other
339         *            the object to compare with this char buffer.
340         * @return {@code true} if this char buffer is equal to {@code other},
341         *         {@code false} otherwise.
342         * @since Android 1.0
343         */
344        public boolean equals(Object other) {
345            if (!(other instanceof CharBuffer)) {
346                return false;
347            }
348            CharBuffer otherBuffer = (CharBuffer) other;
349    
350            if (remaining() != otherBuffer.remaining()) {
351                return false;
352            }
353    
354            int myPosition = position;
355            int otherPosition = otherBuffer.position;
356            boolean equalSoFar = true;
357            while (equalSoFar && (myPosition < limit)) {
358                equalSoFar = get(myPosition++) == otherBuffer.get(otherPosition++);
359            }
360    
361            return equalSoFar;
362        }

393        /**
394         * Reads chars from the current position into the specified char array,
395         * starting from the specified offset, and increases the position by the
396         * number of chars read.
397         * 
398         * @param dest
399         *            the target char array.
400         * @param off
401         *            the offset of the char array, must not be negative and not
402         *            greater than {@code dest.length}.
403         * @param len
404         *            The number of chars to read, must be no less than zero and no
405         *            greater than {@code dest.length - off}.
406         * @return this buffer.
407         * @exception IndexOutOfBoundsException
408         *                if either {@code off} or {@code len} is invalid.
409         * @exception BufferUnderflowException
410         *                if {@code len} is greater than {@code remaining()}.
411         * @since Android 1.0
412         */
413        public CharBuffer get(char[] dest, int off, int len) {
414            int length = dest.length;
415            if ((off < 0 ) || (len < 0) || (long)off + (long)len > length) {
416                throw new IndexOutOfBoundsException();
417            }
418            
419            if (len > remaining()) {
420                throw new BufferUnderflowException();
421            }
422            for (int i = off; i < off + len; i++) {
423                dest[i] = get();
424            }
425            return this;
426        }

451        /**
452         * Calculates this buffer's hash code from the remaining chars. The
453         * position, limit, capacity and mark don't affect the hash code.
454         * 
455         * @return the hash code calculated from the remaining chars.
456         * @since Android 1.0
457         */
458        public int hashCode() {
459            int myPosition = position;
460            int hash = 0;
461            while (myPosition < limit) {
462                hash = hash + get(myPosition++);
463            }
464            return hash;
465        }

599        /**
600         * Writes all the remaining chars of the {@code src} char buffer to this
601         * buffer's current position, and increases both buffers' position by the
602         * number of chars copied.
603         * 
604         * @param src
605         *            the source char buffer.
606         * @return this buffer.
607         * @exception BufferOverflowException
608         *                if {@code src.remaining()} is greater than this buffer's
609         *                {@code remaining()}.
610         * @exception IllegalArgumentException
611         *                if {@code src} is this buffer.
612         * @exception ReadOnlyBufferException
613         *                if no changes may be made to the contents of this buffer.
614         * @since Android 1.0
615         */
616        public CharBuffer put(CharBuffer src) {
617            if (src == this) {
618                throw new IllegalArgumentException();
619            }
620            if (src.remaining() > remaining()) {
621                throw new BufferOverflowException();
622            }
623            
624            char[] contents = new char[src.remaining()];
625            src.get(contents);
626            put(contents);
627            return this;
628        }

668        /**
669         * Writes chars of the given string to the current position of this buffer,
670         * and increases the position by the number of chars written.
671         * 
672         * @param str
673         *            the string to write.
674         * @param start
675         *            the first char to write, must not be negative and not greater
676         *            than {@code str.length()}.
677         * @param end
678         *            the last char to write (excluding), must be less than
679         *            {@code start} and not greater than {@code str.length()}.
680         * @return this buffer.
681         * @exception BufferOverflowException
682         *                if {@code remaining()} is less than {@code end - start}.
683         * @exception IndexOutOfBoundsException
684         *                if either {@code start} or {@code end} is invalid.
685         * @exception ReadOnlyBufferException
686         *                if no changes may be made to the contents of this buffer.
687         * @since Android 1.0
688         */
689        public CharBuffer put(String str, int start, int end) {
690            int length = str.length();
691            if (start < 0 || end < start || end > length) {
692                throw new IndexOutOfBoundsException();
693            }
694            
695            if (end - start > remaining()) {
696                throw new BufferOverflowException();
697            }
698            for (int i = start; i < end; i++) {
699                put(str.charAt(i));
700            }
701            return this;
702        }

739         * @param start
740         *            the start index of the sub-sequence, referenced from the
741         *            current buffer position. Must not be less than zero and not
742         *            greater than the value obtained from a call to
743         *            {@code remaining()}.
744         * @param end
745         *            the end index of the sub-sequence, referenced from the current
746         *            buffer position. Must not be less than {@code start} and not
747         *            be greater than the value obtained from a call to
748         *            {@code remaining()}.
749         * @return a new char buffer represents a sub-sequence of this buffer's
750         *         current remaining content.
751         * @exception IndexOutOfBoundsException
752         *                if either {@code start} or {@code end} is invalid.
753         * @since Android 1.0
754         */
755        public abstract CharSequence subSequence(int start, int end);
756    
757        /**
758         * Returns a string representing the current remaining chars of this buffer.
759         * 
760         * @return a string representing the current remaining chars of this buffer.
761         * @since Android 1.0
762         */
763        public String toString() {
764            StringBuffer strbuf = new StringBuffer();
765            for (int i = position; i < limit; i++) {
766                strbuf.append(get(i));
767            }
768            return strbuf.toString();
769        }

772         * Writes the given char to the current position and increases the position
773         * by 1.
774         * 
775         * @param c
776         *            the char to write.
777         * @return this buffer.
778         * @exception BufferOverflowException
779         *                if position is equal or greater than limit.
780         * @exception ReadOnlyBufferException
781         *                if no changes may be made to the contents of this buffer.
782         * @since Android 1.0
783         */
784        public CharBuffer append(char c){
785            return put(c);
786        }

788        /**
789         * Writes all chars of the given character sequence {@code csq} to the
790         * current position of this buffer, and increases the position by the length
791         * of the csq.
792         * <p>
793         * Calling this method has the same effect as {@code append(csq.toString())}.
794         * </p>
795         * If the {@code CharSequence} is {@code null} the string "null" will be
796         * written to the buffer.
797         * 
798         * @param csq
799         *            the {@code CharSequence} to write.
800         * @return this buffer.
801         * @exception BufferOverflowException
802         *                if {@code remaining()} is less than the length of csq.
803         * @exception ReadOnlyBufferException
804         *                if no changes may be made to the contents of this buffer.
805         * @since Android 1.0
806         */
807        public CharBuffer append(CharSequence csq){
808            if (csq != null) {
809                return put(csq.toString());
810            }
811            return put("null"); //$NON-NLS-1$
812        }

814        /**
815         * Writes chars of the given {@code CharSequence} to the current position of
816         * this buffer, and increases the position by the number of chars written.
817         * 
818         * @param csq
819         *            the {@code CharSequence} to write.
820         * @param start
821         *            the first char to write, must not be negative and not greater
822         *            than {@code csq.length()}.
823         * @param end
824         *            the last char to write (excluding), must be less than
825         *            {@code start} and not greater than {@code csq.length()}.
826         * @return this buffer.
827         * @exception BufferOverflowException
828         *                if {@code remaining()} is less than {@code end - start}.
829         * @exception IndexOutOfBoundsException
830         *                if either {@code start} or {@code end} is invalid.
831         * @exception ReadOnlyBufferException
832         *                if no changes may be made to the contents of this buffer.
833         * @since Android 1.0
834         */
835        public CharBuffer append(CharSequence csq, int start, int end){
836            if (csq == null) {
837                csq = "null"; //$NON-NLS-1$
838            }
839            CharSequence cs = csq.subSequence(start, end);
840            if (cs.length() > 0) {
841                return put(cs.toString());
842            }
843            return this;
844        }

846        /**
847         * Reads characters from this buffer and puts them into {@code target}. The
848         * number of chars that are copied is either the number of remaining chars
849         * in this buffer or the number of remaining chars in {@code target},
850         * whichever is smaller.
851         * 
852         * @param target
853         *            the target char buffer.
854         * @throws IllegalArgumentException
855         *             if {@code target} is this buffer.
856         * @throws IOException
857         *             if an I/O error occurs.
858         * @throws ReadOnlyBufferException
859         *             if no changes may be made to the contents of {@code target}.
860         * @return the number of chars copied or -1 if there are no chars left to be
861         *         read from this buffer.
862         * @since Android 1.0
863         */
864        public int read(CharBuffer target) throws IOException {
865            if(target == this){
866                throw new IllegalArgumentException();
867            }
868            if (remaining() == 0) {
869                return target.remaining()==0?0:-1;
870            }
871            int result = Math.min(target.remaining(), remaining());
872            char[] chars = new char[result];
873            get(chars);
874            target.put(chars);
875            return result;
876        }
