018    package java.nio;
019    
020    /**
021     * A buffer of doubles.
022     * <p>
023     * A double buffer can be created in either one of the following ways:
024     * </p>
025     * <ul>
026     * <li>{@link #allocate(int) Allocate} a new double array and create a buffer
027     * based on it;</li>
028     * <li>{@link #wrap(double[]) Wrap} an existing double array to create a new
029     * buffer;</li>
030     * <li>Use
031     * {@link java.nio.ByteBuffer#asDoubleBuffer() ByteBuffer.asDoubleBuffer} to
032     * create a double buffer based on a byte buffer.</li>
033     * </ul>
034     * 
035     * @since Android 1.0
036     */
037    public abstract class DoubleBuffer extends Buffer implements Comparable<DoubleBuffer> {
038    
039        /**
040         * Creates a double buffer based on a newly allocated double array.
041         * 
042         * @param capacity
043         *            the capacity of the new buffer.
044         * @return the created double buffer.
045         * @throws IllegalArgumentException
046         *             if {@code capacity} is less than zero.
047         * @since Android 1.0
048         */
049        public static DoubleBuffer allocate(int capacity) {
050            if (capacity < 0) {
051                throw new IllegalArgumentException();
052            }
053            return BufferFactory.newDoubleBuffer(capacity);
054        }
055    
056        /**
057         * Creates a new double buffer by wrapping the given double array.
058         * <p>
059         * Calling this method has the same effect as
060         * {@code wrap(array, 0, array.length)}.
061         * </p>
062         * 
063         * @param array
064         *            the double array which the new buffer will be based on.
065         * @return the created double buffer.
066         * @since Android 1.0
067         */
068        public static DoubleBuffer wrap(double[] array) {
069            return wrap(array, 0, array.length);
070        }
071    
072        /**
073         * Creates a new double buffer by wrapping the given double array.
074         * <p>
075         * The new buffer's position will be {@code start}, limit will be
076         * {@code start + len}, capacity will be the length of the array.
077         * </p>
078         * 
079         * @param array
080         *            the double array which the new buffer will be based on.
081         * @param start
082         *            the start index, must not be negative and not greater than
083         *            {@code array.length}.
084         * @param len
085         *            the length, must not be negative and not greater than
086         *            {@code array.length - start}.
087         * @return the created double buffer.
088         * @exception IndexOutOfBoundsException
089         *                if either {@code start} or {@code len} is invalid.
090         * @since Android 1.0
091         */
092        public static DoubleBuffer wrap(double[] array, int start, int len) {
093            int length = array.length;
094            if (start < 0 || len < 0 || (long)start + (long)len > length) {
095                throw new IndexOutOfBoundsException();
096            }
097    
098            DoubleBuffer buf = BufferFactory.newDoubleBuffer(array);
099            buf.position = start;
100            buf.limit = start + len;
101    
102            return buf;
103        }
104    
105        /**
106         * Constructs a {@code DoubleBuffer} with given capacity.
107         * 
108         * @param capacity
109         *            the capacity of the buffer.
110         */
111        DoubleBuffer(int capacity) {
112            super(capacity);
113            // BEGIN android-added
114            _elementSizeShift = 3;
115            // END android-added
116        }
117    
118        /**
119         * Returns the double array which this buffer is based on, if there is one.
120         * 
121         * @return the double array which this buffer is based on.
122         * @exception ReadOnlyBufferException
123         *                if this buffer is based on an array but it is read-only.
124         * @exception UnsupportedOperationException
125         *                if this buffer is not based on an array.
126         * @since Android 1.0
127         */
128        public final double[] array() {
129            return protectedArray();
130        }
131    
132        /**
133         * Returns the offset of the double array which this buffer is based on, if
134         * there is one.
135         * <p>
136         * The offset is the index of the array corresponding to the zero position
137         * of the buffer.
138         * </p>
139         * 
140         * @return the offset of the double array which this buffer is based on.
141         * @exception ReadOnlyBufferException
142         *                if this buffer is based on an array, but it is read-only.
143         * @exception UnsupportedOperationException
144         *                if this buffer is not based on an array.
145         * @since Android 1.0
146         */
147        public final int arrayOffset() {
148            return protectedArrayOffset();
149        }
150    
151        // BEGIN android-added
152        @Override 
153        Object _array() {
154            if (hasArray()) {
155                return array();
156            }
157            return null;
158        }
159    
160        @Override 
161        int _arrayOffset() {
162            if (hasArray()) {
163                return arrayOffset();
164            }
165            return 0;
166        }

169        /**
170         * Returns a read-only buffer that shares its content with this buffer.
171         * <p>
172         * The returned buffer is guaranteed to be a new instance, even if this
173         * buffer is read-only itself. The new buffer's position, limit, capacity
174         * and mark are the same as this buffer's.
175         * </p>
176         * <p>
177         * The new buffer shares its content with this buffer, which means that this
178         * buffer's change of content will be visible to the new buffer. The two
179         * buffer's position, limit and mark are independent.
180         * </p>
181         * 
182         * @return a read-only version of this buffer.
183         * @since Android 1.0
184         */
185        public abstract DoubleBuffer asReadOnlyBuffer();
186    
187        /**
188         * Compacts this double buffer.
189         * <p>
190         * The remaining doubles will be moved to the head of the buffer, staring
191         * from position zero. Then the position is set to {@code remaining()}; the
192         * limit is set to capacity; the mark is cleared.
193         * </p>
194         * 
195         * @return this buffer.
196         * @exception ReadOnlyBufferException
197         *                if no changes may be made to the contents of this buffer.
198         * @since Android 1.0
199         */
200        public abstract DoubleBuffer compact();
201    
202        /**
203         * Compare the remaining doubles of this buffer to another double buffer's
204         * remaining doubles.
205         * 
206         * @param otherBuffer
207         *            another double buffer.
208         * @return a negative value if this is less than {@code other}; 0 if this
209         *         equals to {@code other}; a positive value if this is greater
210         *         than {@code other}.
211         * @exception ClassCastException
212         *                if {@code other} is not a double buffer.
213         * @since Android 1.0
214         */
215        public int compareTo(DoubleBuffer otherBuffer) {
216            int compareRemaining = (remaining() < otherBuffer.remaining()) ? remaining()
217                    : otherBuffer.remaining();
218            int thisPos = position;
219            int otherPos = otherBuffer.position;
220            // BEGIN android-changed
221            double thisDouble, otherDouble;
222            while (compareRemaining > 0) {
223                thisDouble = get(thisPos);
224                otherDouble = otherBuffer.get(otherPos);
225                // checks for double and NaN inequality
226                if ((thisDouble != otherDouble)
227                        && ((thisDouble == thisDouble) || (otherDouble == otherDouble))) {
228                    return thisDouble < otherDouble ? -1 : 1;
229                }
230                thisPos++;
231                otherPos++;
232                compareRemaining--;
233            }
234            // END android-changed
235            return remaining() - otherBuffer.remaining();
236        }
237    
238        /**
239         * Returns a duplicated buffer that shares its content with this buffer.
240         * <p>
241         * The duplicated buffer's position, limit, capacity and mark are the same
242         * as this buffer's. The duplicated buffer's read-only property and byte
243         * order are the same as this buffer's, too.
244         * </p>
245         * <p>
246         * The new buffer shares its content with this buffer, which means either
247         * buffer's change of content will be visible to the other. The two buffer's
248         * position, limit and mark are independent.
249         * </p>
250         * 
251         * @return a duplicated buffer that shares its content with this buffer.
252         * @since Android 1.0
253         */
254        public abstract DoubleBuffer duplicate();
255    
256        /**
257         * Checks whether this double buffer is equal to another object.
258         * <p>
259         * If {@code other} is not a double buffer then {@code false} is returned.
260         * Two double buffers are equal if and only if their remaining doubles are
261         * exactly the same. Position, limit, capacity and mark are not considered.
262         * </p>
263         * 
264         * @param other
265         *            the object to compare with this double buffer.
266         * @return {@code true} if this double buffer is equal to {@code other},
267         *         {@code false} otherwise.
268         * @since Android 1.0
269         */
270        public boolean equals(Object other) {
271            if (!(other instanceof DoubleBuffer)) {
272                return false;
273            }
274            DoubleBuffer otherBuffer = (DoubleBuffer) other;
275    
276            if (remaining() != otherBuffer.remaining()) {
277                return false;
278            }
279    
280            int myPosition = position;
281            int otherPosition = otherBuffer.position;
282            boolean equalSoFar = true;
283            while (equalSoFar && (myPosition < limit)) {
284                equalSoFar = get(myPosition++) == otherBuffer.get(otherPosition++);
285            }
286    
287            return equalSoFar;
288        }
289    
290        /**
291         * Returns the double at the current position and increases the position by
292         * 1.
293         * 
294         * @return the double at the current position.
295         * @exception BufferUnderflowException
296         *                if the position is equal or greater than limit.
297         * @since Android 1.0
298         */
299        public abstract double get();
300    
301        /**
302         * Reads doubles from the current position into the specified double array
303         * and increases the position by the number of doubles read.
304         * <p>
305         * Calling this method has the same effect as
306         * {@code get(dest, 0, dest.length)}.
307         * </p>
308         * 
309         * @param dest
310         *            the destination double array.
311         * @return this buffer.
312         * @exception BufferUnderflowException
313         *                if {@code dest.length} is greater than {@code remaining()}.
314         * @since Android 1.0
315         */
316        public DoubleBuffer get(double[] dest) {
317            return get(dest, 0, dest.length);
318        }
319    
320        /**
321         * Reads doubles from the current position into the specified double array,
322         * starting from the specified offset, and increases the position by the
323         * number of doubles read.
324         * 
325         * @param dest
326         *            the target double array.
327         * @param off
328         *            the offset of the double array, must not be negative and not
329         *            greater than {@code dest.length}.
330         * @param len
331         *            the number of doubles to read, must be no less than zero and
332         *            not greater than {@code dest.length - off}.
333         * @return this buffer.
334         * @exception IndexOutOfBoundsException
335         *                if either {@code off} or {@code len} is invalid.
336         * @exception BufferUnderflowException
337         *                if {@code len} is greater than {@code remaining()}.
338         * @since Android 1.0
339         */
340        public DoubleBuffer get(double[] dest, int off, int len) {
341            int length = dest.length;
342            if (off < 0 || len < 0 || (long)off + (long)len > length) {
343                throw new IndexOutOfBoundsException();
344            }
345            
346            if (len > remaining()) {
347                throw new BufferUnderflowException();
348            }
349            for (int i = off; i < off + len; i++) {
350                dest[i] = get();
351            }
352            return this;
353        }
354    
355        /**
356         * Returns a double at the specified index; the position is not changed.
357         * 
358         * @param index
359         *            the index, must not be negative and less than limit.
360         * @return a double at the specified index.
361         * @exception IndexOutOfBoundsException
362         *                if index is invalid.
363         * @since Android 1.0
364         */
365        public abstract double get(int index);
366    
367        /**
368         * Indicates whether this buffer is based on a double array and is
369         * read/write.
370         * 
371         * @return {@code true} if this buffer is based on a double array and
372         *         provides read/write access, {@code false} otherwise.
373         * @since Android 1.0
374         */
375        public final boolean hasArray() {
376            return protectedHasArray();
377        }
378    
379        /**
380         * Calculates this buffer's hash code from the remaining chars. The
381         * position, limit, capacity and mark don't affect the hash code.
382         * 
383         * @return the hash code calculated from the remaining chars.
384         * @since Android 1.0
385         */
386        public int hashCode() {
387            int myPosition = position;
388            int hash = 0;
389            long l;
390            while (myPosition < limit) {
391                l = Double.doubleToLongBits(get(myPosition++));
392                hash = hash + ((int) l) ^ ((int) (l >> 32));
393            }
394            return hash;
395        }
396    
397        /**
398         * Indicates whether this buffer is direct. A direct buffer will try its
399         * best to take advantage of native memory APIs and it may not stay in the
400         * Java heap, so it is not affected by garbage collection.
401         * <p>
402         * A double buffer is direct if it is based on a byte buffer and the byte
403         * buffer is direct.
404         * </p>
405         * 
406         * @return {@code true} if this buffer is direct, {@code false} otherwise.
407         * @since Android 1.0
408         */
409        public abstract boolean isDirect();
410    
411        /**
412         * Returns the byte order used by this buffer when converting doubles
413         * from/to bytes.
414         * <p>
415         * If this buffer is not based on a byte buffer, then this always returns
416         * the platform's native byte order.
417         * </p>
418         * 
419         * @return the byte order used by this buffer when converting doubles
420         *         from/to bytes.
421         * @since Android 1.0
422         */
423        public abstract ByteOrder order();

444        abstract boolean protectedHasArray();

446        /**
447         * Writes the given double to the current position and increases the
448         * position by 1.
449         * 
450         * @param d
451         *            the double to write.
452         * @return this buffer.
453         * @exception BufferOverflowException
454         *                if position is equal or greater than limit.
455         * @exception ReadOnlyBufferException
456         *                if no changes may be made to the contents of this buffer.
457         * @since Android 1.0
458         */
459        public abstract DoubleBuffer put(double d);
460    
461        /**
462         * Writes doubles from the given double array to the current position and
463         * increases the position by the number of doubles written.
464         * <p>
465         * Calling this method has the same effect as
466         * {@code put(src, 0, src.length)}.
467         * </p>
468         * 
469         * @param src
470         *            the source double array.
471         * @return this buffer.
472         * @exception BufferOverflowException
473         *                if {@code remaining()} is less than {@code src.length}.
474         * @exception ReadOnlyBufferException
475         *                if no changes may be made to the contents of this buffer.
476         * @since Android 1.0
477         */
478        public final DoubleBuffer put(double[] src) {
479            return put(src, 0, src.length);
480        }
481    
482        /**
483         * Writes doubles from the given double array, starting from the specified
484         * offset, to the current position and increases the position by the number
485         * of doubles written.
486         * 
487         * @param src
488         *            the source double array.
489         * @param off
490         *            the offset of double array, must not be negative and not
491         *            greater than {@code src.length}.
492         * @param len
493         *            the number of doubles to write, must be no less than zero and
494         *            not greater than {@code src.length - off}.
495         * @return this buffer.
496         * @exception BufferOverflowException
497         *                if {@code remaining()} is less than {@code len}.
498         * @exception IndexOutOfBoundsException
499         *                if either {@code off} or {@code len} is invalid.
500         * @exception ReadOnlyBufferException
501         *                if no changes may be made to the contents of this buffer.
502         * @since Android 1.0
503         */
504        public DoubleBuffer put(double[] src, int off, int len) {
505            int length = src.length;
506            if (off < 0 || len < 0 || (long)off + (long)len > length) {
507                throw new IndexOutOfBoundsException();
508            }
509            
510            if (len > remaining()) {
511                throw new BufferOverflowException();
512            }
513            for (int i = off; i < off + len; i++) {
514                put(src[i]);
515            }
516            return this;
517        }
518    
519        /**
520         * Writes all the remaining doubles of the {@code src} double buffer to this
521         * buffer's current position, and increases both buffers' position by the
522         * number of doubles copied.
523         * 
524         * @param src
525         *            the source double buffer.
526         * @return this buffer.
527         * @exception BufferOverflowException
528         *                if {@code src.remaining()} is greater than this buffer's
529         *                {@code remaining()}.
530         * @exception IllegalArgumentException
531         *                if {@code src} is this buffer.
532         * @exception ReadOnlyBufferException
533         *                if no changes may be made to the contents of this buffer.
534         * @since Android 1.0
535         */
536        public DoubleBuffer put(DoubleBuffer src) {
537            if (src == this) {
538                throw new IllegalArgumentException();
539            }
540            if (src.remaining() > remaining()) {
541                throw new BufferOverflowException();
542            }
543            double[] doubles = new double[src.remaining()];
544            src.get(doubles);
545            put(doubles);
546            return this;
547        }
548    
549        /**
550         * Write a double to the specified index of this buffer and the position is
551         * not changed.
552         * 
553         * @param index
554         *            the index, must not be negative and less than the limit.
555         * @param d
556         *            the double to write.
557         * @return this buffer.
558         * @exception IndexOutOfBoundsException
559         *                if index is invalid.
560         * @exception ReadOnlyBufferException
561         *                if no changes may be made to the contents of this buffer.
562         * @since Android 1.0
563         */
564        public abstract DoubleBuffer put(int index, double d);

566        /**
567         * Returns a sliced buffer that shares its content with this buffer.
568         * <p>
569         * The sliced buffer's capacity will be this buffer's {@code remaining()},
570         * and its zero position will correspond to this buffer's current position.
571         * The new buffer's position will be 0, limit will be its capacity, and its
572         * mark is cleared. The new buffer's read-only property and byte order are
573         * the same as this buffer's.
574         * </p>
575         * <p>
576         * The new buffer shares its content with this buffer, which means either
577         * buffer's change of content will be visible to the other. The two buffer's
578         * position, limit and mark are independent.
579         * </p>
580         * 
581         * @return a sliced buffer that shares its content with this buffer.
582         * @since Android 1.0
583         */
584        public abstract DoubleBuffer slice();
585    
586        /**
587         * Returns a string representing the state of this double buffer.
588         * 
589         * @return A string representing the state of this double buffer.
590         * @since Android 1.0
591         */
592        public String toString() {
593            StringBuffer buf = new StringBuffer();
594            buf.append(getClass().getName());
595            buf.append(", status: capacity="); //$NON-NLS-1$
596            buf.append(capacity());
597            buf.append(" position="); //$NON-NLS-1$
598            buf.append(position());
599            buf.append(" limit="); //$NON-NLS-1$    
600            buf.append(limit());
601            return buf.toString();
602        }
