018    package java.nio;
019    
020    /**
021     * A buffer of shorts.
022     * <p>
023     * A short buffer can be created in either of the following ways:
024     * </p>
025     * <ul>
026     * <li>{@link #allocate(int) Allocate} a new short array and create a buffer
027     * based on it;</li>
028     * <li>{@link #wrap(short[]) Wrap} an existing short array to create a new
029     * buffer;</li>
030     * <li>Use {@link java.nio.ByteBuffer#asShortBuffer() ByteBuffer.asShortBuffer}
031     * to create a short buffer based on a byte buffer.</li>
032     * </ul>
033     * 
034     * @since Android 1.0
035     */
036    public abstract class ShortBuffer extends Buffer implements Comparable<ShortBuffer> {
037    
038        /**
039         * Creates a short buffer based on a newly allocated short array.
040         * 
041         * @param capacity
042         *            the capacity of the new buffer.
043         * @return the created short buffer.
044         * @throws IllegalArgumentException
045         *             if {@code capacity} is less than zero.
046         * @since Android 1.0
047         */
048        public static ShortBuffer allocate(int capacity) {
049            if (capacity < 0) {
050                throw new IllegalArgumentException();
051            }
052            return BufferFactory.newShortBuffer(capacity);
053        }
054    
055        /**
056         * Creates a new short buffer by wrapping the given short array.
057         * <p>
058         * Calling this method has the same effect as
059         * {@code wrap(array, 0, array.length)}.
060         * </p>
061         * 
062         * @param array
063         *            the short array which the new buffer will be based on.
064         * @return the created short buffer.
065         * @since Android 1.0
066         */
067        public static ShortBuffer wrap(short[] array) {
068            return wrap(array, 0, array.length);
069        }
070    
071        /**
072         * Creates a new short buffer by wrapping the given short array.
073         * <p>
074         * The new buffer's position will be {@code start}, limit will be
075         * {@code start + len}, capacity will be the length of the array.
076         * </p>
077         * 
078         * @param array
079         *            the short array which the new buffer will be based on.
080         * @param start
081         *            the start index, must not be negative and not greater than
082         *            {@code array.length}.
083         * @param len
084         *            the length, must not be negative and not greater than
085         *            {@code array.length - start}.
086         * @return the created short buffer.
087         * @exception IndexOutOfBoundsException
088         *                if either {@code start} or {@code len} is invalid.
089         * @since Android 1.0
090         */
091        public static ShortBuffer wrap(short[] array, int start, int len) {
092                    if (array == null) {
093                            throw new NullPointerException();
094                    }
095                    if (start< 0 || len < 0 || (long)start + (long)len > array.length) {
096                            throw new IndexOutOfBoundsException();
097                    }
098    
099            ShortBuffer buf = BufferFactory.newShortBuffer(array);
100            buf.position = start;
101            buf.limit = start + len;
102    
103            return buf;
104        }
105    
106        /**
107         * Constructs a {@code ShortBuffer} with given capacity.
108         *
109         * @param capacity
110         *            The capacity of the buffer
111         */
112        ShortBuffer(int capacity) {
113            super(capacity);
114            // BEGIN android-added
115            _elementSizeShift = 1;
116            // END android-added
117        }

119        /**
120         * Returns the short array which this buffer is based on, if there is one.
121         * 
122         * @return the short array which this buffer is based on.
123         * @exception ReadOnlyBufferException
124         *                if this buffer is based on an array, but it is read-only.
125         * @exception UnsupportedOperationException
126         *                if this buffer is not based on an array.
127         * @since Android 1.0
128         */
129        public final short[] array() {
130            return protectedArray();
131        }
132    
133        /**
134         * Returns the offset of the short array which this buffer is based on, if
135         * there is one.
136         * <p>
137         * The offset is the index of the array corresponding to the zero position
138         * of the buffer.
139         * </p>
140         * 
141         * @return the offset of the short array which this buffer is based on.
142         * @exception ReadOnlyBufferException
143         *                if this buffer is based on an array, but it is read-only.
144         * @exception UnsupportedOperationException
145         *                if this buffer is not based on an array.
146         * @since Android 1.0
147         */
148        public final int arrayOffset() {
149            return protectedArrayOffset();
150        }
151    
152        // BEGIN android-added
153        @Override Object _array() {
154            if (hasArray()) {
155                return array();
156            }
157            return null;
158        }
159    
160        @Override int _arrayOffset() {
161            if (hasArray()) {
162                return arrayOffset();
163            }
164            return 0;
165        }

168        /**
169         * Returns a read-only buffer that shares its content with this buffer.
170         * <p>
171         * The returned buffer is guaranteed to be a new instance, even if this
172         * buffer is read-only itself. The new buffer's position, limit, capacity
173         * and mark are the same as this buffer's.
174         * </p>
175         * <p>
176         * The new buffer shares its content with this buffer, which means this
177         * buffer's change of content will be visible to the new buffer. The two
178         * buffer's position, limit and mark are independent.
179         * </p>
180         * 
181         * @return a read-only version of this buffer.
182         * @since Android 1.0
183         */
184        public abstract ShortBuffer asReadOnlyBuffer();
185    
186        /**
187         * Compacts this short buffer.
188         * <p>
189         * The remaining shorts will be moved to the head of the buffer, starting
190         * from position zero. Then the position is set to {@code remaining()}; the
191         * limit is set to capacity; the mark is cleared.
192         * </p>
193         * 
194         * @return this buffer.
195         * @exception ReadOnlyBufferException
196         *                if no changes may be made to the contents of this buffer.
197         * @since Android 1.0
198         */
199        public abstract ShortBuffer compact();
200    
201        /**
202         * Compare the remaining shorts of this buffer to another short buffer's
203         * remaining shorts.
204         * 
205         * @param otherBuffer
206         *            another short buffer.
207         * @return a negative value if this is less than {@code otherBuffer}; 0 if
208         *         this equals to {@code otherBuffer}; a positive value if this is
209         *         greater than {@code otherBuffer}.
210         * @exception ClassCastException
211         *                if {@code otherBuffer} is not a short buffer.
212         * @since Android 1.0
213         */
214        public int compareTo(ShortBuffer otherBuffer) {
215            int compareRemaining = (remaining() < otherBuffer.remaining()) ? remaining()
216                    : otherBuffer.remaining();
217            int thisPos = position;
218            int otherPos = otherBuffer.position;
219            short thisByte, otherByte;
220            while (compareRemaining > 0) {
221                thisByte = get(thisPos);
222                otherByte = otherBuffer.get(otherPos);
223                if (thisByte != otherByte) {
224                    return thisByte < otherByte ? -1 : 1;
225                }
226                thisPos++;
227                otherPos++;
228                compareRemaining--;
229            }
230            return remaining() - otherBuffer.remaining();
231        }
232    
233        /**
234         * Returns a duplicated buffer that shares its content with this buffer.
235         * <p>
236         * The duplicated buffer's position, limit, capacity and mark are the same
237         * as this buffer. The duplicated buffer's read-only property and byte order
238         * are the same as this buffer's.
239         * </p>
240         * <p>
241         * The new buffer shares its content with this buffer, which means either
242         * buffer's change of content will be visible to the other. The two buffer's
243         * position, limit and mark are independent.
244         * </p>
245         * 
246         * @return a duplicated buffer that shares its content with this buffer.
247         * @since Android 1.0
248         */
249        public abstract ShortBuffer duplicate();
250    
251        /**
252         * Checks whether this short buffer is equal to another object.
253         * <p>
254         * If {@code other} is not a short buffer then {@code false} is returned.
255         * Two short buffers are equal if and only if their remaining shorts are
256         * exactly the same. Position, limit, capacity and mark are not considered.
257         * </p>
258         * 
259         * @param other
260         *            the object to compare with this short buffer.
261         * @return {@code true} if this short buffer is equal to {@code other},
262         *         {@code false} otherwise.
263         * @since Android 1.0
264         */
265        public boolean equals(Object other) {
266            if (!(other instanceof ShortBuffer)) {
267                return false;
268            }
269            ShortBuffer otherBuffer = (ShortBuffer) other;
270    
271            if (remaining() != otherBuffer.remaining()) {
272                return false;
273            }
274    
275            int myPosition = position;
276            int otherPosition = otherBuffer.position;
277            boolean equalSoFar = true;
278            while (equalSoFar && (myPosition < limit)) {
279                equalSoFar = get(myPosition++) == otherBuffer.get(otherPosition++);
280            }
281    
282            return equalSoFar;
283        }
284    
285        /**
286         * Returns the short at the current position and increases the position by
287         * 1.
288         * 
289         * @return the short at the current position.
290         * @exception BufferUnderflowException
291         *                if the position is equal or greater than limit.
292         * @since Android 1.0
293         */
294        public abstract short get();
295    
296        /**
297         * Reads shorts from the current position into the specified short array and
298         * increases the position by the number of shorts read.
299         * <p>
300         * Calling this method has the same effect as
301         * {@code get(dest, 0, dest.length)}.
302         * </p>
303         * 
304         * @param dest
305         *            the destination short array.
306         * @return this buffer.
307         * @exception BufferUnderflowException
308         *                if {@code dest.length} is greater than {@code remaining()}.
309         * @since Android 1.0
310         */
311        public ShortBuffer get(short[] dest) {
312            return get(dest, 0, dest.length);
313        }
314    
315        /**
316         * Reads shorts from the current position into the specified short array,
317         * starting from the specified offset, and increases the position by the
318         * number of shorts read.
319         * 
320         * @param dest
321         *            the target short array.
322         * @param off
323         *            the offset of the short array, must not be negative and not
324         *            greater than {@code dest.length}.
325         * @param len
326         *            the number of shorts to read, must be no less than zero and
327         *            not greater than {@code dest.length - off}.
328         * @return this buffer.
329         * @exception IndexOutOfBoundsException
330         *                if either {@code off} or {@code len} is invalid.
331         * @exception BufferUnderflowException
332         *                if {@code len} is greater than {@code remaining()}.
333         * @since Android 1.0
334         */
335        public ShortBuffer get(short[] dest, int off, int len) {
336            int length = dest.length;
337            if (off < 0 || len < 0 || (long)off + (long)len > length) {
338                throw new IndexOutOfBoundsException();
339            }
340            if (len > remaining()) {
341                throw new BufferUnderflowException();
342            }
343            for (int i = off; i < off + len; i++) {
344                dest[i] = get();
345            }
346            return this;
347        }
348    
349        /**
350         * Returns the short at the specified index; the position is not changed.
351         * 
352         * @param index
353         *            the index, must not be negative and less than limit.
354         * @return a short at the specified index.
355         * @exception IndexOutOfBoundsException
356         *                if index is invalid.
357         * @since Android 1.0
358         */
359        public abstract short get(int index);
360    
361        /**
362         * Indicates whether this buffer is based on a short array and is
363         * read/write.
364         * 
365         * @return {@code true} if this buffer is based on a short array and
366         *         provides read/write access, {@code false} otherwise.
367         * @since Android 1.0
368         */
369        public final boolean hasArray() {
370            return protectedHasArray();
371        }
372    
373        /**
374         * Calculates this buffer's hash code from the remaining chars. The
375         * position, limit, capacity and mark don't affect the hash code.
376         * 
377         * @return the hash code calculated from the remaining shorts.
378         * @since Android 1.0
379         */
380        public int hashCode() {
381            int myPosition = position;
382            int hash = 0;
383            while (myPosition < limit) {
384                hash = hash + get(myPosition++);
385            }
386            return hash;
387        }
388    
389        /**
390         * Indicates whether this buffer is direct. A direct buffer will try its
391         * best to take advantage of native memory APIs and it may not stay in the
392         * Java heap, so it is not affected by garbage collection.
393         * <p>
394         * A short buffer is direct if it is based on a byte buffer and the byte
395         * buffer is direct.
396         * </p>
397         * 
398         * @return {@code true} if this buffer is direct, {@code false} otherwise.
399         * @since Android 1.0
400         */
401        public abstract boolean isDirect();
402    
403        /**
404         * Returns the byte order used by this buffer when converting shorts from/to
405         * bytes.
406         * <p>
407         * If this buffer is not based on a byte buffer, then always return the
408         * platform's native byte order.
409         * </p>
410         * 
411         * @return the byte order used by this buffer when converting shorts from/to
412         *         bytes.
413         * @since Android 1.0
414         */
415        public abstract ByteOrder order();

418         * Child class implements this method to realize {@code array()}.

420         * @return see {@code array()}

422        abstract short[] protectedArray();
423

425         * Child class implements this method to realize {@code arrayOffset()}.

429        abstract int protectedArrayOffset();
430    
431        /**
432         * Child class implements this method to realize {@code hasArray()}.
433         *
434         * @return see {@code hasArray()}
435         */
436        abstract boolean protectedHasArray();

438        /**
439         * Writes the given short to the current position and increases the position
440         * by 1.
441         * 
442         * @param s
443         *            the short to write.
444         * @return this buffer.
445         * @exception BufferOverflowException
446         *                if position is equal or greater than limit.
447         * @exception ReadOnlyBufferException
448         *                if no changes may be made to the contents of this buffer.
449         * @since Android 1.0
450         */
451        public abstract ShortBuffer put(short s);
452    
453        /**
454         * Writes shorts from the given short array to the current position and
455         * increases the position by the number of shorts written.
456         * <p>
457         * Calling this method has the same effect as
458         * {@code put(src, 0, src.length)}.
459         * </p>
460         * 
461         * @param src
462         *            the source short array.
463         * @return this buffer.
464         * @exception BufferOverflowException
465         *                if {@code remaining()} is less than {@code src.length}.
466         * @exception ReadOnlyBufferException
467         *                if no changes may be made to the contents of this buffer.
468         * @since Android 1.0
469         */
470        public final ShortBuffer put(short[] src) {
471            return put(src, 0, src.length);
472        }
473    
474        /**
475         * Writes shorts from the given short array, starting from the specified
476         * offset, to the current position and increases the position by the number
477         * of shorts written.
478         * 
479         * @param src
480         *            the source short array.
481         * @param off
482         *            the offset of short array, must not be negative and not
483         *            greater than {@code src.length}.
484         * @param len
485         *            the number of shorts to write, must be no less than zero and
486         *            not greater than {@code src.length - off}.
487         * @return this buffer.
488         * @exception BufferOverflowException
489         *                if {@code remaining()} is less than {@code len}.
490         * @exception IndexOutOfBoundsException
491         *                if either {@code off} or {@code len} is invalid.
492         * @exception ReadOnlyBufferException
493         *                if no changes may be made to the contents of this buffer.
494         * @since Android 1.0
495         */
496        public ShortBuffer put(short[] src, int off, int len) {
497            int length = src.length;
498            if (off < 0 || len < 0 || (long)off + (long)len > length) {
499                throw new IndexOutOfBoundsException();
500            }
501            
502            if (len > remaining()) {
503                throw new BufferOverflowException();
504            }
505            for (int i = off; i < off + len; i++) {
506                put(src[i]);
507            }
508            return this;
509        }
510    
511        /**
512         * Writes all the remaining shorts of the {@code src} short buffer to this
513         * buffer's current position, and increases both buffers' position by the
514         * number of shorts copied.
515         * 
516         * @param src
517         *            the source short buffer.
518         * @return this buffer.
519         * @exception BufferOverflowException
520         *                if {@code src.remaining()} is greater than this buffer's
521         *                {@code remaining()}.
522         * @exception IllegalArgumentException
523         *                if {@code src} is this buffer.
524         * @exception ReadOnlyBufferException
525         *                if no changes may be made to the contents of this buffer.
526         * @since Android 1.0
527         */
528        public ShortBuffer put(ShortBuffer src) {
529            if (src == this) {
530                throw new IllegalArgumentException();
531            }
532            if (src.remaining() > remaining()) {
533                throw new BufferOverflowException();
534            }
535            short[] contents = new short[src.remaining()];
536            src.get(contents);
537            put(contents);
538            return this;
539        }

544         * 
545         * @param index
546         *            the index, must not be negative and less than the limit.
547         * @param s
548         *            the short to write.
549         * @return this buffer.
550         * @exception IndexOutOfBoundsException
551         *                if index is invalid.
552         * @exception ReadOnlyBufferException
553         *                if no changes may be made to the contents of this buffer.
554         * @since Android 1.0
555         */
556        public abstract ShortBuffer put(int index, short s);

558        /**
559         * Returns a sliced buffer that shares its content with this buffer.
560         * <p>
561         * The sliced buffer's capacity will be this buffer's {@code remaining()},
562         * and its zero position will correspond to this buffer's current position.
563         * The new buffer's position will be 0, limit will be its capacity, and its
564         * mark is cleared. The new buffer's read-only property and byte order are
565         * same as this buffer's.
566         * </p>
567         * <p>
568         * The new buffer shares its content with this buffer, which means either
569         * buffer's change of content will be visible to the other. The two buffer's
570         * position, limit and mark are independent.
571         * </p>
572         * 
573         * @return a sliced buffer that shares its content with this buffer.
574         * @since Android 1.0
575         */
576        public abstract ShortBuffer slice();
577    
578        /**
579         * Returns a string representing the state of this short buffer.
580         * 
581         * @return a string representing the state of this short buffer.
582         * @since Android 1.0
583         */
584        public String toString() {
585            StringBuffer buf = new StringBuffer();
586            buf.append(getClass().getName());
587            buf.append(", status: capacity="); //$NON-NLS-1$
588            buf.append(capacity());
589            buf.append(" position="); //$NON-NLS-1$
590            buf.append(position());
591            buf.append(" limit="); //$NON-NLS-1$
592            buf.append(limit());
593            return buf.toString();
594        }
