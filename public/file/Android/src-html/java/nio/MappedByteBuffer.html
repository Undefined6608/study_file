017    package java.nio;
018    
019    import org.apache.harmony.luni.platform.IMemorySystem;
020    import org.apache.harmony.luni.platform.MappedPlatformAddress;
021    import org.apache.harmony.luni.platform.PlatformAddress;
022    import org.apache.harmony.nio.internal.DirectBuffer;
023    
024    
025    /**
026     * {@code MappedByteBuffer} is a special kind of direct byte buffer which maps a
027     * region of file to memory.
028     * <p>
029     * {@code MappedByteBuffer} can be created by calling
030     * {@link java.nio.channels.FileChannel#map(java.nio.channels.FileChannel.MapMode, long, long) FileChannel.map}.
031     * Once created, the mapping between the byte buffer and the file region remains
032     * valid until the byte buffer is garbage collected.
033     * </p>
034     * <p>
035     * All or part of a {@code MappedByteBuffer}'s content may change or become
036     * inaccessible at any time, since the mapped file region can be modified by
037     * another thread or process at any time. If this happens, the behavior of the
038     * {@code MappedByteBuffer} is undefined.
039     * </p>
040     * 
041     * @since Android 1.0
042     */
043    public abstract class MappedByteBuffer extends ByteBuffer {
044    
045        final DirectByteBuffer wrapped;
046    
047        private int mapMode;
048    
049        MappedByteBuffer(ByteBuffer directBuffer) {
050            super(directBuffer.capacity);
051            if (!directBuffer.isDirect()) {
052                throw new IllegalArgumentException();
053            }
054            this.wrapped = (DirectByteBuffer) directBuffer;
055    
056        }
057    
058        MappedByteBuffer(PlatformAddress addr, int capa, int offset, int mode) {
059            super(capa);
060            mapMode = mode;
061            switch (mapMode) {
062            case IMemorySystem.MMAP_READ_ONLY:
063                wrapped = new ReadOnlyDirectByteBuffer(addr, capa, offset);
064                break;
065            case IMemorySystem.MMAP_READ_WRITE:
066            case IMemorySystem.MMAP_WRITE_COPY:
067                wrapped = new ReadWriteDirectByteBuffer(addr, capa, offset);
068                break;
069            default:
070                throw new IllegalArgumentException();
071            }
072            addr.autoFree();
073        }
074    
075        /**
076         * Indicates whether this buffer's content is loaded. If the result is true
077         * there is a high probability that the whole buffer memory is currently
078         * loaded in RAM. If it is false it is unsure if it is loaded or not.
079         * 
080         * @return {@code true} if this buffer's content is loaded, {@code false}
081         *         otherwise.
082         * @since Android 1.0
083         */
084        public final boolean isLoaded() {
085            return ((MappedPlatformAddress)((DirectBuffer) wrapped).getBaseAddress()).mmapIsLoaded();
086        }
087    
088        /**
089         * Loads this buffer's content into memory but it is not guaranteed to
090         * succeed.
091         * 
092         * @return this buffer.
093         * @since Android 1.0
094         */
095        public final MappedByteBuffer load() {
096            ((MappedPlatformAddress)((DirectBuffer) wrapped).getBaseAddress()).mmapLoad();
097            return this;
098        }
099    
100        /**
101         * Writes all changes of the buffer to the mapped file. If the mapped file
102         * is stored on a local device, it is guaranteed that the changes are
103         * written to the file. No such guarantee is given if the file is located on
104         * a remote device.
105         * 
106         * @return this buffer.
107         * @since Android 1.0
108         */
109        public final MappedByteBuffer force() {
110            if (mapMode == IMemorySystem.MMAP_READ_WRITE) {
111                ((MappedPlatformAddress)((DirectBuffer) wrapped).getBaseAddress()).mmapFlush();
112            }
113            return this;
114        }
115    }
