054        /**
055         * Returns an input stream on the given channel. The resulting stream has
056         * the following properties:
057         * <ul>
058         * <li>If the stream is closed, then the underlying channel is closed as
059         * well.</li>
060         * <li>It is thread safe.</li>
061         * <li>It throws an {@link IllegalBlockingModeException} if the channel is
062         * in non-blocking mode and {@code read} is called.</li>
063         * <li>Neither {@code mark} nor {@code reset} is supported.</li>
064         * <li>It is not buffered.</li>
065         * </ul>
066         * 
067         * @param channel
068         *            the channel to be wrapped by an InputStream.
069         * @return an InputStream that takes bytes from the given byte channel.
070         * @since Android 1.0
071         */
072        public static InputStream newInputStream(ReadableByteChannel channel) {
073            return new ReadableByteChannelInputStream(channel);
074        }
075    
076        /**
077         * Returns an output stream on the given channel. The resulting stream has
078         * the following properties:
079         * <ul>
080         * <li>If the stream is closed, then the underlying channel is closed as
081         * well.</li>
082         * <li>It is thread safe.</li>
083         * <li>It throws an {@link IllegalBlockingModeException} if the channel is
084         * in non-blocking mode and {@code write} is called.</li>
085         * <li>It is not buffered.</li>
086         * </ul>
087         * 
088         * @param channel
089         *            the channel to be wrapped by an OutputStream.
090         * @return an OutputStream that puts bytes onto the given byte channel.
091         * @since Android 1.0
092         */
093        public static OutputStream newOutputStream(WritableByteChannel channel) {
094            return new WritableByteChannelOutputStream(channel);
095        }

214        /*
215         * wrap a byte array to a ByteBuffer
216         */
217        static ByteBuffer wrapByteBuffer(byte[] bytes, int offset, int length) {
218            ByteBuffer buffer = ByteBuffer.wrap(bytes);
219            int newLimit = offset + length <= buffer.capacity() ? offset + length
220                    : buffer.capacity();
221            buffer.limit(newLimit);
222            buffer.position(offset);
223            return buffer;
224        }

230        private static class ChannelInputStream extends InputStream {
231    
232            protected ReadableByteChannel channel;
233    
234            public ChannelInputStream(ReadableByteChannel aChannel) {
235                super();
236                channel = aChannel;
237            }
238    
239            /*
240             * @see java.io.InputStream#read()
241             */
242            @Override
243            public synchronized int read() throws IOException {
244                byte[] oneByte = new byte[1];
245                int n = read(oneByte);
246                if (n == 1) {
247                    // reads a single byte 0-255
248                    return oneByte[0] & 0xff;
249                }
250                return -1;
251            }
252    
253            /*
254             * @see java.io.InputStream#close()
255             */
256            @Override
257            public synchronized void close() throws IOException {
258                channel.close();
259            }
260        }

262        /*
263         * Wrapper class used for newInputStream(ReadableByteChannel channel)
264         */
265        private static class ReadableByteChannelInputStream extends
266                ChannelInputStream {
267    
268            /*
269             * @param someChannel
270             */
271            public ReadableByteChannelInputStream(ReadableByteChannel aChannel) {
272                super(aChannel);
273            }
274    
275            /*
276             * @see java.io.InputStream#read(byte[], int, int)
277             */
278            @Override
279            public synchronized int read(byte[] target, int offset, int length)
280                    throws IOException {
281                // avoid int overflow, check null target
282                if (length + offset > target.length || length < 0 || offset < 0) {
283                    throw new ArrayIndexOutOfBoundsException();
284                }
285                if (0 == length) {
286                    return 0;
287                }
288                if (channel instanceof SelectableChannel) {
289                    if (!((SelectableChannel) channel).isBlocking()) {
290                        throw new IllegalBlockingModeException();
291                    }
292                }
293                ByteBuffer buffer = ByteBuffer.wrap(target, offset, length);
294                return channel.read(buffer);
295            }
296        }

329        /*
330         * Wrapper class used for newOutputStream(WritableByteChannel channel)
331         */
332        private static class WritableByteChannelOutputStream extends OutputStream {
333    
334            private WritableByteChannel channel;
335    
336            /*
337             * @param someChannel
338             */
339            public WritableByteChannelOutputStream(WritableByteChannel aChannel) {
340                super();
341                channel = aChannel;
342            }
343    
344            /*
345             * @see java.io.OutputStream#write(int)
346             */
347            public synchronized void write(int oneByte) throws IOException {
348                byte[] wrappedByte = new byte[1];
349                wrappedByte[0] = (byte) oneByte;
350                write(wrappedByte);
351            }
352    
353            /*
354             * @see java.io.OutputStream#write(byte[], int, int)
355             */
356            public synchronized void write(byte[] source, int offset, int length)
357                    throws IOException {
358                // avoid int overflow, check null source
359                if (length + offset > source.length || length < 0 || offset < 0) {
360                    throw new ArrayIndexOutOfBoundsException();
361                }
362                if (0 == length) {
363                    return;
364                }
365                if (channel instanceof SelectableChannel) {
366                    if (!((SelectableChannel) channel).isBlocking()) {
367                        throw new IllegalBlockingModeException();
368                    }
369                }
370                ByteBuffer buffer = ByteBuffer.wrap(source, offset, length);
371                channel.write(buffer);
372            }
373    
374            /*
375             * @see java.io.OutputStream#close()
376             */
377            public synchronized void close() throws IOException {
378                channel.close();
379            }
380        }

382        /*
383         * Wrapper class used for newChannel(InputStream inputStream)
384         */
385        private static class ReadableByteChannelImpl extends
386                AbstractInterruptibleChannel implements ReadableByteChannel {
387            private InputStream inputStream;
388    
389            ReadableByteChannelImpl(InputStream aInputStream) {
390                super();
391                inputStream = aInputStream;
392            }
393    
394            /*
395             * @see java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)
396             */
397            public synchronized int read(ByteBuffer target) throws IOException {
398                if (!isOpen()) {
399                    throw new ClosedChannelException();
400                }
401                int bytesRemain = target.remaining();
402                byte[] bytes = new byte[bytesRemain];
403                int readCount = 0;
404                try {
405                    begin();
406                    readCount = inputStream.read(bytes);
407                } finally {
408                    end(readCount >= 0);
409                }
410                if (readCount > 0) {
411                    target.put(bytes, 0, readCount);
412                }
413                return readCount;
414            }
415    
416            /*
417             * @see java.nio.channels.spi.AbstractInterruptibleChannel#implCloseChannel()
418             */
419            protected void implCloseChannel() throws IOException {
420                inputStream.close();
421            }
422        }

424        /*
425         * Wrapper class used for newChannel(OutputStream outputStream)
426         */
427        private static class WritableByteChannelImpl extends
428                AbstractInterruptibleChannel implements WritableByteChannel {
429            private OutputStream outputStream;
430    
431            WritableByteChannelImpl(OutputStream aOutputStream) {
432                super();
433                outputStream = aOutputStream;
434            }
435    
436            /*
437             * @see java.nio.channels.WritableByteChannel#write(java.nio.ByteBuffer)
438             */
439            public synchronized int write(ByteBuffer source) throws IOException {
440                if (!isOpen()) {
441                    throw new ClosedChannelException();
442                }
443                int bytesRemain = source.remaining();
444                if (bytesRemain == 0) {
445                    return 0;
446                }
447                byte[] buf = new byte[bytesRemain];
448                source.get(buf);
449                try {
450                    begin();
451                    outputStream.write(buf, 0, bytesRemain);
452                } finally {
453                    end(bytesRemain >= 0);
454                }
455                return bytesRemain;
456            }
457    
458            /*
459             * @see java.nio.channels.spi.AbstractInterruptibleChannel#implCloseChannel()
460             */
461            protected void implCloseChannel() throws IOException {
462                outputStream.close();
463            }
464        }

466        /*
467         * Wrapper class used for newReader(ReadableByteChannel channel,
468         * CharsetDecoder decoder, int minBufferCapacity)
469         */
470        private static class ByteChannelReader extends Reader {
471    
472            private InputStream inputStream;
473    
474            private static final int BUFFER_SIZE = 8192;
475    
476            CharsetDecoder decoder;
477    
478            ByteBuffer bytes;
479    
480            CharBuffer chars;
481    
482            /*
483             * @param inputStream @param dec @param minBufferCapacity
484             */
485            public ByteChannelReader(InputStream aInputStream,
486                    CharsetDecoder aDecoder, int minBufferCapacity) {
487                super(aInputStream);
488                aDecoder.reset();
489                inputStream = aInputStream;
490                int bufferSize = Math.max(minBufferCapacity, BUFFER_SIZE);
491                bytes = ByteBuffer.allocate(bufferSize);
492                chars = CharBuffer.allocate(bufferSize);
493                decoder = aDecoder;
494                chars.limit(0);
495            }
496    
497            /*
498             * @see java.io.Reader#close()
499             */
500            public void close() throws IOException {
501                synchronized (lock) {
502                    decoder = null;
503                    if (inputStream != null) {
504                        inputStream.close();
505                        inputStream = null;
506                    }
507                }
508            }
509    
510            /*
511             * @see java.io.Reader#ready()
512             */
513            public boolean ready() {
514                synchronized (lock) {
515                    if (null == inputStream) {
516                        return false;
517                    }
518                    try {
519                        return chars.limit() > chars.position()
520                                || inputStream.available() > 0;
521                    } catch (IOException e) {
522                        return false;
523                    }
524                }
525            }
526    
527            /*
528             * @see java.io.Reader#read()
529             */
530            public int read() throws IOException {
531                return IOUtil.readInputStreamReader(inputStream,
532                        bytes, chars, decoder, lock);
533            }
534    
535            /*
536             * @see java.io.Reader#read(char[], int, int)
537             */
538            public int read(char[] buf, int offset, int length) throws IOException {
539                return IOUtil.readInputStreamReader(buf, offset,
540                        length, inputStream, bytes, chars, decoder, lock);
541            }
542        }

544        /*
545         * Wrapper class used for newWriter(WritableByteChannel channel,
546         * CharsetEncoder encoder, int minBufferCapacity)
547         */
548        private static class ByteChannelWriter extends Writer {
549    
550            private static final int BUFFER_SIZE = 8192;
551    
552            private OutputStream outputStream;
553    
554            private CharsetEncoder encoder;
555    
556            private ByteBuffer byteBuf;
557    
558            /*
559             * @param outputStream @param enc @param minBufferCap
560             */
561            public ByteChannelWriter(OutputStream aOutputStream,
562                    CharsetEncoder aEncoder, int minBufferCap) {
563                super(aOutputStream);
564                aEncoder.charset();
565                outputStream = aOutputStream;
566                byteBuf = ByteBuffer.allocate(Math.max(minBufferCap, BUFFER_SIZE));
567                encoder = aEncoder;
568            }
569    
570            /*
571             * @see java.io.Writer#close()
572             */
573            public void close() throws IOException {
574                synchronized (lock) {
575                    if (encoder != null) {
576                        flush();
577                        outputStream.flush();
578                        outputStream.close();
579                        encoder = null;
580                        byteBuf = null;
581                    }
582                }
583            }
584    
585            /*
586             * @see java.io.Writer#flush()
587             */
588            public void flush() throws IOException {
589                IOUtil.flushOutputStreamWriter(outputStream,
590                        byteBuf, encoder, lock);
591            }
592    
593            /*
594             * @see java.io.Writer#write(char[], int, int)
595             */
596            public void write(char[] buf, int offset, int count) throws IOException {
597                IOUtil.writeOutputStreamWriter(buf, offset, count,
598                        outputStream, byteBuf, encoder, lock);
599            }
600    
601            /*
602             * @see java.io.Writer#write(int)
603             */
604            public void write(int oneChar) throws IOException {
605                IOUtil.writeOutputStreamWriter(oneChar,
606                        outputStream, byteBuf, encoder, lock);
607            }
608    
609            /*
610             * @see java.io.Writer#write(java.lang.String, int, int)
611             */
612            public void write(String str, int offset, int count) throws IOException {
613                IOUtil.writeOutputStreamWriter(str, offset, count,
614                        outputStream, byteBuf, encoder, lock);
615            }
616        }
