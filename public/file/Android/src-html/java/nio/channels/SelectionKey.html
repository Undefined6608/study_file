017    package java.nio.channels;
018    
019    import java.nio.channels.Selector;
020    
021    /**
022     * A {@code SelectionKey} represents the relationship between a channel and a
023     * selector for which the channel is registered.
024     * <h3>Operation set</h3>
025     * An operation set is represented by an integer value. The bits of an operation
026     * set represent categories of operations for a key's channel: Accepting socket
027     * connections ({@code OP_ACCEPT}), connecting with a socket ({@code OP_CONNECT}),
028     * reading ({@code OP_READ}) and writing ({@code OP_WRITE}).
029     * <h4>Interest set</h4>
030     * The interest set is an operation set that defines the operations that a
031     * {@link SelectableChannel channel} is interested in performing.
032     * <h4>Ready set</h4>
033     * The ready set is an operation set that shows the operations that a
034     * {@code channel} is ready to execute.
035     * 
036     * @since Android 1.0
037     */
038    public abstract class SelectionKey {
039    
040        /**
041         * Interest set mask bit for socket-accept operations.
042         * 
043         * @since Android 1.0
044         */
045        public static final int OP_ACCEPT = 16;
046    
047        /**
048         * Interest set mask bit for socket-connect operations.
049         * 
050         * @since Android 1.0
051         */
052        public static final int OP_CONNECT = 8;
053    
054        /**
055         * Interesting operation mask bit for read operations.
056         * 
057         * @since Android 1.0
058         */
059        public static final int OP_READ = 1;
060    
061        /**
062         * Interest set mask bit for write operations.
063         * 
064         * @since Android 1.0
065         */
066        public static final int OP_WRITE = 4;
067    
068        private volatile Object attachment = null;
069    
070        /**
071         * Constructs a new {@code SelectionKey}.
072         * 
073         * @since Android 1.0
074         */
075        protected SelectionKey() {
076            super();
077        }
078    
079        /**
080         * Attaches an object to this key. It is acceptable to attach {@code null},
081         * this discards the old attachment.
082         * 
083         * @param anObject
084         *            the object to attach, or {@code null} to discard the current
085         *            attachment.
086         * @return the last attached object or {@code null} if no object has been
087         *         attached.
088         * @since Android 1.0
089         */
090        public final Object attach(Object anObject) {
091            Object oldAttachment = attachment;
092            attachment = anObject;
093            return oldAttachment;
094        }
095    
096        /**
097         * Gets the attached object.
098         * 
099         * @return the attached object or {@code null} if no object has been
100         *         attached.
101         * @since Android 1.0
102         */
103        public final Object attachment() {
104            return attachment;
105        }
106    
107        /**
108         * Cancels this key.
109         * <p>
110         * A key that has been canceled is no longer valid. Calling this method on
111         * an already canceled key does nothing.
112         * </p>
113         * <p>
114         * Calling this method is safe at any time. The call might block until
115         * another ongoing call to a method of this selector has finished. The
116         * reason is that it is synchronizing on the key set of the selector. After
117         * this call finishes, the key will have been added to the selectors
118         * canceled-keys set and will not be included in any future selects of this
119         * selector.
120         * </p>
121         * 
122         * @since Android 1.0
123         */
124        public abstract void cancel();
125    
126        /**
127         * Gets the channel of this key.
128         * 
129         * @return the channel of this key.
130         * @since Android 1.0
131         */
132        public abstract SelectableChannel channel();
133    
134        /**
135         * Gets this key's {@link SelectionKey interest set}. The returned set has
136         * only those bits set that are valid for this key's channel.
137         * 
138         * @return the interest set of this key.
139         * @throws CancelledKeyException
140         *             if the key has already been canceled.
141         * @since Android 1.0
142         */
143        public abstract int interestOps();
144    
145        /**
146         * Sets the {@link SelectionKey interest set} for this key.
147         * 
148         * @param operations
149         *            the new interest set.
150         * @return this key.
151         * @throws IllegalArgumentException
152         *             if a bit in {@code operations} is not in the set of
153         *             {@link SelectableChannel#validOps() valid operations} of this
154         *             key's channel.
155         * @throws CancelledKeyException
156         *             if the key has already been canceled.
157         * @since Android 1.0
158         */
159        public abstract SelectionKey interestOps(int operations);
160    
161        /**
162         * Indicates whether this key's channel is interested in the accept
163         * operation and is ready to accept new connections. A call to this method
164         * is equal to executing {@code (readyOps() & OP_ACCEPT) == OP_ACCEPT}.
165         * 
166         * @return {@code true} if the channel is interested in the accept operation
167         *         and is ready to accept new connections, {@code false} otherwise.
168         * @throws CancelledKeyException
169         *             if the key has already been canceled.
170         * @since Android 1.0
171         */
172        public final boolean isAcceptable() {
173            return (readyOps() & OP_ACCEPT) == OP_ACCEPT;
174        }
175    
176        /**
177         * Indicates whether this key's channel is interested in the connect
178         * operation and is ready to connect. A call to this method is equal to
179         * executing {@code (readyOps() & OP_CONNECT) == OP_CONNECT}.
180         * 
181         * @return {@code true} if the channel is interested in the connect
182         *         operation and is ready to connect, {@code false} otherwise.
183         * @throws CancelledKeyException
184         *             if the key has already been canceled.
185         * @since Android 1.0
186         */    
187        public final boolean isConnectable() {
188            return (readyOps() & OP_CONNECT) == OP_CONNECT;
189        }
190    
191        /**
192         * Indicates whether this key's channel is interested in the read operation
193         * and is ready to read. A call to this method is equal to executing
194         * {@code (readyOps() & OP_READ) == OP_READ}.
195         * 
196         * @return {@code true} if the channel is interested in the read operation
197         *         and is ready to read, {@code false} otherwise.
198         * @throws CancelledKeyException
199         *             if the key has already been canceled.
200         * @since Android 1.0
201         */
202        public final boolean isReadable() {
203            return (readyOps() & OP_READ) == OP_READ;
204        }
205    
206        /**
207         * Indicates whether this key is valid. A key is valid as long as it has not
208         * been canceled.
209         * 
210         * @return {@code true} if this key has not been canceled, {@code false}
211         *         otherwise.
212         * @since Android 1.0
213         */
214        public abstract boolean isValid();
215    
216        /**
217         * Indicates whether this key's channel is interested in the write operation
218         * and is ready to write. A call to this method is equal to executing
219         * {@code (readyOps() & OP_WRITE) == OP_WRITE}.
220         * 
221         * @return {@code true} if the channel is interested in the wrie operation
222         *         and is ready to write, {@code false} otherwise.
223         * @throws CancelledKeyException
224         *             if the key has already been canceled.
225         * @since Android 1.0
226         */
227        public final boolean isWritable() {
228            return (readyOps() & OP_WRITE) == OP_WRITE;
229        }
230    
231        /**
232         * Gets the set of operations that are ready. The returned set has only
233         * those bits set that are valid for this key's channel.
234         * 
235         * @return the operations for which this key's channel is ready.
236         * @throws CancelledKeyException
237         *             if the key has already been canceled.
238         * @since Android 1.0
239         */
240        public abstract int readyOps();
241    
242        /**
243         * Gets the selector for which this key's channel is registered.
244         * 
245         * @return the related selector.
246         * @since Android 1.0
247         */
248        public abstract Selector selector();
249    }
