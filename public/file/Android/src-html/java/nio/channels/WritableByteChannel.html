017    package java.nio.channels;
018    
019    
020    import java.io.IOException;
021    import java.nio.ByteBuffer;
022    
023    /**
024     * A {@code WritableByteChannel} is a type of {@link Channel} that can write
025     * bytes.
026     * <p>
027     * Write operations are synchronous on a {@code WritableByteChannel}, that is,
028     * if a write is already in progress on the channel then subsequent writes will
029     * block until the first write completes. It is undefined whether non-write
030     * operations will block.
031     * 
032     * @since Android 1.0
033     */
034    public interface WritableByteChannel extends Channel {
035    
036        /**
037         * Writes bytes from the given buffer to the channel.
038         * <p>
039         * The maximum number of bytes that will be written is the
040         * <code>remaining()</code> number of bytes in the buffer when the method
041         * invoked. The bytes will be written from the buffer starting at the
042         * buffer's <code>position</code>.
043         * </p>
044         * <p>
045         * The call may block if other threads are also attempting to write on the
046         * same channel.
047         * </p>
048         * <p>
049         * Upon completion, the buffer's <code>position()</code> is updated to the
050         * end of the bytes that were written. The buffer's <code>limit()</code>
051         * is unmodified.
052         * </p>
053         * 
054         * @param buffer
055         *            the byte buffer containing the bytes to be written.
056         * @return the number of bytes actually written.
057         * @throws NonWritableChannelException
058         *             if the channel was not opened for writing.
059         * @throws ClosedChannelException
060         *             if the channel was already closed.
061         * @throws AsynchronousCloseException
062         *             if another thread closes the channel during the write.
063         * @throws ClosedByInterruptException
064         *             if another thread interrupt the calling thread during the
065         *             write.
066         * @throws IOException
067         *             another IO exception occurs, details are in the message.
068         * @since Android 1.0
069         */
070        public int write(ByteBuffer buffer) throws IOException;
071    }
