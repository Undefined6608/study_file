018    package java.nio.channels;
019    
020    import java.io.IOException;
021    import java.net.DatagramSocket;
022    import java.net.SocketAddress;
023    import java.nio.ByteBuffer;
024    import java.nio.channels.spi.AbstractSelectableChannel;
025    import java.nio.channels.spi.SelectorProvider;
026    
027    import org.apache.harmony.luni.platform.Platform;
028    
029    /**
030     * A {@code DatagramChannel} is a selectable channel that represents a partial
031     * abstraction of a datagram socket. The {@code socket} method of this class can
032     * return the related {@code DatagramSocket} instance, which can handle the
033     * socket.
034     * <p>
035     * A datagram channel is open but not connected when created with the
036     * {@code open()} method. After it is connected, it will keep the connected
037     * status until it is disconnected or closed. The benefit of a connected channel
038     * is the reduced effort of security checks during send and receive. When
039     * invoking {@code read} or {@code write}, a connected channel is required.
040     * </p>
041     * <p>
042     * Datagram channels are thread-safe; only one thread can read or write at the
043     * same time.
044     * </p>
045     * 
046     * @since Android 1.0
047     */
048    public abstract class DatagramChannel extends AbstractSelectableChannel
049            implements ByteChannel, ScatteringByteChannel, GatheringByteChannel {
050    
051        static {
052            Platform.getNetworkSystem().oneTimeInitialization(true);
053        }
054        
055        /**
056         * Constructs a new {@code DatagramChannel}.
057         * 
058         * @param selectorProvider
059         *            an instance of SelectorProvider.
060         * @since Android 1.0
061         */
062        protected DatagramChannel(SelectorProvider selectorProvider) {
063            super(selectorProvider);
064        }
065    
066        /**
067         * Creates an opened and not-connected datagram channel.
068         * <p>
069         * This channel is created by calling the <code>openDatagramChannel</code>
070         * method of the default {@link SelectorProvider} instance.
071         * </p>
072         * 
073         * @return the new channel which is open but not connected.
074         * @throws IOException
075         *             if some I/O error occurs.
076         * @since Android 1.0
077         */
078        public static DatagramChannel open() throws IOException {
079            return SelectorProvider.provider().openDatagramChannel();
080        }
081    
082        /**
083         * Gets the valid operations of this channel. Datagram channels support read
084         * and write operations, so this method returns (
085         * <code>SelectionKey.OP_READ</code> | <code>SelectionKey.OP_WRITE</code> ).
086         * 
087         * @see java.nio.channels.SelectableChannel#validOps()
088         * @return valid operations in bit-set.
089         * @since Android 1.0
090         */
091        public final int validOps() {
092            return (SelectionKey.OP_READ | SelectionKey.OP_WRITE);
093        }
094    
095        /**
096         * Returns the related datagram socket of this channel, which does not
097         * define additional public methods to those defined by
098         * {@link DatagramSocket}.
099         * 
100         * @return the related DatagramSocket instance.
101         * @since Android 1.0
102         */
103        public abstract DatagramSocket socket();
104    
105        /**
106         * Returns whether this channel's socket is connected or not.
107         * 
108         * @return <code>true</code> if this channel's socket is connected;
109         *         <code>false</code> otherwise.
110         * @since Android 1.0
111         */
112        public abstract boolean isConnected();
113    
114        /**
115         * Connects the socket of this channel to a remote address, which is the
116         * only communication peer for getting and sending datagrams after being
117         * connected.
118         * <p>
119         * This method can be called at any time without affecting the read and
120         * write operations being processed at the time the method is called. The
121         * connection status does not change until the channel is disconnected or
122         * closed.
123         * </p>
124         * <p>
125         * This method executes the same security checks as the connect method of
126         * the {@link DatagramSocket} class.
127         * </p>
128         * 
129         * @param address
130         *            the address to be connected to.
131         * @return this channel.
132         * @throws ClosedChannelException
133         *             if the channel is already closed.
134         * @throws AsynchronousCloseException
135         *             if the channel is closed by another thread while this method
136         *             is in operation.
137         * @throws ClosedByInterruptException
138         *             if another thread interrupts the calling thread while the
139         *             operation is in progress. The calling thread will have the
140         *             interrupt state set and the channel will be closed.
141         * @throws SecurityException
142         *             if there is a security manager, and the address is not
143         *             permitted to be accessed.
144         * @throws IOException
145         *             if some other I/O error occurrs.
146         * @since Android 1.0
147         */
148        public abstract DatagramChannel connect(SocketAddress address)
149                throws IOException;
150    
151        /**
152         * Disconnects the socket of this channel, which has been connected before
153         * in order to send and receive datagrams.
154         * <p>
155         * This method can be called at any time without affecting the read and
156         * write operations being underway. It does not have any effect if the
157         * socket is not connected or the channel is closed.
158         * </p>
159         * 
160         * @return this channel.
161         * @throws IOException
162         *             some other I/O error occurs.
163         * @since Android 1.0
164         */
165        public abstract DatagramChannel disconnect() throws IOException;
166    
167        /**
168         * Gets a datagram from this channel.
169         * <p>
170         * This method transfers a datagram from the channel into the target byte
171         * buffer. If this channel is in blocking mode, it waits for the datagram
172         * and returns its address when it is available. If this channel is in
173         * non-blocking mode and no datagram is available, it returns {@code null}
174         * immediately. The transfer starts at the current position of the buffer,
175         * and if there is not enough space remaining in the buffer to store the
176         * datagram then the part of the datagram that does not fit is discarded.
177         * </p>
178         * <p>
179         * This method can be called at any time and it will block if there is
180         * another thread that has started a read operation on the channel.
181         * </p>
182         * <p>
183         * This method executes the same security checks as the receive method of
184         * the {@link DatagramSocket} class.
185         * </p>
186         * 
187         * @param target
188         *            the byte buffer to store the received datagram.
189         * @return the address of the datagram if the transfer is performed, or null
190         *         if the channel is in non-blocking mode and no datagram is
191         *         available.
192         * @throws ClosedChannelException
193         *             if the channel is already closed.
194         * @throws AsynchronousCloseException
195         *             if the channel is closed by another thread while this method
196         *             is in operation.
197         * @throws ClosedByInterruptException
198         *             if another thread interrupts the calling thread while the
199         *             operation is in progress. The calling thread will have the
200         *             interrupt state set and the channel will be closed.
201         * @throws SecurityException
202         *             if there is a security manager, and the address is not
203         *             permitted to be accessed.
204         * @throws IOException
205         *             some other I/O error occurs.
206         * @since Android 1.0
207         */
208        public abstract SocketAddress receive(ByteBuffer target) throws IOException;
209    
210        /**
211         * Sends a datagram through this channel. The datagram consists of the
212         * remaining bytes in {@code source}.
213         * <p>
214         * If this channel is in blocking mode then the datagram is sent as soon as
215         * there is enough space in the underlying output buffer. If this channel is
216         * in non-blocking mode then the datagram is only sent if there is enough
217         * space in the underlying output buffer at that moment. The transfer action
218         * is just like a regular write operation.
219         * </p>
220         * <p>
221         * This method can be called at any time and it will block if another thread
222         * has started a send operation on this channel.
223         * </p>
224         * <p>
225         * This method executes the same security checks as the send method of the
226         * {@link DatagramSocket} class.
227         * </p>
228         * 
229         * @param source
230         *            the byte buffer with the datagram to be sent.
231         * @param address
232         *            the destination address for the datagram.
233         * @return the number of bytes sent. This is the number of bytes remaining
234         *         in {@code source} or zero if the channel is in non-blocking mode
235         *         and there is not enough space for the datagram in the underlying
236         *         output buffer.
237         * @throws ClosedChannelException
238         *             if the channel is already closed.
239         * @throws AsynchronousCloseException
240         *             if the channel is closed by another thread while this method
241         *             is in operation.
242         * @throws ClosedByInterruptException
243         *             if another thread interrupts the calling thread while the
244         *             operation is in progress. The calling thread will have the
245         *             interrupt state set and the channel will be closed.
246         * @throws SecurityException
247         *             if there is a security manager, and the address is not
248         *             permitted to access.
249         * @throws IOException
250         *             some other I/O error occurs.
251         * @since Android 1.0
252         */
253        public abstract int send(ByteBuffer source, SocketAddress address)
254                throws IOException;
255    
256        /**
257         * Reads a datagram from this channel into the byte buffer.
258         * <p>
259         * The precondition for calling this method is that the channel is connected
260         * and the incoming datagram is from the connected address. If the buffer is
261         * not big enough to store the datagram, the part of the datagram that does
262         * not fit in the buffer is discarded. Otherwise, this method has the same
263         * behavior as the {@code read} method in the {@link ReadableByteChannel}
264         * interface.
265         * </p>
266         * 
267         * @see java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)
268         * @param target
269         *            the byte buffer to store the received datagram.
270         * @return a non-negative number as the number of bytes read, or -1 as the
271         *         read operation reaches the end of stream.
272         * @throws NotYetConnectedException
273         *             if the channel is not connected yet.
274         * @throws ClosedChannelException
275         *             if the channel is already closed.
276         * @throws AsynchronousCloseException
277         *             if the channel is closed by another thread while this method
278         *             is in operation.
279         * @throws ClosedByInterruptException
280         *             if another thread interrupts the calling thread while the
281         *             operation is in progress. The calling thread will have the
282         *             interrupt state set and the channel will be closed.
283         * @throws IOException
284         *             some other I/O error occurs.
285         * @since Android 1.0
286         */
287        public abstract int read(ByteBuffer target) throws IOException;
288    
289        /**
290         * Reads a datagram from this channel into an array of byte buffers.
291         * <p>
292         * The precondition for calling this method is that the channel is connected
293         * and the incoming datagram is from the connected address. If the buffers
294         * do not have enough remaining space to store the datagram, the part of the
295         * datagram that does not fit in the buffers is discarded. Otherwise, this
296         * method has the same behavior as the {@code read} method in the
297         * {@link ScatteringByteChannel} interface.
298         * </p>
299         * 
300         * @see java.nio.channels.ScatteringByteChannel#read(java.nio.ByteBuffer[],
301         *      int, int)
302         * @param targets
303         *            the byte buffers to store the received datagram.
304         * @param offset
305         *            a non-negative offset in the array of buffers, pointing to the
306         *            starting buffer to store the bytes transferred, must not be
307         *            bigger than {@code targets.length}.
308         * @param length
309         *            a non-negative length to indicate the maximum number of
310         *            buffers to be filled, must not be bigger than
311         *            {@code targets.length - offset}.
312         * @return a non-negative number as the number of bytes read, or -1 if the
313         *         read operation reaches the end of stream.
314         * @throws NotYetConnectedException
315         *             if the channel is not connected yet.
316         * @throws ClosedChannelException
317         *             if the channel is already closed.
318         * @throws AsynchronousCloseException
319         *             if the channel is closed by another thread while this method
320         *             is in operation.
321         * @throws ClosedByInterruptException
322         *             if another thread interrupts the calling thread while the
323         *             operation is in progress. The calling thread will have the
324         *             interrupt state set and the channel will be closed.
325         * @throws IOException
326         *             some other I/O error occurs.
327         * @since Android 1.0
328         */
329        public abstract long read(ByteBuffer[] targets, int offset, int length)
330                throws IOException;
331    
332        /**
333         * Reads a datagram from this channel into an array of byte buffers.
334         * <p>
335         * The precondition for calling this method is that the channel is connected
336         * and the incoming datagram is from the connected address. If the buffers
337         * do not have enough remaining space to store the datagram, the part of the
338         * datagram that does not fit in the buffers is discarded. Otherwise, this
339         * method has the same behavior as the {@code read} method in the
340         * {@link ScatteringByteChannel} interface.
341         * </p>
342         * 
343         * @see java.nio.channels.ScatteringByteChannel#read(java.nio.ByteBuffer[])
344         * @param targets
345         *            the byte buffers to store the received datagram.
346         * @return a non-negative number as the number of bytes read, or -1 if the
347         *         read operation reaches the end of stream.
348         * @throws NotYetConnectedException
349         *             if the channel is not connected yet.
350         * @throws ClosedChannelException
351         *             if the channel is already closed.
352         * @throws AsynchronousCloseException
353         *             if the channel is closed by another thread while this method
354         *             is in operation.
355         * @throws ClosedByInterruptException
356         *             if another thread interrupts the calling thread while the
357         *             operation is in progress. The calling thread will have the
358         *             interrupt state set and the channel will be closed.
359         * @throws IOException
360         *             some other I/O error occurs.
361         * @since Android 1.0
362         */
363        public synchronized final long read(ByteBuffer[] targets)
364                throws IOException {
365            return read(targets, 0, targets.length);
366        }
367    
368        /**
369         * Writes a datagram from the byte buffer to this channel.
370         * <p>
371         * The precondition of calling this method is that the channel is connected
372         * and the datagram is sent to the connected address. Otherwise, this method
373         * has the same behavior as the {@code write} method in the
374         * {@link WritableByteChannel} interface.
375         * </p>
376         * 
377         * @see java.nio.channels.WritableByteChannel#write(java.nio.ByteBuffer)
378         * @param source
379         *            the byte buffer as the source of the datagram.
380         * @return a non-negative number of bytes written.
381         * @throws NotYetConnectedException
382         *             if the channel is not connected yet.
383         * @throws ClosedChannelException
384         *             if the channel is already closed.
385         * @throws AsynchronousCloseException
386         *             if the channel is closed by another thread while this method
387         *             is in operation.
388         * @throws ClosedByInterruptException
389         *             if another thread interrupts the calling thread while the
390         *             operation is in progress. The calling thread will have the
391         *             interrupt state set and the channel will be closed.
392         * @throws IOException
393         *             some other I/O error occurs.
394         * @since Android 1.0
395         */
396        public abstract int write(ByteBuffer source) throws IOException;
397    
398        /**
399         * Writes a datagram from the byte buffers to this channel.
400         * <p>
401         * The precondition of calling this method is that the channel is connected
402         * and the datagram is sent to the connected address. Otherwise, this method
403         * has the same behavior as the {@code write} method in the
404         * {@link GatheringByteChannel} interface.
405         * 
406         * @see java.nio.channels.GatheringByteChannel#write(java.nio.ByteBuffer[],
407         *      int, int)
408         * @param sources
409         *            the byte buffers as the source of the datagram.
410         * @param offset
411         *            a non-negative offset in the array of buffers, pointing to the
412         *            starting buffer to be retrieved, must be no larger than
413         *            {@code sources.length}.
414         * @param length
415         *            a non-negative length to indicate the maximum number of
416         *            buffers to be submitted, must be no bigger than
417         *            {@code sources.length - offset}.
418         * @return the number of bytes written. If this method is called, it returns
419         *         the number of bytes that where remaining in the byte buffers. If
420         *         the channel is in non-blocking mode and there was not enough
421         *         space for the datagram in the buffer, it may return zero.
422         * @throws NotYetConnectedException
423         *             if the channel is not connected yet.
424         * @throws ClosedChannelException
425         *             if the channel is already closed.
426         * @throws AsynchronousCloseException
427         *             if the channel is closed by another thread while this method
428         *             is in operation.
429         * @throws ClosedByInterruptException
430         *             if another thread interrupts the calling thread while the
431         *             operation is in progress. The calling thread will have the
432         *             interrupt state set and the channel will be closed.
433         * @throws IOException
434         *             some other I/O error occurs.
435         * @since Android 1.0
436         */
437        public abstract long write(ByteBuffer[] sources, int offset, int length)
438                throws IOException;
439    
440        /**
441         * Writes a datagram from the byte buffers to this channel.

443         * The precondition of calling this method is that the channel is connected

451         * @return the number of bytes written. If this method is called, it returns
452         *         the number of bytes that where remaining in the byte buffer. If
