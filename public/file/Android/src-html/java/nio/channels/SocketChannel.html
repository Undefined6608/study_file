018    package java.nio.channels;
019    
020    import java.io.IOException;
021    import java.net.Socket;
022    import java.net.SocketAddress;
023    import java.nio.ByteBuffer;
024    import java.nio.channels.spi.AbstractSelectableChannel;
025    import java.nio.channels.spi.SelectorProvider;
026    
027    import org.apache.harmony.luni.platform.Platform;
028    
029    /**
030     * A {@code SocketChannel} is a selectable channel that provides a partial
031     * abstraction of stream connecting socket. {@code socket()} returns the related
032     * {@link Socket} instance which can handle the socket.
033     * <p>
034     * A socket channel is open but not connected when created by {@code open()}.
035     * After connecting it by calling {@code connect(SocketAddress)}, it will remain
036     * connected until it gets closed. If the connection is non-blocking then
037     * {@code connect(SocketAddress)} is used to initiate the connection, followed
038     * by a call of {@code finishConnect()} to perform the final steps of
039     * connecting. {@code isConnectionPending()} indicates if the connection is
040     * blocked or not; {@code isConnected()} indicates if the socket is finally
041     * connected or not.
042     * </p>
043     * <p>
044     * The input and output sides of a channel can be shut down independently and
045     * asynchronously without closing the channel. The {@code shutdownInput} method
046     * is used for the input side of a channel and subsequent read operations return
047     * -1, which means end of stream. If another thread is blocked in a read
048     * operation when the shutdown occurs, the read will end without effect and
049     * return end of stream. The {@code shutdownOutput} method is used for the
050     * output side of the channel; subsequent write operations throw a
051     * {@link ClosedChannelException}. If the output is shut down and another thread
052     * is blocked in a write operation, an {@link AsynchronousCloseException} will
053     * be thrown to the pending thread.
054     * </p>
055     * <p>
056     * Socket channels are thread-safe, no more than one thread can read or write at
057     * any given time. The {@code connect(SocketAddress)} and {@code
058     * finishConnect()} methods are synchronized against each other; when they are
059     * processing, calls to {@code read} and {@code write} will block.
060     * </p>
061     * 
062     * @since Android 1.0
063     */
064    public abstract class SocketChannel extends AbstractSelectableChannel implements
065            ByteChannel, ScatteringByteChannel, GatheringByteChannel {
066    
067        static {
068            Platform.getNetworkSystem().oneTimeInitialization(true);
069        }
070        
071        /**
072         * Constructs a new {@code SocketChannel}.
073         * 
074         * @param selectorProvider
075         *            an instance of SelectorProvider.
076         * @since Android 1.0
077         */
078        protected SocketChannel(SelectorProvider selectorProvider) {
079            super(selectorProvider);
080        }
081    
082        /**
083         * Creates an open and unconnected socket channel.
084         * <p>
085         * This channel is created by calling {@code openSocketChannel()} of the
086         * default {@link SelectorProvider} instance.
087         * </p>
088         * 
089         * @return the new channel which is open but unconnected.
090         * @throws IOException
091         *             if an I/O error occurs.
092         * @since Android 1.0
093         */
094        public static SocketChannel open() throws IOException {
095            return SelectorProvider.provider().openSocketChannel();
096        }
097    
098        /**
099         * Creates a socket channel and connects it to a socket address.
100         * <p>
101         * This method performs a call to {@code open()} followed by a call to
102         * {@code connect(SocketAdress)}.
103         * </p>
104         * 
105         * @param address
106         *            the socket address to be connected to.
107         * @return the new connected channel.
108         * @throws AsynchronousCloseException
109         *             if this channel is closed by another thread while this method
110         *             is executing.
111         * @throws ClosedByInterruptException
112         *             if another thread interrupts the calling thread while this
113         *             operation is executing. The calling thread will have the
114         *             interrupt state set and the channel will be closed.
115         * @throws SecurityException
116         *             if there is a security manager and it denies the access of
117         *             {@code address}.
118         * @throws UnresolvedAddressException
119         *             if the address is not resolved.
120         * @throws UnsupportedAddressTypeException
121         *             if the address type is not supported.
122         * @throws IOException
123         *             if an I/O error occurs.
124         * @since Android 1.0
125         */
126        public static SocketChannel open(SocketAddress address) throws IOException {
127            SocketChannel socketChannel = open();
128            if (null != socketChannel) {
129                socketChannel.connect(address);
130            }
131            return socketChannel;
132        }
133    
134        /**
135         * Gets the valid operations of this channel. Socket channels support
136         * connect, read and write operation, so this method returns
137         * {@code SelectionKey.OP_CONNECT | SelectionKey.OP_READ | SelectionKey.OP_WRITE}.
138         * 
139         * @return the operations supported by this channel.
140         * @see java.nio.channels.SelectableChannel#validOps()
141         * @since Android 1.0
142         */
143        public final int validOps() {
144            return (SelectionKey.OP_CONNECT | SelectionKey.OP_READ | SelectionKey.OP_WRITE);
145        }
146    
147        /**
148         * Returns the socket assigned to this channel, which does not declare any public
149         * methods that are not declared in {@code Socket}.
150         * 
151         * @return the socket assigned to this channel.
152         * @since Android 1.0
153         */
154        public abstract Socket socket();
155    
156        /**
157         * Indicates whether this channel's socket is connected.
158         * 
159         * @return {@code true} if this channel's socket is connected, {@code false}
160         *         otherwise.
161         * @since Android 1.0
162         */
163        public abstract boolean isConnected();
164    
165        /**
166         * Indicates whether this channel's socket is still trying to connect.
167         * 
168         * @return {@code true} if the connection is initiated but not finished;
169         *         {@code false} otherwise.
170         * @since Android 1.0
171         */
172        public abstract boolean isConnectionPending();
173    
174        /**
175         * Connects this channel's socket with a remote address.
176         * <p>
177         * If this channel is blocking, this method will suspend until connecting is
178         * finished or an I/O exception occurrs. If the channel is non-blocking,
179         * this method will return {@code true} if the connection is finished at
180         * once or return {@code false} when the connection must be finished later
181         * by calling {@code finishConnect()}.
182         * </p>
183         * <p>
184         * This method can be called at any moment and can block other read and
185         * write operations while connecting. It executes the same security checks
186         * as the connect method of the {@code Socket} class.
187         * </p>
188         * 
189         * @param address
190         *            the address to connect with.
191         * @return {@code true} if the connection is finished, {@code false}
192         *         otherwise.
193         * @throws AlreadyConnectedException
194         *             if the channel is already connected.
195         * @throws ConnectionPendingException
196         *             a non-blocking connecting operation is already executing on
197         *             this channel.
198         * @throws ClosedChannelException
199         *             if this channel is closed.
200         * @throws AsynchronousCloseException
201         *             if this channel is closed by another thread while this method
202         *             is executing.
203         * @throws ClosedByInterruptException
204         *             if another thread interrupts the calling thread while this
205         *             operation is in progress. The calling thread will have the
206         *             interrupt state set and this channel will be closed.
207         * @throws UnresolvedAddressException
208         *             if the address is not resolved.
209         * @throws UnsupportedAddressTypeException
210         *             if the address type is not supported.
211         * @throws SecurityException
212         *             if there is a security manager and it denies the access of
213         *             {@code address}.
214         * @throws IOException
215         *             if an I/O error occurs.
216         * @since Android 1.0
217         */
218        public abstract boolean connect(SocketAddress address) throws IOException;
219    
220        /**
221         * Completes the connection process initiated by a call of {@code
222         * connect(SocketAddress)}.
223         * <p>
224         * This method returns {@code true} if the connection is finished already
225         * and returns {@code false} if the channel is non-blocking and the
226         * connection is not finished yet.
227         * </p>
228         * <p>
229         * If this channel is in blocking mode, this method will suspend and return
230         * {@code true} when the connection is finished. It closes this channel and
231         * throws an exception if the connection fails.
232         * </p>
233         * <p>
234         * This method can be called at any moment and it can block other {@code
235         * read} and {@code write} operations while connecting.
236         * </p>
237         * 
238         * @return {@code true} if the connection is successfully finished, {@code
239         *         false} otherwise.
240         * @throws NoConnectionPendingException
241         *             if the channel is not connected and the connection process
242         *             has not been initiated.
243         * @throws ClosedChannelException
244         *             if this channel is closed.
245         * @throws AsynchronousCloseException
246         *             if this channel is closed by another thread while this method
247         *             is executing.
248         * @throws ClosedByInterruptException
249         *             if another thread interrupts the calling thread while this
250         *             operation is in progress. The calling thread has the
251         *             interrupt state set, and this channel is closed.
252         * @throws IOException
253         *             if an I/O error occurs.
254         * @since Android 1.0
255         */
256        public abstract boolean finishConnect() throws IOException;
257    
258        /**
259         * Reads bytes from this socket channel into the given buffer.
260         * <p>
261         * The maximum number of bytes that will be read is the remaining number of
262         * bytes in the buffer when the method is invoked. The bytes will be copied
263         * into the buffer starting at the buffer's current position.
264         * </p>
265         * <p>
266         * The call may block if other threads are also attempting to read from this
267         * channel.
268         * </p>
269         * <p>
270         * Upon completion, the buffer's position is set to the end of the bytes
271         * that have been read. The buffer's limit is not changed.
272         * </p>
273         * 
274         * @param target
275         *            the byte buffer to receive the bytes.
276         * @return the number of bytes actually read.
277         * @throws AsynchronousCloseException
278         *             if another thread closes the channel during the read.
279         * @throws NotYetConnectedException
280         *             if this channel is not yet connected.
281         * @throws ClosedByInterruptException
282         *             if another thread interrupts the calling thread while this
283         *             operation is in progress. The interrupt state of the calling
284         *             thread is set and the channel is closed.
285         * @throws ClosedChannelException
286         *             if this channel is closed.
287         * @throws IOException
288         *             if another I/O error occurs.
289         * @since Android 1.0
290         */
291        public abstract int read(ByteBuffer target) throws IOException;
292    
293        /**
294         * Reads bytes from this socket channel and stores them in a subset of the
295         * specified array of buffers. The subset is defined by {@code offset} and
296         * {@code length}, indicating the first buffer and the number of buffers to
297         * use. This method attempts to read as many bytes as can be stored in the
298         * buffer subset from this channel and returns the number of bytes actually
299         * read.
300         * <p>
301         * If a read operation is in progress, subsequent threads will block until
302         * the read is completed and will then contend for the ability to read.
303         * </p>
304         * 
305         * @param targets
306         *            the array of byte buffers into which the bytes will be copied.
307         * @param offset
308         *            the index of the first buffer to store bytes in.
309         * @param length
310         *            the maximum number of buffers to store bytes in.
311         * @return the number of bytes actually read.
312         * @throws AsynchronousCloseException
313         *             if this channel is closed by another thread during this read
314         *             operation.
315         * @throws ClosedByInterruptException
316         *             if another thread interrupts the calling thread while this
317         *             operation is in progress. The interrupt state of the calling
318         *             thread is set and the channel is closed.
319         * @throws ClosedChannelException
320         *             if this channel is closed.
321         * @throws IndexOutOfBoundsException
322         *             if {@code offset < 0} or {@code length < 0}, or if {@code
323         *             offset + length} is greater than the size of {@code targets}.
324         * @throws IOException
325         *             if another I/O error occurs.
326         * @throws NotYetConnectedException
327         *             if this channel is not yet connected.
328         * @since Android 1.0
329         */
330        public abstract long read(ByteBuffer[] targets, int offset, int length)
331                throws IOException;
332    
333        /**
334         * Reads bytes from this socket channel and stores them in the specified
335         * array of buffers. This method attempts to read as many bytes as can be
336         * stored in the buffer array from this channel and returns the number of
337         * bytes actually read.
338         * <p>
339         * If a read operation is in progress, subsequent threads will block until
340         * the read is completed and will then contend for the ability to read.
341         * </p>
342         * <p>
343         * Calling this method is equivalent to calling {@code read(targets, 0,
344         * targets.length);}
345         * </p>
346         * 
347         * @param targets
348         *            the array of byte buffers into which the bytes will be copied.
349         * @return the number of bytes actually read.
350         * @throws AsynchronousCloseException
351         *             if this channel is closed by another thread during this read
352         *             operation.
353         * @throws ClosedByInterruptException
354         *             if another thread interrupts the calling thread while this
355         *             operation is in progress. The interrupt state of the calling
356         *             thread is set and the channel is closed.
357         * @throws ClosedChannelException
358         *             if this channel is closed.
359         * @throws IOException
360         *             if another I/O error occurs.
361         * @throws NotYetConnectedException
362         *             if this channel is not yet connected.
363         * @since Android 1.0
364         */
365        public synchronized final long read(ByteBuffer[] targets)
366                throws IOException {
367            return read(targets, 0, targets.length);
368        }
369    
370        /**
371         * Writes bytes from the given byte buffer to this socket channel. The
372         * maximum number of bytes that are written is the remaining number of bytes
373         * in the buffer when this method is invoked. The bytes are taken from the
374         * buffer starting at the buffer's position.
375         * <p>
376         * The call may block if other threads are also attempting to write to the
377         * same channel.
378         * </p>
379         * <p>
380         * Upon completion, the buffer's position is updated to the end of the bytes
381         * that have been written. The buffer's limit is not changed.
382         * </p>
383         * 
384         * @param source
385         *            the byte buffer containing the bytes to be written.
386         * @return the number of bytes actually written.
387         * @throws AsynchronousCloseException
388         *             if another thread closes the channel during the write.
389         * @throws ClosedByInterruptException
390         *             if another thread interrupts the calling thread while this
391         *             operation is in progress. The interrupt state of the calling
392         *             thread is set and the channel is closed.
393         * @throws ClosedChannelException
394         *             if the channel was already closed.
395         * @throws IOException
396         *             if another I/O error occurs.
397         * @throws NotYetConnectedException
398         *             if this channel is not connected yet.
399         * @since Android 1.0
400         */
401        public abstract int write(ByteBuffer source) throws IOException;
402    
403        /**
404         * Writes bytes from a subset of the specified array of buffers into this
405         * socket channel. The subset is defined by {@code offset} and {@code
406         * length}, indicating the first buffer and the number of buffers to use.
407         * <p>
408         * If a write operation is in progress, subsequent threads will block until
409         * the write is completed and then contend for the ability to write.
410         * </p>
411         * 
412         * @param sources
413         *            the array of byte buffers that is the source for bytes written
414         *            to this channel.
415         * @param offset
416         *            the index of the first buffer in {@code buffers }to get bytes
417         *            from.
418         * @param length
419         *            the number of buffers to get bytes from.
420         * @return the number of bytes actually written to this channel.
421         * @throws AsynchronousCloseException
422         *             if this channel is closed by another thread during this write
423         *             operation.
424         * @throws ClosedByInterruptException
425         *             if another thread interrupts the calling thread while this
426         *             operation is in progress. The interrupt state of the calling
427         *             thread is set and the channel is closed.
428         * @throws ClosedChannelException
429         *             if this channel is closed.
430         * @throws IndexOutOfBoundsException
431         *             if {@code offset < 0} or {@code length < 0}, or if {@code
432         *             offset + length} is greater than the size of {@code sources}.
433         * @throws IOException
434         *             if another I/O error occurs.
435         * @throws NotYetConnectedException
436         *             if this channel is not yet connected.
437         * @since Android 1.0
438         */
439        public abstract long write(ByteBuffer[] sources, int offset, int length)
440                throws IOException;
441    
442        /**
443         * Writes bytes from all the given byte buffers to this socket channel.
444         * <p>
445         * Calling this method is equivalent to calling {@code write(sources, 0,
446         * sources.length);}
447         * </p>
448         * 
449         * @param sources
450         *            the buffers containing bytes to write.
451         * @return the number of bytes actually written.
452         * @throws AsynchronousCloseException
453         *             if this channel is closed by another thread during this write
454         *             operation.
455         * @throws ClosedByInterruptException
456         *             if another thread interrupts the calling thread while this
457         *             operation is in progress. The interrupt state of the calling
458         *             thread is set and the channel is closed.
459         * @throws ClosedChannelException
460         *             if this channel is closed.
461         * @throws IOException
462         *             if another I/O error occurs.
463         * @throws NotYetConnectedException
464         *             if this channel is not yet connected.
465         * @since Android 1.0
466         */
467        public synchronized final long write(ByteBuffer[] sources)
468                throws IOException {
469            return write(sources, 0, sources.length);
470        }
