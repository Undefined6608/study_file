018    package java.nio.channels.spi;
019    
020    import java.io.IOException;
021    import java.nio.channels.CancelledKeyException;
022    import java.nio.channels.ClosedChannelException;
023    import java.nio.channels.IllegalBlockingModeException;
024    import java.nio.channels.IllegalSelectorException;
025    import java.nio.channels.SelectableChannel;
026    import java.nio.channels.SelectionKey;
027    import java.nio.channels.Selector;
028    import java.util.ArrayList;
029    import java.util.List;
030    
031    /**
032     * {@code AbstractSelectableChannel} is the base implementation class for
033     * selectable channels. It declares methods for registering, unregistering and
034     * closing selectable channels. It is thread-safe.
035     * 
036     * @since Android 1.0
037     */
038    public abstract class AbstractSelectableChannel extends SelectableChannel {
039    
040        private final SelectorProvider provider;
041    
042        /*
043         * The collection of key.
044         */
045        private List<SelectionKey> keyList = new ArrayList<SelectionKey>();
046    
047        private class BlockingLock {
048        }
049    
050        private final Object blockingLock = new BlockingLock();
051    
052        boolean isBlocking = true;
053    
054        /**
055         * Constructs a new {@code AbstractSelectableChannel}.
056         * 
057         * @param selectorProvider
058         *            the selector provider that creates this channel.
059         * @since Android 1.0
060         */
061        protected AbstractSelectableChannel(SelectorProvider selectorProvider) {
062            super();
063            provider = selectorProvider;
064        }
065    
066        /**
067         * Returns the selector provider that has created this channel.
068         * 
069         * @see java.nio.channels.SelectableChannel#provider()
070         * @return this channel's selector provider.
071         * @since Android 1.0
072         */
073        public final SelectorProvider provider() {
074            return provider;
075        }
076    
077        /**
078         * Indicates whether this channel is registered with one or more selectors.
079         * 
080         * @return {@code true} if this channel is registered with a selector,
081         *         {@code false} otherwise.
082         * @since Android 1.0
083         */
084        synchronized public final boolean isRegistered() {
085            return !keyList.isEmpty();
086        }
087    
088        /**
089         * Gets this channel's selection key for the specified selector.
090         * 
091         * @param selector
092         *            the selector with which this channel has been registered.
093         * @return the selection key for the channel or {@code null} if this channel
094         *         has not been registered with {@code selector}.
095         * @since Android 1.0
096         */
097        synchronized public final SelectionKey keyFor(Selector selector) {
098            for (int i = 0; i < keyList.size(); i++) {
099                SelectionKey key = keyList.get(i);
100                if (null != key && key.selector() == selector) {
101                    return key;
102                }
103            }
104            return null;
105        }
106    
107        /**
108         * Registers this channel with the specified selector for the specified
109         * interest set. If the channel is already registered with the selector, the
110         * {@link SelectionKey interest set} is updated to {@code interestSet} and
111         * the corresponding selection key is returned. If the channel is not yet
112         * registered, this method calls the {@code register} method of
113         * {@code selector} and adds the selection key to this channel's key set.
114         * 
115         * @param selector
116         *            the selector with which to register this channel.
117         * @param interestSet
118         *            this channel's {@link SelectionKey interest set}.
119         * @param attachment
120         *            the object to attach, can be {@code null}.
121         * @return the selection key for this registration.
122         * @throws CancelledKeyException
123         *             if this channel is registered but its key has been canceled.
124         * @throws ClosedChannelException
125         *             if this channel is closed.
126         * @throws IllegalArgumentException
127         *             if {@code interestSet} is not supported by this channel.
128         * @throws IllegalBlockingModeException
129         *             if this channel is in blocking mode.
130         * @throws IllegalSelectorException
131         *             if this channel does not have the same provider as the given
132         *             selector.
133         * @since Android 1.0
134         */
135        public final SelectionKey register(Selector selector, int interestSet,
136                Object attachment) throws ClosedChannelException {
137            if (!isOpen()) {
138                throw new ClosedChannelException();
139            }
140            if (!((interestSet & ~validOps()) == 0)) {
141                throw new IllegalArgumentException();
142            }
143    
144            synchronized (blockingLock) {
145                if (isBlocking) {
146                    throw new IllegalBlockingModeException();
147                }
148                if (!selector.isOpen()) {
149                    if (0 == interestSet) {
150                        // throw ISE exactly to keep consistency
151                        throw new IllegalSelectorException();
152                    }
153                    // throw NPE exactly to keep consistency
154                    throw new NullPointerException();
155                }
156                if (0 == interestSet) {
157                    // throw ISE exactly to keep consistency
158                    throw new IllegalSelectorException();
159                }
160                SelectionKey key = keyFor(selector);
161                if (null == key) {
162                    key = ((AbstractSelector) selector).register(this, interestSet,
163                            attachment);
164                    keyList.add(key);
165                } else {
166                    if (!key.isValid()) {
167                        throw new CancelledKeyException();
168                    }
169                    key.interestOps(interestSet);
170                    key.attach(attachment);
171                }
172                return key;
173            }
174        }
175    
176        /**
177         * Implements the channel closing behavior. Calls
178         * {@code implCloseSelectableChannel()} first, then loops through the list
179         * of selection keys and cancels them, which unregisters this channel from
180         * all selectors it is registered with.
181         * 
182         * @throws IOException
183         *             if a problem occurs while closing the channel.
184         * @since Android 1.0
185         */
186        synchronized protected final void implCloseChannel() throws IOException {
187            implCloseSelectableChannel();
188            for (int i = 0; i < keyList.size(); i++) {
189                SelectionKey key = keyList.get(i);
190                if (null != key) {
191                    key.cancel();
192                }
193            }
194        }
195    
196        /**
197         * Implements the closing function of the SelectableChannel. This method is
198         * called from {@code implCloseChannel()}.
199         * 
200         * @throws IOException
201         *             if an I/O exception occurs.
202         * @since Android 1.0
203         */
204        protected abstract void implCloseSelectableChannel() throws IOException;
205    
206        /**
207         * Indicates whether this channel is in blocking mode.
208         * 
209         * @return {@code true} if this channel is blocking, {@code false}
210         *         otherwise.
211         * @since Android 1.0
212         */
213        public final boolean isBlocking() {
214            synchronized (blockingLock) {
215                return isBlocking;
216            }
217        }
218    
219        /**
220         * Gets the object used for the synchronization of {@code register} and
221         * {@code configureBlocking}.
222         * 
223         * @return the synchronization object.
224         * @since Android 1.0
225         */
226        public final Object blockingLock() {
227            return blockingLock;
228        }
229    
230        /**
231         * Sets the blocking mode of this channel. A call to this method blocks if
232         * other calls to this method or to {@code register} are executing. The
233         * actual setting of the mode is done by calling
234         * {@code implConfigureBlocking(boolean)}.
235         * 
236         * @param blockingMode
237         *            {@code true} for setting this channel's mode to blocking,
238         *            {@code false} to set it to non-blocking.
239         * @return this channel.
240         * @throws ClosedChannelException
241         *             if this channel is closed.
242         * @throws IllegalBlockingModeException
243         *             if {@code block} is {@code true} and this channel has been
244         *             registered with at least one selector.
245         * @throws IOException
246         *             if an I/O error occurs.
247         * @since Android 1.0
248         */
249        public final SelectableChannel configureBlocking(boolean blockingMode)
250                throws IOException {
251            if (isOpen()) {
252                synchronized (blockingLock) {
253                    if (isBlocking == blockingMode) {
254                        return this;
255                    }
256                    if (blockingMode && isRegistered()) {
257                        throw new IllegalBlockingModeException();
258                    }
259                    implConfigureBlocking(blockingMode);
260                    isBlocking = blockingMode;
261                }
262                return this;
263            }
264            throw new ClosedChannelException();
265    
266        }
267    
268        /**
269         * Implements the setting of the blocking mode.
270         * 
271         * @param blockingMode
272         *            {@code true} for setting this channel's mode to blocking,
273         *            {@code false} to set it to non-blocking.
274         * @throws IOException
275         *             if an I/O error occurs.
276         * @since Android 1.0
277         */
278        protected abstract void implConfigureBlocking(boolean blockingMode)
279                throws IOException;
280    
281        /*
282         * package private for deregister method in AbstractSelector.
283         */
284        synchronized void deregister(SelectionKey k) {
285            if (null != keyList) {
286                keyList.remove(k);
287            }
288        }
289
