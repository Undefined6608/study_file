019

021    import java.io.IOException;
022    import java.io.InputStreamReader;
023    import java.net.URL;
024    import java.nio.channels.Channel;
025    import java.nio.channels.DatagramChannel;
026    import java.nio.channels.Pipe;
027    import java.nio.channels.ServerSocketChannel;
028    import java.nio.channels.SocketChannel;
029    import java.security.AccessController;
030    import java.security.PrivilegedAction;
031    import java.util.Enumeration;
032    
033    import org.apache.harmony.luni.platform.Platform;
034    import org.apache.harmony.nio.internal.SelectorProviderImpl;
035    
036    /**
037     * {@code SelectorProvider} is an abstract base class that declares methods for
038     * providing instances of {@link DatagramChannel}, {@link Pipe},
039     * {@link java.nio.channels.Selector} , {@link ServerSocketChannel}, and
040     * {@link SocketChannel}. All the methods of this class are thread-safe.
041     * <p>
042     * A provider instance can be retrieved through a system property or the
043     * configuration file in a jar file; if no provide is available that way then
044     * the system default provider is returned.
045     * </p>
046     * 
047     * @since Android 1.0
048     */
049    public abstract class SelectorProvider extends Object {
050    
051        private static final String SYMBOL_COMMENT = "#"; //$NON-NLS-1$
052    
053        private static final String PROVIDER_IN_SYSTEM_PROPERTY = "java.nio.channels.spi.SelectorProvider"; //$NON-NLS-1$
054    
055        private static final String PROVIDER_IN_JAR_RESOURCE = "META-INF/services/java.nio.channels.spi.SelectorProvider"; //$NON-NLS-1$
056    
057        private static SelectorProvider provider = null;
058        
059        private static Channel inheritedChannel; 
060    
061        /**
062         * Constructs a new {@code SelectorProvider}.
063         * 
064         * @throws SecurityException
065         *             if there is a security manager installed that does not permit
066         *             the runtime permission labeled "selectorProvider".
067         * @since Android 1.0
068         */
069        protected SelectorProvider() {
070            super();
071            if (null != System.getSecurityManager()) {
072                System.getSecurityManager().checkPermission(
073                        new RuntimePermission("selectorProvider")); //$NON-NLS-1$
074            }
075        }
076    
077        /**
078         * Gets a provider instance by executing the following steps when called for
079         * the first time:
080         * <ul>
081         * <li> if the system property "java.nio.channels.spi.SelectorProvider" is
082         * set, the value of this property is the class name of the provider
083         * returned; </li>
084         * <li>if there is a provider-configuration file named
085         * "java.nio.channels.spi.SelectorProvider" in META-INF/services of a jar
086         * file valid in the system class loader, the first class name is the
087         * provider's class name; </li>
088         * <li> otherwise, a system default provider will be returned.</li>
089         * </ul>
090         * 
091         * @return the provider.
092         * @since Android 1.0
093         */
094        synchronized public static SelectorProvider provider() {
095            if (null == provider) {
096                provider = loadProviderByProperty();
097                if (null == provider) {
098                    provider = loadProviderByJar();
099                }
100                if (null == provider) {
101                    provider = AccessController
102                            .doPrivileged(new PrivilegedAction<SelectorProvider>() {
103                                public SelectorProvider run() {
104                                    return new SelectorProviderImpl();
105                                }
106                            });
107                }
108            }
109            return provider;
110        }
111    
112        /*
113         * load the provider in the jar file of class path.
114         */
115        static SelectorProvider loadProviderByJar() {
116            Enumeration<URL> enumeration = null;
117    
118            ClassLoader classLoader = AccessController.doPrivileged(
119                    new PrivilegedAction<ClassLoader>() {
120                        public ClassLoader run() {
121                            return ClassLoader.getSystemClassLoader();
122                        }
123                    });
124            try {
125                enumeration = classLoader.getResources(PROVIDER_IN_JAR_RESOURCE);
126            } catch (IOException e) {
127                throw new Error(e);
128            }
129            if (null == enumeration) {
130                return null;
131            }
132            // for every jar, read until we find the provider name.
133            while (enumeration.hasMoreElements()) {
134                BufferedReader br = null;
135                String className = null;
136                try {
137                    // BEGIN android-modified
138                    br = new BufferedReader(
139                            new InputStreamReader(
140                                    (enumeration.nextElement()).openStream()),
141                            8192);
142                    // END android-modified
143                } catch (Exception e) {
144                    continue;
145                }
146                try {
147                    // only the first class is loaded ,as spec says, not the same as
148                    // we do before.
149                    while ((className = br.readLine()) != null) {
150                        className = className.trim();
151                        int siteComment = className.indexOf(SYMBOL_COMMENT);
152                        className = (-1 == siteComment) ? className : className
153                                .substring(0, siteComment);
154                        if (0 < className.length()) {
155                            return (SelectorProvider) classLoader.loadClass(
156                                    className).newInstance();                  
157                        }
158                    }
159                } catch (Exception e) {
160                    throw new Error(e);
161                // BEGIN android-added
162                // copied from a newer version of harmony
163                } finally {
164                    try {
165                        br.close();
166                    } catch (IOException ioe) {
167                        // Ignore
168                    }
169                // END android-added
170                }
171            }
172            return null;
173        }
174    
175        /*
176         * load by system property.
177         */
178        static SelectorProvider loadProviderByProperty() {
179            return AccessController.doPrivileged(
180                    new PrivilegedAction<SelectorProvider>() {
181                        public SelectorProvider run() {
182                            try {
183                                final String className =
184                                    System.getProperty(PROVIDER_IN_SYSTEM_PROPERTY);
185                                if (null != className) {
186                                    Class<?> spClass = ClassLoader
187                                            .getSystemClassLoader().loadClass(
188                                                    className);
189                                    return (SelectorProvider)spClass.newInstance();
190                                }
191                                return null;
192                            } catch (Exception e) {
193                                throw new Error(e);
194                            }
195                        }
196                    });
197        }
198    
199        /**
200         * Creates a new open {@code DatagramChannel}.
201         * 
202         * @return the new channel.
203         * @throws IOException
204         *             if an I/O error occurs.
205         * @since Android 1.0
206         */
207        public abstract DatagramChannel openDatagramChannel() throws IOException;
208    
209        /**
210         * Creates a new {@code Pipe}.
211         * 
212         * @return the new pipe.
213         * @throws IOException
214         *             if an I/O error occurs.
215         * @since Android 1.0
216         */
217        public abstract Pipe openPipe() throws IOException;
218    
219        /**
220         * Creates a new selector.
221         * 
222         * @return the new selector.
223         * @throws IOException
224         *             if an I/O error occurs.
225         * @since Android 1.0
226         */
227        public abstract AbstractSelector openSelector() throws IOException;
228    
229        /**
230         * Creates a new open {@code ServerSocketChannel}.
231         * 
232         * @return the new channel.
233         * @throws IOException
234         *             if an I/O error occurs.
235         * @since Android 1.0
236         */
237        public abstract ServerSocketChannel openServerSocketChannel()
238                throws IOException;
239    
240        /**
241         * Create a new open {@code SocketChannel}.
242         * 
243         * @return the new channel.
244         * @throws IOException
245         *             if an I/O error occurs.
246         * @since Android 1.0
247         */
248        public abstract SocketChannel openSocketChannel() throws IOException;
249    
250        /**
251         * Returns the channel inherited from the instance that created this
252         * virtual machine.
253         * 
254         * @return the channel.
255         * @throws IOException
256         *             if an I/O error occurs.
257         * @throws SecurityException
258         *             if there is a security manager installed that does not permit
259         *             the runtime permission labeled "selectorProvider".
260         * @since Android 1.0
261         */
262        public Channel inheritedChannel() throws IOException {
263            // BEGIN android-added
264            SecurityManager smngr = System.getSecurityManager();
265            if (smngr != null) {
266                smngr.checkPermission(
267                        new RuntimePermission("inheritedChannel")); //$NON-NLS-1$
268            }
269            // END android-added
270            if (null == inheritedChannel) {
271                inheritedChannel = Platform.getNetworkSystem().inheritedChannel();
272            }
273            return inheritedChannel;
274        }
