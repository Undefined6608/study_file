017    package java.nio.channels;
018    
019    
020    import java.io.IOException;
021    import java.nio.ByteBuffer;
022    
023    /**
024     * The interface for channels that can write a set of buffers in a single
025     * operation. The corresponding interface for read operations is
026     * {@link ScatteringByteChannel}.
027     * 
028     * @since Android 1.0
029     */
030    public interface GatheringByteChannel extends WritableByteChannel {
031    
032        /**
033         * Writes bytes from all the given buffers to a channel.
034         * <p>
035         * This method is equivalent to: {@code write(buffers, 0, buffers.length);}
036         * </p>
037         * 
038         * @param buffers
039         *            the buffers containing bytes to be written.
040         * @return the number of bytes actually written.
041         * @throws AsynchronousCloseException
042         *             if the channel is closed by another thread during this write
043         *             operation.
044         * @throws ClosedByInterruptException
045         *             if another thread interrupts the calling thread while the
046         *             operation is in progress. The interrupt state of the calling 
047         *             thread is set and the channel is closed.
048         * @throws ClosedChannelException
049         *             if the channel is closed.
050         * @throws IndexOutOfBoundsException
051         *             if {@code offset < 0} or {@code length < 0}, or if
052         *             {@code offset + length} is greater than the size of
053         *             {@code buffers}.
054         * @throws IOException
055         *             if another I/O error occurs; details are in the message.
056         * @throws NonWritableChannelException
057         *             if the channel has not been opened in a mode that permits
058         *             writing.
059         * @since Android 1.0
060         */
061        public long write(ByteBuffer[] buffers) throws IOException;
062    
063        /**
064         * Writes bytes from a subset of the specified array of buffers to a
065         * channel. The subset is defined by {@code offset} and {@code length},
066         * indicating the first buffer and the number of buffers to use.
067         * <p>
068         * If a write operation is in progress, subsequent threads will block until
069         * the write is completed and then contend for the ability to write.
070         * </p>
071         * 
072         * @param buffers
073         *            the array of byte buffers that is the source for bytes written
074         *            to the channel.
075         * @param offset
076         *            the index of the first buffer in {@code buffers }to get bytes
077         *            from.
078         * @param length
079         *            the number of buffers to get bytes from.
080         * @return the number of bytes actually written.
081         * @throws AsynchronousCloseException
082         *             if the channel is closed by another thread during this write
083         *             operation.
084         * @throws ClosedByInterruptException
085         *             if another thread interrupts the calling thread while the
086         *             operation is in progress. The interrupt state of the calling 
087         *             thread is set and the channel is closed.
088         * @throws ClosedChannelException
089         *             if the channel is closed.
090         * @throws IndexOutOfBoundsException
091         *             if {@code offset < 0} or {@code length < 0}, or if
092         *             {@code offset + length} is greater than the size of
093         *             {@code buffers}.
094         * @throws IOException
095         *             if another I/O error occurs; details are in the message.
096         * @throws NonWritableChannelException
097         *             if the channel was not opened for writing.
098         * @since Android 1.0
099         */
100        public long write(ByteBuffer[] buffers, int offset, int length)
101                throws IOException;
102    }
