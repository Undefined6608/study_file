018    package java.nio.channels;
019    
020    import java.io.IOException;
021    
022    /**
023     * A {@code FileLock} represents a locked region of a file.
024     * <p>
025     * Locks have certain properties that enable collaborating processes to avoid
026     * the lost update problem or reading inconsistent data. Logically, a file lock
027     * can be <em>exclusive</em> or <em>shared</em>. Multiple processes can hold
028     * shared locks on the same region of a file, but only a single process can hold
029     * an exclusive lock on a given region of a file and no other process can
030     * simultaneously hold a shared lock overlapping the exclusive lock. An
031     * application can determine whether a {@code FileLock} is shared or exclusive
032     * via the {@code isShared()} method.
033     * </p>
034     * <p>
035     * Locks held by a particular process cannot overlap one another. Applications
036     * can determine whether a proposed lock will overlap by using the {@code
037     * overlaps(long, long)}) method. Locks held in other processes may overlap
038     * locks held in this process. Locks are shared amongst all threads in the
039     * acquiring process, and are therefore unsuitable for intra-process
040     * synchronization.
041     * </p>
042     * <p>
043     * Once a lock is acquired, it is immutable in all its state except {@code
044     * isValid()}. The lock will initially be valid, but may be rendered invalid by
045     * explicit removal of the lock, using {@code release()}, or implicitly by
046     * closing the channel or exiting the process (terminating the virtual machine).
047     * </p>
048     * <h3>Platform dependencies</h3>
049     * <p>
050     * Locks are intended to be true platform operating system file locks, and
051     * therefore locks held by the virtual machine process will be visible to other
052     * operating system processes.
053     * </p>
054     * <p>
055     * The characteristics of the underlying operating system locks will show
056     * through in the Java implementation. For example, some platforms' locks are
057     * 'mandatory' -- meaning the operating system enforces the locks on processes
058     * that attempt to access locked regions of files; whereas other platforms'
059     * locks are only 'advisory' -- meaning that processes are required to
060     * collaborate to ensure locks are acquired and there is a potential for
061     * processes to not play well. To be on the safe side, it is best to assume that
062     * the platform is adopting advisory locks and always acquire shared locks when
063     * reading a region of a file.
064     * </p>
065     * <p>
066     * On some platforms, the presence of a lock will prevent the file from being
067     * memory-mapped. On some platforms, closing a channel on a given file handle
068     * will release all the locks held on that file -- even if there are other
069     * channels open on the same file; their locks will also be released. The safe
070     * option here is to ensure that you only acquire locks on a single channel for
071     * a particular file and that becomes the synchronization point.
072     * </p>
073     * <p>
074     * Further care should be exercised when locking files maintained on network
075     * file systems, since they often have further limitations.
076     * </p>
077     * 
078     * @since Android 1.0
079     */
080    public abstract class FileLock {
081    
082        // The underlying file channel.
083        private final FileChannel channel;
084    
085        // The lock starting position.
086        private final long position;
087    
088        // The lock length in bytes
089        private final long size;
090    
091        // If true then shared, if false then exclusive
092        private final boolean shared;
093    
094        /**
095         * Constructs a new file lock instance for a given channel. The constructor
096         * enforces the starting position, length and sharing mode of the lock.
097         * 
098         * @param channel
099         *            the underlying file channel that holds the lock.
100         * @param position
101         *            the starting point for the lock.
102         * @param size
103         *            the length of the lock in number of bytes.
104         * @param shared
105         *            the lock's sharing mode of lock; {@code true} is shared,
106         *            {@code false} is exclusive.
107         * @since Android 1.0
108         */
109        protected FileLock(FileChannel channel, long position, long size,
110                boolean shared) {
111            super();
112            if (position < 0 || size < 0 || position + size < 0) {
113                throw new IllegalArgumentException();
114            }
115            this.channel = channel;
116            this.position = position;
117            this.size = size;
118            this.shared = shared;
119        }
120    
121        /**
122         * Returns the lock's {@link FileChannel}.
123         * 
124         * @return the channel.
125         * @since Android 1.0
126         */
127        public final FileChannel channel() {
128            return channel;
129        }
130    
131        /**
132         * Returns the lock's starting position in the file.
133         * 
134         * @return the lock position.
135         * @since Android 1.0
136         */
137        public final long position() {
138            return position;
139        }
140    
141        /**
142         * Returns the length of the file lock in bytes.
143         * 
144         * @return the size of the file lock in bytes.
145         * @since Android 1.0
146         */
147        public final long size() {
148            return size;
149        }
150    
151        /**
152         * Indicates if the file lock is shared with other processes or if it is
153         * exclusive.
154         * 
155         * @return {@code true} if the lock is a shared lock, {@code false} if it is
156         *         exclusive.
157         * @since Android 1.0
158         */
159        public final boolean isShared() {
160            return shared;
161        }
162    
163        /**
164         * Indicates if the receiver's lock region overlaps the region described
165         * in the parameter list.
166         * 
167         * @param start
168         *            the starting position for the comparative lock.
169         * @param length
170         *            the length of the comparative lock.
171         * @return {@code true} if there is an overlap, {@code false} otherwise.
172         * @since Android 1.0
173         */
174        public final boolean overlaps(long start, long length) {
175            final long end = position + size - 1;
176            final long newEnd = start + length - 1;
177            if (end < start || position > newEnd) {
178                return false;
179            }
180            return true;
181        }
182    
183        /**
184         * Indicates whether this lock is a valid file lock. The lock is
185         * valid unless the underlying channel has been closed or it has been
186         * explicitly released.
187         * 
188         * @return {@code true} if the lock is valid, {@code false} otherwise.
189         * @since Android 1.0
190         */
191        public abstract boolean isValid();
192    
193        /**
194         * Releases this particular lock on the file. If the lock is invalid then
195         * this method has no effect. Once released, the lock becomes invalid.
196         * 
197         * @throws ClosedChannelException
198         *             if the channel is already closed when an attempt to release
199         *             the lock is made.
200         * @throws IOException
201         *             if another I/O error occurs.
202         * @since Android 1.0
203         */
204        public abstract void release() throws IOException;
205    
206        /**
207         * Returns a string that shows the details of the lock suitable for display
208         * to an end user.
209         * 
210         * @return the display string.
211         * @since Android 1.0
212         */
213        public final String toString() {
214            StringBuffer buffer = new StringBuffer(64); // Guess length of string
215            buffer.append("FileLock: [position="); //$NON-NLS-1$
216            buffer.append(position);
217            buffer.append(", size="); //$NON-NLS-1$
218            buffer.append(size);
219            buffer.append(", shared="); //$NON-NLS-1$
220            buffer.append(Boolean.toString(shared));
221            buffer.append("]"); //$NON-NLS-1$
222            return buffer.toString();
223        }
224    }
