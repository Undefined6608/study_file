034    /**
035     * A class for turning a byte stream into a character stream. Data read from the
036     * source input stream is converted into characters by either a default or a
037     * provided character converter. The default encoding is taken from the
038     * "file.encoding" system property. {@code InputStreamReader} contains a buffer
039     * of bytes read from the source stream and converts these into characters as
040     * needed. The buffer size is 8K.
041     * 
042     * @see OutputStreamWriter
043     * 
044     * @since Android 1.0
045     */
046    public class InputStreamReader extends Reader {
047        private InputStream in;
048    
049        private static final int BUFFER_SIZE = 8192;
050    
051        private boolean endOfInput = false;
052    
053        CharsetDecoder decoder;
054    
055        ByteBuffer bytes = ByteBuffer.allocate(BUFFER_SIZE);
056    
057        /**
058         * Constructs a new {@code InputStreamReader} on the {@link InputStream}
059         * {@code in}. This constructor sets the character converter to the encoding
060         * specified in the "file.encoding" property and falls back to ISO 8859_1
061         * (ISO-Latin-1) if the property doesn't exist.
062         * 
063         * @param in
064         *            the input stream from which to read characters.
065         * @since Android 1.0
066         */
067        public InputStreamReader(InputStream in) {
068            super(in);
069            this.in = in;
070            String encoding = AccessController
071                    .doPrivileged(new PriviAction<String>(
072                            "file.encoding", "ISO8859_1")); //$NON-NLS-1$//$NON-NLS-2$
073            decoder = Charset.forName(encoding).newDecoder().onMalformedInput(
074                    CodingErrorAction.REPLACE).onUnmappableCharacter(
075                    CodingErrorAction.REPLACE);
076        }
077    
078        /**
079         * Constructs a new InputStreamReader on the InputStream {@code in}. The
080         * character converter that is used to decode bytes into characters is
081         * identified by name by {@code enc}. If the encoding cannot be found, an
082         * UnsupportedEncodingException error is thrown.
083         * 
084         * @param in
085         *            the InputStream from which to read characters.
086         * @param enc
087         *            identifies the character converter to use.
088         * @throws NullPointerException
089         *             if {@code enc} is {@code null}.
090         * @throws UnsupportedEncodingException
091         *             if the encoding specified by {@code enc} cannot be found.
092         * @since Android 1.0
093         */
094        public InputStreamReader(InputStream in, final String enc)
095                throws UnsupportedEncodingException {
096            super(in);
097            if (enc == null) {
098                throw new NullPointerException();
099            }
100            this.in = in;
101            try {
102                decoder = Charset.forName(enc).newDecoder().onMalformedInput(
103                        CodingErrorAction.REPLACE).onUnmappableCharacter(
104                        CodingErrorAction.REPLACE);
105            } catch (IllegalArgumentException e) {
106                throw new UnsupportedEncodingException();
107            }
108        }
109    
110        /**
111         * Constructs a new InputStreamReader on the InputStream {@code in} and
112         * CharsetDecoder {@code dec}.
113         * 
114         * @param in
115         *            the source InputStream from which to read characters.
116         * @param dec
117         *            the CharsetDecoder used by the character conversion.
118         * @since Android 1.0
119         */
120        public InputStreamReader(InputStream in, CharsetDecoder dec) {
121            super(in);
122            dec.averageCharsPerByte();
123            this.in = in;
124            decoder = dec;
125        }
126    
127        /**
128         * Constructs a new InputStreamReader on the InputStream {@code in} and
129         * Charset {@code charset}.
130         * 
131         * @param in
132         *            the source InputStream from which to read characters.
133         * @param charset
134         *            the Charset that defines the character converter
135         * @since Android 1.0
136         */
137        public InputStreamReader(InputStream in, Charset charset) {
138            super(in);
139            this.in = in;
140            decoder = charset.newDecoder().onMalformedInput(
141                    CodingErrorAction.REPLACE).onUnmappableCharacter(
142                    CodingErrorAction.REPLACE);
143        }
144    
145        /**
146         * Closes this reader. This implementation closes the source InputStream and
147         * releases all local storage.
148         * 
149         * @throws IOException
150         *             if an error occurs attempting to close this reader.
151         * @since Android 1.0
152         */
153        @Override
154        public void close() throws IOException {
155            synchronized (lock) {
156                // BEGIN android-added
157                if (decoder != null) {
158                    decoder.reset();
159                }
160                // END android-added
161                decoder = null;
162                if (in != null) {
163                    in.close();
164                    in = null;
165                }
166            }
167        }
168    
169        /**
170         * Returns the name of the encoding used to convert bytes into characters.
171         * The value {@code null} is returned if this reader has been closed.
172         * 
173         * @return the name of the character converter or {@code null} if this
174         *         reader is closed.
175         * @since Android 1.0
176         */
177        public String getEncoding() {
178            if (!isOpen()) {
179                return null;
180            }
181            return HistoricalNamesUtil.getHistoricalName(decoder.charset().name());
182        }

338        /**
339         * Reads a single character from this reader and returns it as an integer
340         * with the two higher-order bytes set to 0. Returns -1 if the end of the
341         * reader has been reached. The byte value is either obtained from
342         * converting bytes in this reader's buffer or by first filling the buffer
343         * from the source InputStream and then reading from the buffer.
344         * 
345         * @return the character read or -1 if the end of the reader has been
346         *         reached.
347         * @throws IOException
348         *             if this reader is closed or some other I/O error occurs.
349         * @since Android 1.0
350         */
351        @Override
352        public int read() throws IOException {
353            synchronized (lock) {
354                if (!isOpen()) {
355                    // K0070=InputStreamReader is closed.
356                    throw new IOException(Msg.getString("K0070")); //$NON-NLS-1$
357                }
358    
359                char buf[] = new char[1];
360                return read(buf, 0, 1) != -1 ? buf[0] : -1;
361            }
362        }
363    
364        /**
365         * Reads at most {@code length} characters from this reader and stores them
366         * at position {@code offset} in the character array {@code buf}. Returns
367         * the number of characters actually read or -1 if the end of the reader has
368         * been reached. The bytes are either obtained from converting bytes in this
369         * reader's buffer or by first filling the buffer from the source
370         * InputStream and then reading from the buffer.
371         * 
372         * @param buf
373         *            the array to store the characters read.
374         * @param offset
375         *            the initial position in {@code buf} to store the characters
376         *            read from this reader.
377         * @param length
378         *            the maximum number of characters to read.
379         * @return the number of characters read or -1 if the end of the reader has
380         *         been reached.
381         * @throws IndexOutOfBoundsException
382         *             if {@code offset < 0} or {@code length < 0}, or if
383         *             {@code offset + length} is greater than the length of
384         *             {@code buf}.
385         * @throws IOException
386         *             if this reader is closed or some other I/O error occurs.
387         * @since Android 1.0
388         */
389        @Override
390        public int read(char[] buf, int offset, int length) throws IOException {
391            synchronized (lock) {
392                if (!isOpen()) {
393                    // K0070=InputStreamReader is closed.
394                    throw new IOException(Msg.getString("K0070")); //$NON-NLS-1$
395                }
396                // BEGIN android-changed
397                // Exception priorities (in case of multiple errors) differ from
398                // RI, but are spec-compliant.
399                // made implicit null check explicit, used (offset | length) < 0
400                // instead of (offset < 0) || (length < 0) to safe one operation
401                if (buf == null) {
402                    throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
403                }
404                if ((offset | length) < 0 || offset > buf.length - length) {
405                    throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
406                }
407                // END android-changed
408                if (length == 0) {
409                    return 0;
410                }
411                
412                // allocate enough space for bytes if the default length is
413                // inadequate
414                int availableLen = in.available();     
415                if (Math.min(availableLen, length) > bytes.capacity()) {
416                    bytes = ByteBuffer.allocate(availableLen);
417                }
418                
419                CharBuffer out = CharBuffer.wrap(buf, offset, length);
420                CoderResult result = CoderResult.UNDERFLOW;
421                byte[] a = bytes.array();
422                boolean has_been_read = false;
423    
424                if (!bytes.hasRemaining() || bytes.limit() == bytes.capacity()) {
425                    // Nothing is available in the buffer...
426                    if (!bytes.hasRemaining()) {
427                        bytes.clear();
428                    }
429                    int readed = in.read(a, bytes.arrayOffset(), bytes.remaining());
430                    if (readed == -1) {
431                        endOfInput = true;
432                        return -1;
433                    }
434                    bytes.limit(readed);
435                    has_been_read = true;
436                }
437    
438                while (out.hasRemaining()) {
439                    if (bytes.hasRemaining()) {
440                        result = decoder.decode(bytes, out, false);
441                        if (!bytes.hasRemaining() && endOfInput) {
442                            decoder.decode(bytes, out, true);
443                            decoder.flush(out);
444                            decoder.reset();
445                            break;
446                        }
447                        if (!out.hasRemaining()
448                                || bytes.position() == bytes.limit()) {
449                            bytes.compact();
450                        }
451                    }
452                    if (in.available() > 0
453                            && (!has_been_read && out.hasRemaining())
454                            || out.position() == 0) {
455                        bytes.compact();
456                        int to_read = bytes.remaining();
457                        int off = bytes.arrayOffset() + bytes.position();
458    
459                        to_read = in.read(a, off, to_read);
460                        if (to_read == -1) {
461                            if (bytes.hasRemaining()) {
462                                bytes.flip();
463                            }
464                            endOfInput = true;
465                            break;
466                        }
467                        has_been_read = true;
468                        if (to_read > 0) {
469                            bytes.limit(bytes.position() + to_read);
470                            bytes.position(0);
471                        }
472                    } else {
473                        break;
474                    }
475                }
476    
477                if (result == CoderResult.UNDERFLOW && endOfInput) {
478                    result = decoder.decode(bytes, out, true);
479                    // FIXME: should flush at first, but seems ICU has a bug that it
480                    // will throw IAE if some malform/unmappable bytes found during
481                    // decoding
482                    // result = decoder.flush(chars);
483                    decoder.reset();
484                }
485                if (result.isMalformed()) {
486                    throw new MalformedInputException(result.length());
487                } else if (result.isUnmappable()) {
488                    throw new UnmappableCharacterException(result.length());
489                }
490                if (result == CoderResult.OVERFLOW && bytes.position() != 0) {
491                    bytes.flip();
492                }
493    
494                return out.position() - offset == 0 ? -1 : out.position() - offset;
495            }
496        }

498        /*
499         * Answer a boolean indicating whether or not this InputStreamReader is
500         * open.
501         */
502        private boolean isOpen() {
503            return in != null;
504        }
505    
506        /**
507         * Indicates whether this reader is ready to be read without blocking. If
508         * the result is {@code true}, the next {@code read()} will not block. If
509         * the result is {@code false} then this reader may or may not block when
510         * {@code read()} is called. This implementation returns {@code true} if
511         * there are bytes available in the buffer or the source stream has bytes
512         * available.
513         * 
514         * @return {@code true} if the receiver will not block when {@code read()}
515         *         is called, {@code false} if unknown or blocking will occur.
516         * @throws IOException
517         *             if this reader is closed or some other I/O error occurs.
518         * @since Android 1.0
519         */
520        @Override
521        public boolean ready() throws IOException {
522            synchronized (lock) {
523                if (in == null) {
524                    // K0070=InputStreamReader is closed.
525                    throw new IOException(Msg.getString("K0070")); //$NON-NLS-1$
526                }
527                try {
528                    return bytes.limit() != bytes.capacity() || in.available() > 0;
529                } catch (IOException e) {
530                    return false;
531                }
532            }
533        }
