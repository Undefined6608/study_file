018    package java.io;
019    
020    /**
021     * The lowest-level representation of a file, device, or
022     * socket. If is often used for wrapping an operating system "handle". Some
023     * I/O classes can be queried for the FileDescriptor they are operating on, and
024     * this descriptor can subsequently be used during the instantiation of another
025     * I/O class, so that the new object will reuse it.
026     * <p>
027     * The FileDescriptor class also contains static fields representing the
028     * system's standard input, output and error streams. These can be used directly
029     * if desired, but it is recommended to go through System.in, System.out, and
030     * System.err streams respectively.
031     * <p>
032     * Applications should not create new FileDescriptors.
033     * 
034     * @see FileInputStream#getFD()
035     * @see FileOutputStream#getFD()
036     * @see RandomAccessFile#getFD()
037     * 
038     * @since Android 1.0
039     */
040    public final class FileDescriptor {
041    
042        /**
043         * The FileDescriptor representing standard input.
044         * 
045         * @since Android 1.0
046         */
047        public static final FileDescriptor in = new FileDescriptor();
048    
049        /**
050         * FileDescriptor representing standard out.
051         * 
052         * @since Android 1.0
053         */
054        public static final FileDescriptor out = new FileDescriptor();
055    
056        /**
057         * FileDescriptor representing standard error.
058         * 
059         * @since Android 1.0
060         */
061        public static final FileDescriptor err = new FileDescriptor();
062    
063        // BEGIN android-changed
064        // file descriptors are only int on android
065        /**
066         * Represents a link to any underlying OS resources for this FileDescriptor.
067         * A value of -1 indicates that this FileDescriptor is invalid.
068         */
069        int descriptor = -1;
070        // END android-changed
071        
072        boolean readOnly = false; 
073    
074        private static native void oneTimeInitialization();
075    
076        static {
077            in.descriptor = 0;
078            out.descriptor = 1;
079            err.descriptor = 2;
080    
081            oneTimeInitialization();
082        }
083    
084        /**
085         * Constructs a new FileDescriptor containing an invalid handle. The
086         * contained handle is usually modified by native code at a later point.
087         * 
088         * @since Android 1.0
089         */
090        public FileDescriptor() {
091            super();
092        }
093    
094        /**
095         * Ensures that data which is buffered within the underlying implementation
096         * is written out to the appropriate device before returning.
097         * 
098         * @throws SyncFailedException
099         *             when the operation fails.
100         * @since Android 1.0
101         */
102        public void sync() throws SyncFailedException {
103            // if the descriptor is a read-only one, do nothing
104            if (!readOnly) {
105                syncImpl();
106            }
107        }
108        
109        private native void syncImpl() throws SyncFailedException;
110    
111        /**
112         * Indicates whether this FileDescriptor is valid.
113         * 
114         * @return {@code true} if this FileDescriptor is valid, {@code false}
115         *         otherwise.
116         * @since Android 1.0
117         */
118        public native boolean valid();
119    }
