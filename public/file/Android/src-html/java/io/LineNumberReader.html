018    package java.io;
019    
020    /**
021     * Wraps an existing {@link Reader} and counts the line terminators encountered
022     * while reading the data. The line number starts at 0 and is incremented any
023     * time {@code '\r'}, {@code '\n'} or {@code "\r\n"} is read. The class has an
024     * internal buffer for its data. The size of the buffer defaults to 8 KB.
025     * 
026     * @since Android 1.0
027     */
028    public class LineNumberReader extends BufferedReader {
029    
030        private int lineNumber;
031    
032        private int markedLineNumber = -1;
033    
034        private boolean lastWasCR;
035    
036        private boolean markedLastWasCR;
037    
038        /**
039         * Constructs a new LineNumberReader on the Reader {@code in}. The internal
040         * buffer gets the default size (8 KB).
041         * 
042         * @param in
043         *            the Reader that is buffered.
044         * @since Android 1.0
045         */
046        public LineNumberReader(Reader in) {
047            super(in);
048        }
049    
050        /**
051         * Constructs a new LineNumberReader on the Reader {@code in}. The size of
052         * the internal buffer is specified by the parameter {@code size}.
053         * 
054         * @param in
055         *            the Reader that is buffered.
056         * @param size
057         *            the size of the buffer to allocate.
058         * @throws IllegalArgumentException
059         *             if {@code size <= 0}.
060         * @since Android 1.0
061         */
062        public LineNumberReader(Reader in, int size) {
063            super(in, size);
064        }
065    
066        /**
067         * Returns the current line number for this reader. Numbering starts at 0.
068         * 
069         * @return the current line number.
070         * @since Android 1.0
071         */
072        public int getLineNumber() {
073            synchronized (lock) {
074                return lineNumber;
075            }
076        }
077    
078        /**
079         * Sets a mark position in this reader. The parameter {@code readlimit}
080         * indicates how many characters can be read before the mark is invalidated.
081         * Sending {@code reset()} will reposition this reader back to the marked
082         * position, provided that {@code readlimit} has not been surpassed. The
083         * line number associated with this marked position is also stored so that
084         * it can be restored when {@code reset()} is called.
085         * 
086         * @param readlimit
087         *            the number of characters that can be read from this stream
088         *            before the mark is invalidated.
089         * @throws IOException
090         *             if an error occurs while setting the mark in this reader.
091         * @see #markSupported()
092         * @see #reset()
093         * @since Android 1.0
094         */
095        @Override
096        public void mark(int readlimit) throws IOException {
097            synchronized (lock) {
098                super.mark(readlimit);
099                markedLineNumber = lineNumber;
100                markedLastWasCR = lastWasCR;
101            }
102        }
103    
104        /**
105         * Reads a single character from the source reader and returns it as an
106         * integer with the two higher-order bytes set to 0. Returns -1 if the end
107         * of the source reader has been reached.
108         * <p>
109         * The line number count is incremented if a line terminator is encountered.
110         * Recognized line terminator sequences are {@code '\r'}, {@code '\n'} and
111         * {@code "\r\n"}. Line terminator sequences are always translated into
112         * {@code '\n'}.
113         * </p>
114         * 
115         * @return the character read or -1 if the end of the source reader has been
116         *         reached.
117         * @throws IOException
118         *             if the reader is closed or another IOException occurs.
119         * @since Android 1.0
120         */
121        @Override
122        public int read() throws IOException {
123            synchronized (lock) {
124                int ch = super.read();
125                if (ch == '\n' && lastWasCR) {
126                    ch = super.read();
127                }
128                lastWasCR = false;
129                switch (ch) {
130                    case '\r':
131                        ch = '\n';
132                        lastWasCR = true;
133                        // fall through
134                    case '\n':
135                        lineNumber++;
136                }
137                return ch;
138            }
139        }
140    
141        /**
142         * Reads at most {@code count} characters from the source reader and stores
143         * them in the character array {@code buffer} starting at {@code offset}.
144         * Returns the number of characters actually read or -1 if no characters
145         * have been read and the end of this reader has been reached.
146         * <p>
147         * The line number count is incremented if a line terminator is encountered.
148         * Recognized line terminator sequences are {@code '\r'}, {@code '\n'} and
149         * {@code "\r\n"}. Line terminator sequences are always translated into
150         * {@code '\n'}.
151         * </p>
152         * 
153         * @param buffer
154         *            the array in which to store the characters read.
155         * @param offset
156         *            the initial position in {@code buffer} to store the characters
157         *            read from this reader.
158         * @param count
159         *            the maximum number of characters to store in {@code buffer}.
160         * @return the number of characters actually read or -1 if the end of the
161         *         source reader has been reached while reading.
162         * @throws IOException
163         *             if this reader is closed or another IOException occurs.
164         * @since Android 1.0
165         */
166        @Override
167        public int read(char[] buffer, int offset, int count) throws IOException {
168            synchronized (lock) {
169                int read = super.read(buffer, offset, count);
170                if (read == -1) {
171                    return -1;
172                }
173                for (int i = 0; i < read; i++) {
174                    char ch = buffer[offset + i];
175                    if (ch == '\r') {
176                        lineNumber++;
177                        lastWasCR = true;
178                    } else if (ch == '\n') {
179                        if (!lastWasCR) {
180                            lineNumber++;
181                        }
182                        lastWasCR = false;
183                    } else {
184                        lastWasCR = false;
185                    }
186                }
187                return read;
188            }
189        }
190    
191        /**
192         * Returns the next line of text available from this reader. A line is
193         * represented by 0 or more characters followed by {@code '\r'},
194         * {@code '\n'}, {@code "\r\n"} or the end of the stream. The returned
195         * string does not include the newline sequence.
196         * 
197         * @return the contents of the line or {@code null} if no characters have
198         *         been read before the end of the stream has been reached.
199         * @throws IOException
200         *             if this reader is closed or another IOException occurs.
201         * @since Android 1.0
202         */
203        @Override
204        public String readLine() throws IOException {
205            synchronized (lock) {
206                /* Typical Line Length */
207                StringBuilder result = new StringBuilder(80);
208                while (true) {
209                    int character = read();
210                    if (character == -1) {
211                        return result.length() != 0 ? result.toString() : null;
212                    }
213                    if (character == '\n') {
214                        return result.toString();
215                    }
216                    result.append((char) character);
217                }
218            }
219        }
220    
221        /**
222         * Resets this reader to the last marked location. It also resets the line
223         * count to what is was when this reader was marked. This implementation
224         * resets the source reader.
225         * 
226         * @throws IOException
227         *             if this reader is already closed, no mark has been set or the
228         *             mark is no longer valid because more than {@code readlimit}
229         *             bytes have been read since setting the mark.
230         * @see #mark(int)
231         * @see #markSupported()
232         * @since Android 1.0
233         */
234        @Override
235        public void reset() throws IOException {
236            synchronized (lock) {
237                super.reset();
238                lineNumber = markedLineNumber;
239                lastWasCR = markedLastWasCR;
240            }
241        }
242    
243        /**
244         * Sets the line number of this reader to the specified {@code lineNumber}.
245         * Note that this may have side effects on the line number associated with
246         * the last marked position.
247         * 
248         * @param lineNumber
249         *            the new line number value.
250         * @see #mark(int)
251         * @see #reset()
252         * @since Android 1.0
253         */
254        public void setLineNumber(int lineNumber) {
255            synchronized (lock) {
256                this.lineNumber = lineNumber;
257            }
258        }
259    
260        /**
261         * Skips {@code count} number of characters in this reader. Subsequent
262         * {@code read()}'s will not return these characters unless {@code reset()}
263         * is used. This implementation skips {@code count} number of characters in
264         * the source reader and increments the line number count whenever line
265         * terminator sequences are skipped.
266         * 
267         * @param count
268         *            the number of characters to skip.
269         * @return the number of characters actually skipped.
270         * @throws IllegalArgumentException
271         *             if {@code count < 0}.
272         * @throws IOException
273         *             if this reader is closed or another IOException occurs.
274         * @see #mark(int)
275         * @see #read()
276         * @see #reset()
277         * @since Android 1.0
278         */
279        @Override
280        public long skip(long count) throws IOException {
281            if (count < 0) {
282                throw new IllegalArgumentException();
283            }
284            synchronized (lock) {
285                for (int i = 0; i < count; i++) {
286                    if (read() == -1) {
287                        return i;
288                    }
289                }
290                return count;
291            }
292        }
293    }
