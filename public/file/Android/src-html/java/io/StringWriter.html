018    package java.io;
019    
020    // BEGIN android-added
021    import org.apache.harmony.luni.util.Msg;
022    // ENd android-added
023    
024    /**
025     * A specialized {@link Writer} that writes characters to a {@code StringBuffer}
026     * in a sequential manner, appending them in the process. The result can later
027     * be queried using the {@link #StringWriter(int)} or {@link #toString()}
028     * methods.
029     * 
030     * @see StringReader
031     * 
032     * @since Android 1.0
033     */
034    public class StringWriter extends Writer {
035    
036        private StringBuffer buf;
037    
038        /**
039         * Constructs a new {@code StringWriter} which has a {@link StringBuffer}
040         * allocated with the default size of 16 characters. The {@code
041         * StringBuffer} is also the {@code lock} used to synchronize access to this
042         * writer.
043         * 
044         * @since Android 1.0
045         */
046        public StringWriter() {
047            super();
048            buf = new StringBuffer(16);
049            lock = buf;
050        }
051    
052        /**
053         * Constructs a new {@code StringWriter} which has a {@link StringBuffer}
054         * allocated with a size of {@code initialSize} characters. The {@code
055         * StringBuffer} is also the {@code lock} used to synchronize access to this
056         * writer.
057         * 
058         * @param initialSize
059         *            the intial size of the target string buffer.
060         * @since Android 1.0
061         */
062        public StringWriter(int initialSize) {
063            if (initialSize < 0) {
064                throw new IllegalArgumentException();
065            }
066            buf = new StringBuffer(initialSize);
067            lock = buf;
068        }
069    
070        /**
071         * Calling this method has no effect. In contrast to most {@code Writer} subclasses,
072         * the other methods in {@code StringWriter} do not throw an {@code IOException} if
073         * {@code close()} has been called.
074         * 
075         * @throws IOException
076         *             if an error occurs while closing this writer.
077         * @since Android 1.0
078         */
079        @Override
080        public void close() throws IOException {
081            /* empty */
082        }
083    
084        /**
085         * Calling this method has no effect.
086         * 
087         * @since Android 1.0
088         */
089        @Override
090        public void flush() {
091            /* empty */
092        }
093    
094        /**
095         * Gets a reference to this writer's internal {@link StringBuffer}. Any
096         * changes made to the returned buffer are reflected in this writer.
097         * 
098         * @return a reference to this writer's internal {@code StringBuffer}.
099         * @since Android 1.0
100         */
101        public StringBuffer getBuffer() {
102            synchronized (lock) {
103                return buf;
104            }
105        }
106    
107        /**
108         * Gets a copy of the contents of this writer as a string.
109         * 
110         * @return this writer's contents as a string.
111         * @since Android 1.0
112         */
113        @Override
114        public String toString() {
115            synchronized (lock) {
116                return buf.toString();
117            }
118        }
119    
120        /**
121         * Writes {@code count} characters starting at {@code offset} in {@code buf}
122         * to this writer's {@code StringBuffer}.
123         * 
124         * @param cbuf
125         *            the non-null character array to write.
126         * @param offset
127         *            the index of the first character in {@code cbuf} to write.
128         * @param count
129         *            the maximum number of characters to write.
130         * @throws IndexOutOfBoundsException
131         *             if {@code offset < 0} or {@code count < 0}, or if {@code
132         *             offset + count} is greater than the size of {@code buf}.
133         * @since Android 1.0
134         */
135        @Override
136        public void write(char[] cbuf, int offset, int count) {
137            // avoid int overflow
138            // BEGIN android-changed
139            // Exception priorities (in case of multiple errors) differ from
140            // RI, but are spec-compliant.
141            // removed redundant check, added null check, used (offset | count) < 0
142            // instead of (offset < 0) || (count < 0) to safe one operation
143            if (cbuf == null) {
144                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
145            }
146            if ((offset | count) < 0 || count > cbuf.length - offset) {
147                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
148            }
149            // END android-changed
150            synchronized (lock) {
151                this.buf.append(cbuf, offset, count);
152            }
153        }
154    
155        /**
156         * Writes one character to this writer's {@code StringBuffer}. Only the two
157         * least significant bytes of the integer {@code oneChar} are written.
158         * 
159         * @param oneChar
160         *            the character to write to this writer's {@code StringBuffer}.
161         * @since Android 1.0
162         */
163        @Override
164        public void write(int oneChar) {
165            synchronized (lock) {
166                buf.append((char) oneChar);
167            }
168        }
169    
170        /**
171         * Writes the characters from the specified string to this writer's {@code
172         * StringBuffer}.
173         * 
174         * @param str
175         *            the non-null string containing the characters to write.
176         * @since Android 1.0
177         */
178        @Override
179        public void write(String str) {
180            synchronized (lock) {
181                buf.append(str);
182            }
183        }
184    
185        /**
186         * Writes {@code count} characters from {@code str} starting at {@code
187         * offset} to this writer's {@code StringBuffer}.
188         * 
189         * @param str
190         *            the non-null string containing the characters to write.
191         * @param offset
192         *            the index of the first character in {@code str} to write.
193         * @param count
194         *            the number of characters from {@code str} to write.
195         * @throws StringIndexOutOfBoundsException
196         *             if {@code offset < 0} or {@code count < 0}, or if {@code
197         *             offset + count} is greater than the length of {@code str}.
198         * @since Android 1.0
199         */
200        @Override
201        public void write(String str, int offset, int count) {
202            String sub = str.substring(offset, offset + count);
203            synchronized (lock) {
204                buf.append(sub);
205            }
206        }
207    
208        /**
209         * Appends the character {@code c} to this writer's {@code StringBuffer}.
210         * This method works the same way as {@link #write(int)}.
211         * 
212         * @param c
213         *            the character to append to the target stream.
214         * @return this writer.
215         * @since Android 1.0
216         */
217        @Override
218        public StringWriter append(char c) {
219            write(c);
220            return this;
221        }
222    
223        /**
224         * Appends the character sequence {@code csq} to this writer's {@code
225         * StringBuffer}. This method works the same way as {@code
226         * StringWriter.write(csq.toString())}. If {@code csq} is {@code null}, then
227         * the string "null" is written to the target stream.
228         * 
229         * @param csq
230         *            the character sequence appended to the target.
231         * @return this writer.
232         * @since Android 1.0
233         */
234        @Override
235        public StringWriter append(CharSequence csq) {
236            if (null == csq) {
237                append(TOKEN_NULL, 0, TOKEN_NULL.length());
238            } else {
239                append(csq, 0, csq.length());
240            }
241            return this;
242        }
243    
244        /**
245         * Appends a subsequence of the character sequence {@code csq} to this
246         * writer's {@code StringBuffer}. This method works the same way as {@code
247         * StringWriter.writer(csq.subsequence(start, end).toString())}. If {@code
248         * csq} is {@code null}, then the specified subsequence of the string "null"
249         * will be written to the target.
250         * 
251         * @param csq
252         *            the character sequence appended to the target.
253         * @param start
254         *            the index of the first char in the character sequence appended
255         *            to the target.
256         * @param end
257         *            the index of the character following the last character of the
258         *            subsequence appended to the target.
259         * @return this writer.
260         * @throws IndexOutOfBoundsException
261         *             if {@code start > end}, {@code start < 0}, {@code end < 0} or
262         *             either {@code start} or {@code end} are greater or equal than
263         *             the length of {@code csq}.
264         * @since Android 1.0
265         */
266        @Override
267        public StringWriter append(CharSequence csq, int start, int end) {
268            if (null == csq) {
269                csq = TOKEN_NULL;
270            }
271            String output = csq.subSequence(start, end).toString();
272            write(output, 0, output.length());
273            return this;
274        }
275    }
