029    /**
030     * Allows reading from and writing to a file in a random-access manner. This is
031     * different from the uni-directional sequential access that a
032     * {@link FileInputStream} or {@link FileOutputStream} provides. If the file is
033     * opened in read/write mode, write operations are available as well. The
034     * position of the next read or write operation can be moved forwards and
035     * backwards after every operation.
036     * 
037     * @since Android 1.0
038     */
039    public class RandomAccessFile implements DataInput, DataOutput, Closeable {
040        /**
041         * The FileDescriptor representing this RandomAccessFile.
042         */
043        private FileDescriptor fd;
044    
045        private boolean syncMetadata = false;
046    
047        // The unique file channel associated with this FileInputStream (lazily
048        // initialized).
049        private FileChannel channel;

051        private IFileSystem fileSystem = Platform.getFileSystem();
052    
053        private boolean isReadOnly;

060        }

065        /**
066         * Constructs a new {@code RandomAccessFile} based on {@code file} and opens
067         * it according to the access string in {@code mode}. 
068         * <p><a id="accessmode"/>
069         * {@code mode} may have one of following values: 
070         * <table border="0">
071         * <tr>
072         * <td>{@code "r"}</td>
073         * <td>The file is opened in read-only mode. An {@code IOException} is
074         * thrown if any of the {@code write} methods is called.</td>
075         * </tr>
076         * <tr>
077         * <td>{@code "rw"}</td>
078         * <td>The file is opened for reading and writing. If the file does not
079         * exist, it will be created.</td>
080         * </tr>
081         * <tr>
082         * <td>{@code "rws"}</td>
083         * <td>The file is opened for reading and writing. Every change of the
084         * file's content or metadata must be written synchronously to the target
085         * device.</td>
086         * </tr>
087         * <tr>
088         * <td>{@code "rwd"}</td>
089         * <td>The file is opened for reading and writing. Every change of the
090         * file's content must be written synchronously to the target device.</td>
091         * </tr>
092         * </table>
093         * </p>
094         * @param file
095         *            the file to open.
096         * @param mode
097         *            the file access <a href="#accessmode">mode</a>, either {@code
098         *            "r"}, {@code "rw"}, {@code "rws"} or {@code "rwd"}.
099         * @throws FileNotFoundException
100         *             if the file cannot be opened or created according to {@code
101         *             mode}.
102         * @throws IllegalArgumentException
103         *             if {@code mode} is not {@code "r"}, {@code "rw"}, {@code
104         *             "rws"} or {@code "rwd"}.
105         * @throws SecurityException
106         *             if a {@code SecurityManager} is installed and it denies
107         *             access request according to {@code mode}.
108         * @see java.lang.SecurityManager#checkRead(FileDescriptor)
109         * @see java.lang.SecurityManager#checkWrite(FileDescriptor)
110         * @since Android 1.0
111         */
112        public RandomAccessFile(File file, String mode)
113                throws FileNotFoundException {
114            super();
115    
116            options = 0;
117            
118            fd = new FileDescriptor();
119           
120            if (mode.equals("r")) { //$NON-NLS-1$
121                isReadOnly = true;
122                fd.readOnly = true;
123                options = IFileSystem.O_RDONLY;
124            } else if (mode.equals("rw") || mode.equals("rws") || mode.equals("rwd")) { //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
125                isReadOnly = false;
126                options = IFileSystem.O_RDWR;
127    
128                if (mode.equals("rws")) { //$NON-NLS-1$
129                    // Sync file and metadata with every write
130                    syncMetadata = true;
131                } else if (mode.equals("rwd")) { //$NON-NLS-1$
132                    // Sync file, but not necessarily metadata
133                    options = IFileSystem.O_RDWRSYNC;
134                }
135            } else {
136                throw new IllegalArgumentException(Msg.getString("K0081")); //$NON-NLS-1$
137            }
138    
139            SecurityManager security = System.getSecurityManager();
140            if (security != null) {
141                security.checkRead(file.getPath());
142                if (!isReadOnly) {
143                    security.checkWrite(file.getPath());
144                }
145            }
146            
147            fd.descriptor = fileSystem.open(file.properPath(true), options);
148    
149            // if we are in "rws" mode, attempt to sync file+metadata
150            if (syncMetadata) {
151                try {
152                    fd.sync();
153                } catch (IOException e) {
154                    // Ignored
155                }
156            }
157        }

189        /**
190         * Closes this file.
191         * 
192         * @throws IOException
193         *             if an error occurs while closing this file.
194         * @since Android 1.0
195         */
196        public void close() throws IOException {
197            // BEGIN android-changed
198            synchronized (this) {
199                if (channel != null && channel.isOpen()) {
200                    channel.close();
201                    channel = null;
202                }
203                if (fd != null && fd.descriptor >= 0) {
204                    fileSystem.close(fd.descriptor);
205                    fd.descriptor = -1;
206                }
207            }
208            // END android-changed
209        }

269         * @throws IOException

273            if (fd.descriptor < 0) {
274                throw new IOException();

278        /**
279         * Returns the length of this file in bytes.
280         * 
281         * @return the file's length in bytes.
282         * @throws IOException
283         *             if this file is closed or some other I/O error occurs.
284         * @since Android 1.0
285         */
286        public long length() throws IOException {
287            openCheck();
288            synchronized (repositionLock) {
289                long currentPosition = fileSystem.seek(fd.descriptor, 0L,
290                        IFileSystem.SEEK_CUR);
291                long endOfFilePosition = fileSystem.seek(fd.descriptor, 0L,
292                        IFileSystem.SEEK_END);
293                fileSystem.seek(fd.descriptor, currentPosition,
294                        IFileSystem.SEEK_SET);
295                return endOfFilePosition;
296            }
297        }

299        /**
300         * Reads a single byte from the current position in this file and returns it
301         * as an integer in the range from 0 to 255. Returns -1 if the end of the
302         * file has been reached. Blocks until one byte has been read, the end of
303         * the file is detected or an exception is thrown.
304         * 
305         * @return the byte read or -1 if the end of the file has been reached.
306         * @throws IOException
307         *             if this file is closed or another I/O error occurs.
308         * @since Android 1.0
309         */
310        public int read() throws IOException {
311            openCheck();
312            byte[] bytes = new byte[1];
313            synchronized (repositionLock) {
314                long readed = fileSystem.read(fd.descriptor, bytes, 0, 1);
315                return readed == -1 ? -1 : bytes[0] & 0xff;
316            }
317        }

337        /**
338         * Reads at most {@code count} bytes from the current position in this file
339         * and stores them in the byte array {@code buffer} starting at {@code
340         * offset}. Blocks until {@code count} bytes have been read, the end of the
341         * file is reached or an exception is thrown.
342         * 
343         * @param buffer
344         *            the array in which to store the bytes read from this file.
345         * @param offset
346         *            the initial position in {@code buffer} to store the bytes read
347         *            from this file.
348         * @param count
349         *            the maximum number of bytes to store in {@code buffer}.
350         * @return the number of bytes actually read or -1 if the end of the stream
351         *         has been reached.
352         * @throws IndexOutOfBoundsException
353         *             if {@code offset < 0} or {@code count < 0}, or if {@code
354         *             offset + count} is greater than the size of {@code buffer}.
355         * @throws IOException
356         *             if this file is closed or another I/O error occurs.
357         * @since Android 1.0
358         */
359        public int read(byte[] buffer, int offset, int count) throws IOException {
360            // have to have four comparisions to not miss integer overflow cases
361            // BEGIN android-changed
362            // Exception priorities (in case of multiple errors) differ from
363            // RI, but are spec-compliant.
364            // made implicit null check explicit, used (offset | count) < 0
365            // instead of (offset < 0) || (count < 0) to safe one operation
366            if (buffer == null) {
367                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
368            }
369            if ((offset | count) < 0 || count > buffer.length - offset) {
370                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
371            }
372            // END android-changed
373            if (0 == count) {
374                return 0;
375            }
376            openCheck();
377            synchronized (repositionLock) {
378                return (int) fileSystem.read(fd.descriptor, buffer, offset, count);
379            }
380        }

498        /**
499         * Read bytes from this file into {@code buffer} starting at offset {@code
500         * offset}. This method blocks until {@code count} number of bytes have been
501         * read.
502         * 
503         * @param buffer
504         *            the buffer to read bytes into.
505         * @param offset
506         *            the initial position in {@code buffer} to store the bytes read
507         *            from this file.
508         * @param count
509         *            the maximum number of bytes to store in {@code buffer}.
510         * @throws EOFException
511         *             if the end of this file is detected.
512         * @throws IndexOutOfBoundsException
513         *             if {@code offset < 0} or {@code count < 0}, or if {@code
514         *             offset + count} is greater than the length of {@code buffer}.
515         * @throws IOException
516         *             if this file is closed or another I/O error occurs.
517         * @throws NullPointerException
518         *             if {@code buffer} is {@code null}.
519         * @since Android 1.0
520         */
521        public final void readFully(byte[] buffer, int offset, int count)
522                throws IOException {
523            if (buffer == null) {
524                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
525            }
526            // avoid int overflow
527            // BEGIN android-changed
528            // Exception priorities (in case of multiple errors) differ from
529            // RI, but are spec-compliant.
530            // removed redundant check, used (offset | count) < 0
531            // instead of (offset < 0) || (count < 0) to safe one operation
532            if ((offset | count) < 0 || count > buffer.length - offset) {
533                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
534            }
535            // END android-changed
536            while (count > 0) {
537                int result = read(buffer, offset, count);
538                if (result < 0) {
539                    throw new EOFException();
540                }
541                offset += result;
542                count -= result;
543            }
544        }

548         * until four bytes have been read, the end of the file is reached or an

565                    + ((buffer[2] & 0xff) << 8) + (buffer[3] & 0xff);

568        /**
569         * Reads a line of text form the current position in this file. A line is
570         * represented by zero or more characters followed by {@code '\n'}, {@code
571         * '\r'}, {@code "\r\n"} or the end of file marker. The string does not
572         * include the line terminating sequence.
573         * <p>
574         * Blocks until a line terminating sequence has been read, the end of the
575         * file is reached or an exception is thrown.
576         * </p>
577         * 
578         * @return the contents of the line or {@code null} if no characters have
579         *         been read before the end of the file has been reached.
580         * @throws IOException
581         *             if this file is closed or another I/O error occurs.
582         * @since Android 1.0
583         */
584        public final String readLine() throws IOException {
585            StringBuilder line = new StringBuilder(80); // Typical line length
586            boolean foundTerminator = false;
587            long unreadPosition = 0;
588            while (true) {
589                int nextByte = read();
590                switch (nextByte) {
591                    case -1:
592                        return line.length() != 0 ? line.toString() : null;
593                    case (byte) '\r':
594                        if (foundTerminator) {
595                            seek(unreadPosition);
596                            return line.toString();
597                        }
598                        foundTerminator = true;
599                        /* Have to be able to peek ahead one byte */
600                        unreadPosition = getFilePointer();
601                        break;
602                    case (byte) '\n':
603                        return line.toString();
604                    default:
605                        if (foundTerminator) {
606                            seek(unreadPosition);
607                            return line.toString();
608                        }
609                        line.append((char) nextByte);
610                }
611            }
612        }

636                    + ((buffer[6] & 0xff) << 8)
637                    + (buffer[7] & 0xff);

684         * returns it as an integer. Blocks until two bytes have been read, the end of
685         * the file is reached or an exception is thrown.
686         * 
687         * @return the next unsigned short value from this file as an int.
688         * @throws EOFException
689         *             if the end of this file is detected.
690         * @throws IOException
691         *             if this file is closed or another I/O error occurs.
692         * @see #writeShort(int)
693         * @since Android 1.0
694         */
695        public final int readUnsignedShort() throws IOException {
696            byte[] buffer = new byte[2];
697            if (read(buffer, 0, buffer.length) != buffer.length) {
698                throw new EOFException();
699            }
700            return ((buffer[0] & 0xff) << 8) + (buffer[1] & 0xff);
701        }

703        /**
704         * Reads a string that is encoded in {@link DataInput modified UTF-8} from
705         * this file. The number of bytes that must be read for the complete string
706         * is determined by the first two bytes read from the file. Blocks until all
707         * required bytes have been read, the end of the file is reached or an
708         * exception is thrown.
709         * 
710         * @return the next string encoded in {@link DataInput modified UTF-8} from
711         *         this file.
712         * @throws EOFException
713         *             if the end of this file is detected.
714         * @throws IOException
715         *             if this file is closed or another I/O error occurs.
716         * @throws UTFDataFormatException
717         *             if the bytes read cannot be decoded into a character string.
718         * @see #writeUTF(String)
719         * @since Android 1.0
720         */
721        public final String readUTF() throws IOException {
722            int utfSize = readUnsignedShort();
723            if (utfSize == 0) {
724                return ""; //$NON-NLS-1$
725            }
726            byte[] buf = new byte[utfSize];
727            if (read(buf, 0, buf.length) != buf.length) {
728                throw new EOFException();
729            }
730            return Util.convertFromUTF8(buf, 0, utfSize);
731        }

759         * Sets the length of this file to {@code newLength}. If the current file is
760         * smaller, it is expanded but the contents from the previous end of the
761         * file to the new end are undefined. The file is truncated if its current
762         * size is bigger than {@code newLength}. If the current file pointer
763         * position is in the truncated part, it is set to the end of the file.
764         * 
765         * @param newLength
766         *            the new file length in bytes.
767         * @throws IllegalArgumentException
768         *             if {@code newLength < 0}.
769         * @throws IOException
770         *             if this file is closed or another I/O error occurs.
771         * @since Android 1.0
772         */
773        public void setLength(long newLength) throws IOException {
774            openCheck();
775            if (newLength < 0) {
776                throw new IllegalArgumentException();
777            }
778            synchronized (repositionLock) {
779                long position = fileSystem.seek(fd.descriptor, 0,
780                        IFileSystem.SEEK_CUR);
781                fileSystem.truncate(fd.descriptor, newLength);
782                seek(position > newLength ? newLength : position);
783            }
784    
785            // if we are in "rws" mode, attempt to sync file+metadata
786            if (syncMetadata) {
787                fd.sync();
788            }
789        }

791        /**
792         * Skips over {@code count} bytes in this file. Less than {@code count}
793         * bytes are skipped if the end of the file is reached or an exception is
794         * thrown during the operation. Nothing is done if {@code count} is
795         * negative.
796         * 
797         * @param count
798         *            the number of bytes to skip.
799         * @return the number of bytes actually skipped.
800         * @throws IOException
801         *             if this file is closed or another I/O error occurs.
802         * @since Android 1.0
803         */
804        public int skipBytes(int count) throws IOException {
805            if (count > 0) {
806                long currentPos = getFilePointer(), eof = length();
807                int newCount = (int) ((currentPos + count > eof) ? eof - currentPos
808                        : count);
809                seek(currentPos + newCount);
810                return newCount;
811            }
812            return 0;
813        }

833        /**
834         * Writes {@code count} bytes from the byte array {@code buffer} to this
835         * file, starting at the current file pointer and using {@code offset} as
836         * the first position within {@code buffer} to get bytes.
837         * 
838         * @param buffer
839         *            the buffer to write to this file.
840         * @param offset
841         *            the index of the first byte in {@code buffer} to write.
842         * @param count
843         *            the number of bytes from {@code buffer} to write.
844         * @throws IndexOutOfBoundsException
845         *             if {@code count < 0}, {@code offset < 0} or {@code count +
846         *             offset} is greater than the size of {@code buffer}.
847         * @throws IOException
848         *             if an I/O error occurs while writing to this file.
849         * @see #read(byte[], int, int)
850         * @see #readFully(byte[], int, int)
851         * @since Android 1.0
852         */
853        public void write(byte[] buffer, int offset, int count) throws IOException {
854            // BEGIN android-changed
855            // Exception priorities (in case of multiple errors) differ from
856            // RI, but are spec-compliant.
857            // made implicit null check explicit,
858            // removed redundant check, used (offset | count) < 0
859            // instead of (offset < 0) || (count < 0) to safe one operation
860            if (buffer == null) {
861                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
862            }
863            if ((offset | count) < 0 || count > buffer.length - offset) {
864                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
865            }
866            // END android-changed
867            if (count == 0) {
868                return;
869            }
870            // BEGIN android-added
871            openCheck();
872            // END android-added
873            synchronized (repositionLock) {
874                fileSystem.write(fd.descriptor, buffer, offset, count);
875            }
876    
877            // if we are in "rws" mode, attempt to sync file+metadata
878            if (syncMetadata) {
879                fd.sync();
880            }
881        }

1097        /**
1098         * Writes a string encoded with {@link DataInput modified UTF-8} to this
1099         * file, starting at the current file pointer.
1100         * 
1101         * @param str
1102         *            the string to write in {@link DataInput modified UTF-8}
1103         *            format.
1104         * @throws IOException
1105         *             if an I/O error occurs while writing to this file.
1106         * @throws UTFDataFormatException
1107         *             if the encoded string is longer than 65535 bytes.
1108         * @see #readUTF()
1109         * @since Android 1.0
1110         */
1111        public final void writeUTF(String str) throws IOException {
1112            int utfCount = 0, length = str.length();
1113            for (int i = 0; i < length; i++) {
1114                int charValue = str.charAt(i);
1115                if (charValue > 0 && charValue <= 127) {
1116                    utfCount++;
1117                } else if (charValue <= 2047) {
1118                    utfCount += 2;
1119                } else {
1120                    utfCount += 3;
1121                }
1122            }
1123            if (utfCount > 65535) {
1124                throw new UTFDataFormatException(Msg.getString("K0068")); //$NON-NLS-1$
1125            }
1126            byte utfBytes[] = new byte[utfCount + 2];
1127            int utfIndex = 2;
1128            for (int i = 0; i < length; i++) {
1129                int charValue = str.charAt(i);
1130                if (charValue > 0 && charValue <= 127) {
1131                    utfBytes[utfIndex++] = (byte) charValue;
1132                } else if (charValue <= 2047) {
1133                    utfBytes[utfIndex++] = (byte) (0xc0 | (0x1f & (charValue >> 6)));
1134                    utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & charValue));
1135                } else {
1136                    utfBytes[utfIndex++] = (byte) (0xe0 | (0x0f & (charValue >> 12)));
1137                    utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & (charValue >> 6)));
1138                    utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & charValue));
1139                }
1140            }
1141            utfBytes[0] = (byte) (utfCount >> 8);
1142            utfBytes[1] = (byte) utfCount;
1143            write(utfBytes);
1144        }
