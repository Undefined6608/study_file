018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * A specialized {@link Writer} for class for writing content to an (internal)
024     * char array. As bytes are written to this writer, the char array may be
025     * expanded to hold more characters. When the writing is considered to be
026     * finished, a copy of the char array can be requested from the class.
027     * 
028     * @see CharArrayReader
029     * 
030     * @since Android 1.0
031     */
032    public class CharArrayWriter extends Writer {
033    
034        /**
035         * The buffer for characters.
036         * 
037         * @since Android 1.0
038         */
039        protected char[] buf;
040    
041        /**
042         * The ending index of the buffer.
043         * 
044         * @since Android 1.0
045         */
046        protected int count;
047    
048        /**
049         * Constructs a new CharArrayWriter which has a buffer allocated with the
050         * default size of 32 characters. This buffer is also used as the
051         * {@code lock} to synchronize access to this writer.
052         * 
053         * @since Android 1.0
054         */
055        public CharArrayWriter() {
056            super();
057            buf = new char[32];
058            lock = buf;
059        }
060    
061        /**
062         * Constructs a new CharArrayWriter which has a buffer allocated with the
063         * size of {@code initialSize} characters. The buffer is also used as the
064         * {@code lock} to synchronize access to this writer.
065         * 
066         * @param initialSize
067         *            the initial size of this CharArrayWriters buffer.
068         * @throws IllegalArgumentException
069         *             if {@code initialSize < 0}.
070         * @since Android 1.0
071         */
072        public CharArrayWriter(int initialSize) {
073            super();
074            if (initialSize < 0) {
075                throw new IllegalArgumentException(Msg.getString("K005e")); //$NON-NLS-1$
076            }
077            buf = new char[initialSize];
078            lock = buf;
079        }
080    
081        /**
082         * Closes this writer. The implementation in CharArrayWriter does nothing.
083         * 
084         * @since Android 1.0
085         */
086        @Override
087        public void close() {
088            /* empty */
089        }
090    
091        private void expand(int i) {
092            /* Can the buffer handle @i more chars, if not expand it */
093            if (count + i <= buf.length) {
094                return;
095            }
096    
097            char[] newbuf = new char[buf.length + (2 * i)];
098            System.arraycopy(buf, 0, newbuf, 0, count);
099            buf = newbuf;
100        }
101    
102        /**
103         * Flushes this writer. The implementation in CharArrayWriter does nothing.
104         * 
105         * @since Android 1.0
106         */
107        @Override
108        public void flush() {
109            /* empty */
110        }
111    
112        /**
113         * Resets this writer. The current write position is reset to the beginning
114         * of the buffer. All written characters are lost and the size of this
115         * writer is set to 0.
116         * 
117         * @since Android 1.0
118         */
119        public void reset() {
120            synchronized (lock) {
121                count = 0;
122            }
123        }
124    
125        /**
126         * Returns the size of this writer, that is the number of characters it
127         * stores. This number changes if this writer is reset or when more
128         * characters are written to it.
129         * 
130         * @return this CharArrayWriter's current size in characters.
131         * @since Android 1.0
132         */
133        public int size() {
134            synchronized (lock) {
135                return count;
136            }
137        }
138    
139        /**
140         * Returns the contents of the receiver as a char array. The array returned
141         * is a copy and any modifications made to this writer after calling this
142         * method are not reflected in the result.
143         * 
144         * @return this CharArrayWriter's contents as a new char array.
145         * @since Android 1.0
146         */
147        public char[] toCharArray() {
148            synchronized (lock) {
149                char[] result = new char[count];
150                System.arraycopy(buf, 0, result, 0, count);
151                return result;
152            }
153        }
154    
155        /**
156         * Returns the contents of this CharArrayWriter as a string. The string
157         * returned is a copy and any modifications made to this writer after
158         * calling this method are not reflected in the result.
159         * 
160         * @return this CharArrayWriters contents as a new string.
161         * @since Android 1.0
162         */
163        @Override
164        public String toString() {
165            synchronized (lock) {
166                return new String(buf, 0, count);
167            }
168        }
169    
170        /**
171         * Writes {@code count} characters starting at {@code offset} in {@code c}
172         * to this writer.
173         * 
174         * @param c
175         *            the non-null array containing characters to write.
176         * @param offset
177         *            the index of the first character in {@code buf} to write.
178         * @param len
179         *            maximum number of characters to write.
180         * @throws IndexOutOfBoundsException
181         *             if {@code offset < 0} or {@code len < 0}, or if
182         *             {@code offset + len} is bigger than the size of {@code c}.
183         * @since Android 1.0
184         */
185        @Override
186        public void write(char[] c, int offset, int len) {
187            // avoid int overflow
188            // BEGIN android-changed
189            // Exception priorities (in case of multiple errors) differ from
190            // RI, but are spec-compliant.
191            // made implicit null check explicit,
192            // removed redundant check,
193            // added null check, used (offset | len) < 0 instead of
194            // (offset < 0) || (len < 0) to safe one operation
195            if (c == null) {
196                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
197            }
198            if ((offset | len) < 0 || len > c.length - offset) {
199                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
200            }
201            // END android-changed
202            synchronized (lock) {
203                expand(len);
204                System.arraycopy(c, offset, this.buf, this.count, len);
205                this.count += len;
206            }
207        }
208    
209        /**
210         * Writes the specified character {@code oneChar} to this writer.
211         * This implementation writes the two low order bytes of the integer
212         * {@code oneChar} to the buffer.
213         * 
214         * @param oneChar
215         *            the character to write.
216         * @since Android 1.0
217         */
218        @Override
219        public void write(int oneChar) {
220            synchronized (lock) {
221                expand(1);
222                buf[count++] = (char) oneChar;
223            }
224        }
225    
226        /**
227         * Writes {@code count} number of characters starting at {@code offset} from
228         * the string {@code str} to this CharArrayWriter.
229         * 
230         * @param str
231         *            the non-null string containing the characters to write.
232         * @param offset
233         *            the index of the first character in {@code str} to write.
234         * @param len
235         *            the number of characters to retrieve and write.
236         * @throws NullPointerException
237         *             if {@code str} is null.
238         * @throws StringIndexOutOfBoundsException
239         *             if {@code offset < 0} or {@code len < 0}, or if
240         *             {@code offset + len} is bigger than the length of
241         *             {@code str}.
242         * @since Android 1.0
243         */
244        @Override
245        public void write(String str, int offset, int len) {
246            if (str == null) {
247                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
248            }
249            // avoid int overflow
250            // BEGIN android-changed
251            // Exception priorities (in case of multiple errors) differ from
252            // RI, but are spec-compliant.
253            // removed redundant check, used (offset | len) < 0
254            // instead of (offset < 0) || (len < 0) to safe one operation
255            if ((offset | len) < 0 || len > str.length() - offset) {
256                throw new StringIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
257            }
258            // END android-changed
259            synchronized (lock) {
260                expand(len);
261                str.getChars(offset, offset + len, buf, this.count);
262                this.count += len;
263            }
264        }
265    
266        /**
267         * Writes the contents of this CharArrayWriter to another Writer. The output
268         * is all the characters that have been written to the receiver since the
269         * last reset or since it was created.
270         * 
271         * @param out
272         *            the non-null Writer on which to write the contents.
273         * @throws NullPointerException
274         *             if {@code out} is null.
275         * @throws IOException
276         *             if an error occurs attempting to write out the contents.
277         * @since Android 1.0
278         */
279        public void writeTo(Writer out) throws IOException {
280            synchronized (lock) {
281                out.write(buf, 0, count);
282            }
283        }
284    
285        /**
286         * Appends a char {@code c} to the CharArrayWriter. The method works the
287         * same way as {@code write(c)}.
288         * 
289         * @param c
290         *            the character appended to the CharArrayWriter.
291         * @return this CharArrayWriter.
292         * @since Android 1.0
293         */
294        @Override
295        public CharArrayWriter append(char c) {
296            write(c);
297            return this;
298        }
299    
300        /**
301         * Appends a CharSequence {@code csq} to the CharArrayWriter. The method
302         * works the same way as {@code write(csq.toString())}. If {@code csq} is
303         * null, then it will be substituted with the string "null".
304         * 
305         * @param csq
306         *            the CharSequence appended to the CharArrayWriter, may be null.
307         * @return this CharArrayWriter.
308         * @since Android 1.0
309         */
310        @Override
311        public CharArrayWriter append(CharSequence csq) {
312            if (null == csq) {
313                append(TOKEN_NULL, 0, TOKEN_NULL.length());
314            } else {
315                append(csq, 0, csq.length());
316            }
317            return this;
318        }
319    
320        /**
321         * Append a subsequence of a CharSequence {@code csq} to the
322         * CharArrayWriter. The first and last characters of the subsequence are
323         * specified by the parameters {@code start} and {@code end}. The
324         * CharArrayWriter.append({@code csq}) works the same way as
325         * {@code CharArrayWriter.write(csq.subSequence(start, end).toString)}. If
326         * {@code csq} is null, then it will be substituted with the string "null".
327         * 
328         * @param csq
329         *            the CharSequence appended to the CharArrayWriter, may be null.
330         * @param start
331         *            the index of the first character in the CharSequence appended
332         *            to the CharArrayWriter.
333         * @param end
334         *            the index of the character after the last one in the
335         *            CharSequence appended to the CharArrayWriter.
336         * @return this CharArrayWriter.
337         * @throws IndexOutOfBoundsException
338         *             if {@code start < 0}, {@code end < 0}, {@code start > end},
339         *             or if {@code end} is greater than the length of {@code csq}.
340         * @since Android 1.0
341         */
342        @Override
343        public CharArrayWriter append(CharSequence csq, int start, int end) {
344            if (null == csq) {
345                csq = TOKEN_NULL;
346            }
347            String output = csq.subSequence(start, end).toString();
348            write(output, 0, output.length());
349            return this;
350        }
351    }
