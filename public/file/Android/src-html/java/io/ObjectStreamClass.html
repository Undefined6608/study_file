185        /**
186         * Adds an extra entry mapping a given class {@code cl} to its class
187         * descriptor, which will be computed (an ObjectStreamClass). If
188         * {@code computeSUID} is true, this method will compute the SUID for
189         * this class.
190         * 
191         * @param cl
192         *            a java.langClass for which to compute the corresponding
193         *            descriptor
194         * @param computeSUID
195         *            a boolean indicating if SUID should be computed or not.
196         * @return the computer class descriptor
197         */
198        private static ObjectStreamClass addToCache(Class<?> cl, boolean computeSUID) {
199    
200            ObjectStreamClass result = new ObjectStreamClass();
201    
202            // Now we fill in the values
203            result.setName(cl.getName());
204            result.setClass(cl);
205            Class<?> superclass = cl.getSuperclass();
206            if (superclass != null) {
207                result.setSuperclass(lookup(superclass));
208            }
209    
210            Field[] declaredFields = null;
211            if (computeSUID) {
212                // Lazy computation, to save speed & space
213                declaredFields = cl.getDeclaredFields();
214                result.setSerialVersionUID((cl.isEnum() || (cl == Enum.class)) ? 0
215                        : computeSerialVersionUID(cl, declaredFields));
216            }
217    
218            boolean serializable = isSerializable(cl);
219            // Serializables need field descriptors
220            if (serializable && !cl.isArray()) {
221                if (declaredFields == null) {
222    
223                    declaredFields = cl.getDeclaredFields();
224                }
225                result.buildFieldDescriptors(declaredFields);
226            } else {
227                // Externalizables or arrays do not need FieldDesc info
228                result.setFields(new ObjectStreamField[0]);
229            }
230    
231            byte flags = 0;
232            boolean externalizable = isExternalizable(cl);
233            if (externalizable) {
234                flags |= ObjectStreamConstants.SC_EXTERNALIZABLE;
235            } else if (serializable) {
236                flags |= ObjectStreamConstants.SC_SERIALIZABLE;
237            }
238            if (getPrivateWriteObjectMethod(cl) != null) {
239                flags |= ObjectStreamConstants.SC_WRITE_METHOD;
240            }
241            result.setFlags(flags);
242            classesAndDescriptors.put(cl, result);
243    
244            return result;
245        }

247        /**
248         * Builds the collection of field descriptors for the receiver
249         * 
250         * @param declaredFields
251         *            collection of java.lang.reflect.Field for which to compute
252         *            field descriptors
253         */
254        void buildFieldDescriptors(Field[] declaredFields) {
255            // We could find the field ourselves in the collection, but calling
256            // reflect is easier. Optimize if needed.
257            final Field f = ObjectStreamClass.fieldSerialPersistentFields(this
258                    .forClass());
259            // If we could not find the emulated fields, we'll have to compute
260            // dumpable fields from reflect fields
261            boolean useReflectFields = f == null; // Assume we will compute the
262            // fields to dump based on the
263            // reflect fields
264    
265            ObjectStreamField[] _fields = null;
266            if (!useReflectFields) {
267                // The user declared a collection of emulated fields. Use them.
268                // We have to be able to fetch its value, even if it is private
269                AccessController.doPrivileged(new PriviAction<Object>(f));
270                try {
271                    // static field, pass null
272                    _fields = (ObjectStreamField[]) f.get(null);
273                } catch (IllegalAccessException ex) {
274                    // WARNING - what should we do if we have no access ? This
275                    // should not happen.
276                    throw new RuntimeException(ex);
277                }
278            } else {
279                // Compute collection of dumpable fields based on reflect fields
280                List<ObjectStreamField> serializableFields = new ArrayList<ObjectStreamField>(
281                        declaredFields.length);
282                // Filter, we are only interested in fields that are serializable
283                for (int i = 0; i < declaredFields.length; i++) {
284                    Field declaredField = declaredFields[i];
285                    int modifiers = declaredField.getModifiers();
286                    boolean shouldBeSerialized = !(Modifier.isStatic(modifiers) || Modifier
287                            .isTransient(modifiers));
288                    if (shouldBeSerialized) {
289                        ObjectStreamField field = new ObjectStreamField(
290                                declaredField.getName(), declaredField.getType());
291                        serializableFields.add(field);
292                    }
293                }
294    
295                if (serializableFields.size() == 0) {
296                    _fields = NO_FIELDS; // If no serializable fields, share the
297                    // special value so that users can test
298                } else {
299                    // Now convert from Vector to array
300                    _fields = new ObjectStreamField[serializableFields.size()];
301                    _fields = serializableFields.toArray(_fields);
302                }
303            }
304            ObjectStreamField.sortFields(_fields);
305            // assign offsets
306            int primOffset = 0, objectOffset = 0;
307            for (int i = 0; i < _fields.length; i++) {
308                Class<?> type = _fields[i].getType();
309                if (type.isPrimitive()) {
310                    _fields[i].offset = primOffset;
311                    primOffset += primitiveSize(type);
312                } else {
313                    _fields[i].offset = objectOffset++;
314                }
315            }
316            fields = _fields;
317        }

319        /**
320         * Compute and return the Serial Version UID of the class {@code cl}.
321         * The value is computed based on the class name, superclass chain, field
322         * names, method names, modifiers, etc.
323         * 
324         * @param cl
325         *            a java.lang.Class for which to compute the SUID
326         * @param fields
327         *            cl.getDeclaredFields(), pre-computed by the caller
328         * @return the value of SUID of this class
329         */
330        private static long computeSerialVersionUID(Class<?> cl, Field[] fields) {
331            /*
332             * First we should try to fetch the static slot 'static final long
333             * serialVersionUID'. If it is defined, return it. If not defined, we
334             * really need to compute SUID using SHAOutputStream
335             */
336            for (int i = 0; i < fields.length; i++) {
337                final Field field = fields[i];
338                if (Long.TYPE == field.getType()) {
339                    int modifiers = field.getModifiers();
340                    if (Modifier.isStatic(modifiers) && Modifier.isFinal(modifiers)) {
341                        if (UID_FIELD_NAME.equals(field.getName())) {
342                            /*
343                             * We need to be able to see it even if we have no
344                             * visibility. That is why we set accessible first (new
345                             * API in reflect 1.2)
346                             */
347                            AccessController.doPrivileged(new PriviAction<Object>(
348                                    field));
349                            try {
350                                // Static field, parameter is ignored
351                                return field.getLong(null);
352                            } catch (IllegalAccessException iae) {
353                                throw new RuntimeException(Msg.getString(
354                                        "K0071", iae)); //$NON-NLS-1$
355                            }
356                        }
357                    }
358                }
359            }
360    
361            MessageDigest digest;
362            try {
363                digest = MessageDigest.getInstance("SHA"); //$NON-NLS-1$
364            } catch (NoSuchAlgorithmException e) {
365                throw new Error(e);
366            }
367            ByteArrayOutputStream sha = new ByteArrayOutputStream();
368            try {
369                DataOutputStream output = new DataOutputStream(sha);
370                output.writeUTF(cl.getName());
371                int classModifiers = CLASS_MODIFIERS_MASK & cl.getModifiers();
372                /*
373                 * Workaround for 1F9LOQO. Arrays are ABSTRACT in JDK, but that is
374                 * not in the specification. Since we want to be compatible for
375                 * X-loading, we have to pretend we have the same shape
376                 */
377                boolean isArray = cl.isArray();
378                if (isArray) {
379                    classModifiers |= Modifier.ABSTRACT;
380                }
381                // Required for JDK UID compatibility
382                if (cl.isInterface() && !Modifier.isPublic(classModifiers)) {
383                    classModifiers &= ~Modifier.ABSTRACT;
384                }
385                output.writeInt(classModifiers);
386    
387                /*
388                 * In JDK1.2 arrays implement Cloneable and Serializable but not in
389                 * JDK 1.1.7. So, JDK 1.2 "pretends" arrays have no interfaces when
390                 * computing SHA-1 to be compatible.
391                 */
392                if (!isArray) {
393                    // Interface information
394                    Class<?>[] interfaces = cl.getInterfaces();
395                    if (interfaces.length > 1) {
396                        // Only attempt to sort if really needed (saves object
397                        // creation, etc)
398                        Comparator<Class<?>> interfaceComparator = new Comparator<Class<?>>() {
399                            public int compare(Class<?> itf1, Class<?> itf2) {
400                                return itf1.getName().compareTo(itf2.getName());
401                            }
402                        };
403                        Arrays.sort(interfaces, interfaceComparator);
404                    }
405    
406                    // Dump them
407                    for (int i = 0; i < interfaces.length; i++) {
408                        output.writeUTF(interfaces[i].getName());
409                    }
410                }
411    
412                // Field information
413                if (fields.length > 1) {
414                    // Only attempt to sort if really needed (saves object creation,
415                    // etc)
416                    Comparator<Field> fieldComparator = new Comparator<Field>() {
417                        public int compare(Field field1, Field field2) {
418                            return field1.getName().compareTo(field2.getName());
419                        }
420                    };
421                    Arrays.sort(fields, fieldComparator);
422                }
423    
424                // Dump them
425                for (int i = 0; i < fields.length; i++) {
426                    Field field = fields[i];
427                    int modifiers = field.getModifiers() & FIELD_MODIFIERS_MASK;
428    
429                    boolean skip = Modifier.isPrivate(modifiers)
430                            && (Modifier.isTransient(modifiers) || Modifier
431                                    .isStatic(modifiers));
432                    if (!skip) {
433                        // write name, modifier & "descriptor" of all but private
434                        // static and private transient
435                        output.writeUTF(field.getName());
436                        output.writeInt(modifiers);
437                        output
438                                .writeUTF(descriptorForFieldSignature(getFieldSignature(field)));
439                    }
440                }
441    
442                /*
443                 * Normally constructors come before methods (because <init> <
444                 * anyMethodName). However, <clinit> is an exception. Besides,
445                 * reflect will not let us get to it.
446                 */
447                if (hasClinit(cl)) {
448                    // write name, modifier & "descriptor"
449                    output.writeUTF(CLINIT_NAME);
450                    output.writeInt(CLINIT_MODIFIERS);
451                    output.writeUTF(CLINIT_SIGNATURE);
452                }
453    
454                // Constructor information
455                Constructor<?>[] constructors = cl.getDeclaredConstructors();
456                if (constructors.length > 1) {
457                    // Only attempt to sort if really needed (saves object creation,
458                    // etc)
459                    Comparator<Constructor<?>> constructorComparator = new Comparator<Constructor<?>>() {
460                        public int compare(Constructor<?> ctr1, Constructor<?> ctr2) {
461                            // All constructors have same name, so we sort based on
462                            // signature
463                            return (getConstructorSignature(ctr1)
464                                    .compareTo(getConstructorSignature(ctr2)));
465                        }
466                    };
467                    Arrays.sort(constructors, constructorComparator);
468                }
469    
470                // Dump them
471                for (int i = 0; i < constructors.length; i++) {
472                    Constructor<?> constructor = constructors[i];
473                    int modifiers = constructor.getModifiers()
474                            & METHOD_MODIFIERS_MASK;
475                    boolean isPrivate = Modifier.isPrivate(modifiers);
476                    if (!isPrivate) {
477                        /*
478                         * write name, modifier & "descriptor" of all but private
479                         * ones
480                         * 
481                         * constructor.getName() returns the constructor name as
482                         * typed, not the VM name
483                         */
484                        output.writeUTF("<init>"); //$NON-NLS-1$
485                        output.writeInt(modifiers);
486                        output.writeUTF(descriptorForSignature(
487                                getConstructorSignature(constructor)).replace('/',
488                                '.'));
489                    }
490                }
491    
492                // Method information
493                Method[] methods = cl.getDeclaredMethods();
494                if (methods.length > 1) {
495                    Comparator<Method> methodComparator = new Comparator<Method>() {
496                        public int compare(Method m1, Method m2) {
497                            int result = m1.getName().compareTo(m2.getName());
498                            if (result == 0) {
499                                // same name, signature will tell which one comes
500                                // first
501                                return getMethodSignature(m1).compareTo(
502                                        getMethodSignature(m2));
503                            }
504                            return result;
505                        }
506                    };
507                    Arrays.sort(methods, methodComparator);
508                }
509    
510                // Dump them
511                for (int i = 0; i < methods.length; i++) {
512                    Method method = methods[i];
513                    int modifiers = method.getModifiers() & METHOD_MODIFIERS_MASK;
514                    boolean isPrivate = Modifier.isPrivate(modifiers);
515                    if (!isPrivate) {
516                        // write name, modifier & "descriptor" of all but private
517                        // ones
518                        output.writeUTF(method.getName());
519                        output.writeInt(modifiers);
520                        output.writeUTF(descriptorForSignature(
521                                getMethodSignature(method)).replace('/', '.'));
522                    }
523                }
524            } catch (IOException e) {
525                throw new RuntimeException(Msg.getString("K0072", e));//$NON-NLS-1$
526            }
527    
528            // now compute the UID based on the SHA
529            byte[] hash = digest.digest(sha.toByteArray());
530    
531            return littleEndianLongAt(hash, 0);
532        }

558        /**
559         * Return the java.lang.reflect.Field {@code serialPersistentFields}
560         * if class {@code cl} implements it. Return null otherwise.
561         * 
562         * @param cl
563         *            a java.lang.Class which to test
564         * @return {@code java.lang.reflect.Field} if the class has
565         *         serialPersistentFields {@code null} if the class does not
566         *         have serialPersistentFields
567         */
568        static Field fieldSerialPersistentFields(Class<?> cl) {
569            try {
570                Field f = cl.getDeclaredField("serialPersistentFields"); //$NON-NLS-1$
571                int modifiers = f.getModifiers();
572                if (Modifier.isStatic(modifiers) && Modifier.isPrivate(modifiers)
573                        && Modifier.isFinal(modifiers)) {
574                    if (f.getType() == ARRAY_OF_FIELDS) {
575                        return f;
576                    }
577                }
578            } catch (NoSuchFieldException nsm) {
579                // Ignored
580            }
581            return null;
582        }

609         * Gets a field descriptor of the class represented by this class
610         * descriptor.
611         * 
612         * @param name
613         *            the name of the desired field.
614         * @return the field identified by {@code name} or {@code null} if there is

616         * @since Android 1.0

618        public ObjectStreamField getField(String name) {
619            ObjectStreamField[] allFields = getFields();
620            for (int i = 0; i < allFields.length; i++) {
621                ObjectStreamField f = allFields[i];
622                if (f.getName().equals(name)) {
623                    return f;
624                }
625            }
626            return null;
627        }

629        /**
630         * Returns the collection of field descriptors for the fields of the
631         * corresponding class
632         * 
633         * @return the receiver's collection of declared fields for the class it
634         *         represents
635         */
636        ObjectStreamField[] fields() {
637            if (fields == null) {
638                synchronized(this){ 
639                    Class<?> forCl = forClass();
640                    if (forCl != null && isSerializable(forCl) && !forCl.isArray()) {
641                        buildFieldDescriptors(forCl.getDeclaredFields());
642                    } else {
643                        // Externalizables or arrays do not need FieldDesc info
644                        setFields(new ObjectStreamField[0]);
645                    }
646                }
647            }
648            return fields;
649        }

652         * Returns a collection of field descriptors for the serialized fields of
653         * the class represented by this class descriptor.
654         * 
655         * @return an array of field descriptors or an array of length zero if there
656         *         are no fields in this descriptor's class.
657         * @since Android 1.0

659        public ObjectStreamField[] getFields() {
660            copyFieldAttributes();
661            return loadFields == null ? fields().clone() : loadFields.clone();

664        /**
665         * If a Class uses "serialPersistentFields" to define the serialized fields, 
666         * this.loadFields cannot get the "unshared" information when deserializing
667         * fields using current implementation of ObjectInputStream. This method 
668         * provides a way to copy the "unshared" attribute from this.fields.
669         *
670         */
671        private void copyFieldAttributes() {
672            if ((loadFields == null) || fields == null) {
673                return;
674            }
675            
676            for (int i = 0; i < loadFields.length; i++) {
677                ObjectStreamField loadField = loadFields[i];
678                String name = loadField.getName();
679                for (int j = 0; j < fields.length; j++) {
680                    ObjectStreamField field = fields[j];
681                    if (name.equals(field.getName())) {
682                        loadField.setUnshared(field.isUnshared());
683                        loadField.setOffset(field.getOffset());
684                        break;
685                    }
686                }
687            }
688        }

893        /**
894         * Return a little endian long stored in a given position of the buffer
895         * 
896         * @param buffer
897         *            a byte array with the byte representation of the number
898         * @param position
899         *            index where the number starts in the byte array
900         * @return the number that was stored in little endian format
901         */
902        private static long littleEndianLongAt(byte[] buffer, int position) {
903            long result = 0;
904            for (int i = position + 7; i >= position; i--) {
905                result = (result << 8) + (buffer[i] & 0xff);
906            }
907            return result;
908        }

910        /**
911         * Returns the descriptor corresponding to the class {@code cl}. If the
912         * class is not serializable or externalizable then {@code null} is
913         * returned.
914         * 
915         * @param cl
916         *            a java.langClass for which to obtain the corresponding
917         *            descriptor
918         * @return the corresponding descriptor if the {@code cl} is serializable or
919         *         externalizable; {@code null} otherwise.
920         * @since Android 1.0
921         */
922        public static ObjectStreamClass lookup(Class<?> cl) {
923            boolean serializable = isSerializable(cl);
924            boolean externalizable = isExternalizable(cl);
925    
926            // Has to be either Serializable or Externalizable
927            if (!serializable && !externalizable) {
928                return null;
929            }
930    
931            return lookupStreamClass(cl, true);
932        }

940         *            a java.langClass for which to obtain the corresponding
941         *            descriptor
942         * @return the corresponding descriptor

946        }

948        /**
949         * Return the descriptor (ObjectStreamClass) corresponding to the class
950         * {@code cl}. Returns an ObjectStreamClass even if instances of the
951         * class cannot be serialized
952         * 
953         * @param cl
954         *            a {@code java.langClass} for which to obtain the
955         *            corresponding descriptor
956         * @param computeSUID
957         *            a boolean indicating if SUID should be computed or not.
958         * @return the corresponding descriptor
959         */
960        private static synchronized ObjectStreamClass lookupStreamClass(
961                Class<?> cl, boolean computeSUID) {
962            // Synchronized because of the lookup table 'classesAndDescriptors'
963            ObjectStreamClass cachedValue = classesAndDescriptors.get(cl);
964            if (cachedValue != null) {
965                return cachedValue;
966            }
967            return addToCache(cl, computeSUID);
968        }

970        /**
971         * Return the java.lang.reflect.Method {@code readResolve} if class
972         * {@code cl} implements it. Return null otherwise.
973         * 
974         * @param cl
975         *            a java.lang.Class which to test
976         * @return {@code java.lang.reflect.Method} if the class implements
977         *         readResolve {@code null} if the class does not implement
978         *         readResolve
979         */
980        static Method methodReadResolve(Class<?> cl) {
981            Class<?> search = cl;
982            while (search != null) {
983                try {
984                    Method method = search.getDeclaredMethod(
985                            "readResolve", (Class[]) null); //$NON-NLS-1$
986                    if (search == cl
987                            || (method.getModifiers() & Modifier.PRIVATE) == 0) {
988                        return method;
989                    }
990                    return null;
991                } catch (NoSuchMethodException nsm) {
992                }
993                search = search.getSuperclass();
994            }
995            return null;
996        }

1108        private int primitiveSize(Class<?> type) {
1109            if (type == Byte.TYPE || type == Boolean.TYPE) {
1110                return 1;
1111            }
1112            if (type == Short.TYPE || type == Character.TYPE) {
1113                return 2;
1114            }
1115            if (type == Integer.TYPE || type == Float.TYPE) {
1116                return 4;
1117            }
1118            if (type == Long.TYPE || type == Double.TYPE) {
1119                return 8;
1120            }
1121            return 0;
1122        }
