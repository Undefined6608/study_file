018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * A specialized {@link Reader} that reads characters from a {@code String} in
024     * a sequential manner.
025     * 
026     * @see StringWriter
027     * 
028     * @since Android 1.0
029     */
030    public class StringReader extends Reader {
031        private String str;
032    
033        private int markpos = -1;
034    
035        private int pos;
036    
037        private int count;
038    
039        /**
040         * Construct a new {@code StringReader} with {@code str} as source. The size
041         * of the reader is set to the {@code length()} of the string and the Object
042         * to synchronize access through is set to {@code str}.
043         * 
044         * @param str
045         *            the source string for this reader.
046         * @since Android 1.0
047         */
048        public StringReader(String str) {
049            super(str);
050            this.str = str;
051            this.count = str.length();
052        }
053    
054        /**
055         * Closes this reader. Once it is closed, read operations on this reader
056         * will throw an {@code IOException}. Only the first invocation of this
057         * method has any effect.
058         * 
059         * @since Android 1.0
060         */
061        @Override
062        public void close() {
063            synchronized (lock) {
064                if (isClosed()) {
065                    return;
066                }
067                str = null;
068            }
069        }
070    
071        /**
072         * Returns a boolean indicating whether this reader is closed.
073         * 
074         * @return {@code true} if closed, otherwise {@code false}.
075         */
076        private boolean isClosed() {
077            return str == null;
078        }
079    
080        /**
081         * Sets a mark position in this reader. The parameter {@code readLimit} is
082         * ignored for this class. Calling {@code reset()} will reposition the
083         * reader back to the marked position.
084         * 
085         * @param readLimit
086         *            ignored for {@code StringReader} instances.
087         * @throws IllegalArgumentException
088         *             if {@code readLimit < 0}.
089         * @throws IOException
090         *             if this reader is closed.
091         * @see #markSupported()
092         * @see #reset()
093         * @since Android 1.0
094         */
095        @Override
096        public void mark(int readLimit) throws IOException {
097            if (readLimit < 0) {
098                throw new IllegalArgumentException();
099            }
100    
101            synchronized (lock) {
102                if (isClosed()) {
103                    throw new IOException(Msg.getString("K0083")); //$NON-NLS-1$
104                }
105                markpos = pos;
106            }
107        }
108    
109        /**
110         * Indicates whether this reader supports the {@code mark()} and {@code
111         * reset()} methods. This implementation returns {@code true}.
112         * 
113         * @return always {@code true}.
114         * @since Android 1.0
115         */
116        @Override
117        public boolean markSupported() {
118            return true;
119        }
120    
121        /**
122         * Reads a single character from the source string and returns it as an
123         * integer with the two higher-order bytes set to 0. Returns -1 if the end
124         * of the source string has been reached.
125         * 
126         * @return the character read or -1 if the end of the source string has been
127         *         reached.
128         * @throws IOException
129         *             if this reader is closed.
130         * @since Android 1.0
131         */
132        @Override
133        public int read() throws IOException {
134            synchronized (lock) {
135                if (isClosed()) {
136                    throw new IOException(Msg.getString("K0083")); //$NON-NLS-1$
137                }
138                if (pos != count) {
139                    return str.charAt(pos++);
140                }
141                return -1;
142            }
143        }
144    
145        /**
146         * Reads at most {@code len} characters from the source string and stores
147         * them at {@code offset} in the character array {@code buf}. Returns the
148         * number of characters actually read or -1 if the end of the source string
149         * has been reached.
150         * 
151         * @param buf
152         *            the character array to store the characters read.
153         * @param offset
154         *            the initial position in {@code buffer} to store the characters
155         *            read from this reader.
156         * @param len
157         *            the maximum number of characters to read.
158         * @return the number of characters read or -1 if the end of the reader has
159         *         been reached.
160         * @throws IndexOutOfBoundsException
161         *             if {@code offset < 0} or {@code len < 0}, or if
162         *             {@code offset + len} is greater than the size of {@code buf}.
163         * @throws IOException
164         *             if this reader is closed.
165         * @since Android 1.0
166         */
167        @Override
168        public int read(char[] buf, int offset, int len) throws IOException {
169            // BEGIN android-note
170            // changed array notation to be consistent with the rest of harmony
171            // END android-note
172            // avoid int overflow
173            // BEGIN android-changed
174            // Exception priorities (in case of multiple errors) differ from
175            // RI, but are spec-compliant.
176            // removed redundant check, added null check, used (offset | len) < 0
177            // instead of (offset < 0) || (len < 0) to safe one operation
178            if (buf == null) {
179                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
180            }
181            if ((offset | len) < 0 || len > buf.length - offset) {
182                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
183            }
184            // END android-changed
185            synchronized (lock) {
186                if (isClosed()) {
187                    throw new IOException(Msg.getString("K0083")); //$NON-NLS-1$
188                }
189                if (pos == this.count) {
190                    return -1;
191                }
192                int end = pos + len > this.count ? this.count : pos + len;
193                str.getChars(pos, end, buf, offset);
194                int read = end - pos;
195                pos = end;
196                return read;
197            }
198        }
199    
200        /**
201         * Indicates whether this reader is ready to be read without blocking. This
202         * implementation always returns {@code true}.
203         * 
204         * @return always {@code true}.
205         * @throws IOException
206         *             if this reader is closed.
207         * @see #read()
208         * @see #read(char[], int, int)
209         * @since Android 1.0
210         */
211        @Override
212        public boolean ready() throws IOException {
213            synchronized (lock) {
214                if (isClosed()) {
215                    throw new IOException(Msg.getString("K0083")); //$NON-NLS-1$
216                }
217                return true;
218            }
219        }
220    
221        /**
222         * Resets this reader's position to the last {@code mark()} location.
223         * Invocations of {@code read()} and {@code skip()} will occur from this new
224         * location. If this reader has not been marked, it is reset to the
225         * beginning of the source string.
226         * 
227         * @throws IOException
228         *             if this reader is closed.
229         * @see #mark(int)
230         * @see #markSupported()
231         * @since Android 1.0
232         */
233        @Override
234        public void reset() throws IOException {
235            synchronized (lock) {
236                if (isClosed()) {
237                    throw new IOException(Msg.getString("K0083")); //$NON-NLS-1$
238                }
239                pos = markpos != -1 ? markpos : 0;
240            }
241        }
242    
243        /**
244         * Skips {@code amount} characters in the source string. Subsequent calls of
245         * {@code read} methods will not return these characters unless {@code
246         * reset()} is used.
247         * 
248         * @param ns
249         *            the maximum number of characters to skip.
250         * @return the number of characters actually skipped or 0 if {@code ns < 0}.
251         * @throws IOException
252         *             if this reader is closed.
253         * @see #mark(int)
254         * @see #markSupported()
255         * @see #reset()
256         * @since Android 1.0
257         */
258        @Override
259        public long skip(long ns) throws IOException {
260            synchronized (lock) {
261                if (isClosed()) {
262                    throw new IOException(Msg.getString("K0083")); //$NON-NLS-1$
263                }
264                if (ns <= 0) {
265                    return 0;
266                }
267                long skipped = 0;
268                if (ns < this.count - pos) {
269                    pos = pos + (int) ns;
270                    skipped = ns;
271                } else {
272                    skipped = this.count - pos;
273                    pos = this.count;
274                }
275                return skipped;
276            }
277        }
278    }
