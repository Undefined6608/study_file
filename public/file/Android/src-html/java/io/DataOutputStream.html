018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * Wraps an existing {@link OutputStream} and writes typed data to it.
024     * Typically, this stream can be read in by DataInputStream. Types that can be
025     * written include byte, 16-bit short, 32-bit int, 32-bit float, 64-bit long,
026     * 64-bit double, byte strings, and {@link DataInput MUTF-8} encoded strings.
027     * 
028     * @see DataInputStream
029     * 
030     * @since Android 1.0
031     */
032    public class DataOutputStream extends FilterOutputStream implements DataOutput {
033    
034        /**
035         * The number of bytes written out so far.
036         * 
037         * @since Android 1.0
038         */
039        protected int written;
040    
041        /**
042         * Constructs a new {@code DataOutputStream} on the {@code OutputStream}
043         * {@code out}. Note that data written by this stream is not in a human
044         * readable form but can be reconstructed by using a {@link DataInputStream}
045         * on the resulting output.
046         * 
047         * @param out
048         *            the target stream for writing.
049         * @since Android 1.0
050         */
051        public DataOutputStream(OutputStream out) {
052            super(out);
053        }
054    
055        /**
056         * Flushes this stream to ensure all pending data is sent out to the target
057         * stream. This implementation then also flushes the target stream.
058         * 
059         * @throws IOException
060         *             if an error occurs attempting to flush this stream.
061         * @since Android 1.0
062         */
063        @Override
064        public void flush() throws IOException {
065            super.flush();
066        }
067    
068        /**
069         * Returns the total number of bytes written to the target stream so far.
070         * 
071         * @return the number of bytes written to the target stream.
072         * @since Android 1.0
073         */
074        public final int size() {
075            if (written < 0) {
076                written = Integer.MAX_VALUE;
077            }
078            return written;
079        }
080    
081        /**
082         * Writes {@code count} bytes from the byte array {@code buffer} starting at
083         * {@code offset} to the target stream.
084         * 
085         * @param buffer
086         *            the buffer to write to the target stream.
087         * @param offset
088         *            the index of the first byte in {@code buffer} to write.
089         * @param count
090         *            the number of bytes from the {@code buffer} to write.
091         * @throws IOException
092         *             if an error occurs while writing to the target stream.
093         * @throws NullPointerException
094         *             if {@code buffer} is {@code null}.
095         * @see DataInputStream#readFully(byte[])
096         * @see DataInputStream#readFully(byte[], int, int)
097         * @since Android 1.0
098         */
099        @Override
100        public void write(byte[] buffer, int offset, int count) throws IOException {
101            // BEGIN android-note
102            // changed array notation to be consistent with the rest of harmony
103            // END android-note
104            if (buffer == null) {
105                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
106            }
107            out.write(buffer, offset, count);
108            written += count;
109        }
110    
111        /**
112         * Writes a byte to the target stream. Only the least significant byte of
113         * the integer {@code oneByte} is written.
114         * 
115         * @param oneByte
116         *            the byte to write to the target stream.
117         * @throws IOException
118         *             if an error occurs while writing to the target stream.
119         * @see DataInputStream#readByte()
120         * @since Android 1.0
121         */
122        @Override
123        public void write(int oneByte) throws IOException {
124            out.write(oneByte);
125            written++;
126        }
127    
128        /**
129         * Writes a boolean to the target stream.
130         * 
131         * @param val
132         *            the boolean value to write to the target stream.
133         * @throws IOException
134         *             if an error occurs while writing to the target stream.
135         * @see DataInputStream#readBoolean()
136         * @since Android 1.0
137         */
138        public final void writeBoolean(boolean val) throws IOException {
139            out.write(val ? 1 : 0);
140            written++;
141        }
142    
143        /**
144         * Writes an 8-bit byte to the target stream. Only the least significant
145         * byte of the integer {@code val} is written.
146         * 
147         * @param val
148         *            the byte value to write to the target stream.
149         * @throws IOException
150         *             if an error occurs while writing to the target stream.
151         * @see DataInputStream#readByte()
152         * @see DataInputStream#readUnsignedByte()
153         * @since Android 1.0
154         */
155        public final void writeByte(int val) throws IOException {
156            out.write(val);
157            written++;
158        }
159    
160        /**
161         * Writes the low order bytes from a string to the target stream.
162         * 
163         * @param str
164         *            the string containing the bytes to write to the target stream.
165         * @throws IOException
166         *             if an error occurs while writing to the target stream.
167         * @see DataInputStream#readFully(byte[])
168         * @see DataInputStream#readFully(byte[],int,int)
169         * @since Android 1.0
170         */
171        public final void writeBytes(String str) throws IOException {
172            if (str.length() == 0) {
173                return;
174            }
175            byte bytes[] = new byte[str.length()];
176            for (int index = 0; index < str.length(); index++) {
177                bytes[index] = (byte) str.charAt(index);
178            }
179            out.write(bytes);
180            written += bytes.length;
181        }
182    
183        /**
184         * Writes a 16-bit character to the target stream. Only the two lower bytes
185         * of the integer {@code val} are written, with the higher one written
186         * first. This corresponds to the Unicode value of {@code val}.
187         * 
188         * @param val
189         *            the character to write to the target stream
190         * @throws IOException
191         *             if an error occurs while writing to the target stream.
192         * @see DataInputStream#readChar()
193         * @since Android 1.0
194         */
195        public final void writeChar(int val) throws IOException {
196            out.write(val >> 8);
197            out.write(val);
198            written += 2;
199        }
200    
201        /**
202         * Writes the 16-bit characters contained in {@code str} to the target
203         * stream.
204         * 
205         * @param str
206         *            the string that contains the characters to write to this
207         *            stream.
208         * @throws IOException
209         *             if an error occurs while writing to the target stream.
210         * @see DataInputStream#readChar()
211         * @since Android 1.0
212         */
213        public final void writeChars(String str) throws IOException {
214            byte newBytes[] = new byte[str.length() * 2];
215            for (int index = 0; index < str.length(); index++) {
216                int newIndex = index == 0 ? index : index * 2;
217                newBytes[newIndex] = (byte) (str.charAt(index) >> 8);
218                newBytes[newIndex + 1] = (byte) str.charAt(index);
219            }
220            out.write(newBytes);
221            written += newBytes.length;
222        }
223    
224        /**
225         * Writes a 64-bit double to the target stream. The resulting output is the
226         * eight bytes resulting from calling Double.doubleToLongBits().
227         * 
228         * @param val
229         *            the double to write to the target stream.
230         * @throws IOException
231         *             if an error occurs while writing to the target stream.
232         * @see DataInputStream#readDouble()
233         * @since Android 1.0
234         */
235        public final void writeDouble(double val) throws IOException {
236            writeLong(Double.doubleToLongBits(val));
237        }
238    
239        /**
240         * Writes a 32-bit float to the target stream. The resulting output is the
241         * four bytes resulting from calling Float.floatToIntBits().
242         * 
243         * @param val
244         *            the float to write to the target stream.
245         * @throws IOException
246         *             if an error occurs while writing to the target stream.
247         * @see DataInputStream#readFloat()
248         * @since Android 1.0
249         */
250        public final void writeFloat(float val) throws IOException {
251            writeInt(Float.floatToIntBits(val));
252        }
253    
254        /**
255         * Writes a 32-bit int to the target stream. The resulting output is the
256         * four bytes, highest order first, of {@code val}.
257         * 
258         * @param val
259         *            the int to write to the target stream.
260         * @throws IOException
261         *             if an error occurs while writing to the target stream.
262         * @see DataInputStream#readInt()
263         * @since Android 1.0
264         */
265        public final void writeInt(int val) throws IOException {
266            out.write(val >> 24);
267            out.write(val >> 16);
268            out.write(val >> 8);
269            out.write(val);
270            written += 4;
271        }
272    
273        /**
274         * Writes a 64-bit long to the target stream. The resulting output is the
275         * eight bytes, highest order first, of {@code val}.
276         * 
277         * @param val
278         *            the long to write to the target stream.
279         * @throws IOException
280         *             if an error occurs while writing to the target stream.
281         * @see DataInputStream#readLong()
282         * @since Android 1.0
283         */
284        public final void writeLong(long val) throws IOException {
285            writeInt((int) (val >> 32));
286            writeInt((int) val);
287        }
288    
289        /**
290         * Writes the specified 16-bit short to the target stream. Only the lower
291         * two bytes of the integer {@code val} are written, with the higher one
292         * written first.
293         * 
294         * @param val
295         *            the short to write to the target stream.
296         * @throws IOException
297         *             if an error occurs while writing to the target stream.
298         * @see DataInputStream#readShort()
299         * @see DataInputStream#readUnsignedShort()
300         * @since Android 1.0
301         */
302        public final void writeShort(int val) throws IOException {
303            writeChar(val);
304        }
305        
306        // BEGIN android-added
307        static final int MAX_BUF_SIZE = 8192;
308        // END android-added
309        
310        /**
311         * Writes the specified encoded in {@link DataInput modified UTF-8} to this
312         * stream.
313         * 
314         * @param str
315         *            the string to write to the target stream encoded in
316         *            {@link DataInput modified UTF-8}.
317         * @throws IOException
318         *             if an error occurs while writing to the target stream.
319         * @throws UTFDataFormatException
320         *             if the encoded string is longer than 65535 bytes.
321         * @see DataInputStream#readUTF()
322         * @since Android 1.0
323         */
324        public final void writeUTF(String str) throws IOException {
325            int length = str.length();
326            // BEGIN android-changed
327            if (length <= MAX_BUF_SIZE / 3) {
328                int size = length * 3;
329                byte[] utfBytes = new byte[size];
330                // boolean makeBuf = true;
331                // synchronized (DataInputStream.byteBuf) {
332                //     if (DataInputStream.useShared) {
333                //         DataInputStream.useShared = false;
334                //         makeBuf = false;
335                //     }
336                // }
337                // if (makeBuf) {
338                //     utfBytes = new byte[size];
339                // } else {
340                //     if (DataInputStream.byteBuf.length < size) {
341                //         DataInputStream.byteBuf = new byte[size];
342                //     }
343                //     utfBytes = DataInputStream.byteBuf;
344                // }
345                int utfIndex = 0;
346                for (int i = 0; i < length; i++) {
347                    int charValue = str.charAt(i);
348                    if (charValue > 0 && charValue <= 127) {
349                        utfBytes[utfIndex++] = (byte) charValue;
350                    } else if (charValue <= 2047) {
351                        utfBytes[utfIndex++] = (byte) (0xc0 | (0x1f & (charValue >> 6)));
352                        utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & charValue));
353                    } else {
354                        utfBytes[utfIndex++] = (byte) (0xe0 | (0x0f & (charValue >> 12)));
355                        utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & (charValue >> 6)));
356                        utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & charValue));
357                    }
358                }
359                writeShort(utfIndex);
360                write(utfBytes, 0, utfIndex);
361                // if (!makeBuf) {
362                //     DataInputStream.useShared = true;
363                // }
364            } else {
365                long utfCount;
366                if (length <= 65535 && (utfCount = countUTFBytes(str)) <= 65535) {
367                    writeShort((int) utfCount);
368                    writeUTFBytes(str, utfCount);
369                } else {
370                    throw new UTFDataFormatException(Msg.getString("K0068")); //$NON-NLS-1$
371                }
372            }
373            // END android-changed
374        }
375    
376        long countUTFBytes(String str) {
377            int utfCount = 0, length = str.length();
378            for (int i = 0; i < length; i++) {
379                int charValue = str.charAt(i);
380                if (charValue > 0 && charValue <= 127) {
381                    utfCount++;
382                } else if (charValue <= 2047) {
383                    utfCount += 2;
384                } else {
385                    utfCount += 3;
386                }
387            }
388            return utfCount;
389        }

391        void writeUTFBytes(String str, long count) throws IOException {
392            boolean single = true;
393            int size = (int) count;
394            // BEGIN android-changed
395            if (count > MAX_BUF_SIZE) {
396                single = false;
397                size = MAX_BUF_SIZE;
398            }
399            byte[] utfBytes = new byte[size];
400            // END android-changed
401            // BEGIN android-removed
402            // boolean makeBuf = true;
403            // if (DataInputStream.useShared) {
404            //     synchronized (DataInputStream.cacheLock) {
405            //         if (DataInputStream.useShared) {
406            //             DataInputStream.useShared = false;
407            //             makeBuf = false;
408            //         }
409            //     }
410            // }
411            // if (makeBuf) {
412            //     utfBytes = new byte[size];
413            // } else {
414            //     // byteBuf is not protected by the cacheLock, so sample it first
415            //     utfBytes = DataInputStream.byteBuf;
416            //     if (utfBytes.length < size) {
417            //         utfBytes = DataInputStream.byteBuf = new byte[size];
418            //     }
419            // }
420            // END android-removed
421    
422            int utfIndex = 0, i = 0, length = str.length();
423            int end = length;
424            while (i < length) {
425                if (!single) {
426                    end = i + ((utfBytes.length - utfIndex) / 3);
427                    if (end > length) {
428                        end = length;
429                    }
430                }
431                for (int j = i; j < end; j++) {
432                    int charValue = str.charAt(j);
433                    if (charValue > 0 && charValue <= 127) {
434                        utfBytes[utfIndex++] = (byte) charValue;
435                    } else if (charValue <= 2047) {
436                        utfBytes[utfIndex++] = (byte) (0xc0 | (0x1f & (charValue >> 6)));
437                        utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & charValue));
438                    } else {
439                        utfBytes[utfIndex++] = (byte) (0xe0 | (0x0f & (charValue >> 12)));
440                        utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & (charValue >> 6)));
441                        utfBytes[utfIndex++] = (byte) (0x80 | (0x3f & charValue));
442                    }
443                }
444                if (single || utfIndex > utfBytes.length - 300) {
445                    write(utfBytes, 0, utfIndex);
446                    if (single) {
447                        return;
448                    }
449                    utfIndex = 0;
450                }
451                i = end;
452            }
453            if (utfIndex > 0) {
454                write(utfBytes, 0, utfIndex);
455            }
456            // BEGIN android-removed
457            // if (!makeBuf) {
458            //     // Update the useShared flag optimistically (see DataInputStream
459            //     // equivalent)
460            //     DataInputStream.useShared = true;
461            // }
462            // END android-removed
