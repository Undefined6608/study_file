018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * Receives information from a communications pipe. When two threads want to
024     * pass data back and forth, one creates a piped output stream and the other one
025     * creates a piped input stream.
026     * 
027     * @see PipedOutputStream
028     * 
029     * @since Android 1.0
030     */
031    public class PipedInputStream extends InputStream {
032    
033        private Thread lastReader, lastWriter;
034    
035        private boolean isClosed = false;
036    
037        /**
038         * The circular buffer through which data is passed.
039         * 
040         * @since Android 1.0
041         */
042        protected byte buffer[];
043    
044        /**
045         * The index in {@code buffer} where the next byte will be written.
046         * 
047         * @since Android 1.0
048         */
049        protected int in = -1;
050    
051        /**
052         * The index in {@code buffer} where the next byte will be read.
053         * 
054         * @since Android 1.0
055         */
056        protected int out = 0;
057    
058        /**
059         * The size of the default pipe in bytes.
060         * 
061         * @since Android 1.0
062         */
063        protected static final int PIPE_SIZE = 1024;
064    
065        /**
066         * Indicates if this pipe is connected.
067         */
068        boolean isConnected = false;
069    
070        /**
071         * Constructs a new unconnected {@code PipedInputStream}. The resulting
072         * stream must be connected to a {@link PipedOutputStream} before data may
073         * be read from it.
074         * 
075         * @since Android 1.0
076         */
077        public PipedInputStream() {
078            /* empty */
079        }
080    
081        /**
082         * Constructs a new {@code PipedInputStream} connected to the
083         * {@link PipedOutputStream} {@code out}. Any data written to the output
084         * stream can be read from the this input stream.
085         * 
086         * @param out
087         *            the piped output stream to connect to.
088         * @throws IOException
089         *             if this stream or {@code out} are already connected.
090         * @since Android 1.0
091         */
092        public PipedInputStream(PipedOutputStream out) throws IOException {
093            connect(out);
094        }
095    
096        /**
097         * Returns the number of bytes that are available before this stream will
098         * block. This implementation returns the number of bytes written to this
099         * pipe that have not been read yet.
100         * 
101         * @return the number of bytes available before blocking.
102         * @throws IOException
103         *             if an error occurs in this stream.
104         * @since Android 1.0
105         */
106        @Override
107        public synchronized int available() throws IOException {
108            if (buffer == null || in == -1) {
109                return 0;
110            }
111            return in <= out ? buffer.length - out + in : in - out;
112        }
113    
114        /**
115         * Closes this stream. This implementation releases the buffer used for the
116         * pipe and notifies all threads waiting to read or write.
117         * 
118         * @throws IOException
119         *             if an error occurs while closing this stream.
120         * @since Android 1.0
121         */
122        @Override
123        public void close() throws IOException {
124            synchronized (this) {
125                /* No exception thrown if already closed */
126                if (buffer != null) {
127                    /* Release buffer to indicate closed. */
128                    buffer = null;
129                }
130            }
131        }
132    
133        /**
134         * Connects this {@code PipedInputStream} to a {@link PipedOutputStream}.
135         * Any data written to the output stream becomes readable in this input
136         * stream.
137         * 
138         * @param src
139         *            the source output stream.
140         * @throws IOException
141         *             if either stream is already connected.
142         * @since Android 1.0
143         */
144        public void connect(PipedOutputStream src) throws IOException {
145            src.connect(this);
146        }
147    
148        /**
149         * Reads a single byte from this stream and returns it as an integer in the
150         * range from 0 to 255. Returns -1 if the end of this stream has been
151         * reached. If there is no data in the pipe, this method blocks until data
152         * is available, the end of the stream is detected or an exception is
153         * thrown.
154         * <p>
155         * Separate threads should be used to read from a {@code PipedInputStream}
156         * and to write to the connected {@link PipedOutputStream}. If the same
157         * thread is used, a deadlock may occur.
158         * </p>
159         * 
160         * @return the byte read or -1 if the end of the source stream has been
161         *         reached.
162         * @throws IOException
163         *             if this stream is closed or not connected to an output
164         *             stream, or if the thread writing to the connected output
165         *             stream is no longer alive.
166         * @since Android 1.0
167         */
168        @Override
169        public synchronized int read() throws IOException {
170            if (!isConnected) {
171                throw new IOException(Msg.getString("K0074")); //$NON-NLS-1$
172            }
173            if (buffer == null) {
174                throw new IOException(Msg.getString("K0075")); //$NON-NLS-1$
175            }
176            /**
177             * Set the last thread to be reading on this PipedInputStream. If
178             * lastReader dies while someone is waiting to write an IOException of
179             * "Pipe broken" will be thrown in receive()
180             */
181            lastReader = Thread.currentThread();
182            try {
183                boolean first = true;
184                while (in == -1) {
185                    // Are we at end of stream?
186                    if (isClosed) {
187                        return -1;
188                    }
189                    if (!first && lastWriter != null && !lastWriter.isAlive()) {
190                        throw new IOException(Msg.getString("K0076")); //$NON-NLS-1$
191                    }
192                    first = false;
193                    // Notify callers of receive()
194                    notifyAll();
195                    wait(1000);
196                }
197            } catch (InterruptedException e) {
198                throw new InterruptedIOException();
199            }
200    
201            byte result = buffer[out++];
202            if (out == buffer.length) {
203                out = 0;
204            }
205            if (out == in) {
206                // empty buffer
207                in = -1;
208                out = 0;
209            }
210            return result & 0xff;
211        }
212    
213        /**
214         * Reads at most {@code count} bytes from this stream and stores them in the
215         * byte array {@code bytes} starting at {@code offset}. Blocks until at
216         * least one byte has been read, the end of the stream is detected or an
217         * exception is thrown.
218         * <p>
219         * Separate threads should be used to read from a {@code PipedInputStream}
220         * and to write to the connected {@link PipedOutputStream}. If the same
221         * thread is used, a deadlock may occur.
222         * </p>
223         * 
224         * @param bytes
225         *            the array in which to store the bytes read.
226         * @param offset
227         *            the initial position in {@code bytes} to store the bytes
228         *            read from this stream.
229         * @param count
230         *            the maximum number of bytes to store in {@code bytes}.
231         * @return the number of bytes actually read or -1 if the end of the stream
232         *         has been reached.
233         * @throws IndexOutOfBoundsException
234         *             if {@code offset < 0} or {@code count < 0}, or if {@code
235         *             offset + count} is greater than the size of {@code bytes}.
236         * @throws InterruptedIOException
237         *             if the thread reading from this stream is interrupted.
238         * @throws IOException
239         *             if this stream is closed or not connected to an output
240         *             stream, or if the thread writing to the connected output
241         *             stream is no longer alive.
242         * @throws NullPointerException
243         *             if {@code bytes} is {@code null}.
244         */
245        @Override
246        public synchronized int read(byte[] bytes, int offset, int count)
247                throws IOException {
248            // BEGIN android-changed
249            if (bytes == null) {
250                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
251            }
252    
253            // Exception priorities (in case of multiple errors) differ from
254            // RI, but are spec-compliant.
255            // removed redundant check, used (offset | count) < 0
256            // instead of (offset < 0) || (count < 0) to safe one operation
257            if ((offset | count) < 0 || count > bytes.length - offset) {
258                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
259            }
260            // END android-changed
261    
262            if (count == 0) {
263                return 0;
264            }
265    
266            if (!isConnected) {
267                throw new IOException(Msg.getString("K0074")); //$NON-NLS-1$
268            }
269    
270            if (buffer == null) {
271                throw new IOException(Msg.getString("K0075")); //$NON-NLS-1$
272            }
273    
274            /**
275             * Set the last thread to be reading on this PipedInputStream. If
276             * lastReader dies while someone is waiting to write an IOException of
277             * "Pipe broken" will be thrown in receive()
278             */
279            lastReader = Thread.currentThread();
280            try {
281                boolean first = true;
282                while (in == -1) {
283                    // Are we at end of stream?
284                    if (isClosed) {
285                        return -1;
286                    }
287                    if (!first && lastWriter != null && !lastWriter.isAlive()) {
288                        throw new IOException(Msg.getString("K0076")); //$NON-NLS-1$
289                    }
290                    first = false;
291                    // Notify callers of receive()
292                    notifyAll();
293                    wait(1000);
294                }
295            } catch (InterruptedException e) {
296                throw new InterruptedIOException();
297            }
298    
299            int copyLength = 0;
300            /* Copy bytes from out to end of buffer first */
301            if (out >= in) {
302                copyLength = count > (buffer.length - out) ? buffer.length - out
303                        : count;
304                System.arraycopy(buffer, out, bytes, offset, copyLength);
305                out += copyLength;
306                if (out == buffer.length) {
307                    out = 0;
308                }
309                if (out == in) {
310                    // empty buffer
311                    in = -1;
312                    out = 0;
313                }
314            }
315    
316            /*
317             * Did the read fully succeed in the previous copy or is the buffer
318             * empty?
319             */
320            if (copyLength == count || in == -1) {
321                return copyLength;
322            }
323    
324            int bytesCopied = copyLength;
325            /* Copy bytes from 0 to the number of available bytes */
326            copyLength = in - out > (count - bytesCopied) ? count - bytesCopied
327                    : in - out;
328            System.arraycopy(buffer, out, bytes, offset + bytesCopied, copyLength);
329            out += copyLength;
330            if (out == in) {
331                // empty buffer
332                in = -1;
333                out = 0;
334            }
335            return bytesCopied + copyLength;
336        }
337    
338        /**
339         * Receives a byte and stores it in this stream's {@code buffer}. This
340         * method is called by {@link PipedOutputStream#write(int)}. The least
341         * significant byte of the integer {@code oneByte} is stored at index
342         * {@code in} in the {@code buffer}.
343         * <p>
344         * This method blocks as long as {@code buffer} is full.
345         * </p>
346         * 
347         * @param oneByte
348         *            the byte to store in this pipe.
349         * @throws InterruptedIOException
350         *             if the {@code buffer} is full and the thread that has called
351         *             this method is interrupted.
352         * @throws IOException
353         *             if this stream is closed or the thread that has last read
354         *             from this stream is no longer alive.
355         */
356        protected synchronized void receive(int oneByte) throws IOException {
357            if (buffer == null || isClosed) {
358                throw new IOException(Msg.getString("K0078")); //$NON-NLS-1$
359            }
360            if (lastReader != null && !lastReader.isAlive()) {
361                throw new IOException(Msg.getString("K0076")); //$NON-NLS-1$
362            }
363            /**
364             * Set the last thread to be writing on this PipedInputStream. If
365             * lastWriter dies while someone is waiting to read an IOException of
366             * "Pipe broken" will be thrown in read()
367             */
368            lastWriter = Thread.currentThread();
369            try {
370                while (buffer != null && out == in) {
371                    notifyAll();
372                    wait(1000);
373                    if (lastReader != null && !lastReader.isAlive()) {
374                        throw new IOException(Msg.getString("K0076")); //$NON-NLS-1$
375                    }
376                }
377            } catch (InterruptedException e) {
378                throw new InterruptedIOException();
379            }
380            if (buffer != null) {
381                if (in == -1) {
382                    in = 0;
383                }
384                buffer[in++] = (byte) oneByte;
385                if (in == buffer.length) {
386                    in = 0;
387                }
388                return;
389            }
390        }
391    
392        synchronized void done() {
393            isClosed = true;
394            notifyAll();
395        }
396    }
