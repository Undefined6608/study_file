018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * Wraps an existing {@link InputStream} and adds functionality to "push back"
024     * bytes that have been read, so that they can be read again. Parsers may find
025     * this useful. The number of bytes which may be pushed back can be specified
026     * during construction. If the buffer of pushed back bytes is empty, bytes are
027     * read from the underlying input stream.
028     * 
029     * @since Android 1.0
030     */
031    public class PushbackInputStream extends FilterInputStream {
032        /**
033         * The buffer that contains pushed-back bytes.
034         * 
035         * @since Android 1.0
036         */
037        protected byte[] buf;
038    
039        /**
040         * The current position within {@code buf}. A value equal to
041         * {@code buf.length} indicates that no bytes are available. A value of 0
042         * indicates that the buffer is full.
043         * 
044         * @since Android 1.0
045         */
046        protected int pos;
047    
048        /**
049         * Constructs a new {@code PushbackInputStream} with the specified input
050         * stream as source. The size of the pushback buffer is set to the default
051         * value of 1 byte.
052         * 
053         * @param in
054         *            the source input stream.
055         * @since Android 1.0
056         */
057        public PushbackInputStream(InputStream in) {
058            super(in);
059            buf = (in == null) ? null : new byte[1];
060            pos = 1;
061        }
062    
063        /**
064         * Constructs a new {@code PushbackInputStream} with {@code in} as source
065         * input stream. The size of the pushback buffer is set to {@code size}.
066         * 
067         * @param in
068         *            the source input stream.
069         * @param size
070         *            the size of the pushback buffer.
071         * @throws IllegalArgumentException
072         *             if {@code size} is negative.
073         * @since Android 1.0
074         */
075        public PushbackInputStream(InputStream in, int size) {
076            super(in);
077            if (size <= 0) {
078                throw new IllegalArgumentException(Msg.getString("K0058")); //$NON-NLS-1$
079            }
080            buf = (in == null) ? null : new byte[size];
081            pos = size;
082        }
083    
084        /**
085         * Returns the number of bytes that are available before this stream will
086         * block. This is the sum of the bytes available in the pushback buffer and
087         * those available from the source stream.
088         * 
089         * @return the number of bytes available before blocking.
090         * @throws IOException
091         *             if this stream is closed or an I/O error occurs in the source
092         *             stream.
093         * @since Android 1.0
094         */
095        @Override
096        public int available() throws IOException {
097            if (buf == null) {
098                throw new IOException();
099            }
100            return buf.length - pos + in.available();
101        }
102    
103        /**
104         * Closes this stream. This implementation closes the source stream
105         * and releases the pushback buffer.
106         * 
107         * @throws IOException
108         *             if an error occurs while closing this stream.
109         * @since Android 1.0
110         */
111        @Override
112        public void close() throws IOException {
113            if (in != null) {
114                in.close();
115                in = null;
116                buf = null;
117            }
118        }
119    
120        /**
121         * Indicates whether this stream supports the {@code mark(int)} and
122         * {@code reset()} methods. {@code PushbackInputStream} does not support
123         * them, so it returns {@code false}.
124         * 
125         * @return always {@code false}.
126         * @see #mark(int)
127         * @see #reset()
128         * @since Android 1.0
129         */
130        @Override
131        public boolean markSupported() {
132            return false;
133        }
134    
135        /**
136         * Reads a single byte from this stream and returns it as an integer in the
137         * range from 0 to 255. If the pushback buffer does not contain any
138         * available bytes then a byte from the source input stream is returned.
139         * Blocks until one byte has been read, the end of the source stream is
140         * detected or an exception is thrown.
141         * 
142         * @return the byte read or -1 if the end of the source stream has been
143         *         reached.
144         * @throws IOException
145         *             if this stream is closed or an I/O error occurs while reading
146         *             from this stream.
147         * @since Android 1.0
148         */
149        @Override
150        public int read() throws IOException {
151            if (buf == null) {
152                throw new IOException();
153            }
154            // Is there a pushback byte available?
155            if (pos < buf.length) {
156                return (buf[pos++] & 0xFF);
157            }
158            // Assume read() in the InputStream will return low-order byte or -1
159            // if end of stream.
160            return in.read();
161        }
162    
163        /**
164         * Reads at most {@code length} bytes from this stream and stores them in
165         * the byte array {@code buffer} starting at {@code offset}. Bytes are read
166         * from the pushback buffer first, then from the source stream if more bytes
167         * are required. Blocks until {@code count} bytes have been read, the end of
168         * the source stream is detected or an exception is thrown.
169         * 
170         * @param buffer
171         *            the array in which to store the bytes read from this stream.
172         * @param offset
173         *            the initial position in {@code buffer} to store the bytes read
174         *            from this stream.
175         * @param length
176         *            the maximum number of bytes to store in {@code buffer}.
177         * @return the number of bytes read or -1 if the end of the source stream
178         *         has been reached.
179         * @throws IndexOutOfBoundsException
180         *             if {@code offset < 0} or {@code length < 0}, or if
181         *             {@code offset + length} is greater than the length of
182         *             {@code buffer}.
183         * @throws IOException
184         *             if this stream is closed or another I/O error occurs while
185         *             reading from this stream.
186         * @throws NullPointerException
187         *             if {@code buffer} is {@code null}.
188         * @since Android 1.0
189         */
190        @Override
191        public int read(byte[] buffer, int offset, int length) throws IOException {
192            if (buf == null) {
193                throw new IOException();
194            }
195            // BEGIN android-changed
196            if (buffer == null) {
197                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
198            }
199            // avoid int overflow
200            // Exception priorities (in case of multiple errors) differ from
201            // RI, but are spec-compliant.
202            // removed redundant check, used (offset | length) < 0
203            // instead of (offset < 0) || (length < 0) to safe one operation
204            if ((offset | length) < 0 || length > buffer.length - offset) {
205                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
206            }
207            // END android-changed
208    
209            int copiedBytes = 0, copyLength = 0, newOffset = offset;
210            // Are there pushback bytes available?
211            if (pos < buf.length) {
212                copyLength = (buf.length - pos >= length) ? length : buf.length
213                        - pos;
214                System.arraycopy(buf, pos, buffer, newOffset, copyLength);
215                newOffset += copyLength;
216                copiedBytes += copyLength;
217                // Use up the bytes in the local buffer
218                pos += copyLength;
219            }
220            // Have we copied enough?
221            if (copyLength == length) {
222                return length;
223            }
224            int inCopied = in.read(buffer, newOffset, length - copiedBytes);
225            if (inCopied > 0) {
226                return inCopied + copiedBytes;
227            }
228            if (copiedBytes == 0) {
229                return inCopied;
230            }
231            return copiedBytes;
232        }
233    
234        /**
235         * Skips {@code count} bytes in this stream. This implementation skips bytes
236         * in the pushback buffer first and then in the source stream if necessary.
237         * 
238         * @param count
239         *            the number of bytes to skip.
240         * @return the number of bytes actually skipped.
241         * @throws IOException
242         *             if this stream is closed or another I/O error occurs.
243         * @since Android 1.0
244         */
245        @Override
246        public long skip(long count) throws IOException {
247            if (in == null) {
248                throw new IOException(Msg.getString("K0059")); //$NON-NLS-1$
249            }
250            if (count <= 0) {
251                return 0;
252            }
253            int numSkipped = 0;
254            if (pos < buf.length) {
255                numSkipped += (count < buf.length - pos) ? count : buf.length - pos;
256                pos += numSkipped;
257            }
258            if (numSkipped < count) {
259                numSkipped += in.skip(count - numSkipped);
260            }
261            return numSkipped;
262        }
263    
264        /**
265         * Pushes all the bytes in {@code buffer} back to this stream. The bytes are
266         * pushed back in such a way that the next byte read from this stream is
267         * buffer[0], then buffer[1] and so on.
268         * <p>
269         * If this stream's internal pushback buffer cannot store the entire
270         * contents of {@code buffer}, an {@code IOException} is thrown. Parts of
271         * {@code buffer} may have already been copied to the pushback buffer when
272         * the exception is thrown.
273         * </p>
274         * 
275         * @param buffer
276         *            the buffer containing the bytes to push back to this stream.
277         * @throws IOException
278         *             if the free space in the internal pushback buffer is not
279         *             sufficient to store the contents of {@code buffer}.
280         * @since Android 1.0
281         */
282        public void unread(byte[] buffer) throws IOException {
283            unread(buffer, 0, buffer.length);
284        }
285    
286        /**
287         * Pushes a subset of the bytes in {@code buffer} back to this stream. The
288         * subset is defined by the start position {@code offset} within
289         * {@code buffer} and the number of bytes specified by {@code length}. The
290         * bytes are pushed back in such a way that the next byte read from this
291         * stream is {@code buffer[offset]}, then {@code buffer[1]} and so on.
292         * <p>
293         * If this stream's internal pushback buffer cannot store the selected
294         * subset of {@code buffer}, an {@code IOException} is thrown. Parts of
295         * {@code buffer} may have already been copied to the pushback buffer when
296         * the exception is thrown.
297         * </p>
298         * 
299         * @param buffer
300         *            the buffer containing the bytes to push back to this stream.
301         * @param offset
302         *            the index of the first byte in {@code buffer} to push back.
303         * @param length
304         *            the number of bytes to push back.
305         * @throws IndexOutOfBoundsException
306         *             if {@code offset < 0} or {@code length < 0}, or if
307         *             {@code offset + length} is greater than the length of
308         *             {@code buffer}.
309         * @throws IOException
310         *             if the free space in the internal pushback buffer is not
311         *             sufficient to store the selected contents of {@code buffer}.
312         * @since Android 1.0
313         */
314        public void unread(byte[] buffer, int offset, int length)
315                throws IOException {
316            if (length > pos) {
317                // Pushback buffer full
318                throw new IOException(Msg.getString("K007e")); //$NON-NLS-1$
319            }
320            // avoid int overflow
321            // BEGIN android-changed
322            // Exception priorities (in case of multiple errors) differ from
323            // RI, but are spec-compliant.
324            // removed redundant check, made implicit null check explicit
325            // used (offset | length) < 0 instead of (offset < 0) || (length < 0)
326            // to safe one operation
327            if (buffer == null) {
328                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
329            }
330            if ((offset | length) < 0 || length > buffer.length - offset) {
331                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
332            }
333            // END android-changed
334    
335            for (int i = offset + length - 1; i >= offset; i--) {
336                unread(buffer[i]);
337            }
338        }
339    
340        /**
341         * Pushes the specified byte {@code oneByte} back to this stream. Only the
342         * least significant byte of the integer {@code oneByte} is pushed back.
343         * This is done in such a way that the next byte read from this stream is
344         * {@code (byte) oneByte}.
345         * <p>
346         * If this stream's internal pushback buffer cannot store the byte, an
347         * {@code IOException} is thrown.
348         * </p>
349         * 
350         * @param oneByte
351         *            the byte to push back to this stream.
352         * @throws IOException
353         *             if this stream is closed or the internal pushback buffer is
354         *             full.
355         * @since Android 1.0
356         */
357        public void unread(int oneByte) throws IOException {
358            if (buf == null) {
359                throw new IOException();
360            }
361            if (pos == 0) {
362                throw new IOException(Msg.getString("K007e")); //$NON-NLS-1$
363            }
364            buf[--pos] = (byte) oneByte;
365        }
366    
367        /**
368         * Marks the current position in this stream. Setting a mark is not
369         * supported in this class; this implementation does nothing.
370         * 
371         * @param readlimit
372         *            the number of bytes that can be read from this stream before
373         *            the mark is invalidated; this parameter is ignored.
374         * @since Android 1.0
375         */
376        @Override
377        public void mark(int readlimit) {
378            return;
379        }
380    
381        /**
382         * Resets this stream to the last marked position. Resetting the stream is
383         * not supported in this class; this implementation always throws an
384         * {@code IOException}.
385         * 
386         * @throws IOException
387         *             if this method is called.
388         * @since Android 1.0
389         */
390        @Override
391        public void reset() throws IOException {
392            throw new IOException();
393        }
