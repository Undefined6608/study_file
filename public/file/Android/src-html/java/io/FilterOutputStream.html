018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * Wraps an existing {@link OutputStream} and performs some transformation on
024     * the output data while it is being written. Transformations can be anything
025     * from a simple byte-wise filtering output data to an on-the-fly compression or
026     * decompression of the underlying stream. Output streams that wrap another
027     * output stream and provide some additional functionality on top of it usually
028     * inherit from this class.
029     * 
030     * @see FilterOutputStream
031     * 
032     * @since Android 1.0
033     */
034    public class FilterOutputStream extends OutputStream {
035    
036        /**
037         * The target output stream for this filter stream.
038         * 
039         * @since Android 1.0
040         */
041        protected OutputStream out;
042    
043        /**
044         * Constructs a new {@code FilterOutputStream} with {@code out} as its
045         * target stream.
046         * 
047         * @param out
048         *            the target stream that this stream writes to.
049         * @since Android 1.0
050         */
051        public FilterOutputStream(OutputStream out) {
052            this.out = out;
053        }
054    
055        /**
056         * Closes this stream. This implementation closes the target stream.
057         * 
058         * @throws IOException
059         *             if an error occurs attempting to close this stream.
060         * @since Android 1.0
061         */
062        @Override
063        public void close() throws IOException {
064            try {
065                flush();
066            } catch (IOException e) {
067                // Ignored
068            }
069            /* Make sure we clean up this stream if exception fires */
070            out.close();
071        }
072    
073        /**
074         * Ensures that all pending data is sent out to the target stream. This
075         * implementation flushes the target stream.
076         * 
077         * @throws IOException
078         *             if an error occurs attempting to flush this stream.
079         * @since Android 1.0
080         */
081        @Override
082        public void flush() throws IOException {
083            out.flush();
084        }
085    
086        /**
087         * Writes the entire contents of the byte array {@code buffer} to this
088         * stream. This implementation writes the {@code buffer} to the target
089         * stream.
090         * 
091         * @param buffer
092         *            the buffer to be written.
093         * @throws IOException
094         *             if an I/O error occurs while writing to this stream.
095         * @since Android 1.0
096         */
097        @Override
098        public void write(byte[] buffer) throws IOException {
099            // BEGIN android-note
100            // changed array notation to be consistent with the rest of harmony
101            // END android-note
102            write(buffer, 0, buffer.length);
103        }
104    
105        /**
106         * Writes {@code count} bytes from the byte array {@code buffer} starting at
107         * {@code offset} to the target stream.
108         * 
109         * @param buffer
110         *            the buffer to write.
111         * @param offset
112         *            the index of the first byte in {@code buffer} to write.
113         * @param count
114         *            the number of bytes in {@code buffer} to write.
115         * @throws IndexOutOfBoundsException
116         *             if {@code offset < 0} or {@code count < 0}, or if
117         *             {@code offset + count} is bigger than the length of
118         *             {@code buffer}.
119         * @throws IOException
120         *             if an I/O error occurs while writing to this stream.
121         * @since Android 1.0
122         */
123        @Override
124        public void write(byte[] buffer, int offset, int count) throws IOException {
125            // BEGIN android-note
126            // changed array notation to be consistent with the rest of harmony
127            // END android-note
128            // avoid int overflow, force null buffer check first
129            // BEGIN android-changed
130            // Exception priorities (in case of multiple errors) differ from
131            // RI, but are spec-compliant.
132            // removed redundant check, made implicit null check explicit,
133            // used (offset | count) < 0 instead of (offset < 0) || (count < 0)
134            // to safe one operation
135            if (buffer == null) {
136                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
137            }
138            if ((offset | count) < 0 || count > buffer.length - offset) {
139                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
140            }
141            // END android-changed
142            for (int i = 0; i < count; i++) {
143                // Call write() instead of out.write() since subclasses could
144                // override the write() method.
145                write(buffer[offset + i]);
146            }
147        }
148    
149        /**
150         * Writes one byte to the target stream. Only the low order byte of the
151         * integer {@code oneByte} is written.
152         * 
153         * @param oneByte
154         *            the byte to be written.
155         * @throws IOException
156         *             if an I/O error occurs while writing to this stream.
157         * @since Android 1.0
158         */
159        @Override
160        public void write(int oneByte) throws IOException {
161            out.write(oneByte);
162        }
163    }
