018    package java.io;
019    
020    import java.nio.ByteBuffer;
021    import java.nio.CharBuffer;
022    import java.nio.charset.Charset;
023    import java.nio.charset.CharsetEncoder;
024    import java.nio.charset.CoderResult;
025    import java.nio.charset.CodingErrorAction;
026    import java.security.AccessController;
027    
028    import org.apache.harmony.luni.util.Msg;
029    import org.apache.harmony.luni.util.PriviAction;
030    
031    /**
032     * A class for turning a character stream into a byte stream. Data written to
033     * the target input stream is converted into bytes by either a default or a
034     * provided character converter. The default encoding is taken from the
035     * "file.encoding" system property. {@code OutputStreamWriter} contains a buffer
036     * of bytes to be written to target stream and converts these into characters as
037     * needed. The buffer size is 8K.
038     * 
039     * @see InputStreamReader
040     * 
041     * @since Android 1.0
042     */
043    public class OutputStreamWriter extends Writer {
044    
045        private OutputStream out;
046    
047        private CharsetEncoder encoder;
048    
049        private ByteBuffer bytes = ByteBuffer.allocate(8192);
050    
051        /**
052         * Constructs a new OutputStreamWriter using {@code out} as the target
053         * stream to write converted characters to. The default character encoding
054         * is used.
055         * 
056         * @param out
057         *            the non-null target stream to write converted bytes to.
058         * @since Android 1.0
059         */
060        public OutputStreamWriter(OutputStream out) {
061            super(out);
062            this.out = out;
063            String encoding = AccessController
064                    .doPrivileged(new PriviAction<String>(
065                            "file.encoding", "ISO8859_1")); //$NON-NLS-1$ //$NON-NLS-2$
066            encoder = Charset.forName(encoding).newEncoder();
067            encoder.onMalformedInput(CodingErrorAction.REPLACE);
068            encoder.onUnmappableCharacter(CodingErrorAction.REPLACE);
069        }
070    
071        /**
072         * Constructs a new OutputStreamWriter using {@code out} as the target
073         * stream to write converted characters to and {@code enc} as the character
074         * encoding. If the encoding cannot be found, an
075         * UnsupportedEncodingException error is thrown.
076         * 
077         * @param out
078         *            the target stream to write converted bytes to.
079         * @param enc
080         *            the string describing the desired character encoding.
081         * @throws NullPointerException
082         *             if {@code enc} is {@code null}.
083         * @throws UnsupportedEncodingException
084         *             if the encoding specified by {@code enc} cannot be found.
085         * @since Android 1.0
086         */
087        public OutputStreamWriter(OutputStream out, final String enc)
088                throws UnsupportedEncodingException {
089            super(out);
090            if (enc == null) {
091                throw new NullPointerException();
092            }
093            this.out = out;
094            try {
095                encoder = Charset.forName(enc).newEncoder();
096            } catch (Exception e) {
097                throw new UnsupportedEncodingException(enc);
098            }
099            encoder.onMalformedInput(CodingErrorAction.REPLACE);
100            encoder.onUnmappableCharacter(CodingErrorAction.REPLACE);
101        }
102    
103        /**
104         * Constructs a new OutputStreamWriter using {@code out} as the target
105         * stream to write converted characters to and {@code cs} as the character
106         * encoding.
107         * 
108         * @param out
109         *            the target stream to write converted bytes to.
110         * @param cs
111         *            the {@code Charset} that specifies the character encoding.
112         * @since Android 1.0
113         */
114        public OutputStreamWriter(OutputStream out, Charset cs) {
115            super(out);
116            this.out = out;
117            encoder = cs.newEncoder();
118            encoder.onMalformedInput(CodingErrorAction.REPLACE);
119            encoder.onUnmappableCharacter(CodingErrorAction.REPLACE);
120        }
121    
122        /**
123         * Constructs a new OutputStreamWriter using {@code out} as the target
124         * stream to write converted characters to and {@code enc} as the character
125         * encoder.
126         * 
127         * @param out
128         *            the target stream to write converted bytes to.
129         * @param enc
130         *            the character encoder used for character conversion.
131         * @since Android 1.0
132         */
133        public OutputStreamWriter(OutputStream out, CharsetEncoder enc) {
134            super(out);
135            enc.charset();
136            this.out = out;
137            encoder = enc;
138        }
139    
140        /**
141         * Closes this writer. This implementation flushes the buffer as well as the
142         * target stream. The target stream is then closed and the resources for the
143         * buffer and converter are released.
144         * <p>
145         * Only the first invocation of this method has any effect. Subsequent calls
146         * do nothing.
147         * </p>
148         * 
149         * @throws IOException
150         *             if an error occurs while closing this writer.
151         * @since Android 1.0
152         */
153        @Override
154        public void close() throws IOException {
155            synchronized (lock) {
156                if (encoder != null) {
157                    encoder.flush(bytes);
158                    flush();
159                    out.flush();
160                    out.close();
161                    encoder = null;
162                    bytes = null;
163                }
164            }
165        }
166    
167        /**
168         * Flushes this writer. This implementation ensures that all buffered bytes
169         * are written to the target stream. After writing the bytes, the target
170         * stream is flushed as well.
171         * 
172         * @throws IOException
173         *             if an error occurs while flushing this writer.
174         * @since Android 1.0
175         */
176        @Override
177        public void flush() throws IOException {
178            synchronized (lock) {
179                checkStatus();
180                int position;
181                if ((position = bytes.position()) > 0) {
182                    bytes.flip();
183                    out.write(bytes.array(), 0, position);
184                    bytes.clear();
185                }
186                out.flush();
187            }
188        }
189    
190        private void checkStatus() throws IOException {
191            if (encoder == null) {
192                // K005d=Writer is closed.
193                throw new IOException(Msg.getString("K005d")); //$NON-NLS-1$
194            }
195        }
196    
197        /**
198         * Gets the name of the encoding that is used to convert characters to
199         * bytes.
200         * 
201         * @return the string describing the converter or {@code null} if this
202         *         writer is closed.
203         * @since Android 1.0
204         */
205        public String getEncoding() {
206            if (encoder == null) {
207                return null;
208            }
209            return InputStreamReader.HistoricalNamesUtil.getHistoricalName(encoder
210                    .charset().name());
211        }
212    
213        /**
214         * Writes {@code count} characters starting at {@code offset} in {@code buf}
215         * to this writer. The characters are immediately converted to bytes by the
216         * character converter and stored in a local buffer. If the buffer gets full
217         * as a result of the conversion, this writer is flushed.
218         * 
219         * @param buf
220         *            the array containing characters to write.
221         * @param offset
222         *            the index of the first character in {@code buf} to write.
223         * @param count
224         *            the maximum number of characters to write.
225         * @throws IndexOutOfBoundsException
226         *             if {@code offset < 0} or {@code count < 0}, or if
227         *             {@code offset + count} is greater than the size of
228         *             {@code buf}.
229         * @throws IOException
230         *             if this writer has already been closed or another I/O error
231         *             occurs.
232         * @since Android 1.0
233         */
234        @Override
235        public void write(char[] buf, int offset, int count) throws IOException {
236            synchronized (lock) {
237                checkStatus();
238                // BEGIN android-changed
239                // Exception priorities (in case of multiple errors) differ from
240                // RI, but are spec-compliant.
241                // made implicit null check explicit,
242                // used (offset | count) < 0 instead of (offset < 0) || (count < 0)
243                // to safe one operation
244                if (buf == null) {
245                    throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
246                }
247                if ((offset | count) < 0 || offset > buf.length - count) {
248                    throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
249                }
250                // END android-changed
251                CharBuffer chars = CharBuffer.wrap(buf, offset, count);
252                convert(chars);
253            }
254        }
255    
256        private void convert(CharBuffer chars) throws IOException {
257            CoderResult result = encoder.encode(chars, bytes, true);
258            while (true) {
259                if (result.isError()) {
260                    throw new IOException(result.toString());
261                } else if (result.isOverflow()) {
262                    // flush the output buffer
263                    flush();
264                    result = encoder.encode(chars, bytes, true);
265                    continue;
266                }
267                break;
268            }
269        }
270    
271        /**
272         * Writes the character {@code oneChar} to this writer. The lowest two bytes
273         * of the integer {@code oneChar} are immediately converted to bytes by the
274         * character converter and stored in a local buffer. If the buffer gets full
275         * by converting this character, this writer is flushed.
276         * 
277         * @param oneChar
278         *            the character to write.
279         * @throws IOException
280         *             if this writer is closed or another I/O error occurs.
281         * @since Android 1.0
282         */
283        @Override
284        public void write(int oneChar) throws IOException {
285            synchronized (lock) {
286                checkStatus();
287                CharBuffer chars = CharBuffer.wrap(new char[] { (char) oneChar });
288                convert(chars);
289            }
290        }
291    
292        /**
293         * Writes {@code count} characters starting at {@code offset} in {@code str}
294         * to this writer. The characters are immediately converted to bytes by the
295         * character converter and stored in a local buffer. If the buffer gets full
296         * as a result of the conversion, this writer is flushed.
297         * 
298         * @param str
299         *            the string containing characters to write.
300         * @param offset
301         *            the start position in {@code str} for retrieving characters.
302         * @param count
303         *            the maximum number of characters to write.
304         * @throws IOException
305         *             if this writer has already been closed or another I/O error
306         *             occurs.
307         * @throws IndexOutOfBoundsException
308         *             if {@code offset < 0} or {@code count < 0}, or if
309         *             {@code offset + count} is bigger than the length of
310         *             {@code str}.
311         * @since Android 1.0
312         */
313        @Override
314        public void write(String str, int offset, int count) throws IOException {
315            synchronized (lock) {
316                // avoid int overflow
317                // BEGIN android-changed
318                // Exception priorities (in case of multiple errors) differ from RI,
319                // but are spec-compliant.
320                // made implicit null check explicit, used (offset | count) < 0
321                // instead of (offset < 0) || (count < 0) to safe one operation
322                if (str == null) {
323                    throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
324                }
325                if ((offset | count) < 0 || offset > str.length() - count) {
326                    throw new StringIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
327                }
328                // END android-changed
329                checkStatus();
330                CharBuffer chars = CharBuffer.wrap(str, offset, count + offset);
331                convert(chars);
332            }
333        }
