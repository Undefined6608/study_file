018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * The base class for all output streams. An output stream is a means of writing
024     * data to a target in a byte-wise manner. Most output streams expect the
025     * {@link #flush()} method to be called before closing the stream, to ensure all
026     * data is actually written through.
027     * <p>
028     * This abstract class does not provide a fully working implementation, so it
029     * needs to be subclassed, and at least the {@link #write(int)} method needs to
030     * be overridden. Overriding some of the non-abstract methods is also often
031     * advised, since it might result in higher efficiency.
032     * <p>
033     * Many specialized output streams for purposes like writing to a file already
034     * exist in this package.
035     * 
036     * @see InputStream
037     * 
038     * @since Android 1.0
039     */
040    public abstract class OutputStream implements Closeable, Flushable {
041    
042        /**
043         * Default constructor.
044         * 
045         * @since Android 1.0
046         */
047        public OutputStream() {
048            super();
049        }
050    
051        /**
052         * Closes this stream. Implementations of this method should free any
053         * resources used by the stream. This implementation does nothing.
054         * 
055         * @throws IOException
056         *             if an error occurs while closing this stream.
057         * @since Android 1.0
058         */
059        public void close() throws IOException {
060            /* empty */
061        }
062    
063        /**
064         * Flushes this stream. Implementations of this method should ensure that
065         * any buffered data is written out. This implementation does nothing.
066         * 
067         * @throws IOException
068         *             if an error occurs while flushing this stream.
069         * @since Android 1.0
070         */
071        public void flush() throws IOException {
072            /* empty */
073        }
074    
075        /**
076         * Writes the entire contents of the byte array {@code buffer} to this
077         * stream.
078         * 
079         * @param buffer
080         *            the buffer to be written.
081         * @throws IOException
082         *             if an error occurs while writing to this stream.
083         * @since Android 1.0
084         */
085        public void write(byte[] buffer) throws IOException {
086            // BEGIN android-note
087            // changed array notation to be consistent with the rest of harmony
088            // END android-note
089            write(buffer, 0, buffer.length);
090        }
091    
092        /**
093         * Writes {@code count} bytes from the byte array {@code buffer} starting at
094         * position {@code offset} to this stream.
095         * 
096         * @param buffer
097         *            the buffer to be written.
098         * @param offset
099         *            the start position in {@code buffer} from where to get bytes.
100         * @param count
101         *            the number of bytes from {@code buffer} to write to this
102         *            stream.
103         * @throws IOException
104         *             if an error occurs while writing to this stream.
105         * @throws IndexOutOfBoundsException
106         *             if {@code offset < 0} or {@code count < 0}, or if
107         *             {@code offset + count} is bigger than the length of
108         *             {@code buffer}.
109         * @since Android 1.0
110         */
111        public void write(byte[] buffer, int offset, int count) throws IOException {
112            // BEGIN android-note
113            // changed array notation to be consistent with the rest of harmony
114            // END android-note
115            // avoid int overflow, check null buffer
116            // BEGIN android-changed
117            // Exception priorities (in case of multiple errors) differ from
118            // RI, but are spec-compliant.
119            // removed redundant check, made implicit null check explicit,
120            // used (offset | count) < 0 instead of (offset < 0) || (count < 0)
121            // to safe one operation
122            if (buffer == null) {
123                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
124            }
125            if ((offset | count) < 0 || count > buffer.length - offset) {
126                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
127            }
128            // END android-changed
129            for (int i = offset; i < offset + count; i++) {
130                write(buffer[i]);
131            }
132        }
133    
134        /**
135         * Writes a single byte to this stream. Only the least significant byte of
136         * the integer {@code oneByte} is written to the stream.
137         * 
138         * @param oneByte
139         *            the byte to be written.
140         * @throws IOException
141         *             if an error occurs while writing to this stream.
142         * @since Android 1.0
143         */
144        public abstract void write(int oneByte) throws IOException;
145    }
