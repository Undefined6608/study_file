018    package java.io;
019    
020    /**
021     * The base class for all writers. A writer is a means of writing data to a
022     * target in a character-wise manner. Most output streams expect the
023     * {@link #flush()} method to be called before closing the stream, to ensure all
024     * data is actually written out.
025     * <p>
026     * This abstract class does not provide a fully working implementation, so it
027     * needs to be subclassed, and at least the {@link #write(char[], int, int)},
028     * {@link #close()} and {@link #flush()} methods needs to be overridden.
029     * Overriding some of the non-abstract methods is also often advised, since it
030     * might result in higher efficiency.
031     * <p>
032     * Many specialized readers for purposes like reading from a file already exist
033     * in this package.
034     * 
035     * @see Reader
036     * 
037     * @since Android 1.0
038     */
039    public abstract class Writer implements Appendable, Closeable, Flushable {
040    
041        static final String TOKEN_NULL = "null"; //$NON-NLS-1$
042    
043        /**
044         * The object used to synchronize access to the writer.
045         * 
046         * @since Android 1.0
047         */
048        protected Object lock;
049    
050        /**
051         * Constructs a new {@code Writer} with {@code this} as the object used to
052         * synchronize critical sections.
053         * 
054         * @since Android 1.0
055         */
056        protected Writer() {
057            super();
058            lock = this;
059        }
060    
061        /**
062         * Constructs a new {@code Writer} with {@code lock} used to synchronize
063         * critical sections.
064         * 
065         * @param lock
066         *            the {@code Object} used to synchronize critical sections.
067         * @throws NullPointerException
068         *             if {@code lock} is {@code null}.
069         * @since Android 1.0
070         */
071        protected Writer(Object lock) {
072            if (lock == null) {
073                throw new NullPointerException();
074            }
075            this.lock = lock;
076        }
077    
078        /**
079         * Closes this writer. Implementations of this method should free any
080         * resources associated with the writer.
081         * 
082         * @throws IOException
083         *             if an error occurs while closing this writer.
084         * @since Android 1.0
085         */
086        public abstract void close() throws IOException;
087    
088        /**
089         * Flushes this writer. Implementations of this method should ensure that
090         * all buffered characters are written to the target.
091         * 
092         * @throws IOException
093         *             if an error occurs while flushing this writer.
094         * @since Android 1.0
095         */
096        public abstract void flush() throws IOException;
097    
098        /**
099         * Writes the entire character buffer {@code buf} to the target.
100         * 
101         * @param buf
102         *            the non-null array containing characters to write.
103         * @throws IOException
104         *             if this writer is closed or another I/O error occurs.
105         * @since Android 1.0
106         */
107        public void write(char[] buf) throws IOException {
108            // BEGIN android-note
109            // changed array notation to be consistent with the rest of harmony
110            // END android-note
111            write(buf, 0, buf.length);
112        }
113    
114        /**
115         * Writes {@code count} characters starting at {@code offset} in {@code buf}
116         * to the target.
117         * 
118         * @param buf
119         *            the non-null character array to write.
120         * @param offset
121         *            the index of the first character in {@code buf} to write.
122         * @param count
123         *            the maximum number of characters to write.
124         * @throws IndexOutOfBoundsException
125         *             if {@code offset < 0} or {@code count < 0}, or if {@code
126         *             offset + count} is greater than the size of {@code buf}.
127         * @throws IOException
128         *             if this writer is closed or another I/O error occurs.
129         * @since Android 1.0
130         */
131        public abstract void write(char[] buf, int offset, int count)
132                throws IOException;
133        // BEGIN android-note
134        // changed array notation to be consistent with the rest of harmony
135        // END android-note
136    
137        /**
138         * Writes one character to the target. Only the two least significant bytes
139         * of the integer {@code oneChar} are written.
140         * 
141         * @param oneChar
142         *            the character to write to the target.
143         * @throws IOException
144         *             if this writer is closed or another I/O error occurs.
145         * @since Android 1.0
146         */
147        public void write(int oneChar) throws IOException {
148            synchronized (lock) {
149                char oneCharArray[] = new char[1];
150                oneCharArray[0] = (char) oneChar;
151                write(oneCharArray);
152            }
153        }
154    
155        /**
156         * Writes the characters from the specified string to the target.
157         * 
158         * @param str
159         *            the non-null string containing the characters to write.
160         * @throws IOException
161         *             if this writer is closed or another I/O error occurs.
162         * @since Android 1.0
163         */
164        public void write(String str) throws IOException {
165            char buf[] = new char[str.length()];
166            str.getChars(0, buf.length, buf, 0);
167            synchronized (lock) {
168                write(buf);
169            }
170        }
171    
172        /**
173         * Writes {@code count} characters from {@code str} starting at {@code
174         * offset} to the target.
175         * 
176         * @param str
177         *            the non-null string containing the characters to write.
178         * @param offset
179         *            the index of the first character in {@code str} to write.
180         * @param count
181         *            the number of characters from {@code str} to write.
182         * @throws IOException
183         *             if this writer is closed or another I/O error occurs.
184         * @throws IndexOutOfBoundsException
185         *             if {@code offset < 0} or {@code count < 0}, or if {@code
186         *             offset + count} is greater than the length of {@code str}.
187         * @since Android 1.0
188         */
189        public void write(String str, int offset, int count) throws IOException {
190            if (count < 0) { // other cases tested by getChars()
191                throw new StringIndexOutOfBoundsException();
192            }
193            char buf[] = new char[count];
194            str.getChars(offset, offset + count, buf, 0);
195    
196            synchronized (lock) {
197                write(buf);
198            }
199        }
200    
201        /**
202         * Appends the character {@code c} to the target. This method works the same
203         * way as {@link #write(int)}.
204         * 
205         * @param c
206         *            the character to append to the target stream.
207         * @return this writer.
208         * @throws IOException
209         *             if this writer is closed or another I/O error occurs.
210         * @since Android 1.0
211         */
212        public Writer append(char c) throws IOException {
213            write(c);
214            return this;
215        }
216    
217        /**
218         * Appends the character sequence {@code csq} to the target. This method
219         * works the same way as {@code Writer.write(csq.toString())}. If {@code
220         * csq} is {@code null}, then the string "null" is written to the target
221         * stream.
222         * 
223         * @param csq
224         *            the character sequence appended to the target.
225         * @return this writer.
226         * @throws IOException
227         *             if this writer is closed or another I/O error occurs.
228         * @since Android 1.0
229         */
230        public Writer append(CharSequence csq) throws IOException {
231            if (null == csq) {
232                write(TOKEN_NULL);
233            } else {
234                write(csq.toString());
235            }
236            return this;
237        }
238    
239        /**
240         * Appends a subsequence of the character sequence {@code csq} to the
241         * target. This method works the same way as {@code
242         * Writer.writer(csq.subsequence(start, end).toString())}. If {@code
243         * csq} is {@code null}, then the specified subsequence of the string "null"
244         * will be written to the target.
245         * 
246         * @param csq
247         *            the character sequence appended to the target.
248         * @param start
249         *            the index of the first char in the character sequence appended
250         *            to the target.
251         * @param end
252         *            the index of the character following the last character of the
253         *            subsequence appended to the target.
254         * @return this writer.
255         * @throws IOException
256         *             if this writer is closed or another I/O error occurs.
257         * @throws IndexOutOfBoundsException
258         *             if {@code start > end}, {@code start < 0}, {@code end < 0} or
259         *             either {@code start} or {@code end} are greater or equal than
260         *             the length of {@code csq}.
261         * @since Android 1.0
262         */
263        public Writer append(CharSequence csq, int start, int end)
264                throws IOException {
265            if (null == csq) {
266                write(TOKEN_NULL.substring(start, end));
267            } else {
268                write(csq.subSequence(start, end).toString());
269            }
270            return this;
271        }
272    }
