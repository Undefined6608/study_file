018    package java.io;
019    
020    // BEGIN android-added
021    import org.apache.harmony.luni.util.Msg;
022    // END android-added
023    
024    /**
025     * The base class for all input streams. An input stream is a means of reading
026     * data from a source in a byte-wise manner.
027     * <p>
028     * Some input streams also support marking a position in the input stream and
029     * returning to this position later. This abstract class does not provide a
030     * fully working implementation, so it needs to be subclassed, and at least the
031     * {@link #read()} method needs to be overridden. Overriding some of the
032     * non-abstract methods is also often advised, since it might result in higher
033     * efficiency.
034     * <p>
035     * Many specialized input streams for purposes like reading from a file already
036     * exist in this package.
037     * 
038     * @see OutputStream
039     * 
040     * @since Android 1.0
041     */
042    public abstract class InputStream extends Object implements Closeable {
043    
044        private static byte[] skipBuf;
045    
046        /**
047         * This constructor does nothing. It is provided for signature
048         * compatibility.
049         * 
050         * @since Android 1.0
051         */
052        public InputStream() {
053            /* empty */
054        }
055    
056        /**
057         * Returns the number of bytes that are available before this stream will
058         * block. This implementation always returns 0. Subclasses should override
059         * and indicate the correct number of bytes available.
060         * 
061         * @return the number of bytes available before blocking.
062         * @throws IOException
063         *             if an error occurs in this stream.
064         * @since Android 1.0
065         */
066        public int available() throws IOException {
067            return 0;
068        }
069    
070        /**
071         * Closes this stream. Concrete implementations of this class should free
072         * any resources during close. This implementation does nothing.
073         * 
074         * @throws IOException
075         *             if an error occurs while closing this stream.
076         * @since Android 1.0
077         */
078        public void close() throws IOException {
079            /* empty */
080        }
081    
082        /**
083         * Sets a mark position in this InputStream. The parameter {@code readlimit}
084         * indicates how many bytes can be read before the mark is invalidated.
085         * Sending {@code reset()} will reposition the stream back to the marked
086         * position provided {@code readLimit} has not been surpassed.
087         * <p>
088         * This default implementation does nothing and concrete subclasses must
089         * provide their own implementation.
090         * </p>
091         * 
092         * @param readlimit
093         *            the number of bytes that can be read from this stream before
094         *            the mark is invalidated.
095         * @see #markSupported()
096         * @see #reset()
097         * @since Android 1.0
098         */
099        public void mark(int readlimit) {
100            /* empty */
101        }
102    
103        /**
104         * Indicates whether this stream supports the {@code mark()} and
105         * {@code reset()} methods. The default implementation returns {@code false}.
106         * 
107         * @return always {@code false}.
108         * @see #mark(int)
109         * @see #reset()
110         * @since Android 1.0
111         */
112        public boolean markSupported() {
113            return false;
114        }
115    
116        /**
117         * Reads a single byte from this stream and returns it as an integer in the
118         * range from 0 to 255. Returns -1 if the end of the stream has been
119         * reached. Blocks until one byte has been read, the end of the source
120         * stream is detected or an exception is thrown.
121         * 
122         * @return the byte read or -1 if the end of stream has been reached.
123         * @throws IOException
124         *             if the stream is closed or another IOException occurs.
125         * @since Android 1.0
126         */
127        public abstract int read() throws IOException;
128    
129        /**
130         * Reads bytes from this stream and stores them in the byte array {@code b}.
131         * 
132         * @param b
133         *            the byte array in which to store the bytes read.
134         * @return the number of bytes actually read or -1 if the end of the stream
135         *         has been reached.
136         * @throws IOException
137         *             if this stream is closed or another IOException occurs.
138         * @since Android 1.0
139         */
140        public int read(byte[] b) throws IOException {
141            // BEGIN android-note
142            // changed array notation to be consistent with the rest of harmony
143            // END android-note
144            return read(b, 0, b.length);
145        }
146    
147        /**
148         * Reads at most {@code length} bytes from this stream and stores them in
149         * the byte array {@code b} starting at {@code offset}.
150         * 
151         * @param b
152         *            the byte array in which to store the bytes read.
153         * @param offset
154         *            the initial position in {@code buffer} to store the bytes read
155         *            from this stream.
156         * @param length
157         *            the maximum number of bytes to store in {@code b}.
158         * @return the number of bytes actually read or -1 if the end of the stream
159         *         has been reached.
160         * @throws IndexOutOfBoundsException
161         *             if {@code offset < 0} or {@code length < 0}, or if
162         *             {@code offset + length} is greater than the length of
163         *             {@code b}.
164         * @throws IOException
165         *             if the stream is closed or another IOException occurs.
166         * @since Android 1.0
167         */
168        public int read(byte[] b, int offset, int length) throws IOException {
169            // BEGIN android-note
170            // changed array notation to be consistent with the rest of harmony
171            // END android-note
172            // avoid int overflow, check null b
173            // BEGIN android-changed
174            // Exception priorities (in case of multiple errors) differ from
175            // RI, but are spec-compliant.
176            // removed redundant check, made implicit null check explicit,
177            // used (offset | length) < 0 instead of (offset < 0) || (length < 0)
178            // to safe one operation
179            if (b == null) {
180                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
181            }
182            if ((offset | length) < 0 || length > b.length - offset) {
183                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
184            }
185            // END android-changed
186            for (int i = 0; i < length; i++) {
187                int c;
188                try {
189                    if ((c = read()) == -1) {
190                        return i == 0 ? -1 : i;
191                    }
192                } catch (IOException e) {
193                    if (i != 0) {
194                        return i;
195                    }
196                    throw e;
197                }
198                b[offset + i] = (byte)c;
199            }
200            return length;
201        }
202    
203        /**
204         * Resets this stream to the last marked location. Throws an
205         * {@code IOException} if the number of bytes read since the mark has been
206         * set is greater than the limit provided to {@code mark}, or if no mark
207         * has been set.
208         * <p>
209         * This implementation always throws an {@code IOException} and concrete
210         * subclasses should provide the proper implementation.
211         * </p>
212         * 
213         * @throws IOException
214         *             if this stream is closed or another IOException occurs.
215         * @since Android 1.0
216         */
217        public synchronized void reset() throws IOException {
218            throw new IOException();
219        }
220    
221        /**
222         * Skips at most {@code n} bytes in this stream. It does nothing and returns
223         * 0 if {@code n} is negative. Less than {@code n} characters are skipped if
224         * the end of this stream is reached before the operation completes.
225         * <p>
226         * This default implementation reads {@code n} bytes into a temporary
227         * buffer. Concrete subclasses should provide their own implementation.
228         * </p>
229         * 
230         * @param n
231         *            the number of bytes to skip.
232         * @return the number of bytes actually skipped.
233         * @throws IOException
234         *             if this stream is closed or another IOException occurs.
235         * @since Android 1.0
236         */
237        public long skip(long n) throws IOException {
238            if (n <= 0) {
239                return 0;
240            }
241            long skipped = 0;
242            int toRead = n < 4096 ? (int) n : 4096;
243            if (skipBuf == null || skipBuf.length < toRead) {
244                skipBuf = new byte[toRead];
245            }
246            while (skipped < n) {
247                int read = read(skipBuf, 0, toRead);
248                if (read == -1) {
249                    return skipped;
250                }
251                skipped += read;
252                if (read < toRead) {
253                    return skipped;
254                }
255                if (n - skipped < toRead) {
256                    toRead = (int) (n - skipped);
257                }
258            }
259            return skipped;
260        }
261    }
