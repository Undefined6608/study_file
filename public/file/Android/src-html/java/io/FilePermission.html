018    package java.io;
019    
020    import java.security.AccessController;
021    import java.security.Permission;
022    import java.security.PermissionCollection;
023    import java.security.PrivilegedAction;
024    
025    import org.apache.harmony.luni.util.Msg;
026    
027    /**
028     * A permission for accessing a file or directory. The FilePermission is made up
029     * of a pathname and a set of actions which are valid for the pathname.
030     * <p>
031     * The {@code File.separatorChar} must be used in all pathnames when
032     * constructing a FilePermission. The following descriptions will assume the
033     * char is {@code /}. A pathname that ends in {@code /*} includes all the files
034     * and directories contained in that directory. If the pathname
035     * ends in {@code /-}, it includes all the files and directories in that
036     * directory <i>recursively</i>. The following pathnames have a special meaning:
037     * </p>
038     * <ul>
039     *   <li>
040     *     "*": all files in the current directory;
041     *   </li>
042     *   <li>
043     *     "-": recursively all files and directories in the current directory;
044     *   </li>
045     *   <li>
046     *     "&lt;&lt;ALL FILES&gt;&gt;": any file and directory in the file system.
047     *   </li>
048     * </ul>
049     * 
050     * @since Android 1.0
051     */
052    public final class FilePermission extends Permission implements Serializable {
053        
054        private static final long serialVersionUID = 7930732926638008763L;
055    
056        // canonical path of this permission
057        private transient String canonPath;
058    
059        // list of actions permitted for socket permission in order
060        private static final String[] actionList = { "read", "write", "execute", //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
061                "delete" }; //$NON-NLS-1$
062    
063        // "canonicalized" action list
064        private String actions;
065    
066        // the numeric representation of this action list
067        // for implies() to check if one action list is the subset of another.
068        transient int mask = -1;
069    
070        // global include all permission?
071        private transient boolean includeAll = false;
072    
073        private transient boolean allDir = false;
074    
075        private transient boolean allSubdir = false;
076    
077        /**
078         * Constructs a new FilePermission with the path and actions specified.
079         * 
080         * @param path
081         *            the pathname of the file or directory to apply the actions to.
082         * @param actions
083         *            the actions for the {@code path}. May be any combination of
084         *            "read", "write", "execute" and "delete".
085         * @throws IllegalArgumentException
086         *             if {@code actions} is {@code null} or an empty string, or if
087         *             it contains a string other than "read", "write", "execute"
088         *             and "delete".
089         * @throws NullPointerException
090         *             if {@code path} is null.
091         * @since Android 1.0
092         */
093        public FilePermission(String path, String actions) {
094            super(path);
095            init(path, actions);
096        }
097    
098        private void init(final String path, String pathActions) {
099            if (pathActions == null || pathActions.equals("")) { //$NON-NLS-1$
100                throw new IllegalArgumentException(Msg.getString("K006d")); //$NON-NLS-1$
101            }
102            this.actions = toCanonicalActionString(pathActions);
103    
104            if (path == null) {
105                throw new NullPointerException(Msg.getString("K006e")); //$NON-NLS-1$
106            }
107            if (path.equals("<<ALL FILES>>")) { //$NON-NLS-1$
108                includeAll = true;
109            } else {
110                canonPath = AccessController
111                        .doPrivileged(new PrivilegedAction<String>() {
112                            public String run() {
113                                try {
114                                    return new File(path).getCanonicalPath();
115                                } catch (IOException e) {
116                                    return path;
117                                }
118                            }
119                        });
120                if (path.equals("*") || path.endsWith(File.separator + "*")) { //$NON-NLS-1$ //$NON-NLS-2$
121                    allDir = true;
122                }
123                if (path.equals("-") || path.endsWith(File.separator + "-")) { //$NON-NLS-1$ //$NON-NLS-2$
124                    allSubdir = true;
125                }
126            }
127        }
128    
129        /**
130         * Returns the string representing this permission's actions. It must be of
131         * the form "read,write,execute,delete", all lower case and in the correct
132         * order if there is more than one action.
133         * 
134         * @param action
135         *            the action name
136         * @return the string representing this permission's actions
137         */
138        private String toCanonicalActionString(String action) {
139            actions = action.trim().toLowerCase();
140    
141            // get the numerical representation of the action list
142            mask = getMask(actions);
143    
144            // convert the mask to a canonical action list.
145            int len = actionList.length;
146            // the test mask - shift the 1 to the leftmost position of the
147            // actionList
148            int highestBitMask = 1 << (len - 1);
149    
150            // if a bit of mask is set, append the corresponding action to result
151            StringBuilder result = new StringBuilder();
152            boolean addedItem = false;
153            for (int i = 0; i < len; i++) {
154                if ((highestBitMask & mask) != 0) {
155                    if (addedItem) {
156                        result.append(","); //$NON-NLS-1$
157                    }
158                    result.append(actionList[i]);
159                    addedItem = true;
160                }
161                highestBitMask = highestBitMask >> 1;
162            }
163            return result.toString();
164        }
165    
166        /**
167         * Returns the numerical representation of the argument.
168         * 
169         * @param actionNames
170         *            the action names
171         * @return the action mask
172         */
173        private int getMask(String actionNames) {
174            int actionInt = 0, head = 0, tail = 0;
175            do {
176                tail = actionNames.indexOf(",", head); //$NON-NLS-1$
177                String action = tail > 0 ? actionNames.substring(head, tail).trim()
178                        : actionNames.substring(head).trim();
179                if (action.equals("read")) { //$NON-NLS-1$
180                    actionInt |= 8;
181                } else if (action.equals("write")) { //$NON-NLS-1$
182                    actionInt |= 4;
183                } else if (action.equals("execute")) { //$NON-NLS-1$
184                    actionInt |= 2;
185                } else if (action.equals("delete")) { //$NON-NLS-1$
186                    actionInt |= 1;
187                } else {
188                    throw new IllegalArgumentException(Msg.getString(
189                            "K006f", action)); //$NON-NLS-1$
190                }
191                head = tail + 1;
192            } while (tail > 0);
193            return actionInt;
194        }
195    
196        /**
197         * Returns the actions associated with this file permission.
198         * 
199         * @return the actions associated with this file permission.
200         * @since Android 1.0
201         */
202        @Override
203        public String getActions() {
204            return actions;
205        }
206    
207        /**
208         * Indicates if this file permission is equal to another. The two are equal
209         * if {@code obj} is a FilePermission, they have the same path, and they
210         * have the same actions.
211         * 
212         * @param obj
213         *            the object to check equality with.
214         * @return {@code true} if this file permission is equal to {@code obj},
215         *         {@code false} otherwise.
216         * @since Android 1.0
217         */
218        @Override
219        public boolean equals(Object obj) {
220            if (obj instanceof FilePermission) {
221                FilePermission fp = (FilePermission) obj;
222                if (fp.actions != actions) {
223                    if (fp.actions == null || !fp.actions.equals(actions)) {
224                        return false;
225                    }
226                }
227    
228                /* Matching actions and both are <<ALL FILES>> ? */
229                if (fp.includeAll || includeAll) {
230                    return fp.includeAll == includeAll;
231                }
232                return fp.canonPath.equals(canonPath);
233            }
234            return false;
235        }
236    
237        /**
238         * Indicates whether the permission {@code p} is implied by this file
239         * permission. This is the case if {@code p} is an instance of
240         * {@code FilePermission}, if {@code p}'s actions are a subset of this
241         * file permission's actions and if {@code p}'s path is implied by this
242         * file permission's path.
243         * 
244         * @param p
245         *            the permission to check.
246         * @return {@code true} if the argument permission is implied by the
247         *         receiver, and {@code false} if it is not.
248         * @since Android 1.0
249         */
250        @Override
251        public boolean implies(Permission p) {
252            int match = impliesMask(p);
253            return match != 0 && match == ((FilePermission) p).mask;
254        }
255    
256        /**
257         * Returns an int describing what masks are implied by a specific
258         * permission.
259         * 
260         * @param p
261         *            the permission
262         * @return the mask applied to the given permission
263         */
264        int impliesMask(Permission p) {
265            if (!(p instanceof FilePermission)) {
266                return 0;
267            }
268            FilePermission fp = (FilePermission) p;
269            int matchedMask = mask & fp.mask;
270            // Can't match any bits?
271            if (matchedMask == 0) {
272                return 0;
273            }
274    
275            // Is this permission <<ALL FILES>>
276            if (includeAll) {
277                return matchedMask;
278            }
279    
280            // We can't imply all files
281            if (fp.includeAll) {
282                return 0;
283            }
284    
285            // Scan the length of p checking all match possibilities
286            // \- implies everything except \
287            int thisLength = canonPath.length();
288            if (allSubdir && thisLength == 2
289                    && !fp.canonPath.equals(File.separator)) {
290                return matchedMask;
291            }
292            // need /- to imply /-
293            if (fp.allSubdir && !allSubdir) {
294                return 0;
295            }
296            // need /- or /* to imply /*
297            if (fp.allDir && !allSubdir && !allDir) {
298                return 0;
299            }
300    
301            boolean includeDir = false;
302            int pLength = fp.canonPath.length();
303            // do not compare the * or -
304            if (allDir || allSubdir) {
305                thisLength--;
306            }
307            if (fp.allDir || fp.allSubdir) {
308                pLength--;
309            }
310            for (int i = 0; i < pLength; i++) {
311                char pChar = fp.canonPath.charAt(i);
312                // Is p longer than this permissions canonLength?
313                if (i >= thisLength) {
314                    if (i == thisLength) {
315                        // Is this permission include all? (must have matched up
316                        // until this point).
317                        if (allSubdir) {
318                            return matchedMask;
319                        }
320                        // Is this permission include a dir? Continue the check
321                        // afterwards.
322                        if (allDir) {
323                            includeDir = true;
324                        }
325                    }
326                    // If not includeDir then is has to be a mismatch.
327                    if (!includeDir) {
328                        return 0;
329                    }
330                    /**
331                     * If we have * for this and find a separator it is invalid. IE:
332                     * this is '/a/*' and p is '/a/b/c' we should fail on the
333                     * separator after the b. Except for root, canonical paths do
334                     * not end in a separator.
335                     */
336                    if (pChar == File.separatorChar) {
337                        return 0;
338                    }
339                } else {
340                    // Are the characters matched?
341                    if (canonPath.charAt(i) != pChar) {
342                        return 0;
343                    }
344                }
345            }
346            // Must have matched up to this point or it's a valid file in an include
347            // all directory
348            if (pLength == thisLength) {
349                if (allSubdir) {
350                    // /- implies /- or /*
351                    return fp.allSubdir || fp.allDir ? matchedMask : 0;
352                }
353                return allDir == fp.allDir ? matchedMask : 0;
354            }
355            return includeDir ? matchedMask : 0;
356        }
357    
358        /**
359         * Returns a new PermissionCollection in which to place FilePermission
360         * objects.
361         * 
362         * @return A new PermissionCollection object suitable for storing
363         *         FilePermission objects.
364         * @since Android 1.0
365         */
366        @Override
367        public PermissionCollection newPermissionCollection() {
368            return new FilePermissionCollection();
369        }
370    
371        /**
372         * Calculates the hash code value for this file permission.
373         * 
374         * @return the hash code value for this file permission.
375         * @since Android 1.0
376         */
377        @Override
378        public int hashCode() {
379            return (canonPath == null ? getName().hashCode() : canonPath.hashCode())
380                    + mask;
381        }
382    
383        private void writeObject(ObjectOutputStream stream) throws IOException {
384            stream.defaultWriteObject();
385        }
386    
387        private void readObject(ObjectInputStream stream) throws IOException,
388                ClassNotFoundException {
389            stream.defaultReadObject();
390            init(getName(), actions);
391        }
