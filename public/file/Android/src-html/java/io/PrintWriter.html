032    /**
033     * Wraps either an existing {@link OutputStream} or an existing {@link Writer}
034     * and provides convenience methods for printing common data types in a human
035     * readable format. No {@code IOException} is thrown by this class. Instead,
036     * callers should use {@link #checkError()} to see if a problem has occurred in
037     * this writer.
038     * 
039     * @since Android 1.0
040     */

045         * @since Android 1.0

249        /**
250         * Flushes this writer and returns the value of the error flag.
251         * 
252         * @return {@code true} if either an {@code IOException} has been thrown
253         *         previously or if {@code setError()} has been called;
254         *         {@code false} otherwise.
255         * @see #setError()         
256         * @since Android 1.0
257         */
258        public boolean checkError() {
259            if (out != null) {
260                flush();
261            }
262            return ioError;
263        }
264    
265        /**
266         * Closes this print writer. Flushes this writer and then closes the target.
267         * If an I/O error occurs, this writer's error flag is set to {@code true}.
268         * 
269         * @since Android 1.0
270         */
271        @Override
272        public void close() {
273            synchronized (lock) {
274                if (out != null) {
275                    try {
276                        out.close();
277                    } catch (IOException e) {
278                        setError();
279                    }
280                    out = null;
281                }
282            }
283        }

285        /**
286         * Ensures that all pending data is sent out to the target. It also
287         * flushes the target. If an I/O error occurs, this writer's error
288         * state is set to {@code true}.
289         * 
290         * @since Android 1.0
291         */
292        @Override
293        public void flush() {
294            synchronized (lock) {
295                if (out != null) {
296                    try {
297                        out.flush();
298                    } catch (IOException e) {
299                        setError();
300                    }
301                } else {
302                    setError();
303                }
304            }
305        }

307        /**
308         * Writes a string formatted by an intermediate {@code Formatter} to the
309         * target using the specified format string and arguments. For the locale,
310         * the default value of the current virtual machine instance is used. If
311         * automatic flushing is enabled then the buffer is flushed as well.
312         * 
313         * @param format
314         *            the format string used for {@link java.util.Formatter#format}.
315         * @param args
316         *            the list of arguments passed to the formatter. If there are
317         *            more arguments than required by the {@code format} string,
318         *            then the additional arguments are ignored.
319         * @return this writer.
320         * @throws IllegalFormatException
321         *             if the format string is illegal or incompatible with the
322         *             arguments, if there are not enough arguments or if any other
323         *             error regarding the format string or arguments is detected.
324         * @throws NullPointerException
325         *             if {@code format} is {@code null}.
326         * @since Android 1.0
327         */
328        public PrintWriter format(String format, Object... args) {
329            return format(Locale.getDefault(), format, args);
330        }

332        /**
333         * Writes a string formatted by an intermediate {@code Formatter} to the
334         * target using the specified locale, format string and arguments. If
335         * automatic flushing is enabled then this writer is flushed.
336         * 
337         * @param l
338         *            the locale used in the method. No localization will be applied
339         *            if {@code l} is {@code null}.
340         * @param format
341         *            the format string used for {@link java.util.Formatter#format}.
342         * @param args
343         *            the list of arguments passed to the formatter. If there are
344         *            more arguments than required by the {@code format} string,
345         *            then the additional arguments are ignored.
346         * @return this writer.
347         * @throws IllegalFormatException
348         *             if the format string is illegal or incompatible with the
349         *             arguments, if there are not enough arguments or if any other
350         *             error regarding the format string or arguments is detected.
351         * @throws NullPointerException
352         *             if {@code format} is {@code null}.
353         * @since Android 1.0
354         */
355        public PrintWriter format(Locale l, String format, Object... args) {
356            if (format == null) {
357                throw new NullPointerException(Msg.getString("K0351")); //$NON-NLS-1$
358            }
359            new Formatter(this, l).format(format, args);
360            if (autoflush) {
361                flush();
362            }
363            return this;
364        }

417         * Print a new line String onto the writer, flushing if autoflush enabled.

420            print(lineSeparator);
421            if (autoflush) {
422                flush();
423            }
424        }

714        /**
715         * Writes {@code count} characters from {@code buffer} starting at {@code
716         * offset} to the target.
717         * <p>
718         * This writer's error flag is set to {@code true} if this writer is closed
719         * or an I/O error occurs.
720         * </p>
721         * 
722         * @param buf
723         *            the buffer to write to the target.
724         * @param offset
725         *            the index of the first character in {@code buffer} to write.
726         * @param count
727         *            the number of characters in {@code buffer} to write.
728         * @throws IndexOutOfBoundsException
729         *             if {@code offset < 0} or {@code count < 0}, or if {@code
730         *             offset + count} is greater than the length of {@code buf}.
731         * @since Android 1.0
732         */
733        @Override
734        public void write(char[] buf, int offset, int count) {
735            // BEGIN android-note

844         * target. This method works the same way as {@code
845         * PrintWriter.print(csq.subsequence(start, end).toString())}. If {@code
846         * csq} is {@code null}, then the specified subsequence of the string "null"

848         * 
849         * @param csq
850         *            the character sequence appended to the target.
851         * @param start
852         *            the index of the first char in the character sequence appended

855         *            the index of the character following the last character of the
856         *            subsequence appended to the target.
857         * @return this writer.
858         * @throws StringIndexOutOfBoundsException
859         *             if {@code start > end}, {@code start < 0}, {@code end < 0} or
860         *             either {@code start} or {@code end} are greater or equal than
861         *             the length of {@code csq}.
862         * @since Android 1.0
863         */
864        @Override
865        public PrintWriter append(CharSequence csq, int start, int end) {
866            if (null == csq) {
867                csq = TOKEN_NULL;
868            }
869            String output = csq.subSequence(start, end).toString();
870            write(output, 0, output.length());
871            return this;
872        }
