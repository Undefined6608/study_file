018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * Places information on a communications pipe. When two threads want to pass
024     * data back and forth, one creates a piped writer and the other creates a piped
025     * reader.
026     * 
027     * @see PipedReader
028     * 
029     * @since Android 1.0
030     */
031    public class PipedWriter extends Writer {
032        /**
033         * The destination PipedReader
034         */
035        private PipedReader dest;
036    
037        private boolean closed;
038    
039        /**
040         * Constructs a new unconnected {@code PipedWriter}. The resulting writer
041         * must be connected to a {@code PipedReader} before data may be written to
042         * it.
043         * 
044         * @see PipedReader
045         * @since Android 1.0
046         */
047        public PipedWriter() {
048            super();
049        }
050    
051        /**
052         * Constructs a new {@code PipedWriter} connected to the {@link PipedReader}
053         * {@code dest}. Any data written to this writer can be read from {@code
054         * dest}.
055         * 
056         * @param dest
057         *            the {@code PipedReader} to connect to.
058         * @throws IOException
059         *             if {@code dest} is already connected.
060         * @since Android 1.0
061         */
062        public PipedWriter(PipedReader dest) throws IOException {
063            super(dest);
064            connect(dest);
065        }
066    
067        /**
068         * Closes this writer. If a {@link PipedReader} is connected to this writer,
069         * it is closed as well and the pipe is disconnected. Any data buffered in
070         * the reader can still be read.
071         * 
072         * @throws IOException
073         *             if an error occurs while closing this writer.
074         * @since Android 1.0
075         */
076        @Override
077        public void close() throws IOException {
078            synchronized (lock) {
079                /* Is the pipe connected? */
080                if (dest != null) {
081                    dest.done();
082                    dest = null;
083                }
084                closed = true;
085            }
086        }
087    
088        /**
089         * Connects this {@code PipedWriter} to a {@link PipedReader}. Any data
090         * written to this writer becomes readable in the reader.
091         * 
092         * @param stream
093         *            the reader to connect to.
094         * @throws IOException
095         *             if this writer is closed or already connected, or if {@code
096         *             stream} is already connected.
097         * @since Android 1.0
098         */
099        public void connect(PipedReader stream) throws IOException {
100            synchronized (lock) {
101                if (this.dest != null) {
102                    throw new IOException(Msg.getString("K0079")); //$NON-NLS-1$
103                }
104                if (closed) {
105                    throw new IOException(Msg.getString("K0078")); //$NON-NLS-1$
106                }
107                stream.establishConnection();
108                this.dest = stream;
109            }
110        }
111    
112        /**
113         * Notifies the readers of this {@code PipedReader} that characters can be read. This
114         * method does nothing if this Writer is not connected.
115         * 
116         * @throws IOException
117         *             if an I/O error occurs while flushing this writer.
118         * @since Android 1.0
119         */
120        @Override
121        public void flush() throws IOException {
122            if (dest != null) {
123                dest.flush();
124            }
125        }
126    
127        /**
128         * Writes {@code count} characters from the character array {@code buffer}
129         * starting at offset {@code index} to this writer. The written data can
130         * then be read from the connected {@link PipedReader} instance.
131         * <p>
132         * Separate threads should be used to write to a {@code PipedWriter} and to
133         * read from the connected {@code PipedReader}. If the same thread is used,
134         * a deadlock may occur.
135         * </p>
136         * 
137         * @param buffer
138         *            the buffer to write.
139         * @param offset
140         *            the index of the first character in {@code buffer} to write.
141         * @param count
142         *            the number of characters from {@code buffer} to write to this
143         *            writer.
144         * @throws IndexOutOfBoundsException
145         *             if {@code offset < 0} or {@code count < 0}, or if {@code
146         *             offset + count} is bigger than the length of {@code buffer}.
147         * @throws InterruptedIOException
148         *             if the pipe is full and the current thread is interrupted
149         *             waiting for space to write data. This case is not currently
150         *             handled correctly.
151         * @throws IOException
152         *             if this writer is closed or not connected, if the target
153         *             reader is closed or if the thread reading from the target
154         *             reader is no longer alive. This case is currently not handled
155         *             correctly.
156         * @throws NullPointerException
157         *             if {@code buffer} is {@code null}.
158         * @since Android 1.0
159         */
160        @Override
161        public void write(char[] buffer, int offset, int count) throws IOException {
162            // BEGIN android-note
163            // changed array notation to be consistent with the rest of harmony
164            // END android-note
165            synchronized (lock) {
166                if (closed) {
167                    throw new IOException(Msg.getString("K0078")); //$NON-NLS-1$
168                }
169                if (dest == null) {
170                    throw new IOException(Msg.getString("K007b")); //$NON-NLS-1$
171                }
172                if (buffer == null) {
173                    throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
174                }
175    
176                // avoid int overflow
177                // BEGIN android-changed
178                // Exception priorities (in case of multiple errors) differ from
179                // RI, but are spec-compliant.
180                // removed redundant check, used (offset | count) < 0
181                // instead of (offset < 0) || (count < 0) to safe one operation
182                if ((offset | count) < 0 || count > buffer.length - offset) {
183                    throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
184                }
185                // END android-changed
186                dest.receive(buffer, offset, count);
187            }
188        }
189    
190        /**
191         * Writes a single character {@code c} to this writer. This character can
192         * then be read from the connected {@link PipedReader} instance.
193         * <p>
194         * Separate threads should be used to write to a {@code PipedWriter} and to
195         * read from the connected {@code PipedReader}. If the same thread is used,
196         * a deadlock may occur.
197         * </p>
198         * 
199         * @param c
200         *            the character to write.
201         * @throws InterruptedIOException
202         *             if the pipe is full and the current thread is interrupted
203         *             waiting for space to write data. This case is not currently
204         *             handled correctly.
205         * @throws IOException
206         *             if this writer is closed or not connected, if the target
207         *             reader is closed or if the thread reading from the target
208         *             reader is no longer alive. This case is currently not handled
209         *             correctly.
210         * @since Android 1.0
211         */
212        @Override
213        public void write(int c) throws IOException {
214            synchronized (lock) {
215                if (closed) {
216                    throw new IOException(Msg.getString("K0078")); //$NON-NLS-1$
217                }
218                if (dest == null) {
219                    throw new IOException(Msg.getString("K007b")); //$NON-NLS-1$
220                }
221                dest.receive((char) c);
222            }
223        }
224    }
