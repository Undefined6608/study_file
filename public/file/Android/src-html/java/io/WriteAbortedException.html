018    package java.io;
019    
020    /**
021     * Signals that the {@link ObjectInputStream#readObject()} method has detected
022     * an exception marker in the input stream. This marker indicates that exception
023     * occurred when the object was serialized, and this marker was inserted instead
024     * of the original object. It is a way to "propagate" an exception from the code
025     * that attempted to write the object to the code that is attempting to read the
026     * object.
027     * 
028     * @see ObjectInputStream#readObject()
029     * 
030     * @since Android 1.0
031     */
032    public class WriteAbortedException extends ObjectStreamException {
033    
034        private static final long serialVersionUID = -3326426625597282442L;
035    
036        /**
037         * The exception that occured when writeObject() was attempting to serialize
038         * the object.
039         * 
040         * @since Android 1.0
041         */
042        public Exception detail;
043    
044        /**
045         * Constructs a new {@code WriteAbortedException} with its stack trace,
046         * detail message and the exception which caused the underlying problem when
047         * serializing the object filled in.
048         * 
049         * @param detailMessage
050         *            the detail message for this exception.
051         * @param rootCause
052         *            the exception that was thrown when serializing the object.
053         * @since Android 1.0
054         */
055        public WriteAbortedException(String detailMessage, Exception rootCause) {
056            super(detailMessage);
057            detail = rootCause;
058            initCause(rootCause);
059        }
060    
061        /**
062         * Gets the extra information message which was provided when this exception
063         * was created. Returns {@code null} if no message was provided at creation
064         * time.
065         * 
066         * @return the exception message.
067         * @since Android 1.0
068         */
069        @Override
070        public String getMessage() {
071            String msg = super.getMessage();
072            if (detail != null) {
073                msg = msg + "; " + detail.toString(); //$NON-NLS-1$
074            }
075            return msg;
076        }
077    
078        /**
079         * Gets the cause of this exception or {@code null} if there is no cause.
080         * 
081         * @return the exception cause.
082         * @since Android 1.0
083         */
084        @Override
085        public Throwable getCause() {
086            return detail;
087        }
088    }
