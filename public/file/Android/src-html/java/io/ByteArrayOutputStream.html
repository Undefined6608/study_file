018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * A specialized {@link OutputStream} for class for writing content to an
024     * (internal) byte array. As bytes are written to this stream, the byte array
025     * may be expanded to hold more bytes. When the writing is considered to be
026     * finished, a copy of the byte array can be requested from the class.
027     * 
028     * @see ByteArrayInputStream
029     * 
030     * @since Android 1.0
031     */
032    public class ByteArrayOutputStream extends OutputStream {
033        /**
034         * The byte array containing the bytes written.
035         * 
036         * @since Android 1.0
037         */
038        protected byte[] buf;
039    
040        /**
041         * The number of bytes written.
042         * 
043         * @since Android 1.0
044         */
045        protected int count;
046    
047        /**
048         * Constructs a new ByteArrayOutputStream with a default size of 32 bytes.
049         * If more than 32 bytes are written to this instance, the underlying byte
050         * array will expand.
051         * 
052         * @since Android 1.0
053         */
054        public ByteArrayOutputStream() {
055            super();
056            buf = new byte[32];
057        }
058    
059        /**
060         * Constructs a new {@code ByteArrayOutputStream} with a default size of
061         * {@code size} bytes. If more than {@code size} bytes are written to this
062         * instance, the underlying byte array will expand.
063         * 
064         * @param size
065         *            initial size for the underlying byte array, must be
066         *            non-negative.
067         * @throws IllegalArgumentException
068         *             if {@code size} < 0.
069         * @since Android 1.0
070         */
071        public ByteArrayOutputStream(int size) {
072            super();
073            if (size >= 0) {
074                buf = new byte[size];
075            } else {
076                throw new IllegalArgumentException(Msg.getString("K005e")); //$NON-NLS-1$
077            }
078        }
079    
080        /**
081         * Closes this stream. This releases system resources used for this stream.
082         * 
083         * @throws IOException
084         *             if an error occurs while attempting to close this stream.
085         * @since Android 1.0
086         */
087        @Override
088        public void close() throws IOException {
089            /**
090             * Although the spec claims "A closed stream cannot perform output
091             * operations and cannot be reopened.", this implementation must do
092             * nothing.
093             */
094            super.close();
095        }
096    
097        private void expand(int i) {
098            /* Can the buffer handle @i more bytes, if not expand it */
099            if (count + i <= buf.length) {
100                return;
101            }
102    
103            byte[] newbuf = new byte[(count + i) * 2];
104            System.arraycopy(buf, 0, newbuf, 0, count);
105            buf = newbuf;
106        }
107    
108        /**
109         * Resets this stream to the beginning of the underlying byte array. All
110         * subsequent writes will overwrite any bytes previously stored in this
111         * stream.
112         * 
113         * @since Android 1.0
114         */
115        public synchronized void reset() {
116            count = 0;
117        }
118    
119        /**
120         * Returns the total number of bytes written to this stream so far.
121         * 
122         * @return the number of bytes written to this stream.
123         * @since Android 1.0
124         */
125        public int size() {
126            return count;
127        }
128    
129        /**
130         * Returns the contents of this ByteArrayOutputStream as a byte array. Any
131         * changes made to the receiver after returning will not be reflected in the
132         * byte array returned to the caller.
133         * 
134         * @return this stream's current contents as a byte array.
135         * @since Android 1.0
136         */
137        public synchronized byte[] toByteArray() {
138            byte[] newArray = new byte[count];
139            System.arraycopy(buf, 0, newArray, 0, count);
140            return newArray;
141        }
142    
143        /**
144         * Returns the contents of this ByteArrayOutputStream as a string. Any
145         * changes made to the receiver after returning will not be reflected in the
146         * string returned to the caller.
147         * 
148         * @return this stream's current contents as a string.
149         * @since Android 1.0
150         */
151    
152        @Override
153        public String toString() {
154            return new String(buf, 0, count);
155        }
156    
157        /**
158         * Returns the contents of this ByteArrayOutputStream as a string. Each byte
159         * {@code b} in this stream is converted to a character {@code c} using the
160         * following function:
161         * {@code c == (char)(((hibyte & 0xff) << 8) | (b & 0xff))}. This method is
162         * deprecated and either {@link #toString()} or {@link #toString(String)}
163         * should be used.
164         * 
165         * @param hibyte
166         *            the high byte of each resulting Unicode character.
167         * @return this stream's current contents as a string with the high byte set
168         *         to {@code hibyte}.
169         * @deprecated Use {@link #toString()}.
170         * @since Android 1.0
171         */
172        @Deprecated
173        public String toString(int hibyte) {
174            char[] newBuf = new char[size()];
175            for (int i = 0; i < newBuf.length; i++) {
176                newBuf[i] = (char) (((hibyte & 0xff) << 8) | (buf[i] & 0xff));
177            }
178            return new String(newBuf);
179        }
180    
181        /**
182         * Returns the contents of this ByteArrayOutputStream as a string converted
183         * according to the encoding declared in {@code enc}.
184         * 
185         * @param enc
186         *            a string representing the encoding to use when translating
187         *            this stream to a string.
188         * @return this stream's current contents as an encoded string.
189         * @throws UnsupportedEncodingException
190         *             if the provided encoding is not supported.
191         * @since Android 1.0
192         */
193        public String toString(String enc) throws UnsupportedEncodingException {
194            return new String(buf, 0, count, enc);
195        }
196    
197        /**
198         * Writes {@code count} bytes from the byte array {@code buffer} starting at
199         * offset {@code index} to this stream.
200         * 
201         * @param buffer
202         *            the buffer to be written.
203         * @param offset
204         *            the initial position in {@code buffer} to retrieve bytes.
205         * @param len
206         *            the number of bytes of {@code buffer} to write.
207         * @throws IndexOutOfBoundsException
208         *             if {@code offset < 0} or {@code len < 0}, or if
209         *             {@code offset + len} is greater than the length of
210         *             {@code buffer}.
211         * @since Android 1.0
212         */
213        @Override
214        public synchronized void write(byte[] buffer, int offset, int len) {
215            // avoid int overflow
216            // BEGIN android-changed
217            // Exception priorities (in case of multiple errors) differ from
218            // RI, but are spec-compliant.
219            // removed redundant check, made implicit null check explicit,
220            // used (offset | len) < 0 instead of (offset < 0) || (len < 0)
221            // to safe one operation
222            if (buffer == null) {
223                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
224            }
225            if ((offset | len) < 0 || len > buffer.length - offset) {
226                throw new IndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
227            }
228            // END android-changed
229            if (len == 0) {
230                return;
231            }
232    
233            /* Expand if necessary */
234            expand(len);
235            System.arraycopy(buffer, offset, buf, this.count, len);
236            this.count += len;
237        }
238    
239        /**
240         * Writes the specified byte {@code oneByte} to the OutputStream. Only the
241         * low order byte of {@code oneByte} is written.
242         * 
243         * @param oneByte
244         *            the byte to be written.
245         * @since Android 1.0
246         */
247        @Override
248        public synchronized void write(int oneByte) {
249            if (count == buf.length) {
250                expand(1);
251            }
252            buf[count++] = (byte)oneByte;
253        }
254    
255        /**
256         * Takes the contents of this stream and writes it to the output stream
257         * {@code out}.
258         * 
259         * @param out
260         *            an OutputStream on which to write the contents of this stream.
261         * @throws IOException
262         *             if an error occurs while writing to {@code out}.
263         * @since Android 1.0
264         */
265        public synchronized void writeTo(OutputStream out) throws IOException {
266            out.write(buf, 0, count);
267        }
268    }
