018    package java.io;
019    
020    // BEGIN android-added
021    import org.apache.harmony.luni.util.Msg;
022    // END android-added
023    
024    /**
025     * Wraps an existing {@link InputStream} and counts the line terminators
026     * encountered while reading the data. Line numbering starts at 0. Recognized
027     * line terminator sequences are {@code '\r'}, {@code '\n'} and {@code "\r\n"}.
028     * When using {@code read}, line terminator sequences are always translated into
029     * {@code '\n'}.
030     * 
031     * @deprecated Use {@link LineNumberReader}
032     * 
033     * @since Android 1.0
034     */
035    @Deprecated
036    public class LineNumberInputStream extends FilterInputStream {
037    
038        private int lineNumber;
039    
040        private int markedLineNumber = -1;
041    
042        private int lastChar = -1;
043    
044        private int markedLastChar;
045    
046        /**
047         * Constructs a new {@code LineNumberInputStream} on the {@link InputStream}
048         * {@code in}. Line numbers are counted for all data read from this stream.
049         * 
050         * @param in
051         *            The non-null input stream to count line numbers.
052         * @since Android 1.0
053         */
054        public LineNumberInputStream(InputStream in) {
055            super(in);
056        }
057    
058        /**
059         * Returns the number of bytes that are available before this stream will
060         * block.
061         * <p>
062         * Note: The source stream may just be a sequence of {@code "\r\n"} bytes
063         * which are converted into {@code '\n'} by this stream. Therefore,
064         * {@code available} returns only {@code in.available() / 2} bytes as
065         * result.
066         * </p>
067         * 
068         * @return the guaranteed number of bytes available before blocking.
069         * @throws IOException
070         *             if an error occurs in this stream.
071         * @since Android 1.0
072         */
073        @Override
074        public int available() throws IOException {
075            return in.available() / 2 + (lastChar == -1 ? 0 : 1);
076        }
077    
078        /**
079         * Returns the current line number for this stream. Numbering starts at 0.
080         * 
081         * @return the current line number.
082         * @since Android 1.0
083         */
084        public int getLineNumber() {
085            return lineNumber;
086        }
087    
088        /**
089         * Sets a mark position in this stream. The parameter {@code readlimit}
090         * indicates how many bytes can be read before the mark is invalidated.
091         * Sending {@code reset()} will reposition this stream back to the marked
092         * position, provided that {@code readlimit} has not been surpassed.
093         * The line number count will also be reset to the last marked
094         * line number count.
095         * <p>
096         * This implementation sets a mark in the filtered stream.
097         * </p>
098         * 
099         * @param readlimit
100         *            the number of bytes that can be read from this stream before
101         *            the mark is invalidated.
102         * @see #markSupported()
103         * @see #reset()
104         * @since Android 1.0
105         */
106        @Override
107        public void mark(int readlimit) {
108            in.mark(readlimit);
109            markedLineNumber = lineNumber;
110            markedLastChar = lastChar;
111        }
112    
113        /**
114         * Reads a single byte from the filtered stream and returns it as an integer
115         * in the range from 0 to 255. Returns -1 if the end of this stream has been
116         * reached.
117         * <p>
118         * The line number count is incremented if a line terminator is encountered.
119         * Recognized line terminator sequences are {@code '\r'}, {@code '\n'} and
120         * {@code "\r\n"}. Line terminator sequences are always translated into
121         * {@code '\n'}.
122         * </p>
123         * 
124         * @return the byte read or -1 if the end of the filtered stream has been
125         *         reached.
126         * @throws IOException
127         *             if the stream is closed or another IOException occurs.
128         * @since Android 1.0
129         */
130        @Override
131        public int read() throws IOException {
132            int currentChar = lastChar;
133            if (currentChar == -1) {
134                currentChar = in.read();
135            } else {
136                lastChar = -1;
137            }
138            switch (currentChar) {
139                case '\r':
140                    currentChar = '\n';
141                    lastChar = in.read();
142                    if (lastChar == '\n') {
143                        lastChar = -1;
144                    }
145                    // fall through
146                case '\n':
147                    lineNumber++;
148            }
149            return currentChar;
150        }
151    
152        /**
153         * Reads at most {@code length} bytes from the filtered stream and stores
154         * them in the byte array {@code buffer} starting at {@code offset}.
155         * Returns the number of bytes actually read or -1 if no bytes have been
156         * read and the end of this stream has been reached.
157         * <p>
158         * The line number count is incremented if a line terminator is encountered.
159         * Recognized line terminator sequences are {@code '\r'}, {@code '\n'} and
160         * {@code "\r\n"}. Line terminator sequences are always translated into
161         * {@code '\n'}.
162         * </p>
163         * 
164         * @param buffer
165         *            the array in which to store the bytes read.
166         * @param offset
167         *            the initial position in {@code buffer} to store the bytes read
168         *            from this stream.
169         * @param length
170         *            the maximum number of bytes to store in {@code buffer}.
171         * @return the number of bytes actually read or -1 if the end of the
172         *         filtered stream has been reached while reading.
173         * @throws IndexOutOfBoundsException
174         *             if {@code offset < 0} or {@code length < 0}, or if
175         *             {@code offset + length} is greater than the length of
176         *             {@code buffer}.
177         * @throws IOException
178         *             if this stream is closed or another IOException occurs.
179         * @throws NullPointerException
180         *             if {@code buffer} is {@code null}.
181         * @since Android 1.0
182         */
183        @Override
184        public int read(byte[] buffer, int offset, int length) throws IOException {
185            // BEGIN android-changed
186            if (buffer == null) {
187                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
188            }
189            // avoid int overflow
190            // Exception priorities (in case of multiple errors) differ from
191            // RI, but are spec-compliant.
192            // removed redundant check, used (offset | length) < 0
193            // instead of (offset < 0) || (length < 0) to safe one operation
194            if ((offset | length) < 0 || length > buffer.length - offset) {
195                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
196            }
197            // END android-changed
198    
199            for (int i = 0; i < length; i++) {
200                int currentChar;
201                try {
202                    currentChar = read();
203                } catch (IOException e) {
204                    if (i != 0) {
205                        return i;
206                    }
207                    throw e;
208                }
209                if (currentChar == -1) {
210                    return i == 0 ? -1 : i;
211                }
212                buffer[offset + i] = (byte) currentChar;
213            }
214            return length;
215        }
216    
217        /**
218         * Resets this stream to the last marked location. It also resets the line
219         * count to what is was when this stream was marked.
220         * 
221         * @throws IOException
222         *             if this stream is already closed, no mark has been set or the
223         *             mark is no longer valid because more than {@code readlimit}
224         *             bytes have been read since setting the mark.
225         * @see #mark(int)
226         * @see #markSupported()
227         * @since Android 1.0
228         */
229        @Override
230        public void reset() throws IOException {
231            in.reset();
232            lineNumber = markedLineNumber;
233            lastChar = markedLastChar;
234        }
235    
236        /**
237         * Sets the line number of this stream to the specified
238         * {@code lineNumber}. Note that this may have side effects on the
239         * line number associated with the last marked position.
240         * 
241         * @param lineNumber
242         *            the new lineNumber value.
243         * @see #mark(int)
244         * @see #reset()
245         * @since Android 1.0
246         */
247        public void setLineNumber(int lineNumber) {
248            this.lineNumber = lineNumber;
249        }
250    
251        /**
252         * Skips {@code count} number of bytes in this stream. Subsequent
253         * {@code read()}'s will not return these bytes unless {@code reset()} is
254         * used. This implementation skips {@code count} number of bytes in the
255         * filtered stream and increments the line number count whenever line
256         * terminator sequences are skipped.
257         * 
258         * @param count
259         *            the number of bytes to skip.
260         * @return the number of bytes actually skipped.
261         * @throws IOException
262         *             if this stream is closed or another IOException occurs.
263         * @see #mark(int)
264         * @see #read()
265         * @see #reset()
266         * @since Android 1.0
267         */
268        @Override
269        public long skip(long count) throws IOException {
270            if (count <= 0) {
271                return 0;
272            }
273            for (int i = 0; i < count; i++) {
274                int currentChar = read();
275                if (currentChar == -1) {
276                    return i;
277                }
278            }
279            return count;
280        }
281    }
