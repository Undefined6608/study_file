018    package java.io;
019    
020    /**
021     * Wraps an existing {@link Reader} and performs some transformation on the
022     * input data while it is being read. Transformations can be anything from a
023     * simple byte-wise filtering input data to an on-the-fly compression or
024     * decompression of the underlying reader. Readers that wrap another reader and
025     * provide some additional functionality on top of it usually inherit from this
026     * class.
027     * 
028     * @see FilterWriter
029     * 
030     * @since Android 1.0
031     */
032    public abstract class FilterReader extends Reader {
033    
034        /**
035         * The target Reader which is being filtered.
036         * 
037         * @since Android 1.0
038         */
039        protected Reader in;
040    
041        /**
042         * Constructs a new FilterReader on the Reader {@code in}.
043         * 
044         * @param in
045         *            The non-null Reader to filter reads on.
046         * @since Android 1.0
047         */
048        protected FilterReader(Reader in) {
049            super(in);
050            this.in = in;
051        }
052    
053        /**
054         * Closes this reader. This implementation closes the filtered reader.
055         * 
056         * @throws IOException
057         *             if an error occurs while closing this reader.
058         * @since Android 1.0
059         */
060        @Override
061        public void close() throws IOException {
062            synchronized (lock) {
063                in.close();
064            }
065        }
066    
067        /**
068         * Sets a mark position in this reader. The parameter {@code readlimit}
069         * indicates how many bytes can be read before the mark is invalidated.
070         * Sending {@code reset()} will reposition this reader back to the marked
071         * position, provided that {@code readlimit} has not been surpassed.
072         * <p>
073         * This implementation sets a mark in the filtered reader.
074         * </p>
075         * 
076         * @param readlimit
077         *            the number of bytes that can be read from this reader before
078         *            the mark is invalidated.
079         * @throws IOException
080         *             if an error occurs while marking this reader.
081         * @see #markSupported()
082         * @see #reset()
083         * @since Android 1.0
084         */
085        @Override
086        public synchronized void mark(int readlimit) throws IOException {
087            synchronized (lock) {
088                in.mark(readlimit);
089            }
090        }
091    
092        /**
093         * Indicates whether this reader supports {@code mark()} and {@code reset()}.
094         * This implementation returns whether the filtered reader supports marking.
095         * 
096         * @return {@code true} if {@code mark()} and {@code reset()} are supported
097         *         by the filtered reader, {@code false} otherwise.
098         * @see #mark(int)
099         * @see #reset()
100         * @see #skip(long)
101         * @since Android 1.0
102         */
103        @Override
104        public boolean markSupported() {
105            synchronized (lock) {
106                return in.markSupported();
107            }
108        }
109    
110        /**
111         * Reads a single character from the filtered reader and returns it as an
112         * integer with the two higher-order bytes set to 0. Returns -1 if the end
113         * of the filtered reader has been reached.
114         * 
115         * @return The character read or -1 if the end of the filtered reader has
116         *         been reached.
117         * @throws IOException
118         *             if an error occurs while reading from this reader.
119         * @since Android 1.0
120         */
121        @Override
122        public int read() throws IOException {
123            synchronized (lock) {
124                return in.read();
125            }
126        }
127    
128        /**
129         * Reads at most {@code count} characters from the filtered reader and stores them
130         * in the byte array {@code buffer} starting at {@code offset}. Returns the
131         * number of characters actually read or -1 if no characters were read and
132         * the end of the filtered reader was encountered.
133         * 
134         * @param buffer
135         *            the char array in which to store the characters read.
136         * @param offset
137         *            the initial position in {@code buffer} to store the characters
138         *            read from this reader.
139         * @param count
140         *            the maximum number of characters to store in {@code buffer}.
141         * @return the number of characters actually read or -1 if the end of the
142         *         filtered reader has been reached while reading.
143         * @throws IOException
144         *             if an error occurs while reading from this reader.
145         * @since Android 1.0
146         */
147        @Override
148        public int read(char[] buffer, int offset, int count) throws IOException {
149            synchronized (lock) {
150                return in.read(buffer, offset, count);
151            }
152        }
153    
154        /**
155         * Indicates whether this reader is ready to be read without blocking. If
156         * the result is {@code true}, the next {@code read()} will not block. If
157         * the result is {@code false}, this reader may or may not block when
158         * {@code read()} is sent.
159         * 
160         * @return {@code true} if this reader will not block when {@code read()}
161         *         is called, {@code false} if unknown or blocking will occur.
162         * @throws IOException
163         *             if the reader is closed or some other I/O error occurs.
164         * @since Android 1.0
165         */
166        @Override
167        public boolean ready() throws IOException {
168            synchronized (lock) {
169                return in.ready();
170            }
171        }
172    
173        /**
174         * Resets this reader's position to the last marked location. Invocations of
175         * {@code read()} and {@code skip()} will occur from this new location. If
176         * this reader was not marked, the behavior depends on the implementation of
177         * {@code reset()} in the Reader subclass that is filtered by this reader.
178         * The default behavior for Reader is to throw an {@code IOException}.
179         * 
180         * @throws IOException
181         *             if a problem occurred or the filtered reader does not support
182         *             {@code mark()} and {@code reset()}.
183         * @see #mark(int)
184         * @see #markSupported()
185         * @since Android 1.0
186         */
187        @Override
188        public void reset() throws IOException {
189            synchronized (lock) {
190                in.reset();
191            }
192        }
193    
194        /**
195         * Skips {@code count} characters in this reader. Subsequent {@code read()}'s
196         * will not return these characters unless {@code reset()} is used. The
197         * default implementation is to skip characters in the filtered reader.
198         * 
199         * @param count
200         *            the maximum number of characters to skip.
201         * @return the number of characters actually skipped.
202         * @throws IOException
203         *             if the filtered reader is closed or some other I/O error
204         *             occurs.
205         * @see #mark(int)
206         * @see #markSupported()
207         * @see #reset()
208         * @since Android 1.0
209         */
210        @Override
211        public long skip(long count) throws IOException {
212            synchronized (lock) {
213                return in.skip(count);
214            }
215        }
216    }
