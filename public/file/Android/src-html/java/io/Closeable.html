017    package java.io;
018    
019    /**
020     * Defines an interface for classes that can (or need to) be closed once they
021     * are not used any longer. This usually includes all sorts of
022     * {@link InputStream}s and {@link OutputStream}s. Calling the {@code close}
023     * method releases resources that the object holds.
024     * 
025     * @since Android 1.0
026     */
027    public interface Closeable {
028    
029        /**
030         * Closes the object and release any system resources it holds. If the
031         * object has already been closed, then invoking this method has no effect.
032         * 
033         * @throws IOException
034         *             if any error occurs when closing the object.
035         * @since Android 1.0
036         */
037        public void close() throws IOException;
038    }
