018    package java.io;
019    
020    import org.apache.harmony.luni.util.Msg;
021    
022    /**
023     * A specialized {@link Reader} for reading the contents of a char array.
024     * 
025     * @see CharArrayWriter
026     * 
027     * @since Android 1.0
028     */
029    public class CharArrayReader extends Reader {
030        /**
031         * The buffer for characters.
032         * 
033         * @since Android 1.0 
034         */
035        protected char buf[];
036    
037        /**
038         * The current buffer position.
039         * 
040         * @since Android 1.0 
041         */
042        protected int pos;
043    
044        /**
045         * The current mark position.
046         * 
047         * @since Android 1.0
048         */
049        protected int markedPos = -1;
050    
051        /**
052         * The ending index of the buffer.
053         *
054         * @since Android 1.0 
055         */
056        protected int count;
057    
058        /**
059         * Constructs a CharArrayReader on the char array {@code buf}. The size of
060         * the reader is set to the length of the buffer and the object to to read
061         * from is set to {@code buf}.
062         * 
063         * @param buf
064         *            the char array from which to read.
065         * @since Android 1.0
066         */
067        public CharArrayReader(char[] buf) {
068            super(buf);
069            this.buf = buf;
070            this.count = buf.length;
071        }
072    
073        /**
074         * Constructs a CharArrayReader on the char array {@code buf}. The size of
075         * the reader is set to {@code length} and the start position from which to
076         * read the buffer is set to {@code offset}.
077         * 
078         * @param buf
079         *            the char array from which to read.
080         * @param offset
081         *            the index of the first character in {@code buf} to read.
082         * @param length
083         *            the number of characters that can be read from {@code buf}.
084         * @throws IllegalArgumentException
085         *             if {@code offset < 0} or {@code length < 0}, or if
086         *             {@code offset} is greater than the size of {@code buf} .
087         * @since Android 1.0
088         */
089        public CharArrayReader(char[] buf, int offset, int length) {
090            super(buf);
091            // BEGIN android-changed
092            // Exception priorities (in case of multiple errors) differ from
093            // RI, but are spec-compliant.
094            // made implicit null check explicit,
095            // removed redundant check, used (offset | length) < 0 instead of 
096            // (offset < 0) || (length < 0) to safe one operation
097            if (buf == null) {
098                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
099            }
100            if ((offset | length) < 0 || offset > buf.length) {
101                throw new IllegalArgumentException(Msg.getString("K002f")); //$NON-NLS-1$
102            }
103            // END android-changed
104            this.buf = buf;
105            this.pos = offset;
106    
107            /* This is according to spec */
108            this.count = this.pos + length < buf.length ? length : buf.length;
109        }
110    
111        /**
112         * This method closes this CharArrayReader. Once it is closed, you can no
113         * longer read from it. Only the first invocation of this method has any
114         * effect.
115         * 
116         * @since Android 1.0
117         */
118        @Override
119        public void close() {
120            synchronized (lock) {
121                if (isOpen()) {
122                    buf = null;
123                }
124            }
125        }
126    
127        /**
128         * Indicates whether this reader is open.
129         * 
130         * @return {@code true} if the reader is open, {@code false} otherwise.
131         */
132        private boolean isOpen() {
133            return buf != null;
134        }
135    
136        /**
137         * Indicates whether this reader is closed.
138         * 
139         * @return {@code true} if the reader is closed, {@code false} otherwise.
140         */
141        private boolean isClosed() {
142            return buf == null;
143        }
144    
145        /**
146         * Sets a mark position in this reader. The parameter {@code readLimit} is
147         * ignored for CharArrayReaders. Calling {@code reset()} will reposition the
148         * reader back to the marked position provided the mark has not been
149         * invalidated.
150         * 
151         * @param readLimit
152         *            ignored for CharArrayReaders.
153         * @throws IOException
154         *             if this reader is closed.
155         * @since Android 1.0
156         */
157        @Override
158        public void mark(int readLimit) throws IOException {
159            synchronized (lock) {
160                if (isClosed()) {
161                    throw new IOException(Msg.getString("K0060")); //$NON-NLS-1$
162                }
163                markedPos = pos;
164            }
165        }
166    
167        /**
168         * Indicates whether this reader supports the {@code mark()} and
169         * {@code reset()} methods.
170         * 
171         * @return {@code true} for CharArrayReader.
172         * @see #mark(int)
173         * @see #reset()
174         * @since Android 1.0
175         */
176        @Override
177        public boolean markSupported() {
178            return true;
179        }
180    
181        /**
182         * Reads a single character from this reader and returns it as an integer
183         * with the two higher-order bytes set to 0. Returns -1 if no more
184         * characters are available from this reader.
185         * 
186         * @return the character read as an int or -1 if the end of the reader has
187         *         been reached.
188         * @throws IOException
189         *             if this reader is closed.
190         * @since Android 1.0
191         */
192        @Override
193        public int read() throws IOException {
194            synchronized (lock) {
195                if (isClosed()) {
196                    throw new IOException(Msg.getString("K0060")); //$NON-NLS-1$
197                }
198                if (pos == count) {
199                    return -1;
200                }
201                return buf[pos++];
202            }
203        }
204    
205        /**
206         * Reads at most {@code count} characters from this CharArrayReader and
207         * stores them at {@code offset} in the character array {@code buf}.
208         * Returns the number of characters actually read or -1 if the end of reader
209         * was encountered.
210         * 
211         * @param buffer
212         *            the character array to store the characters read.
213         * @param offset
214         *            the initial position in {@code buffer} to store the characters
215         *            read from this reader.
216         * @param len
217         *            the maximum number of characters to read.
218         * @return number of characters read or -1 if the end of the reader has been
219         *         reached.
220         * @throws IndexOutOfBoundsException
221         *             if {@code offset < 0} or {@code len < 0}, or if
222         *             {@code offset + len} is bigger than the size of
223         *             {@code buffer}.
224         * @throws IOException
225         *             if this reader is closed.
226         * @since Android 1.0
227         */
228        @Override
229        public int read(char[] buffer, int offset, int len) throws IOException {
230            // BEGIN android-note
231            // changed array notation to be consistent with the rest of harmony
232            // END android-note
233            // avoid int overflow
234            // BEGIN android-changed
235            // Exception priorities (in case of multiple errors) differ from
236            // RI, but are spec-compliant.
237            // made implicit null check explicit,
238            // removed redundant check, used (offset | len) < 0 instead of
239            // (offset < 0) || (len < 0) to safe one operation
240            if (buffer == null) {
241                throw new NullPointerException(Msg.getString("K0047")); //$NON-NLS-1$
242            }
243            if ((offset | len) < 0 || len > buffer.length - offset) {
244                throw new ArrayIndexOutOfBoundsException(Msg.getString("K002f")); //$NON-NLS-1$
245            }
246            // END android-changed
247            synchronized (lock) {
248                if (isClosed()) {
249                    throw new IOException(Msg.getString("K0060")); //$NON-NLS-1$
250                }
251                if (pos < this.count) {
252                    int bytesRead = pos + len > this.count ? this.count - pos : len;
253                    System.arraycopy(this.buf, pos, buffer, offset, bytesRead);
254                    pos += bytesRead;
255                    return bytesRead;
256                }
257                return -1;
258            }
259        }
260    
261        /**
262         * Indicates whether this reader is ready to be read without blocking.
263         * Returns {@code true} if the next {@code read} will not block. Returns
264         * {@code false} if this reader may or may not block when {@code read} is
265         * called. The implementation in CharArrayReader always returns {@code true}
266         * even when it has been closed.
267         * 
268         * @return {@code true} if this reader will not block when {@code read} is
269         *         called, {@code false} if unknown or blocking will occur.
270         * @throws IOException
271         *             if this reader is closed.
272         * @since Android 1.0
273         */
274        @Override
275        public boolean ready() throws IOException {
276            synchronized (lock) {
277                if (isClosed()) {
278                    throw new IOException(Msg.getString("K0060")); //$NON-NLS-1$
279                }
280                return pos != count;
281            }
282        }
283    
284        /**
285         * Resets this reader's position to the last {@code mark()} location.
286         * Invocations of {@code read()} and {@code skip()} will occur from this new
287         * location. If this reader has not been marked, it is reset to the
288         * beginning of the string.
289         * 
290         * @throws IOException
291         *             if this reader is closed.
292         * @since Android 1.0
293         */
294        @Override
295        public void reset() throws IOException {
296            synchronized (lock) {
297                if (isClosed()) {
298                    throw new IOException(Msg.getString("K0060")); //$NON-NLS-1$
299                }
300                pos = markedPos != -1 ? markedPos : 0;
301            }
302        }
303    
304        /**
305         * Skips {@code count} number of characters in this reader. Subsequent
306         * {@code read()}s will not return these characters unless {@code reset()}
307         * is used. This method does nothing and returns 0 if {@code n} is negative.
308         * 
309         * @param n
310         *            the number of characters to skip.
311         * @return the number of characters actually skipped.
312         * @throws IOException
313         *             if this reader is closed.
314         * @since Android 1.0
315         */
316        @Override
317        public long skip(long n) throws IOException {
318            synchronized (lock) {
319                if (isClosed()) {
320                    throw new IOException(Msg.getString("K0060")); //$NON-NLS-1$
321                }
322                if (n <= 0) {
323                    return 0;
324                }
325                long skipped = 0;
326                if (n < this.count - pos) {
327                    pos = pos + (int) n;
328                    skipped = n;
329                } else {
330                    skipped = this.count - pos;
331                    pos = this.count;
332                }
333                return skipped;
334            }
335        }
336    }
