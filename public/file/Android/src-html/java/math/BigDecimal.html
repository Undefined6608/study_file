300        /**
301         * Constructs a new {@code BigDecimal} instance from a string representation
302         * given as a character array.
303         * 
304         * @param in
305         *            array of characters containing the string representation of
306         *            this {@code BigDecimal}.
307         * @param offset
308         *            first index to be copied.
309         * @param len
310         *            number of characters to be used.
311         * @throws NullPointerException
312         *             if {@code in == null}.
313         * @throws NumberFormatException
314         *             if {@code offset < 0} or {@code len <= 0} or {@code
315         *             offset+len-1 < 0} or {@code offset+len-1 >= in.length}.
316         * @throws NumberFormatException
317         *             if in does not contain a valid string representation of a big
318         *             decimal.
319         * @since Android 1.0
320         */
321        public BigDecimal(char[] in, int offset, int len) {
322            int begin = offset; // first index to be copied
323            int last = offset + (len - 1); // last index to be copied
324            String scaleString = null; // buffer for scale
325            StringBuilder unscaledBuffer; // buffer for unscaled value
326            long newScale; // the new scale
327    
328            if (in == null) {
329                throw new NullPointerException();
330            }
331            if ((last >= in.length) || (offset < 0) || (len <= 0) || (last < 0)) {
332                throw new NumberFormatException();
333            }
334            unscaledBuffer = new StringBuilder(len);
335            int bufLength = 0;
336            // To skip a possible '+' symbol
337            if ((offset <= last) && (in[offset] == '+')) {
338                offset++;
339                begin++;
340            }
341            int counter = 0;
342            boolean wasNonZero = false;
343            // Accumulating all digits until a possible decimal point
344            for (; (offset <= last) && (in[offset] != '.')
345            && (in[offset] != 'e') && (in[offset] != 'E'); offset++) {
346                if (!wasNonZero) {
347                    if (in[offset] == '0') {
348                        counter++;
349                    } else {
350                        wasNonZero = true;    
351                    }                
352                };
353    
354            }
355            unscaledBuffer.append(in, begin, offset - begin);
356            bufLength += offset - begin;
357            // A decimal point was found
358            if ((offset <= last) && (in[offset] == '.')) {
359                offset++;
360                // Accumulating all digits until a possible exponent
361                begin = offset;
362                for (; (offset <= last) && (in[offset] != 'e')
363                && (in[offset] != 'E'); offset++) {
364                    if (!wasNonZero) {
365                        if (in[offset] == '0') {
366                            counter++;
367                        } else {
368                            wasNonZero = true;    
369                        }                
370                    };
371                }
372                scale = offset - begin;
373                bufLength +=scale;
374                unscaledBuffer.append(in, begin, scale);
375            } else {
376                scale = 0;
377            }
378            // An exponent was found
379            if ((offset <= last) && ((in[offset] == 'e') || (in[offset] == 'E'))) {
380                offset++;
381                // Checking for a possible sign of scale
382                begin = offset;
383                if ((offset <= last) && (in[offset] == '+')) {
384                    offset++;
385                    if ((offset <= last) && (in[offset] != '-')) {
386                        begin++;
387                    }
388                }
389                // Accumulating all remaining digits
390                scaleString = String.valueOf(in, begin, last + 1 - begin);
391                // Checking if the scale is defined            
392                newScale = (long)scale - Integer.parseInt(scaleString);
393                scale = (int)newScale;
394                if (newScale != scale) {
395                    // math.02=Scale out of range.
396                    throw new NumberFormatException(Messages.getString("math.02")); //$NON-NLS-1$
397                }
398            }
399            // Parsing the unscaled value
400            if (bufLength < 19) {
401                smallValue = Long.parseLong(unscaledBuffer.toString());
402                bitLength = bitLength(smallValue);
403            } else {
404                setUnscaledValue(new BigInteger(unscaledBuffer.toString()));
405            }        
406            precision = unscaledBuffer.length() - counter;
407            if (unscaledBuffer.charAt(0) == '-') {
408                precision --;
409            }    
410        }

528        /**
529         * Constructs a new {@code BigDecimal} instance from the 64bit double
530         * {@code val}. The constructed big decimal is equivalent to the given
531         * double. For example, {@code new BigDecimal(0.1)} is equal to {@code
532         * 0.1000000000000000055511151231257827021181583404541015625}. This happens
533         * as {@code 0.1} cannot be represented exactly in binary.
534         * <p>
535         * To generate a big decimal instance which is equivalent to {@code 0.1} use
536         * the {@code BigDecimal(String)} constructor.
537         * 
538         * @param val
539         *            double value to be converted to a {@code BigDecimal} instance.
540         * @throws NumberFormatException
541         *             if {@code val} is infinity or not a number.
542         * @since Android 1.0
543         */
544        public BigDecimal(double val) {
545            if (Double.isInfinite(val) || Double.isNaN(val)) {
546                // math.03=Infinity or NaN
547                throw new NumberFormatException(Messages.getString("math.03")); //$NON-NLS-1$
548            }
549            long bits = Double.doubleToLongBits(val); // IEEE-754
550            long mantisa;
551            int trailingZeros;
552            // Extracting the exponent, note that the bias is 1023
553            scale = 1075 - (int)((bits >> 52) & 0x7FFL);
554            // Extracting the 52 bits of the mantisa.
555            mantisa = (scale == 1075) ? (bits & 0xFFFFFFFFFFFFFL) << 1
556                    : (bits & 0xFFFFFFFFFFFFFL) | 0x10000000000000L;
557            if (mantisa == 0) {
558                scale = 0;
559                precision = 1;
560            }
561            // To simplify all factors '2' in the mantisa 
562            if (scale > 0) {
563                trailingZeros = Math.min(scale, Long.numberOfTrailingZeros(mantisa));
564                mantisa >>>= trailingZeros;
565                scale -= trailingZeros;
566            }
567            // Calculating the new unscaled value and the new scale
568            if((bits >> 63) != 0) {
569                mantisa = -mantisa;
570            }
571            int mantisaBits = bitLength(mantisa);
572            if (scale < 0) {
573                bitLength = mantisaBits == 0 ? 0 : mantisaBits - scale;
574                if(bitLength < 64) {
575                    smallValue = mantisa << (-scale);
576                } else {
577                    // BEGIN android-changed
578                    BigInt bi = new BigInt();
579                    bi.putLongInt(mantisa);
580                    bi.shift(-scale);
581                    intVal = new BigInteger(bi);
582                    // END android-changed
583                }
584                scale = 0;
585            } else if (scale > 0) {
586                // m * 2^e =  (m * 5^(-e)) * 10^e
587                if(scale < LONG_FIVE_POW.length
588                        && mantisaBits+LONG_FIVE_POW_BIT_LENGTH[scale] < 64) {
589                    smallValue = mantisa * LONG_FIVE_POW[scale];
590                    bitLength = bitLength(smallValue);
591                } else {
592                    setUnscaledValue(Multiplication.multiplyByFivePow(BigInteger.valueOf(mantisa), scale));
593                }
594            } else { // scale == 0
595                smallValue = mantisa;
596                bitLength = mantisaBits;
597            }
598        }

856        public BigDecimal add(BigDecimal augend) {

858            // Fast return when some operand is zero
859            if (this.isZero()) {
860                if (diffScale <= 0) {
861                    return augend;
862                }
863                if (augend.isZero()) {
864                    return this;
865                }
866            } else if (augend.isZero()) {
867                if (diffScale >= 0) {
868                    return this;
869                }
870            }
871            // Let be:  this = [u1,s1]  and  augend = [u2,s2]
872            if (diffScale == 0) {
873                // case s1 == s2: [u1 + u2 , s1]
874                if (Math.max(this.bitLength, augend.bitLength) + 1 < 64) {
875                    return valueOf(this.smallValue + augend.smallValue, this.scale);
876                }
877                return new BigDecimal(this.getUnscaledValue().add(augend.getUnscaledValue()), this.scale);
878            } else if (diffScale > 0) {
879                // case s1 > s2 : [(u1 + u2) * 10 ^ (s1 - s2) , s1]
880                return addAndMult10(this, augend, diffScale);
881            } else {// case s2 > s1 : [(u2 + u1) * 10 ^ (s2 - s1) , s2]
882                return addAndMult10(augend, this, -diffScale);
883            }

913            BigDecimal larger; // operand with the largest unscaled value
914            BigDecimal smaller; // operand with the smallest unscaled value
915            BigInteger tempBI;
916            long diffScale = (long)this.scale - augend.scale;
917            int largerSignum;
918            // Some operand is zero or the precision is infinity

923            // Cases where there is room for optimizations
924            if (this.aproxPrecision() < diffScale - 1) {

927            } else if (augend.aproxPrecision() < -diffScale - 1) {

929                smaller = augend;

931                return add(augend).round(mc);

967            // Fast return when some operand is zero
968            if (this.isZero()) {
969                if (diffScale <= 0) {
970                    return subtrahend.negate();
971                }
972                if (subtrahend.isZero()) {
973                    return this;
974                }
975            } else if (subtrahend.isZero()) {
976                if (diffScale >= 0) {
977                    return this;
978                }
979            }
980            // Let be: this = [u1,s1] and subtrahend = [u2,s2] so:
981            if (diffScale == 0) {
982                // case s1 = s2 : [u1 - u2 , s1]
983                if (Math.max(this.bitLength, subtrahend.bitLength) + 1 < 64) {
984                    return valueOf(this.smallValue - subtrahend.smallValue,this.scale);
985                }

990                        Math.max(this.bitLength,subtrahend.bitLength+LONG_TEN_POW_BIT_LENGTH[diffScale])+1<64) {

998                        Math.max(this.bitLength+LONG_TEN_POW_BIT_LENGTH[diffScale],subtrahend.bitLength)+1<64) {
999                    return valueOf(this.smallValue*LONG_TEN_POW[diffScale]-subtrahend.smallValue,subtrahend.scale);
1000                }
1001                return new BigDecimal(Multiplication.multiplyByTenPow(this.getUnscaledValue(),diffScale)

1201        private static BigDecimal divideBigIntegers(BigInteger scaledDividend, BigInteger scaledDivisor, int scale, RoundingMode roundingMode) {
1202            
1203            BigInteger[] quotAndRem = scaledDividend.divideAndRemainder(scaledDivisor);  // quotient and remainder
1204            // If after division there is a remainder...
1205            BigInteger quotient = quotAndRem[0];
1206            BigInteger remainder = quotAndRem[1];
1207            if (remainder.signum() == 0) {
1208                return new BigDecimal(quotient, scale);
1209            }
1210            int sign = scaledDividend.signum() * scaledDivisor.signum();
1211            int compRem;                                      // 'compare to remainder'
1212            if(scaledDivisor.bitLength() < 63) { // 63 in order to avoid out of long after <<1
1213                long rem = remainder.longValue();
1214                long divisor = scaledDivisor.longValue();
1215                compRem = longCompareTo(Math.abs(rem) << 1,Math.abs(divisor));
1216                // To look if there is a carry
1217                compRem = roundingBehavior(quotient.testBit(0) ? 1 : 0,
1218                        sign * (5 + compRem), roundingMode);
1219                
1220            } else {
1221                // Checking if:  remainder * 2 >= scaledDivisor 
1222                compRem = remainder.abs().shiftLeft(1).compareTo(scaledDivisor.abs());
1223                compRem = roundingBehavior(quotient.testBit(0) ? 1 : 0,
1224                        sign * (5 + compRem), roundingMode);
1225            }
1226                if (compRem != 0) {
1227                if(quotient.bitLength() < 63) {
1228                    return valueOf(quotient.longValue() + compRem,scale);
1229                }
1230                quotient = quotient.add(BigInteger.valueOf(compRem));
1231                return new BigDecimal(quotient, scale);
1232            }
1233            // Constructing the result with the appropriate unscaled value
1234            return new BigDecimal(quotient, scale);
1235        }

1237        private static BigDecimal dividePrimitiveLongs(long scaledDividend, long scaledDivisor, int scale, RoundingMode roundingMode) {
1238            long quotient = scaledDividend / scaledDivisor;
1239            long remainder = scaledDividend % scaledDivisor;
1240            int sign = Long.signum( scaledDividend ) * Long.signum( scaledDivisor );
1241            if (remainder != 0) {
1242                // Checking if:  remainder * 2 >= scaledDivisor

1247                        sign * (5 + compRem),
1248                        roundingMode);
1249            }
1250            // Constructing the result with the appropriate unscaled value
1251            return valueOf(quotient, scale);

1306        /**
1307         * Returns a new {@code BigDecimal} whose value is {@code this / divisor}.
1308         * The scale of the result is the difference of the scales of {@code this}
1309         * and {@code divisor}. If the exact result requires more digits, then the
1310         * scale is adjusted accordingly. For example, {@code 1/128 = 0.0078125}
1311         * which has a scale of {@code 7} and precision {@code 5}.
1312         * 
1313         * @param divisor
1314         *            value by which {@code this} is divided.
1315         * @return {@code this / divisor}.
1316         * @throws NullPointerException
1317         *             if {@code divisor == null}.
1318         * @throws ArithmeticException
1319         *             if {@code divisor == 0}.
1320         * @throws ArithmeticException
1321         *             if the result cannot be represented exactly.
1322         * @since Android 1.0
1323         */
1324        public BigDecimal divide(BigDecimal divisor) {
1325            BigInteger p = this.getUnscaledValue();
1326            BigInteger q = divisor.getUnscaledValue();
1327            BigInteger gcd; // greatest common divisor between 'p' and 'q'
1328            BigInteger quotAndRem[];
1329            long diffScale = (long)scale - divisor.scale;
1330            int newScale; // the new scale for final quotient
1331            int k; // number of factors "2" in 'q'
1332            int l = 0; // number of factors "5" in 'q'
1333            int i = 1;
1334            int lastPow = FIVE_POW.length - 1;
1335    
1336            if (divisor.isZero()) {
1337                // math.04=Division by zero
1338                throw new ArithmeticException(Messages.getString("math.04")); //$NON-NLS-1$
1339            }
1340            if (p.signum() == 0) {
1341                return zeroScaledBy(diffScale);
1342            }
1343            // To divide both by the GCD
1344            gcd = p.gcd(q);
1345            p = p.divide(gcd);
1346            q = q.divide(gcd);
1347            // To simplify all "2" factors of q, dividing by 2^k
1348            k = q.getLowestSetBit();
1349            q = q.shiftRight(k);
1350            // To simplify all "5" factors of q, dividing by 5^l
1351            do {
1352                quotAndRem = q.divideAndRemainder(FIVE_POW[i]);
1353                if (quotAndRem[1].signum() == 0) {
1354                    l += i;
1355                    if (i < lastPow) {
1356                        i++;
1357                    }
1358                    q = quotAndRem[0];
1359                } else {
1360                    if (i == 1) {
1361                        break;
1362                    }
1363                    i = 1;
1364                }
1365            } while (true);
1366            // If  abs(q) != 1  then the quotient is periodic
1367            if (!q.abs().equals(BigInteger.ONE)) {
1368                // math.05=Non-terminating decimal expansion; no exact representable decimal result.
1369                throw new ArithmeticException(Messages.getString("math.05")); //$NON-NLS-1$
1370            }
1371            // The sign of the is fixed and the quotient will be saved in 'p'
1372            if (q.signum() < 0) {
1373                p = p.negate();
1374            }
1375            // Checking if the new scale is out of range
1376            newScale = toIntScale(diffScale + Math.max(k, l));
1377            // k >= 0  and  l >= 0  implies that  k - l  is in the 32-bit range
1378            i = k - l;
1379            
1380            p = (i > 0) ? Multiplication.multiplyByFivePow(p, i)
1381            : p.shiftLeft(-i);
1382            return new BigDecimal(p, newScale);
1383        }

1385        /**
1386         * Returns a new {@code BigDecimal} whose value is {@code this / divisor}.
1387         * The result is rounded according to the passed context {@code mc}. If the
1388         * passed math context specifies precision {@code 0}, then this call is
1389         * equivalent to {@code this.divide(divisor)}.
1390         * 
1391         * @param divisor
1392         *            value by which {@code this} is divided.
1393         * @param mc
1394         *            rounding mode and precision for the result of this operation.
1395         * @return {@code this / divisor}.
1396         * @throws NullPointerException
1397         *             if {@code divisor == null} or {@code mc == null}.
1398         * @throws ArithmeticException
1399         *             if {@code divisor == 0}.
1400         * @throws ArithmeticException
1401         *             if {@code mc.getRoundingMode() == UNNECESSARY} and rounding
1402         *             is necessary according {@code mc.getPrecision()}.
1403         * @since Android 1.0
1404         */
1405        public BigDecimal divide(BigDecimal divisor, MathContext mc) {
1406            /* Calculating how many zeros must be append to 'dividend'
1407             * to obtain a  quotient with at least 'mc.precision()' digits */
1408            long traillingZeros = mc.getPrecision() + 2L
1409                    + divisor.aproxPrecision() - aproxPrecision();
1410            long diffScale = (long)scale - divisor.scale;
1411            long newScale = diffScale; // scale of the final quotient
1412            int compRem; // to compare the remainder
1413            int i = 1; // index   
1414            int lastPow = TEN_POW.length - 1; // last power of ten
1415            BigInteger integerQuot; // for temporal results
1416            BigInteger quotAndRem[] = {getUnscaledValue()};
1417            // In special cases it reduces the problem to call the dual method
1418            if ((mc.getPrecision() == 0) || (this.isZero())
1419            || (divisor.isZero())) {
1420                return this.divide(divisor);
1421            }
1422            if (traillingZeros > 0) {
1423                // To append trailing zeros at end of dividend
1424                quotAndRem[0] = getUnscaledValue().multiply( Multiplication.powerOf10(traillingZeros) );
1425                newScale += traillingZeros;
1426            }
1427            quotAndRem = quotAndRem[0].divideAndRemainder( divisor.getUnscaledValue() );
1428            integerQuot = quotAndRem[0];
1429            // Calculating the exact quotient with at least 'mc.precision()' digits
1430            if (quotAndRem[1].signum() != 0) {
1431                // Checking if:   2 * remainder >= divisor ?
1432                compRem = quotAndRem[1].shiftLeft(1).compareTo( divisor.getUnscaledValue() );
1433                // quot := quot * 10 + r;     with 'r' in {-6,-5,-4, 0,+4,+5,+6}
1434                integerQuot = integerQuot.multiply(BigInteger.TEN)
1435                .add(BigInteger.valueOf(quotAndRem[0].signum() * (5 + compRem)));
1436                newScale++;
1437            } else {
1438                // To strip trailing zeros until the preferred scale is reached
1439                while (!integerQuot.testBit(0)) {
1440                    quotAndRem = integerQuot.divideAndRemainder(TEN_POW[i]);
1441                    if ((quotAndRem[1].signum() == 0)
1442                            && (newScale - i >= diffScale)) {
1443                        newScale -= i;
1444                        if (i < lastPow) {
1445                            i++;
1446                        }
1447                        integerQuot = quotAndRem[0];
1448                    } else {
1449                        if (i == 1) {
1450                            break;
1451                        }
1452                        i = 1;
1453                    }
1454                }
1455            }
1456            // To perform rounding
1457            return new BigDecimal(integerQuot, toIntScale(newScale), mc);
1458        }

1461         * Returns a new {@code BigDecimal} whose value is the integral part of
1462         * {@code this / divisor}. The quotient is rounded down towards zero to the
1463         * next integer. For example, {@code 0.5/0.2 = 2}.
1464         * 
1465         * @param divisor
1466         *            value by which {@code this} is divided.
1467         * @return integral part of {@code this / divisor}.
1468         * @throws NullPointerException
1469         *             if {@code divisor == null}.
1470         * @throws ArithmeticException
1471         *             if {@code divisor == 0}.
1472         * @since Android 1.0
1473         */
1474        public BigDecimal divideToIntegralValue(BigDecimal divisor) {
1475            BigInteger integralValue; // the integer of result
1476            BigInteger powerOfTen; // some power of ten
1477            BigInteger quotAndRem[] = {getUnscaledValue()};
1478            long newScale = (long)this.scale - divisor.scale;
1479            long tempScale = 0;
1480            int i = 1;
1481            int lastPow = TEN_POW.length - 1;
1482    
1483            if (divisor.isZero()) {
1484                // math.04=Division by zero
1485                throw new ArithmeticException(Messages.getString("math.04")); //$NON-NLS-1$
1486            }
1487            if ((divisor.aproxPrecision() + newScale > this.aproxPrecision() + 1L)
1488            || (this.isZero())) {
1489                /* If the divisor's integer part is greater than this's integer part,
1490                 * the result must be zero with the appropriate scale */
1491                integralValue = BigInteger.ZERO;
1492            } else if (newScale == 0) {
1493                integralValue = getUnscaledValue().divide( divisor.getUnscaledValue() );
1494            } else if (newScale > 0) {
1495                powerOfTen = Multiplication.powerOf10(newScale);
1496                integralValue = getUnscaledValue().divide( divisor.getUnscaledValue().multiply(powerOfTen) );
1497                integralValue = integralValue.multiply(powerOfTen);
1498            } else {// (newScale < 0)
1499                powerOfTen = Multiplication.powerOf10(-newScale);
1500                integralValue = getUnscaledValue().multiply(powerOfTen).divide( divisor.getUnscaledValue() );
1501                // To strip trailing zeros approximating to the preferred scale
1502                while (!integralValue.testBit(0)) {
1503                    quotAndRem = integralValue.divideAndRemainder(TEN_POW[i]);
1504                    if ((quotAndRem[1].signum() == 0)
1505                            && (tempScale - i >= newScale)) {
1506                        tempScale -= i;
1507                        if (i < lastPow) {
1508                            i++;
1509                        }
1510                        integralValue = quotAndRem[0];
1511                    } else {
1512                        if (i == 1) {
1513                            break;
1514                        }
1515                        i = 1;
1516                    }
1517                }
1518                newScale = tempScale;
1519            }
1520            return ((integralValue.signum() == 0)
1521            ? zeroScaledBy(newScale)
1522                    : new BigDecimal(integralValue, toIntScale(newScale)));
1523        }

1525        /**
1526         * Returns a new {@code BigDecimal} whose value is the integral part of
1527         * {@code this / divisor}. The quotient is rounded down towards zero to the
1528         * next integer. The rounding mode passed with the parameter {@code mc} is
1529         * not considered. But if the precision of {@code mc > 0} and the integral
1530         * part requires more digits, then an {@code ArithmeticException} is thrown.
1531         * 
1532         * @param divisor
1533         *            value by which {@code this} is divided.
1534         * @param mc
1535         *            math context which determines the maximal precision of the
1536         *            result.
1537         * @return integral part of {@code this / divisor}.
1538         * @throws NullPointerException
1539         *             if {@code divisor == null} or {@code mc == null}.
1540         * @throws ArithmeticException
1541         *             if {@code divisor == 0}.
1542         * @throws ArithmeticException
1543         *             if {@code mc.getPrecision() > 0} and the result requires more
1544         *             digits to be represented.
1545         * @since Android 1.0
1546         */
1547        public BigDecimal divideToIntegralValue(BigDecimal divisor, MathContext mc) {
1548            int mcPrecision = mc.getPrecision();
1549            int diffPrecision = this.precision() - divisor.precision();
1550            int lastPow = TEN_POW.length - 1;
1551            long diffScale = (long)this.scale - divisor.scale;
1552            long newScale = diffScale;
1553            long quotPrecision = diffPrecision - diffScale + 1;
1554            BigInteger quotAndRem[] = new BigInteger[2];
1555            // In special cases it call the dual method
1556            if ((mcPrecision == 0) || (this.isZero()) || (divisor.isZero())) {
1557                return this.divideToIntegralValue(divisor);
1558            }
1559            // Let be:   this = [u1,s1]   and   divisor = [u2,s2]
1560            if (quotPrecision <= 0) {
1561                quotAndRem[0] = BigInteger.ZERO;
1562            } else if (diffScale == 0) {
1563                // CASE s1 == s2:  to calculate   u1 / u2 
1564                quotAndRem[0] = this.getUnscaledValue().divide( divisor.getUnscaledValue() );
1565            } else if (diffScale > 0) {
1566                // CASE s1 >= s2:  to calculate   u1 / (u2 * 10^(s1-s2)  
1567                quotAndRem[0] = this.getUnscaledValue().divide(
1568                        divisor.getUnscaledValue().multiply(Multiplication.powerOf10(diffScale)) );
1569                // To chose  10^newScale  to get a quotient with at least 'mc.precision()' digits
1570                newScale = Math.min(diffScale, Math.max(mcPrecision - quotPrecision + 1, 0));
1571                // To calculate: (u1 / (u2 * 10^(s1-s2)) * 10^newScale
1572                quotAndRem[0] = quotAndRem[0].multiply(Multiplication.powerOf10(newScale));
1573            } else {// CASE s2 > s1:   
1574                /* To calculate the minimum power of ten, such that the quotient 
1575                 *   (u1 * 10^exp) / u2   has at least 'mc.precision()' digits. */
1576                long exp = Math.min(-diffScale, Math.max((long)mcPrecision - diffPrecision, 0));
1577                long compRemDiv;
1578                // Let be:   (u1 * 10^exp) / u2 = [q,r]  
1579                quotAndRem = this.getUnscaledValue().multiply(Multiplication.powerOf10(exp)).
1580                        divideAndRemainder(divisor.getUnscaledValue());
1581                newScale += exp; // To fix the scale
1582                exp = -newScale; // The remaining power of ten
1583                // If after division there is a remainder...
1584                if ((quotAndRem[1].signum() != 0) && (exp > 0)) {
1585                    // Log10(r) + ((s2 - s1) - exp) > mc.precision ?
1586                    compRemDiv = (new BigDecimal(quotAndRem[1])).precision()
1587                    + exp - divisor.precision();
1588                    if (compRemDiv == 0) {
1589                        // To calculate:  (r * 10^exp2) / u2
1590                        quotAndRem[1] = quotAndRem[1].multiply(Multiplication.powerOf10(exp)).
1591                                divide(divisor.getUnscaledValue());
1592                        compRemDiv = Math.abs(quotAndRem[1].signum());
1593                    }
1594                    if (compRemDiv > 0) {
1595                        // The quotient won't fit in 'mc.precision()' digits
1596                        // math.06=Division impossible
1597                        throw new ArithmeticException(Messages.getString("math.06")); //$NON-NLS-1$
1598                    }
1599                }
1600            }
1601            // Fast return if the quotient is zero
1602            if (quotAndRem[0].signum() == 0) {
1603                return zeroScaledBy(diffScale);
1604            }
1605            BigInteger strippedBI = quotAndRem[0];
1606            BigDecimal integralValue = new BigDecimal(quotAndRem[0]);
1607            long resultPrecision = integralValue.precision();
1608            int i = 1;
1609            // To strip trailing zeros until the specified precision is reached
1610            while (!strippedBI.testBit(0)) {
1611                quotAndRem = strippedBI.divideAndRemainder(TEN_POW[i]);
1612                if ((quotAndRem[1].signum() == 0) &&
1613                        ((resultPrecision - i >= mcPrecision)
1614                        || (newScale - i >= diffScale)) ) {
1615                    resultPrecision -= i;
1616                    newScale -= i;
1617                    if (i < lastPow) {
1618                        i++;
1619                    }
1620                    strippedBI = quotAndRem[0];
1621                } else {
1622                    if (i == 1) {
1623                        break;
1624                    }
1625                    i = 1;
1626                }
1627            }
1628            // To check if the result fit in 'mc.precision()' digits
1629            if (resultPrecision > mcPrecision) {
1630                // math.06=Division impossible
1631                throw new ArithmeticException(Messages.getString("math.06")); //$NON-NLS-1$
1632            }
1633            integralValue.scale = toIntScale(newScale);
1634            integralValue.setUnscaledValue(strippedBI);
1635            return integralValue;
1636        }

1790            int m = Math.abs(n);
1791            int mcPrecision = mc.getPrecision();
1792            int elength = (int)Math.log10(m) + 1;   // decimal digits in 'n'
1793            int oneBitMask; // mask of bits
1794            BigDecimal accum; // the single accumulator
1795            MathContext newPrecision = mc; // MathContext by default
1796    
1797            // In particular cases, it reduces the problem to call the other 'pow()'
1798            if ((n == 0) || ((isZero()) && (n > 0))) {
1799                return pow(n);
1800            }
1801            if ((m > 999999999) || ((mcPrecision == 0) && (n < 0))
1802                    || ((mcPrecision > 0) && (elength > mcPrecision))) {
1803                // math.07=Invalid Operation
1804                throw new ArithmeticException(Messages.getString("math.07")); //$NON-NLS-1$
1805            }
1806            if (mcPrecision > 0) {
1807                newPrecision = new MathContext( mcPrecision + elength + 1,
1808                        mc.getRoundingMode());
1809            }
1810            // The result is calculated as if 'n' were positive        
1811            accum = round(newPrecision);
1812            oneBitMask = Integer.highestOneBit(m) >> 1;
1813    
1814            while (oneBitMask > 0) {
1815                accum = accum.multiply(accum, newPrecision);
1816                if ((m & oneBitMask) == oneBitMask) {
1817                    accum = accum.multiply(this, newPrecision);
1818                }
1819                oneBitMask >>= 1;
1820            }
1821            // If 'n' is negative, the value is divided into 'ONE'
1822            if (n < 0) {
1823                accum = ONE.divide(accum, newPrecision);
1824            }
1825            // The final value is rounded to the destination precision
1826            accum.inplaceRound(mc);
1827            return accum;
1828        }

2741        /**
2742         * Returns this {@code BigDecimal} as a double value. If {@code this} is too
2743         * big to be represented as an float, then {@code Double.POSITIVE_INFINITY}
2744         * or {@code Double.NEGATIVE_INFINITY} is returned.
2745         * <p>
2746         * Note, that if the unscaled value has more than 53 significant digits,
2747         * then this decimal cannot be represented exactly in a double variable. In
2748         * this case the result is rounded.
2749         * <p>
2750         * For example, if the instance {@code x1 = new BigDecimal("0.1")} cannot be
2751         * represented exactly as a double, and thus {@code x1.equals(new
2752         * BigDecimal(x1.doubleValue())} returns {@code false} for this case.
2753         * <p>
2754         * Similarly, if the instance {@code new BigDecimal(9007199254740993L)} is
2755         * converted to a double, the result is {@code 9.007199254740992E15}.
2756         * <p>
2757         * 
2758         * @return this {@code BigDecimal} as a double value.
2759         * @since Android 1.0
2760         */
2761        @Override
2762        public double doubleValue() {
2763            int sign = signum();
2764            int exponent = 1076; // bias + 53
2765            int lowestSetBit;
2766            int discardedSize;
2767            long powerOfTwo = this.bitLength - (long)(scale / LOG10_2);
2768            long bits; // IEEE-754 Standard
2769            long tempBits; // for temporal calculations     
2770            BigInteger mantisa;
2771    
2772            if ((powerOfTwo < -1074) || (sign == 0)) {
2773                // Cases which 'this' is very small            
2774                return (sign * 0.0d);
2775            } else if (powerOfTwo > 1025) {
2776                // Cases which 'this' is very large            
2777                return (sign * Double.POSITIVE_INFINITY);
2778            }
2779            mantisa = getUnscaledValue().abs();
2780            // Let be:  this = [u,s], with s > 0
2781            if (scale <= 0) {
2782                // mantisa = abs(u) * 10^s
2783                mantisa = mantisa.multiply(Multiplication.powerOf10(-scale));
2784            } else {// (scale > 0)
2785                BigInteger quotAndRem[];
2786                BigInteger powerOfTen = Multiplication.powerOf10(scale);
2787                int k = 100 - (int)powerOfTwo;
2788                int compRem;
2789    
2790                if (k > 0) {
2791                    /* Computing (mantisa * 2^k) , where 'k' is a enough big
2792                     * power of '2' to can divide by 10^s */
2793                    mantisa = mantisa.shiftLeft(k);
2794                    exponent -= k;
2795                }
2796                // Computing (mantisa * 2^k) / 10^s
2797                quotAndRem = mantisa.divideAndRemainder(powerOfTen);
2798                // To check if the fractional part >= 0.5
2799                compRem = quotAndRem[1].shiftLeft(1).compareTo(powerOfTen);
2800                // To add two rounded bits at end of mantisa
2801                mantisa = quotAndRem[0].shiftLeft(2).add(
2802                        BigInteger.valueOf((compRem * (compRem + 3)) / 2 + 1));
2803                exponent -= 2;
2804            }
2805            lowestSetBit = mantisa.getLowestSetBit();
2806            discardedSize = mantisa.bitLength() - 54;
2807            if (discardedSize > 0) {// (n > 54)
2808                // mantisa = (abs(u) * 10^s) >> (n - 54)
2809                bits = mantisa.shiftRight(discardedSize).longValue();
2810                tempBits = bits;
2811                // #bits = 54, to check if the discarded fraction produces a carry             
2812                if ((((bits & 1) == 1) && (lowestSetBit < discardedSize))
2813                        || ((bits & 3) == 3)) {
2814                    bits += 2;
2815                }
2816            } else {// (n <= 54)
2817                // mantisa = (abs(u) * 10^s) << (54 - n)                
2818                bits = mantisa.longValue() << -discardedSize;
2819                tempBits = bits;
2820                // #bits = 54, to check if the discarded fraction produces a carry:
2821                if ((bits & 3) == 3) {
2822                    bits += 2;
2823                }
2824            }
2825            // Testing bit 54 to check if the carry creates a new binary digit
2826            if ((bits & 0x40000000000000L) == 0) {
2827                // To drop the last bit of mantisa (first discarded)
2828                bits >>= 1;
2829                // exponent = 2^(s-n+53+bias)
2830                exponent += discardedSize;
2831            } else {// #bits = 54
2832                bits >>= 2;
2833                exponent += discardedSize + 1;
2834            }
2835            // To test if the 53-bits number fits in 'double'            
2836            if (exponent > 2046) {// (exponent - bias > 1023)
2837                return (sign * Double.POSITIVE_INFINITY);
2838            } else if (exponent <= 0) {// (exponent - bias <= -1023)
2839                // Denormalized numbers (having exponent == 0)
2840                if (exponent < -53) {// exponent - bias < -1076
2841                    return (sign * 0.0d);
2842                }
2843                // -1076 <= exponent - bias <= -1023 
2844                // To discard '- exponent + 1' bits
2845                bits = tempBits >> 1;
2846                tempBits = bits & (-1L >>> (63 + exponent));
2847                bits >>= (-exponent );
2848                // To test if after discard bits, a new carry is generated
2849                if (((bits & 3) == 3) || (((bits & 1) == 1) && (tempBits != 0)
2850                && (lowestSetBit < discardedSize))) {
2851                    bits += 1;
2852                }
2853                exponent = 0;
2854                bits >>= 1;
2855            }
2856            // Construct the 64 double bits: [sign(1), exponent(11), mantisa(52)]
2857            bits = (sign & 0x8000000000000000L) | ((long)exponent << 52)
2858                    | (bits & 0xFFFFFFFFFFFFFL);
2859            return Double.longBitsToDouble(bits);
2860        }

2885         * It does all rounding work of the public method
2886         * {@code round(MathContext)}, performing an inplace rounding
2887         * without creating a new object.
2888         * 
2889         * @param mc
2890         *            the {@code MathContext} for perform the rounding.
2891         * @see #round(MathContext)
2892         */
2893        private void inplaceRound(MathContext mc) {
2894            int mcPrecision = mc.getPrecision();
2895            // BEGIN android-changed
2896            if (aproxPrecision() < mcPrecision || mcPrecision == 0) {
2897                return;
2898            }
2899            // END android-changed
2900            int discardedPrecision = precision() - mcPrecision;
2901            // If no rounding is necessary it returns immediately
2902            if ((discardedPrecision <= 0)) {
2903                return;
2904            }
2905            // When the number is small perform an efficient rounding
2906            if (this.bitLength < 64) {
2907                smallRound(mc, discardedPrecision);
2908                return;
2909            }
2910            // Getting the integer part and the discarded fraction
2911            BigInteger sizeOfFraction = Multiplication.powerOf10(discardedPrecision);
2912            BigInteger[] integerAndFraction = getUnscaledValue().divideAndRemainder(sizeOfFraction);
2913            long newScale = (long)scale - discardedPrecision;
2914            int compRem;
2915            BigDecimal tempBD;
2916            // If the discarded fraction is non-zero, perform rounding
2917            if (integerAndFraction[1].signum() != 0) {
2918                // To check if the discarded fraction >= 0.5
2919                compRem = (integerAndFraction[1].abs().shiftLeft(1).compareTo(sizeOfFraction));
2920                // To look if there is a carry
2921                compRem =  roundingBehavior( integerAndFraction[0].testBit(0) ? 1 : 0,
2922                        integerAndFraction[1].signum() * (5 + compRem),
2923                        mc.getRoundingMode());
2924                if (compRem != 0) {
2925                    integerAndFraction[0] = integerAndFraction[0].add(BigInteger.valueOf(compRem));
2926                }
2927                tempBD = new BigDecimal(integerAndFraction[0]);
2928                // If after to add the increment the precision changed, we normalize the size
2929                if (tempBD.precision() > mcPrecision) {
2930                    integerAndFraction[0] = integerAndFraction[0].divide(BigInteger.TEN);
2931                    newScale--;
2932                }
2933            }
2934            // To update all internal fields
2935            scale = toIntScale(newScale);
2936            precision = mcPrecision;
2937            setUnscaledValue(integerAndFraction[0]);
2938        }
