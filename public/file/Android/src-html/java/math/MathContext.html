027    /**
028     * Immutable objects describing settings such as rounding mode and digit
029     * precision for the numerical operations provided by class {@link BigDecimal}.
030     * 
031     * @since Android 1.0
032     */
033    public final class MathContext implements Serializable {
034        // BEGIN android-note
035        // copied from newer version of harmony
036        // added final modifier
037        // END android-note

041        /**

044         * {@link RoundingMode#HALF_EVEN} rounding.

049                RoundingMode.HALF_EVEN);

058        public static final MathContext DECIMAL32 = new MathContext(7,

083        /**
084         * The number of digits to be used for an operation; results are rounded to
085         * this precision.
086         */
087        private int precision;
088    
089        /**
090         * A {@code RoundingMode} object which specifies the algorithm to be used
091         * for rounding.
092         */
093        private RoundingMode roundingMode;
094    
095        /**
096         * An array of {@code char} containing: {@code
097         * 'p','r','e','c','i','s','i','o','n','='}. It's used to improve the
098         * methods related to {@code String} conversion.
099         * 
100         * @see #MathContext(String)
101         * @see #toString()
102         */
103        private final static char[] chPrecision = { 'p', 'r', 'e', 'c', 'i', 's',
104                'i', 'o', 'n', '=' };
105    
106        /**
107         * An array of {@code char} containing: {@code
108         * 'r','o','u','n','d','i','n','g','M','o','d','e','='}. It's used to
109         * improve the methods related to {@code String} conversion.
110         * 
111         * @see #MathContext(String)
112         * @see #toString()
113         */
114        private final static char[] chRoundingMode = { 'r', 'o', 'u', 'n', 'd',
115                'i', 'n', 'g', 'M', 'o', 'd', 'e', '=' };

119        /**
120         * Constructs a new {@code MathContext} with the specified precision and
121         * with the rounding mode {@link RoundingMode#HALF_UP HALF_UP}. If the
122         * precision passed is zero, then this implies that the computations have to
123         * be performed exact, the rounding mode in this case is irrelevant.
124         * 
125         * @param precision
126         *            the precision for the new {@code MathContext}.
127         * @throws IllegalArgumentException
128         *             if {@code precision < 0}.
129         * 
130         * @since Android 1.0
131         */
132        public MathContext(int precision) {
133            // BEGIN android-note
134            // parameter names changed.
135            // END android-note
136            this(precision, RoundingMode.HALF_UP);
137        }
138    
139        /**
140         * Constructs a new {@code MathContext} with the specified precision and
141         * with the specified rounding mode. If the precision passed is zero, then
142         * this implies that the computations have to be performed exact, the
143         * rounding mode in this case is irrelevant.
144         * 
145         * @param precision
146         *            the precision for the new {@code MathContext}.
147         * @param roundingMode
148         *            the rounding mode for the new {@code MathContext}.
149         * @throws IllegalArgumentException
150         *             if {@code precision < 0}.
151         * @throws NullPointerException
152         *             if {@code roundingMode} is {@code null}.
153         * 
154         * @since Android 1.0
155         */
156        public MathContext(int precision, RoundingMode roundingMode) {
157            // BEGIN android-note
158            // parameter names changed.
159            // END android-note
160            if (precision < 0) {
161                // math.0C=Digits < 0
162                throw new IllegalArgumentException(Messages.getString("math.0C")); //$NON-NLS-1$
163            }
164            if (roundingMode == null) {
165                // math.0D=null RoundingMode
166                throw new NullPointerException(Messages.getString("math.0D")); //$NON-NLS-1$
167            }
168            this.precision = precision;
169            this.roundingMode = roundingMode;
170        }
171    
172        /**
173         * Constructs a new {@code MathContext} from a string. The string has to
174         * specify the precision and the rounding mode to be used and has to follow
175         * the following syntax: "precision=&lt;precision&gt; roundingMode=&lt;roundingMode&gt;"
176         * This is the same form as the one returned by the {@link #toString}
177         * method.
178         * 
179         * @param val
180         *            a string describing the precision and rounding mode for the
181         *            new {@code MathContext}.
182         * @throws IllegalArgumentException
183         *             if the string is not in the correct format or if the
184         *             precision specified is < 0.
185         * 
186         * @since Android 1.0
187         */
188        public MathContext(String val) {
189            char[] charVal = val.toCharArray();
190            int i; // Index of charVal
191            int j; // Index of chRoundingMode
192            int digit; // It will contain the digit parsed
193    
194            if ((charVal.length < 27) || (charVal.length > 45)) {
195                // math.0E=bad string format
196                throw new IllegalArgumentException(Messages.getString("math.0E")); //$NON-NLS-1$
197            }
198            // Parsing "precision=" String
199            for (i = 0; (i < chPrecision.length) && (charVal[i] == chPrecision[i]); i++) {
200                ;
201            }
202    
203            if (i < chPrecision.length) {
204                // math.0E=bad string format
205                throw new IllegalArgumentException(Messages.getString("math.0E")); //$NON-NLS-1$
206            }
207            // Parsing the value for "precision="...
208            digit = Character.digit(charVal[i], 10);
209            if (digit == -1) {
210                // math.0E=bad string format
211                throw new IllegalArgumentException(Messages.getString("math.0E")); //$NON-NLS-1$
212            }
213            // BEGIN android-changed
214            this.precision = digit;
215            // END android-changed
216            i++;
217    
218            do {
219                digit = Character.digit(charVal[i], 10);
220                if (digit == -1) {
221                    if (charVal[i] == ' ') {
222                        // It parsed all the digits
223                        i++;
224                        break;
225                    }
226                    // It isn't  a valid digit, and isn't a white space
227                    // math.0E=bad string format
228                    throw new IllegalArgumentException(Messages.getString("math.0E")); //$NON-NLS-1$
229                }
230                // Accumulating the value parsed
231                this.precision = this.precision * 10 + digit;
232                if (this.precision < 0) {
233                    // math.0E=bad string format
234                    throw new IllegalArgumentException(Messages.getString("math.0E")); //$NON-NLS-1$
235                }
236                i++;
237            } while (true);
238            // Parsing "roundingMode="
239            for (j = 0; (j < chRoundingMode.length)
240                    && (charVal[i] == chRoundingMode[j]); i++, j++) {
241                ;
242            }
243    
244            if (j < chRoundingMode.length) {
245                // math.0E=bad string format
246                throw new IllegalArgumentException(Messages.getString("math.0E")); //$NON-NLS-1$
247            }
248            // Parsing the value for "roundingMode"...
249            this.roundingMode = RoundingMode.valueOf(String.valueOf(charVal, i,
250                    charVal.length - i));
251        }
252    
253        /* Public Methods */
254    
255        /**
256         * Returns the precision. The precision is the number of digits used for an
257         * operation. Results are rounded to this precision. The precision is
258         * guaranteed to be non negative. If the precision is zero, then the
259         * computations have to be performed exact, results are not rounded in this
260         * case.
261         * 
262         * @return the precision.
263         * 
264         * @since Android 1.0
265         */
266        public int getPrecision() {
267            return precision;
268        }
269    
270        /**
271         * Returns the rounding mode. The rounding mode is the strategy to be used
272         * to round results.
273         * <p> 
274         * The rounding mode is one of 
275         * {@link RoundingMode#UP},
276         * {@link RoundingMode#DOWN}, 
277         * {@link RoundingMode#CEILING}, 
278         * {@link RoundingMode#FLOOR},
279         * {@link RoundingMode#HALF_UP}, 
280         * {@link RoundingMode#HALF_DOWN}, 
281         * {@link RoundingMode#HALF_EVEN}, or
282         * {@link RoundingMode#UNNECESSARY}.
283         * 
284         * @return the rounding mode.
285         * 
286         * @since Android 1.0
287         */
288        public RoundingMode getRoundingMode() {
289            return roundingMode;
290        }
291    
292        /**
293         * Returns true if x is a {@code MathContext} with the same precision
294         * setting and the same rounding mode as this {@code MathContext} instance.
295         * 
296         * @param x
297         *            object to be compared.
298         * @return {@code true} if this {@code MathContext} instance is equal to the
299         *         {@code x} argument; {@code false} otherwise.
300         * 
301         * @since Android 1.0
302         */
303        @Override
304        public boolean equals(Object x) {
305            return ((x instanceof MathContext)
306                    && (((MathContext) x).getPrecision() == precision) && (((MathContext) x)
307                    .getRoundingMode() == roundingMode));
308        }
309    
310        /**
311         * Returns the hash code for this {@code MathContext} instance.
312         * 
313         * @return the hash code for this {@code MathContext}.
314         * 
315         * @since Android 1.0
316         */
317        @Override
318        public int hashCode() {
319            // Make place for the necessary bits to represent 8 rounding modes
320            return ((precision << 3) | roundingMode.ordinal());
321        }
322    
323        /**
324         * Returns the string representation for this {@code MathContext} instance.
325         * The string has the form 
326         * {@code
327         * "precision=&lt;precision&gt; roundingMode=&lt;roundingMode&gt;"
328         * } where {@code &lt;precision&gt;} is an integer describing the number 
329         * of digits used for operations and {@code &lt;roundingMode&gt;} is the
330         * string representation of the rounding mode.
331         * 
332         * @return a string representation for this {@code MathContext} instance
333         * @since Android 1.0
334         */
335        @Override
336        public String toString() {
337            StringBuffer sb = new StringBuffer(45);
338    
339            sb.append(chPrecision);
340            sb.append(precision);
341            sb.append(' ');
342            sb.append(chRoundingMode);
343            sb.append(roundingMode);
344            return sb.toString();
345        }
346    
347        /**
348         * Makes checks upon deserialization of a {@code MathContext} instance.
349         * Checks whether {@code precision >= 0} and {@code roundingMode != null}
350         * 
351         * @throws StreamCorruptedException
352         *             if {@code precision < 0}
353         * @throws StreamCorruptedException
354         *             if {@code roundingMode == null}
355         */
356        private void readObject(ObjectInputStream s) throws IOException,
357                ClassNotFoundException {
358            s.defaultReadObject();
359            if (precision < 0) {
360                // math.0F=bad precision value
361                throw new StreamCorruptedException(Messages.getString("math.0F")); //$NON-NLS-1$
362            }
363            if (roundingMode == null) {
364                // math.10=null roundingMode
365                throw new StreamCorruptedException(Messages.getString("math.10")); //$NON-NLS-1$
366            }
367        }
