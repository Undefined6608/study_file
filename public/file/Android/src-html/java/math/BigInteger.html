049    /**
050     * This class represents immutable integer numbers of arbitrary length. Large
051     * numbers are typically used in security applications and therefore BigIntegers
052     * offer dedicated functionality like the generation of large prime numbers or
053     * the computation of modular inverse.
054     * <p>
055     * Since the class was modeled to offer all the functionality as the {@link Integer}
056     * class does, it provides even methods that operate bitwise on a two's
057     * complement representation of large integers. Note however that the
058     * implementations favors an internal representation where magnitude and sign
059     * are treated separately. Hence such operations are inefficient and should be
060     * discouraged. In simple words: Do NOT implement any bit fields based on
061     * BigInteger.
062     * <p>
063     * <b>Implementation Note:</b> <br>
064     * The native OpenSSL library with its BIGNUM operations covers all the
065     * meaningful functionality (everything but bit level operations).
066     * 
067     * @since Android 1.0
068     */
069    public class BigInteger extends Number implements Comparable<BigInteger>,
070            Serializable {
071

077        transient private boolean oldReprIsValid = false;

079        void establishOldRepresentation(String caller) {
080            if (!oldReprIsValid) {
081                sign = bigInt.sign();
082                if (sign != 0) digits = bigInt.littleEndianIntsMagnitude();
083                else digits = new int[] { 0 };
084                numberLength = digits.length;
085                oldReprIsValid = true;
086            }
087        }
088    
089        // The name is confusing:
090        // This method is called whenever the old representation has been written.
091        // It ensures that the new representation will be established on demand.
092        //
093        BigInteger withNewRepresentation(String caller) {
094            bigIntIsValid = false;
095            return this;
096        }
097    
098        void validate(String caller, String param) {
099            if (bigIntIsValid) {
100                if (bigInt == null)
101                    System.out.print("Claiming bigIntIsValid BUT bigInt == null, ");
102                else if (bigInt.getNativeBIGNUM() == 0)
103                    System.out.print("Claiming bigIntIsValid BUT bigInt.bignum == 0, ");
104            }
105            else {
106                if (oldReprIsValid) { // establish new representation
107                    if (bigInt == null) bigInt = new BigInt();
108                    bigInt.putLittleEndianInts(digits, (sign < 0));
109                    bigIntIsValid = true;
110                }
111                else {
112                    throw new IllegalArgumentException(caller + ":" + param);
113                }
114            }
115        }

245        /**
246         * Constructs a random non-negative {@code BigInteger} instance in the range
247         * [0, 2^(numBits)-1].
248         * 
249         * @param numBits
250         *            maximum length of the new {@code BigInteger} in bits.
251         * @param rnd
252         *            is an optional random generator to be used.
253         * @throws IllegalArgumentException
254         *             if {@code numBits} < 0.
255         * 
256         * @since Android 1.0
257         */
258        public BigInteger(int numBits, Random rnd) {
259            if (numBits < 0) {
260                // math.1B=numBits must be non-negative
261                throw new IllegalArgumentException(Messages.getString("math.1B")); //$NON-NLS-1$
262            }
263            if (numBits == 0) {
264                sign = 0;
265                numberLength = 1;
266                digits = new int[] { 0 };
267            } else {
268                sign = 1;
269                numberLength = (numBits + 31) >> 5;
270                digits = new int[numberLength];
271                for (int i = 0; i < numberLength; i++) {
272                    digits[i] = rnd.nextInt();
273                }
274                // Using only the necessary bits
275                digits[numberLength - 1] >>>= (-numBits) & 31;
276                cutOffLeadingZeroes();
277            }
278            oldReprIsValid = true;
279            withNewRepresentation("BigInteger(int numBits, Random rnd)");
280        }

282        /**
283         * Constructs a random {@code BigInteger} instance in the range [0,
284         * 2^(bitLength)-1] which is probably prime. The probability that the
285         * returned {@code BigInteger} is prime is beyond (1-1/2^certainty).
286         * <p>
287         * <b>Implementation Note:</b>
288         * Currently {@code rnd} is ignored. The implementation always uses
289         * method {@code bn_rand} from the OpenSSL library. {@code bn_rand} 
290         * generates cryptographically strong pseudo-random numbers.
291         * @see <a href="http://www.openssl.org/docs/crypto/BN_rand.html">
292         * Specification of random generator used from OpenSSL library</a>
293         * 
294         * @param bitLength
295         *            length of the new {@code BigInteger} in bits.
296         * @param certainty
297         *            tolerated primality uncertainty.
298         * @param rnd
299         *            is an optional random generator to be used.
300         * @throws ArithmeticException
301         *             if {@code bitLength} < 2.
302         * 
303         * @since Android 1.0
304         */
305        public BigInteger(int bitLength, int certainty, Random rnd) {
306            if (bitLength < 2) {
307                // math.1C=bitLength < 2
308                throw new ArithmeticException(Messages.getString("math.1C")); //$NON-NLS-1$
309            }
310            bigInt = BigInt.generatePrimeDefault(bitLength, rnd, null);
311            bigIntIsValid = true;
312            // !oldReprIsValid
313        }

337        /**
338         * Constructs a new {@code BigInteger} instance from the string
339         * representation. The string representation consists of an optional minus
340         * sign followed by a non-empty sequence of digits in the specified radix.
341         * For the conversion the method {@code Character.digit(char, radix)} is
342         * used.
343         * 
344         * @param val
345         *            string representation of the new {@code BigInteger}.
346         * @param radix
347         *            the base to be used for the conversion.
348         * @throws NullPointerException
349         *             if {@code val == null}.
350         * @throws NumberFormatException
351         *             if {@code val} is not a valid representation of a {@code
352         *             BigInteger} or if {@code radix < Character.MIN_RADIX} or
353         *             {@code radix > Character.MAX_RADIX}.
354         *             
355         * @since Android 1.0
356         */
357        public BigInteger(String val, int radix) {
358            if (val == null) {
359                throw new NullPointerException();
360            }
361            if (radix == 10) {
362                bigInt = new BigInt();
363                bigInt.putDecString(val);
364                bigIntIsValid = true;
365                // !oldReprIsValid
366            } else if (radix == 16) {
367                bigInt = new BigInt();
368                bigInt.putHexString(val);
369                bigIntIsValid = true;
370                // !oldReprIsValid
371            } else {
372                if ((radix < Character.MIN_RADIX) || (radix > Character.MAX_RADIX)) {
373                    // math.11=Radix out of range
374                    throw new NumberFormatException(Messages.getString("math.11")); //$NON-NLS-1$
375                }
376                if (val.length() == 0) {
377                    // math.12=Zero length BigInteger
378                    throw new NumberFormatException(Messages.getString("math.12")); //$NON-NLS-1$
379                }
380                BigInteger.setFromString(this, val, radix);
381                // oldReprIsValid == true;
382            }
383        }

385        /**
386         * Constructs a new {@code BigInteger} instance with the given sign and the
387         * given magnitude. The sign is given as an integer (-1 for negative, 0 for
388         * zero, 1 for positive). The magnitude is specified as a byte array. The
389         * most significant byte is the entry at index 0.
390         * 
391         * @param signum
392         *            sign of the new {@code BigInteger} (-1 for negative, 0 for
393         *            zero, 1 for positive).
394         * @param magnitude
395         *            magnitude of the new {@code BigInteger} with the most
396         *            significant byte first.
397         * @throws NullPointerException
398         *             if {@code magnitude == null}.
399         * @throws NumberFormatException
400         *             if the sign is not one of -1, 0, 1 or if the sign is zero and
401         *             the magnitude contains non-zero entries.
402         *             
403         * @since Android 1.0
404         */
405        public BigInteger(int signum, byte[] magnitude) {
406            if (magnitude == null) {
407                throw new NullPointerException();
408            }
409            if ((signum < -1) || (signum > 1)) {
410                // math.13=Invalid signum value
411                throw new NumberFormatException(Messages.getString("math.13")); //$NON-NLS-1$
412            }
413            if (signum == 0) {
414                for (byte element : magnitude) {
415                    if (element != 0) {
416                        // math.14=signum-magnitude mismatch
417                        throw new NumberFormatException(Messages.getString("math.14")); //$NON-NLS-1$
418                    }
419                }
420            }
421            bigInt = new BigInt();
422            bigInt.putBigEndian(magnitude, (signum < 0));
423            bigIntIsValid = true;
424        }

426        /**
427         * Constructs a new {@code BigInteger} from the given two's complement
428         * representation. The most significant byte is the entry at index 0. The
429         * most significant bit of this entry determines the sign of the new {@code
430         * BigInteger} instance. The given array must not be empty.
431         * 
432         * @param val
433         *            two's complement representation of the new {@code BigInteger}.
434         * @throws NullPointerException
435         *             if {@code val == null}.
436         * @throws NumberFormatException
437         *             if the length of {@code val} is zero.
438         *             
439         * @since Android 1.0
440         */
441        public BigInteger(byte[] val) {
442            if (val.length == 0) {
443                // math.12=Zero length BigInteger
444                throw new NumberFormatException(Messages.getString("math.12")); //$NON-NLS-1$
445            }
446            bigInt = new BigInt();
447            bigInt.putBigEndianTwosComplement(val);
448            bigIntIsValid = true;
449        }

465            if (val < 0) {

469                return MINUS_ONE;
470            } else if (val <= 10) {

472            } else {// (val > 10)
473                return new BigInteger(1, val);
474            }

596        /**
597         * Returns a new {@code BigInteger} whose value is {@code this << n}. The
598         * result is equivalent to {@code this * 2^n} if n >= 0. The shift distance
599         * may be negative which means that {@code this} is shifted right. The
600         * result then corresponds to {@code floor(this / 2^(-n))}.
601         * <p>
602         * <b>Implementation Note:</b> Usage of this method on negative values is
603         * not recommended as the current implementation is not efficient.
604         * 
605         * @param n
606         *            shift distance.
607         * @return {@code this << n} if {@code n >= 0}; {@code this >> (-n)}.
608         *         otherwise
609         * 
610         * @since Android 1.0
611         */
612        public BigInteger shiftLeft(int n) {
613            if (n == 0) return this;
614            int sign = signum();
615            if (sign == 0) return this;
616            if ((sign > 0) || (n >= 0)) {
617                validate1("shiftLeft", this);
618                return new BigInteger(BigInt.shift(bigInt, n));
619            }
620            else {
621                // Negative numbers faking 2's complement:
622                // Not worth optimizing this:
623                // Sticking to Harmony Java implementation.
624                //
625                return BitLevel.shiftRight(this, -n);
626            }
627        }

635         * {@code bitLength() + 1} bits long.
636         * <p>
637         * The value will fit into an {@code int} if {@code bitLength() < 32} or
638         * into a {@code long} if {@code bitLength() < 64}.
639         * 
640         * @return the length of the minimal two's complement representation for
641         *         {@code this} without the sign bit.

653        /**
654         * Tests whether the bit at position n in {@code this} is set. The result is
655         * equivalent to {@code this & (2^n) != 0}.
656         * <p>
657         * <b>Implementation Note:</b> Usage of this method is not recommended as
658         * the current implementation is not efficient.
659         * 
660         * @param n
661         *            position where the bit in {@code this} has to be inspected.
662         * @return {@code this & (2^n) != 0}.
663         * @throws ArithmeticException
664         *             if {@code n < 0}.
665         * 
666         * @since Android 1.0
667         */
668        public boolean testBit(int n) {
669            if (n < 0) {
670                // math.15=Negative bit address
671                throw new ArithmeticException(Messages.getString("math.15")); //$NON-NLS-1$
672            }
673            int sign = signum();
674            if ((sign > 0) && bigIntIsValid && !oldReprIsValid) {
675                validate1("testBit", this);
676                return bigInt.isBitSet(n);
677            }
678            else {
679                // Negative numbers faking 2's complement:
680                // Not worth optimizing this:
681                // Sticking to Harmony Java implementation.
682                //
683                establishOldRepresentation("testBit");
684                if (n == 0) {
685                    return ((digits[0] & 1) != 0);
686                }
687                int intCount = n >> 5;
688                if (intCount >= numberLength) {
689                    return (sign < 0);
690                }
691                int digit = digits[intCount];
692                n = (1 << (n & 31)); // int with 1 set to the needed position
693                if (sign < 0) {
694                    int firstNonZeroDigit = getFirstNonzeroDigit();
695                    if (  intCount < firstNonZeroDigit  ){
696                        return false;
697                    }else if( firstNonZeroDigit == intCount ){
698                        digit = -digit;
699                    }else{
700                        digit = ~digit;
701                    }
702                }
703                return ((digit & n) != 0);
704            }
705        }

1409        /**
1410         * Returns the two's complement representation of this BigInteger in a byte
1411         * array.
1412         * 
1413         * @return two's complement representation of {@code this}
1414         */
1415        private byte[] twosComplement() {
1416            establishOldRepresentation("twosComplement()");
1417            if( this.sign == 0 ){
1418                return new byte[]{0};
1419            }
1420            BigInteger temp = this;
1421            int bitLen = bitLength();
1422            int iThis = getFirstNonzeroDigit();
1423            int bytesLen = (bitLen >> 3) + 1;
1424            /* Puts the little-endian int array representing the magnitude
1425             * of this BigInteger into the big-endian byte array. */
1426            byte[] bytes = new byte[bytesLen];
1427            int firstByteNumber = 0;
1428            int highBytes;
1429            int digitIndex = 0;
1430            int bytesInInteger = 4;
1431            int digit;
1432            int hB;
1433    
1434            if (bytesLen - (numberLength << 2) == 1) {
1435                bytes[0] = (byte) ((sign < 0) ? -1 : 0);
1436                highBytes = 4;
1437                firstByteNumber++;
1438            } else {
1439                hB = bytesLen & 3;
1440                highBytes = (hB == 0) ? 4 : hB;
1441            }
1442            
1443            digitIndex = iThis;
1444            bytesLen -= iThis << 2;
1445            
1446            if (sign < 0) {
1447                digit = -temp.digits[digitIndex];
1448                digitIndex++;
1449                if(digitIndex == numberLength){
1450                    bytesInInteger = highBytes;
1451                }
1452                for (int i = 0; i < bytesInInteger; i++, digit >>= 8) {
1453                    bytes[--bytesLen] = (byte) digit;
1454                }
1455                while( bytesLen > firstByteNumber ){
1456                    digit = ~temp.digits[digitIndex];
1457                    digitIndex++;
1458                    if(digitIndex == numberLength){
1459                        bytesInInteger = highBytes;
1460                    }
1461                    for (int i = 0; i < bytesInInteger; i++, digit >>= 8) {
1462                        bytes[--bytesLen] = (byte) digit;
1463                    }
1464                }
1465            } else {
1466                while (bytesLen > firstByteNumber) {
1467                    digit = temp.digits[digitIndex];
1468                    digitIndex++;
1469                    if (digitIndex == numberLength) {
1470                        bytesInInteger = highBytes;
1471                    }
1472                    for (int i = 0; i < bytesInInteger; i++, digit >>= 8) {
1473                        bytes[--bytesLen] = (byte) digit;
1474                    }
1475                }
1476            }
1477            return bytes;
1478        }

1481        static int multiplyByInt(int res[], int a[], final int aSize, final int factor) {
1482            long carry = 0;
1483    
1484            for (int i = 0; i < aSize; i++) {
1485                carry += (a[i] & 0xFFFFFFFFL) * (factor & 0xFFFFFFFFL);
1486                res[i] = (int)carry;
1487                carry >>>= 32;
1488            }
1489            return (int)carry;
1490        }

1492        static int inplaceAdd(int a[], final int aSize, final int addend) {
1493            long carry = addend & 0xFFFFFFFFL;
1494    
1495            for (int i = 0; (carry != 0) && (i < aSize); i++) {
1496                carry += a[i] & 0xFFFFFFFFL;
1497                a[i] = (int) carry;
1498                carry >>= 32;
1499            }
1500            return (int) carry;
1501        }

1503        /** @see BigInteger#BigInteger(String, int) */
1504        private static void setFromString(BigInteger bi, String val, int radix) {
1505            int sign;
1506            int[] digits;
1507            int numberLength;
1508            int stringLength = val.length();
1509            int startChar;
1510            int endChar = stringLength;
1511    
1512            if (val.charAt(0) == '-') {
1513                sign = -1;
1514                startChar = 1;
1515                stringLength--;
1516            } else {
1517                sign = 1;
1518                startChar = 0;
1519            }
1520            /*
1521             * We use the following algorithm: split a string into portions of n
1522             * characters and convert each portion to an integer according to the
1523             * radix. Then convert an exp(radix, n) based number to binary using the
1524             * multiplication method. See D. Knuth, The Art of Computer Programming,
1525             * vol. 2.
1526             */
1527    
1528            int charsPerInt = Conversion.digitFitInInt[radix];
1529            int bigRadixDigitsLength = stringLength / charsPerInt;
1530            int topChars = stringLength % charsPerInt;
1531    
1532            if (topChars != 0) {
1533                bigRadixDigitsLength++;
1534            }
1535            digits = new int[bigRadixDigitsLength];
1536            // Get the maximal power of radix that fits in int
1537            int bigRadix = Conversion.bigRadices[radix - 2];
1538            // Parse an input string and accumulate the BigInteger's magnitude
1539            int digitIndex = 0; // index of digits array
1540            int substrEnd = startChar + ((topChars == 0) ? charsPerInt : topChars);
1541            int newDigit;
1542    
1543            for (int substrStart = startChar; substrStart < endChar; substrStart = substrEnd, substrEnd = substrStart
1544                    + charsPerInt) {
1545                int bigRadixDigit = Integer.parseInt(val.substring(substrStart,
1546                        substrEnd), radix);
1547                newDigit = multiplyByInt(digits, digits, digitIndex, bigRadix);
1548                newDigit += inplaceAdd(digits, digitIndex, bigRadixDigit);
1549                digits[digitIndex++] = newDigit;
1550            }
1551            numberLength = digitIndex;
1552            bi.sign = sign;
1553            bi.numberLength = numberLength;
1554            bi.digits = digits;
1555            bi.cutOffLeadingZeroes();
1556            bi.oldReprIsValid = true;
1557            bi.withNewRepresentation("Cordoba-BigInteger: private static setFromString");
1558        }

1562        final void cutOffLeadingZeroes() {
1563            while ((numberLength > 0) && (digits[--numberLength] == 0)) {
1564                ;
1565            }
1566            if (digits[numberLength++] == 0) {
1567                sign = 0;
