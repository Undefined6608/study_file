018    package java.sql;
019    
020    import java.io.InputStream;
021    import java.io.Reader;
022    import java.math.BigDecimal;
023    import java.net.URL;
024    
025    /**
026     * The interface for an output stream used to write attributes of an SQL <i>User
027     * Defined Type</i> (UDT) to the database. This interface is used for custom
028     * mapping of types and is called by the JDBC driver. It is not intended to be
029     * used by applications.
030     * <p>
031     * When an object which implements the {@code SQLData} interface is used as an
032     * argument to an SQL statement, the JDBC driver calls the method {@code
033     * SQLData.getSQLType} to establish the type of the SQL UDT that is being
034     * passed. The driver then creates an {@code SQLOutput} stream and passes it to
035     * the {@code SQLData.writeSQL} method, which in turn uses the appropriate
036     * {@code SQLOutput} writer methods to write the data from the {@code SQLData}
037     * object into the stream according to the defined mapping.
038     * </p>
039     * 
040     * @see SQLData
041     * 
042     * @since Android 1.0
043     */
044    public interface SQLOutput {
045    
046        /**
047         * Write a {@code String} value into the output stream.
048         * 
049         * @param theString
050         *            the {@code String} to write.
051         * @throws SQLException
052         *             if a database error occurs.
053         * @since Android 1.0
054         */
055        public void writeString(String theString) throws SQLException;
056    
057        /**
058         * Write a {@code boolean} value into the output stream.
059         * 
060         * @param theFlag
061         *            the {@code boolean} value to write.
062         * @throws SQLException
063         *             if a database error occurs.
064         * @since Android 1.0
065         */
066        public void writeBoolean(boolean theFlag) throws SQLException;
067    
068        /**
069         * Write a {@code byte} value into the output stream.
070         * 
071         * @param theByte
072         *            the {@code byte} value to write.
073         * @throws SQLException
074         *             if a database error occurs.
075         * @since Android 1.0
076         */
077        public void writeByte(byte theByte) throws SQLException;
078    
079        /**
080         * Write a {@code short} value into the output stream.
081         * 
082         * @param theShort
083         *            the {@code short} value to write.
084         * @throws SQLException
085         *             if a database error occurs.
086         * @since Android 1.0
087         */
088        public void writeShort(short theShort) throws SQLException;
089    
090        /**
091         * Write an {@code int} value into the output stream.
092         * 
093         * @param theInt
094         *            the {@code int} value to write.
095         * @throws SQLException
096         *             if a database error occurs.
097         * @since Android 1.0
098         */
099        public void writeInt(int theInt) throws SQLException;
100    
101        /**
102         * Write a {@code long} value into the output stream.
103         * 
104         * @param theLong
105         *            the {@code long} value to write.
106         * @throws SQLException
107         *             if a database error occurs.
108         * @since Android 1.0
109         */
110        public void writeLong(long theLong) throws SQLException;
111    
112        /**
113         * Write a {@code float} value into the output stream.
114         * 
115         * @param theFloat
116         *            the {@code float} value to write.
117         * @throws SQLException
118         *             if a database error occurs.
119         * @since Android 1.0
120         */
121        public void writeFloat(float theFloat) throws SQLException;
122    
123        /**
124         * Write a {@code double} value into the output stream.
125         * 
126         * @param theDouble
127         *            the {@code double} value to write.
128         * @throws SQLException
129         *             if a database error occurs.
130         * @since Android 1.0
131         */
132        public void writeDouble(double theDouble) throws SQLException;
133    
134        /**
135         * Write a {@code java.math.BigDecimal} value into the output stream.
136         * 
137         * @param theBigDecimal
138         *            the {@code BigDecimal} value to write.
139         * @throws SQLException
140         *             if a database error occurs.
141         * @since Android 1.0
142         */
143        public void writeBigDecimal(BigDecimal theBigDecimal) throws SQLException;
144    
145        /**
146         * Write an array of bytes into the output stream.
147         * 
148         * @param theBytes
149         *            the array of bytes to write.
150         * @throws SQLException
151         *             if a database error occurs.
152         * @since Android 1.0
153         */
154        public void writeBytes(byte[] theBytes) throws SQLException;
155    
156        /**
157         * Write a {@code java.sql.Date} value into the output stream.
158         * 
159         * @param theDate
160         *            the {@code Date} value to write.
161         * @throws SQLException
162         *             if a database error occurs.
163         * @see Date
164         * @since Android 1.0
165         */
166        public void writeDate(Date theDate) throws SQLException;
167    
168        /**
169         * Write a {@code java.sql.Time} value into the output stream.
170         * 
171         * @param theTime
172         *            the {@code Time} value to write.
173         * @throws SQLException
174         *             if a database error occurs.
175         * @see Time
176         * @since Android 1.0
177         */
178        public void writeTime(Time theTime) throws SQLException;
179    
180        /**
181         * Write a {@code java.sql.Timestamp} value into the output stream.
182         * 
183         * @param theTimestamp
184         *            the {@code Timestamp} value to write.
185         * @throws SQLException
186         *             if a database error occurs.
187         * @see Timestamp
188         * @since Android 1.0
189         */
190        public void writeTimestamp(Timestamp theTimestamp) throws SQLException;
191    
192        /**
193         * Write a stream of unicode characters into the output stream.
194         * 
195         * @param theStream
196         *            the stream of unicode characters to write, as a {@code
197         *            java.io.Reader} object.
198         * @throws SQLException
199         *             if a database error occurs.
200         * @since Android 1.0
201         */
202        public void writeCharacterStream(Reader theStream) throws SQLException;
203    
204        /**
205         * Write a stream of ASCII characters into the output stream.
206         * 
207         * @param theStream
208         *            the stream of ASCII characters to write, as a {@code
209         *            java.io.InputStream} object
210         * @throws SQLException
211         *             if a database error occurs.
212         * @since Android 1.0
213         */
214        public void writeAsciiStream(InputStream theStream) throws SQLException;
215    
216        /**
217         * Write a stream of uninterpreted bytes into the output stream.
218         * 
219         * @param theStream
220         *            the stream of bytes to write, as a {@code java.io.InputStream}
221         *            object
222         * @throws SQLException
223         *             if a database error occurs.
224         * @since Android 1.0
225         */
226        public void writeBinaryStream(InputStream theStream) throws SQLException;
227    
228        /**
229         * Write an {@code SQLData} object into the output stream.
230         * <p>
231         * If the {@code SQLData} object is null, writes {@code NULL} to the stream.
232         * </p>
233         * <p>
234         * Otherwise, calls the {@code SQLData.writeSQL} method of the object, which
235         * writes the object's attributes to the stream by calling the appropriate
236         * SQLOutput writer methods for each attribute, in order. The order of the
237         * attributes is the order they are listed in the SQL definition of the User
238         * Defined Type.
239         * </p>
240         * 
241         * @param theObject
242         *            the {@code SQLData} object to write.
243         * @throws SQLException
244         *             if a database error occurs.
245         * @see SQLData
246         * @since Android 1.0
247         */
248        public void writeObject(SQLData theObject) throws SQLException;
249    
250        /**
251         * Write an SQL {@code Ref} value into the output stream.
252         * 
253         * @param theRef
254         *            the {@code java.sql.Ref} object to write.
255         * @throws SQLException
256         *             if a database error occurs.
257         * @see Ref
258         * @since Android 1.0
259         */
260        public void writeRef(Ref theRef) throws SQLException;
261    
262        /**
263         * Write an SQL {@code Blob} value into the output stream.
264         * 
265         * @param theBlob
266         *            the {@code java.sql.Blob} object to write.
267         * @throws SQLException
268         *             if a database error occurs.
269         * @see Blob
270         * @since Android 1.0
271         */
272        public void writeBlob(Blob theBlob) throws SQLException;
273    
274        /**
275         * Write an SQL {@code Clob} value into the output stream.
276         * 
277         * @param theClob
278         *            the {@code java.sql.Clob} object to write.
279         * @throws SQLException
280         *             if a database error occurs.
281         * @see Clob
282         * @since Android 1.0
283         */
284        public void writeClob(Clob theClob) throws SQLException;
285    
286        /**
287         * Write an SQL {@code Struct} value into the output stream.
288         * 
289         * @param theStruct
290         *            the {@code java.sql.Struct} object to write.
291         * @throws SQLException
292         *             if a database error occurs.
293         * @see Struct
294         * @since Android 1.0
295         */
296        public void writeStruct(Struct theStruct) throws SQLException;
297    
298        /**
299         * Write an SQL {@code Array} value into the output stream.
300         * 
301         * @param theArray
302         *            the {@code java.sql.Array} object to write.
303         * @throws SQLException
304         *             if a database error occurs.
305         * @see Array
306         * @since Android 1.0
307         */
308        public void writeArray(Array theArray) throws SQLException;
309    
310        /**
311         * Write an SQL {@code DATALINK} value into the output stream.
312         * 
313         * @param theURL
314         *            the datalink value as a {@code java.net.URL} to write.
315         * @throws SQLException
316         *             if a database error occurs.
317         * @see java.net.URL
318         * @since Android 1.0
319         */
320        public void writeURL(URL theURL) throws SQLException;
321    }
