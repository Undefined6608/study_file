018    package java.sql;

022    /**
023     * A connection represents a link from a Java application to a database. All SQL
024     * statements and results are returned within the context of a connection.
025     * Database statements that are executed within this context form a
026     * database session which forms one or more closed transactions. Especially In distributed applications, multiple concurrent connections may exist accessing the same values of the database.
027     * which may lead to the following phenomena (referred to as <i>transaction isolation levels</i>):
028     * <ul>
029     * <li><i>dirty reads</i>:<br>
030     * reading values from table rows that are not committed.</br></li>
031     * <li><i>non-repeatable reads</i>:<br>
032     * reading table rows more than once in a transaction but getting back different
033     * data because other transactions have altered the rows between the reads.</br></li>
034     * <li><i>phantom reads</i>:<br>
035     * retrieving additional "phantom" rows in the course of repeated table reads
036     * because other transactions have inserted additional rows that satisfy an
037     * SQL {@code WHERE} clause</br></li>
038     * </ul>
039     *  
040     * @since Android 1.0
041     */
042    public interface Connection {
043    
044        /**
045         * A constant indicating that transactions are not supported.
046         * 
047         * @since Android 1.0
048         */
049        public static final int TRANSACTION_NONE = 0;
050    
051        /**
052         * No <i>dirty reads</i> are permitted, therefore transactions may not read
053         * a row containing uncommitted values - but does not prevent an application
054         * from <i>non-repeatable reads</i> and <i>phantom reads</i>.
055         * 
056         * @since Android 1.0
057         */
058        public static final int TRANSACTION_READ_COMMITTED = 2;
059    
060        /**
061         * In the case that reading uncommitted values is allowed, the following
062         * incidents may happen which may lead to an invalid results:
063         * <ul>
064         * <li><i>dirty reads</i></li>
065         * <li><i>non-repeatable reads</i></li>
066         * <li><i>phantom reads</i></li>
067         * </ul>
068         * 
069         * @since Android 1.0
070         */
071        public static final int TRANSACTION_READ_UNCOMMITTED = 1;
072    
073        /**
074         * A constant indicating that <i>dirty reads</i> and <i>non-repeatable
075         * reads</i> are <b>prevented</b> but <i>phantom reads</i> can occur.
076         * 
077         * @since Android 1.0
078         */
079        public static final int TRANSACTION_REPEATABLE_READ = 4;
080    
081        /**
082         * The constant that indicates that the following incidents are <b>all
083         * prevented</b> (the opposite of {@link #TRANSACTION_READ_UNCOMMITTED}):
084         * <ul>
085         * <li><i>dirty reads</i></li>
086         * <li><i>non-repeatable reads</i></li>
087         * <li><i>phantom reads</i></li>
088         * </ul>
089         * 
090         * @since Android 1.0
091         */
092        public static final int TRANSACTION_SERIALIZABLE = 8;
093    
094        /**
095         * Discards all warnings that may have arisen for this connection.
096         * Subsequent calls to {@link #getWarnings()} will return {@code null}
097         * up until a new warning condition occurs.
098         * 
099         * @throws SQLException
100         *             if there is a problem accessing the database.
101         */
102        public void clearWarnings() throws SQLException;
103    
104        /**
105         * Causes the instant release of all database and driver connection
106         * resources associated with this object. Any subsequent invocations of this
107         * method have no effect.
108         * <p>
109         * It is strongly recommended that all connections are closed before they
110         * are dereferenced by the application ready for garbage collection.
111         * Although the {@code finalize} method of the connection closes the
112         * connection before garbage collection takes place, it is not advisable to
113         * leave the {@code close} operation to take place in this way. Mainly
114         * because undesired side-effects may appear.
115         * </p>
116         * 
117         * @throws SQLException
118         *             if there is a problem accessing the database.
119         */
120        public void close() throws SQLException;
121    
122        /**
123         * Commits all of the changes made since the last {@code commit} or
124         * {@code rollback} of the associated transaction. All locks in the database
125         * held by this connection are also relinquished. Calling this operation on
126         * connection objects in {@code auto-commit} mode leads to an error.
127         * 
128         * @throws SQLException
129         *             if there is a problem accessing the database or if the target
130         *             connection instance is in auto-commit mode.
131         */
132        public void commit() throws SQLException;
133    
134        /**
135         * Returns a new instance of {@code Statement} for issuing SQL commands to
136         * the remote database.
137         * <p>
138         * {@code ResultSets} generated by the returned statement will default to
139         * type {@code ResultSet.TYPE_FORWARD_ONLY} and concurrency level {@code
140         * ResultSet.CONCUR_READ_ONLY}.
141         * 
142         * @return a {@code Statement} object with default settings.
143         * @throws SQLException
144         *             if there is a problem accessing the database.
145         * @see ResultSet
146         */
147        public Statement createStatement() throws SQLException;
148    
149        /**
150         * Returns a new instance of {@code Statement} whose associated {@code
151         * ResultSet}s have the characteristics specified in the type and
152         * concurrency arguments.
153         * 
154         * @param resultSetType
155         *            one of the following type specifiers:
156         *            <ul>
157         *            <li>{@link ResultSet#TYPE_SCROLL_SENSITIVE} </li> <li>
158         *            {@link ResultSet#TYPE_SCROLL_INSENSITIVE} </li> <li>
159         *            {@link ResultSet#TYPE_FORWARD_ONLY}</li>
160         *            </ul>
161         * @param resultSetConcurrency
162         *            one of the following concurrency mode specifiers:
163         *            <ul>
164         *            <li>{@link ResultSet#CONCUR_UPDATABLE}</li> <li>
165         *            {@link ResultSet#CONCUR_READ_ONLY}</li>
166         *            </ul>
167         * @return a new instance of {@code Statement} capable of manufacturing
168         *         {@code ResultSet}s that satisfy the specified {@code
169         *         resultSetType} and {@code resultSetConcurrency} values.
170         * @throws SQLException
171         *             if there is a problem accessing the database
172         */
173        public Statement createStatement(int resultSetType, int resultSetConcurrency)
174                throws SQLException;
175    
176        /**
177         * Returns a new instance of {@code Statement} whose associated
178         * {@code ResultSet}s will have the characteristics specified in the
179         * type, concurrency and holdability arguments.
180         * 
181         * @param resultSetType
182         *            one of the following type specifiers:
183         *            <ul>
184         *            <li>{@link ResultSet#TYPE_SCROLL_SENSITIVE}</li>
185         *            <li>{@link ResultSet#TYPE_SCROLL_INSENSITIVE}</li>
186         *            <li>{@link ResultSet#TYPE_FORWARD_ONLY}</li>
187         *            </ul>
188         * @param resultSetConcurrency
189         *            one of the following concurrency mode specifiers:
190         *            <ul>
191         *            <li>{@link ResultSet#CONCUR_UPDATABLE}</li>
192         *            <li>{@link ResultSet#CONCUR_READ_ONLY}</li>
193         *            </ul>
194         * @param resultSetHoldability
195         *            one of the following holdability mode specifiers:
196         *            <ul>
197         *            <li>{@link ResultSet#HOLD_CURSORS_OVER_COMMIT}</li>
198         *            <li>{@link ResultSet#CLOSE_CURSORS_AT_COMMIT}</li>
199         *            </ul>
200         * @return a new instance of {@code Statement} capable of
201         *         manufacturing {@code ResultSet}s that satisfy the
202         *         specified {@code resultSetType},
203         *         {@code resultSetConcurrency} and
204         *         {@code resultSetHoldability} values.
205         * @throws SQLException
206         *             if there is a problem accessing the database.
207         */
208        public Statement createStatement(int resultSetType,
209                int resultSetConcurrency, int resultSetHoldability)
210                throws SQLException;
211

259         */

262        /**
263         * Returns the transaction isolation level for this connection.

268         * @see #TRANSACTION_NONE
269         * @see #TRANSACTION_READ_COMMITTED
270         * @see #TRANSACTION_READ_UNCOMMITTED
271         * @see #TRANSACTION_REPEATABLE_READ
272         * @see #TRANSACTION_SERIALIZABLE
273         * @since Android 1.0
274         */
275        public int getTransactionIsolation() throws SQLException;
276    
277        /**
278         * Returns the type mapping associated with this {@code Connection} object.

281         * @return the Type Map as a {@code java.util.Map}.

288        /**
289         * Gets the first instance of any {@code SQLWarning} objects that may have

603        /**
604         * Creates a default {@code PreparedStatement} that can retrieve the

607         * names of each column in the statement's associated database table

614         * statement will not reach the database server until it is executed. This
615         * will have a bearing on precisely <i>when</i> {@code SQLException}
616         * instances get raised.
617         * </p>
618         * <p>
619         * By default, ResultSets from the returned object will be
620         * {@link ResultSet#TYPE_FORWARD_ONLY} type with a
621         * {@link ResultSet#CONCUR_READ_ONLY} concurrency mode.
622         * </p>
623         * 
624         * @param sql
625         *            the SQL statement.
626         * @param columnNames
627         *            the names of the columns for which auto-generated keys should
628         *            be made available.
629         * @return the PreparedStatement containing the supplied SQL statement.
630         * @throws SQLException
631         *             if a problem occurs accessing the database.
632         * @since Android 1.0
633         */
634        public PreparedStatement prepareStatement(String sql, String[] columnNames)
635                throws SQLException;

676        /**
677         * Sets this connection's auto-commit mode {@code on} or {@code off}.
678         * <p>
679         * Putting a Connection into auto-commit mode means that all associated SQL
680         * statements are run and committed as separate transactions.
681         * By contrast, setting auto-commit to {@code off} means that associated SQL
682         * statements get grouped into transactions that need to be completed by
683         * explicit calls to either the {@link #commit()} or {@link #rollback()}
684         * methods.
685         * </p>
686         * Auto-commit is the default mode for new connection instances.
687         * <p>
688         * When in this mode, commits will automatically occur upon successful SQL
689         * statement completion or upon successful completion of an execute.
690         * Statements are not considered successfully completed until all associated
691         * {@code ResultSet}s and output parameters have been obtained or closed.
692         * </p>
693         * <p>
694         * Calling this operation during an uncommitted transaction will result in
695         * it being committed.
696         * </p>
697         * 
698         * @param autoCommit
699         *            {@code boolean} indication of whether to put the target
700         *            connection into auto-commit mode ({@code true}) or not (
701         *            {@code false}).
702         * @throws SQLException
703         *             if there is a problem accessing the database.
704         * @since Android 1.0
705         */
706        public void setAutoCommit(boolean autoCommit) throws SQLException;
