018    package java.sql;
019    
020    import java.util.Properties;
021    
022    /**
023     * An interface to a JDBC driver.
024     * <p>
025     * The JDBC driver uses URLs to specify the location of specific data. URL
026     * format typically takes the form " {@code xxxx:yyyy:SpecificData}", where "
027     * {@code xxxx:yyyy}" is referred to as the <i>subprotocol</i> and is normally
028     * the same for all of a particular driver. " {@code SpecificData}" is a string
029     * which identifies the particular data source that the driver should use.
030     * </p>
031     * <p>
032     * A driver needs to be registered with a {@link DriverManager}. It is
033     * registered and instantiated by calling {@code Class.forName("DriverURL")}
034     * with the URL string as argument.
035     * </p>
036     * @see DriverManager
037     *  
038     * @since Android 1.0
039     */
040    public interface Driver {
041    
042        /**
043         * Returns whether the driver thinks that it can open a connection to the
044         * given URL.
045         * 
046         * @param url
047         *            the URL to connect to.
048         * @return {@code true} if the driver thinks that is can open a connection
049         *         to the supplied URL, {@code false} otherwise. Typically, the
050         *         driver will respond {@code true} if it thinks that it can handle
051         *         the subprotocol specified by the driver.
052         * @throws SQLException
053         *          if a database error occurs.
054         * @since Android 1.0
055         */
056        public boolean acceptsURL(String url) throws SQLException;
057    
058        /**
059         * Attempts to make a database connection to a data source specified by a
060         * supplied URL.
061         * 
062         * @param url
063         *            the URL to connect.
064         * @param info
065         *            some properties that should be used in establishing the
066         *            connection. The properties consist of name/value pairs of
067         *            strings. Normally, a connection to a database requires at
068         *            least two properties - for {@code "user"} and {@code
069         *            "password"} in order to pass authentication to the database.
070         * @return the connection to the database.
071         * @throws SQLException
072         *             if a database error occurs.
073         * @since Android 1.0
074         */
075        public Connection connect(String url, Properties info) throws SQLException;
076    
077        /**
078         * Gets the driver's major version number.
079         * 
080         * @return the major version number of the driver - typically starts at 1.
081         * @since Android 1.0
082         */
083        public int getMajorVersion();
084    
085        /**
086         * Gets the driver's minor version number.
087         * 
088         * @return the minor version number of the driver - typically starts at 0.
089         * @since Android 1.0
090         */
091        public int getMinorVersion();
092    
093        /**
094         * Gets information about possible properties for this driver.
095         * <p>
096         * This method is intended to provide a listing of possible properties that
097         * the client of the driver must supply in order to establish a connection
098         * to a database. Note that the returned array of properties may change
099         * depending on the supplied list of property values.
100         * </p>
101         * 
102         * @param url
103         *            the URL of the database. An application may call this method
104         *            iteratively as the property list is built up - for example,
105         *            when displaying a dialog to an end-user as part of the
106         *            database login process.
107         * @param info
108         *            a set of tag/value pairs giving data that a user may be 
109         *            prompted to provide in order to connect to the database.
110         * @return an array of {@code DriverPropertyInfo} records which provide
111         *         details on which additional properties are required (in addition
112         *         to those supplied in the {@code info} parameter) in order to 
113         *         connect to the database.
114         * @throws SQLException
115         *             if a database error occurs.
116         * @since Android 1.0
117         */
118        public DriverPropertyInfo[] getPropertyInfo(String url, Properties info)
119                throws SQLException;
120    
121        /**
122         * Reports whether this driver is a genuine JDBC CompliantTM driver. The
123         * driver may only return {@code true} if it passes all the JDBC compliance
124         * tests.
125         * <p>
126         * A driver may not be fully compliant if the underlying database has
127         * limited functionality.
128         * </p>
129         * 
130         * @return {@code true} if the driver is fully JDBC compliant, {@code false}
131         *         otherwise.
132         * @since Android 1.0
133         */
134        public boolean jdbcCompliant();
135    
136    }
