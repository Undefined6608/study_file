019    
020    import java.text.DecimalFormat;
021    import java.text.ParsePosition;
022    import java.text.SimpleDateFormat;
023    import java.util.Date;

025    import org.apache.harmony.sql.internal.nls.Messages;

027    /**
028     * A Java representation of the SQL {@code TIMESTAMP} type. It provides the
029     * capability of representing the SQL {@code TIMESTAMP} nanosecond value, in
030     * addition to the regular date/time value which has millisecond resolution.
031     * <p>
032     * The {@code Timestamp} class consists of a regular date/time value, where only
033     * the integral seconds value is stored, plus a nanoseconds value where the
034     * fractional seconds are stored.
035     * </p><p>
036     * The addition of the nanosecond value field to the {@code Timestamp} object
037     * makes it significantly different from the {@code java.util.Date} object which
038     * it extends. Users should be aware that {@code Timestamp} objects are not
039     * interchangable with {@code java.util.Date} objects when used outside the
040     * confines of the {@code java.sql} package.
041     * </p>
042     * 
043     * @see Date
044     * @see Time
045     * @see java.util.Date
046     * @since Android 1.0
047     */
048    public class Timestamp extends Date {
049    
050        private static final long serialVersionUID = 2745179027874758501L;
051    
052        // The nanoseconds time value of the Timestamp
053        private int nanos;
054    
055        /**
056         * Returns a {@code Timestamp} corresponding to the time specified by the
057         * supplied values for <i>Year</i>, <i>Month</i>, <i>Date</i>, <i>Hour</i>,
058         * <i>Minutes</i>, <i>Seconds</i> and <i>Nanoseconds</i>.
059         * 
060         * @deprecated Please use the constructor {@link #Timestamp(long)}.
061         * @param theYear
062         *            specified as the year minus 1900.
063         * @param theMonth
064         *            specified as an integer in the range [0,11].
065         * @param theDate
066         *            specified as an integer in the range [1,31].
067         * @param theHour
068         *            specified as an integer in the range [0,23].
069         * @param theMinute
070         *            specified as an integer in the range [0,59].
071         * @param theSecond
072         *            specified as an integer in the range [0,59].
073         * @param theNano
074         *            which defines the nanosecond value of the timestamp specified
075         *            as an integer in the range [0,999'999'999]
076         * @throws IllegalArgumentException
077         *             if any of the parameters is out of range.
078         * @since Android 1.0
079         */
080        @SuppressWarnings("deprecation")
081        @Deprecated
082        public Timestamp(int theYear, int theMonth, int theDate, int theHour,
083                int theMinute, int theSecond, int theNano)
084                throws IllegalArgumentException {
085            super(theYear, theMonth, theDate, theHour, theMinute, theSecond);
086            if (theNano < 0 || theNano > 999999999) {
087                throw new IllegalArgumentException();
088            }
089            nanos = theNano;
090        }

092        /**
093         * Returns a {@code Timestamp} object corresponding to the time represented
094         * by a supplied time value.
095         * 
096         * @param theTime
097         *            a time value in the format of milliseconds since the Epoch
098         *            (January 1 1970 00:00:00.000 GMT).
099         * @since Android 1.0
100         */
101        public Timestamp(long theTime) {
102            super(theTime);
103            /*
104             * Now set the time for this Timestamp object - which deals with the
105             * nanosecond value as well as the base time
106             */
107            this.setTime(theTime);
108        }
109    
110        /**
111         * Returns {@code true} if this timestamp object is later than the supplied
112         * timestamp, otherwise returns {@code false}.
113         * 
114         * @param theTimestamp
115         *            the timestamp to compare with this timestamp object.
116         * @return {@code true} if this {@code Timestamp} object is later than the
117         *         supplied timestamp, {@code false} otherwise.
118         * @since Android 1.0
119         */
120        public boolean after(Timestamp theTimestamp) {
121            long thisTime = this.getTime();
122            long compareTime = theTimestamp.getTime();
123    
124            // If the time value is later, the timestamp is later
125            if (thisTime > compareTime) {
126                return true;
127            }
128            // If the time value is earlier, the timestamp is not later
129            else if (thisTime < compareTime) {
130                return false;
131            }
132            /*
133             * Otherwise the time values are equal in which case the nanoseconds
134             * value determines whether this timestamp is later...
135             */
136            else if (this.getNanos() > theTimestamp.getNanos()) {
137                return true;
138            } else {
139                return false;
140            }
141        }
142    
143        /**
144         * Returns {@code true} if this {@code Timestamp} object is earlier than the
145         * supplied timestamp, otherwise returns {@code false}.
146         * 
147         * @param theTimestamp
148         *            the timestamp to compare with this {@code Timestamp} object.
149         * @return {@code true} if this {@code Timestamp} object is earlier than the
150         *         supplied timestamp, {@code false} otherwise.
151         * @since Android 1.0
152         */
153        public boolean before(Timestamp theTimestamp) {
154            long thisTime = this.getTime();
155            long compareTime = theTimestamp.getTime();
156    
157            // If the time value is later, the timestamp is later
158            if (thisTime < compareTime) {
159                return true;
160            }
161            // If the time value is earlier, the timestamp is not later
162            else if (thisTime > compareTime) {
163                return false;
164            }
165            /*
166             * Otherwise the time values are equal in which case the nanoseconds
167             * value determines whether this timestamp is later...
168             */
169            else if (this.getNanos() < theTimestamp.getNanos()) {
170                return true;
171            } else {
172                return false;
173            }
174        }
175    
176        /**
177         * Compares this {@code Timestamp} object with a supplied {@code Timestamp}
178         * object.
179         * 
180         * @param theObject
181         *            the timestamp to compare with this {@code Timestamp} object,
182         *            passed as an {@code Object}.
183         * @return <dd>
184         *         <dl>
185         *         {@code 0} if the two {@code Timestamp} objects are equal in time
186         *         </dl>
187         *         <dl>
188         *         a value {@code < 0} if this {@code Timestamp} object is before
189         *         the supplied {@code Timestamp} and a value
190         *         </dl>
191         *         <dl>
192         *         {@code > 0} if this {@code Timestamp} object is after the
193         *         supplied {@code Timestamp}
194         *         </dl>
195         *         </dd>
196         * @throws ClassCastException
197         *             if the supplied object is not a {@code Timestamp} object.
198         * @since Android 1.0
199         */
200        @Override
201        public int compareTo(Date theObject) throws ClassCastException {
202            return this.compareTo((Timestamp) theObject);
203        }
204    
205        /**
206         * Compares this {@code Timestamp} object with a supplied {@code Timestamp}
207         * object.
208         * 
209         * @param theTimestamp
210         *            the timestamp to compare with this {@code Timestamp} object,
211         *            passed in as a {@code Timestamp}.
212         * @return one of the following:
213         *         <ul>
214         *         <li>{@code 0}, if the two {@code Timestamp} objects are
215         *         equal in time</li>
216         *         <li>{@code < 0}, if this {@code Timestamp} object is before the
217         *         supplied {@code Timestamp}</li>
218         *         <li> {@code > 0}, if this {@code Timestamp} object is after the
219         *         supplied {@code Timestamp}</li>
220         *         </ul>
221         * @since Android 1.0
222         */
223        public int compareTo(Timestamp theTimestamp) {
224            int result = super.compareTo(theTimestamp);
225            if (result == 0) {
226                int thisNano = this.getNanos();
227                int thatNano = theTimestamp.getNanos();
228                if (thisNano > thatNano) {
229                    return 1;
230                } else if (thisNano == thatNano) {
231                    return 0;
232                } else {
233                    return -1;
234                }
235            }
236            return result;
237        }

239        /**
240         * Tests to see if this timestamp is equal to a supplied object.
241         * 
242         * @param theObject
243         *            the object to which this timestamp is compared.
244         * @return {@code true} if this {@code Timestamp} object is equal to the
245         *         supplied {@code Timestamp} object<br>{@code false} if the object
246         *         is not a {@code Timestamp} object or if the object is a {@code
247         *         Timestamp} but represents a different instant in time.
248         * @since Android 1.0
249         */
250        @Override
251        public boolean equals(Object theObject) {
252            if (theObject instanceof Timestamp) {
253                return equals((Timestamp) theObject);
254            }
255            return false;
256        }
257    
258        /**
259         * Tests to see if this timestamp is equal to a supplied timestamp.
260         * 
261         * @param theTimestamp
262         *            the timestamp to compare with this {@code Timestamp} object,
263         *            passed as an {@code Object}.
264         * @return {@code true} if this {@code Timestamp} object is equal to the
265         *         supplied {@code Timestamp} object, {@code false} otherwise.
266         * @since Android 1.0
267         */
268        public boolean equals(Timestamp theTimestamp) {
269            if (theTimestamp == null) {
270                return false;
271            }
272            return (this.getTime() == theTimestamp.getTime())
273                    && (this.getNanos() == theTimestamp.getNanos());
274        }

276        /**
277         * Gets this {@code Timestamp}'s nanosecond value
278         * 
279         * @return The timestamp's nanosecond value, an integer between 0 and
280         *         999,999,999.
281         * @since Android 1.0
282         */
283        public int getNanos() {
284            return nanos;
285        }
286    
287        /**
288         * Returns the time represented by this {@code Timestamp} object, as a long
289         * value containing the number of milliseconds since the Epoch (January 1
290         * 1970, 00:00:00.000 GMT).
291         * 
292         * @return the number of milliseconds that have passed since January 1 1970,
293         *         00:00:00.000 GMT.
294         * @since Android 1.0
295         */
296        @Override
297        public long getTime() {
298            long theTime = super.getTime();
299            theTime = theTime + (nanos / 1000000);
300            return theTime;
301        }
302    
303        /**
304         * Sets the nanosecond value for this {@code Timestamp}.
305         * 
306         * @param n
307         *            number of nanoseconds.
308         * @throws IllegalArgumentException
309         *             if number of nanoseconds smaller than 0 or greater than
310         *             999,999,999.
311         * @since Android 1.0
312         */
313        public void setNanos(int n) throws IllegalArgumentException {
314            if ((n < 0) || (n > 999999999)) {
315                // sql.0=Value out of range
316                throw new IllegalArgumentException(Messages.getString("sql.0")); //$NON-NLS-1$
317            }
318            nanos = n;
319        }
320    
321        /**
322         * Sets the time represented by this {@code Timestamp} object to the
323         * supplied time, defined as the number of milliseconds since the Epoch
324         * (January 1 1970, 00:00:00.000 GMT).
325         * 
326         * @param theTime
327         *            number of milliseconds since the Epoch (January 1 1970,
328         *            00:00:00.000 GMT).
329         * @since Android 1.0
330         */
331        @Override
332        public void setTime(long theTime) {
333            /*
334             * Deal with the nanoseconds value. The supplied time is in milliseconds -
335             * so we must extract the milliseconds value and multiply by 1000000 to
336             * get nanoseconds. Things are more complex if theTime value is
337             * negative, since then the time value is the time before the Epoch but
338             * the nanoseconds value of the Timestamp must be positive - so we must
339             * take the "raw" milliseconds value and subtract it from 1000 to get to
340             * the true nanoseconds value Simultaneously, recalculate the time value
341             * to the exact nearest second and reset the Date time value
342             */
343            int milliseconds = (int) (theTime % 1000);
344            theTime = theTime - milliseconds;
345            if (milliseconds < 0) {
346                theTime = theTime - 1000;
347                milliseconds = 1000 + milliseconds;
348            }
349            super.setTime(theTime);
350            setNanos(milliseconds * 1000000);
351        }

353        /**
354         * Returns the timestamp formatted as a String in the JDBC Timestamp Escape
355         * format, which is {@code "yyyy-mm-dd hh:mm:ss.nnnnnnnnn"}.
356         * 
357         * @return A string representing the instant defined by the {@code
358         *         Timestamp}, in JDBC Timestamp escape format.
359         * @since Android 1.0
360         */
361        @SuppressWarnings("deprecation")
362        @Override
363        public String toString() {
364            /*
365             * Use a DecimalFormat to lay out the nanosecond value as a simple
366             * string of 9 integers, with leading Zeros
367             */
368            DecimalFormat decimalFormat = new DecimalFormat("0"); //$NON-NLS-1$
369            decimalFormat.setMinimumIntegerDigits(9);
370            decimalFormat.setMaximumIntegerDigits(9);
371            String theNanos = decimalFormat.format(nanos);
372            theNanos = stripTrailingZeros(theNanos);
373            
374            String year = format((getYear() + 1900), 4);
375            String month = format((getMonth() + 1), 2);
376            String date = format(getDate(), 2);
377            String hours = format(getHours(), 2);
378            String minutes = format(getMinutes(), 2);
379            String seconds = format(getSeconds(), 2);
380    
381            return year + '-' + month + '-' + date + ' ' + hours + ':' + minutes
382                    + ':' + seconds + '.' + theNanos;
383        }
384    
385        /*
386         * Private method to format the time
387         */
388        private String format(int date, int digits) {
389            StringBuilder dateStringBuffer = new StringBuilder(String.valueOf(date));
390            while (dateStringBuffer.length() < digits) {
391                dateStringBuffer = dateStringBuffer.insert(0,'0');
392            }
393            return dateStringBuffer.toString();
394        }
395        
396        /*
397         * Private method to strip trailing '0' characters from a string. @param
398         * inputString the starting string @return a string with the trailing zeros
399         * stripped - will leave a single 0 at the beginning of the string
400         */
401        private String stripTrailingZeros(String inputString) {
402            String finalString;
403    
404            int i;
405            for (i = inputString.length(); i > 0; i--) {
406                if (inputString.charAt(i - 1) != '0') {
407                    break;
408                }
409                /*
410                 * If the string has a 0 as its first character, return a string
411                 * with a single '0'
412                 */
413                if (i == 1) {
414                    return "0"; //$NON-NLS-1$
415                }
416            }
417    
418            finalString = inputString.substring(0, i);
419            return finalString;
420        }

422        /**
423         * Creates a {@code Timestamp} object with a time value equal to the time
424         * specified by a supplied String holding the time in JDBC timestamp escape
425         * format, which is {@code "yyyy-mm-dd hh:mm:ss.nnnnnnnnn}"
426         * 
427         * @param s
428         *            the {@code String} containing a time in JDBC timestamp escape
429         *            format.
430         * @return A {@code Timestamp} object with time value as defined by the
431         *         supplied {@code String}.
432         * @throws IllegalArgumentException
433         *             if the provided string is {@code null}.
434         * @since Android 1.0
435         */
436        public static Timestamp valueOf(String s) throws IllegalArgumentException {
437            if (s == null) {
438                // sql.3=Argument cannot be null
439                throw new IllegalArgumentException(Messages.getString("sql.3")); //$NON-NLS-1$
440            }
441    
442            SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss"); //$NON-NLS-1$
443            ParsePosition pp = new ParsePosition(0);
444    
445            /*
446             * First parse out the yyyy-MM-dd HH:mm:ss component of the String into
447             * a Date object using the SimpleDateFormat. This should stop after the
448             * seconds value, according to the definition of SimpleDateFormat.parse,
449             * with the ParsePosition indicating the index of the "." which should
450             * precede the nanoseconds value
451             */
452            Date theDate;
453            try {
454                theDate = df.parse(s, pp);
455            } catch (Exception e) {
456                throw new IllegalArgumentException(Messages.getString("sql.2")); //$NON-NLS-1$
457            }
458    
459            if (theDate == null) {
460                throw new IllegalArgumentException(Messages.getString("sql.2")); //$NON-NLS-1$
461            }
462    
463            /*
464             * If we get here, the Date part of the string was OK - now for the
465             * nanoseconds value. Strictly, this requires the remaining part of the
466             * String to look like ".nnnnnnnnn". However, we accept anything with a
467             * '.' followed by 1 to 9 digits - we also accept nothing (no fractions
468             * of a second). Anything else is interpreted as incorrect format which
469             * will generate an IllegalArgumentException
470             */
471            int position = pp.getIndex();
472            int remaining = s.length() - position;
473            int theNanos;
474    
475            if (remaining == 0) {
476                // First, allow for the case where no fraction of a second is given:
477                theNanos = 0;
478            } else {
479                /*
480                 * Case where fraction of a second is specified: Require 1 character
481                 * plus the "." in the remaining part of the string...
482                 */
483                if ((s.length() - position) < ".n".length()) { //$NON-NLS-1$
484                    throw new IllegalArgumentException(Messages.getString("sql.2")); //$NON-NLS-1$
485                }
486    
487                /*
488                 * If we're strict, we should not allow any EXTRA characters after
489                 * the 9 digits
490                 */
491                if ((s.length() - position) > ".nnnnnnnnn".length()) { //$NON-NLS-1$
492                    throw new IllegalArgumentException(Messages.getString("sql.2")); //$NON-NLS-1$
493                }
494    
495                // Require the next character to be a "."
496                if (s.charAt(position) != '.') {
497                    // sql.4=Bad input string format: expected '.' not {0}
498                    throw new NumberFormatException(Messages.getString("sql.4", s.charAt(position))); //$NON-NLS-1$
499                }
500                // Get the length of the number string - need to account for the '.'
501                int nanoLength = s.length() - position - 1;
502    
503                // Get the 9 characters following the "." as an integer
504                String theNanoString = s.substring(position + 1, position + 1
505                        + nanoLength);
506                /*
507                 * We must adjust for the cases where the nanos String was not 9
508                 * characters long by padding out with zeros
509                 */
510                theNanoString = theNanoString + "000000000"; //$NON-NLS-1$
511                theNanoString = theNanoString.substring(0, 9);
512    
513                try {
514                    theNanos = Integer.parseInt(theNanoString);
515                } catch (Exception e) {
516                    // If we get here, the string was not a number
517                    throw new IllegalArgumentException(Messages.getString("sql.2")); //$NON-NLS-1$
518                }
519            }
520    
521            if (theNanos < 0 || theNanos > 999999999) {
522                throw new IllegalArgumentException(Messages.getString("sql.2")); //$NON-NLS-1$
523            }
524    
525            Timestamp theTimestamp = new Timestamp(theDate.getTime());
526            theTimestamp.setNanos(theNanos);
527    
528            return theTimestamp;
529        }
