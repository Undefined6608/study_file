026    /**
027     * An interface for a precompiled SQL Statement.
028     * <p>
029     * An SQL Statement is put into a {@code PreparedStatement} and is precompiled
030     * so that it can be executed efficiently multiple times.
031     * </p>
032     * <p>
033     * Setter methods are supplied in the {@code PreparedStatement} interface for
034     * the setting of {@code IN} parameters for the statement. The setter method
035     * used for each {@code IN} parameter must match the parameter's type.
036     * </p>
037     *  
038     * @since Android 1.0
039     */
040    public interface PreparedStatement extends Statement {
041    
042        /**
043         * Add a set of parameters to the {@code PreparedStatement}'s command batch.
044         * 
045         * @throws SQLException
046         *             if a database error happens.
047         * @since Android 1.0
048         */
049        public void addBatch() throws SQLException;
050    
051        /**
052         * Clear the current parameter values.
053         * <p>
054         * Typically, parameter values are retained for multiple executions of the
055         * {@code Statement}. Setting a parameter value replaces the previous value. This
056         * method clears the values for all parameters, releasing all resources used
057         * by those parameters.
058         * </p>
059         * 
060         * @throws SQLException
061         *             if a database error happens.
062         * @since Android 1.0
063         */
064        public void clearParameters() throws SQLException;
065    
066        /**
067         * Executes the SQL statement in this {@code PreparedStatement}.
068         * <p>
069         * A {@code PreparedStatement} may return multiple results. The execute
070         * method executes the {@code PreparedStatement} and returns a flag 
071         * indicating the kind of result produced by the action. The methods 
072         * {@code getResultSet} or {@code getUpdateCount} are used to retrieve 
073         * the first result, and the second and subsequent results are 
074         * retrieved with {@code getMoreResults}.
075         * </p>
076         * 
077         * @return {@code true} if the result of the execution is a {@code
078         *         ResultSet}, {@code false} if there is no result or if the result
079         *         is an update count.
080         * @throws SQLException
081         *             if a database error happens.
082         * @since Android 1.0
083         */
084        public boolean execute() throws SQLException;
085    
086        /**
087         * Executes the SQL query in the {@code PreparedStatement} and returns the
088         * {@code ResultSet} generated by the query.
089         * 
090         * @return the {@code ResultSet} generated by the query, never {@code null}.
091         * @throws SQLException
092         *             if a database error happens or if the SQL statement does not
093         *             produce a {@code ResultSet}.
094         * @since Android 1.0
095         */
096        public ResultSet executeQuery() throws SQLException;
097    
098        /**
099         * Invokes the SQL command contained within the prepared statement. This
100         * must be {@code INSERT}, {@code UPDATE}, {@code DELETE}, or a command that
101         * returns nothing.
102         * 
103         * @return the number of affected rows for {@code INSERT}, {@code UPDATE} or {@code
104         *         DELETE} statements, {@code 0} for statements that return nothing.
105         * @throws SQLException
106         *             if a database error happens or if the SQL statement returns a
107         *             {@code ResultSet}.
108         * @since Android 1.0
109         */
110        public int executeUpdate() throws SQLException;
111    
112        /**
113         * Returns a {@code ResultSetMetaData} describing the {@code
114         * ResultSet} that would be produced by execution of the {@code PreparedStatement}.
115         * <p>
116         * It is possible to know the metadata for the {@code ResultSet} without
117         * executing the {@code PreparedStatement}, because the {@code
118         * PreparedStatement} is precompiled. As a result the metadata can be
119         * queried ahead of time without actually executing the statement.
120         * </p>
121         * 
122         * @return a {@code ResultSetMetaData} object with the information about the
123         *         columns of the {@code ResultSet}, if the driver can return a
124         *         {@code ResultSetMetaData}. {@code null} otherwise.
125         * @throws SQLException
126         *             if there is a database error.
127         * @since Android 1.0
128         */
129        public ResultSetMetaData getMetaData() throws SQLException;
130    
131        /**
132         * Gets information about the parameters of the {@code PreparedStatement}.
133         * 
134         * @return a {@code ParameterMetaData} object which holds information about
135         *         the number, type, and properties of the parameters of this {@code
136         *         PreparedStatement}.
137         * @throws SQLException
138         *             if a database error happens.
139         * @since Android 1.0
140         */
141        public ParameterMetaData getParameterMetaData() throws SQLException;

145         *

147         *            the parameter number index, where the first parameter has
148         *            index 1.
149         * @param theArray
150         *            a {@code java.sql.Array} giving the new value of the parameter at {@code
151         *            parameterIndex}.

163         * <p>
164         * This is a good method for setting an SQL {@code LONVARCHAR} parameter
165         * where the length of the data is large. Data is read from the {@code
166         * InputStream} until end-of-file is reached or the specified number of
167         * bytes is copied.
168         * </p>
169         * 
170         * @param parameterIndex
171         *            the parameter number index, where the first parameter has
172         *            index 1.
173         * @param theInputStream
174         *            the ASCII {@code InputStream} carrying the data to which the
175         *            parameter at {@code parameterIndex} is set.
176         * @param length
177         *            the number of bytes in the {@code InputStream} to copy to the
178         *            parameter.
179         * @throws SQLException
180         *             if a database error happens.
181         * @since Android 1.0

184                int length) throws SQLException;

455         * Sets the value of a specified parameter to SQL {@code NULL}. This version
456         * of {@code setNull} should be used for <i>User Defined Types</i> (UDTs)

463         * parameter the type name is the type name of the referenced type.
464         * </p>
465         * 
466         * @param paramIndex
467         *            the parameter number index, where the first parameter has
468         *            index 1.
469         * @param sqlType
470         *            the SQL type of the parameter, as defined in {@code
471         *            java.sql.Types}.
472         * @param typeName
473         *            the fully qualified name of a UDT or REF type - ignored if the
474         *            parameter is not a UDT.
475         * @throws SQLException
476         *             if a database error happens.
477         * @see Types
478         * @since Android 1.0
479         */

481                throws SQLException;

483        /**
484         * Sets the value of a specified parameter using a supplied object.
485         * <p>
486         * There is a standard mapping from Java types to SQL types, defined in the
487         * JDBC specification. The passed object is then transformed into the
488         * appropriate SQL type, and then transferred to the database. {@code
489         * setObject} can be used to pass abstract data types unique to the
490         * database, by using a JDBC driver specific Java type. If the object's
491         * class implements the interface {@code SQLData}, the JDBC driver calls
492         * {@code SQLData.writeSQL} to write it to the SQL data stream. If the
493         * object's class implements {@code Ref}, {@code Blob}, {@code Clob},
494         * {@code Struct}, or {@code Array}, the driver passes it to the database as
495         * a value of the corresponding SQL type.
496         * </p>
497         * 
498         * @param parameterIndex
499         *            the parameter number index, where the first parameter has
500         *            index 1.
501         * @param theObject
502         *            the object containing the value to which the parameter at
503         *            {@code parameterIndex} is set.
504         * @throws SQLException
505         *             if a database error happens.
506         * @since Android 1.0
507         */

516         * implements the interface {@code SQLData}), the JDBC driver will call the method
517         * {@code SQLData.writeSQL} to write it to the SQL data stream. If the
518         * object's class implements {@code Ref}, {@code Blob}, {@code Clob},
519         * {@code Struct}, or {@code Array}, the driver will pass it to the database
520         * in the form of the relevant SQL type.
521         * </p>
522         * 
523         * @param parameterIndex
524         *            the parameter index, where the first parameter has index 1.
525         * @param theObject
526         *            the Object containing the value to which the parameter at
527         *            {@code parameterIndex} is set.
528         * @param targetSqlType

530         *            java.sql.Types}.
