018    package java.sql;
019    
020    /**
021     * Interface used for executing static SQL statements to retrieve query results.
022     * The resulting table rows are returned as {@code ResultSet}s. For any given
023     * {@code Statement} object, only one {@code ResultSet} can be opened at one
024     * time. A call to any of the execution methods of {@code Statement} will cause
025     * any previously created {@code ResultSet} object for that {@code Statement} to
026     * be closed implicitly.
027     * <p>
028     * To have multiple {@code ResultSet} objects opened concurrently, multiple
029     * {@code Statement} objects must be created and then executed.
030     * </p>
031     * <p>
032     * To obtain such an executable statement one needs to invoke {@code
033     * Connection#createStatement}.
034     * </p>
035     * 
036     * @see ResultSet
037     * @see Connection#createStatement
038     * 
039     * @since Android 1.0
040     */
041    public interface Statement {
042    
043        /**
044         * Passing this constant to {@link #getMoreResults} implies that all {@code
045         * ResultSet} objects previously kept open should be closed.
046         * 
047         * @since Android 1.0
048         */
049        public static final int CLOSE_ALL_RESULTS = 3;
050    
051        /**
052         * Passing this constant to {@link #getMoreResults} implies that the current
053         * {@code ResultSet} object should be closed.
054         * 
055         * @since Android 1.0
056         */
057        public static final int CLOSE_CURRENT_RESULT = 1;
058    
059        /**
060         * Indicates that an error was encountered during execution of a batch
061         * statement.
062         * 
063         * @since Android 1.0
064         */
065        public static final int EXECUTE_FAILED = -3;
066    
067        /**
068         * Passing this constant to <i>getMoreResults</i> implies that the current
069         * {@code ResultSet} object should not be closed.
070         * 
071         * @since Android 1.0
072         */
073        public static final int KEEP_CURRENT_RESULT = 2;
074    
075        /**
076         * Indicates that generated keys should not be accessible for retrieval.
077         * 
078         * @since Android 1.0
079         */
080        public static final int NO_GENERATED_KEYS = 2;
081    
082        /**
083         * Indicates that generated keys should be accessible for retrieval.
084         * 
085         * @since Android 1.0
086         */
087        public static final int RETURN_GENERATED_KEYS = 1;
088    
089        /**
090         * Indicates that a batch statement was executed with a successful result,
091         * but a count of the number of rows it affected is unavailable.
092         * 
093         * @since Android 1.0
094         */
095        public static final int SUCCESS_NO_INFO = -2;
096    
097        /**
098         * Adds a specified SQL command to the list of commands for this {@code
099         * Statement}.
100         * <p>
101         * The list of commands is executed by invoking the {@code executeBatch}
102         * method.
103         * </p>
104         * 
105         * @param sql
106         *            the SQL command as a String. Typically an {@code INSERT} or
107         *            {@code UPDATE} statement.
108         * @throws SQLException
109         *             if an error occurs accessing the database or the database
110         *             does not support batch updates.
111         * @since Android 1.0
112         */
113        public void addBatch(String sql) throws SQLException;
114    
115        /**
116         * Cancels this statement's execution if both the database and the JDBC
117         * driver support aborting an SQL statement in flight. This method can be
118         * used by one thread to stop a statement that is executed on another
119         * thread.
120         * 
121         * @throws SQLException
122         *             if an error occurs accessing the database.
123         * @since Android 1.0
124         */
125        public void cancel() throws SQLException;
126    
127        /**
128         * Clears the current list of SQL commands for this statement.
129         * 
130         * @throws SQLException
131         *             if an error occurs accessing the database or the database
132         *             does not support batch updates.
133         * @since Android 1.0
134         */
135        public void clearBatch() throws SQLException;
136    
137        /**
138         * Clears all {@code SQLWarnings} from this statement.
139         * 
140         * @throws SQLException
141         *             if an error occurs accessing the database.
142         * @since Android 1.0
143         */
144        public void clearWarnings() throws SQLException;
145    
146        /**
147         * Releases this statement's database and JDBC driver resources.
148         * <p>
149         * Using this method to release these resources as soon as possible is
150         * strongly recommended.
151         * </p>
152         * <p>
153         * One should not rely on the resources being automatically released when
154         * finalized during garbage collection. Doing so can result in unpredictable
155         * behavior for the application.
156         * </p>
157         * 
158         * @throws SQLException
159         *             if an error occurs accessing the database.
160         */
161        public void close() throws SQLException;
162    
163        /**
164         * Executes a supplied SQL statement. This may return multiple {@code
165         * ResultSet}s.
166         * <p>
167         * Use the {@code getResultSet} or {@code getUpdateCount} methods to get the
168         * first result and {@code getMoreResults} to get any subsequent results.
169         * </p>
170         * 
171         * @param sql
172         *            the SQL statement to execute
173         * @return {@code true} if the first result is a {@code ResultSet}, {@code
174         *         false} if the first result is an update count or if there is no
175         *         result.
176         * @throws SQLException
177         *             if an error occurs accessing the database.
178         * @since Android 1.0
179         */
180        public boolean execute(String sql) throws SQLException;
181    
182        /**
183         * Executes a supplied SQL statement. This may return multiple {@code
184         * ResultSet}s. This method allows control of whether auto-generated Keys
185         * should be made available for retrieval, if the SQL statement is an
186         * {@code INSERT} statement.
187         * <p>
188         * Use the {@code getResultSet} or {@code getUpdateCount} methods to get the
189         * first result and {@code getMoreResults} to get any subsequent results.
190         * </p>
191         * 
192         * @param sql
193         *            the SQL statement to execute.
194         * @param autoGeneratedKeys
195         *            a flag indicating whether to make auto generated keys
196         *            available for retrieval. This parameter must be one of {@code
197         *            Statement.NO_GENERATED_KEYS} or {@code
198         *            Statement.RETURN_GENERATED_KEYS}.
199         * @return {@code true} if results exists and the first result is a {@code
200         *         ResultSet}, {@code false} if the first result is an update count
201         *         or if there is no result.
202         * @throws SQLException
203         *             if an error occurs accessing the database.
204         * @since Android 1.0
205         */
206        public boolean execute(String sql, int autoGeneratedKeys)
207                throws SQLException;
208    
209        /**
210         * Executes the supplied SQL statement. This may return multiple {@code
211         * ResultSet}s. This method allows retrieval of auto generated keys
212         * specified by the supplied array of column indexes, if the SQL statement
213         * is an {@code INSERT} statement.
214         * <p>
215         * Use the {@code getResultSet} or {@code getUpdateCount} methods to get the
216         * first result and {@code getMoreResults} to get any subsequent results.
217         * </p>
218         * 
219         * @param sql
220         *            the SQL statement to execute.
221         * @param columnIndexes
222         *            an array of indexes of the columns in the inserted row which
223         *            should be made available for retrieval via the {@code
224         *            getGeneratedKeys} method.
225         * @return {@code true} if the first result is a {@code ResultSet}, {@code
226         *         false} if the first result is an update count or if there is no
227         *         result.
228         * @throws SQLException
229         *             if an error occurs accessing the database.
230         * @since Android 1.0
231         */
232        public boolean execute(String sql, int[] columnIndexes) throws SQLException;
233    
234        /**
235         * Executes the supplied SQL statement. This may return multiple {@code
236         * ResultSet}s. This method allows retrieval of auto generated keys
237         * specified by the supplied array of column indexes, if the SQL statement
238         * is an {@code INSERT} statement.
239         * <p>
240         * Use the {@code getResultSet} or {@code getUpdateCount} methods to get the
241         * first result and {@code getMoreResults} to get any subsequent results.
242         * </p>
243         * 
244         * @param sql
245         *            the SQL statement to execute.
246         * @param columnNames
247         *            an array of column names in the inserted row which should be
248         *            made available for retrieval via the {@code getGeneratedKeys}
249         *            method.
250         * @return {@code true} if the first result is a {@code ResultSet}, {@code
251         *         false} if the first result is an update count or if there is no
252         *         result
253         * @throws SQLException
254         *             if an error occurs accessing the database.
255         * @since Android 1.0
256         */
257        public boolean execute(String sql, String[] columnNames)
258                throws SQLException;
259    
260        /**
261         * Submits a batch of SQL commands to the database. Returns an array of
262         * update counts, if all the commands execute successfully.
263         * <p>
264         * If one of the commands in the batch fails, this method can throw a
265         * {@link BatchUpdateException} and the JDBC driver may or may not process
266         * the remaining commands. The JDBC driver must behave consistently with the
267         * underlying database, following the "all or nothing" principle. If the
268         * driver continues processing, the array of results returned contains the
269         * same number of elements as there are commands in the batch, with a
270         * minimum of one of the elements having the {@code EXECUTE_FAILED} value.
271         * 
272         * @return an array of update counts, with one entry for each command in the
273         *         batch. The elements are ordered according to the order in which
274         *         the commands were added to the batch.
275         *         <p>
276         *         <ol>
277         *         <li>If the value of an element is &ge; 0, the corresponding
278         *         command completed successfully and the value is the <i>update
279         *         count</i> (the number of rows in the database affected by the
280         *         command) for that command.</li>
281         *         <li>If the value is {@code SUCCESS_NO_INFO}, the command
282         *         completed successfully but the number of rows affected is
283         *         unknown.
284         *         <li>
285         *         <li>If the value is {@code EXECUTE_FAILED}, the command failed.
286         *         </ol>
287         * @throws SQLException
288         *             if an error occurs accessing the database.
289         */
290        public int[] executeBatch() throws SQLException;
291    
292        /**
293         * Executes a supplied SQL statement. Returns a single {@code ResultSet}.
294         * 
295         * @param sql
296         *            an SQL statement to execute. Typically a {@code SELECT}
297         *            statement
298         * @return a {@code ResultSet} containing the data produced by the SQL
299         *         statement. Never null.
300         * @throws SQLException
301         *             if an error occurs accessing the database or if the statement
302         *             produces anything other than a single {@code ResultSet}.
303         * @since Android 1.0
304         */
305        public ResultSet executeQuery(String sql) throws SQLException;
306    
307        /**
308         * Executes the supplied SQL statement. The statement may be an {@code
309         * INSERT}, {@code UPDATE} or {@code DELETE} statement or a statement which
310         * returns nothing.
311         * 
312         * @param sql
313         *            an SQL statement to execute - an SQL {@code INSERT}, {@code
314         *            UPDATE}, {@code DELETE} or a statement which returns nothing
315         * @return the count of updated rows, or 0 for a statement that returns
316         *         nothing.
317         * @throws SQLException
318         *             if an error occurs accessing the database or if the statement
319         *             produces a {@code ResultSet}.
320         * @since Android 1.0
321         */
322        public int executeUpdate(String sql) throws SQLException;
323    
324        /**
325         * Executes the supplied SQL statement. This method allows control of
326         * whether auto-generated Keys should be made available for retrieval.
327         * 
328         * @param sql
329         *            an SQL statement to execute - an SQL {@code INSERT}, {@code
330         *            UPDATE}, {@code DELETE} or a statement which does not return
331         *            anything.
332         * @param autoGeneratedKeys
333         *            a flag that indicates whether to allow retrieval of auto
334         *            generated keys. Parameter must be one of {@code
335         *            Statement.RETURN_GENERATED_KEYS} or {@code
336         *            Statement.NO_GENERATED_KEYS}
337         * @return the number of updated rows, or 0 if the statement returns
338         *         nothing.
339         * @throws SQLException
340         *             if an error occurs accessing the database or if the statement
341         *             produces a {@code ResultSet}.
342         * @since Android 1.0
343         */
344        public int executeUpdate(String sql, int autoGeneratedKeys)
345                throws SQLException;
346    
347        /**
348         * Executes the supplied SQL statement. This method allows retrieval of auto
349         * generated keys specified by the supplied array of column indexes.
350         * 
351         * @param sql
352         *            an SQL statement to execute - an SQL {@code INSERT}, {@code
353         *            UPDATE}, {@code DELETE} or a statement which returns nothing
354         * @param columnIndexes
355         *            an array of indexes of the columns in the inserted row which
356         *            should be made available for retrieval via the {@code
357         *            getGeneratedKeys} method.
358         * @return the count of updated rows, or 0 for a statement that returns
359         *         nothing.
360         * @throws SQLException
361         *             if an error occurs accessing the database or if the statement
362         *             produces a {@code ResultSet}.
363         * @since Android 1.0
364         */
365        public int executeUpdate(String sql, int[] columnIndexes)
366                throws SQLException;
367    
368        /**
369         * Executes the supplied SQL statement. This method allows retrieval of auto
370         * generated keys specified by the supplied array of column names.
371         * 
372         * @param sql
373         *            an SQL statement to execute - an SQL {@code INSERT}, {@code
374         *            UPDATE}, {@code DELETE} or a statement which returns nothing
375         * @param columnNames
376         *            an array of column names in the inserted row which should be
377         *            made available for retrieval via the {@code getGeneratedKeys}
378         *            method.
379         * @return the count of updated rows, or 0 for a statement that returns
380         *         nothing.
381         * @throws SQLException
382         *             if an error occurs accessing the database or if the statement
383         *             produces a {@code ResultSet}.
384         * @since Android 1.0
385         */
386        public int executeUpdate(String sql, String[] columnNames)
387                throws SQLException;
388    
389        /**
390         * Gets the {@code Connection} object which created this statement.
391         * 
392         * @return the {@code Connection} through which this statement is
393         *         transmitted to the database.
394         * @throws SQLException
395         *             if an error occurs accessing the database.
396         * @since Android 1.0
397         */
398        public Connection getConnection() throws SQLException;
399    
400        /**
401         * Gets the default direction for fetching rows for {@code ResultSet}s
402         * generated from this statement.
403         * 
404         * @return the default fetch direction, one of:
405         *         <ul>
406         *         <li>ResultSet.FETCH_FORWARD</li> <li>ResultSet.FETCH_REVERSE</li>
407         *         <li>ResultSet.FETCH_UNKNOWN</li>
408         *         </ul>
409         * @throws SQLException
410         *             if an error occurs accessing the database.
411         * @since Android 1.0
412         */
413        public int getFetchDirection() throws SQLException;
414    
415        /**
416         * Gets the default number of rows for a fetch for the {@code ResultSet}
417         * objects returned from this statement.
418         * 
419         * @return the default fetch size for {@code ResultSet}s produced by this
420         *         statement.
421         * @throws SQLException
422         *             if an error occurs accessing the database.
423         * @since Android 1.0
424         */
425        public int getFetchSize() throws SQLException;
426    
427        /**
428         * Returns auto generated keys created by executing this statement.
429         * 
430         * @return a {@code ResultSet} containing the auto generated keys - empty if
431         *         no keys are generated by this statement.
432         * @throws SQLException
433         *             if an error occurs accessing the database.
434         * @since Android 1.0
435         */
436        public ResultSet getGeneratedKeys() throws SQLException;
437    
438        /**
439         * Gets the maximum number of bytes which can be returned as values from
440         * character and binary type columns in a {@code ResultSet} derived from this
441         * statement. This limit applies to {@code BINARY}, {@code VARBINARY},
442         * {@code LONGVARBINARY}, {@code CHAR}, {@code VARCHAR}, and {@code
443         * LONGVARCHAR} types. Any data exceeding the maximum size is abandoned
444         * without announcement.
445         * 
446         * @return the current size limit, where {@code 0} means that there is no
447         *         limit.
448         * @throws SQLException
449         *             if an error occurs accessing the database.
450         * @since Android 1.0
451         */
452        public int getMaxFieldSize() throws SQLException;
453    
454        /**
455         * Gets the maximum number of rows that a {@code ResultSet} can contain when
456         * produced from this statement. If the limit is exceeded, the excess rows
457         * are discarded silently.
458         * 
459         * @return the current row limit, where {@code 0} means that there is no
460         *         limit.
461         * @throws SQLException
462         *             if an error occurs accessing the database.
463         * @since Android 1.0
464         */
465        public int getMaxRows() throws SQLException;
466    
467        /**
468         * Moves to this statement's next result. Returns {@code true} if it is a
469         * {@code ResultSet}. Any current {@code ResultSet} objects previously
470         * obtained with {@code getResultSet()} are closed implicitly.
471         * 
472         * @return {@code true} if the next result is a {@code ResultSet}, {@code
473         *         false} if the next result is not a {@code ResultSet} or if there
474         *         are no more results. Note that if there is no more data, this
475         *         method will return {@code false} and {@code getUpdateCount} will
476         *         return -1.
477         * @throws SQLException
478         *             if an error occurs accessing the database.
479         * @since Android 1.0
480         */
481        public boolean getMoreResults() throws SQLException;
482    
483        /**
484         * Moves to this statement's next result. Returns {@code true} if the next
485         * result is a {@code ResultSet}. Any current {@code ResultSet} objects
486         * previously obtained with {@code getResultSet()} are handled as indicated
487         * by a supplied Flag parameter.
488         * 
489         * @param current
490         *            a flag indicating what to do with existing {@code ResultSet}s.
491         *            This parameter must be one of {@code
492         *            Statement.CLOSE_ALL_RESULTS}, {@code
493         *            Statement.CLOSE_CURRENT_RESULT} or {@code
494         *            Statement.KEEP_CURRENT_RESULT}.
495         * @return {@code true} if the next result exists and is a {@code ResultSet}
496         *         , {@code false} if the next result is not a {@code ResultSet} or
497         *         if there are no more results. Note that if there is no more data,
498         *         this method will return {@code false} and {@code getUpdateCount}
499         *         will return -1.
500         * @throws SQLException
501         *             if an error occurs accessing the database.
502         * @since Android 1.0
503         */
504        public boolean getMoreResults(int current) throws SQLException;
505    
506        /**
507         * Gets the timeout value for the statement's execution time. The JDBC
508         * driver will wait up to this value for the execution to complete - after
509         * the limit is exceeded an SQL {@code Exception} is thrown.
510         * 
511         * @return the current query timeout value, where {@code 0} indicates that
512         *         there is no current timeout.
513         * @throws SQLException
514         *             if an error occurs accessing the database.
515         * @since Android 1.0
516         */
517        public int getQueryTimeout() throws SQLException;
518    
519        /**
520         * Gets the current result. Should only be called once per result.
521         * 
522         * @return the {@code ResultSet} for the current result. {@code null} if the
523         *         result is an update count or if there are no more results.
524         * @throws SQLException
525         *             if an error occurs accessing the database.
526         * @since Android 1.0
527         */
528        public ResultSet getResultSet() throws SQLException;
529    
530        /**
531         * Gets the concurrency setting for {@code ResultSet} objects generated by
532         * this statement.
533         * 
534         * @return {@code ResultSet.CONCUR_READ_ONLY} or {@code
535         *         ResultSet.CONCUR_UPDATABLE}.
536         * @throws SQLException
537         *             if an error occurs accessing the database.
538         * @since Android 1.0
539         */
540        public int getResultSetConcurrency() throws SQLException;
541    
542        /**
543         * Gets the cursor hold setting for {@code ResultSet} objects generated by
544         * this statement.
545         * 
546         * @return {@code ResultSet.HOLD_CURSORS_OVER_COMMIT} or {@code
547         *         ResultSet.CLOSE_CURSORS_AT_COMMIT}
548         * @throws SQLException
549         *             if there is an error while accessing the database.
550         * @since Android 1.0
551         */
552        public int getResultSetHoldability() throws SQLException;
553    
554        /**
555         * Gets the {@code ResultSet} type setting for {@code ResultSet}s derived
556         * from this statement.
557         * 
558         * @return {@code ResultSet.TYPE_FORWARD_ONLY} for a {@code ResultSet} where
559         *         the cursor can only move forwards, {@code
560         *         ResultSet.TYPE_SCROLL_INSENSITIVE} for a {@code ResultSet} which
561         *         is scrollable but is not sensitive to changes made by others,
562         *         {@code ResultSet.TYPE_SCROLL_SENSITIVE} for a {@code ResultSet}
563         *         which is scrollable but is sensitive to changes made by others.
564         * @throws SQLException
565         *             if there is an error accessing the database.
566         * @since Android 1.0
567         */
568        public int getResultSetType() throws SQLException;
569    
570        /**
571         * Gets an update count for the current result if it is not a {@code
572         * ResultSet}.
573         * 
574         * @return the current result as an update count. {@code -1} if the current
575         *         result is a {@code ResultSet} or if there are no more results.
576         * @throws SQLException
577         *             if an error occurs accessing the database.
578         * @since Android 1.0
579         */
580        public int getUpdateCount() throws SQLException;
581    
582        /**
583         * Retrieves the first {@code SQLWarning} reported by calls on this
584         * statement. If there are multiple warnings, subsequent warnings are
585         * chained to the first one. The chain of warnings is cleared each time the
586         * statement is executed.
587         * <p>
588         * Warnings associated with reads from the {@code ResultSet} returned from
589         * executing the statement will be attached to the {@code ResultSet}, not the
590         * statement object.
591         * </p>
592         * 
593         * @return an SQLWarning, null if there are no warnings
594         * @throws SQLException
595         *             if an error occurs accessing the database.
596         * @since Android 1.0
597         */
598        public SQLWarning getWarnings() throws SQLException;
599    
600        /**
601         * Sets the SQL cursor name. This name is used by subsequent statement
602         * execute methods.
603         * <p>
604         * Cursor names must be unique within one Connection.
605         * </p>
606         * <p>
607         * With the cursor name set, it can then be used in SQL positioned
608         * update or delete statements to determine the current row in a {@code
609         * ResultSet} generated from this statement. The positioned update or delete
610         * must be done with a different statement than this one.
611         * </p>
612         * 
613         * @param name
614         *            the Cursor name as a string,
615         * @throws SQLException
616         *             if an error occurs accessing the database.
617         * @since Android 1.0
618         */
619        public void setCursorName(String name) throws SQLException;
620    
621        /**
622         * Sets Escape Processing mode.
623         * <p>
624         * If Escape Processing is on, the JDBC driver will do escape substitution
625         * on an SQL statement before sending it for execution. This does not apply
626         * to {@link PreparedStatement}s since they are processed when created,
627         * before this method can be called.
628         * </p>
629         * 
630         * @param enable
631         *            {@code true} to set escape processing mode <i>on</i>, {@code
632         *            false} to turn it <i>off</i>.
633         * @throws SQLException
634         *             if an error occurs accessing the database.
635         * @since Android 1.0
636         */
637        public void setEscapeProcessing(boolean enable) throws SQLException;
638    
639        /**
640         * Sets the fetch direction - a hint to the JDBC driver about the direction
641         * of processing of rows in {@code ResultSet}s created by this statement.
642         * The default fetch direction is {@code FETCH_FORWARD}.
643         * 
644         * @param direction
645         *            which fetch direction to use. This parameter should be one of
646         *            <ul>
647         *            <li>{@code ResultSet.FETCH_UNKNOWN}</li>
648         *            <li>{@code ResultSet.FETCH_FORWARD}</li>
649         *            <li>{@code ResultSet.FETCH_REVERSE}</li>
650         *            </ul>
651         * @throws SQLException
652         *             if there is an error while accessing the database or if the
653         *             fetch direction is unrecognized.
654         */
655        public void setFetchDirection(int direction) throws SQLException;
656    
657        /**
658         * Sets the fetch size. This is a hint to the JDBC driver about how many
659         * rows should be fetched from the database when more are required by
660         * application processing.
661         * 
662         * @param rows
663         *            the number of rows that should be fetched. {@code 0} tells the driver
664         *            to ignore the hint. Should be less than {@code getMaxRows} for
665         *            this statement. Should not be negative.
666         * @throws SQLException
667         *             if an error occurs accessing the database, or if the rows
668         *             parameter is out of range.
669         * @since Android 1.0
670         */
671        public void setFetchSize(int rows) throws SQLException;

673        /**
