018    package java.sql;
019    
020    /**
021     * An interface for the custom mapping of an SQL <i>User Defined Type</i> (UDT)
022     * to a Java class. The Java class object is added to the connection's type map
023     * paired with the SQL name of the corresponding UDT.
024     * <p>
025     * Usually within an implementation of {@code SQLData}, there is a corresponding
026     * field for every attribute of an SQL type, but only one field, if the type is
027     * SQL {@code DISTINCT}. When the UDT is returned within a {@code ResultSet}, it
028     * is accessed with the {@link ResultSet#getObject} method and is returned as an
029     * object which is an instance of the class defined by the {@code SQLData}
030     * mapping. The application can use this object just like any other Java object
031     * and can store changes back into the database using the
032     * {@link PreparedStatement#setObject} method which performs the reverse mapping
033     * into the SQL {@code UDT}.
034     * </p>
035     * Normally the implementation of a custom mapping is generated by
036     * a tool requiring the name of the SQL {@code UDT}, the name
037     * of the class which it is going to be mapped to, and the field names to which
038     * the UDT attributes are mapped. The tool can then implement the {@code
039     * SQLData}, {@code readSQL}, and {@code writeSQL} methods. {@code readSQL} reads
040     * attributes from an {@code SQLInput} object, and {@code writeSQL} writes them.
041     * This is done via {@code SQLInput} and {@code SQLOutput} method calls
042     * respectively.
043     * <p>
044     * Ordinarily an application would not call {@code SQLData} methods directly.
045     * Similarly {@code SQLInput} and {@code SQLOutput} methods are not usually
046     * called directly.
047     * </p>
048     * 
049     * @since Android 1.0
050     */
051    public interface SQLData {
052    
053        /**
054         * Gets the SQL name of the <i>User Defined Type</i> (UDT) that this object
055         * represents. This method, usually invoked by the JDBC driver, retrieves
056         * the name of the UDT instance associated with this {@code SQLData} object.
057         * 
058         * @return a string with UDT type name for this object mapping, passed to
059         *         {@code readSQL} when the object was created.
060         * @throws SQLException
061         *             if a database error occurs.
062         * @since Android 1.0
063         */
064        public String getSQLTypeName() throws SQLException;
065    
066        /**
067         * Reads data from the database into this object. This method follows these
068         * steps:
069         * <p>
070         * <ul>
071         * <li>Utilize the passed input stream to read the attributes or entries of
072         * the SQL type</li>
073         * <li>This is carried out by reading each entry from the input stream,
074         * ordered as they are in the SQL definition.</li>
075         * <li>Assign the data to the appropriate fields or elements. This is done
076         * by calling the relevant reader method for the type involved (e.g. {@code
077         * SQLInput.readString}, {@code SQLInputreadBigDecimal}). If the type is
078         * distinct, then read its only data entry. For structured types, read every
079         * entry.</li>
080         * </ul>
081         * </p>
082         * <p>
083         * The supplied input stream is typically initialized by the calling JDBC
084         * driver with the type map before {@code readSQL} is called.
085         * </p>
086         * 
087         * @param stream
088         *            the {@code SQLInput} stream from which the type map data is
089         *            read for the custom mapping.
090         * @param typeName
091         *            the SQL type name for the type which is being mapped.
092         * @throws SQLException
093         *             if a database error occurs.
094         * @see SQLInput
095         * @since Android 1.0
096         */
097        public void readSQL(SQLInput stream, String typeName) throws SQLException;
098    
099        /**
100         * Writes the object to a supplied {@code SQLOutput} data stream, writing it
101         * out as an SQL value to the data source.
102         * <p>
103         * This method follows the following steps:
104         * <ul>
105         * <li>Write each attribute of the SQL type to the output stream.</li>
106         * <li>Write each item by calling a method on the output stream, in the
107         * order they appear in the SQL definition of the type. Use the appropriate
108         * {@code SQLOutput} methods (e.g. {@code writeInt}, {@code writeString}).
109         * Write a single data element for a distinct type. For a structured type,
110         * write a value for each attribute of the the SQL type.</li>
111         * </ul>
112         * </p>
113         * 
114         * @param stream
115         *            the {@code SQLOutput} stream to use to write out the data for
116         *            the custom mapping.
117         * @throws SQLException
118         *             if a database error occurs.
119         * @see SQLOutput
120         * @since Android 1.0
121         */
122        public void writeSQL(SQLOutput stream) throws SQLException;
123    }
