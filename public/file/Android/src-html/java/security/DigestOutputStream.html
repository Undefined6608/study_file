023    package java.security;
024    
025    import java.io.FilterOutputStream;
026    import java.io.IOException;
027    import java.io.OutputStream;
028    
029    /**
030     * {@code DigestOutputStream} is a {@code FilterOutputStream} which maintains an
031     * associated message digest.
032     */
033    public class DigestOutputStream extends FilterOutputStream {
034    
035        /**
036         * The message digest for this stream.
037         * 
038         * @since Android 1.0
039         */
040        protected MessageDigest digest;
041    
042        // Indicates whether digest functionality is on or off
043        private boolean isOn = true;
044    
045        /**
046         * Constructs a new instance of this {@code DigestOutputStream}, using the
047         * given {@code stream} and the {@code digest}.
048         * 
049         * @param stream
050         *            the output stream.
051         * @param digest
052         *            the message digest.
053         * @since Android 1.0
054         */
055        public DigestOutputStream(OutputStream stream, MessageDigest digest) {
056            super(stream);
057            this.digest = digest;
058        }
059    
060        /**
061         * Returns the message digest for this stream.
062         * 
063         * @return the message digest for this stream.
064         * @since Android 1.0
065         */
066        public MessageDigest getMessageDigest() {
067            return digest;
068        }
069    
070        /**
071         * Sets the message digest which this stream will use.
072         * 
073         * @param digest
074         *            the message digest which this stream will use.
075         * @since Android 1.0
076         */
077        public void setMessageDigest(MessageDigest digest) {
078            this.digest = digest;
079        }
080    
081        /**
082         * Writes the specified {@code int} to the stream. Updates the digest if
083         * this function is {@link #on(boolean)}.
084         * 
085         * @param b
086         *            the byte to be written.
087         * @throws IOException
088         *             if writing to the stream causes a {@code IOException}
089         * @since Android 1.0
090         */
091        public void write(int b) throws IOException {
092            // update digest only if digest functionality is on
093            if (isOn) {
094                digest.update((byte)b);
095            }
096            // write the byte
097            out.write(b);
098        }
099    
100        /**
101         * Writes {@code len} bytes into the stream, starting from the specified
102         * offset. Updates the digest if this function is {@link #on(boolean)}.
103         * 
104         * @param b
105         *            the buffer to write to.
106         * @param off
107         *            the index of the first byte in {@code b} to write.
108         * @param len
109         *            the number of bytes in {@code b} to write.
110         * @throws IOException
111         *             if writing to the stream causes an {@code IOException}.
112         * @since Android 1.0
113         */
114        public void write(byte[] b, int off, int len) throws IOException {
115            // update digest only if digest functionality is on
116            if (isOn) {
117                digest.update(b, off, len);
118            }
119            // write len bytes
120            out.write(b, off, len);
121        }
122    
123        /**
124         * Enables or disables the digest function (default is on).
125         * 
126         * @param on
127         *            {@code true} if the digest should be computed, {@code false}
128         *            otherwise.
129         * @since Android 1.0
130         */
131        public void on(boolean on) {
132            isOn = on;
133        }
134    
135        /**
136         * Returns a string containing a concise, human-readable description of this
137         * {@code DigestOutputStream} including the digest.
138         * 
139         * @return a printable representation for this {@code DigestOutputStream}.
140         * @since Android 1.0
141         */
142        public String toString() {
143            return super.toString() + ", " + digest.toString() + //$NON-NLS-1$
144                (isOn ? ", is on" : ", is off"); //$NON-NLS-1$ //$NON-NLS-2$
145        }
146    }
