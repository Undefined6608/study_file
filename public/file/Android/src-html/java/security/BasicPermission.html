023    package java.security;
024    
025    import java.io.IOException;
026    import java.io.Serializable;
027    
028    import org.apache.harmony.security.internal.nls.Messages;
029    
030    /**
031     * {@code BasicPermission} is the common base class of all permissions which
032     * have a name but no action lists. A {@code BasicPermission} is granted or it
033     * is not.
034     * <p>
035     * Names of a BasicPermission follow the dot separated, hierarchical property
036     * naming convention. Asterisk '*' can be used as wildcards. Either by itself,
037     * matching anything, or at the end of the name, immediately preceded by a '.'.
038     * For example:
039     * 
040     * <pre>
041     * com.google.android.*  grants all permissions under the com.google.android permission hierarchy
042     * *                     grants all permissions
043     * </pre>
044     * </p><p>
045     * While this class ignores the action list in the
046     * {@link #BasicPermission(String, String)} constructor, subclasses may
047     * implement actions on top of this class.
048     * </p>
049     */
050    public abstract class BasicPermission extends Permission implements
051        Serializable {
052    
053        private static final long serialVersionUID = 6279438298436773498L;
054    
055        /**
056         * Constructs a new instance of {@code BasicPermission} with the specified
057         * name.
058         * 
059         * @param name
060         *            the name of the permission.
061         * @throws NullPointerException if {@code name} is {@code null}.
062         * @throws IllegalArgumentException if {@code name.length() == 0}.
063         * @since Android 1.0
064         */
065        public BasicPermission(String name) {
066            super(name);
067            checkName(name);
068        }
069    
070        /**
071         * Constructs a new instance of {@code BasicPermission} with the specified
072         * name. The {@code action} parameter is ignored.
073         * 
074         * @param name
075         *            the name of the permission.
076         * @param action
077         *            is ignored.
078         * @throws NullPointerException
079         *             if {@code name} is {@code null}.
080         * @throws IllegalArgumentException
081         *             if {@code name.length() == 0}.
082         * @since Android 1.0
083         */
084        public BasicPermission(String name, String action) {
085            super(name);
086            checkName(name);
087        }
088    
089        /**
090         * Checks name parameter
091         */ 
092        private final void checkName(String name) {
093            if (name == null) {
094                throw new NullPointerException(Messages.getString("security.28")); //$NON-NLS-1$
095            }
096            if (name.length() == 0) {
097                throw new IllegalArgumentException(Messages.getString("security.29")); //$NON-NLS-1$
098            }
099        }
100    
101        /**
102         * Compares the specified object with this {@code BasicPermission} for
103         * equality. Returns {@code true} if the specified object has the same class
104         * and the two {@code Permissions}s have the same name.
105         * <p>
106         * The {@link #implies(Permission)} method should be used for making access
107         * control checks.
108         * </p>
109         * 
110         * @param obj
111         *            object to be compared for equality with this {@code
112         *            BasicPermission}.
113         * @return {@code true} if the specified object is equal to this {@code
114         *         BasicPermission}, otherwise {@code false}.
115         * @since Android 1.0
116         */
117        public boolean equals(Object obj) {
118            if (obj == this) {
119                return true;
120            }
121    
122            if (obj != null && obj.getClass() == this.getClass()) {
123                return this.getName().equals(((Permission)obj).getName());
124            }
125            return false;
126        }
127    
128        /**
129         * Returns the hash code value for this {@code BasicPermission}. Returns the
130         * same hash code for {@code BasicPermission}s that are equal to each other
131         * as required by the general contract of {@link Object#hashCode}.
132         * 
133         * @return the hash code value for this {@code BasicPermission}.
134         * @see Object#equals(Object)
135         * @see BasicPermission#equals(Object)
136         * @since Android 1.0
137         */
138        public int hashCode() {
139            return getName().hashCode();
140        }
141    
142        /**
143         * Returns the actions associated with this permission. Since {@code
144         * BasicPermission} instances have no actions, an empty string is returned.
145         * 
146         * @return an empty string.
147         * @since Android 1.0
148         */
149        public String getActions() {
150            return ""; //$NON-NLS-1$
151        }
152    
153        /**
154         * Indicates whether the specified permission is implied by this permission.
155         * 
156         * @param permission
157         *            the permission to check against this permission.
158         * @return {@code true} if the specified permission is implied by this
159         *         permission, {@code false} otherwise.
160         * @since Android 1.0
161         */
162        public boolean implies(Permission permission) {
163            if (permission != null && permission.getClass() == this.getClass()) {
164                return nameImplies(getName(), permission.getName());
165            }
166            return false;
167        }
168    
169        /**
170         * Checks if {@code thisName} implies {@code thatName},
171         * accordingly to hierarchical property naming convention.
172         * It is assumed that names cannot be {@code null} or empty.
173         */
174        static boolean nameImplies(String thisName, String thatName) {
175            if (thisName == thatName) {
176                return true;
177            }
178            int end = thisName.length();
179            if (end > thatName.length()) {
180                return false;
181            }
182            if (thisName.charAt(--end) == '*'
183                && (end == 0 || thisName.charAt(end - 1) == '.')) {
184                //wildcard found
185                end--;
186            } else if (end != (thatName.length()-1)) {
187                //names are not equal
188                return false;
189            }
190            for (int i = end; i >= 0; i--) {
191                if (thisName.charAt(i) != thatName.charAt(i)) {
192                    return false;
193                }
194            }
195            return true;
196        }
197    
198        /**
199         * Returns an empty {@link PermissionCollection} for holding permissions.
200         * <p>
201         * For {@code PermissionCollection} (and subclasses which do not override
202         * this method), the collection which is returned does <em>not</em> invoke
203         * the {@link #implies(Permission)} method of the permissions which are
204         * stored in it when checking if the collection implies a permission.
205         * Instead, it assumes that if the type of the permission is correct, and
206         * the name of the permission is correct, there is a match.
207         * </p>
208         * 
209         * @return an empty {@link PermissionCollection} for holding permissions.
210         * @see BasicPermissionCollection
211         * @since Android 1.0
212         */
213        public PermissionCollection newPermissionCollection() {
214            return new BasicPermissionCollection();
215        }
216    
217        /**
218         * Checks name after default deserialization.
219         */
220        private void readObject(java.io.ObjectInputStream in) throws IOException,
221            ClassNotFoundException {
222            in.defaultReadObject();
223            checkName(this.getName());
224        }
225    }
