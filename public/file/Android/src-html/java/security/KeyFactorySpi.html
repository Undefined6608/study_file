023    package java.security;
024    
025    import java.security.spec.InvalidKeySpecException;
026    import java.security.spec.KeySpec;
027    
028    /**
029     * {@code KeyFactorySpi} is the Service Provider Interface (SPI) definition for
030     * {@link KeyFactory}.
031     * 
032     * @see KeyFactory
033     * @since Android 1.0
034     */
035    public abstract class KeyFactorySpi {
036        
037        /**
038         * Generates a instance of {@code PublicKey} from the given key
039         * specification.
040         * 
041         * @param keySpec
042         *            the specification of the public key.
043         * @return the public key.
044         * @throws InvalidKeySpecException
045         *             if the specified {@code keySpec} is invalid.
046         * @since Android 1.0
047         */
048        protected abstract PublicKey engineGeneratePublic(KeySpec keySpec) 
049                                        throws InvalidKeySpecException;
050        
051        /**
052         * Generates a instance of {@code PrivateKey} from the given key
053         * specification.
054         * 
055         * @param keySpec
056         *            the specification of the private key.
057         * @return the private key.
058         * @throws InvalidKeySpecException
059         *             if the specified {@code keySpec} is invalid.
060         * @since Android 1.0
061         */
062        protected abstract PrivateKey engineGeneratePrivate(KeySpec keySpec)
063                                        throws InvalidKeySpecException;
064        
065        /**
066         * Returns the key specification for the specified key.
067         * 
068         * @param key
069         *            the key from which the specification is requested.
070         * @param keySpec
071         *            the type of the requested {@code KeySpec}.
072         * @return the key specification for the specified key.
073         * @throws InvalidKeySpecException
074         *             if the key can not be processed, or the requested requested
075         *             {@code KeySpec} is inappropriate for the given key.
076         * @since Android 1.0
077         */
078        protected abstract <T extends KeySpec> T engineGetKeySpec(Key key, Class<T> keySpec)
079                                        throws InvalidKeySpecException;
080        //FIXME 1.5 signature: protected abstract <T extends KeySpec> T engineGetKeySpec(Key key, Class<T> keySpec) throws InvalidKeySpecException
081        
082        /**
083         * Translates the given key into a key from this key factory.
084         * 
085         * @param key
086         *            the key to translate.
087         * @return the translated key.
088         * @throws InvalidKeyException
089         *             if the specified key can not be translated by this key
090         *             factory.
091         * @since Android 1.0
092         */
093        protected abstract Key engineTranslateKey(Key key) throws InvalidKeyException;
094    
095    }
