027    package java.security;
028    
029    import java.io.IOException;
030    import java.io.InputStream;
031    import java.io.OutputStream;
032    
033    /**
034     * {@code Certificate} represents an identity certificate, such as X.509 or PGP.
035     * Note: A {@code Certificate} instances does not make any statement about the
036     * validity of itself. It's in the responsibility of the application to verify
037     * the validity of its certificates.
038     * 
039     * @deprecated Replaced by behavior in {@link javax.security.cert}
040     * @see java.security.cert.Certificate
041     * @since Android 1.0
042     */
043    @Deprecated
044    public interface Certificate {
045    
046        /**
047         * Decodes a certificate from the given {@code InputStream}. The format of
048         * the data to encode must be that identified by {@link #getFormat()} and
049         * encoded by {@link #encode(OutputStream)}.
050         * 
051         * @param stream
052         *            the {@code InputStream} to read from.
053         * @throws KeyException
054         *             if certificate information is incomplete or incorrect.
055         * @throws IOException
056         *             if an exception is thrown by accessing the provided stream.
057         * @see #encode(OutputStream)
058         * @see #getFormat()
059         * @since Android 1.0
060         */
061        public void decode(InputStream stream) throws KeyException, IOException;
062    
063        /**
064         * Encodes this certificate to an output stream. The
065         * {@link #decode(InputStream)} method must be able to decode the format
066         * written by this method.
067         * 
068         * @param stream
069         *            the {@code OutputStream} to encode this certificate to.
070         * @throws KeyException
071         *             if certificate information is incomplete or incorrect.
072         * @throws IOException
073         *             if an exception is thrown by accessing the provided stream.
074         * @see #decode(InputStream)
075         * @since Android 1.0
076         */
077        public void encode(OutputStream stream) throws KeyException, IOException;
078    
079        /**
080         * Returns a string identifying the format of this certificate.
081         * 
082         * @return a string identifying the format of this certificate.
083         * @since Android 1.0
084         */
085        public String getFormat();
086    
087        /**
088         * Returns the guarantor of this certificate. That guarantor guarantees,
089         * that the public key of this certificate is from the principal returned by
090         * {@link #getPrincipal()}.
091         * 
092         * @return the guarantor of this certificate.
093         * @see #getPrincipal()
094         * @since Android 1.0
095         */
096        public Principal getGuarantor();
097    
098        /**
099         * Returns the principal of this certificate. The principal is guaranteed by
100         * the guarantor returned by {@link #getGuarantor()}.
101         * 
102         * @return the principal of this certificate.
103         * @see #getGuarantor()
104         * @since Android 1.0
105         */
106        public Principal getPrincipal();
107    
108        /**
109         * Returns the public key of this certificate. The public key is guaranteed
110         * by the guarantor to belong to the principal.
111         * 
112         * @return the public key of this certificate.
113         * @see #getGuarantor()
114         * @see Certificate#getPrincipal()
115         * @since Android 1.0
116         */
117        public PublicKey getPublicKey();
118    
119        /**
120         * Returns a string containing a concise, human-readable description of the
121         * this {@code Certificate}.
122         * 
123         * @param detailed
124         *            whether or not this method should return detailed information.
125         * @return a string representation of this certificate.
126         * @since Android 1.0
127         */
128        public String toString(boolean detailed);
129    }
