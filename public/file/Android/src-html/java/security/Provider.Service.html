162        /**
163         * Clears all properties used to look up services implemented by this
164         * {@code Provider}.
165         * <p>
166         * If a {@code SecurityManager} is installed, code calling this method needs
167         * the {@code SecurityPermission} {@code clearProviderProperties.NAME}
168         * (where NAME is the provider name) to be granted, otherwise a {@code
169         * SecurityException} will be thrown.
170         * </p>
171         * 
172         * @throws SecurityException
173         *             if a {@code SecurityManager} is installed and the caller does
174         *             not have permission to invoke this method.
175         * @since Android 1.0
176         */
177        public synchronized void clear() {
178            SecurityManager sm = System.getSecurityManager();
179            if (sm != null) {
180                sm.checkSecurityAccess("clearProviderProperties." + name); //$NON-NLS-1$
181            }
182            super.clear();
183            if (serviceTable != null) {
184                serviceTable.clear();
185            }
186            if (propertyServiceTable != null) {
187                propertyServiceTable.clear();
188            }
189            if (aliasTable != null) {
190                aliasTable.clear();
191            }
192            if (propertyAliasTable != null) {
193                propertyAliasTable.clear();
194            }
195            // BEGIN android-changed
196            changedProperties = null;
197            // END android-changed
198            putProviderInfo();
199            if (providerNumber != -1) {
200                // if registered then refresh Services
201                Services.setNeedRefresh();
202            }
203            servicesChanged();
204        }

213         * Copies all from the provided map to this {@code Provider}.
214         * <p>
215         * If a {@code SecurityManager} is installed, code calling this method needs
216         * the {@code SecurityPermission} {@code putProviderProperty.NAME} (where
217         * NAME is the provider name) to be granted, otherwise a {@code
218         * SecurityException} will be thrown.
219         * </p>
220         * 
221         * @param t
222         *            the mappings to copy to this provider.
223         * @throws SecurityException
224         *             if a {@code SecurityManager} is installed and the caller does
225         *             not have permission to invoke this method.
226         * @since Android 1.0
227         */
228        public synchronized void putAll(Map<?,?> t) {
229    
230            // Implementation note:
231            // checkSecurityAccess method call is NOT specified
232            // Do it as in put(Object key, Object value).
233    
234            SecurityManager sm = System.getSecurityManager();
235            if (sm != null) {
236                sm.checkSecurityAccess("putProviderProperty." + name); //$NON-NLS-1$
237            }
238            myPutAll(t);
239        }

241        private void myPutAll(Map<?,?> t) {
242            if (changedProperties == null) {
243                changedProperties = new Properties();
244            }
245            Iterator<? extends Map.Entry<?, ?>> it = t.entrySet().iterator();
246            Object key;
247            Object value;
248            while (it.hasNext()) {
249                Map.Entry<?, ?> entry = it.next();
250                key = entry.getKey();
251                if (key instanceof String && ((String) key).startsWith("Provider.")) { //$NON-NLS-1$
252                    // Provider service type is reserved
253                    continue;
254                }
255                value = entry.getValue();
256                super.put(key, value);
257                if (changedProperties.remove(key) == null) {
258                    removeFromPropertyServiceTable(key);
259                }
260                changedProperties.put(key, value);
261            }
262            if (providerNumber != -1) {
263                // if registered then refresh Services
264                Services.setNeedRefresh();
265            }
266        }

280        /**
281         * Maps the specified {@code key} property name to the specified {@code
282         * value}.
283         * <p>
284         * If a {@code SecurityManager} is installed, code calling this method needs
285         * the {@code SecurityPermission} {@code putProviderProperty.NAME} (where
286         * NAME is the provider name) to be granted, otherwise a {@code
287         * SecurityException} will be thrown.
288         * </p>
289         * 
290         * @param key
291         *            the name of the property.
292         * @param value
293         *            the value of the property.
294         * @return the value that was previously mapped to the specified {@code key}
295         *         ,or {@code null} if it did not have one.
296         * @throws SecurityException
297         *             if a {@code SecurityManager} is installed and the caller does
298         *             not have permission to invoke this method.
299         * @since Android 1.0
300         */
301        public synchronized Object put(Object key, Object value) {
302            SecurityManager sm = System.getSecurityManager();
303            if (sm != null) {
304                sm.checkSecurityAccess("putProviderProperty." + name); //$NON-NLS-1$
305            }
306            if (key instanceof String && ((String) key).startsWith("Provider.")) { //$NON-NLS-1$
307                // Provider service type is reserved
308                return null;
309            }
310            if (providerNumber != -1) {
311                // if registered then refresh Services
312                Services.setNeedRefresh();
313            }
314            if (changedProperties != null && changedProperties.remove(key) == null) {
315                removeFromPropertyServiceTable(key);
316            }
317            if (changedProperties == null) {
318                changedProperties = new Properties();
319            }
320            changedProperties.put(key, value);
321            return super.put(key, value);
322        }

324        /**
325         * Removes the specified {@code key} and its associated value from this
326         * {@code Provider}.
327         * <p>
328         * If a {@code SecurityManager} is installed, code calling this method needs
329         * the {@code SecurityPermission} {@code removeProviderProperty.NAME} (where
330         * NAME is the provider name) to be granted, otherwise a {@code
331         * SecurityException} will be thrown.
332         * </p>
333         * 
334         * @param key
335         *            the name of the property
336         * @return the value that was mapped to the specified {@code key} ,or
337         *         {@code null} if no mapping was present
338         * @throws SecurityException
339         *             if a {@code SecurityManager} is installed and the caller does
340         *             not have the permission to invoke this method.
341         * @since Android 1.0
342         */
343        public synchronized Object remove(Object key) {
344            SecurityManager sm = System.getSecurityManager();
345            if (sm != null) {
346                sm.checkSecurityAccess("removeProviderProperty." + name); //$NON-NLS-1$
347            }
348            if (key instanceof String && ((String) key).startsWith("Provider.")) { //$NON-NLS-1$
349                // Provider service type is reserved
350                return null;
351            }
352            if (providerNumber != -1) {
353                // if registered then refresh Services
354                Services.setNeedRefresh();
355            }
356            if (changedProperties != null && changedProperties.remove(key) == null) {
357                removeFromPropertyServiceTable(key);
358                // BEGIN android-added
359                if (changedProperties.size() == 0) {
360                    changedProperties = null;
361                }
362                // END android-added
363            }
364            return super.remove(key);
365        }

367        /**
368         * returns true if the provider implements the specified algorithm. Caller
369         * must specify the cryptographic service and specify constraints via the
370         * attribute name the attribute value
371         * 
372         * @param serv
373         *            Crypto service.
374         * @param alg
375         *            Algorithm or type.
376         * @param attribute
377         *            The attribute name or {@code null}.
378         * @param val
379         *            The attribute value.
380         * @return
381         */
382        boolean implementsAlg(String serv, String alg, String attribute, String val) {
383            String servAlg = serv + "." + alg; //$NON-NLS-1$
384            String prop = getPropertyIgnoreCase(servAlg);
385            if (prop == null) {
386                alg = getPropertyIgnoreCase("Alg.Alias." + servAlg); //$NON-NLS-1$
387                if (alg != null) {
388                    servAlg = serv + "." + alg; //$NON-NLS-1$
389                    prop = getPropertyIgnoreCase(serv + "." + alg); //$NON-NLS-1$
390                }
391            }
392            if (prop != null) {
393                if (attribute == null) {
394                    return true;
395                } else {
396                    return checkAttribute(serv + "." + alg, attribute, val); //$NON-NLS-1$
397                }
398            }
399            return false;
400        }
401    
402        // returns true if the implementation meets the constraint expressed by the
403        // specified attribute name/value pair.
404        private boolean checkAttribute(String servAlg, String attribute, String val) {
405            
406            String attributeValue = getPropertyIgnoreCase(servAlg + ' ' + attribute);
407            if (attributeValue != null) {
408                if (attribute.equalsIgnoreCase("KeySize")) { //$NON-NLS-1$
409                    // BEGIN android-changed
410                    if (Integer.parseInt(attributeValue) >= Integer.parseInt(val)) {
411                        return true;
412                    }
413                    // END android-changed
414                } else { // other attributes
415                    if (attributeValue.equalsIgnoreCase(val)) {
416                        return true;
417                    }
418                }
419            }
420            return false;
421        }

443        /**
444         * Get the service of the specified type
445         *  
446         */
447        synchronized Provider.Service getService(String type) {
448            updatePropertyServiceTable();
449            if (lastServicesByType != null && type.equals(lastType)) {
450                return lastServicesByType;
451            }
452            Provider.Service service;
453            for (Iterator<Service> it = getServices().iterator(); it.hasNext();) {
454                service = it.next();
455                if (type.equals(service.type)) {
456                    lastType = type;
457                    lastServicesByType = service;
458                    return service;
459                }
460            }
461            return null;
462        }

464        /**
465         * Returns the service with the specified {@code type} implementing the
466         * specified {@code algorithm}, or {@code null} if no such implementation
467         * exists.
468         * <p>
469         * If two services match the requested type and algorithm, the one added
470         * with the {@link #putService(Service)} is returned (as opposed to the one
471         * added via {@link #put(Object, Object)}.
472         * 
473         * @param type
474         *            the type of the service (for example {@code KeyPairGenerator})
475         * @param algorithm
476         *            the algorithm name (case insensitive)
477         * @return the requested service, or {@code null} if no such implementation
478         *         exists
479         * @since Android 1.0
480         */
481        public synchronized Provider.Service getService(String type,
482                String algorithm) {
483            if (type == null || algorithm == null) {
484                throw new NullPointerException();
485            }
486    
487            if (type.equals(lastServiceName)
488                    && algorithm.equalsIgnoreCase(lastAlgorithm)) {
489                return returnedService;
490            }
491    
492            String alg = algorithm.toUpperCase();
493            Object o = null;
494            if (serviceTable != null) {
495                o = serviceTable.get(type, alg);
496            }
497            if (o == null && aliasTable != null) {
498                o = aliasTable.get(type, alg);
499            }
500            if (o == null) {
501                updatePropertyServiceTable();
502            }
503            if (o == null && propertyServiceTable != null) {
504                o = propertyServiceTable.get(type, alg);
505            }
506            if (o == null && propertyAliasTable != null) {
507                o = propertyAliasTable.get(type, alg);
508            }
509    
510            if (o != null) {
511                lastServiceName = type;
512                lastAlgorithm = algorithm;
513                returnedService = (Provider.Service) o;
514                return returnedService;
515            }
516            return null;
517        }

522         * 
523         * @return an unmodifiable {@code Set} of all services registered by this
524         *         provider
525         * @since Android 1.0
526         */
527        public synchronized Set<Provider.Service> getServices() {
528            updatePropertyServiceTable();
529            if (lastServicesSet != null) {
530                return lastServicesSet;
531            }
532            if (serviceTable != null) {
533                lastServicesSet = new HashSet<Service>(serviceTable.values());
534            } else {
535                lastServicesSet = new HashSet<Service>();
536            }

539            }
540            lastServicesSet = Collections.unmodifiableSet(lastServicesSet);
541            return lastServicesSet;

544        /**
545         * Adds a {@code Service} to this {@code Provider}. If a service with the
546         * same name was registered via this method, it is replace.
547         * <p>
548         * If a {@code SecurityManager} is installed, code calling this method needs
549         * the {@code SecurityPermission} {@code putProviderProperty.NAME} (where
550         * NAME is the provider name) to be granted, otherwise a {@code
551         * SecurityException} will be thrown.
552         * 
553         * @param s
554         *            the {@code Service} to register
555         * @throws SecurityException
556         *             if a {@code SecurityManager} is installed and the caller does
557         *             not have permission to invoke this method
558         * @since Android 1.0
559         */
560        protected synchronized void putService(Provider.Service s) {
561            if (s == null) {
562                throw new NullPointerException();
563            }
564            SecurityManager sm = System.getSecurityManager();
565            if (sm != null) {
566                sm.checkSecurityAccess("putProviderProperty." + name); //$NON-NLS-1$
567            }
568            if ("Provider".equals(s.getType())) { // Provider service type cannot be //$NON-NLS-1$
569                                                  // added
570                return;
571            }
572            servicesChanged();
573            if (serviceTable == null) {
574                serviceTable = new TwoKeyHashMap<String, String, Service>(128);
575            }
576            serviceTable.put(s.type, s.algorithm.toUpperCase(), s);
577            if (s.aliases != null) {
578                if (aliasTable == null) {
579                    aliasTable = new TwoKeyHashMap<String, String, Service>(256);
580                }
581                for (Iterator<String> it = s.getAliases(); it.hasNext();) {
582                    aliasTable.put(s.type, (it.next()).toUpperCase(), s);
583                }
584            }
585            serviceInfoToProperties(s);
586        }

588        /**
589         * Removes a previously registered {@code Service} from this {@code
590         * Provider}.
591         * <p>
592         * If a {@code SecurityManager} is installed, code calling this method needs
593         * the {@code SecurityPermission} {@code removeProviderProperty.NAME} (where
594         * NAME is the provider name) to be granted, otherwise a {@code
595         * SecurityException} will be thrown.
596         * 
597         * @param s
598         *            the {@code Service} to remove
599         * @throws SecurityException
600         *             if a {@code SecurityManager} is installed and the caller does
601         *             not have permission to invoke this method
602         * @throws NullPointerException
603         *             if {@code s} is {@code null}
604         * @since Android 1.0
605         */
606        protected synchronized void removeService(Provider.Service s) {
607            if (s == null) {
608                throw new NullPointerException();
609            }
610            SecurityManager sm = System.getSecurityManager();
611            if (sm != null) {
612                sm.checkSecurityAccess("removeProviderProperty." + name); //$NON-NLS-1$
613            }
614            servicesChanged();
615            if (serviceTable != null) {
616                serviceTable.remove(s.type, s.algorithm.toUpperCase());
617            }
618            if (aliasTable != null && s.aliases != null) {
619                for (Iterator<String> it = s.getAliases(); it.hasNext();) {
620                    aliasTable.remove(s.type, (it.next()).toUpperCase());
621                }
622            }
623            serviceInfoFromProperties(s);
624        }

626        // Add Service information to the provider's properties.
627        private void serviceInfoToProperties(Provider.Service s) {
628            super.put(s.type + "." + s.algorithm, s.className); //$NON-NLS-1$
629            if (s.aliases != null) {
630                for (Iterator<String> i = s.aliases.iterator(); i.hasNext();) {
631                    super.put("Alg.Alias." + s.type + "." + i.next(), s.algorithm); //$NON-NLS-1$ //$NON-NLS-2$
632                }
633            }
634            if (s.attributes != null) {
635                for (Iterator<Map.Entry<String, String>> i = s.attributes.entrySet().iterator(); i.hasNext();) {
636                    Map.Entry<String, String> entry = i.next();
637                    super.put(s.type + "." + s.algorithm + " " + entry.getKey(), //$NON-NLS-1$ //$NON-NLS-2$
638                            entry.getValue());
639                }
640            }
641            if (providerNumber != -1) {
642                // if registered then refresh Services
643                Services.setNeedRefresh();
644            }
645        }

647        // Remove Service information from the provider's properties.
648        private void serviceInfoFromProperties(Provider.Service s) {
649            super.remove(s.type + "." + s.algorithm); //$NON-NLS-1$
650            if (s.aliases != null) {
651                for (Iterator<String> i = s.aliases.iterator(); i.hasNext();) {
652                    super.remove("Alg.Alias." + s.type + "." + i.next()); //$NON-NLS-1$ //$NON-NLS-2$
653                }
654            }
655            if (s.attributes != null) {
656                for (Iterator<Map.Entry<String, String>> i = s.attributes.entrySet().iterator(); i.hasNext();) {
657                    Map.Entry<String, String> entry = i.next();
658                    super.remove(s.type + "." + s.algorithm + " " + entry.getKey()); //$NON-NLS-1$ //$NON-NLS-2$
659                }
660            }
661            if (providerNumber != -1) {
662                // if registered then refresh Services
663                Services.setNeedRefresh();
664            }
665        }

667        // Remove property information from provider Services
668        private void removeFromPropertyServiceTable(Object key) {
669            if (key == null || !(key instanceof String)) {
670                return;
671            }
672            String k = (String) key;
673            if (k.startsWith("Provider.")) { // Provider service type is reserved //$NON-NLS-1$
674                return;
675            }
676            Provider.Service s;
677            String serviceName;
678            String algorithm = null;
679            String attribute = null;
680            int i;
681            if (k.startsWith("Alg.Alias.")) { // Alg.Alias.<crypto_service>.<aliasName>=<stanbdardName> //$NON-NLS-1$
682                String aliasName;
683                String service_alias = k.substring(10);
684                i = service_alias.indexOf("."); //$NON-NLS-1$
685                serviceName = service_alias.substring(0, i);
686                aliasName = service_alias.substring(i + 1);
687                if (propertyAliasTable != null) {
688                    propertyAliasTable.remove(serviceName, aliasName.toUpperCase());
689                }
690                if (propertyServiceTable != null) {
691                    for (Iterator<Service> it = propertyServiceTable.values().iterator(); it
692                            .hasNext();) {
693                        s = it.next();
694                        if (s.aliases.contains(aliasName)) {
695                            s.aliases.remove(aliasName);
696                            return;
697                        }
698                    }
699                }
700                return;
701            }
702            int j = k.indexOf("."); //$NON-NLS-1$
703            if (j == -1) { // unknown format
704                return;
705            }
706    
707            i = k.indexOf(" "); //$NON-NLS-1$
708            if (i == -1) { // <crypto_service>.<algorithm_or_type>=<className>
709                serviceName = k.substring(0, j);
710                algorithm = k.substring(j + 1);
711                if (propertyServiceTable != null) {
712                    Provider.Service ser = propertyServiceTable.remove(serviceName, algorithm.toUpperCase());
713                    if (ser != null && propertyAliasTable != null
714                            && ser.aliases != null) {
715                        for (Iterator<String> it = ser.aliases.iterator(); it.hasNext();) {
716                            propertyAliasTable.remove(serviceName, (it
717                                    .next()).toUpperCase());
718                        }
719                    }
720                }
721            } else { // <crypto_service>.<algorithm_or_type>
722                     // <attribute_name>=<attrValue>
723                attribute = k.substring(i + 1);
724                serviceName = k.substring(0, j);
725                algorithm = k.substring(j + 1, i);
726                if (propertyServiceTable != null) {
727                    Object o = propertyServiceTable.get(serviceName, algorithm
728                            .toUpperCase());
729                    if (o != null) {
730                        s = (Provider.Service) o;
731                        s.attributes.remove(attribute);
732                    }
733                }
734            }

737        // Update provider Services if the properties was changed
738        private void updatePropertyServiceTable() {
739            Object _key;
740            Object _value;
741            Provider.Service s;
742            String serviceName;
743            String algorithm;
744            if (changedProperties == null || changedProperties.isEmpty()) {
745                return;
746            }
747            for (Iterator<Map.Entry<Object, Object>> it = changedProperties.entrySet().iterator(); it
748                    .hasNext();) {
749                Map.Entry<Object, Object> entry = it.next();
750                _key = entry.getKey();
751                _value = entry.getValue();
752                if (_key == null || _value == null || !(_key instanceof String)
753                        || !(_value instanceof String)) {
754                    continue;
755                }
756                String key = (String) _key;
757                String value = (String) _value;
758                if (key.startsWith("Provider")) { // Provider service type is reserved //$NON-NLS-1$
759                    continue;
760                }
761                int i;
762                if (key.startsWith("Alg.Alias.")) { // Alg.Alias.<crypto_service>.<aliasName>=<stanbdardName> //$NON-NLS-1$
763                    String aliasName;
764                    String service_alias = key.substring(10);
765                    i = service_alias.indexOf("."); //$NON-NLS-1$
766                    serviceName = service_alias.substring(0, i);
767                    aliasName = service_alias.substring(i + 1);
768                    algorithm = value;
769                    String algUp = algorithm.toUpperCase();
770                    Object o = null;
771                    if (propertyServiceTable == null) {
772                        propertyServiceTable = new TwoKeyHashMap<String, String, Service>(128);
773                    } else {
774                        o = propertyServiceTable.get(serviceName, algUp);
775                    }
776                    if (o != null) {
777                        s = (Provider.Service) o;
778                        // BEGIN android-changed
779                        s.addAlias(aliasName);
780                        // END android-changed
781                        if (propertyAliasTable == null) {
782                            propertyAliasTable = new TwoKeyHashMap<String, String, Service>(256);
783                        }
784                        propertyAliasTable.put(serviceName,
785                                aliasName.toUpperCase(), s);
786                    } else {
787                        String className = (String) changedProperties
788                                .get(serviceName + "." + algorithm); //$NON-NLS-1$
789                        if (className != null) {
790                            List<String> l = new ArrayList<String>();
791                            l.add(aliasName);
792                            s = new Provider.Service(this, serviceName, algorithm,
793                                    className, l, new HashMap<String, String>());
794                            propertyServiceTable.put(serviceName, algUp, s);
795                            if (propertyAliasTable == null) {
796                                propertyAliasTable = new TwoKeyHashMap<String, String, Service>(256);
797                            }
798                            propertyAliasTable.put(serviceName, aliasName
799                                    .toUpperCase(), s);
800                        }
801                    }
802                    continue;
803                }
804                int j = key.indexOf("."); //$NON-NLS-1$
805                if (j == -1) { // unknown format
806                    continue;
807                }
808                i = key.indexOf(" "); //$NON-NLS-1$
809                if (i == -1) { // <crypto_service>.<algorithm_or_type>=<className>
810                    serviceName = key.substring(0, j);
811                    algorithm = key.substring(j + 1);
812                    String alg = algorithm.toUpperCase();
813                    Object o = null;
814                    if (propertyServiceTable != null) {
815                        o = propertyServiceTable.get(serviceName, alg);
816                    }
817                    if (o != null) {
818                        s = (Provider.Service) o;
819                        s.className = value;
820                    } else {
821                        // BEGIN android-changed
822                        s = new Provider.Service(this, serviceName, algorithm,
823                                value, Collections.<String>emptyList(),
824                                Collections.<String,String>emptyMap());
825                        // END android-changed
826                        if (propertyServiceTable == null) {
827                            propertyServiceTable = new TwoKeyHashMap<String, String, Service>(128);
828                        }
829                        propertyServiceTable.put(serviceName, alg, s);
830    
831                    }
832                } else { // <crypto_service>.<algorithm_or_type>
833                         // <attribute_name>=<attrValue>
834                    serviceName = key.substring(0, j);
835                    algorithm = key.substring(j + 1, i);
836                    String attribute = key.substring(i + 1);
837                    String alg = algorithm.toUpperCase();
838                    Object o = null;
839                    if (propertyServiceTable != null) {
840                        o = propertyServiceTable.get(serviceName, alg);
841                    }
842                    if (o != null) {
843                        s = (Provider.Service) o;
844                        // BEGIN android-changed
845                        s.putAttribute(attribute, value);
846                        // END android-changed
847                    } else {
848                        String className = (String) changedProperties
849                                .get(serviceName + "." + algorithm); //$NON-NLS-1$
850                        if (className != null) {
851                            Map<String, String> m = new HashMap<String, String>();
852                            m.put(attribute, value);
853                            s = new Provider.Service(this, serviceName, algorithm,
854                                    className, new ArrayList<String>(), m);
855                            if (propertyServiceTable == null) {
856                                propertyServiceTable = new TwoKeyHashMap<String, String, Service>(128);
857                            }
858                            propertyServiceTable.put(serviceName, alg, s);
859                        }
860                    }
861                }
862            }
863            servicesChanged();
864            // BEGIN android-changed
865            changedProperties = null;
866            // END android-changed
867        }

886        // properties. Key is not case-sensitive.
887        // 
888        // @param prop
889        // @return the property value with the specified key value.
890        private String getPropertyIgnoreCase(String key) {
891            String res = getProperty(key);
892            if (res != null) {
893                return res;
894            }
895            for (Enumeration<?> e = propertyNames(); e.hasMoreElements();) {
896                String pname = (String) e.nextElement();
897                if (key.equalsIgnoreCase(pname)) {
898                    return getProperty(pname);
899                }
900            }
901            return null;
