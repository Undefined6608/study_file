023    package java.security;
024    
025    import java.nio.ByteBuffer;
026    
027    import org.apache.harmony.security.fortress.Engine;
028    import org.apache.harmony.security.internal.nls.Messages;
029    
030    /**
031     * {@code MessageDigest} is an engine class which is capable of generating one
032     * way hash values for arbitrary input, utilizing the algorithm it was
033     * initialized with.
034     * 
035     * @see MessageDigestSpi
036     * @since Android 1.0
037     */
038    public abstract class MessageDigest extends MessageDigestSpi {
039        
040        // The service name
041        private static final String SERVICE = "MessageDigest"; //$NON-NLS-1$
042    
043        // Used to access common engine functionality
044        private static Engine engine = new Engine(SERVICE);
045    
046        // The provider
047        private Provider provider;
048    
049        // The algorithm.
050        private String algorithm;
051    
052        /**
053         * Constructs a new instance of {@code MessageDigest} with the name of
054         * the algorithm to use.
055         * 
056         * @param algorithm
057         *            the name of algorithm to use
058         * @since Android 1.0
059         */
060        protected MessageDigest(String algorithm) {
061            this.algorithm = algorithm;
062        }
063    
064        /**
065         * Returns a new instance of {@code MessageDigest} that utilizes the
066         * specified algorithm.
067         * 
068         * @param algorithm
069         *            the name of the algorithm to use
070         * @return a new instance of {@code MessageDigest} that utilizes the
071         *         specified algorithm
072         * @throws NoSuchAlgorithmException
073         *             if the specified algorithm is not available
074         * @throws NullPointerException
075         *             if {@code algorithm} is {@code null}
076         * @since Android 1.0
077         */
078        public static MessageDigest getInstance(String algorithm)
079                throws NoSuchAlgorithmException {
080            if (algorithm == null) {
081                throw new NullPointerException(Messages.getString("security.01")); //$NON-NLS-1$
082            }
083            MessageDigest result;
084            synchronized (engine) {
085                engine.getInstance(algorithm, null);
086                if (engine.spi instanceof MessageDigest) {
087                    result = (MessageDigest) engine.spi;
088                    result.algorithm = algorithm;
089                    result.provider = engine.provider;
090                    return result;
091                } else {
092                    result = new MessageDigestImpl((MessageDigestSpi) engine.spi,
093                            engine.provider, algorithm);
094                    return result;
095                }
096            }
097        }
098    
099        /**
100         * Returns a new instance of {@code MessageDigest} that utilizes the
101         * specified algorithm from the specified provider.
102         * 
103         * @param algorithm
104         *            the name of the algorithm to use
105         * @param provider
106         *            the name of the provider
107         * @return a new instance of {@code MessageDigest} that utilizes the
108         *         specified algorithm from the specified provider
109         * @throws NoSuchAlgorithmException
110         *             if the specified algorithm is not available
111         * @throws NoSuchProviderException
112         *             if the specified provider is not available
113         * @throws NullPointerException
114         *             if {@code algorithm} is {@code null}
115         * @since Android 1.0
116         */
117        public static MessageDigest getInstance(String algorithm, String provider)
118                throws NoSuchAlgorithmException, NoSuchProviderException {
119            if ((provider == null) || (provider.length() == 0)) {
120                throw new IllegalArgumentException(Messages.getString("security.02")); //$NON-NLS-1$
121            }
122            Provider p = Security.getProvider(provider);
123            if (p == null) {
124                throw new NoSuchProviderException(Messages.getString("security.03", provider)); //$NON-NLS-1$
125            }
126            return getInstance(algorithm, p);
127        }
128    
129        /**
130         * Returns a new instance of {@code MessageDigest} that utilizes the
131         * specified algorithm from the specified provider.
132         * 
133         * @param algorithm
134         *            the name of the algorithm to use
135         * @param provider
136         *            the provider
137         * @return a new instance of {@code MessageDigest} that utilizes the
138         *         specified algorithm from the specified provider
139         * @throws NoSuchAlgorithmException
140         *             if the specified algorithm is not available
141         * @throws NullPointerException
142         *             if {@code algorithm} is {@code null}
143         * @since Android 1.0
144         */
145        public static MessageDigest getInstance(String algorithm, Provider provider)
146                throws NoSuchAlgorithmException {
147            if (provider == null) {
148                throw new IllegalArgumentException(Messages.getString("security.04")); //$NON-NLS-1$
149            }
150            if (algorithm == null) {
151                throw new NullPointerException(Messages.getString("security.01")); //$NON-NLS-1$
152            }
153            MessageDigest result;
154            synchronized (engine) {
155                engine.getInstance(algorithm, provider, null);
156                if (engine.spi instanceof MessageDigest) {
157                    result = (MessageDigest) engine.spi;
158                    result.algorithm = algorithm;
159                    result.provider = provider;
160                    return result;
161                } else {
162                    result = new MessageDigestImpl((MessageDigestSpi) engine.spi,
163                            provider, algorithm);
164                    return result;
165                }
166            }
167        }
168    
169        /**
170         * Puts this {@code MessageDigest} back in an initial state, such that it is
171         * ready to compute a one way hash value.
172         * 
173         * @since Android 1.0
174         */
175        public void reset() {
176            engineReset();
177        }
178    
179        /**
180         * Updates this {@code MessageDigest} using the given {@code byte}.
181         * 
182         * @param arg0
183         *            the {@code byte} to update this {@code MessageDigest} with
184         * @see #reset()
185         * @since Android 1.0
186         */
187        public void update(byte arg0) {
188            engineUpdate(arg0);
189        }
190    
191        /**
192         * Updates this {@code MessageDigest} using the given {@code byte[]}.
193         * 
194         * @param input
195         *            the {@code byte} array
196         * @param offset
197         *            the index of the first byte in {@code input} to update from
198         * @param len
199         *            the number of bytes in {@code input} to update from
200         * @throws IllegalArgumentException
201         *             if {@code offset} or {@code len} are not valid in respect to
202         *             {@code input}
203         * @since Android 1.0
204         */
205        public void update(byte[] input, int offset, int len) {
206            if (input == null ||
207                    // offset < 0 || len < 0 ||
208                    // checks for negative values are commented out intentionally
209                    // see HARMONY-1120 for details
210                    (long) offset + (long) len > input.length) {
211                throw new IllegalArgumentException(Messages
212                        .getString("security.05")); //$NON-NLS-1$
213            }
214            engineUpdate(input, offset, len);
215        }
216    
217        /**
218         * Updates this {@code MessageDigest} using the given {@code byte[]}.
219         * 
220         * @param input
221         *            the {@code byte} array
222         * @throws NullPointerException
223         *             if {@code input} is {@code null}
224         * @since Android 1.0
225         */
226        public void update(byte[] input) {
227            if (input == null) {
228                throw new NullPointerException(Messages.getString("security.06")); //$NON-NLS-1$
229            }
230            engineUpdate(input, 0, input.length);
231        }
232    
233        /**
234         * Computes and returns the final hash value for this {@link MessageDigest}.
235         * After the digest is computed the receiver is reset.
236         * 
237         * @return the computed one way hash value
238         * @see #reset
239         * @since Android 1.0
240         */
241        public byte[] digest() {
242            return engineDigest();
243        }
244    
245        /**
246         * Computes and stores the final hash value for this {@link MessageDigest}.
247         * After the digest is computed the receiver is reset.
248         * 
249         * @param buf
250         *            the buffer to store the result
251         * @param offset
252         *            the index of the first byte in {@code buf} to store
253         * @param len
254         *            the number of bytes allocated for the digest
255         * @return the number of bytes written to {@code buf}
256         * @throws DigestException
257         *             if an error occures
258         * @throws IllegalArgumentException
259         *             if {@code offset} or {@code len} are not valid in respect to
260         *             {@code buf}
261         * @see #reset()
262         * @since Android 1.0
263         */
264        public int digest(byte[] buf, int offset, int len) throws DigestException {
265            if (buf == null ||
266                    // offset < 0 || len < 0 ||
267                    // checks for negative values are commented out intentionally
268                    // see HARMONY-1148 for details
269                    (long) offset + (long) len > buf.length) {
270                throw new IllegalArgumentException(Messages
271                        .getString("security.05")); //$NON-NLS-1$
272            }
273            return engineDigest(buf, offset, len);
274        }
275    
276        /**
277         * Performs the final update and then computes and returns the final hash
278         * value for this {@link MessageDigest}. After the digest is computed the
279         * receiver is reset.
280         * 
281         * @param input
282         *            the {@code byte} array
283         * @return the computed one way hash value
284         * @see #reset()
285         * @since Android 1.0
286         */
287        public byte[] digest(byte[] input) {
288            update(input);
289            return digest();
290        }
291    
292        /**
293         * Returns a string containing a concise, human-readable description of this
294         * {@code MessageDigest} including the name of its algorithm.
295         * 
296         * @return a printable representation for this {@code MessageDigest}
297         * @since Android 1.0
298         */
299        public String toString() {
300            return "MESSAGE DIGEST " + algorithm; //$NON-NLS-1$
301        }
302    
303        /**
304         * Indicates whether to digest are equal by performing a simply
305         * byte-per-byte compare of the two digests.
306         * 
307         * @param digesta
308         *            the first digest to be compared
309         * @param digestb
310         *            the second digest to be compared
311         * @return {@code true} if the two hashes are equal, {@code false} otherwise
312         * @since Android 1.0
313         */
314        public static boolean isEqual(byte[] digesta, byte[] digestb) {
315            if (digesta.length != digestb.length) {
316                return false;
317            }
318            for (int i = 0; i < digesta.length; i++) {
319                if (digesta[i] != digestb[i]) {
320                    return false;
321                }
322            }
323            return true;
324        }
325    
326        /**
327         * Returns the name of the algorithm of this {@code MessageDigest}.
328         * 
329         * @return the name of the algorithm of this {@code MessageDigest}
330         * @since Android 1.0
331         */
332        public final String getAlgorithm() {
333            return algorithm;
334        }
335    
336        /**
337         * Returns the provider associated with this {@code MessageDigest}.
338         * 
339         * @return the provider associated with this {@code MessageDigest}
340         * @since Android 1.0
341         */
342        public final Provider getProvider() {
343            return provider;
344        }
345    
346        /**
347         * Returns the engine digest length in bytes. If the implementation does not
348         * implement this function or is not an instance of {@code Cloneable},
349         * {@code 0} is returned.
350         * 
351         * @return the digest length in bytes, or {@code 0}
352         * @since Android 1.0
353         */
354        public final int getDigestLength() {
355            int l = engineGetDigestLength();
356            if (l != 0) {
357                return l;
358            }
359            if (!(this instanceof Cloneable)) {
360                return 0;
361            }
362            try {
363                MessageDigest md = (MessageDigest) clone();
364                return md.digest().length;
365            } catch (CloneNotSupportedException e) {
366                return 0;
367            }
368        }
369    
370        public Object clone() throws CloneNotSupportedException {
371            if (this instanceof Cloneable) {
372                return super.clone();
373            } else {
374                throw new CloneNotSupportedException();
375            }
376        }
377    
378        /**
379         * Updates this {@code MessageDigest} using the given {@code input}.
380         * 
381         * @param input
382         *            the {@code ByteBuffer}
383         * @since Android 1.0
384         */
385        public final void update(ByteBuffer input) {
386            engineUpdate(input);
387        }
388    
389        /**
390         * 
391         * The internal MessageDigest implementation
392         * 
393         */
394        private static class MessageDigestImpl extends MessageDigest {
395            
396            // MessageDigestSpi implementation
397            private MessageDigestSpi spiImpl;
398    
399            // MessageDigestImpl ctor
400            private MessageDigestImpl(MessageDigestSpi messageDigestSpi,
401                    Provider provider, String algorithm) {
402                super(algorithm);
403                super.provider = provider;
404                spiImpl = messageDigestSpi;
405            }
406    
407            // engineReset() implementation
408            protected void engineReset() {
409                spiImpl.engineReset();
410            }
411    
412            // engineDigest() implementation
413            protected byte[] engineDigest() {
414                return spiImpl.engineDigest();
415            }
416    
417            // engineGetDigestLength() implementation
418            protected int engineGetDigestLength() {
419                return spiImpl.engineGetDigestLength();
420            }
421    
422            // engineUpdate() implementation
423            protected void engineUpdate(byte arg0) {
424                spiImpl.engineUpdate(arg0);
425            }
426    
427            // engineUpdate() implementation
428            protected void engineUpdate(byte[] arg0, int arg1, int arg2) {
429                spiImpl.engineUpdate(arg0, arg1, arg2);
430            }
431    
432            // Returns a clone if the spiImpl is cloneable
433            public Object clone() throws CloneNotSupportedException {
434                if (spiImpl instanceof Cloneable) {
435                    MessageDigestSpi spi = (MessageDigestSpi) spiImpl.clone();
436                    return new MessageDigestImpl(spi, getProvider(), getAlgorithm());
437                } else {
438                    throw new CloneNotSupportedException();
439                }
440            }
441        }
