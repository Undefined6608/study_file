023    package java.security;
024    
025    import java.io.IOException;
026    import java.io.Serializable;
027    
028    /**
029     * {@code GuardedObject} controls access to an object, by checking all requests
030     * for the object with a {@code Guard}.
031     * 
032     * @since Android 1.0
033     */
034    public class GuardedObject implements Serializable {
035    
036        private static final long serialVersionUID = -5240450096227834308L;
037    
038        private final Object object;
039    
040        private final Guard guard;
041    
042        /**
043         * Constructs a new instance of {@code GuardedObject} which protects access
044         * to the specified {@code Object} using the specified {@code Guard}.
045         * 
046         * @param object
047         *            the {@code Object} to protect.
048         * @param guard
049         *            the {@code Guard} which protects the specified {@code Object},
050         *            maybe {@code null}.
051         * @since Android 1.0
052         */
053        public GuardedObject(Object object, Guard guard) {
054            this.object = object;
055            this.guard = guard;
056        }
057    
058        /**
059         * Returns the guarded {@code Object} if the associated {@code Guard}
060         * permits access. If access is not granted, then a {@code
061         * SecurityException} is thrown.
062         * 
063         * @return the guarded object.
064         * @exception SecurityException
065         *                if access is not granted to the guarded object.
066         * @since Android 1.0
067         */
068        public Object getObject() throws SecurityException {
069            if (guard != null) {
070                guard.checkGuard(object);
071            }
072            return object;
073        }
074    
075        /**
076         * Checks the guard (if there is one) before performing a default
077         * serialization.
078         */
079        private void writeObject(java.io.ObjectOutputStream out) throws IOException {
080            if (guard != null) {
081                guard.checkGuard(object);
082            }
083            out.defaultWriteObject();
084        }
085    }
