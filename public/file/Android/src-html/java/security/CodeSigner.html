023    package java.security;
024    
025    import java.io.Serializable;
026    import java.security.cert.CertPath;
027    
028    import org.apache.harmony.security.internal.nls.Messages;
029    
030    /**
031     * {@code CodeSigner} represents a signer of code. Instances are immutable.
032     * 
033     * @since Android 1.0
034     */
035    public final class CodeSigner implements Serializable {
036    
037        private static final long serialVersionUID = 6819288105193937581L;
038    
039        private CertPath signerCertPath;
040    
041        private Timestamp timestamp;
042    
043        // Cached hash code value
044        private transient int hash;
045    
046        /**
047         * Constructs a new instance of {@code CodeSigner}.
048         * 
049         * @param signerCertPath
050         *            the certificate path associated with this code signer.
051         * @param timestamp
052         *            the time stamp associated with this code signer, maybe {@code
053         *            null}.
054         * @throws NullPointerException
055         *             if {@code signerCertPath} is {@code null}.
056         * @since Android 1.0
057         */
058        public CodeSigner(CertPath signerCertPath, Timestamp timestamp) {
059            if (signerCertPath == null) {
060                throw new NullPointerException(Messages.getString("security.10")); //$NON-NLS-1$
061            }
062            this.signerCertPath = signerCertPath;
063            this.timestamp = timestamp;
064        }
065    
066        /**
067         * Compares the specified object with this {@code CodeSigner} for equality.
068         * Returns {@code true} if the specified object is also an instance of
069         * {@code CodeSigner}, the two {@code CodeSigner} encapsulate the same
070         * certificate path and the same time stamp, if present in both.
071         * 
072         * @param obj
073         *            object to be compared for equality with this {@code
074         *            CodeSigner}.
075         * @return {@code true} if the specified object is equal to this {@code
076         *         CodeSigner}, otherwise {@code false}.
077         * @since Android 1.0
078         */
079        public boolean equals(Object obj) {
080            if (obj == this) {
081                return true;
082            }
083            if (obj instanceof CodeSigner) {
084                CodeSigner that = (CodeSigner) obj;
085                if (!signerCertPath.equals(that.signerCertPath)) {
086                    return false;
087                }
088                return timestamp == null ? that.timestamp == null : timestamp
089                        .equals(that.timestamp);
090            }
091            return false;
092        }
093    
094        /**
095         * Returns the certificate path associated with this {@code CodeSigner}.
096         * 
097         * @return the certificate path associated with this {@code CodeSigner}.
098         * @since Android 1.0
099         */
100        public CertPath getSignerCertPath() {
101            return signerCertPath;
102        }
103    
104        /**
105         * Returns the time stamp associated with this {@code CodeSigner}.
106         * 
107         * @return the time stamp associated with this {@code CodeSigner}, maybe
108         *         {@code null}.
109         * @since Android 1.0
110         */
111        public Timestamp getTimestamp() {
112            return timestamp;
113        }
114    
115        /**
116         * Returns the hash code value for this {@code CodeSigner}. Returns the same
117         * hash code for {@code CodeSigner}s that are equal to each other as
118         * required by the general contract of {@link Object#hashCode}.
119         * 
120         * @return the hash code value for this {@code CodeSigner}.
121         * @see Object#equals(Object)
122         * @see CodeSigner#equals(Object)
123         * @since Android 1.0
124         */
125        public int hashCode() {
126            if (hash == 0) {
127                hash = signerCertPath.hashCode()
128                        ^ (timestamp == null ? 0 : timestamp.hashCode());
129            }
130            return hash;
131        }
132    
133        /**
134         * Returns a string containing a concise, human-readable description of the
135         * this {@code CodeSigner} including its first certificate and its time
136         * stamp, if present.
137         * 
138         * @return a printable representation for this {@code CodeSigner}.
139         * @since Android 1.0
140         */
141        public String toString() {
142            // There is no any special reason for '256' here, it's taken abruptly
143            // FIXME: 1.5 StringBuffer => StringBuilder
144            StringBuffer buf = new StringBuffer(256);
145            // The javadoc says nothing, and the others implementations behavior seems as 
146            // dumping only the first certificate. Well, let's do the same.
147            buf.append("CodeSigner [").append(signerCertPath.getCertificates().get(0)); //$NON-NLS-1$
148            if( timestamp != null ) {
149                buf.append("; ").append(timestamp); //$NON-NLS-1$
150            }
151            buf.append("]"); //$NON-NLS-1$
152            return buf.toString();
153        }
154    }
