023    package java.security;
024    
025    import java.io.Serializable;
026    
027    /**
028     * {@code Permission} is the common base class of all permissions that
029     * participate in the access control security framework around
030     * {@link AccessController} and {@link AccessControlContext}. A permission
031     * constitutes of a name and associated actions.
032     * 
033     * @since Android 1.0
034     */
035    public abstract class Permission implements Guard, Serializable {
036    
037        private static final long serialVersionUID = -5636570222231596674L;
038    
039        private final String name;
040    
041        /**
042         * Compares the specified object with this {@code Permission} for equality
043         * and returns {@code true} if the specified object is equal, {@code false}
044         * otherwise.
045         * <p>
046         * The {@link #implies(Permission)} method should be used for making access
047         * control checks.
048         * </p>
049         * 
050         * @param obj
051         *            object to be compared for equality with this {@code
052         *            Permission}.
053         * @return {@code true} if the specified object is equal to this {@code
054         *         Permission}, otherwise {@code false}.
055         * @since Android 1.0
056         */
057        public abstract boolean equals(Object obj);
058    
059        /**
060         * Returns the hash code value for this {@code Permission}. Returns the same
061         * hash code for {@code Permission}s that are equal to each other as
062         * required by the general contract of {@link Object#hashCode}.
063         * 
064         * @return the hash code value for this {@code Permission}.
065         * @see Object#equals(Object)
066         * @see Permission#equals(Object)
067         * @since Android 1.0
068         */
069        public abstract int hashCode();
070    
071        /**
072         * Returns a comma separated string identifying the actions associated with
073         * this permission. The returned actions are in canonical form. For example:
074         * 
075         * <pre>
076         * sp0 = new SocketPermission(&quot;www.google.com&quot;, &quot;connect,resolve&quot;)
077         * sp1 = new SocketPermission(&quot;www.google.com&quot;, &quot;resolve,connect&quot;)
078         * sp0.getActions().equals(sp1.getActions()) //yields true
079         * </pre>
080         * 
081         * Both permissions return "connect,resolve" (in that order) if {@code
082         * #getActions()} is invoked. Returns an empty String, if no actions are
083         * associated with this permission.
084         * 
085         * @return the actions associated with this permission or an empty string if
086         *         no actions are associated with this permission.
087         * @since Android 1.0
088         */
089        public abstract String getActions();
090    
091        /**
092         * Indicates whether the specified permission is implied by this permission.
093         * The {@link AccessController} uses this method to check whether permission
094         * protected access is allowed with the present policy.
095         * 
096         * @param permission
097         *            the permission to check against this permission.
098         * @return {@code true} if the specified permission is implied by this
099         *         permission, {@code false} otherwise.
100         * @since Android 1.0
101         */
102        public abstract boolean implies(Permission permission);
103    
104        /**
105         * Constructs a new instance of {@code Permission} with its name.
106         * 
107         * @param name
108         *            the name of the permission.
109         * @since Android 1.0
110         */
111        public Permission(String name) {
112            this.name = name;
113        }
114    
115        /**
116         * Returns the name of this permission.
117         * 
118         * @return the name of this permission.
119         * @since Android 1.0
120         */
121        public final String getName() {
122            return name;
123        }
124    
125        /**
126         * Invokes {@link SecurityManager#checkPermission(Permission)} with this
127         * permission as its argument. This method implements the {@link Guard}
128         * interface.
129         * 
130         * @param obj
131         *            as specified in {@link Guard#checkGuard(Object)} but ignored
132         *            in this implementation.
133         * @throws SecurityException
134         *             if this permission is not granted.
135         * @see Guard
136         * @see SecurityManager#checkPermission(Permission)
137         * @since Android 1.0
138         */
139        public void checkGuard(Object obj) throws SecurityException {
140            SecurityManager sm = System.getSecurityManager();
141            if (sm != null) {
142                sm.checkPermission(this);
143            }
144        }
145    
146        /**
147         * Returns a specific {@link PermissionCollection} container for permissions
148         * of this type. Returns {@code null} if any permission collection can be
149         * used.
150         * <p>
151         * Subclasses may override this method to return an appropriate collection
152         * for the specific permissions they implement.
153         * </p>
154         * 
155         * @return an empty {@link PermissionCollection} or {@code null} if any
156         *         permission collection can be used.
157         * @since Android 1.0
158         */
159        public PermissionCollection newPermissionCollection() {
160            return null;
161        }
162    
163        /**
164         * Returns a string containing a concise, human-readable description of the
165         * this {@code Permission} including its name and its actions.
166         * 
167         * @return a printable representation for this {@code Permission}.
168         * @since Android 1.0
169         */
170        public String toString() {
171            String actions = getActions();
172            actions = (actions == null || actions.length() == 0) ? "" : " " //$NON-NLS-1$ //$NON-NLS-2$
173                    + getActions();
174            return "(" + getClass().getName() + " " + getName() + actions + ")"; //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
175        }
176    }
