022    package java.security;
023    
024    import java.nio.ByteBuffer;
025    import java.security.spec.AlgorithmParameterSpec;
026    
027    import org.apache.harmony.security.internal.nls.Messages;
028    
029    /**
030     * {@code SignatureSpi} is the <i>Service Provider Interface</i> (<b>SPI</b>)
031     * definition for {@link Signature}.
032     * 
033     * @see Signature
034     * @since Android 1.0
035     */
036    public abstract class SignatureSpi {
037    
038        /**
039         * Implementation specific source of randomness.
040         * 
041         * @since Android 1.0
042         */
043        protected SecureRandom appRandom;
044    
045        /**
046         * Initializes this {@code SignatureSpi} instance for signature
047         * verification, using the public key of the identity whose signature is
048         * going to be verified.
049         * 
050         * @param publicKey
051         *            the public key.
052         * @throws InvalidKeyException
053         *             if {@code publicKey} is not valid.
054         * @since Android 1.0
055         */
056        protected abstract void engineInitVerify(PublicKey publicKey)
057                throws InvalidKeyException;
058    
059        /**
060         * Initializes this {@code SignatureSpi} instance for signing, using the
061         * private key of the identity whose signature is going to be generated.
062         * 
063         * @param privateKey
064         *            the private key.
065         * @throws InvalidKeyException
066         *             if {@code privateKey} is not valid.
067         * @since Android 1.0
068         */
069        protected abstract void engineInitSign(PrivateKey privateKey)
070                throws InvalidKeyException;
071    
072        /**
073         * Initializes this {@code SignatureSpi} instance for signing, using the
074         * private key of the identity whose signature is going to be generated and
075         * the specified source of randomness.
076         * 
077         * @param privateKey
078         *            the private key.
079         * @param random
080         *            the {@code SecureRandom} to use.
081         * @throws InvalidKeyException
082         *             if {@code privateKey} is not valid.
083         * @since Android 1.0
084         */
085        protected void engineInitSign(PrivateKey privateKey, SecureRandom random)
086                throws InvalidKeyException {
087            appRandom = random;
088            engineInitSign(privateKey);
089        }
090    
091        /**
092         * Updates the data to be verified or to be signed, using the specified
093         * {@code byte}.
094         * 
095         * @param b
096         *            the byte to update with.
097         * @throws SignatureException
098         *             if this {@code SignatureSpi} instance is not initialized
099         *             properly.
100         * @since Android 1.0
101         */
102        protected abstract void engineUpdate(byte b) throws SignatureException;
103    
104        /**
105         * Updates the data to be verified or to be signed, using the given {@code
106         * byte[]}, starting form the specified index for the specified length.
107         * 
108         * @param b
109         *            the byte array to update with.
110         * @param off
111         *            the start index in {@code b} of the data.
112         * @param len
113         *            the number of bytes to use.
114         * @throws SignatureException
115         *             if this {@code SignatureSpi} instance is not initialized
116         *             properly.
117         * @since Android 1.0
118         */
119        protected abstract void engineUpdate(byte[] b, int off, int len)
120                throws SignatureException;
121    
122        /**
123         * Updates the data to be verified or to be signed, using the specified
124         * {@code ByteBuffer}.
125         * 
126         * @param input
127         *            the {@code ByteBuffer} to update with.
128         * @throws RuntimeException
129         *             since {@code SignatureException} is not specified for this
130         *             method it throws a {@code RuntimeException} if underlying
131         *             {@link #engineUpdate(byte[], int, int)} throws {@code
132         *             SignatureException}.
133         * @since Android 1.0
134         */
135        protected void engineUpdate(ByteBuffer input) {
136            if (!input.hasRemaining()) {
137                return;
138            }
139            byte[] tmp;
140            if (input.hasArray()) {
141                tmp = input.array();
142                int offset = input.arrayOffset();
143                int position = input.position();
144                int limit = input.limit();
145                try {
146                    engineUpdate(tmp, offset + position, limit - position);
147                } catch (SignatureException e) { 
148                    throw new RuntimeException(e); //Wrap SignatureException
149                }
150                input.position(limit);
151            } else {
152                tmp = new byte[input.limit() - input.position()];
153                input.get(tmp);
154                try {
155                    engineUpdate(tmp, 0, tmp.length);
156                } catch (SignatureException e) {
157                    throw new RuntimeException(e); //Wrap SignatureException
158                }
159            }
160        }
161    
162        /**
163         * Generates and returns the signature of all updated data.
164         * <p>
165         * This {@code SignatureSpi} instance is reset to the state of its last
166         * initialization for signing and thus can be used for another signature
167         * from the same identity.
168         * </p>
169         * 
170         * @return the signature of all updated data.
171         * @throws SignatureException
172         *             if this {@code SignatureSpi} instance is not initialized
173         *             properly.
174         * @since Android 1.0
175         */
176        protected abstract byte[] engineSign() throws SignatureException;
177    
178        /**
179         * Generates and stores the signature of all updated data in the provided
180         * {@code byte[]} at the specified position with the specified length.
181         * <p>
182         * This {@code SignatureSpi} instance is reset to the state of its last
183         * initialization for signing and thus can be used for another signature
184         * from the same identity.
185         * </p>
186         * 
187         * @param outbuf
188         *            the buffer to store the signature.
189         * @param offset
190         *            the index of the first byte in {@code outbuf} to store.
191         * @param len
192         *            the number of bytes allocated for the signature.
193         * @return the number of bytes stored in {@code outbuf}.
194         * @throws SignatureException
195         *             if this {@code SignatureSpi} instance is not initialized
196         *             properly.
197         * @throws IllegalArgumentException
198         *             if {@code offset} or {@code len} are not valid in respect to
199         *             {@code outbuf}.
200         * @since Android 1.0
201         */
202        protected int engineSign(byte[] outbuf, int offset, int len)
203                throws SignatureException {
204            byte tmp[] = engineSign();
205            if (tmp == null) {
206                return 0;
207            }
208            if (len < tmp.length) {
209                throw new SignatureException(Messages.getString("security.2D")); //$NON-NLS-1$
210            }
211            if (offset < 0) {
212                throw new SignatureException(Messages.getString("security.1C")); //$NON-NLS-1$
213            }
214            if (offset + len > outbuf.length) {
215                throw new SignatureException(Messages.getString("security.05")); //$NON-NLS-1$
216            }
217            System.arraycopy(tmp, 0, outbuf, offset, tmp.length);
218            return tmp.length;
219        }
220    
221        /**
222         * Indicates whether the given {@code sigBytes} can be verified using the
223         * public key or a certificate of the signer.
224         * <p>
225         * This {@code SignatureSpi} instance is reset to the state of its last
226         * initialization for verifying and thus can be used to verify another
227         * signature of the same signer.
228         * </p>
229         * 
230         * @param sigBytes
231         *            the signature to verify.
232         * @return {@code true} if the signature was verified, {@code false}
233         *         otherwise.
234         * @throws SignatureException
235         *             if this {@code SignatureSpi} instance is not initialized
236         *             properly.
237         * @since Android 1.0
238         */
239        protected abstract boolean engineVerify(byte[] sigBytes)
240                throws SignatureException;
241    
242        /**
243         * Indicates whether the given {@code sigBytes} starting at index {@code
244         * offset} with {@code length} bytes can be verified using the public key or
245         * a certificate of the signer.
246         * <p>
247         * This {@code SignatureSpi} instance is reset to the state of its last
248         * initialization for verifying and thus can be used to verify another
249         * signature of the same signer.
250         * </p>
251         * 
252         * @param sigBytes
253         *            the {@code byte[]} containing the signature to verify.
254         * @param offset
255         *            the start index in {@code sigBytes} of the signature
256         * @param length
257         *            the number of bytes allocated for the signature.
258         * @return {@code true} if the signature was verified, {@code false}
259         *         otherwise.
260         * @throws SignatureException
261         *             if this {@code SignatureSpi} instance is not initialized
262         *             properly.
263         * @throws IllegalArgumentException
264         *             if {@code offset} or {@code length} are not valid in respect
265         *             to {@code sigBytes}.
266         * @since Android 1.0
267         */
268        protected boolean engineVerify(byte[] sigBytes, int offset, int length)
269                throws SignatureException {
270            byte tmp[] = new byte[length];
271            System.arraycopy(sigBytes, offset, tmp, 0, length);
272            return engineVerify(tmp);
273        }
274    
275        /**
276         * Sets the specified parameter to the given value.
277         * 
278         * @param param
279         *            the name of the parameter.
280         * @param value
281         *            the parameter value.
282         * @throws InvalidParameterException
283         *             if the parameter is invalid, already set or is not allowed to
284         *             be changed.
285         * @deprecated Use {@link #engineSetParameter(AlgorithmParameterSpec)}
286         * @since Android 1.0
287         */
288        @Deprecated
289        protected abstract void engineSetParameter(String param, Object value)
290                throws InvalidParameterException;
291    
292        /**
293         * Sets the specified {@code AlgorithmParameterSpec}.
294         * 
295         * @param params
296         *            the parameter to set.
297         * @throws InvalidAlgorithmParameterException
298         *             if the parameter is invalid, already set or is not allowed to
299         *             be changed.
300         * @since Android 1.0
301         */
302        protected void engineSetParameter(AlgorithmParameterSpec params)
303                throws InvalidAlgorithmParameterException {
304            throw new UnsupportedOperationException();
305        }
306    
307        /**
308         * Returns the {@code AlgorithmParameters} of this {@link SignatureSpi}
309         * instance.
310         * 
311         * @return the {@code AlgorithmParameters} of this {@link SignatureSpi}
312         *         instance, maybe {@code null}.
313         * @since Android 1.0
314         */
315        protected AlgorithmParameters engineGetParameters() {
316            throw new UnsupportedOperationException();
317        }
318

322        /**
323         * Returns the value of the parameter with the specified name.
324         * 
325         * @param param
326         *            the name of the requested parameter value.
327         * @return the value of the parameter with the specified name, maybe {@code
328         *         null}.
329         * @throws InvalidParameterException
330         *             if {@code param} is not a valid parameter for this {@code
331         *             SignatureSpi} or an other error occurs.
332         * @deprecated There is no generally accepted parameter naming convention.
333         * @since Android 1.0
334         */
335        @Deprecated
336        protected abstract Object engineGetParameter(String param)
337                throws InvalidParameterException;
338    
339        public Object clone() throws CloneNotSupportedException {
340            if (this instanceof Cloneable) {
341                return super.clone();
342            } else {
343                throw new CloneNotSupportedException();
344            }
345        }
