023    package java.security;
024    
025    import java.io.IOException;
026    import java.io.NotSerializableException;
027    import java.io.ObjectInputStream;
028    import java.io.ObjectStreamException;
029    import java.io.Serializable;
030    import java.security.spec.X509EncodedKeySpec;
031    import java.security.spec.PKCS8EncodedKeySpec;
032    import java.security.spec.InvalidKeySpecException;
033    
034    import javax.crypto.spec.SecretKeySpec;
035    
036    import org.apache.harmony.security.internal.nls.Messages;
037    
038    /**
039     * {@code KeyRep} is a standardized representation for serialized {@link Key}
040     * objects.
041     * 
042     * @since Android 1.0
043     */
044    public class KeyRep implements Serializable {
045    
046        private static final long serialVersionUID = -4757683898830641853L;
047        // Key type
048        private final Type type;
049        // Key algorithm name
050        private final String algorithm;
051        // Key encoding format
052        private final String format;
053        // Key encoding
054        private byte[] encoded;
055    
056        /**
057         * Constructs a new instance of {@code KeyRep} with the specified arguments.
058         * The arguments should be obtained from the {@code Key} object that has to
059         * be serialized.
060         * 
061         * @param type
062         *            the type of the key.
063         * @param algorithm
064         *            the algorithm (obtained by {@link Key#getAlgorithm()}).
065         * @param format
066         *            the format of the key (obtained by {@link Key#getFormat()}).
067         * @param encoded
068         *            the encoded {@code byte[]} (obtained by
069         *            {@link Key#getEncoded()}).
070         * @throws NullPointerException
071         *             if {@code type, algorithm, format or encoded} is {@code null}
072         *             .
073         * @since Android 1.0
074         */
075        public KeyRep(Type type,
076                String algorithm, String format, byte[] encoded) {
077            this.type = type;
078            this.algorithm = algorithm;
079            this.format = format;
080            this.encoded = encoded;
081            if(this.type == null) {
082                throw new NullPointerException(Messages.getString("security.07")); //$NON-NLS-1$
083            }
084            if(this.algorithm == null) {
085                throw new NullPointerException(Messages.getString("security.08")); //$NON-NLS-1$
086            }
087            if(this.format == null) {
088                throw new NullPointerException(Messages.getString("security.09")); //$NON-NLS-1$
089            }
090            if(this.encoded == null) {
091                throw new NullPointerException(Messages.getString("security.0A")); //$NON-NLS-1$
092            }
093        }
094    
095        /**
096         * Resolves and returns the {@code Key} object. Three {@link Type}|format
097         * combinations are supported:
098         * <ul>
099         * <li> {@code Type.PRIVATE} | "PKCS#8" : returns a {@link PrivateKey}
100         * instance, generated from a key factory (suitable for the algorithm) that
101         * is initialized with a {@link PKCS8EncodedKeySpec} using the encoded key
102         * bytes.
103         * <li> {@code Type.SECRET} | "RAW" : returns a {@link SecretKeySpec}
104         * instance, created with the encoded key bytes and the algorithm.
105         * <li> {@code Type.PUBLIC} | "X.509": returns a {@link PublicKey} instance,
106         * generated from a key factory (suitable for the algorithm) that is
107         * initialized with a {@link X509EncodedKeySpec} using the encoded key
108         * bytes.
109         * </ul>
110         * 
111         * @return the resolved {@code Key} object.
112         * @throws ObjectStreamException
113         *             if the {@code Type}|format combination is not recognized, or
114         *             the resolution of any key parameter fails.
115         * @since Android 1.0
116         */
117        protected Object readResolve() throws ObjectStreamException {
118            switch (type) {
119            case SECRET:
120                if ("RAW".equals(format)) { //$NON-NLS-1$
121                    try {
122                        return new SecretKeySpec(encoded, algorithm);
123                    } catch (IllegalArgumentException e) {
124                        throw new NotSerializableException(
125                                Messages.getString("security.0B", e)); //$NON-NLS-1$
126                    }
127                }
128                throw new NotSerializableException(
129                    Messages.getString("security.0C", type, format)); //$NON-NLS-1$
130            case PUBLIC:
131                if ("X.509".equals(format)) { //$NON-NLS-1$
132                    try {
133                        KeyFactory kf = KeyFactory.getInstance(algorithm);
134                        return kf.generatePublic(new X509EncodedKeySpec(encoded));
135                    } catch (NoSuchAlgorithmException e) {
136                        throw new NotSerializableException(
137                                Messages.getString("security.0D", e)); //$NON-NLS-1$
138                    }
139                    catch (InvalidKeySpecException e) {
140                        throw new NotSerializableException(
141                                Messages.getString("security.0D", e)); //$NON-NLS-1$
142                    }
143                }
144                throw new NotSerializableException(
145                    Messages.getString("security.0C", type, format)); //$NON-NLS-1$
146            case PRIVATE:
147                if ("PKCS#8".equals(format)) { //$NON-NLS-1$
148                    try {
149                        KeyFactory kf = KeyFactory.getInstance(algorithm);
150                        return kf.generatePrivate(new PKCS8EncodedKeySpec(encoded));
151                    } catch (NoSuchAlgorithmException e) {
152                        throw new NotSerializableException(
153                                Messages.getString("security.0D", e)); //$NON-NLS-1$
154                    }
155                    catch (InvalidKeySpecException e) {
156                        throw new NotSerializableException(
157                                Messages.getString("security.0D", e)); //$NON-NLS-1$
158                    }
159                }
160                throw new NotSerializableException(
161                    Messages.getString("security.0C", type, format)); //$NON-NLS-1$
162            }
163            throw new NotSerializableException(Messages.getString("security.0E", type)); //$NON-NLS-1$
164        }
165    
166        // Makes defensive copy of key encoding
167        private void readObject(ObjectInputStream is)
168            throws IOException, ClassNotFoundException {
169            is.defaultReadObject();
170            byte[] new_encoded = new byte[encoded.length];
171            System.arraycopy(encoded, 0, new_encoded, 0, new_encoded.length);
172            encoded = new_encoded;    
173        }
174    
175        /**
176         * {@code Type} enumerates the supported key types.
177         * @since Android 1.0
178         */
179        public static enum Type {
180            
181            /**
182             * Type for secret keys.
183             */
184            SECRET,
185            /**
186             * Type for public keys.
187             */
188            PUBLIC,
189            /**
190             * Type for private keys.
191             */
192            PRIVATE
193        }
194    }
