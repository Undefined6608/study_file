018    package java.security.spec;
019    
020    import org.apache.harmony.security.internal.nls.Messages;
021    
022    /**
023     * The parameter specification for the RSA-PSS Signature scheme.
024     * <p>
025     * Defined in the <a
026     * href="http://www.rsa.com/rsalabs/pubs/PKCS/html/pkcs-1.html">PKCS #1 v2.1</a>
027     * standard.
028     * </p>
029     * 
030     * @since Android 1.0
031     */
032    public class PSSParameterSpec implements AlgorithmParameterSpec {   
033    
034        /**
035         * The default parameter specification. It specifies the following parameters:
036         * <ul>
037         * <li>message digest: {@code "SHA-1"}</li>
038         * <li>mask generation function (<i>mgf</i>): {@code "MGF1"}</li>
039         * <li>parameters for the <i>mgf</i>: {@link MGF1ParameterSpec#SHA1}</li>
040         * <li>salt length: {@code 20}</li>
041         * <li>trailer field: {@code -1}</li>
042         * </ul>
043         * 
044         * @since Android 1.0
045         */
046        public static final PSSParameterSpec DEFAULT = new PSSParameterSpec(20);
047    
048        // Message digest algorithm name
049        private final String mdName;
050        // Mask generation function algorithm name
051        private final String mgfName;
052        // Mask generation function parameters
053        private final AlgorithmParameterSpec mgfSpec;
054        // Trailer field value
055        private final int trailerField;
056        // Salt length in bits
057        private final int saltLen;
058    
059        /**
060         * Creates a new {@code PSSParameterSpec} with the specified salt length 
061         * and the default values.
062         * 
063         * @param saltLen
064         *            the salt length (in bits).
065         * @throws IllegalArgumentException
066         *             if {@code saltLen} is negative.
067         * @since Android 1.0
068         */
069        public PSSParameterSpec(int saltLen) {
070            if (saltLen < 0) {
071                throw new IllegalArgumentException(Messages.getString("security.7F")); //$NON-NLS-1$
072            }
073            this.saltLen = saltLen;
074            this.mdName = "SHA-1"; //$NON-NLS-1$
075            this.mgfName = "MGF1"; //$NON-NLS-1$
076            this.mgfSpec = MGF1ParameterSpec.SHA1;
077            this.trailerField = 1;
078        }
079    
080        /**
081         * Creates a new {@code PSSParameterSpec} with the specified message digest
082         * name, mask generation function name, mask generation function parameters,
083         * salt length, and trailer field value.
084         * 
085         * @param mdName
086         *            the name of the message digest algorithm.
087         * @param mgfName
088         *            the name of the mask generation function algorithm.
089         * @param mgfSpec
090         *            the parameter for the mask generation function algorithm.
091         * @param saltLen
092         *            the salt length (in bits).
093         * @param trailerField
094         *            the trailer field value.
095         * @throws IllegalArgumentException
096         *             if {@code saltLen} or {@code trailerField} is negative.
097         * @since Android 1.0
098         */
099        public PSSParameterSpec(String mdName, String mgfName,
100                AlgorithmParameterSpec mgfSpec, int saltLen, int trailerField) {
101    
102            if (mdName == null) {
103                throw new NullPointerException(Messages.getString("security.80")); //$NON-NLS-1$
104            }
105            if (mgfName == null) {
106                throw new NullPointerException(Messages.getString("security.81")); //$NON-NLS-1$
107            }
108            if (saltLen < 0) {
109                throw new IllegalArgumentException(Messages.getString("security.7F")); //$NON-NLS-1$
110            }
111            if (trailerField < 0) {
112                throw new IllegalArgumentException(Messages.getString("security.82")); //$NON-NLS-1$
113            }
114            this.mdName = mdName;
115            this.mgfName = mgfName;
116            this.mgfSpec = mgfSpec;
117            this.saltLen = saltLen;
118            this.trailerField = trailerField;
119        }
120    
121        /**
122         * Returns the length of the salt (in bits).
123         * 
124         * @return the length of the salt (in bits).
125         * @since Android 1.0
126         */
127        public int getSaltLength() {
128            return saltLen;
129        }
130    
131        /**
132         * Returns the name of the message digest algorithm.
133         * 
134         * @return the name of the message digest algorithm.
135         * @since Android 1.0
136         */
137        public String getDigestAlgorithm() {
138            return mdName;
139        }
140    
141        /**
142         * Returns the name of the mask generation function algorithm.
143         * 
144         * @return the name of the mask generation function algorithm.
145         * @since Android 1.0
146         */
147        public String getMGFAlgorithm() {
148            return mgfName;
149        }
150    
151        /**
152         * Returns the parameter for the mask generation function algorithm.
153         * 
154         * @return the parameter for the mask generation function algorithm, or
155         *         {@code null} if none specified.
156         * @since Android 1.0
157         */
158        public AlgorithmParameterSpec getMGFParameters() {
159            return mgfSpec;
160        }
161    
162        /**
163         * Returns the trailer field value.
164         * 
165         * @return the trailer field value.
166         * @since Android 1.0
167         */
168        public int getTrailerField() {
169            return trailerField;
170        }
171    }
