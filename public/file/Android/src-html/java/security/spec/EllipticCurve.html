018    package java.security.spec;
019    
020    import java.math.BigInteger;
021    import java.util.Arrays;
022    
023    import org.apache.harmony.security.internal.nls.Messages;
024    
025    /**
026     * An Elliptic Curve with its necessary values.
027     * 
028     * @since Android 1.0
029     */
030    public class EllipticCurve {
031    
032        // Underlying finite field
033        private final ECField field;
034    
035        // The first coefficient of the equation defining this elliptic curve
036        private final BigInteger a;
037    
038        // The second coefficient of the equation defining this elliptic curve
039        private final BigInteger b;
040    
041        // Bytes used during this elliptic curve generation,
042        // if it was generated randomly
043        private final byte[] seed;
044    
045        // Hash code
046        private volatile int hash;
047    
048        /**
049         * Creates a new {@code EllipticCurve} with the specified field,
050         * coefficients and seed.
051         * 
052         * @param field
053         *            the finite field of this elliptic curve.
054         * @param a
055         *            the coefficient {@code a}.
056         * @param b
057         *            the coefficient {@code b}.
058         * @param seed
059         *            the seed used for the generation of the curve.
060         * @throws IllegalArgumentException
061         *             if the specified coefficients are not in the specified field.
062         * @since Android 1.0
063         */
064        public EllipticCurve(ECField field, BigInteger a, BigInteger b, byte[] seed) {
065            this.field = field;
066            if (this.field == null) {
067                throw new NullPointerException(Messages.getString("security.7A")); //$NON-NLS-1$
068            }
069            this.a = a;
070            if (this.a == null) {
071                throw new NullPointerException(Messages.getString("security.7B")); //$NON-NLS-1$
072            }
073            this.b = b;
074            if (this.b == null) {
075                throw new NullPointerException(Messages.getString("security.7C")); //$NON-NLS-1$
076            }
077            // make defensive copy
078            if (seed == null) {
079                this.seed = null;
080            } else {
081                this.seed = new byte[seed.length];
082                System.arraycopy(seed, 0, this.seed, 0, this.seed.length);
083            }
084            // check parameters for ECFieldFp and ECFieldF2m.
085            // Check invariant: a and b must be in the field.
086            // Check conditions for custom ECField are not specified.
087            if (this.field instanceof ECFieldFp) {
088                BigInteger p = ((ECFieldFp) this.field).getP();
089                if (this.a.signum() < 0 || this.a.compareTo(p) >= 0) {
090                    throw new IllegalArgumentException(Messages.getString("security.7D")); //$NON-NLS-1$
091                }
092                if (this.b.signum() < 0 || this.b.compareTo(p) >= 0) {
093                    throw new IllegalArgumentException(Messages.getString("security.7E")); //$NON-NLS-1$
094                }
095            } else if (this.field instanceof ECFieldF2m) {
096                int fieldSizeInBits = this.field.getFieldSize();
097                if (!(this.a.bitLength() <= fieldSizeInBits)) {
098                    throw new IllegalArgumentException(Messages.getString("security.7D")); //$NON-NLS-1$
099                }
100                if (!(this.b.bitLength() <= fieldSizeInBits)) {
101                    throw new IllegalArgumentException(Messages.getString("security.7E")); //$NON-NLS-1$
102                }
103            }
104        }
105    
106        /**
107         * Creates a new {@code EllipticCurve} with the specified field and
108         * coefficients.
109         * 
110         * @param field
111         *            the finite field of this elliptic curve.
112         * @param a
113         *            the coefficient {@code a}.
114         * @param b
115         *            the coefficient {@code b}.
116         * @throws IllegalArgumentException
117         *             if the specified coefficients are not in the specified field.
118         * @since Android 1.0
119         */
120        public EllipticCurve(ECField field, BigInteger a, BigInteger b) {
121            this(field, a, b, null);
122        }
123    
124        /**
125         * Returns the coefficient {@code a} of this elliptic curve.
126         * 
127         * @return the coefficient {@code a} of this elliptic curve.
128         * @since Android 1.0
129         */
130        public BigInteger getA() {
131            return a;
132        }
133    
134        /**
135         * Returns the coefficient {@code b} of this elliptic curve.
136         * 
137         * @return the coefficient {@code b} of this elliptic curve.
138         * @since Android 1.0
139         */
140        public BigInteger getB() {
141            return b;
142        }
143    
144        /**
145         * Returns the finite field of this elliptic curve.
146         * 
147         * @return the finite field of this elliptic curve.
148         * @since Android 1.0
149         */
150        public ECField getField() {
151            return field;
152        }
153    
154        /**
155         * Returns a copy of the seed that was used to generate this elliptic curve.
156         * 
157         * @return a copy of the seed that was used to generate this elliptic curve,
158         *         or {@code null} if none specified.
159         * @since Android 1.0
160         */
161        public byte[] getSeed() {
162            if (seed == null) {
163                return null;
164            } else {
165                // return copy
166                byte[] ret = new byte[seed.length];
167                System.arraycopy(seed, 0, ret, 0, ret.length);
168                return ret;
169            }
170        }
171    
172        /**
173         * Returns whether the specified object equals to this elliptic curve.
174         * 
175         * @param other
176         *            the object to compare.
177         * @return {@code true} if the specified object is equal to this elliptic
178         *         curve, otherwise {@code false}.
179         * @since Android 1.0
180         */
181        public boolean equals(Object other) {
182            if (this == other) {
183                return true;
184            }
185            if (!(other instanceof EllipticCurve)) {
186                return false;
187            }
188            EllipticCurve otherEc = (EllipticCurve) other;
189            return this.field.equals(otherEc.field) && this.a.equals(otherEc.a)
190                    && this.b.equals(otherEc.b)
191                    && Arrays.equals(this.seed, otherEc.seed);
192        }
193    
194        /**
195         * Returns the hashcode of this elliptic curve.
196         * 
197         * @return the hashcode of this elliptic curve.
198         * @since Android 1.0
199         */
200        public int hashCode() {
201            // hash init is delayed
202            if (hash == 0) {
203                int hash0 = 11;
204                hash0 = hash0 * 31 + field.hashCode();
205                hash0 = hash0 * 31 + a.hashCode();
206                hash0 = hash0 * 31 + b.hashCode();
207                if (seed != null) {
208                    for (int i = 0; i < seed.length; i++) {
209                        hash0 = hash0 * 31 + seed[i];
210                    }
211                } else {
212                    hash0 = hash0 * 31;
213                }
214                hash = hash0;
215            }
216            return hash;
217        }
218    }
