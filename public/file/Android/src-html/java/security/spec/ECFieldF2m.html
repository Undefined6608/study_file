018    package java.security.spec;
019    
020    import java.math.BigInteger;
021    import java.util.Arrays;
022    
023    import org.apache.harmony.security.internal.nls.Messages;
024    
025    /**
026     * The parameters specifying a <i>characteristic 2 finite field</i> of an
027     * elliptic curve.
028     * 
029     * @since Android 1.0
030     */
031    public class ECFieldF2m implements ECField {
032        // Mid terms array length for trinomial basis
033        private static final int TPB_MID_LEN = 1;
034        // Mid terms array length for pentanomial basis
035        private static final int PPB_MID_LEN = 3;
036        // All terms number for trinomial basis
037        private static final int TPB_LEN = TPB_MID_LEN + 2;
038        // All terms number for pentanomial basis
039        private static final int PPB_LEN = PPB_MID_LEN + 2;
040        // m value
041        private final int m;
042        // Reduction polynomial
043        private final BigInteger rp;
044        // Mid term(s) of reduction polynomial
045        private final int[] ks;
046    
047        /**
048         * Creates a new {@code ECFieldF2m} with {@code 2^m} elements with a normal
049         * basis.
050         * 
051         * @param m
052         *            the exponent {@code m} for the number of elements.
053         * @throws IllegalArgumentException
054         *             if {@code m <= zero}.
055         * @since Android 1.0
056         */
057        public ECFieldF2m(int m) {
058            this.m = m;
059            if (this.m <= 0) {
060                throw new IllegalArgumentException(Messages.getString("security.75")); //$NON-NLS-1$
061            }
062            this.rp = null;
063            this.ks = null;
064        }
065    
066        /**
067         * Creates a new {@code ECFieldF2m} with {@code 2^m} elements with a polynomial
068         * basis and the reduction polynomial based on {@code rp}.
069         * <p>
070         * The reduction polynomial must be either <i>trinomial</i> or
071         * <i>pentanomial</i>.
072         * </p>
073         * 
074         * @param m
075         *            the exponent {@code m} for the number of elements.
076         * @param rp
077         *            the base of the reduction polynomial with the n-th bit
078         *            corresponding to the n-th coefficient of the reduction
079         *            polynomial.
080         * @throws IllegalArgumentException
081         *             if {@code m <= zero} or the {@code rp} is invalid.
082         * @since Android 1.0
083         */
084        public ECFieldF2m(int m, BigInteger rp) {
085            this.m = m;
086            if (this.m <= 0) {
087                throw new IllegalArgumentException(Messages.getString("security.75")); //$NON-NLS-1$
088            }
089            this.rp = rp;
090            if (this.rp == null) {
091                throw new NullPointerException(Messages.getString("security.76")); //$NON-NLS-1$
092            }
093            // the leftmost bit must be (m+1)-th one,
094            // set bits count must be 3 or 5,
095            // bits 0 and m must be set
096            int rp_bc = this.rp.bitCount();
097            if ((this.rp.bitLength() != (m+1)) ||
098                (rp_bc != TPB_LEN && rp_bc != PPB_LEN) ||
099                (!this.rp.testBit(0) || !this.rp.testBit(m)) ) {
100                throw new IllegalArgumentException(Messages.getString("security.77")); //$NON-NLS-1$
101            }
102    
103            // setup ks using rp:
104            // allocate for mid terms only
105            ks = new int[rp_bc-2];
106            // find midterm orders and set ks accordingly
107            BigInteger rpTmp = rp.clearBit(0);
108            for (int i=ks.length-1; i>=0; i-- ) {
109                ks[i] = rpTmp.getLowestSetBit();
110                rpTmp = rpTmp.clearBit(ks[i]);
111            }
112        }
113    
114        /**
115         * Creates a new {@code ECFieldF2m} with {@code 2^m} elements with
116         * a polynomial basis and the reduction polynomial based on {@code ks}.
117         * <p>
118         * The reduction polynomial must be either <i>trinomial</i> or
119         * <i>pentanomial</i>.
120         * </p>
121         * 
122         * @param m
123         *            the exponent {@code m} for the number of elements.
124         * @param ks
125         *            the base of the reduction polynomial with coefficients
126         *            given in descending order.     
127         * @throws IllegalArgumentException
128         *             if {@code m <= zero} or the reduction polynomial is not
129         *             valid.
130         * @since Android 1.0
131         */
132        public ECFieldF2m(int m, int[] ks) {
133            this.m = m;
134            if (this.m <= 0) {
135                throw new IllegalArgumentException(Messages.getString("security.75")); //$NON-NLS-1$
136            }
137            // Defensively copies array parameter
138            // to prevent subsequent modification.
139            // NPE as specified if ks is null
140            this.ks = new int[ks.length];
141            System.arraycopy(ks, 0, this.ks, 0, this.ks.length);
142    
143            // no need to check for null already
144            if (this.ks.length != TPB_MID_LEN && this.ks.length != PPB_MID_LEN) {
145                // must be either trinomial or pentanomial basis
146                throw new IllegalArgumentException(Messages.getString("security.78")); //$NON-NLS-1$
147            }
148            // trinomial basis:
149            // check that m > k >= 1, where k is ks[0]
150            // pentanomial basis:
151            // check that m > k3 > k2 > k1 >= 1
152            // and kx in descending order, where
153            // k3 is ks[0], k2 is ks[1], k1 is ks[2]
154            boolean checkFailed = false;
155            int prev = this.m;
156            for (int i=0; i<this.ks.length; i++) {
157                if (this.ks[i] < prev) {
158                    prev = this.ks[i];
159                    continue;
160                }
161                checkFailed = true;
162                break;
163            }
164            if (checkFailed || prev < 1) {
165                throw new IllegalArgumentException(Messages.getString("security.79")); //$NON-NLS-1$
166            }
167    
168            // Setup rp using ks:
169            // bits 0 and m always set
170            BigInteger rpTmp = BigInteger.ONE.setBit(this.m);
171            // set remaining bits according to ks
172            for (int i=0; i<this.ks.length; i++) {
173                rpTmp = rpTmp.setBit(this.ks[i]);
174            }
175            rp = rpTmp;
176        }
177    
178        /**
179         * Returns whether the specified object equals to this finite field.
180         * 
181         * @param obj
182         *            the object to compare to this finite field.
183         * @return {@code true} if the specified object is equal to this finite field,
184         *         otherwise {@code false}.
185         * @since Android 1.0
186         */
187        public boolean equals(Object obj) {
188            // object equals to itself
189            if (this == obj) {
190                return true;
191            }
192            if (obj instanceof ECFieldF2m) {
193                ECFieldF2m o = (ECFieldF2m)obj;
194                // check m
195                if (this.m == o.m) {
196                    // check rp
197                    if (this.rp == null) {
198                        if (o.rp == null) {
199                            // fields both with normal basis
200                            return true;
201                        }
202                    } else {
203                        // at least this field with polynomial basis
204                        // check that rp match
205                        // return this.rp.equals(o.rp);
206                        return Arrays.equals(this.ks, o.ks);
207                    }
208                }
209            }
210            return false;
211        }
212    
213        /**
214         * Returns the size of this finite field (in bits).
215         * 
216         * @return the size of this finite field (in bits).
217         * @since Android 1.0
218         */
219        public int getFieldSize() {
220            return m;
221        }
222    
223        /**
224         * Returns the exponent {@code m} for this finite field, with {@code 2^m} as
225         * the number of elements.
226         * 
227         * @return the exponent {@code m} for this finite field
228         * @since Android 1.0
229         */
230        public int getM() {
231            return m;
232        }
233    
234        /**
235         * Returns a copy of the integer array containing the order of the middle
236         * term(s) of the reduction polynomial for a polynomial basis.
237         * 
238         * @return a copy of the integer array containing the order of the middle
239         *         term(s) of the reduction polynomial for a polynomial basis or
240         *         {@code null} for a normal basis.
241         * @since Android 1.0
242         */
243        public int[] getMidTermsOfReductionPolynomial() {
244            // Defensively copies private array
245            // to prevent subsequent modification
246            // was: return ks == null ? null : (int[])ks.clone();
247            if (ks == null) {
248                return null;
249            } else {
250                int[] ret = new int[ks.length];
251                System.arraycopy(ks, 0, ret, 0, ret.length);
252                return ret;
253            }
254        }
255    
256        /**
257         * Returns the base of the reduction polynomial with the n-th bit
258         * corresponding to the n-th coefficient of the reduction polynomial for a
259         * polynomial basis.
260         * 
261         * @return the base of the reduction polynomial with the n-th bit
262         *         corresponding to the n-th coefficient of the reduction polynomial
263         *         for a polynomial basis or {@code null} for a normal basis.
264         * @since Android 1.0
265         */
266        public BigInteger getReductionPolynomial() {
267            return rp;
268        }
269    
270        /**
271         * Returns the hashcode value for this finite field.
272         * 
273         * @return the hashcode value for this finite field.
274         * @since Android 1.0
275         */
276        public int hashCode() {
277            return rp == null ? m : m + rp.hashCode();
278        }
279    }
