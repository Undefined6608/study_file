018    package java.security.spec;
019    
020    import java.math.BigInteger;
021    
022    /**
023     * The parameter specification for generating an RSA key pair. 
024     * 
025     * @since Android 1.0
026     */
027    public class RSAKeyGenParameterSpec implements AlgorithmParameterSpec {    
028    
029        /**
030         * The value of the public exponent {@code F0} = 3.
031         * 
032         * @since Android 1.0
033         */
034        public static final BigInteger F0 = BigInteger.valueOf(3L);
035    
036        /**
037         * The value of the public exponent {@code F4} = 65537.
038         * 
039         * @since Android 1.0
040         */
041        public static final BigInteger F4 = BigInteger.valueOf(65537L);
042    
043        // Key size
044        private final int keysize;
045        // Public Exponent
046        private final BigInteger publicExponent;
047    
048        /**
049         * Creates a new {@code RSAKeyGenParameterSpec} with the specified key size
050         * and public exponent.
051         * 
052         * @param keysize
053         *            the size of the modulus (number of bits).
054         * @param publicExponent
055         *            the value of the public exponent.
056         * @since Android 1.0
057         */
058        public RSAKeyGenParameterSpec(int keysize, BigInteger publicExponent) {
059            this.keysize = keysize;
060            this.publicExponent = publicExponent;
061        }
062    
063        /**
064         * Returns the size of the modulus (number of bits).
065         * 
066         * @return the size of the modulus (number of bits).
067         * @since Android 1.0
068         */
069        public int getKeysize() {
070            return keysize;
071        }
072    
073        /**
074         * Returns the value of the public exponent.
075         * 
076         * @return the value of the public exponent.
077         * @since Android 1.0
078         */
079        public BigInteger getPublicExponent() {
080            return publicExponent;
081        }
082    }
