018    package java.security.spec;
019    
020    import java.math.BigInteger;
021    
022    import org.apache.harmony.security.internal.nls.Messages;
023    
024    /**
025     * The parameter specification used with Elliptic Curve Cryptography (ECC). 
026     * 
027     * @since Android 1.0
028     */
029    public class ECParameterSpec implements AlgorithmParameterSpec {
030        // Elliptic curve for which this is parameter
031        private final EllipticCurve curve;
032        // Distinguished point on the elliptic curve called generator or base point
033        private final ECPoint generator;
034        // Order of the generator
035        private final BigInteger order;
036        // Cofactor
037        private final int cofactor;
038    
039        /**
040         * Creates a new {@code ECParameterSpec} with the specified elliptic curve,
041         * the base point, the order of the generator (or base point) and the
042         * co-factor.
043         * 
044         * @param curve
045         *            the elliptic curve.
046         * @param generator
047         *            the generator (or base point).
048         * @param order
049         *            the order of the generator.
050         * @param cofactor
051         *            the co-factor.
052         * @throws IllegalArgumentException
053         *             if {@code order <= zero} or {@code cofactor <= zero}.
054         * @since Android 1.0
055         */
056        public ECParameterSpec(EllipticCurve curve, ECPoint generator,
057                BigInteger order, int cofactor) {
058            this.curve = curve;
059            this.generator = generator;
060            this.order = order;
061            this.cofactor = cofactor;
062            // throw NullPointerException if curve, generator or order is null
063            if (this.curve == null) {
064                throw new NullPointerException(Messages.getString("security.83", "curve")); //$NON-NLS-1$ //$NON-NLS-2$
065            }
066            if (this.generator == null) {
067                throw new NullPointerException(Messages.getString("security.83", "generator")); //$NON-NLS-1$ //$NON-NLS-2$
068            }
069            if (this.order == null) {
070                throw new NullPointerException(Messages.getString("security.83", "order")); //$NON-NLS-1$ //$NON-NLS-2$
071            }
072            // throw IllegalArgumentException if order or cofactor is not positive
073            if (!(this.order.compareTo(BigInteger.ZERO) > 0)) {
074                throw new
075                IllegalArgumentException(Messages.getString("security.86", "order")); //$NON-NLS-1$ //$NON-NLS-2$
076            }
077            if (!(this.cofactor > 0)) {
078                throw new
079                IllegalArgumentException(Messages.getString("security.86", "cofactor")); //$NON-NLS-1$ //$NON-NLS-2$
080            }
081        }
082    
083        /**
084         * Returns the {@code cofactor}.
085         * 
086         * @return the {@code cofactor}.
087         * @since Android 1.0
088         */
089        public int getCofactor() {
090            return cofactor;
091        }
092    
093        /**
094         * Returns the elliptic curve.
095         * 
096         * @return the elliptic curve.
097         * @since Android 1.0
098         */
099        public EllipticCurve getCurve() {
100            return curve;
101        }
102    
103        /**
104         * Returns the generator (or base point).
105         * 
106         * @return the generator (or base point).
107         * @since Android 1.0
108         */
109        public ECPoint getGenerator() {
110            return generator;
111        }
112    
113        /**
114         * Returns the order of the generator.
115         * 
116         * @return the order of the generator.
117         * @since Android 1.0
118         */
119        public BigInteger getOrder() {
120            return order;
121        }
122    }
