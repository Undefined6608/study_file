200        /**
201         * Insert the given {@code Provider} at the specified {@code position}. The
202         * positions define the preference order in which providers are searched for
203         * requested algorithms.
204         * <p>
205         * If a {@code SecurityManager} is installed, code calling this method needs
206         * the {@code SecurityPermission} {@code insertProvider.NAME} (where NAME is
207         * the provider name) to be granted, otherwise a {@code SecurityException}
208         * will be thrown.
209         * </p>
210         * 
211         * @param provider
212         *            the provider to insert.
213         * @param position
214         *            the position (starting from 1).
215         * @return the actual position or {@code -1} if the given {@code provider}
216         *         was already in the list. The actual position may be different
217         *         from the desired position.
218         * @throws SecurityException
219         *             if a {@code SecurityManager} is installed and the caller does
220         *             not have permission to invoke this method.
221         * @since Android 1.0
222         */
223        public static synchronized int insertProviderAt(Provider provider,
224                int position) {
225            // check security access; check that provider is not already
226            // installed, else return -1; if (position <1) or (position > max
227            // position) position = max position + 1; insert provider, shift up
228            // one position for next providers; Note: The position is 1-based
229            SecurityManager sm = System.getSecurityManager();
230            if (sm != null) {
231                sm.checkSecurityAccess("insertProvider." + provider.getName()); //$NON-NLS-1$
232            }
233            if (getProvider(provider.getName()) != null) {
234                return -1;
235            }
236            int result = Services.insertProviderAt(provider, position);
237            renumProviders();
238            return result;
239        }

264        /**
265         * Removes the {@code Provider} with the specified name form the collection
266         * of providers. If the the {@code Provider} with the specified name is
267         * removed, all provider at a greater position are shifted down one
268         * position.
269         * <p>
270         * Returns silently if {@code name} is {@code null} or no provider with the
271         * specified name is installed.
272         * </p>
273         * <p>
274         * If a {@code SecurityManager} is installed, code calling this method needs
275         * the {@code SecurityPermission} {@code removeProvider.NAME} (where NAME is
276         * the provider name) to be granted, otherwise a {@code SecurityException}
277         * will be thrown.
278         * </p>
279         * 
280         * @param name
281         *            the name of the provider to remove.
282         * @throws SecurityException
283         *             if a {@code SecurityManager} is installed and the caller does
284         *             not have permission to invoke this method.
285         * @since Android 1.0
286         */
287        public static synchronized void removeProvider(String name) {
288            // It is not clear from spec.:
289            // 1. if name is null, should we checkSecurityAccess or not? 
290            //    throw SecurityException or not?
291            // 2. as 1 but provider is not installed
292            // 3. behavior if name is empty string?
293    
294            Provider p;
295            if ((name == null) || (name.length() == 0)) {
296                return;
297            }
298            p = getProvider(name);
299            if (p == null) {
300                return;
301            }
302            SecurityManager sm = System.getSecurityManager();
303            if (sm != null) {
304                sm.checkSecurityAccess("removeProvider." + name); //$NON-NLS-1$
305            }
306            Services.removeProvider(p.getProviderNumber());
307            renumProviders();
308            p.setProviderNumber(-1);
309        }

336        /**
337         * Returns the array of providers which meet the user supplied string
338         * filter. The specified filter must be supplied in one of two formats:
339         * <nl>
340         * <li> CRYPTO_SERVICE_NAME.ALGORITHM_OR_TYPE 
341         * <p>
342         * (for example: "MessageDigest.SHA") 
343         * <li> CRYPTO_SERVICE_NAME.ALGORITHM_OR_TYPE
344         * ATTR_NAME:ATTR_VALUE
345         * <p>
346         * (for example: "Signature.MD2withRSA KeySize:512")
347         * </nl>
348         * 
349         * @param filter
350         *            case-insensitive filter.
351         * @return the providers which meet the user supplied string filter {@code
352         *         filter}. A {@code null} value signifies that none of the
353         *         installed providers meets the filter specification.
354         * @throws InvalidParameterException
355         *             if an unusable filter is supplied.
356         * @throws NullPointerException
357         *             if {@code filter} is {@code null}.
358         * @since Android 1.0
359         */
360        public static Provider[] getProviders(String filter) {
361            if (filter == null) {
362                throw new NullPointerException(Messages.getString("security.2A")); //$NON-NLS-1$
363            }
364            if (filter.length() == 0) {
365                throw new InvalidParameterException(
366                        Messages.getString("security.2B")); //$NON-NLS-1$
367            }
368            HashMap<String, String> hm = new HashMap<String, String>();
369            int i = filter.indexOf(":"); //$NON-NLS-1$
370            if ((i == filter.length() - 1) || (i == 0)) {
371                throw new InvalidParameterException(
372                        Messages.getString("security.2B")); //$NON-NLS-1$
373            }
374            if (i < 1) {
375                hm.put(filter, ""); //$NON-NLS-1$
376            } else {
377                hm.put(filter.substring(0, i), filter.substring(i + 1));
378            }
379            return getProviders(hm);
380        }

382        /**
383         * Returns the array of providers which meet the user supplied set of
384         * filters. The filter must be supplied in one of two formats:
385         * <nl>
386         * <li> CRYPTO_SERVICE_NAME.ALGORITHM_OR_TYPE
387         * <p>
388         * for example: "MessageDigest.SHA" The value associated with the key must
389         * be an empty string. <li> CRYPTO_SERVICE_NAME.ALGORITHM_OR_TYPE
390         * ATTR_NAME:ATTR_VALUE
391         * <p>
392         * for example: "Signature.MD2withRSA KeySize:512" where "KeySize:512" is
393         * the value of the filter map entry.
394         * </nl>
395         * 
396         * @param filter
397         *            case-insensitive filter.
398         * @return the providers which meet the user supplied string filter {@code
399         *         filter}. A {@code null} value signifies that none of the
400         *         installed providers meets the filter specification.
401         * @throws InvalidParameterException
402         *             if an unusable filter is supplied.
403         * @throws NullPointerException
404         *             if {@code filter} is {@code null}.
405         * @since Android 1.0
406         */
407        public static synchronized Provider[] getProviders(Map<String,String> filter) {
408            if (filter == null) {
409                throw new NullPointerException(Messages.getString("security.2A")); //$NON-NLS-1$
410            }
411            if (filter.isEmpty()) {
412                return null;
413            }
414            java.util.List<Provider> result = Services.getProvidersList();
415            Set keys = filter.entrySet();
416            Map.Entry entry;
417            for (Iterator it = keys.iterator(); it.hasNext();) {
418                entry = (Map.Entry) it.next();
419                String key = (String) entry.getKey();
420                String val = (String) entry.getValue();
421                String attribute = null;
422                int i = key.indexOf(" "); //$NON-NLS-1$
423                int j = key.indexOf("."); //$NON-NLS-1$
424                if (j == -1) {
425                    throw new InvalidParameterException(
426                            Messages.getString("security.2B")); //$NON-NLS-1$
427                }
428                if (i == -1) { // <crypto_service>.<algorithm_or_type>
429                    if (val.length() != 0) {
430                        throw new InvalidParameterException(
431                                Messages.getString("security.2B")); //$NON-NLS-1$
432                    }
433                } else { // <crypto_service>.<algorithm_or_type> <attribute_name>
434                    if (val.length() == 0) {
435                        throw new InvalidParameterException(
436                                Messages.getString("security.2B")); //$NON-NLS-1$
437                    }
438                    attribute = key.substring(i + 1);
439                    if (attribute.trim().length() == 0) {
440                        throw new InvalidParameterException(
441                                Messages.getString("security.2B")); //$NON-NLS-1$
442                    }
443                    key = key.substring(0, i);
444                }
445                String serv = key.substring(0, j);
446                String alg = key.substring(j + 1);
447                if (serv.length() == 0 || alg.length() == 0) {
448                    throw new InvalidParameterException(
449                            Messages.getString("security.2B")); //$NON-NLS-1$
450                }
451                Provider p;
452                for (int k = 0; k < result.size(); k++) {
453                    try {
454                        p = (Provider) result.get(k);
455                    } catch (IndexOutOfBoundsException e) {
456                        break;
457                    }
458                    if (!p.implementsAlg(serv, alg, attribute, val)) {
459                        result.remove(p);
460                        k--;
461                    }
462                }
463            }
464            if (result.size() > 0) {
465                return result.toArray(new Provider[result.size()]);
466            } else {
467                return null;
468            }
469        }

472         * Returns the value of the security property named by the argument.
473         * <p>
474         * If a {@code SecurityManager} is installed, code calling this method needs

476         * specified {@code key}) to be granted, otherwise a {@code

479         * 
480         * @param key
481         *            the name of the requested security property.

484         *             if a {@code SecurityManager} is installed and the caller does
485         *             not have permission to invoke this method.
486         * @since Android 1.0
487         */
488        public static String getProperty(String key) {
489            if (key == null) {
490                throw new NullPointerException(Messages.getString("security.2C")); //$NON-NLS-1$
491            }
492            SecurityManager sm = System.getSecurityManager();
493            if (sm != null) {
494                sm.checkSecurityAccess("getProperty." + key); //$NON-NLS-1$
495            }
496            return secprops.getProperty(key);
497        }

525        /**
526         * Returns a {@code Set} of all registered algorithms for the specified
527         * cryptographic service. {@code "Signature"}, {@code "Cipher"} and {@code
528         * "KeyStore"} are examples for such kind of services.
529         * 
530         * @param serviceName
531         *            the case-insensitive name of the service.
532         * @return a {@code Set} of all registered algorithms for the specified
533         *         cryptographic service, or an empty {@code Set} if {@code
534         *         serviceName} is {@code null} or if no registered provider
535         *         provides the requested service.
536         * @since Android 1.0
537         */
538        public static Set<String> getAlgorithms(String serviceName) {
539            Set<String> result = new HashSet<String>();
540            Provider[] p = getProviders();
541            for (int i = 0; i < p.length; i++) {
542                for (Iterator it = p[i].getServices().iterator(); it.hasNext();) {
543                    Provider.Service s = (Provider.Service) it.next();
544                    if (s.getType().equalsIgnoreCase(serviceName)) {
545                        result.add(s.getAlgorithm());
546                    }
547                }
548            }
549            return result;
550        }

552        /**
553         * 
554         * Update sequence numbers of all providers.
555         *  
556         */
557        private static void renumProviders() {
558            Provider[] p = Services.getProviders();
559            for (int i = 0; i < p.length; i++) {
560                p[i].setProviderNumber(i + 1);
561            }
562        }
