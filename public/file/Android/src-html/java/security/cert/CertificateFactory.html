019

022    import java.security.NoSuchProviderException;
023    import java.security.Provider;
024    import java.security.Security;
025    import java.util.Collection;
026    import java.util.Iterator;

033    /**
034     * This class implements the functionality of a certificate factory algorithm,
035     * relying on parsing a stream of bytes.
036     * <p>
037     * It defines methods for parsing certificate chains (certificate paths) and
038     * <i>Certificate Revocation Lists</i> (CRLs).
039     * </p>
040     * 
041     * @since Android 1.0
042     */
043    public class CertificateFactory {
044    
045        // Store CertificateFactory service name
046        private static final String SERVICE = "CertificateFactory"; //$NON-NLS-1$
047    
048        // Used to access common engine functionality
049        private static Engine engine = new Engine(SERVICE);
050    
051        // Store used provider
052        private final Provider provider;
053    
054        // Store used CertificateFactorySpi implementation
055        private final CertificateFactorySpi spiImpl;
056    
057        // Store used type
058        private final String type;
059    
060        /**
061         * Creates a new {@code CertificateFactory} instance.
062         * 
063         * @param certFacSpi
064         *            the implementation delegate.
065         * @param provider
066         *            the associated provider.
067         * @param type
068         *            the certificate type.
069         * @since Android 1.0
070         */
071        protected CertificateFactory(CertificateFactorySpi certFacSpi,
072                Provider provider, String type) {
073            this.provider = provider;
074            this.type = type;
075            this.spiImpl = certFacSpi;
076        }
077    
078        /**
079         * Creates a new {@code CertificateFactory} instance that provides the
080         * requested certificate type.
081         * 
082         * @param type
083         *            the certificate type.
084         * @return the new {@code CertificateFactory} instance.
085         * @throws CertificateException
086         *             if the specified certificate type is not available at any
087         *             installed provider.
088         * @throws NullPointerException
089         *             if {@code type} is {@code null}.
090         * @since Android 1.0
091         */
092        public static final CertificateFactory getInstance(String type)
093                throws CertificateException {
094            if (type == null) {
095                throw new NullPointerException(Messages.getString("security.07")); //$NON-NLS-1$
096            }
097            try {
098                synchronized (engine) {
099                    engine.getInstance(type, null);
100                    return new CertificateFactory((CertificateFactorySpi) engine.spi,
101                            engine.provider, type);
102                }
103            } catch (NoSuchAlgorithmException e) {
104                throw new CertificateException(e);
105            }
106        }
107    
108        /**
109         * Creates a new {@code CertificateFactory} instance from the specified
110         * provider that provides the requested certificate type.
111         * 
112         * @param type
113         *            the certificate type.
114         * @param provider
115         *            the name of the provider providing certificates of the
116         *            specified type.
117         * @return the new {@code CertificateFactory} instance.
118         * @throws CertificateException
119         *             if the specified certificate type is not available by the
120         *             specified provider.
121         * @throws NoSuchProviderException
122         *             if no provider with the specified name can be found.
123         * @throws IllegalArgumentException
124         *             if the specified provider name is {@code null} or empty.
125         * @throws NullPointerException
126         *             it {@code type} is {@code null}.
127         * @since Android 1.0
128         */
129        public static final CertificateFactory getInstance(String type,
130                String provider) throws CertificateException,
131                NoSuchProviderException {
132            if ((provider == null) || (provider.length() == 0)) {
133                throw new IllegalArgumentException(Messages.getString("security.02")); //$NON-NLS-1$
134            }
135            Provider impProvider = Security.getProvider(provider);
136            if (impProvider == null) {
137                throw new NoSuchProviderException(provider);
138            }
139            return getInstance(type, impProvider);
140        }
141    
142        /**
143         * Creates a new {@code CertificateFactory} instance from the specified
144         * provider that provides the requested certificate type.
145         * 
146         * @param type
147         *            the certificate type.
148         * @param provider
149         *            the name of the provider providing certificates of the
150         *            specified type.
151         * @return the new {@code CertificateFactory} instance.
152         * @throws CertificateException
153         *             if the specified certificate type is not available at the
154         *             specified provider.
155         * @throws IllegalArgumentException
156         *             if the specified provider is {@code null}.
157         * @throws NullPointerException
158         *             is {@code type} is {@code null}.
159         * @since Android 1.0
160         */
161        public static final CertificateFactory getInstance(String type,
162                Provider provider) throws CertificateException {
163            if (provider == null) {
164                throw new IllegalArgumentException(Messages.getString("security.04")); //$NON-NLS-1$
165            }
166            if (type == null) {
167                throw new NullPointerException(Messages.getString("security.07")); //$NON-NLS-1$
168            }
169            try {
170                synchronized (engine) {
171                    engine.getInstance(type, provider, null);
172                    return new CertificateFactory((CertificateFactorySpi) engine.spi,
173                            provider, type);
174                }
175            } catch (NoSuchAlgorithmException e) {
176                throw new CertificateException(e.getMessage());
177            }
178        }
179    
180        /**
181         * Returns the {@code Provider} of the certificate factory represented by
182         * the certificate.
183         * 
184         * @return the provider of this certificate factory.
185         * @since Android 1.0
186         */
187        public final Provider getProvider() {
188            return provider;
189        }
190    
191        /**
192         * Returns the Certificate type.
193         * 
194         * @return type of certificate being used.
195         * @since Android 1.0
196         */
197        public final String getType() {
198            return type;
199        }
200    
201        /**
202         * Generates and initializes a {@code Certificate} from the provided input
203         * stream.
204         * 
205         * @param inStream
206         *            the stream from where data is read to create the {@code
207         *            Certificate}.
208         * @return an initialized Certificate.
209         * @throws CertificateException
210         *             if parsing problems are detected.
211         * @since Android 1.0
212         */
213        public final Certificate generateCertificate(InputStream inStream)
214                throws CertificateException {
215            return spiImpl.engineGenerateCertificate(inStream);
216        }
217    
218        /**
219         * Returns an {@code Iterator} over the supported {@code CertPath} encodings
220         * (as Strings). The first element is the default encoding scheme to apply.
221         * 
222         * @return an iterator over supported {@link CertPath} encodings (as
223         *         Strings).
224         * @since Android 1.0
225         */
226        public final Iterator<String> getCertPathEncodings() {
227            return spiImpl.engineGetCertPathEncodings();
228        }
229    
230        /**
231         * Generates a {@code CertPath} (a certificate chain) from the provided
232         * {@code InputStream}. The default encoding scheme is applied.
233         * 
234         * @param inStream
235         *            {@code InputStream} with encoded data.
236         * @return a {@code CertPath} initialized from the provided data.
237         * @throws CertificateException
238         *             if parsing problems are detected.
239         * @since Android 1.0
240         */
241        public final CertPath generateCertPath(InputStream inStream)
242                throws CertificateException {
243            Iterator<String> it = getCertPathEncodings();
244            if (!it.hasNext()) {
245                throw new CertificateException(Messages.getString("security.74")); //$NON-NLS-1$
246            }
247            return spiImpl.engineGenerateCertPath(inStream, it.next());
248        }
249    
250        /**
251         * Generates a {@code CertPath} (a certificate chain) from the provided
252         * {@code InputStream} and the specified encoding scheme.
253         * 
254         * @param inStream
255         *            {@code InputStream} containing certificate path data in
256         *            specified encoding.
257         * @param encoding
258         *            encoding of the data in the input stream.
259         * @return a {@code CertPath} initialized from the provided data.
260         * @throws CertificateException
261         *             if parsing problems are detected.
262         * @throws UnsupportedOperationException
263         *             if the provider does not implement this method.
264         * @since Android 1.0
265         */
266        public final CertPath generateCertPath(InputStream inStream, String encoding)
267                throws CertificateException {
268            return spiImpl.engineGenerateCertPath(inStream, encoding);
269        }
270    
271        /**
272         * Generates a {@code CertPath} from the provided list of certificates. The
273         * encoding is the default encoding.
274         * 
275         * @param certificates
276         *            the list containing certificates in a format supported by the
277         *            {@code CertificateFactory}.
278         * @return a {@code CertPath} initialized from the provided data.
279         * @throws CertificateException
280         *             if parsing problems are detected.
281         * @throws UnsupportedOperationException
282         *             if the provider does not implement this method.
283         * @since Android 1.0
284         */
285        public final CertPath generateCertPath(List<? extends Certificate> certificates)
286                throws CertificateException {
287            return spiImpl.engineGenerateCertPath(certificates);
288        }
289    
290        /**
291         * Generates and initializes a collection of (unrelated) certificates from
292         * the provided input stream.
293         * 
294         * @param inStream
295         *            the stream from which the data is read to create the
296         *            collection.
297         * @return an initialized collection of certificates.
298         * @throws CertificateException
299         *             if parsing problems are detected.
300         * @since Android 1.0
301         */
302        public final Collection<? extends Certificate> generateCertificates(InputStream inStream)
303                throws CertificateException {
304            return spiImpl.engineGenerateCertificates(inStream);
305        }
306    
307        /**
308         * Generates and initializes a <i>Certificate Revocation List</i> (CRL) from
309         * the provided input stream.
310         * 
311         * @param inStream
312         *            the stream from where data is read to create the CRL.
313         * @return an initialized CRL.
314         * @exception CRLException
315         *                if parsing problems are detected.
316         * @since Android 1.0
317         */
318        public final CRL generateCRL(InputStream inStream) throws CRLException {
319            return spiImpl.engineGenerateCRL(inStream);
320        }
321    
322        /**
323         * Generates and initializes a collection of <i>Certificate Revocation
324         * List</i> (CRL) from the provided input stream.
325         * 
326         * @param inStream
327         *            the stream from which the data is read to create the CRLs.
328         * @return an initialized collection of CRLs.
329         * @exception CRLException
330         *                if parsing problems are detected.
331         * @since Android 1.0
332         */
333        public final Collection<? extends CRL> generateCRLs(InputStream inStream)
334                throws CRLException {
335            return spiImpl.engineGenerateCRLs(inStream);
336        }
