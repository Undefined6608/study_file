018    package java.security.cert;
019    
020    import java.io.ByteArrayInputStream;
021    import java.io.NotSerializableException;
022    import java.io.ObjectStreamException;
023    import java.io.ObjectStreamField;
024    import java.io.Serializable;
025    import java.security.InvalidKeyException;
026    import java.security.NoSuchAlgorithmException;
027    import java.security.NoSuchProviderException;
028    import java.security.PublicKey;
029    import java.security.SignatureException;
030    import java.util.Arrays;
031    
032    import org.apache.harmony.security.internal.nls.Messages;
033    
034    /**
035     * Abstract class to represent identity certificates. It represents a way to
036     * verify the binding of a Principal and its public key. Examples are X.509,
037     * PGP, and SDSI.
038     * 
039     * @since Android 1.0
040     */
041    public abstract class Certificate implements Serializable {
042    
043        private static final long serialVersionUID = -3585440601605666277L;
044    
045        // The standard name of the certificate type
046        private final String type;
047    
048        /**
049         * Creates a new {@code Certificate} with the specified type.
050         * 
051         * @param type
052         *        the certificate type.
053         * @since Android 1.0
054         */
055        protected Certificate(String type) {
056            this.type = type;
057        }
058    
059        /**
060         * Returns the certificate type.
061         * 
062         * @return the certificate type.
063         * @since Android 1.0
064         */
065        public final String getType() {
066            return type;
067        }
068    
069        /**
070         * Compares the argument to the certificate, and returns {@code true} if they
071         * represent the <em>same</em> object using a class specific comparison. The
072         * implementation in Object returns {@code true} only if the argument is the
073         * exact same object as the callee (==).
074         * 
075         * @param other
076         *            the object to compare with this object.
077         * @return {@code true} if the object is the same as this object, {@code
078         *         false} if it is different from this object.
079         * @see #hashCode
080         * @since Android 1.0
081         */
082        public boolean equals(Object other) {
083            // obj equal to itself
084            if (this == other) {
085                return true;
086            }
087            if (other instanceof Certificate) {
088                try {
089                    // check that encoded forms match
090                    return Arrays.equals(this.getEncoded(),
091                            ((Certificate)other).getEncoded());
092                } catch (CertificateEncodingException e) {
093                    throw new RuntimeException(e);
094                }
095            }
096            return false;
097        }
098    
099        /**
100         * Returns an integer hash code for the certificate. Any two objects which
101         * return {@code true} when passed to {@code equals} must return the same
102         * value for this method.
103         * 
104         * @return the certificate's hash
105         * @see #equals
106         * @since Android 1.0
107         */
108        public int hashCode() {
109            try {
110                byte[] encoded = getEncoded();
111                int hash = 0;
112                for (int i=0; i<encoded.length; i++) {
113                    hash += i*encoded[i];
114                }
115                return hash;
116            } catch (CertificateEncodingException e) {
117                throw new RuntimeException(e);
118            }
119        }
120    
121        /**
122         * Returns the encoded representation for this certificate.
123         * 
124         * @return the encoded representation for this certificate.
125         * @throws CertificateEncodingException
126         *             if the encoding fails.
127         * @since Android 1.0
128         */
129        public abstract byte[] getEncoded() throws CertificateEncodingException;
130    
131        /**
132         * Verifies that this certificate was signed with the given public key.
133         * 
134         * @param key
135         *            PublicKey public key for which verification should be
136         *            performed.
137         * @throws CertificateException
138         *             if encoding errors are detected.
139         * @throws NoSuchAlgorithmException
140         *             if an unsupported algorithm is detected.
141         * @throws InvalidKeyException
142         *             if an invalid key is detected.
143         * @throws NoSuchProviderException
144         *             if there is no default provider.
145         * @throws SignatureException
146         *             if signature errors are detected.
147         * @since Android 1.0
148         */
149        public abstract void verify(PublicKey key)
150            throws CertificateException,
151                   NoSuchAlgorithmException,
152                   InvalidKeyException,
153                   NoSuchProviderException,
154                   SignatureException;
155    
156        /**
157         * Verifies that this certificate was signed with the given public key. It
158         * Uses the signature algorithm given by the provider.
159         * 
160         * @param key
161         *            PublicKey public key for which verification should be
162         *            performed.
163         * @param sigProvider
164         *            String the name of the signature provider.
165         * @exception CertificateException
166         *                if encoding errors are detected.
167         * @exception NoSuchAlgorithmException
168         *                if an unsupported algorithm is detected.
169         * @exception InvalidKeyException
170         *                if an invalid key is detected.
171         * @exception NoSuchProviderException
172         *                if the specified provider does not exists.
173         * @exception SignatureException
174         *                if signature errors are detected.
175         * @since Android 1.0
176         */
177        public abstract void verify(PublicKey key, String sigProvider)
178            throws CertificateException,
179                   NoSuchAlgorithmException,
180                   InvalidKeyException,
181                   NoSuchProviderException,
182                   SignatureException;
183    
184        /**
185         * Returns a string containing a concise, human-readable description of the
186         * certificate.
187         * 
188         * @return a printable representation for the certificate.
189         * @since Android 1.0
190         */
191        public abstract String toString();
192    
193        /**
194         * Returns the public key corresponding to this certificate.
195         * 
196         * @return the public key corresponding to this certificate.
197         * @since Android 1.0
198         */
199        public abstract PublicKey getPublicKey();
200    
201        /**
202         * Returns an alternate object to be serialized.
203         * 
204         * @return the object to serialize.
205         * @throws ObjectStreamException
206         *             if the creation of the alternate object fails.
207         * @since Android 1.0
208         */
209        protected Object writeReplace() throws ObjectStreamException {
210            try {
211                return new CertificateRep(getType(), getEncoded());
212            } catch (CertificateEncodingException e) {  
213                throw new NotSerializableException (
214                        Messages.getString("security.66", e)); //$NON-NLS-1$
215            }
216        }
217    
218        /**
219         * The alternate {@code Serializable} class to be used for serialization and
220         * deserialization of {@code Certificate} objects. 
221         * 
222         * @since Android 1.0
223         */
224        protected static class CertificateRep implements Serializable {
225    
226            private static final long serialVersionUID = -8563758940495660020L;
227            // The standard name of the certificate type
228            private final String type;
229            // The certificate data
230            private final byte[] data;
231    
232            // Force default serialization to use writeUnshared/readUnshared
233            // for the certificate data
234            private static final ObjectStreamField[] serialPersistentFields = {
235                 new ObjectStreamField("type", String.class), //$NON-NLS-1$
236                 new ObjectStreamField("data", byte[].class, true) //$NON-NLS-1$
237            };
238    
239            /**
240             * Creates a new {@code CertificateRep} instance with the specified
241             * certificate type and encoded data.
242             * 
243             * @param type
244             *            the certificate type.
245             * @param data
246             *            the encoded data.
247             * @since Android 1.0
248             */
249            protected CertificateRep(String type, byte[] data) {
250                this.type = type;
251                this.data = data;
252            }
253    
254            /**
255             * Deserializes a {@code Certificate} from a serialized {@code
256             * CertificateRep} object.
257             * 
258             * @return the deserialized {@code Certificate}.
259             * @throws ObjectStreamException
260             *             if deserialization fails.
261             * @since Android 1.0
262             */
263            protected Object readResolve() throws ObjectStreamException {
264                try {
265                    CertificateFactory cf = CertificateFactory.getInstance(type);
266                    return cf.generateCertificate(new ByteArrayInputStream(data));
267                } catch (Throwable t) {
268                    throw new NotSerializableException(
269                            Messages.getString("security.68", t)); //$NON-NLS-1$
270                }
271            }
272        }
273    }
