018    package java.security.cert;
019    
020    import java.io.ByteArrayInputStream;
021    import java.io.NotSerializableException;
022    import java.io.ObjectStreamException;
023    import java.io.ObjectStreamField;
024    import java.io.Serializable;
025    import java.util.Iterator;
026    import java.util.List;
027    
028    import org.apache.harmony.security.internal.nls.Messages;
029    
030    /**
031     * An immutable certificate path that can be validated. All certificates in the
032     * path are of the same type (i.e., X509).
033     * <p>
034     * A {@code CertPath} can be represented as a byte array in at least one
035     * supported encoding scheme (i.e. PkiPath or PKCS7) when serialized.
036     * </p>
037     * <p>
038     * When a {@code List} of the certificates is obtained it must be immutable.
039     * </p>
040     * <p>
041     * A {@code CertPath} must be thread-safe without requiring coordinated access.
042     * </p>
043     * 
044     * @see Certificate
045     * @since Android 1.0
046     */
047    public abstract class CertPath implements Serializable {
048    
049        private static final long serialVersionUID = 6068470306649138683L;
050        // Standard name of the type of certificates in this path
051        private final String type;
052    
053        /**
054         * Creates a new {@code CertPath} instance for the specified certificate
055         * type.
056         * 
057         * @param type
058         *            the certificate type.
059         * @since Android 1.0
060         */
061        protected CertPath(String type) {
062            this.type = type;
063        }
064    
065        /**
066         * Returns the type of {@code Certificate} in this instance.
067         * 
068         * @return the certificate type.
069         * @since Android 1.0
070         */
071        public String getType() {
072            return type;
073        }
074    
075        /**
076         * Returns {@code true} if {@code Certificate}s in the list are the same
077         * type and the lists are equal (and by implication the certificates
078         * contained within are the same).
079         * 
080         * @param other
081         *            {@code CertPath} to be compared for equality.
082         * @return {@code true} if the object are equal, {@code false} otherwise.
083         * @since Android 1.0
084         */
085        @Override
086        public boolean equals(Object other) {
087            if (this == other) {
088                return true;
089            }
090            if (other instanceof CertPath) {
091                CertPath o = (CertPath)other;
092                if (getType().equals(o.getType())) {
093                    if (getCertificates().equals(o.getCertificates())) {
094                        return true;
095                    }
096                }
097            }
098            return false;
099        }
100    
101        /**
102         * Overrides {@code Object.hashCode()}. The function is defined as follows: <br>
103         * {@code hashCode = 31 * path.getType().hashCode() +
104         * path.getCertificates().hashCode();}</br> </p>
105         * 
106         * @return the hash code for this instance.
107         * @since Android 1.0
108         */
109        @Override
110        public int hashCode() {
111            int hash = getType().hashCode();
112            hash = hash*31 + getCertificates().hashCode();
113            return hash;
114        }
115    
116        /**
117         * Returns a {@code String} representation of this {@code CertPath}
118         * instance.
119         * 
120         * @return a string representation of this instance.
121         * @since Android 1.0
122         */
123        public String toString() {
124            StringBuffer sb = new StringBuffer(getType());
125            sb.append(" Cert Path, len="); //$NON-NLS-1$
126            sb.append(getCertificates().size());
127            sb.append(": [\n"); //$NON-NLS-1$
128            int n=1;
129            // BEGIN android-changed
130            for (Iterator<? extends Certificate> i=getCertificates().iterator();
131                          i.hasNext(); n++) {
132                sb.append("---------------certificate "); //$NON-NLS-1$
133                sb.append(n);
134                sb.append("---------------\n"); //$NON-NLS-1$
135                sb.append(((Certificate)i.next()).toString());
136            }
137            // END android-changed
138            sb.append("\n]"); //$NON-NLS-1$
139            return sb.toString();
140        }
141    
142        /**
143         * Returns an immutable List of the {@code Certificate}s contained
144         * in the {@code CertPath}.
145         * 
146         * @return a list of {@code Certificate}s in the {@code CertPath}.
147         * @since Android 1.0
148         */
149        public abstract List<? extends Certificate> getCertificates();
150    
151        /**
152         * Returns an encoding of the {@code CertPath} using the default encoding.
153         * 
154         * @return default encoding of the {@code CertPath}.
155         * @throws CertificateEncodingException
156         *             if the encoding fails.
157         * @since Android 1.0
158         */
159        public abstract byte[] getEncoded()
160            throws CertificateEncodingException;
161    
162        /**
163         * Returns an encoding of the {@code CertPath} using the specified encoding.
164         * 
165         * @param encoding
166         *            encoding that should be generated.
167         * @return default encoding of the {@code CertPath}.
168         * @throws CertificateEncodingException
169         *             if the encoding fails.
170         * @since Android 1.0
171         */
172        public abstract byte[] getEncoded(String encoding)
173            throws CertificateEncodingException;
174    
175        /**
176         * Returns an {@code Iterator} over the supported encodings for a
177         * representation of the certificate path.
178         * 
179         * @return {@code Iterator} over supported encodings (as {@code String}s).
180         * @since Android 1.0
181         */
182        public abstract Iterator<String> getEncodings();
183    
184        /**
185         * Returns an alternate object to be serialized.
186         * 
187         * @return an alternate object to be serialized.
188         * @throws ObjectStreamException
189         *             if the creation of the alternate object fails.
190         * @since Android 1.0
191         */
192        protected Object writeReplace() throws ObjectStreamException {
193            try {
194                return new CertPathRep(getType(), getEncoded());
195            } catch (CertificateEncodingException e) {
196                throw new NotSerializableException (
197                        Messages.getString("security.66", e)); //$NON-NLS-1$
198            }
199        }
200    
201        /**
202         * The alternate {@code Serializable} class to be used for serialization and
203         * deserialization on {@code CertPath} objects.
204         * 
205         * @since Android 1.0
206         */
207        protected static class CertPathRep implements Serializable {
208    
209            private static final long serialVersionUID = 3015633072427920915L;
210            // Standard name of the type of certificates in this path
211            private final String type;
212            // cert path data
213            private final byte[] data;
214    
215            // Force default serialization to use writeUnshared/readUnshared
216            // for cert path data
217            private static final ObjectStreamField[] serialPersistentFields = {
218                 new ObjectStreamField("type", String.class), //$NON-NLS-1$
219                 new ObjectStreamField("data", byte[].class, true) //$NON-NLS-1$
220            };
221    
222            /**
223             * Creates a new {@code CertPathRep} instance with the specified type
224             * and encoded data.
225             * 
226             * @param type
227             *            the certificate type.
228             * @param data
229             *            the encoded data.
230             * @since Android 1.0
231             */
232            protected CertPathRep(String type, byte[] data) {
233                this.type = type;
234                this.data = data;
235            }
236    
237            /**
238             * Deserializes a {@code CertPath} from a serialized {@code CertPathRep}
239             * object.
240             * 
241             * @return the deserialized {@code CertPath}.
242             * @throws ObjectStreamException
243             *             if deserialization fails.
244             * @since Android 1.0
245             */
246            protected Object readResolve() throws ObjectStreamException {
247                try {
248                    CertificateFactory cf = CertificateFactory.getInstance(type);
249                    return cf.generateCertPath(new ByteArrayInputStream(data));
250                } catch (Throwable t) {
251                    throw new NotSerializableException(
252                            Messages.getString("security.67", t)); //$NON-NLS-1$
253                }
254            }
255        }
256    }
