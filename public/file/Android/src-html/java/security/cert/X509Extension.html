018    package java.security.cert;
019    
020    import java.util.Set;
021    
022    /**
023     * The interface specifying an X.509 Certificate or CRL extension.
024     * 
025     * @since Android 1.0
026     */
027    public interface X509Extension {
028    
029        /**
030         * Returns the set of OIDs of the extension(s) marked as CRITICAL, that this
031         * implementation manages.
032         * 
033         * @return the set of extension OIDs marked as CRITIAL, an empty set if none
034         *         are marked as CRITICAL, or {@code null} if no extensions are
035         *         present.
036         * @since Android 1.0
037         */
038        public Set<String> getCriticalExtensionOIDs();
039    
040        /**
041         * Returns the extension value as DER-encoded OCTET string for the specified
042         * OID.
043         * 
044         * @param oid
045         *            the object identifier to get the extension value for.
046         * @return the extension value as DER-encoded OCTET string, or {@code null}
047         *         if no extension for the specified OID can be found.
048         * @since Android 1.0
049         */
050        public byte[] getExtensionValue(String oid);
051    
052        /**
053         * Returns the set of OIDs of the extension(s) marked as NON-CRITICAL, that
054         * this implementation manages.
055         * 
056         * @return the set of extension OIDs marked as NON-CRITIAL, an empty set if
057         *         none are marked as NON-.CRITICAL, or {@code null} if no
058         *         extensions are present.
059         * @since Android 1.0
060         */
061        public Set<String> getNonCriticalExtensionOIDs();
062    
063        /**
064         * Returns whether this instance has an extension marked as CRITICAL that it
065         * cannot support.
066         * 
067         * @return {@code true} if an unsupported CRITICAL extension is present,
068         *         {@code false} otherwise.
069         * @since Android 1.0
070         */
071        public boolean hasUnsupportedCriticalExtension();
072    }
