018    package java.security.cert;
019    
020    import java.security.AccessController;
021    import java.security.InvalidAlgorithmParameterException;
022    import java.security.NoSuchAlgorithmException;
023    import java.security.NoSuchProviderException;
024    import java.security.Provider;
025    import java.security.Security;
026    
027    import org.apache.harmony.security.fortress.Engine;
028    import org.apache.harmony.security.internal.nls.Messages;
029    
030    
031    /**
032     * This class implements the functionality of a builder for an unverified
033     * <i>Certification Path</i>s from a specified certificate to a trust anchor.
034     * 
035     * @since Android 1.0
036     */
037    public class CertPathBuilder {
038    
039        // Store CertPathBuilder service name
040        private static final String SERVICE = "CertPathBuilder"; //$NON-NLS-1$
041    
042        // Used to access common engine functionality
043        private static Engine engine = new Engine(SERVICE);
044    
045        // Store default property name
046        private static final String PROPERTYNAME = "certpathbuilder.type"; //$NON-NLS-1$
047    
048        // Default value of CertPathBuilder type. It returns if certpathbuild.type
049        // property is not defined in java.security file
050        private static final String DEFAULTPROPERTY = "PKIX"; //$NON-NLS-1$
051    
052        // Store used provider
053        private final Provider provider;
054    
055        // Store spi implementation
056        private CertPathBuilderSpi spiImpl;
057    
058        // Store algorithm name
059        private final String algorithm;
060    
061        /**
062         * Creates a new {@code CertPathBuilder}.
063         * 
064         * @param builderSpi
065         *            the implementation delegate.
066         * @param provider
067         *            the provider.
068         * @param algorithm
069         *            the desired algorithm available at the provider.
070         * @since Android 1.0
071         */
072        protected CertPathBuilder(CertPathBuilderSpi builderSpi, Provider provider,
073                String algorithm) {
074            this.provider = provider;
075            this.algorithm = algorithm;
076            this.spiImpl = builderSpi;
077        }
078    
079        /**
080         * Returns the algorithm name of this instance.
081         * 
082         * @return the algorithm name of this instance.
083         * @since Android 1.0
084         */
085        public final String getAlgorithm() {
086            return algorithm;
087        }
088    
089        /**
090         * Returns the provider of this instance.
091         * 
092         * @return the provider of this instance.
093         * @since Android 1.0
094         */
095        public final Provider getProvider() {
096            return provider;
097        }
098    
099        /**
100         * Creates a new {@code CertPathBuilder} instance with the specified
101         * algorithm.
102         * 
103         * @param algorithm
104         *            the name of the algorithm.
105         * @return a builder for the requested algorithm.
106         * @throws NullPointerException
107         *             if the algorithm is {@code null}.
108         * @throws NoSuchAlgorithmException
109         *             if no installed provider can provide the algorithm.
110         * @since Android 1.0
111         */
112        public static CertPathBuilder getInstance(String algorithm)
113                throws NoSuchAlgorithmException {
114            if (algorithm == null) {
115                throw new NullPointerException(Messages.getString("security.01")); //$NON-NLS-1$
116            }
117            synchronized (engine) {
118                engine.getInstance(algorithm, null);
119                return new CertPathBuilder((CertPathBuilderSpi) engine.spi,
120                        engine.provider, algorithm);
121            }
122        }
123    
124        /**
125         * Creates a new {@code CertPathBuilder} instance from the specified
126         * provider providing the specified algorithm.
127         * 
128         * @param algorithm
129         *            the name of the algorithm.
130         * @param provider
131         *            the name of the provider.
132         * @return a builder for the requested algorithm.
133         * @throws NoSuchAlgorithmException
134         *             if the specified provider cannot provide the algorithm.
135         * @throws NoSuchProviderException
136         *             if no provider with the specified name can be found.
137         * @throws NullPointerException
138         *             if algorithm is {@code null}.
139         * @throws IllegalArgumentException
140         *             if provider is {@code null} or empty.
141         * @since Android 1.0
142         */
143        public static CertPathBuilder getInstance(String algorithm, String provider)
144                throws NoSuchAlgorithmException, NoSuchProviderException {
145            if ((provider == null) || (provider.length() == 0)) {
146                throw new IllegalArgumentException(Messages.getString("security.02")); //$NON-NLS-1$
147            }
148            Provider impProvider = Security.getProvider(provider);
149            if (impProvider == null) {
150                throw new NoSuchProviderException(provider);
151            }
152            return getInstance(algorithm, impProvider);
153    
154        }
155    
156        /**
157         * Creates a new {@code CertPathBuilder} instance from the specified
158         * provider providing the specified algorithm.
159         * 
160         * @param algorithm
161         *            the name of the algorithm.
162         * @param provider
163         *            the provider.
164         * @return a builder for the requested algorithm
165         * @throws NoSuchAlgorithmException
166         *             if the specified provider cannot provide the algorithm.
167         * @throws IllegalArgumentException
168         *             if provider is {@code null}.
169         * @throws NullPointerException
170         *             if algorithm is {@code null}.
171         * @since Android 1.0
172         */
173        public static CertPathBuilder getInstance(String algorithm,
174                Provider provider) throws NoSuchAlgorithmException {
175            if (provider == null) {
176                throw new IllegalArgumentException(Messages.getString("security.04")); //$NON-NLS-1$
177            }
178            if (algorithm == null) {
179                throw new NullPointerException(Messages.getString("security.01")); //$NON-NLS-1$
180            }
181            synchronized (engine) {
182                engine.getInstance(algorithm, provider, null);
183                return new CertPathBuilder((CertPathBuilderSpi) engine.spi, provider,
184                        algorithm);
185            }
186        }
187    
188        /**
189         * Builds a certification path with the specified algorithm parameters.
190         * 
191         * @param params
192         *            the algorithm parameters.
193         * @return the built certification path.
194         * @throws CertPathBuilderException
195         *             if the build fails.
196         * @throws InvalidAlgorithmParameterException
197         *             if the specified parameters cannot be used to build with this
198         *             builder.
199         * @see CertPathBuilderResult
200         * @since Android 1.0
201         */
202        public final CertPathBuilderResult build(CertPathParameters params)
203                throws CertPathBuilderException, InvalidAlgorithmParameterException {
204            return spiImpl.engineBuild(params);
205        }
206    
207        /**
208         * Returns the default {@code CertPathBuilder} type from the <i>Security
209         * Properties</i>.
210         * 
211         * @return the default {@code CertPathBuilder} type from the <i>Security
212         *         Properties</i>, or the string "{@code PKIX}" if it cannot be
213         *         determined.
214         * @since Android 1.0
215         */
216        public static final String getDefaultType() {
217            String defaultType = AccessController
218                    .doPrivileged(new java.security.PrivilegedAction<String>() {
219                        public String run() {
220                            return Security.getProperty(PROPERTYNAME);
221                        }
222                    });
223            return (defaultType != null ? defaultType : DEFAULTPROPERTY);
224        }
225    }
