074        /**
075         * Creates a new {@code PKIXParameters} instance with the specified set of
076         * <i>trusted</i> certificate authorities.
077         * 
078         * @param trustAnchors
079         *            the trusted CAs.
080         * @throws InvalidAlgorithmParameterException
081         *             if {@code trustAnchors} is empty.
082         * @since Android 1.0
083         */
084        public PKIXParameters(Set<TrustAnchor> trustAnchors)
085            throws InvalidAlgorithmParameterException {
086            if (trustAnchors == null) {
087                throw new NullPointerException(Messages.getString("security.6F")); //$NON-NLS-1$
088            }
089            checkTrustAnchors(trustAnchors);
090            this.trustAnchors = new HashSet<TrustAnchor>(trustAnchors);
091        }

093        /**
094         * Creates a new {@code PKIXParameters} instance with the trusted {@code
095         * X509Certificate} entries from the specified {@code KeyStore}.
096         * 
097         * @param keyStore
098         *            the key store containing trusted certificates.
099         * @throws KeyStoreException
100         *             if the {@code keyStore} is not initialized.
101         * @throws InvalidAlgorithmParameterException
102         *             if {@code keyStore} does not contained any trusted
103         *             certificate entry.
104         * @since Android 1.0
105         */
106        public PKIXParameters(KeyStore keyStore)
107            throws KeyStoreException,
108                   InvalidAlgorithmParameterException {
109            if (keyStore == null) {
110                throw new NullPointerException(Messages.getString("security.41")); //$NON-NLS-1$
111            }
112            // Will throw KeyStoreException if
113            // keyStore has not been initialized (loaded)
114            if (keyStore.size() == 0) {
115                throw new InvalidAlgorithmParameterException(
116                        Messages.getString("security.6A")); //$NON-NLS-1$
117            }
118            // keyStore is not null and loaded
119            trustAnchors = new HashSet<TrustAnchor>();
120            for (Enumeration i = keyStore.aliases(); i.hasMoreElements();) {
121                String alias = (String) i.nextElement();
122                if (keyStore.isCertificateEntry(alias)) {
123                    // this is trusted certificate entry
124                    // check if it is X509Cerificate
125                    Certificate c = keyStore.getCertificate(alias);
126                    // add only X509Cerificate
127                    // ignore all other types
128                    if (c instanceof X509Certificate) {
129                        trustAnchors.add(new TrustAnchor((X509Certificate)c, null));
130                    }
131                }
132            }
133            checkTrustAnchors(trustAnchors);
134        }

191        /**
192         * Returns the list of checkers for the certification path.
193         * <p>
194         * The list is unmodifiable and the entries in the list are cloned.
195         * </p>
196         * 
197         * @return the list of checkers for the certification path.
198         * @since Android 1.0
199         */
200        public List<PKIXCertPathChecker> getCertPathCheckers() {
201            if (certPathCheckers == null) {
202                // set to empty List if has not been set yet
203                certPathCheckers = new ArrayList<PKIXCertPathChecker>();
204            }
205            if (certPathCheckers.isEmpty()) {
206                // no content - no need to copy,
207                // just return immutable view of the same
208                // empty List each time
209                return Collections.unmodifiableList(certPathCheckers);
210            }
211            // List is not empty - do deep copy
212            ArrayList<PKIXCertPathChecker> modifiableList = 
213                new ArrayList<PKIXCertPathChecker>();
214            for (Iterator<PKIXCertPathChecker> i 
215                    = certPathCheckers.iterator(); i.hasNext();) {
216                modifiableList.add((PKIXCertPathChecker)i.next().clone());
217            }
218            return Collections.unmodifiableList(modifiableList);
219        }

221        /**
222         * Sets the list of checkers for the certification path.
223         * <p>
224         * The list is copied and the entries are cloned.
225         * </p>
226         * 
227         * @param certPathCheckers
228         *            the list of checkers for the certification path, or {@code
229         *            null} to clear the checkers.
230         * @since Android 1.0
231         */
232        public void setCertPathCheckers(List<PKIXCertPathChecker> certPathCheckers) {
233            if (certPathCheckers == null || certPathCheckers.isEmpty()) {
234                // empty list or null provided
235                if (this.certPathCheckers != null &&
236                   !this.certPathCheckers.isEmpty()) {
237                    // discard non-empty list
238                    this.certPathCheckers = null;
239                }
240                return;
241            }
242            // non-empty list provided - do deep copy
243            this.certPathCheckers = new ArrayList<PKIXCertPathChecker>();
244            for (Iterator<PKIXCertPathChecker> i 
245                    = certPathCheckers.iterator(); i.hasNext();) {
246                this.certPathCheckers.add((PKIXCertPathChecker)i.next().clone());
247            }
248        }

251         * Adds the specified {@code PKIXCertPathChecker} to the list of
252         * certification path checkers.
253         * 
254         * @param checker
255         *            the {@code PKIXCertPathChecker} to add, if {@code null}, it
256         *            will be ignored.
257         * @since Android 1.0
258         */
259        public void addCertPathChecker(PKIXCertPathChecker checker) {
260            if (checker == null) {
261                // do nothing if null provided
262                return;
263            }
264            if (certPathCheckers == null) {
265                // set to empty List if has not been set yet
266                certPathCheckers = new ArrayList<PKIXCertPathChecker>();
267            }
268            // add a copy to avoid possible modifications
269            certPathCheckers.add((PKIXCertPathChecker) checker.clone());
270        }

303        public void setCertStores(List<CertStore> certStores) {
304            if (certStores == null || certStores.isEmpty()) {

572         * Clones this {@code PKIXParameters} instance.
573         * 
574         * @return the cloned instance.
575         * @since Android 1.0
576         */
577        public Object clone() {
578            try {
579                // do shallow copy first
580                PKIXParameters ret = (PKIXParameters)super.clone();
581                // copy fields containing references to mutable objects
582                if (this.certStores != null) {
583                    ret.certStores = new ArrayList(this.certStores);
584                }
585                if (this.certPathCheckers != null) {
586                    ret.certPathCheckers = new ArrayList(this.certPathCheckers);
587                }
588                return ret;
589            } catch (CloneNotSupportedException e) {
590                throw new Error(e);
591            }
592        }

608            sb.append("\n Policy Mapping Inhibited: "); //$NON-NLS-1$
609            sb.append(policyMappingInhibited);
610            sb.append("\n Any Policy Inhibited: "); //$NON-NLS-1$
611            sb.append(anyPolicyInhibited);
612            sb.append("\n Policy Qualifiers Rejected: "); //$NON-NLS-1$
613            sb.append(policyQualifiersRejected);

616                    ? "any" : initialPolicies.toString()); //$NON-NLS-1$
617            sb.append("\n Cert Stores: "); //$NON-NLS-1$
618            sb.append((certStores==null||certStores.isEmpty())?
619                    "no":certStores.toString()); //$NON-NLS-1$
620            sb.append("\n Validity Date: "); //$NON-NLS-1$
621            sb.append(date);
622            sb.append("\n Cert Path Checkers: "); //$NON-NLS-1$
623            sb.append((certPathCheckers==null||certPathCheckers.isEmpty())?
624                    "no":certPathCheckers.toString()); //$NON-NLS-1$
625            sb.append("\n Signature Provider: "); //$NON-NLS-1$
626            sb.append(sigProvider);
627            sb.append("\n Target Certificate Constraints: "); //$NON-NLS-1$
628            sb.append(targetCertConstraints);
629            sb.append("\n]"); //$NON-NLS-1$
630            return sb.toString();
631        }

637        //
638        // Checks that 'trustAnchors' contains
639        // only TrustAnchor instances.
640        // Throws InvalidAlgorithmParameterException if trustAnchors set is empty.
641        //
642        private void checkTrustAnchors(Set trustAnchors)
643            throws InvalidAlgorithmParameterException {
644            if (trustAnchors.isEmpty()) {
645                throw new InvalidAlgorithmParameterException(
646                        Messages.getString("security.6D")); //$NON-NLS-1$
647            }
648            for (Iterator i = trustAnchors.iterator(); i.hasNext();) {
649                if (!(i.next() instanceof TrustAnchor)) {
650                    throw new ClassCastException(
651                 Messages.getString("security.6E")); //$NON-NLS-1$
652                }
653            }
654        }
