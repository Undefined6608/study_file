031    /**
032     * Abstract base class for X.509 certificates.
033     * <p>
034     * This represents a standard way for accessing the attributes of X.509
035     * certificates.
036     * </p>
037     * <p>
038     * The basic X.509 v3 format described in ASN.1:
039     * 
040     * <pre>
041     * Certificate  ::=  SEQUENCE  {
042     *     tbsCertificate       TBSCertificate,
043     *     signatureAlgorithm   AlgorithmIdentifier,
044     *     signature            BIT STRING  }
045     *     
046     * TBSCertificate  ::=  SEQUENCE  {
047     *      version         [0]  EXPLICIT Version DEFAULT v1,
048     *      serialNumber         CertificateSerialNumber,
049     *      signature            AlgorithmIdentifier,
050     *      issuer               Name,
051     *      validity             Validity,
052     *      subject              Name,
053     *      subjectPublicKeyInfo SubjectPublicKeyInfo,
054     *      issuerUniqueID  [1]  IMPLICIT UniqueIdentifier OPTIONAL,
055     *                           -- If present, version must be v2 or v3
056     *      subjectUniqueID [2]  IMPLICIT UniqueIdentifier OPTIONAL,
057     *                           -- If present, version must be v2 or v3
058     *      extensions      [3]  EXPLICIT Extensions OPTIONAL
059     *                           -- If present, version must be v3
060     *      }
061     * </pre>
062     * </p>
063     * <p>
064     * For more information consult RFC 2459
065     * "Internet X.509 Public Key Infrastructure Certificate and CRL Profile" at <a
066     * href
067     * ="http://www.ietf.org/rfc/rfc2459.txt">http://www.ietf.org/rfc/rfc2459.txt
068     * </a> .
069     * </p>
070     * 
071     * @since Android 1.0
072     */
073    public abstract class X509Certificate
074            extends Certificate implements X509Extension {
075    
076        private static final long serialVersionUID = -2491127588187038216L;
077    
078        /**
079         * Creates a new {@code X509Certificate}.
080         * 
081         * @since Android 1.0
082         */
083        protected X509Certificate() {
084            super("X.509"); //$NON-NLS-1$
085        }
086    
087        /**
088         * Checks whether the certificate is currently valid.
089         * <p>
090         * The validity defined in ASN.1:
091         * 
092         * <pre>
093         * validity             Validity
094         * 
095         * Validity ::= SEQUENCE { 
096         *      notBefore       CertificateValidityDate, 
097         *      notAfter        CertificateValidityDate }
098         * 
099         * CertificateValidityDate ::= CHOICE { 
100         *      utcTime         UTCTime, 
101         *      generalTime     GeneralizedTime }
102         * </pre>
103         * 
104         * </p>
105         * 
106         * @throws CertificateExpiredException
107         *             if the certificate has expired.
108         * @throws CertificateNotYetValidException
109         *             if the certificate is not yet valid.
110         * @since Android 1.0
111         */
112        public abstract void checkValidity()
113                throws CertificateExpiredException, CertificateNotYetValidException;
114    
115        /**
116         * Checks whether the certificate is valid at the specified date.
117         * 
118         * @param date
119         *            the date to check the validity against.
120         * @throws CertificateExpiredException
121         *             if the certificate has expired.
122         * @throws CertificateNotYetValidException
123         *             if the certificate is not yet valid.
124         * @see #checkValidity()
125         * @since Android 1.0
126         */
127        public abstract void checkValidity(Date date)
128                throws CertificateExpiredException, CertificateNotYetValidException;
129    
130        /**
131         * Returns the certificates {@code version} (version number).
132         * <p>
133         * The version defined is ASN.1:
134         * 
135         * <pre>
136         * Version ::=  INTEGER  {  v1(0), v2(1), v3(2)  }
137         * </pre>
138         * 
139         * </p>
140         * 
141         * @return the version number.
142         * @since Android 1.0
143         */
144        public abstract int getVersion();
145    
146        /**
147         * Returns the {@code serialNumber} of the certificate.
148         * <p>
149         * The ASN.1 definition of {@code serialNumber}:
150         * 
151         * <pre>
152         * CertificateSerialNumber  ::=  INTEGER
153         * </pre>
154         * 
155         * </p>
156         * 
157         * @return the serial number.
158         * @since Android 1.0
159         */
160        public abstract BigInteger getSerialNumber();
161    
162        /**
163         * Returns the {@code issuer} (issuer distinguished name) as an
164         * implementation specific {@code Principal} object.
165         * <p>
166         * The ASN.1 definition of {@code issuer}:
167         * 
168         * <pre>
169         *  issuer      Name
170         * 
171         *  Name ::= CHOICE {
172         *      RDNSequence }
173         * 
174         *    RDNSequence ::= SEQUENCE OF RelativeDistinguishedName
175         * 
176         *    RelativeDistinguishedName ::= SET OF AttributeTypeAndValue
177         * 
178         *    AttributeTypeAndValue ::= SEQUENCE {
179         *      type     AttributeType,
180         *      value    AttributeValue }
181         * 
182         *    AttributeType ::= OBJECT IDENTIFIER
183         * 
184         *    AttributeValue ::= ANY DEFINED BY AttributeType
185         * </pre>
186         * 
187         * </p>
188         * <b>replaced by:</b> {@link #getIssuerX500Principal()}.
189         * 
190         * @return the {@code issuer} as an implementation specific {@code
191         *         Principal}.
192         * @since Android 1.0
193         */
194        public abstract Principal getIssuerDN() ;
195    
196        /**
197         * Returns the {@code issuer} (issuer distinguished name) as an {@code
198         * X500Principal}.
199         * 
200         * @return the {@code issuer} (issuer distinguished name).
201         * @since Android 1.0
202         */
203        public X500Principal getIssuerX500Principal() {
204    
205            try {
206                // TODO if there is no X.509 certificate provider installed
207                // should we try to access Harmony X509CertImpl via classForName?
208                CertificateFactory factory = CertificateFactory
209                        .getInstance("X.509"); //$NON-NLS-1$
210    
211                X509Certificate cert = (X509Certificate) factory
212                        .generateCertificate(new ByteArrayInputStream(getEncoded()));
213    
214                return cert.getIssuerX500Principal();
215    
216            } catch (Exception e) {
217                throw new RuntimeException(Messages.getString("security.59"), e); //$NON-NLS-1$
218            }
219        }
220    
221        /**
222         * Returns the {@code subject} (subject distinguished name) as an
223         * implementation specific {@code Principal} object.
224         * <p>
225         * The ASN.1 definition of {@code subject}:
226         * 
227         * <pre>
228         * subject      Name
229         * 
230         *  Name ::= CHOICE {
231         *      RDNSequence }
232         * 
233         *    RDNSequence ::= SEQUENCE OF RelativeDistinguishedName
234         * 
235         *    RelativeDistinguishedName ::= SET OF AttributeTypeAndValue
236         * 
237         *    AttributeTypeAndValue ::= SEQUENCE {
238         *      type     AttributeType,
239         *      value    AttributeValue }
240         * 
241         *    AttributeType ::= OBJECT IDENTIFIER
242         * 
243         *    AttributeValue ::= ANY DEFINED BY AttributeType
244         * </pre>
245         * 
246         * </p>
247         * <p>
248         * <b>replaced by:</b> {@link #getSubjectX500Principal()}.
249         * </p>
250         * 
251         * @return the {@code subject} (subject distinguished name).
252         * @since Android 1.0
253         */
254        public abstract Principal getSubjectDN();
255    
256        /**
257         * Returns the {@code subject} (subject distinguished name) as an {@code
258         * X500Principal}.
259         * 
260         * @return the {@code subject} (subject distinguished name)
261         * @since Android 1.0
262         */
263        public X500Principal getSubjectX500Principal() {
264    
265            try {
266                // TODO if there is no X.509 certificate provider installed
267                // should we try to access Harmony X509CertImpl via classForName?
268                CertificateFactory factory = CertificateFactory
269                        .getInstance("X.509"); //$NON-NLS-1$
270    
271                X509Certificate cert = (X509Certificate) factory
272                        .generateCertificate(new ByteArrayInputStream(getEncoded()));
273    
274                return cert.getSubjectX500Principal();
275    
276            } catch (Exception e) {
277                throw new RuntimeException(Messages.getString("security.5A"), e); //$NON-NLS-1$
278            }
279        }
280    
281        /**
282         * Returns the {@code notBefore} date from the validity period of the
283         * certificate.
284         * 
285         * @return the start of the validity period.
286         * @since Android 1.0
287         */
288        public abstract Date getNotBefore();
289    
290        /**
291         * Returns the {@code notAfter} date of the validity period of the
292         * certificate.
293         * 
294         * @return the end of the validity period.
295         * @since Android 1.0
296         */
297        public abstract Date getNotAfter();
298    
299        /**
300         * Returns the {@code tbsCertificate} information from this certificate in
301         * DER-encoded format.
302         * 
303         * @return the DER-encoded certificate information.
304         * @throws CertificateEncodingException
305         *             if an error occurs in encoding
306         * @since Android 1.0
307         */
308        public abstract byte[] getTBSCertificate()
309                                        throws CertificateEncodingException;
310    
311        /**
312         * Returns the raw signature bits from the certificate.
313         * 
314         * @return the raw signature bits from the certificate.
315         * @since Android 1.0
316         */
317        public abstract byte[] getSignature();
318    
319        /**
320         * Returns the name of the algorithm for the certificate signature.
321         * 
322         * @return the signature algorithm name.
323         * @since Android 1.0
324         */
325        public abstract String getSigAlgName();
326    
327        /**
328         * Returns the OID of the signature algorithm from the certificate.
329         * 
330         * @return the OID of the signature algorithm.
331         * @since Android 1.0
332         */
333        public abstract String getSigAlgOID();
334    
335        /**
336         * Returns the parameters of the signature algorithm in DER-encoded format.
337         * 
338         * @return the parameters of the signature algorithm, or {@code null} if
339         *         none are used.
340         * @since Android 1.0
341         */
342        public abstract byte[] getSigAlgParams();
343    
344        /**
345         * Returns the {@code issuerUniqueID} from the certificate.
346         * 
347         * @return the {@code issuerUniqueID} or {@code null} if there's none in the
348         *         certificate.
349         * @since Android 1.0
350         */
351        public abstract boolean[] getIssuerUniqueID();
352    
353        /**
354         * Returns the {@code subjectUniqueID} from the certificate.
355         * 
356         * @return the {@code subjectUniqueID} or null if there's none in the
357         *         certificate.
358         * @since Android 1.0
359         */
360        public abstract boolean[] getSubjectUniqueID();
361    
362        /**
363         * Returns the {@code KeyUsage} extension as a {@code boolean} array.
364         * <p>
365         * The ASN.1 definition of {@code KeyUsage}:
366         * 
367         * <pre>
368         * KeyUsage ::= BIT STRING {
369         *      digitalSignature        (0),
370         *      nonRepudiation          (1),
371         *      keyEncipherment         (2),
372         *      dataEncipherment        (3),
373         *      keyAgreement            (4),
374         *      keyCertSign             (5),
375         *      cRLSign                 (6),
376         *      encipherOnly            (7),
377         *      decipherOnly            (8) }
378         * 
379         * </pre>
380         * 
381         * </p>
382         * 
383         * @return the {@code KeyUsage} extension or {@code null} if there's none in
384         *         the certificate.
385         * @since Android 1.0
386         */
387        public abstract boolean[] getKeyUsage();
388    
389        /**
390         * Returns a read-only list of OID strings representing the {@code
391         * ExtKeyUsageSyntax} field of the extended key usage extension.
392         * 
393         * @return the extended key usage extension, or {@code null} if there's none
394         *         in the certificate.
395         * @throws CertificateParsingException
396         *             if the extension decoding fails.
397         * @since Android 1.0
398         */
399        public List<String> getExtendedKeyUsage()
400                            throws CertificateParsingException {
401            return null;
402        }
403    
404        /**
405         * Returns the path length of the certificate constraints from the {@code
406         * BasicContraints} extension.
407         * 
408         * @return the path length of the certificate constraints if the extension
409         *         is present or {@code -1} if the extension is not present. {@code
410         *         Integer.MAX_VALUE} if there's not limit.
411         * @since Android 1.0
412         */
413        public abstract int getBasicConstraints();
414    
415        /**
416         * Returns a read-only list of the subject alternative names from the
417         * {@code SubjectAltName} extension.
418         * <p>
419         * The ASN.1 definition of {@code SubjectAltName}:
420         * 
421         * <pre>
422         * SubjectAltName ::= GeneralNames
423         * 
424         * GeneralNames ::= SEQUENCE SIZE (1..MAX) OF GeneralName
425         * 
426         * GeneralName ::= CHOICE {
427         *      otherName                       [0]     AnotherName,
428         *      rfc822Name                      [1]     IA5String,
429         *      dNSName                         [2]     IA5String,
430         *      x400Address                     [3]     ORAddress,
431         *      directoryName                   [4]     Name,
432         *      ediPartyName                    [5]     EDIPartyName,
433         *      uniformResourceIdentifier       [6]     IA5String,
434         *      iPAddress                       [7]     OCTET STRING,
435         *      registeredID                    [8]     OBJECT IDENTIFIER }
436         * 
437         * </pre>
438         * 
439         * </p>
440         * 
441         * @return the subject alternative names or {@code null} if there are none
442         *         in the certificate.
443         * @throws CertificateParsingException
444         *             if decoding of the extension fails.
445         * @since Android 1.0
446         */
447        public Collection<List<?>> getSubjectAlternativeNames()
448                                        throws CertificateParsingException {
449            return null;
450        }
451    
452        /**
453         * Returns a read-only list of the issuer alternative names from the {@code
454         * IssuerAltName} extension.
455         * <p>
456         * The ASN.1 definition of {@code IssuerAltName}:
457         * 
458         * <pre>
459         * IssuerAltName ::= GeneralNames
460         * 
461         * GeneralNames ::= SEQUENCE SIZE (1..MAX) OF GeneralName
462         * 
463         * GeneralName ::= CHOICE {
464         *      otherName                       [0]     AnotherName,
465         *      rfc822Name                      [1]     IA5String,
466         *      dNSName                         [2]     IA5String,
467         *      x400Address                     [3]     ORAddress,
468         *      directoryName                   [4]     Name,
469         *      ediPartyName                    [5]     EDIPartyName,
470         *      uniformResourceIdentifier       [6]     IA5String,
471         *      iPAddress                       [7]     OCTET STRING,
472         *      registeredID                    [8]     OBJECT IDENTIFIER }
473         * 
474         * </pre>
475         * 
476         * </p>
477         * 
478         * @return the issuer alternative names of {@code null} if there are none in
479         *         the certificate.
480         * @throws CertificateParsingException
481         *             if decoding of the extension fails.
482         * @since Android 1.0
483         */
484        public Collection<List<?>> getIssuerAlternativeNames()
485                                        throws CertificateParsingException {
486            return null;
487        }
