033    /**
034     * {@code KeyStoreSpi} is the Service Provider Interface (SPI) definition for
035     * {@link KeyStore}.
036     * 
037     * @see KeyStore
038     * @since Android 1.0
039     */
040    public abstract class KeyStoreSpi {
041    
042        /**
043         * Returns the key with the given alias, using the password to recover the
044         * key from the store.
045         * 
046         * @param alias
047         *            the alias for the entry.
048         * @param password
049         *            the password used to recover the key.
050         * @return the key with the specified alias, or {@code null} if the
051         *         specified alias is not bound to an entry.
052         * @throws NoSuchAlgorithmException
053         *             if the algorithm for recovering the key is not available.
054         * @throws UnrecoverableKeyException
055         *             if the key can not be recovered.
056         * @since Android 1.0
057         */
058        public abstract Key engineGetKey(String alias, char[] password)
059                throws NoSuchAlgorithmException, UnrecoverableKeyException;
060

062         * Returns the certificate chain for the entry with the given alias.
063         * 
064         * @param alias
065         *            the alias for the entry
066         * @return the certificate chain for the entry with the given alias, or
067         *         {@code null} if the specified alias is not bound to an entry.
068         * @since Android 1.0
069         */
070        public abstract Certificate[] engineGetCertificateChain(String alias);

073         * Returns the trusted certificate for the entry with the given alias.
074         * 
075         * @param alias
076         *            the alias for the entry.
077         * @return the trusted certificate for the entry with the given alias, or
078         *         {@code null} if the specified alias is not bound to an entry.
079         * @since Android 1.0
080         */
081        public abstract Certificate engineGetCertificate(String alias);

083        /**
084         * Returns the creation date of the entry with the given alias.
085         * 
086         * @param alias
087         *            the alias for the entry.
088         * @return the creation date, or {@code null} if the specified alias is not
089         *         bound to an entry.
090         * @since Android 1.0
091         */
092        public abstract Date engineGetCreationDate(String alias);

094        /**
095         * Associates the given alias with the key, password and certificate chain.
096         * <p>
097         * If the specified alias already exists, it will be reassigned.
098         * </p>
099         * 
100         * @param alias
101         *            the alias for the key.
102         * @param key
103         *            the key.
104         * @param password
105         *            the password.
106         * @param chain
107         *            the certificate chain.
108         * @throws KeyStoreException
109         *             if the specified key can not be protected, or if this
110         *             operation fails for another reason.
111         * @throws IllegalArgumentException
112         *             if {@code key} is a {@code PrivateKey} and {@code chain} does
113         *             not contain any certificates.
114         * @since Android 1.0
115         */
116        public abstract void engineSetKeyEntry(String alias, Key key,
117                char[] password, Certificate[] chain) throws KeyStoreException;

119        /**
120         * Associates the given alias with a key and a certificate chain.
121         * <p>
122         * If the specified alias already exists, it will be reassigned.
123         * </p>
124         * 
125         * @param alias
126         *            the alias for the key.
127         * @param key
128         *            the key in an encoded format.
129         * @param chain
130         *            the certificate chain.
131         * @throws KeyStoreException
132         *             if this operation fails.
133         * @throws IllegalArgumentException
134         *             if {@code key} is a {@code PrivateKey} and {@code chain}
135         *             does.
136         * @since Android 1.0
137         */
138        public abstract void engineSetKeyEntry(String alias, byte[] key,
139                Certificate[] chain) throws KeyStoreException;

281        /**
282         * Loads this {@code KeyStoreSpi} from the given {@code InputStream}.
283         * Utilizes the given password to verify the stored data.
284         * 
285         * @param stream
286         *            the {@code InputStream} to load this {@code KeyStoreSpi}'s
287         *            data from.
288         * @param password
289         *            the password to verify the stored data, maybe {@code null}.
290         * @throws IOException
291         *             if a problem occurred while reading from the stream.
292         * @throws NoSuchAlgorithmException
293         *             if the required algorithm is not available.
294         * @throws CertificateException
295         *             if the an exception occurred while loading the certificates
296         *             of this code {@code KeyStoreSpi}.
297         * @since Android 1.0
298         */
299        public abstract void engineLoad(InputStream stream, char[] password)
300                throws IOException, NoSuchAlgorithmException, CertificateException;
301    
302        /**
303         * Loads this {@code KeyStoreSpi} using the specified {@code
304         * LoadStoreParameter}.
305         * 
306         * @param param
307         *            the {@code LoadStoreParameter} that specifies how to load this
308         *            {@code KeyStoreSpi}, maybe {@code null}.
309         * @throws IOException
310         *             if a problem occurred while reading from the stream.
311         * @throws NoSuchAlgorithmException
312         *             if the required algorithm is not available.
313         * @throws CertificateException
314         *             if the an exception occurred while loading the certificates
315         *             of this code {@code KeyStoreSpi}.
316         * @throws IllegalArgumentException
317         *             if the given {@link KeyStore.LoadStoreParameter} is not
318         *             recognized.
319         * @since Android 1.0
320         */
321        public void engineLoad(KeyStore.LoadStoreParameter param)
322                throws IOException, NoSuchAlgorithmException, CertificateException {
323            if (param == null) {
324                engineLoad(null, null);
325                return;
326            }
327            char[] pwd;
328            KeyStore.ProtectionParameter pp = param.getProtectionParameter();
329            if (pp instanceof KeyStore.PasswordProtection) {
330                try {
331                    pwd = ((KeyStore.PasswordProtection) pp).getPassword();
332                    engineLoad(null, pwd);
333                    return;
334                } catch (IllegalStateException e) {
335                    throw new IllegalArgumentException(e);
336                }
337            }
338            if (pp instanceof KeyStore.CallbackHandlerProtection) {
339                try {
340                    pwd = getPasswordFromCallBack(pp);
341                    engineLoad(null, pwd);
342                    return;
343                } catch (UnrecoverableEntryException e) {
344                    throw new IllegalArgumentException(e);
345                }
346            }
347            throw new UnsupportedOperationException(
348                    Messages.getString("security.35")); //$NON-NLS-1$
349        }

351        /**
352         * Returns the {@code Entry} with the given alias, using the specified
353         * {@code ProtectionParameter}.
354         * 
355         * @param alias
356         *            the alias of the requested entry.
357         * @param protParam
358         *            the {@code ProtectionParameter}, used to protect the requested
359         *            entry, maybe {@code null}.
360         * @return he {@code Entry} with the given alias, using the specified
361         *         {@code ProtectionParameter}.
362         * @throws NoSuchAlgorithmException
363         *             if the required algorithm is not available.
364         * @throws UnrecoverableEntryException
365         *             if the entry can not be recovered.
366         * @throws KeyStoreException
367         *             if this operation fails
368         * @since Android 1.0
369         */
370        public KeyStore.Entry engineGetEntry(String alias,
371                KeyStore.ProtectionParameter protParam) throws KeyStoreException,
372                NoSuchAlgorithmException, UnrecoverableEntryException {
373            if (!engineContainsAlias(alias)) {
374                return null;
375            }
376            if (engineIsCertificateEntry(alias)) {
377                return new KeyStore.TrustedCertificateEntry(
378                        engineGetCertificate(alias));
379            }
380            char[] passW = null;
381            if (protParam != null) {
382                if (protParam instanceof KeyStore.PasswordProtection) {
383                    try {
384                        passW = ((KeyStore.PasswordProtection) protParam)
385                                .getPassword();
386                    } catch (IllegalStateException ee) {
387                        throw new KeyStoreException(Messages.getString("security.36"), ee); //$NON-NLS-1$
388                    }
389                } else if (protParam instanceof KeyStore.CallbackHandlerProtection) {
390                    passW = getPasswordFromCallBack(protParam);
391                } else {
392                    throw new UnrecoverableEntryException(
393                            Messages.getString("security.37", //$NON-NLS-1$
394                                    protParam.toString()));
395                }
396            }
397            if (engineIsKeyEntry(alias)) {
398                try {
399                    Key key = engineGetKey(alias, passW);
400                    if (key instanceof PrivateKey) {
401                        return new KeyStore.PrivateKeyEntry((PrivateKey) key,
402                                engineGetCertificateChain(alias));
403                    }
404                    if (key instanceof SecretKey) {
405                        return new KeyStore.SecretKeyEntry((SecretKey) key);
406                    }
407                } catch (UnrecoverableKeyException e) {
408                    throw new KeyStoreException(e);
409                }
410            }
411            throw new NoSuchAlgorithmException(Messages.getString("security.38")); //$NON-NLS-1$
412        }
413    
414        /**
415         * Stores the given {@code Entry} in this {@code KeyStoreSpi} and associates
416         * the entry with the given {@code alias}. The entry is protected by the
417         * specified {@code ProtectionParameter}.
418         * <p>
419         * If the specified alias already exists, it will be reassigned.
420         * </p>
421         * 
422         * @param alias
423         *            the alias for the entry.
424         * @param entry
425         *            the entry to store.
426         * @param protParam
427         *            the {@code ProtectionParameter} to protect the entry.
428         * @throws KeyStoreException
429         *             if this operation fails.
430         * @since Android 1.0
431         */
432        public void engineSetEntry(String alias, KeyStore.Entry entry,
433                KeyStore.ProtectionParameter protParam) throws KeyStoreException {
434            if (entry == null) {
435                throw new KeyStoreException(Messages.getString("security.39")); //$NON-NLS-1$
436            }
437    
438            if (engineContainsAlias(alias)) {
439                engineDeleteEntry(alias);
440            }
441    
442            if (entry instanceof KeyStore.TrustedCertificateEntry) {
443                KeyStore.TrustedCertificateEntry trE = (KeyStore.TrustedCertificateEntry) entry;
444                engineSetCertificateEntry(alias, trE.getTrustedCertificate());
445                return;
446            }
447    
448            char[] passW = null;
449            if (protParam instanceof KeyStore.PasswordProtection) {
450                try {
451                    passW = ((KeyStore.PasswordProtection) protParam).getPassword();
452                } catch (IllegalStateException ee) {
453                    throw new KeyStoreException(Messages.getString("security.36"), ee); //$NON-NLS-1$
454                }
455            } else {
456                if (protParam instanceof KeyStore.CallbackHandlerProtection) {
457                    try {
458                        passW = getPasswordFromCallBack(protParam);
459                    } catch (Exception e) {
460                        throw new KeyStoreException(e);
461                    }
462                } else {
463                    throw new KeyStoreException(
464                            Messages.getString("security.3A")); //$NON-NLS-1$
465                }
466            }
467    
468            if (entry instanceof KeyStore.PrivateKeyEntry) {
469                KeyStore.PrivateKeyEntry prE = (KeyStore.PrivateKeyEntry) entry;
470                engineSetKeyEntry(alias, prE.getPrivateKey(), passW, prE
471                        .getCertificateChain());
472                return;
473            }
474    
475            if (entry instanceof KeyStore.SecretKeyEntry) {
476                KeyStore.SecretKeyEntry skE = (KeyStore.SecretKeyEntry) entry;
477                engineSetKeyEntry(alias, skE.getSecretKey(), passW, null);
478                //            engineSetKeyEntry(alias, skE.getSecretKey().getEncoded(), null);
479                return;
480            }
481    
482            throw new KeyStoreException(
483                    Messages.getString("security.3B", entry.toString())); //$NON-NLS-1$
484        }

486        /**
487         * Indicates whether the entry for the given alias is assignable to the
488         * provided {@code Class}.
489         * 
490         * @param alias
491         *            the alias for the entry.
492         * @param entryClass
493         *            the type of the entry.
494         * @return {@code true} if the {@code Entry} for the alias is assignable to
495         *         the specified {@code entryClass}.
496         * @since Android 1.0
497         */
498        public boolean engineEntryInstanceOf(String alias,
499                Class<? extends KeyStore.Entry> entryClass) {
500            if (!engineContainsAlias(alias)) {
501                return false;
502            }
503    
504            try {
505                if (engineIsCertificateEntry(alias)) {
506                    return entryClass
507                            .isAssignableFrom(Class
508                                    .forName("java.security.KeyStore$TrustedCertificateEntry")); //$NON-NLS-1$
509                }
510    
511                if (engineIsKeyEntry(alias)) {
512                    if (entryClass.isAssignableFrom(Class
513                            .forName("java.security.KeyStore$PrivateKeyEntry"))) { //$NON-NLS-1$
514                        return engineGetCertificate(alias) != null;
515                    }
516    
517                    if (entryClass.isAssignableFrom(Class
518                            .forName("java.security.KeyStore$SecretKeyEntry"))) { //$NON-NLS-1$
519                        return engineGetCertificate(alias) == null;
520                    }
521                }
522            } catch (ClassNotFoundException ignore) {}
523    
524            return false;
525        }

527        /*
528         * This method returns password which is encapsulated in
529         * CallbackHandlerProtection object If there is no implementation of
530         * CallbackHandler then this method returns null
531         */
532        static char[] getPasswordFromCallBack(KeyStore.ProtectionParameter protParam)
533                throws UnrecoverableEntryException {
534    
535            if (protParam == null) {
536                return null;
537            }
538    
539            if (!(protParam instanceof KeyStore.CallbackHandlerProtection)) {
540                throw new UnrecoverableEntryException(
541                        Messages.getString("security.3C")); //$NON-NLS-1$
542            }
543    
544            String clName = Security
545                    .getProperty("auth.login.defaultCallbackHandler"); //$NON-NLS-1$
546            if (clName == null) {
547                throw new UnrecoverableEntryException(
548                        Messages.getString("security.3D")); //$NON-NLS-1$
549    
550            }
551    
552            try {
553                Class<?> cl = Class.forName(clName);
554                CallbackHandler cbHand = (CallbackHandler) cl.newInstance();
555                PasswordCallback[] pwCb = { new PasswordCallback("password: ", true) }; //$NON-NLS-1$
556                cbHand.handle(pwCb);
557                return pwCb[0].getPassword();
558            } catch (Exception e) {
559                throw new UnrecoverableEntryException(e.toString());
560            }
561        }
