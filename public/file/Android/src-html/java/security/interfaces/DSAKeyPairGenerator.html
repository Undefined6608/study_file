018    package java.security.interfaces;
019    
020    import java.security.InvalidParameterException;
021    import java.security.SecureRandom;
022    
023    /**
024     * The interface for key generators that can generate DSA key pairs.
025     * 
026     * @since Android 1.0
027     */
028    public interface DSAKeyPairGenerator {
029    
030        /**
031         * Initializes this generator with the prime ({@code p}), subprime ({@code
032         * q}), and base ({@code g}) values from the specified parameters.
033         * 
034         * @param params
035         *            the parameter values.
036         * @param random
037         *            the source of randomness.
038         * @throws InvalidParameterException
039         *             if the specified parameter values are {@code null} or
040         *             invalid.
041         * @since Android 1.0
042         */
043        public void initialize(DSAParams params, SecureRandom random)
044                throws InvalidParameterException;
045    
046        /**
047         * Initializes this generator for the specified modulus length. Valid values
048         * for the modulus length are the multiples of 8 between 512 and 1024.
049         * <p>
050         * The parameter {@code genParams} specifies whether this method should
051         * generate new prime ({@code p}), subprime ({@code q}), and base ({@code g})
052         * values or whether
053         * it will use the pre-calculated values for the specified modulus
054         * length. Default parameters are available for modulus lengths of 512 and 1024
055         * bits.
056         * </p>
057         * 
058         * @param modlen
059         *            the length of the modulus in bits.
060         * @param genParams
061         *            whether new values should be generated.
062         * @param random
063         *            the source of randomness.
064         * @throws InvalidParameterException
065         *             if the specified modulus length is not valid, or if there are
066         *             no pre-calculated values and {@code genParams} is {@code
067         *             false}.
068         * @since Android 1.0
069         */
070        public void initialize(int modlen, boolean genParams, SecureRandom random)
071                throws InvalidParameterException;
072    }
