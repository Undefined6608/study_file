078         * {@code URL} and the {@code Certificate}s.
079         *

116        /**
117         * Compares the specified object with this {@code CodeSource} for equality.
118         * Returns {@code true} if the specified object is also an instance of
119         * {@code CodeSource}, points to the same {@code URL} location and the two
120         * code sources encapsulate the same {@code Certificate}s. The order of the
121         * {@code Certificate}s is ignored by this method.
122         * 
123         * @param obj
124         *            object to be compared for equality with this {@code
125         *            CodeSource}.
126         * @return {@code true} if the specified object is equal to this {@code
127         *         CodeSource}, otherwise {@code false}.
128         * @since Android 1.0
129         */
130        public boolean equals(Object obj) {
131            if (obj == this) {
132                return true;
133            }
134    
135            if (!(obj instanceof CodeSource)) {
136                return false;
137            }
138    
139            CodeSource that = (CodeSource) obj;
140    
141            if (this.location != null) {
142                if (that.location == null) {
143                    return false;
144                }
145                if (!this.location.equals(that.location)) {
146                    return false;
147                }
148            } else if (that.location != null) {
149                return false;
150            }
151    
152            // do not use this.certs, as we also need to take care about 
153            // CodeSigners' certificates
154            Certificate[] thizCerts = getCertificatesNoClone();
155            Certificate[] thatCerts = that.getCertificatesNoClone();
156            if (!PolicyUtils.matchSubset(thizCerts, thatCerts)) {
157                return false;
158            }
159            if (!PolicyUtils.matchSubset(thatCerts, thizCerts)) {
160                return false;
161            }
162            return true;
163        }

165        /**
166         * Returns the certificates of this {@code CodeSource}. If the
167         * {@link #CodeSource(URL, CodeSigner[])} constructor was used to create
168         * this instance, the certificates are obtained from the supplied signers.
169         * <p>
170         * External modifications of the returned {@code Certificate[]} has no
171         * impact on this {@code CodeSource}.
172         * </p>
173         * 
174         * @return the certificates of this {@code CodeSource} or {@code null} if
175         *         there is none.
176         * @since Android 1.0
177         */
178        public final Certificate[] getCertificates() {
179            getCertificatesNoClone();
180            if (certs == null) {
181                return null;
182            }
183            Certificate[] tmp = new Certificate[certs.length];
184            System.arraycopy(certs, 0, tmp, 0, certs.length);
185            return tmp;
186        }

188        // Acts exactly as {@link #getCertificates()} does, but does not clone the 
189        // array before returning (and returns reference to <code>this.certs</code>
190        // if this array is not null).<br>
191        // @return a reference to the certificates array, or null if there are no 
192        // certificates associated.  
193        private Certificate[] getCertificatesNoClone() {
194            if (certs != null) {
195                return certs;
196            }
197    
198            if (signers == null) {
199                return null;
200            }
201            // Extract Certificates from the CodeSigner-s
202            ArrayList<Certificate> v = new ArrayList<Certificate>();
203            for (int i = 0; i < signers.length; i++) {
204                v.addAll(signers[i].getSignerCertPath().getCertificates());
205            }
206    
207            certs = v.toArray(new Certificate[v.size()]);
208            return certs;
209        }

211        /**
212         * Returns the {@code CodeSigner}s of this {@code CodeSource}. If the
213         * {@link #CodeSource(URL, Certificate[])} constructor was used to create
214         * this instance, the signers are obtained from the supplied certificates.
215         * Only X.509 certificates are analyzed.
216         * 
217         * @return the signers of this {@code CodeSource}, or {@code null} if there
218         *         is none.
219         * @since Android 1.0
220         */
221        public final CodeSigner[] getCodeSigners() {
222            if (signers != null) {
223                CodeSigner[] tmp = new CodeSigner[signers.length];
224                System.arraycopy(signers, 0, tmp, 0, tmp.length);
225                return tmp;
226            }
227            if(certs == null || factory != null){
228                // factory != null means we've done this exercise already.
229                return null;
230            }
231    
232            X500Principal prevIssuer = null;
233            ArrayList<Certificate> list = new ArrayList<Certificate>(certs.length);
234            ArrayList<CodeSigner> asigners = new ArrayList<CodeSigner>();
235    
236            // The presumption is that the chains of certificates are placed
237            // according to the CertPath agreement:
238            //
239            // the lowest certs first; the CAs are at the last
240            //
241            // So the following loop scans trough the certs and checks
242            // that every next certificate is an Issuer of the previous one. 
243            // Any certificate that is not an Issuer of the previous one starts a 
244            // new chain (== a new CertPath) 
245    
246            for (int i = 0; i < certs.length; i++) {
247                if (!(certs[i] instanceof X509Certificate)) {
248                    // Only X509Certificate-s are taken into account - see API spec.
249                    continue;
250                }
251                X509Certificate x509 = (X509Certificate) certs[i];
252                if (prevIssuer == null) {
253                    // start a very first chain
254                    prevIssuer = x509.getIssuerX500Principal();
255                    list.add(x509);
256                } else {
257                    X500Principal subj = x509.getSubjectX500Principal();
258                    if (!prevIssuer.equals(subj)) {
259                        // Ok, this ends the previous chain, 
260                        // so transform this one into CertPath ...
261                        CertPath cpath = makeCertPath(list);
262                        if (cpath != null) {
263                            asigners.add(new CodeSigner(cpath, null));
264                        }
265                        // ... and start a new one
266                        list.clear();
267                    }// else { it's still the same chain }
268                    prevIssuer = x509.getSubjectX500Principal();
269                    list.add(x509);
270                }
271            }
272            if (!list.isEmpty()) {
273                CertPath cpath = makeCertPath(list);
274                if (cpath != null) {
275                    asigners.add(new CodeSigner(cpath, null));
276                }
277            }
278            if (asigners.isEmpty()) {
279                // 'signers' is 'null' already
280                return null;
281            }
282            signers = new CodeSigner[asigners.size()];
283            asigners.toArray(signers);
284            CodeSigner[] tmp = new CodeSigner[asigners.size()];
285            System.arraycopy(signers, 0, tmp, 0, tmp.length);
286            return tmp;
287        }

289        // Makes an CertPath from a given List of X509Certificate-s. 
290        // @param list
291        // @return CertPath, or null if CertPath cannot be made  
292        private CertPath makeCertPath(List<? extends Certificate> list) {
293            if (factory == null) {
294                try {
295                    factory = CertificateFactory.getInstance("X.509"); //$NON-NLS-1$
296                } catch (CertificateException ex) {
297                    //? throw new Error("X.509 is a 'must be'", ex);
298                    return null;
299                }
300            }
301            try {
302                return factory.generateCertPath(list);
303            } catch (CertificateException ex) {
304                // ignore(ex)
305            }
306            return null;
307        }

338        /**
339         * Indicates whether the specified code source is implied by this {@code
340         * CodeSource}. Returns {@code true} if all of the following conditions are
341         * {@code true}, otherwise {@code false}:
342         * <p>
343         * <ul>
344         * <li>{@code cs} is not {@code null}
345         * <li>if this {@code CodeSource} has associated certificates, all
346         * certificates are present in {@code cs}. The certificates are extracted
347         * from the signers if signers are present.
348         * <li>if this {@code CodeSource}'s location is not {@code null}, the
349         * following conditions are checked
350         * <ul>
351         * <li>this {@code CodeSource}'s location is not {@code null}
352         * <li>this {@code CodeSource}'s location protocol is equal to {@code cs}'s
353         * location protocol
354         * <li>if this {@code CodeSource}'s location host is not {@code null}, the
355         * following conditions are checked
356         * <ul>
357         * <li>{@code cs}'s host is not {@code null}
358         * <li>the {@link SocketPermission} of this {@code CodeSource}'s location
359         * host implies the {@code SocketPermission} of {@code cs}'s location host
360         * </ul>
361         * <li>if this {@code CodeSource}'s location port != -1 the port of {@code
362         * cs}'s location is equal to this {@code CodeSource}'s location port
363         * <li>this {@code CodeSource}'s location file matches {@code cs}'s file
364         * whereas special wildcard matching applies as described below
365         * <li>this {@code CodeSource}'s location reference is equal to to {@code
366         * cs}'s location reference
367         * </ul>
368         * </ul>
369         * <p>
370         * Note: If this {@code CodeSource} has a {@code null} location and not any
371         * certificates, this method returns {@code true}.
372         * <p>
373         * Matching rules for the {@code CodeSource}'s location file:
374         * <ul>
375         * <li>if this {@code CodeSource}'s location file ends with {@code "/-"},
376         * then {@code cs}'s file must start with {@code CodeSource}'s location file
377         * (exclusive the trailing '-')
378         * <li>if this {@code CodeSource}'s location file ends with {@code "/*"},
379         * then {@code cs}'s file must start with {@code CodeSource}'s location file
380         * (exclusive the trailing '*') and must not have any further '/'
381         * <li>if this {@code CodeSource}'s location file ends with {@code "/"},
382         * then {@code cs}'s file must start with {@code CodeSource}'s location file
383         * <li>if this {@code CodeSource}'s location file does not end with {@code
384         * "/"}, then {@code cs}'s file must start with {@code CodeSource}'s
385         * location file with the '/' appended to it.
386         * </ul>
387         * Examples for locations that imply the location
388         * "http://code.google.com/android/security.apk":
389         * 
390         * <pre>
391         * http:
392         * http://&#42;/android/*
393         * http://*.google.com/android/*
394         * http://code.google.com/android/-
395         * http://code.google.com/android/security.apk
396         * </pre>
397         * 
398         * @param cs
399         *            the code source to check.
400         * @return {@code true} if the argument code source is implied by this
401         *         {@code CodeSource}, otherwise {@code false}.
402         * @since Android 1.0
403         */
404        public boolean implies(CodeSource cs) {
405            //
406            // Here, javadoc:N refers to the appropriate item in the API spec for 
407            // the CodeSource.implies()
408            // The info was taken from the 1.5 final API spec
409    
410            // javadoc:1
411            if (cs == null) {
412                return false;
413            }
414    
415            // javadoc:2
416            // with a comment: the javadoc says only about certificates and does 
417            // not explicitly mention CodeSigners' certs.
418            // It seems more convenient to use getCerts() to get the real 
419            // certificates - with a certificates got form the signers
420            Certificate[] thizCerts = getCertificatesNoClone();
421            if (thizCerts != null) {
422                Certificate[] thatCerts = cs.getCertificatesNoClone();
423                if (thatCerts == null
424                        || !PolicyUtils.matchSubset(thizCerts, thatCerts)) {
425                    return false;
426                }
427            }
428    
429            // javadoc:3
430            if (this.location != null) {
431                //javadoc:3.1
432                if (cs.location == null) {
433                    return false;
434                }
435                //javadoc:3.2
436                if (this.location.equals(cs.location)) {
437                    return true;
438                }
439                //javadoc:3.3
440                if (!this.location.getProtocol().equals(cs.location.getProtocol())) {
441                    return false;
442                }
443                //javadoc:3.4
444                String thisHost = this.location.getHost();
445                if (thisHost != null) {
446                    String thatHost = cs.location.getHost();
447                    if (thatHost == null) {
448                        return false;
449                    }
450    
451                    // 1. According to the spec, an empty string will be considered 
452                    // as "localhost" in the SocketPermission
453                    // 2. 'file://' URLs will have an empty getHost()
454                    // so, let's make a special processing of localhost-s, I do 
455                    // believe this'll improve performance of file:// code sources 
456    
457                    //
458                    // Don't have to evaluate both the boolean-s each time.
459                    // It's better to evaluate them directly under if() statement.
460                    // 
461                    // boolean thisIsLocalHost = thisHost.length() == 0 || "localhost".equals(thisHost);
462                    // boolean thatIsLocalHost = thatHost.length() == 0 || "localhost".equals(thatHost);
463                    // 
464                    // if( !(thisIsLocalHost && thatIsLocalHost) &&
465                    // !thisHost.equals(thatHost)) {
466    
467                    if (!((thisHost.length() == 0 || "localhost".equals(thisHost)) && (thatHost //$NON-NLS-1$
468                            .length() == 0 || "localhost".equals(thatHost))) //$NON-NLS-1$
469                            && !thisHost.equals(thatHost)) {
470    
471                        // Obvious, but very slow way....
472                        // 
473                        // SocketPermission thisPerm = new SocketPermission(
474                        //          this.location.getHost(), "resolve");
475                        // SocketPermission thatPerm = new SocketPermission(
476                        //          cs.location.getHost(), "resolve");
477                        // if (!thisPerm.implies(thatPerm)) { 
478                        //      return false;
479                        // }
480                        //
481                        // let's cache it: 
482    
483                        if (this.sp == null) {
484                            this.sp = new SocketPermission(thisHost, "resolve"); //$NON-NLS-1$
485                        }
486    
487                        if (cs.sp == null) {
488                            cs.sp = new SocketPermission(thatHost, "resolve"); //$NON-NLS-1$
489                        }
490    
491                        if (!this.sp.implies(cs.sp)) {
492                            return false;
493                        }
494                    } // if( ! this.location.getHost().equals(cs.location.getHost())
495                } // if (this.location.getHost() != null)
496    
497                //javadoc:3.5
498                if (this.location.getPort() != -1) {
499                    if (this.location.getPort() != cs.location.getPort()) {
500                        return false;
501                    }
502                }
503    
504                //javadoc:3.6
505                String thisFile = this.location.getFile();
506                String thatFile = cs.location.getFile();
507    
508                if (thisFile.endsWith("/-")) { //javadoc:3.6."/-" //$NON-NLS-1$
509                    if (!thatFile.startsWith(thisFile.substring(0, thisFile
510                            .length() - 2))) {
511                        return false;
512                    }
513                } else if (thisFile.endsWith("/*")) { //javadoc:3.6."/*" //$NON-NLS-1$
514                    if (!thatFile.startsWith(thisFile.substring(0, thisFile
515                            .length() - 2))) {
516                        return false;
517                    }
518                    // no further separators(s) allowed
519                    if (thatFile.indexOf("/", thisFile.length() - 1) != -1) { //$NON-NLS-1$
520                        return false;
521                    }
522                } else {
523                    // javadoc:3.6."/"
524                    if (!thisFile.equals(thatFile)) {
525                        if (!thisFile.endsWith("/")) { //$NON-NLS-1$
526                            if (!thatFile.equals(thisFile + "/")) { //$NON-NLS-1$
527                                return false;
528                            }
529                        } else {
530                            return false;
531                        }
532                    }
533                }
534    
535                //javadoc:3.7
536                if (this.location.getRef() != null) {
537                    if (!this.location.getRef().equals(cs.location.getRef())) {
538                        return false;
539                    }
540                }
541                // ok, every check was made, and they all were successful. 
542                // it's ok to return true.
543            } // if this.location != null
544    
545            // javadoc: a note about CodeSource with null location and null Certs 
546            // is applicable here 
547            return true;
548        }

550        /**
551         * Returns a string containing a concise, human-readable description of the
552         * this {@code CodeSource} including its location, its certificates and its
553         * signers.
554         * 
555         * @return a printable representation for this {@code CodeSource}.
556         * @since Android 1.0
557         */
558        public String toString() {
559            StringBuilder buf = new StringBuilder();
560            buf.append("CodeSource, url="); //$NON-NLS-1$
561            buf.append(location == null ? "<null>" : location.toString()); //$NON-NLS-1$
562    
563            if (certs == null) {
564                buf.append(", <no certificates>"); //$NON-NLS-1$
565            } else {
566                buf.append("\nCertificates [\n"); //$NON-NLS-1$
567                for (int i = 0; i < certs.length; i++) {
568                    buf.append(i + 1).append(") ").append(certs[i]).append("\n"); //$NON-NLS-1$ //$NON-NLS-2$
569                }
570                buf.append("]\n"); //$NON-NLS-1$
571            }
572            if (signers != null) {
573                buf.append("\nCodeSigners [\n"); //$NON-NLS-1$
574                for (int i = 0; i < signers.length; i++) {
575                    buf.append(i + 1).append(") ").append(signers[i]).append("\n"); //$NON-NLS-1$ //$NON-NLS-2$
576                }
577                buf.append("]\n"); //$NON-NLS-1$
578            }
579            return buf.toString();
580        }

582        private void writeObject(ObjectOutputStream oos) throws IOException {
583    
584            oos.defaultWriteObject();
585    
586            if (certs == null || certs.length == 0) {
587                oos.writeInt(0);
588            } else {
589                oos.writeInt(certs.length);
590                for (int i = 0; i < certs.length; i++) {
591                    try {
592                        oos.writeUTF(certs[i].getType());
593                        byte[] data = certs[i].getEncoded();
594                        // hope there are no certificates with 'data==null'
595                        oos.writeInt(data.length);
596                        oos.write(data);
597                    } catch (CertificateEncodingException ex) {
598                        throw (IOException) new IOException(
599                                Messages.getString("security.18")).initCause(ex); //$NON-NLS-1$
600                    }
601                }
602            }
603            if (signers != null && signers.length != 0) {
604                oos.writeObject(signers);
605            }
606        }

608        private void readObject(ObjectInputStream ois) throws IOException,
609                ClassNotFoundException {
610            
611            ois.defaultReadObject();
612            
613            int certsCount = ois.readInt();
614            certs = null;
615            if (certsCount != 0) {
616                certs = new Certificate[certsCount];
617                for (int i = 0; i < certsCount; i++) {
618                    String type = ois.readUTF();
619                    CertificateFactory factory;
620                    try {
621                        factory = CertificateFactory.getInstance(type);
622                    } catch (CertificateException ex) {
623                        throw new ClassNotFoundException(
624                                Messages.getString("security.19", type), //$NON-NLS-1$
625                                ex);
626                    }
627                    int dataLen = ois.readInt();
628                    byte[] data = new byte[dataLen];
629                    ois.readFully(data);
630                    ByteArrayInputStream bais = new ByteArrayInputStream(data);
631                    try {
632                        certs[i] = factory.generateCertificate(bais);
633                    } catch (CertificateException ex) {
634                        throw (IOException) new IOException(
635                                Messages.getString("security.1A")).initCause(ex); //$NON-NLS-1$
636                    }
637                }
638            }
639            try {
640                signers = (CodeSigner[]) ois.readObject();
641            } catch (OptionalDataException ex) {
642                if (!ex.eof) {
643                    throw ex;
644                }
645                // no signers (ex.eof==true <= no data left) is allowed
646            }
647        }
