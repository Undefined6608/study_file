018    package java.security.acl;
019    
020    import java.security.Principal;
021    import java.util.Enumeration;
022    
023    /**
024     * The <i>Access Control List</i> (<b>ACL</b>) interface definition.
025     * <p>
026     * An ACL is a set of {@link AclEntry} objects.
027     * </p>
028     * <p>
029     * An {@code AclEntry} is a list of {@link Permission}s that are granted 
030     * (<i>positive</i>) or denied
031     * (<i>negative</i>) to a {@link Principal}.
032     * </p>
033     * <p>
034     * An {@code Acl} has a list of owners ({@link Owner}) which are principals as well {@code
035     * Principal}. Only those principals which are the {@code Acl}'s owners are allowed to modify the {@code
036     * Acl}.
037     * </p>
038     * <p>
039     * The <i>ACL</i> has to conform to the following rules:
040     * <ul>
041     * <li>For each {@code Principal} there can be only one <i>positive</i> and one
042     * <i>negative</i> {@code AclEntry}.</li>
043     * <li>If the two {@code AclEntry}'s (<i>positive</i> and <i>negative</i>) for a
044     * specific {@code Principal} grant and deny the same {@code Permission} to that
045     * {@code Principal}, then that {@code Permission} is treated as
046     * neither granted nor denied to that {@code Principal}.</li>
047     * <li>Permissions associated with an individual {@code Principal} always
048     * override permissions of the group(s) to which the individual belongs.</li>
049     * <li>If there is no {@code AclEntry} associated with a specific {@code
050     * Principal}, then it is interpreted as an empty list of permissions.</li>
051     * </ul>
052     * </p>
053     * 
054     * @since Android 1.0
055     */
056    public interface Acl extends Owner {
057    
058        /**
059         * Sets the name of this <i>ACL</i> instance.
060         * 
061         * @param caller
062         *            the invoking {@code Principal}.
063         * @param name
064         *            the name to be set.
065         * @throws NotOwnerException
066         *             if the invoking {@code Principal} is not an owner of this
067         *             <i>ACL</i>.
068         * @since Android 1.0
069         */
070        void setName(Principal caller, String name) throws NotOwnerException;
071    
072        /**
073         * Returns the name of this <i>ACL</i> instance.
074         * 
075         * @return the name of this <i>ACL</i> instance.
076         * @since Android 1.0
077         */
078        String getName();
079    
080        /**
081         * Adds an {@code AclEntry} to this <i>ACL</i> instance.
082         * <p>
083         * If the <i>ACL</i> already has an {@code AclEntry} of the same type (<i>
084         * positive</i> or <i>negative</i>) and principal, then the new entry is not added.
085         * </p>
086         * 
087         * @param caller
088         *            the invoking {@code Principal}.
089         * @param entry
090         *            the ACL entry to add.
091         * @return {@code true} if the entry is added, {@code false} if there is already an entry of
092         *         the same type for the same principal
093         * @throws NotOwnerException
094         *             if the invoking {@code Principal} is not an owner of this
095         *             <i>ACL</i>.
096         * @since Android 1.0
097         */
098        boolean addEntry(Principal caller, AclEntry entry) throws NotOwnerException;
099        
100        /**
101         * Removes an {@code AclEntry} from this <i>ACL</i> instance.
102         * 
103         * @param caller
104         *            the invoking {@code Principal}.
105         * @param entry
106         *            the ACL entry to remove.
107         * @return {@code true} if the entry is removed, {@code false} if the entry is not in this
108         *         <i>ACL</i>.
109         * @throws NotOwnerException
110         *             if the invoking {@code Principal} is not an owner of this
111         *             <i>ACL</i>.
112         * @since Android 1.0
113         */
114        boolean removeEntry(Principal caller, AclEntry entry) 
115                    throws NotOwnerException;
116        
117        /**
118         * Returns the set of allowed permissions for the specified {@code
119         * Principal}.
120         * <p>
121         * If the specified principal has no entry in this ACL, an empty set is
122         * returned.
123         * </p>
124         * <p>
125         * The allowed permissions are collected according to the following rules:
126         * <ul>
127         * <li>The two permission lists (<i>positive</i> and <i>negative</i>) of the
128         * principal's groups ({@link Group}) are collected. The positive (granted)
129         * permissions are the union of all group's positive permissions that the
130         * principal belongs to, the negative (denied) permissions are the union of
131         * all group's negative permissions that the principal belongs to. If a
132         * specific permission is in both the positive and the negative list, it is
133         * removed from both lists.</li>
134         * <li>The individual permissions (<i>positive</i> and <i>negative</i>) of
135         * the principal override the group permissions. The positive individual
136         * permissions override the group's negative permissions and the negative
137         * individual permissions override the grpup's positive permissions.</li>
138         * </ul>
139         * </p>
140         * 
141         * @param user
142         *            the principal to get the allowed permissions for.
143         * @return the set of allowed permissions for the specified principal.
144         * @since Android 1.0
145         */
146        Enumeration<Permission> getPermissions(Principal user); 
147        
148        /**
149         * Returns an {@code Enumeration} of the {@code AclEntry} of this
150         * <i>ACL</i>.
151         * 
152         * @return an {@code Enumeration} of the {@code AclEntry} of this
153         *         <i>ACL</i>.
154         * @since Android 1.0
155         */
156        Enumeration<AclEntry> entries();
157        
158        /**
159         * Checks whether the specified principal is granted the specified
160         * permission.
161         * <p>
162         * The list of granted permissions is determined according to the rules
163         * specified by {@code getPermissions}.
164         * </p>
165         * 
166         * @param principal
167         *            the principal the check the permissions for.
168         * @param permission
169         *            the permission to check for the principal.
170         * @return {@code true} if the principal is granted the permission, otherwise {@code false}.
171         * @see #getPermissions(Principal)
172         * @since Android 1.0
173         */
174        boolean checkPermission(Principal principal, Permission permission);
175        
176        /**
177         * Returns the string representation of this ACL.
178         * 
179         * @return the string representation of this ACL.
180         * @since Android 1.0
181         */
182        String toString();
183    }
