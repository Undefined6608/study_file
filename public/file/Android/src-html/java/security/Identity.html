091            this(name);
092            if (scope != null) {

098        /**
099         * Adds a {@code Certificate} to this {@code Identity}.
100         * <p>
101         * If a {@code SecurityManager} is installed, code calling this method needs
102         * the {@code SecurityPermission} {@code addIdentityCertificate} to be
103         * granted, otherwise a {@code SecurityException} will be thrown.
104         * </p>
105         * 
106         * @param certificate
107         *            the {@code Certificate} to be added to this {@code Identity}.
108         * @throws KeyManagementException
109         *             if the certificate is not valid.
110         * @throws SecurityException
111         *             if a {@code SecurityManager} is installed and the caller does
112         *             not have permission to invoke this method.
113         * @since Android 1.0
114         */
115        public void addCertificate(Certificate certificate)
116                throws KeyManagementException {
117            SecurityManager sm = System.getSecurityManager();
118            if (sm != null) {
119                sm.checkSecurityAccess("addIdentityCertificate"); //$NON-NLS-1$
120            }
121            PublicKey certPK = certificate.getPublicKey();
122            if (publicKey != null) {
123                if (!checkKeysEqual(publicKey, certPK)) {
124                    throw new KeyManagementException(Messages.getString("security.13")); //$NON-NLS-1$
125                }
126            } else {
127                publicKey = certPK;
128            }
129            if (certificates == null) {
130                certificates = new Vector<Certificate>();
131            }
132            certificates.add(certificate);
133        }

138        private static boolean checkKeysEqual(PublicKey pk1, PublicKey pk2) {
139            // first, they should have the same format
140            // second, their encoded form must be the same
141    
142            // assert(pk1 != null);
143            // assert(pk2 != null);
144    
145            String format1 = pk1.getFormat();
146            String format2;
147            if ((pk2 == null)
148                    || (((format2 = pk2.getFormat()) != null) ^ (format1 != null))
149                    || ((format1 != null) && !format1.equals(format2))) {
150                return false;
151            }
152    
153            return Arrays.equals(pk1.getEncoded(), pk2.getEncoded());
154        }

159        /**
160         * Removes the specified {@code Certificate} from this {@code Identity}.
161         * <p>
162         * If a {@code SecurityManager} is installed, code calling this method needs
163         * the {@code SecurityPermission} {@code "removeIdentityCertificate"} to be
164         * granted, otherwise a {@code SecurityException} will be thrown.
165         * <p>
166         * 
167         * @param certificate
168         *            the {@code Certificate} to be removed.
169         * @throws KeyManagementException
170         *             if the certificate is not found.
171         * @throws SecurityException
172         *             if a {@code SecurityManager} is installed and the caller does
173         *             not have permission to invoke this method.
174         * @since Android 1.0
175         */
176        public void removeCertificate(Certificate certificate)
177                throws KeyManagementException {
178            SecurityManager sm = System.getSecurityManager();
179            if (sm != null) {
180                sm.checkSecurityAccess("removeIdentityCertificate"); //$NON-NLS-1$
181            }
182            if (certificates != null) {
183                // BEGIN android-added
184                if (!certificates.contains(certificate)) {
185                    throw new KeyManagementException("Certificate not found");
186                }
187                // END android-added
188                certificates.removeElement(certificate);
189            }
190        }

195        /**
196         * Returns the certificates for this {@code Identity}. External
197         * modifications of the returned array has no impact on this {@code
198         * Identity}.
199         * 
200         * @return the {@code Certificates} for this {@code Identity}
201         * @since Android 1.0
202         */
203        public Certificate[] certificates() {
204            if (certificates == null) {
205                return new Certificate[0];
206            }
207            Certificate[] ret = new Certificate[certificates.size()];
208            certificates.copyInto(ret);
209            return ret;
210        }

215        /**
216         * Compares the specified {@code Identity} with this {@code Identity} for
217         * equality and returns {@code true} if the specified object is equal,
218         * {@code false} otherwise.
219         * <p>
220         * To be equal, two {@code Identity} objects need to have the same name and
221         * the same public keys.
222         * </p>
223         * 
224         * @param identity
225         *            the identity to check for equality.
226         * @return {@code true} if the {@code Identity} objects are equal, {@code
227         *         false} otherwise.
228         * @since Android 1.0
229         */
230        protected boolean identityEquals(Identity identity) {
231            if (!name.equals(identity.name)) {
232                return false;
233            }
234    
235            if (publicKey == null) {
236                return (identity.publicKey == null);
237            }
238    
239            return checkKeysEqual(publicKey, identity.publicKey);
240        }

278        /**
279         * Sets the specified {@code PublicKey} to this {@code Identity}.
280         * <p>
281         * If a {@code SecurityManager} is installed, code calling this method needs
282         * the {@code SecurityPermission} {@code setIdentityPublicKey} to be
283         * granted, otherwise a {@code SecurityException} will be thrown.
284         * </p>
285         * 
286         * @param key
287         *            the {@code PublicKey} to be set.
288         * @throws KeyManagementException
289         *             if another {@code Identity} in the same scope as this {@code
290         *             Identity} already has the same {@code PublicKey}.
291         * @throws SecurityException
292         *             if a {@code SecurityManager} is installed and the caller does
293         *             not have permission to invoke this method.
294         * @since Android 1.0
295         */
296        public void setPublicKey(PublicKey key) throws KeyManagementException {
297            SecurityManager sm = System.getSecurityManager();
298            if (sm != null) {
299                sm.checkSecurityAccess("setIdentityPublicKey"); //$NON-NLS-1$
300            }
301            // this check does not always work  
302            if ((scope != null) && (key != null)) {
303                Identity i = scope.getIdentity(key);
304                //System.out.println("###DEBUG## Identity: "+i);
305                if ((i != null) && (i != this)) {
306                    throw new KeyManagementException(Messages.getString("security.14")); //$NON-NLS-1$
307                }
308            }
309            this.publicKey = key;
310            certificates = null;
311        }

367        /**
368         * Compares the specified object with this {@code Identity} for equality and
369         * returns {@code true} if the specified object is equal, {@code false}
370         * otherwise. {@code Identity} objects are considered equal, if they have
371         * the same name and are in the same scope.
372         * 
373         * @param obj
374         *            object to be compared for equality with this {@code
375         *            Identity}.
376         * @return {@code true} if the specified object is equal to this {@code
377         *         Identity}, otherwise {@code false}.
378         * @since Android 1.0
379         */
380        public final boolean equals(Object obj) {
381            if (this == obj) {
382                return true;
383            }
384            if (!(obj instanceof Identity)) {
385                return false;
386            }
387            Identity i = (Identity) obj;
388            if ((name == i.name || (name != null && name.equals(i.name)))
389                    && (scope == i.scope || (scope != null && scope.equals(i.scope)))) {
390                return true;
391            }
392            return identityEquals(i);
393        }

412         * Returns the hash code value for this {@code Identity}. Returns the same
413         * hash code for {@code Identity}s that are equal to each other as required
414         * by the general contract of {@link Object#hashCode}.
415         * 
416         * @return the hash code value for this {@code Identity}.
417         * @see Object#equals(Object)
418         * @see Identity#equals(Object)
419         * @since Android 1.0
420         */
421        public int hashCode() {
422            int hash = 0;
423            if (name != null) {
424                hash += name.hashCode();
425            }
426            if (scope != null) {
427                hash += scope.hashCode();
428            }
429            return hash;
430        }

435        /**
436         * Returns a string containing a concise, human-readable description of the
437         * this {@code Identity} including its name and its scope.
438         * <p>
439         * If a {@code SecurityManager} is installed, code calling this method
440         * needs the {@code SecurityPermission} {@code printIdentity} to be granted,
441         * otherwise a {@code SecurityException} will be thrown.
442         * </p>
443         * 
444         * @return a printable representation for this {@code Identity}.
445         * @throws SecurityException
446         *             if a {@code SecurityManager} is installed and the caller does
447         *             not have permission to invoke this method.
448         * @since Android 1.0
449         */
450        public String toString() {
451            SecurityManager sm = System.getSecurityManager();
452            if (sm != null) {
453                sm.checkSecurityAccess("printIdentity"); //$NON-NLS-1$
454            }
455            String s = (this.name == null? "" : this.name);
456            if (scope != null) {
457                s += " [" + scope.getName() + "]"; //$NON-NLS-1$ //$NON-NLS-2$
458            }
459            return s;
460        }
