088        /**
089         * Creates a new unconnected socket using the given proxy type. When a
090         * {@code SocketImplFactory} is defined it creates the internal socket
091         * implementation, otherwise the default socket implementation will be used
092         * for this socket.
093         * <p>
094         * Example that will create a socket connection through a {@code SOCKS}
095         * proxy server: <br>
096         * {@code Socket sock = new Socket(new Proxy(Proxy.Type.SOCKS, new
097         * InetSocketAddress("test.domain.org", 2130)));}
098         * </p>
099         * 
100         * @param proxy
101         *            the specified proxy for this socket.
102         * @throws IllegalArgumentException
103         *             if the argument {@code proxy} is {@code null} or of an
104         *             invalid type.
105         * @throws SecurityException
106         *             if a security manager exists and it denies the permission to
107         *             connect to the given proxy.
108         * @see SocketImplFactory
109         * @see SocketImpl
110         * @since Android 1.0
111         */
112        public Socket(Proxy proxy) {
113            if (null == proxy || Proxy.Type.HTTP == proxy.type()) {
114                // KA023=Proxy is null or invalid type
115                throw new IllegalArgumentException(Msg.getString("KA023")); //$NON-NLS-1$
116            }
117            InetSocketAddress address = (InetSocketAddress) proxy.address();
118            if (null != address) {
119                InetAddress addr = address.getAddress();
120                String host;
121                if (null != addr) {
122                    host = addr.getHostAddress();
123                } else {
124                    host = address.getHostName();
125                }
126                int port = address.getPort();
127                checkConnectPermission(host, port);
128            }
129            impl = factory != null ? factory.createSocketImpl()
130                    : SocketImplProvider.getSocketImpl(proxy);
131            this.proxy = proxy;
132        }

319        /**
320         * Checks whether the connection destination satisfies the security policy
321         * and the validity of the port range.
322         * 
323         * @param destAddr
324         *            the destination host address.
325         * @param dstPort
326         *            the port on the destination host.
327         */
328        void checkDestination(InetAddress destAddr, int dstPort) {
329            if (dstPort < 0 || dstPort > 65535) {
330                throw new IllegalArgumentException(Msg.getString("K0032")); //$NON-NLS-1$
331            }
332            // BEGIN android-changed
333            checkConnectPermission(destAddr.getHostAddress(), dstPort);
334            // END android-changed
335        }

338         * Checks whether the connection destination satisfies the security policy.
339         * 
340         * @param hostname
341         *            the destination hostname.
342         * @param dstPort
343         *            the port on the destination host.
344         */
345        private void checkConnectPermission(String hostname, int dstPort) {
346            SecurityManager security = System.getSecurityManager();
347            if (security != null) {
348                security.checkConnect(hostname, dstPort);
349            }

628        /**
629         * Sets the state of the {@code SocketOptions.SO_LINGER} with the given
630         * timeout in seconds. The timeout value for this option is silently limited
631         * to the maximum of {@code 65535}.
632         * 
633         * @param on
634         *            the state whether this option is enabled or not.
635         * @param timeout
636         *            the linger timeout value in seconds.
637         * @throws SocketException
638         *             if an error occurs while setting the option.
639         * @see SocketOptions#SO_LINGER
640         * @since Android 1.0
641         */
642        public void setSoLinger(boolean on, int timeout) throws SocketException {
643            checkClosedAndCreate(true);
644            if (on && timeout < 0) {
645                throw new IllegalArgumentException(Msg.getString("K0045")); //$NON-NLS-1$
646            }
647            // BEGIN android-changed
648            /*
649             * The spec indicates that the right way to turn off an option
650             * is to pass Boolean.FALSE, so that's what we do here.
651             */
652            if (on) {
653                if (timeout > 65535) {
654                    timeout = 65535;
655                }
656                impl.setOption(SocketOptions.SO_LINGER, Integer.valueOf(timeout));
657            } else {
658                impl.setOption(SocketOptions.SO_LINGER, Boolean.FALSE);
659            }
660            // END android-changed
661        }

714         *             operations.
715         */
716        void startupSocket(InetAddress dstAddress, int dstPort,
717                InetAddress localAddress, int localPort, boolean streaming)
718                throws IOException {
719    
720            if (localPort < 0 || localPort > 65535) {
721                throw new IllegalArgumentException(Msg.getString("K0046")); //$NON-NLS-1$
722            }
723    
724            InetAddress addr = localAddress == null ? InetAddress.ANY
725                    : localAddress;
726            synchronized (this) {
727                impl.create(streaming);
728                isCreated = true;
729                try {
730                    if (!streaming || !NetUtil.usingSocks(proxy)) {
731                        impl.bind(addr, localPort);
732                    }
733                    isBound = true;
734                    impl.connect(dstAddress, dstPort);
735                    isConnected = true;
736                } catch (IOException e) {
737                    impl.close();
738                    throw e;
739                }
740            }
741        }

798        /**
799         * Checks whether the socket is closed, and throws an exception. Otherwise
800         * creates the underlying SocketImpl.
801         * 
802         * @throws SocketException
803         *             if the socket is closed.
804         */
805        private void checkClosedAndCreate(boolean create) throws SocketException {
806            if (isClosed()) {
807                throw new SocketException(Msg.getString("K003d")); //$NON-NLS-1$
808            }
809            if (!create) {
810                if (!isConnected()) {
811                    throw new SocketException(Msg.getString("K0320")); //$NON-NLS-1$
812                    // a connected socket must be created
813                }
814    
815                /*
816                 * return directly to fix a possible bug, if !create, should return
817                 * here
818                 */
819                return;
820            }
821            if (isCreated) {
822                return;
823            }
824            synchronized (this) {
825                if (isCreated) {
826                    return;
827                }
828                try {
829                    impl.create(true);
830                } catch (SocketException e) {
831                    throw e;
832                } catch (IOException e) {
833                    throw new SocketException(e.toString());
834                }
835                isCreated = true;
836            }
837        }

899        /**
900         * Binds this socket to the given local host address and port specified by
901         * the SocketAddress {@code localAddr}. If {@code localAddr} is set to
902         * {@code null}, this socket will be bound to an available local address on
903         * any free port.
904         * 
905         * @param localAddr
906         *            the specific address and port on the local machine to bind to.
907         * @throws IllegalArgumentException
908         *             if the given SocketAddress is invalid or not supported.
909         * @throws IOException
910         *             if the socket is already bound or an error occurs while
911         *             binding.
912         * @since Android 1.0
913         */
914        public void bind(SocketAddress localAddr) throws IOException {
915            checkClosedAndCreate(true);
916            if (isBound()) {
917                throw new BindException(Msg.getString("K0315")); //$NON-NLS-1$
918            }
919    
920            int port = 0;
921            InetAddress addr = InetAddress.ANY;
922            if (localAddr != null) {
923                if (!(localAddr instanceof InetSocketAddress)) {
924                    throw new IllegalArgumentException(Msg.getString(
925                            "K0316", localAddr.getClass())); //$NON-NLS-1$
926                }
927                InetSocketAddress inetAddr = (InetSocketAddress) localAddr;
928                if ((addr = inetAddr.getAddress()) == null) {
929                    throw new SocketException(Msg.getString(
930                            "K0317", inetAddr.getHostName())); //$NON-NLS-1$
931                }
932                port = inetAddr.getPort();
933            }
934    
935            synchronized (this) {
936                try {
937                    impl.bind(addr, port);
938                    isBound = true;
939                } catch (IOException e) {
940                    impl.close();
941                    throw e;
942                }
943            }
944        }

963        /**
964         * Connects this socket to the given remote host address and port specified
965         * by the SocketAddress {@code remoteAddr} with the specified timeout. The
966         * connecting method will block until the connection is established or an
967         * error occurred.
968         * 
969         * @param remoteAddr
970         *            the address and port of the remote host to connect to.
971         * @param timeout
972         *            the timeout value in milliseconds or {@code 0} for an infinite
973         *            timeout.
974         * @throws IllegalArgumentException
975         *             if the given SocketAddress is invalid or not supported or the
976         *             timeout value is negative.
977         * @throws IOException
978         *             if the socket is already connected or an error occurs while
979         *             connecting.
980         * @since Android 1.0
981         */
982        public void connect(SocketAddress remoteAddr, int timeout)
983                throws IOException {
984            checkClosedAndCreate(true);
985            if (timeout < 0) {
986                throw new IllegalArgumentException(Msg.getString("K0036")); //$NON-NLS-1$
987            }
988            if (isConnected()) {
989                throw new SocketException(Msg.getString("K0079")); //$NON-NLS-1$
990            }
991            if (remoteAddr == null) {
992                throw new IllegalArgumentException(Msg.getString("K0318")); //$NON-NLS-1$
993            }
994    
995            if (!(remoteAddr instanceof InetSocketAddress)) {
996                throw new IllegalArgumentException(Msg.getString(
997                        "K0316", remoteAddr.getClass())); //$NON-NLS-1$
998            }
999            InetSocketAddress inetAddr = (InetSocketAddress) remoteAddr;
1000            InetAddress addr;
1001            if ((addr = inetAddr.getAddress()) == null) {
1002                throw new UnknownHostException(Msg.getString("K0317", remoteAddr));//$NON-NLS-1$
1003            }
1004            int port = inetAddr.getPort();
1005    
1006            checkDestination(addr, port);
1007            synchronized (connectLock) {
1008                try {
1009                    if (!isBound()) {
1010                        // socket allready created at this point by earlier call or
1011                        // checkClosedAndCreate this caused us to lose socket
1012                        // options on create
1013                        // impl.create(true);
1014                        if (!NetUtil.usingSocks(proxy)) {
1015                            impl.bind(InetAddress.ANY, 0);
1016                        }
1017                        isBound = true;
1018                    }
1019                    impl.connect(remoteAddr, timeout);
1020                    isConnected = true;
1021                } catch (IOException e) {
1022                    impl.close();
1023                    throw e;
1024                }
1025            }
1026        }
