405        /**
406         * Returns the permission object (in this case {@code SocketPermission})
407         * with the host and the port number as the target name and {@code
408         * "resolve, connect"} as the action list. If the port number of this URL
409         * instance is lower than {@code 0} the port will be set to {@code 80}.
410         * 
411         * @return the permission object required for this connection.
412         * @throws IOException
413         *             if an IO exception occurs during the creation of the
414         *             permission object.
415         * @since Android 1.0
416         */
417        @Override
418        public java.security.Permission getPermission() throws IOException {
419            int port = url.getPort();
420            if (port < 0) {
421                port = 80;
422            }
423            return new SocketPermission(url.getHost() + ":" + port, //$NON-NLS-1$
424                    "connect, resolve"); //$NON-NLS-1$
425        }

441        /**
442         * Returns the response code returned by the remote HTTP server.
443         * 
444         * @return the response code, -1 if no valid response code.
445         * @throws IOException
446         *             if there is an IO error during the retrieval.
447         * @see #getResponseMessage
448         * @since Android 1.0
449         */
450        public int getResponseCode() throws IOException {
451            // Call getInputStream() first since getHeaderField() doesn't return
452            // exceptions
453            getInputStream();
454            String response = getHeaderField(0);
455            if (response == null) {
456                return -1;
457            }
458            response = response.trim();
459            int mark = response.indexOf(" ") + 1; //$NON-NLS-1$
460            if (mark == 0) {
461                return -1;
462            }
463            int last = mark + 3;
464            if (last > response.length()) {
465                last = response.length();
466            }
467            responseCode = Integer.parseInt(response.substring(mark, last));
468            if (last + 1 <= response.length()) {
469                responseMessage = response.substring(last + 1);
470            }
471            return responseCode;
472        }
473    
474        /**
475         * Returns the response message returned by the remote HTTP server.
476         * 
477         * @return the response message. {@code null} if no such response exists.
478         * @throws IOException
479         *             if there is an error during the retrieval.
480         * @see #getResponseCode()
481         * @since Android 1.0
482         */
483        public String getResponseMessage() throws IOException {
484            if (responseMessage != null) {
485                return responseMessage;
486            }
487            getResponseCode();
488            return responseMessage;
489        }

503            if (security != null) {
504                security.checkSetFactory();
505            }
506            followRedirects = auto;
507        }

509        /**
510         * Sets the request command which will be sent to the remote HTTP server.
511         * This method can only be called before the connection is made.
512         * 
513         * @param method
514         *            the string representing the method to be used.
515         * @throws ProtocolException
516         *             if this is called after connected, or the method is not
517         *             supported by this HTTP implementation.
518         * @see #getRequestMethod()
519         * @see #method
520         * @since Android 1.0
521         */
522        public void setRequestMethod(String method) throws ProtocolException {
523            if (connected) {
524                throw new ProtocolException(Msg.getString("K0037")); //$NON-NLS-1$
525            }
526            for (int i = 0; i < methodTokens.length; i++) {
527                if (methodTokens[i].equals(method)) {
528                    // if there is a supported method that matches the desired
529                    // method, then set the current method and return
530                    this.method = methodTokens[i];
531                    return;
532                }
533            }
534            // if none matches, then throw ProtocolException
535            throw new ProtocolException();
536        }

589        /**
590         * If the length of a HTTP request body is known ahead, sets fixed length to
591         * enable streaming without buffering. Sets after connection will cause an
592         * exception.
593         * 
594         * @see #setChunkedStreamingMode
595         * @param contentLength
596         *            the fixed length of the HTTP request body.
597         * @throws IllegalStateException
598         *             if already connected or an other mode already set.
599         * @throws IllegalArgumentException
600         *             if {@code contentLength} is less than zero.
601         * @since Android 1.0
602         */
603        public void setFixedLengthStreamingMode(int contentLength) {
604            if (super.connected) {
605                throw new IllegalStateException(Msg.getString("K0079")); //$NON-NLS-1$
606            }
607            if (0 < chunkLength) {
608                throw new IllegalStateException(Msg.getString("KA003")); //$NON-NLS-1$
609            }
610            if (0 > contentLength) {
611                throw new IllegalArgumentException(Msg.getString("K0051")); //$NON-NLS-1$
612            }
613            this.fixedContentLength = contentLength;
614        }
615    
616        /**
617         * If the length of a HTTP request body is NOT known ahead, enable chunked
618         * transfer encoding to enable streaming with buffering. Notice that not all
619         * http servers support this mode. Sets after connection will cause an
620         * exception.
621         * 
622         * @see #setFixedLengthStreamingMode
623         * @param chunklen
624         *            the length of a chunk.
625         * @throws IllegalStateException
626         *             if already connected or an other mode already set.
627         * @since Android 1.0
628         */
629        public void setChunkedStreamingMode(int chunklen) {
630            if (super.connected) {
631                throw new IllegalStateException(Msg.getString("K0079")); //$NON-NLS-1$
632            }
633            if (0 <= fixedContentLength) {
634                throw new IllegalStateException(Msg.getString("KA003")); //$NON-NLS-1$
635            }
636            if (0 >= chunklen) {
637                chunkLength = DEFAULT_CHUNK_LENGTH;
638            } else {
639                chunkLength = chunklen;
640            }
641        }
