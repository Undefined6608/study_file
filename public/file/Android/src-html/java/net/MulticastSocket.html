067        /**
068         * Gets the network address used by this socket. This is useful on
069         * multihomed machines.
070         * 
071         * @return the address of the network interface through which the datagram
072         *         packets are sent or received.
073         * @exception SocketException
074         *                if an error occurs while getting the interface address.
075         * @since Android 1.0
076         */
077        public InetAddress getInterface() throws SocketException {
078            checkClosedAndBind(false);
079            if (interfaceSet == null) {
080                InetAddress ipvXaddress = (InetAddress) impl
081                        .getOption(SocketOptions.IP_MULTICAST_IF);
082                if (ipvXaddress.isAnyLocalAddress()) {
083                    // the address was not set at the IPV4 level so check the IPV6
084                    // level
085                    NetworkInterface theInterface = getNetworkInterface();
086                    if (theInterface != null) {
087                        Enumeration<InetAddress> addresses = theInterface
088                                .getInetAddresses();
089                        if (addresses != null) {
090                            while (addresses.hasMoreElements()) {
091                                InetAddress nextAddress = addresses.nextElement();
092                                if (nextAddress instanceof Inet6Address) {
093                                    return nextAddress;
094                                }
095                            }
096                        }
097                    }
098                }
099                return ipvXaddress;
100            }
101            return interfaceSet;
102        }

104        /**
105         * Gets the network interface used by this socket. This is useful on
106         * multihomed machines.
107         * 
108         * @return the network interface used by this socket or {@code null} if no
109         *         interface is set.
110         * @exception SocketException
111         *                if an error occurs while getting the interface.
112         * @since Android 1.0
113         */
114        public NetworkInterface getNetworkInterface() throws SocketException {
115            checkClosedAndBind(false);
116    
117            // check if it is set at the IPV6 level. If so then use that. Otherwise
118            // do it at the IPV4 level
119            Integer theIndex = Integer.valueOf(0);
120            try {
121                theIndex = (Integer) impl.getOption(SocketOptions.IP_MULTICAST_IF2);
122            } catch (SocketException e) {
123                // we may get an exception if IPV6 is not enabled.
124            }
125    
126            if (theIndex.intValue() != 0) {
127                Enumeration<NetworkInterface> theInterfaces = NetworkInterface
128                        .getNetworkInterfaces();
129                while (theInterfaces.hasMoreElements()) {
130                    NetworkInterface nextInterface = theInterfaces.nextElement();
131                    if (nextInterface.getIndex() == theIndex.intValue()) {
132                        return nextInterface;
133                    }
134                }
135            }
136    
137            // ok it was not set at the IPV6 level so try at the IPV4 level
138            InetAddress theAddress = (InetAddress) impl
139                    .getOption(SocketOptions.IP_MULTICAST_IF);
140            if (theAddress != null) {
141                if (!theAddress.isAnyLocalAddress()) {
142                    return NetworkInterface.getByInetAddress(theAddress);
143                }
144    
145                // not set as we got the any address so return a dummy network
146                // interface with only the any address. We do this to be
147                // compatible
148                InetAddress theAddresses[] = new InetAddress[1];
149                if ((Socket.preferIPv4Stack() == false)
150                        && (InetAddress.preferIPv6Addresses() == true)) {
151                    theAddresses[0] = Inet6Address.ANY;
152                } else {
153                    theAddresses[0] = InetAddress.ANY;
154                }
155                return new NetworkInterface(null, null, theAddresses,
156                        NetworkInterface.UNSET_INTERFACE_INDEX);
157            }
158    
159            // ok not set at all so return null
160            return null;
161        }

197        /**
198         * Adds this socket to the specified multicast group. A socket must join a
199         * group before data may be received. A socket may be a member of multiple
200         * groups but may join any group only once.
201         * 
202         * @param groupAddr
203         *            the multicast group to be joined.
204         * @exception IOException
205         *                if an error occurs while joining a group.
206         * @since Android 1.0
207         */
208        public void joinGroup(InetAddress groupAddr) throws IOException {
209            checkClosedAndBind(false);
210            if (!groupAddr.isMulticastAddress()) {
211                throw new IOException(Msg.getString("K0039")); //$NON-NLS-1$
212            }
213            SecurityManager security = System.getSecurityManager();
214            if (security != null) {
215                security.checkMulticast(groupAddr);
216            }
217            impl.join(groupAddr);
218        }

220        /**
221         * Adds this socket to the specified multicast group. A socket must join a
222         * group before data may be received. A socket may be a member of multiple
223         * groups but may join any group only once.
224         * 
225         * @param groupAddress
226         *            the multicast group to be joined.
227         * @param netInterface
228         *            the network interface on which the datagram packets will be
229         *            received.
230         * @exception IOException
231         *                if the specified address is not a multicast address.
232         * @exception SecurityException
233         *                if the caller is not authorized to join the group.
234         * @exception IllegalArgumentException
235         *                if no multicast group is specified.
236         * @since Android 1.0
237         */
238        public void joinGroup(SocketAddress groupAddress,
239                NetworkInterface netInterface) throws IOException {
240            checkClosedAndBind(false);
241            if (null == groupAddress) {
242                throw new IllegalArgumentException(Msg.getString("K0318")); //$NON-NLS-1$
243            }
244    
245            if ((netInterface != null) && (netInterface.getFirstAddress() == null)) {
246                // this is ok if we could set it at the
247                throw new SocketException(Msg.getString("K0335")); //$NON-NLS-1$
248            }
249    
250            if (!(groupAddress instanceof InetSocketAddress)) {
251                throw new IllegalArgumentException(Msg.getString(
252                        "K0316", groupAddress.getClass())); //$NON-NLS-1$
253            }
254    
255            InetAddress groupAddr = ((InetSocketAddress) groupAddress).getAddress();
256    
257            if (groupAddr == null) {
258                throw new SocketException(Msg.getString("K0331")); //$NON-NLS-1$
259            }
260    
261            if (!groupAddr.isMulticastAddress()) {
262                throw new IOException(Msg.getString("K0039")); //$NON-NLS-1$
263            }
264    
265            SecurityManager security = System.getSecurityManager();
266            if (security != null) {
267                security.checkMulticast(groupAddr);
268            }
269            impl.joinGroup(groupAddress, netInterface);
270        }

344        /**
345         * Send the packet on this socket. The packet must satisfy the security
346         * policy before it may be sent.
347         * 
348         * @param pack
349         *            the {@code DatagramPacket} to send
350         * @param ttl
351         *            the TTL setting for this transmission, overriding the socket
352         *            default
353         * @exception IOException
354         *                if an error occurs while sending data or setting options.
355         * @deprecated use {@link #setTimeToLive}.
356         * @since Android 1.0
357         */
358        @Deprecated
359        public void send(DatagramPacket pack, byte ttl) throws IOException {
360            checkClosedAndBind(false);
361            InetAddress packAddr = pack.getAddress();
362            SecurityManager security = System.getSecurityManager();
363            if (security != null) {
364                if (packAddr.isMulticastAddress()) {
365                    security.checkMulticast(packAddr, ttl);
366                } else {
367                    security.checkConnect(packAddr.getHostName(), pack.getPort());
368                }
369            }
370            int currTTL = getTimeToLive();
371            if (packAddr.isMulticastAddress() && (byte) currTTL != ttl) {
372                try {
373                    setTimeToLive(ttl & 0xff);
374                    impl.send(pack);
375                } finally {
376                    setTimeToLive(currTTL);
377                }
378            } else {
379                impl.send(pack);
380            }
381        }

383        /**
384         * Sets the interface address used by this socket. This allows to send
385         * multicast packets on a different interface than the default interface of
386         * the local system. This is useful on multihomed machines.
387         * 
388         * @param addr
389         *            the multicast interface network address to set.
390         * @exception SocketException
391         *                if an error occurs while setting the network interface
392         *                address option.
393         * @since Android 1.0
394         */
395        public void setInterface(InetAddress addr) throws SocketException {
396            checkClosedAndBind(false);
397            if (addr == null) {
398                throw new NullPointerException();
399            }
400            if (addr.isAnyLocalAddress()) {
401                impl.setOption(SocketOptions.IP_MULTICAST_IF, InetAddress.ANY);
402            } else if (addr instanceof Inet4Address) {
403                impl.setOption(SocketOptions.IP_MULTICAST_IF, addr);
404                // keep the address used to do the set as we must return the same
405                // value and for IPv6 we may not be able to get it back uniquely
406                interfaceSet = addr;
407            }
408    
409            /*
410             * now we should also make sure this works for IPV6 get the network
411             * interface for the address and set the interface using its index
412             * however if IPV6 is not enabled then we may get an exception. if IPV6
413             * is not enabled
414             */
415            NetworkInterface theInterface = NetworkInterface.getByInetAddress(addr);
416            if ((theInterface != null) && (theInterface.getIndex() != 0)) {
417                try {
418                    impl.setOption(SocketOptions.IP_MULTICAST_IF2, Integer
419                            .valueOf(theInterface.getIndex()));
420                } catch (SocketException e) {
421                    // Ignored
422                }
423            } else if (addr.isAnyLocalAddress()) {
424                try {
425                    impl.setOption(SocketOptions.IP_MULTICAST_IF2, Integer
426                            .valueOf(0));
427                } catch (SocketException e) {
428                    // Ignored
429                }
430            } else if (addr instanceof Inet6Address) {
431                throw new SocketException(Msg.getString("K0338")); //$NON-NLS-1$
432            }
433        }

435        /**
436         * Sets the network interface used by this socket. This is useful for
437         * multihomed machines.
438         * 
439         * @param netInterface
440         *            the multicast network interface to set.
441         * @exception SocketException
442         *                if an error occurs while setting the network interface
443         *                option.
444         * @since Android 1.0
445         */
446        public void setNetworkInterface(NetworkInterface netInterface)
447                throws SocketException {
448    
449            checkClosedAndBind(false);
450    
451            if (netInterface == null) {
452                // throw a socket exception indicating that we do not support this
453                throw new SocketException(Msg.getString("K0334")); //$NON-NLS-1$
454            }
455    
456            InetAddress firstAddress = netInterface.getFirstAddress();
457            if (firstAddress == null) {
458                // this is ok if we could set it at the
459                throw new SocketException(Msg.getString("K0335")); //$NON-NLS-1$
460            }
461    
462            if (netInterface.getIndex() == NetworkInterface.UNSET_INTERFACE_INDEX) {
463                // set the address using IP_MULTICAST_IF to make sure this
464                // works for both IPV4 and IPV6
465                impl.setOption(SocketOptions.IP_MULTICAST_IF, InetAddress.ANY);
466    
467                try {
468                    // we have the index so now we pass set the interface
469                    // using IP_MULTICAST_IF2. This is what is used to set
470                    // the interface on systems which support IPV6
471                    impl.setOption(SocketOptions.IP_MULTICAST_IF2, Integer
472                            .valueOf(NetworkInterface.NO_INTERFACE_INDEX));
473                } catch (SocketException e) {
474                    // for now just do this, -- could be narrowed?
475                }
476            }
477    
478            /*
479             * Now try to set using IPV4 way. However, if interface passed in has no
480             * IP addresses associated with it then we cannot do it. first we have
481             * to make sure there is an IPV4 address that we can use to call set
482             * interface otherwise we will not set it
483             */
484            Enumeration<InetAddress> theAddresses = netInterface.getInetAddresses();
485            boolean found = false;
486            firstAddress = null;
487            while ((theAddresses.hasMoreElements()) && (found != true)) {
488                InetAddress theAddress = theAddresses.nextElement();
489                if (theAddress instanceof Inet4Address) {
490                    firstAddress = theAddress;
491                    found = true;
492                }
493            }
494            if (netInterface.getIndex() == NetworkInterface.NO_INTERFACE_INDEX) {
495                // the system does not support IPV6 and does not provide
496                // indexes for the network interfaces. Just pass in the
497                // first address for the network interface
498                if (firstAddress != null) {
499                    impl.setOption(SocketOptions.IP_MULTICAST_IF, firstAddress);
500                } else {
501                    /*
502                     * we should never get here as there should not be any network
503                     * interfaces which have no IPV4 address and which does not have
504                     * the network interface index not set correctly
505                     */
506                    throw new SocketException(Msg.getString("K0335")); //$NON-NLS-1$
507                }
508            } else {
509                // set the address using IP_MULTICAST_IF to make sure this
510                // works for both IPV4 and IPV6
511                if (firstAddress != null) {
512                    impl.setOption(SocketOptions.IP_MULTICAST_IF, firstAddress);
513                }
514    
515                try {
516                    // we have the index so now we pass set the interface
517                    // using IP_MULTICAST_IF2. This is what is used to set
518                    // the interface on systems which support IPV6
519                    impl.setOption(SocketOptions.IP_MULTICAST_IF2, Integer
520                            .valueOf(netInterface.getIndex()));
521                } catch (SocketException e) {
522                    // for now just do this -- could be narrowed?
523                }
524            }
525    
526            interfaceSet = null;
527        }

530         * Sets the time-to-live (TTL) for multicast packets sent on this socket.
531         * Valid TTL values are between 0 and 255 inclusive.
532         * 
533         * @param ttl
534         *            the default time-to-live field value for packets sent on this
535         *            socket. {@code 0 <= ttl <= 255}.
536         * @exception IOException

540        public void setTimeToLive(int ttl) throws IOException {
541            checkClosedAndBind(false);
542            if (ttl < 0 || ttl > 255) {
543                throw new IllegalArgumentException(Msg.getString("K003c")); //$NON-NLS-1$
544            }
545            impl.setTimeToLive(ttl);
546        }

567        @Override
568        synchronized void createSocket(int aPort, InetAddress addr)
569                throws SocketException {
570            impl = factory != null ? factory.createDatagramSocketImpl()
571                    : SocketImplProvider.getMulticastSocketImpl();
572            impl.create();
573            try {
574                // the required default options are now set in the VM where they
575                // should be
576                impl.bind(aPort, addr);
577                isBound = true;
578            } catch (SocketException e) {
579                close();
580                throw e;
581            }
582        }

602        /**
603         * Gets the state of the {@code SocketOptions.IP_MULTICAST_LOOP}.
604         * 
605         * @return {@code true} if the IP multicast loop is enabled, {@code false}
606         *         otherwise.
607         * @throws SocketException
608         *             if the socket is closed or the option is invalid.
609         * @since Android 1.0
610         */
611        public boolean getLoopbackMode() throws SocketException {
612            checkClosedAndBind(false);
613            return !((Boolean) impl.getOption(SocketOptions.IP_MULTICAST_LOOP))
614                    .booleanValue();
615        }

623         * @throws SocketException
624         *             if the socket is closed or the option is invalid.
625         * @since Android 1.0
626         */
627        public void setLoopbackMode(boolean loop) throws SocketException {
628            checkClosedAndBind(false);
629            impl.setOption(SocketOptions.IP_MULTICAST_LOOP, loop ? Boolean.FALSE
630                    : Boolean.TRUE);
