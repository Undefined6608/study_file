017    package java.net;
018    
019    import java.io.IOException;
020    import java.io.OutputStream;
021    
022    /**
023     * {@code CacheRequest} is a kind of channel for storing resource data in the
024     * {@code ResponseCache}. A protocol handler calls the {@code OutputStream}
025     * which is provided by the {@code CacheRequest} object, to store the resource
026     * data into the cache. It also allows the user to interrupt and abort the
027     * current store operation by calling the method {@code abort}. If an {@code
028     * IOException} occurs while reading the response or writing data to the cache,
029     * the current cache store operation is abandoned.
030     * 
031     * @see ResponseCache
032     * @since Android 1.0
033     */
034    public abstract class CacheRequest {
035    
036        /**
037         * This implementation does nothing.
038         * 
039         * @since Android 1.0
040         */
041        public CacheRequest() {
042            super();
043        }
044    
045        /**
046         * Aborts the current cache operation. If an {@code IOException} occurs
047         * while reading the response or writing resource data to the cache, the
048         * current cache store operation is aborted.
049         * 
050         * @since Android 1.0
051         */
052        public abstract void abort();
053    
054        /**
055         * Returns an {@code OutputStream} which is used to write the response body.
056         * 
057         * @return an {@code OutputStream} which is used to write the response body.
058         * @throws IOException
059         *             if an I/O error is encountered during writing response body
060         *             operation.
061         * @since Android 1.0
062         */
063        public abstract OutputStream getBody() throws IOException;
064    }
