094        /**
095         * Constructs an IPv6 address according to the given {@code host}, {@code
096         * addr} and {@code scope_id}.
097         * 
098         * @param host
099         *            the host name associated with the address.
100         * @param addr
101         *            the network address.
102         * @param scope_id
103         *            the scope id for link- or site-local addresses.
104         * @return the Inet6Address instance representing the IP address.
105         * @throws UnknownHostException
106         *             if the address is null or has an invalid length.
107         * @since Android 1.0
108         */
109        public static Inet6Address getByAddress(String host, byte[] addr,
110                int scope_id) throws UnknownHostException {
111            if (null == addr || 16 != addr.length) {
112                // KA020=Illegal IPv6 address
113                throw new UnknownHostException(Msg.getString("KA020")); //$NON-NLS-1$
114            }
115            if (scope_id < 0) {
116                scope_id = 0;
117            }
118            return new Inet6Address(addr, host, scope_id);
119        }
120    
121        /**
122         * Gets an IPv6 address instance according to the given {@code host},
123         * {@code addr} and {@code nif}. {@code scope_id} is set according to the
124         * given {@code nif} and the {@code addr} type (for example site-local or
125         * link-local).
126         * 
127         * @param host
128         *            the hostname associated with the address.
129         * @param addr
130         *            the network address.
131         * @param nif
132         *            the network interface that this address is associated with.
133         * @return the Inet6Address instance representing the IP address.
134         * @throws UnknownHostException
135         *             if the address is {@code null} or has an invalid length or
136         *             the interface doesn't have a numeric scope id for the given
137         *             address type.
138         * @since Android 1.0
139         */
140        public static Inet6Address getByAddress(String host, byte[] addr,
141                NetworkInterface nif) throws UnknownHostException {
142    
143            Inet6Address address = Inet6Address.getByAddress(host, addr, 0);
144    
145            // if nif is null, nothing needs to be set.
146            if (null == nif) {
147                return address;
148            }
149    
150            // find the first address which matches the type addr,
151            // then set the scope_id, ifname and scopedIf.
152            Enumeration<InetAddress> addressList = nif.getInetAddresses();
153            while (addressList.hasMoreElements()) {
154                InetAddress ia = addressList.nextElement();
155                if (ia.getAddress().length == 16) {
156                    Inet6Address v6ia = (Inet6Address) ia;
157                    boolean isSameType = v6ia.compareLocalType(address);
158                    if (isSameType) {
159                        address.scope_id_set = true;
160                        address.scope_id = v6ia.scope_id;
161                        address.scope_ifname_set = true;
162                        address.ifname = nif.getName();
163                        address.scopedIf = nif;
164                        break;
165                    }
166                }
167            }
168            // if no address matches the type of addr, throws an
169            // UnknownHostException.
170            if (!address.scope_id_set) {
171                // KA021=Scope id is not found for the given address
172                throw new UnknownHostException(Msg.getString("KA021")); //$NON-NLS-1$
173            }
174            return address;
175        }
176    
177        /**
178         * Returns {@code true} if one of following cases applies:
179         * <p>
180         * <li>1. both addresses are site local</li>
181         * <li>2. both addresses are link local</li>
182         * <li>3. {@code ia} is neither site local nor link local</li>
183         * </p>
184         */
185        private boolean compareLocalType(Inet6Address ia) {
186            if (ia.isSiteLocalAddress() && isSiteLocalAddress()) {
187                return true;
188            }
189            if (ia.isLinkLocalAddress() && isLinkLocalAddress()) {
190                return true;
191            }
192            if (!ia.isSiteLocalAddress() && !ia.isLinkLocalAddress()) {
193                return true;
194            }
195            return false;
196        }

216        /**
217         * Returns whether this address is an IP multicast address or not. Valid
218         * IPv6 multicast addresses are binary prefixed with 11111111 or FF (hex).
219         * 
220         * @return {@code true} if this address is in the multicast group, {@code
221         *         false} otherwise.
222         * @since Android 1.0
223         */
224        @Override
225        public boolean isMulticastAddress() {
226            // Multicast addresses are prefixed with 11111111 (255)
227            return ipaddress[0] == -1;
228        }
229    
230        /**
231         * Returns whether this address is a unspecified wildcard address "::" or
232         * not.
233         * 
234         * @return {@code true} if this instance represents a wildcard address,
235         *         {@code false} otherwise.
236         * @since Android 1.0
237         */
238        @Override
239        public boolean isAnyLocalAddress() {
240            for (int i = 0; i < ipaddress.length; i++) {
241                if (ipaddress[i] != 0) {
242                    return false;
243                }
244            }
245            return true;
246        }
247    
248        /**
249         * Returns whether this address is the loopback address or not. The only
250         * valid IPv6 loopback address is "::1".
251         * 
252         * @return {@code true} if this instance represents the loopback address,
253         *         {@code false} otherwise.
254         * @since Android 1.0
255         */
256        @Override
257        public boolean isLoopbackAddress() {
258    
259            // The last word must be 1
260            if (ipaddress[15] != 1) {
261                return false;
262            }
263    
264            // All other words must be 0
265            for (int i = 0; i < 15; i++) {
266                if (ipaddress[i] != 0) {
267                    return false;
268                }
269            }
270    
271            return true;
272        }
273    
274        /**
275         * Returns whether this address is a link-local address or not. A valid IPv6
276         * link-local address is prefixed with 1111111010.
277         * 
278         * @return {@code true} if this instance represents a link-local address,
279         *         {@code false} otherwise.
280         * @since Android 1.0
281         */
282        @Override
283        public boolean isLinkLocalAddress() {
284    
285            // the first 10 bits need to be 1111111010 (1018)
286            return (ipaddress[0] == -2) && ((ipaddress[1] & 255) >>> 6) == 2;
287        }
288    
289        /**
290         * Returns whether this address is a site-local address or not. A valid IPv6
291         * site-local address is prefixed with 1111111011.
292         * 
293         * @return {@code true} if this instance represents a site-local address,
294         *         {@code false} otherwise.
295         * @since Android 1.0
296         */
297        @Override
298        public boolean isSiteLocalAddress() {
299    
300            // the first 10 bits need to be 1111111011 (1019)
301            return (ipaddress[0] == -2) && ((ipaddress[1] & 255) >>> 6) == 3;
302        }
303    
304        /**
305         * Returns whether this address is a global multicast address or not. A
306         * valid IPv6 global multicast address is 11111111xxxx1110 or FF0E hex.
307         * 
308         * @return {@code true} if this instance represents a global multicast
309         *         address, {@code false} otherwise.
310         * @since Android 1.0
311         */
312        @Override
313        public boolean isMCGlobal() {
314            // the first byte should be 0xFF and the lower 4 bits
315            // of the second byte should be 0xE
316            return (ipaddress[0] == -1) && (ipaddress[1] & 15) == 14;
317        }
318    
319        /**
320         * Returns whether this address is a node-local multicast address or not. A
321         * valid IPv6 node-local multicast address is prefixed with
322         * 11111111xxxx0001.
323         * 
324         * @return {@code true} if this instance represents a node-local multicast
325         *         address, {@code false} otherwise.
326         * @since Android 1.0
327         */
328        @Override
329        public boolean isMCNodeLocal() {
330            // the first byte should be 0xFF and the lower 4 bits

333        }

424         * Gets the hashcode of the represented IP address.
425         * 
426         * @return the appropriate hashcode value.

451        /**
452         * Returns whether this address is IPv4 compatible or not. An IPv4
453         * compatible address is prefixed with 96 bits of 0's. The last 32-bits are
454         * varied corresponding with the 32-bit IPv4 address space.
455         * 
456         * @return {@code true} if this instance represents an IPv4 compatible
457         *         address, {@code false} otherwise.
458         * @since Android 1.0
459         */
460        public boolean isIPv4CompatibleAddress() {
461            for (int i = 0; i < 12; i++) {
462                if (ipaddress[i] != 0) {
463                    return false;
464                }
465            }
466            return true;
467        }

469        private static final ObjectStreamField[] serialPersistentFields = {
470                new ObjectStreamField("ipaddress", new byte[0].getClass()), //$NON-NLS-1$
471                new ObjectStreamField("scope_id", Integer.TYPE), //$NON-NLS-1$
472                new ObjectStreamField("scope_id_set", Boolean.TYPE), //$NON-NLS-1$
473                new ObjectStreamField("scope_ifname_set", Boolean.TYPE), //$NON-NLS-1$
474                new ObjectStreamField("ifname", String.class), }; //$NON-NLS-1$
475    
476        private void writeObject(ObjectOutputStream stream) throws IOException {
477            ObjectOutputStream.PutField fields = stream.putFields();
478            if (ipaddress == null) {
479                fields.put("ipaddress", null); //$NON-NLS-1$
480            } else {
481                fields.put("ipaddress", ipaddress); //$NON-NLS-1$
482            }
483    
484            fields.put("scope_id", scope_id); //$NON-NLS-1$
485            fields.put("scope_id_set", scope_id_set); //$NON-NLS-1$
486            fields.put("scope_ifname_set", scope_ifname_set); //$NON-NLS-1$
487            fields.put("ifname", ifname); //$NON-NLS-1$
488            stream.writeFields();
489        }
490    
491        private void readObject(ObjectInputStream stream) throws IOException,
492                ClassNotFoundException {
493            ObjectInputStream.GetField fields = stream.readFields();
494            ipaddress = (byte[]) fields.get("ipaddress", null); //$NON-NLS-1$
495            scope_id = fields.get("scope_id", 0); //$NON-NLS-1$
496            scope_id_set = fields.get("scope_id_set", false); //$NON-NLS-1$
497            ifname = (String) fields.get("ifname", null); //$NON-NLS-1$
498            scope_ifname_set = fields.get("scope_ifname_set", false); //$NON-NLS-1$
499            if (scope_ifname_set && null != ifname) {
500                scopedIf = NetworkInterface.getByName(ifname);
501            }
502        }

504        /**
505         * Returns a string containing a concise, human-readable description of this
506         * IP address.
507         * 
508         * @return the description, as host/address.
509         * @since Android 1.0
510         */
511        @Override
512        public String toString() {
513            if (ifname != null) {
514                return super.toString() + "%" + ifname; //$NON-NLS-1$
515            }
516            if (scope_id != 0) {
517                return super.toString() + "%" + scope_id; //$NON-NLS-1$
518            }
519            return super.toString();
520        }
