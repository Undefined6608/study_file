018    package java.net;
019    
020    import java.io.IOException;
021    import java.security.cert.Certificate;
022    import java.util.jar.Attributes;
023    import java.util.jar.JarEntry;
024    import java.util.jar.JarFile;
025    import java.util.jar.Manifest;
026    
027    /**
028     * This class establishes a connection to a {@code jar:} URL using the {@code
029     * JAR} protocol. A {@code JarURLConnection} instance can refer to either a JAR
030     * archive file or to an entry of such a file. {@code jar:} URLs are specified
031     * as follows: <i>jar:{archive-url}!/{entry}</i> where "!/" is called a
032     * separator. This separator is important to determine if an archive or an entry
033     * of an archive is referred.
034     * <p>
035     * Examples:
036     * <li>Archive: {@code jar:http://www.example.com/applets/archive.jar!/}</li>
037     * <li>File Entry: {@code
038     * jar:http://www.example.com/applets/archive.jar!/test.class}</li>
039     * <li>Directory Entry: {@code
040     * jar:http://www.example.com/applets/archive.jar!/applets/}</li>
041     * 
042     * @since Android 1.0
043     */
044    public abstract class JarURLConnection extends URLConnection {
045    
046        /**
047         * The location part of the represented URL.
048         * 
049         * @since Android 1.0
050         */
051        protected URLConnection jarFileURLConnection;
052    
053        private String entryName;
054    
055        private URL fileURL;
056    
057        // the file component of the URL
058        private String file;
059    
060        /**
061         * Constructs an instance of {@code JarURLConnection} that refers to the
062         * specified URL.
063         * 
064         * @param url
065         *            the URL that contains the location to connect to.
066         * @throws MalformedURLException
067         *             if an invalid URL has been entered.
068         * @since Android 1.0
069         */
070        protected JarURLConnection(URL url) throws MalformedURLException {
071            super(url);
072            file = url.getFile();
073            int sepIdx;
074            if ((sepIdx = file.indexOf("!/")) < 0) { //$NON-NLS-1$
075                throw new MalformedURLException();
076            }
077            if (file.length() == sepIdx + 2) {
078                return;
079            }
080            entryName = file.substring(sepIdx + 2, file.length());
081            if (null != url.getRef()) {
082                entryName += "#" + url.getRef(); //$NON-NLS-1$
083            }
084        }
085    
086        /**
087         * Returns all attributes of the {@code JarEntry} referenced by this {@code
088         * JarURLConnection}.
089         * 
090         * @return the attributes of the referenced {@code JarEntry}.
091         * @exception IOException
092         *                if an I/O exception occurs while retrieving the
093         *                JAR-entries.
094         * @since Android 1.0
095         */
096        public Attributes getAttributes() throws java.io.IOException {
097            JarEntry jEntry = getJarEntry();
098            return (jEntry == null) ? null : jEntry.getAttributes();
099        }
100    
101        /**
102         * Returns all certificates of the {@code JarEntry} referenced by this
103         * {@code JarURLConnection} instance. This method will return {@code null}
104         * until the {@code InputStream} has been completely verified.
105         * 
106         * @return the certificates of the {@code JarEntry} as an array.
107         * @exception IOException
108         *                if there is an I/O exception occurs while getting the
109         *                {@code JarEntry}.
110         * @since Android 1.0
111         */
112        public Certificate[] getCertificates() throws java.io.IOException {
113            JarEntry jEntry = getJarEntry();
114            if (jEntry == null) {
115                return null;
116            }
117    
118            return jEntry.getCertificates();
119        }
120    
121        /**
122         * Gets the name of the entry referenced by this {@code JarURLConnection}.
123         * The return value will be {@code null} if this instance refers to a JAR
124         * file rather than an JAR file entry.
125         * 
126         * @return the {@code JarEntry} name this instance refers to.
127         * @since Android 1.0
128         */
129        public String getEntryName() {
130            return entryName;
131        }
132    
133        /**
134         * Gets the {@code JarEntry} object of the entry referenced by this {@code
135         * JarURLConnection}.
136         * 
137         * @return the referenced {@code JarEntry} object or {@code null} if no
138         *         entry name is specified.
139         * @throws IOException
140         *             if an error occurs while getting the file or file-entry.
141         * @since Android 1.0
142         */
143        public JarEntry getJarEntry() throws IOException {
144            if (!connected) {
145                connect();
146            }
147            if (entryName == null) {
148                return null;
149            }
150            // The entry must exist since the connect succeeded
151            return getJarFile().getJarEntry(entryName);
152        }
153    
154        /**
155         * Gets the manifest file associated with this JAR-URL.
156         * 
157         * @return the manifest of the referenced JAR-file.
158         * @throws IOException
159         *             if an error occurs while getting the manifest file.
160         * @since Android 1.0
161         */
162        public Manifest getManifest() throws java.io.IOException {
163            return getJarFile().getManifest();
164        }
165    
166        /**
167         * Gets the {@code JarFile} object referenced by this {@code
168         * JarURLConnection}.
169         * 
170         * @return the referenced JarFile object.
171         * @exception IOException
172         *                if an I/O exception occurs while retrieving the JAR-file.
173         * @since Android 1.0
174         */
175        public abstract JarFile getJarFile() throws java.io.IOException;
176    
177        /**
178         * Gets the URL to the JAR-file referenced by this {@code JarURLConnection}.
179         * 
180         * @return the URL to the JAR-file or {@code null} if there was an error
181         *         retrieving the URL.
182         * @since Android 1.0
183         */
184        public URL getJarFileURL() {
185            if (fileURL != null) {
186                return fileURL;
187            }
188            try {
189                return fileURL = new URL(url.getFile().substring(0,
190                        url.getFile().indexOf("!/"))); //$NON-NLS-1$
191            } catch (MalformedURLException e) {
192                return null;
193            }
194        }
195    
196        /**
197         * Gets all attributes of the manifest file referenced by this {@code
198         * JarURLConnection}. If this instance refers to a JAR-file rather than a
199         * JAR-file entry, {@code null} will be returned.
200         * 
201         * @return the attributes of the manifest file or {@code null}.
202         * @exception IOException
203         *                if an I/O exception occurs while retrieving the {@code
204         *                JarFile}.
205         * @since Android 1.0
206         */
207        public Attributes getMainAttributes() throws java.io.IOException {
208            Manifest m = getJarFile().getManifest();
209            return (m == null) ? null : m.getMainAttributes();
210        }
211    }
