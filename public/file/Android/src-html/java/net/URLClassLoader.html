102            /**
103             * Overrides the {@code loadClass()} of {@code ClassLoader}. It calls
104             * the security manager's {@code checkPackageAccess()} before
105             * attempting to load the class.
106             * 
107             * @return the Class object.
108             * @param className
109             *            String the name of the class to search for.
110             * @param resolveClass
111             *            boolean indicates if class should be resolved after
112             *            loading.
113             * @throws ClassNotFoundException
114             *             If the class could not be found.
115             */
116            @Override
117            protected synchronized Class<?> loadClass(String className,
118                    boolean resolveClass) throws ClassNotFoundException {
119                SecurityManager sm = System.getSecurityManager();
120                if (sm != null && !checkingPackageAccess) {
121                    int index = className.lastIndexOf('.');
122                    if (index > 0) { // skip if class is from a default package
123                        try {
124                            checkingPackageAccess = true;
125                            sm.checkPackageAccess(className.substring(0, index));
126                        } finally {
127                            checkingPackageAccess = false;
128                        }
129                    }
130                }
131                return super.loadClass(className, resolveClass);
132            }
133        }

175         * Adds the specified URL to the search list.
176         * 
177         * @param url
178         *            the URL which is to add.
179         * @since Android 1.0
180         */
181        protected void addURL(URL url) {
182            try {
183                URL search = createSearchURL(url);
184                urls = addURL(urls, search);
185                orgUrls = addURL(orgUrls, url);
186                synchronized (extensions) {
187                    extensions.put(search, null);
188                }
189            } catch (MalformedURLException e) {
190            }

193        /**
194         * Returns an array with the given URL added to the given array.
195         * 
196         * @param urlArray
197         *            {@code java.net.URL[]} the source array
198         * @param url
199         *            {@code java.net.URL} the URL to be added
200         * @return java.net.URL[] an array made of the given array and the new URL
201         */
202        URL[] addURL(URL[] urlArray, URL url) {
203            URL[] newPath = new URL[urlArray.length + 1];
204            System.arraycopy(urlArray, 0, newPath, 0, urlArray.length);
205            newPath[urlArray.length] = url;
206            Hashtable<String, URL[]>[] newIndexes = newHashtableArray(indexes.length + 1);
207            System.arraycopy(indexes, 0, newIndexes, 0, indexes.length);
208            indexes = newIndexes;
209            return newPath;
210        }

212        /**
213         * Returns all known URLs which point to the specified resource.
214         * 
215         * @param name
216         *            the name of the requested resource.
217         * @return the enumeration of URLs which point to the specified resource.
218         * @throws IOException
219         *             if an I/O error occurs while attempting to connect.
220         * @since Android 1.0
221         */
222        @Override
223        public Enumeration<URL> findResources(final String name) throws IOException {
224            if (name == null) {
225                return null;
226            }
227            Vector<URL> result = AccessController.doPrivileged(
228                    new PrivilegedAction<Vector<URL>>() {
229                        public Vector<URL> run() {
230                            return findResources(urls, name, new Vector<URL>());
231                        }
232                    }, currentContext);
233            SecurityManager sm;
234            int length = result.size();
235            if (length > 0 && (sm = System.getSecurityManager()) != null) {
236                Vector<URL> reduced = new Vector<URL>(length);
237                for (int i = 0; i < length; i++) {
238                    URL url = result.elementAt(i);
239                    try {
240                        sm.checkPermission(url.openConnection().getPermission());
241                        reduced.addElement(url);
242                    } catch (IOException e) {
243                    } catch (SecurityException e) {
244                    }
245                }
246                result = reduced;
247            }
248            return result.elements();
249        }

251        /**
252         * Returns a Vector of URLs among the given ones that contain the specified
253         * resource.
254         * 
255         * @return Vector the enumeration of URLs that contain the specified
256         *         resource.
257         * @param searchURLs
258         *            {@code java.net.URL[]} the array to be searched.
259         * @param name
260         *            {@code java.lang.String} the name of the requested resource.
261         */
262        Vector<URL> findResources(URL[] searchURLs, String name, Vector<URL> result) {
263            boolean findInExtensions = searchURLs == urls;
264            for (int i = 0; i < searchURLs.length; i++) {
265                if (!invalidUrls.contains(searchURLs[i])) {
266                    URL[] search = new URL[] { searchURLs[i] };
267                    URL res = findResourceImpl(search, name);
268                    if (!invalidUrls.contains(search[0])) {
269                        if (res != null && !result.contains(res)) {
270                            result.addElement(res);
271                        }
272                        if (findInExtensions) {
273                            findInExtensions(explore(searchURLs[i], i), name, i,
274                                    result, false);
275                        }
276                    }
277                }
278            }
279            return result;
280        }

282        /**
283         * Returns an {@code Object[]} containing a class, a URL, and a vector of
284         * URLs, two of which are {@code null}, according to the caller, which is
285         * identified by the {@code int} type.
286         * 
287         * @return Object[] a 3-element array : {Class, URL, Vector}. The non-null
288         *         element contains the resource(s) found, which are searched in in
289         *         {@code indexes[i]}.
290         * @param i
291         *            the index of 'indexes' array to use.
292         * @param name
293         *            the resource to look for : either a resource or a class.
294         * @param resources
295         *            {@code boolean} indicates that a vector of URL should be
296         *            returned as the non {@code null} element in {@code Object[]}.
297         * @param url
298         *            if {@code true} a URL should be returned as the non-null
299         *            element, if {@code false} a class should be returned.
300         */
301        Object findInIndex(int i, String name, Vector<URL> resources, boolean url) {
302            Hashtable<String, URL[]> index = indexes[i];
303            if (index != null) {
304                int pos = name.lastIndexOf("/"); //$NON-NLS-1$
305                // only keep the directory part of the resource
306                // as index.list only keeps track of directories and root files
307                String indexedName = (pos > 0) ? name.substring(0, pos) : name;
308                URL[] jarURLs;
309                if (resources != null) {
310                    jarURLs = index.get(indexedName);
311                    if (jarURLs != null) {
312                        findResources(jarURLs, name, resources);
313                    }
314                } else if (url) {
315                    jarURLs = index.get(indexedName);
316                    if (jarURLs != null) {
317                        return findResourceImpl(jarURLs, name);
318                    }
319                } else {
320                    String clsName = name;
321                    String partialName = clsName.replace('.', '/');
322                    int position;
323                    if ((position = partialName.lastIndexOf('/')) != -1) {
324                        String packageName = partialName.substring(0, position);
325                        jarURLs = index.get(packageName);
326                    } else {
327                        String className = partialName.substring(0, partialName
328                                .length())
329                                + ".class"; //$NON-NLS-1$
330                        jarURLs = index.get(className);
331                    }
332                    if (jarURLs != null) {
333                        Class<?> c = findClassImpl(jarURLs, clsName);
334                        // InvalidJarException is thrown when a mapping for a class
335                        // is not valid, i.e. we can't find the class by following
336                        // the mapping.
337                        if (c == null) {
338                            throw new InvalidJarIndexException();
339                        }
340                        return c;
341                    }
342                }
343            }
344            return null;
345        }

347        /**
348         * Returns an {@code Object[]} containing a Class, a URL, and a Vector of
349         * URLs, two of which are {@code null}, according to the caller, which is
350         * identified by the {@code int} type.
351         * 
352         * @return Object[] a 3-element array : {Class, URL, Vector}. The non-null
353         *         element contains the resource(s) found, which are searched in
354         *         newExtensions.
355         * @param newExtensions
356         *            URL[] the URLs to look in for.
357         * @param name
358         *            the resource to look for : either a resource or a class.
359         * @param i
360         *            the index of 'indexes' array to use.
361         * @param resources
362         *            indicates that a Vector of URL should be returned as the
363         *            non-null element in {@code Object[]}.
364         * @param url
365         *            if {@code true} a URL should be returned as the non-null
366         *            element, if {@code false} a class should be returned.
367         */
368        Object findInExtensions(URL[] newExtensions, String name, int i,
369                Vector<URL> resources, boolean url) {
370            if (newExtensions != null) {
371                for (int k = 0; k < newExtensions.length; k++) {
372                    if (newExtensions[k] != null) {
373                        URL[] search = new URL[] { newExtensions[k] };
374                        if (resources != null) {
375                            URL res = findResourceImpl(search, name);
376                            if (!invalidUrls.contains(search[0])) { // the URL does
377                                // not exist
378                                if (res != null && !resources.contains(res)) {
379                                    resources.addElement(res);
380                                }
381                                findInExtensions(explore(newExtensions[k], i),
382                                        name, i, resources, url);
383                            }
384                        } else {
385                            Object result;
386                            if (url) {
387                                result = findResourceImpl(search, name);
388                            } else {
389                                result = findClassImpl(search, name);
390                            }
391                            if (result != null) {
392                                return result;
393                            }
394                            if (!invalidUrls.contains(search[0])) { // the URL
395                                // exists
396                                result = findInExtensions(explore(newExtensions[k],
397                                        i), name, i, null, url);
398                                if (result != null) {
399                                    return result;
400                                }
401                            }
402                        }
403                    }
404                }
405            } else {
406                try {
407                    return findInIndex(i, name, resources, url);
408                } catch (InvalidJarIndexException ex) {
409                    // Ignore misleading/wrong jar index
410                    return null;
411                }
412            }
413            return null;
414        }

416        /**
417         * Converts an input stream into a byte array.
418         * 
419         * @return byte[] the byte array
420         * @param is
421         *            the input stream
422         */
423        private static byte[] getBytes(InputStream is, boolean readAvailable)
424                throws IOException {
425            if (readAvailable) {
426                byte[] buf = new byte[is.available()];
427                is.read(buf, 0, buf.length);
428                is.close();
429                return buf;
430            }
431            byte[] buf = new byte[4096];
432            int size = is.available();
433            if (size < 1024) {
434                size = 1024;
435            }
436            ByteArrayOutputStream bos = new ByteArrayOutputStream(size);
437            int count;
438            while ((count = is.read(buf)) > 0) {
439                bos.write(buf, 0, count);
440            }
441            return bos.toByteArray();
442        }

444        /**
445         * Gets all permissions for the specified {@code codesource}. First, this
446         * method retrieves the permissions from the system policy. If the protocol
447         * is "file:/" then a new permission, {@code FilePermission}, granting the
448         * read permission to the file is added to the permission collection.
449         * Otherwise, connecting to and accepting connections from the URL is
450         * granted.
451         * 
452         * @param codesource
453         *            the code source object whose permissions have to be known.
454         * @return the list of permissions according to the code source object.
455         * @since Android 1.0
456         */
457        @Override
458        protected PermissionCollection getPermissions(final CodeSource codesource) {
459            PermissionCollection pc = super.getPermissions(codesource);
460            URL u = codesource.getLocation();
461            if (u.getProtocol().equals("jar")) { //$NON-NLS-1$
462                try {
463                    // Create a URL for the resource the jar refers to
464                    u = ((JarURLConnection) u.openConnection()).getJarFileURL();
465                } catch (IOException e) {
466                    // This should never occur. If it does continue using the jar
467                    // URL
468                }
469            }
470            if (u.getProtocol().equals("file")) { //$NON-NLS-1$
471                String path = u.getFile();
472                String host = u.getHost();
473                if (host != null && host.length() > 0) {
474                    path = "//" + host + path; //$NON-NLS-1$
475                }
476    
477                if (File.separatorChar != '/') {
478                    path = path.replace('/', File.separatorChar);
479                }
480                if (isDirectory(u)) {
481                    pc.add(new FilePermission(path + "-", "read")); //$NON-NLS-1$ //$NON-NLS-2$
482                } else {
483                    pc.add(new FilePermission(path, "read")); //$NON-NLS-1$
484                }
485            } else {
486                String host = u.getHost();
487                if (host.length() == 0) {
488                    host = "localhost"; //$NON-NLS-1$
489                }
490                pc.add(new SocketPermission(host, "connect, accept")); //$NON-NLS-1$
491            }
492            return pc;
493        }

562        /**
563         * Constructs a new {@code URLClassLoader} instance. The newly created
564         * instance will have the specified {@code ClassLoader} as its parent and
565         * use the specified factory to create stream handlers. URLs that end with
566         * "/" are assumed to be directories, otherwise they are assumed to be JAR
567         * files.
568         * 
569         * @param searchUrls
570         *            the list of URLs where a specific class or file could be
571         *            found.
572         * @param parent
573         *            the {@code ClassLoader} to assign as this loader's parent.
574         * @param factory
575         *            the factory that will be used to create protocol-specific
576         *            stream handlers.
577         * @throws SecurityException
578         *             if a security manager exists and its {@code
579         *             checkCreateClassLoader()} method doesn't allow creation of
580         *             new {@code ClassLoader}s.
581         * @since Android 1.0
582         */
583        public URLClassLoader(URL[] searchUrls, ClassLoader parent,
584                URLStreamHandlerFactory factory) {
585            super(parent);
586            // Required for pre-v1.2 security managers to work
587            SecurityManager security = System.getSecurityManager();
588            if (security != null) {
589                security.checkCreateClassLoader();
590            }
591            this.factory = factory;
592            // capture the context of the thread that creates this URLClassLoader
593            currentContext = AccessController.getContext();
594            int nbUrls = searchUrls.length;
595            urls = new URL[nbUrls];
596            orgUrls = new URL[nbUrls];
597            // Search each jar for CLASS-PATH attribute in manifest
598            extensions = new HashMap<URL, URL[]>(nbUrls * 2);
599            for (int i = 0; i < nbUrls; i++) {
600                try {
601                    urls[i] = createSearchURL(searchUrls[i]);
602                    extensions.put(urls[i], null);
603                } catch (MalformedURLException e) {
604                }
605                orgUrls[i] = searchUrls[i];
606            }
607            // Search each jar for META-INF/INDEX.LIST
608            indexes = newHashtableArray(nbUrls);
609        }

613         * class could be found, a class object representing the loaded class will
614         * be returned.
615         * 
616         * @return the class that has been loaded.
617         * @param clsName
618         *            the name of the class which has to be found.
619         * @throws ClassNotFoundException
620         *             if the specified class cannot be loaded.
621         * @since Android 1.0
622         */
623        @Override
624        protected Class<?> findClass(final String clsName)
625                throws ClassNotFoundException {
626            Class<?> cls = AccessController.doPrivileged(
627                    new PrivilegedAction<Class<?>>() {
628                        public Class<?> run() {
629                            return findClassImpl(urls, clsName);
630                        }
631                    }, currentContext);
632            if (cls != null) {
633                return cls;
634            }
635            throw new ClassNotFoundException(clsName);
636        }

638        /**
639         * Returns an URL that will be checked if it contains the class or resource.
640         * If the file component of the URL is not a directory, a Jar URL will be
641         * created.
642         * 
643         * @return java.net.URL a test URL
644         */
645        private URL createSearchURL(URL url) throws MalformedURLException {
646            if (url == null) {
647                return url;
648            }
649    
650            String protocol = url.getProtocol();
651    
652            if (isDirectory(url) || protocol.equals("jar")) { //$NON-NLS-1$
653                return url;
654            }
655            if (factory == null) {
656                return new URL("jar", "", //$NON-NLS-1$ //$NON-NLS-2$
657                        -1, url.toString() + "!/"); //$NON-NLS-1$
658            }
659            return new URL("jar", "", //$NON-NLS-1$ //$NON-NLS-2$
660                    -1, url.toString() + "!/", //$NON-NLS-1$
661                    factory.createURLStreamHandler(protocol));
662        }

664        /**
665         * Returns an URL referencing the specified resource or {@code null} if the
666         * resource could not be found.
667         * 
668         * @param name
669         *            the name of the requested resource.
670         * @return the URL which points to the given resource.
671         * @since Android 1.0
672         */
673        @Override
674        public URL findResource(final String name) {
675            if (name == null) {
676                return null;
677            }
678            URL result = AccessController.doPrivileged(new PrivilegedAction<URL>() {
679                public URL run() {
680                    return findResourceImpl(urls, name);
681                }
682            }, currentContext);
683            SecurityManager sm;
684            if (result != null && (sm = System.getSecurityManager()) != null) {
685                try {
686                    sm.checkPermission(result.openConnection().getPermission());
687                } catch (IOException e) {
688                    return null;

987        /**
988         * @param in
989         *            InputStream the stream to read lines from
990         * @return List a list of String lines
991         */
992        private List<String> readLines(InputStream in) throws IOException {
993            byte[] buff = new byte[144];
994            List<String> lines = new ArrayList<String>();
995            int pos = 0;
996            int next;
997            while ((next = in.read()) != -1) {
998                if (next == '\n') {
999                    lines.add(new String(buff, 0, pos, "UTF8")); //$NON-NLS-1$
1000                    pos = 0;
1001                    continue;
1002                }
1003                if (next == '\r') {
1004                    lines.add(new String(buff, 0, pos, "UTF8")); //$NON-NLS-1$
1005                    pos = 0;
1006                    if ((next = in.read()) == '\n') {
1007                        continue;
1008                    }
1009                }
1010                if (pos == buff.length) {
1011                    byte[] newBuf = new byte[buff.length * 2];
1012                    System.arraycopy(buff, 0, newBuf, 0, buff.length);
1013                    buff = newBuf;
1014                }
1015                buff[pos++] = (byte) next;
1016            }
1017            if (pos > 0) {
1018                lines.add(new String(buff, 0, pos, "UTF8")); //$NON-NLS-1$
1019            }
1020            return lines;
1021        }

1259                internal = extensions.get(url);
1260            }
1261            if (internal != null) {
1262                return internal;
1263            }
1264            if (indexes[indexNumber] != null) {
1265                return null;
1266            }

1269                return null;
1270            }
1271    
1272            JarFile jf = resCache.get(url);
1273            // Add mappings from INDEX.LIST
1274            ZipEntry ze = jf.getEntry("META-INF/INDEX.LIST"); //$NON-NLS-1$
1275            if (ze != null) {
1276                if (url.equals(urls[indexNumber])) {
1277                    try {
1278                        Hashtable<String, URL[]> index = new Hashtable<String, URL[]>(
1279                                15);
1280                        InputStream indexIS = jf.getInputStream(ze);
1281                        List<String> lines = readLines(indexIS);
1282                        indexIS.close();
1283                        ListIterator<String> iterator = lines.listIterator();
1284                        // Ignore the 2 first lines (index version)
1285                        iterator.next();
1286                        iterator.next();
1287                        // Add mappings from resource to jar file
1288                        URL fileURL = ((JarURLConnection) url.openConnection())
1289                                .getJarFileURL();
1290                        String file = fileURL.getFile();
1291                        String parentFile = new File(file).getParent();
1292                        parentFile = parentFile.replace(File.separatorChar, '/');
1293                        if (parentFile.charAt(0) != '/') {
1294                            parentFile = "/" + parentFile; //$NON-NLS-1$
1295                        }
1296                        URL parentURL = new URL(fileURL.getProtocol(), fileURL
1297                                .getHost(), fileURL.getPort(), parentFile);
1298                        while (iterator.hasNext()) {
1299                            URL jar = new URL("jar:" //$NON-NLS-1$
1300                                    + parentURL.toExternalForm() + "/" //$NON-NLS-1$
1301                                    + iterator.next() + "!/"); //$NON-NLS-1$
1302                            String resource = null;
1303                            while (iterator.hasNext()
1304                                    && !(resource = iterator.next()).equals("")) { //$NON-NLS-1$
1305                                if (index.containsKey(resource)) {
1306                                    URL[] jars = index.get(resource);
1307                                    URL[] newJars = new URL[jars.length + 1];
1308                                    System.arraycopy(jars, 0, newJars, 0,
1309                                            jars.length);
1310                                    newJars[jars.length] = jar;
1311                                    index.put(resource, newJars);
1312                                } else {
1313                                    URL[] jars = { jar };
