033    /**
034     * Regulates the access to network operations available through sockets through
035     * permissions. A permission consists of a target (a host), and an associated
036     * action list. The target should identify the host by either indicating the
037     * (possibly wildcarded (eg. {@code .company.com})) DNS style name of the host
038     * or its IP address in standard {@code nn.nn.nn.nn} ("dot") notation. The
039     * action list can be made up of one or more of the following actions separated
040     * by a comma:
041     * <dl>
042     * <dt>connect</dt>
043     * <dd>requests permission to connect to the host</dd>
044     * <dt>listen</dt>
045     * <dd>requests permission to listen for connections from the host</dd>
046     * <dt>accept</dt>
047     * <dd>requests permission to accept connections from the host</dd>
048     * <dt>resolve</dt>
049     * <dd>requests permission to resolve the hostname</dd>
050     * </dl>
051     * Note that {@code resolve} is implied when any (or none) of the others are
052     * present.
053     * <p>
054     * Access to a particular port can be requested by appending a colon and a
055     * single digit to the name (eg. {@code .company.com:7000}). A range of port
056     * numbers can also be specified, by appending a pattern of the form
057     * <i>LOW-HIGH</i> where <i>LOW</i> and <i>HIGH</i> are valid port numbers. If
058     * either <i>LOW</i> or <i>HIGH</i> is omitted it is equivalent to entering the
059     * lowest or highest possible value respectively. For example:
060     * 
061     * <pre>
062     * {@code SocketPermission(&quot;www.company.com:7000-&quot;, &quot;connect,accept&quot;)}
063     * </pre>
064     * 
065     * represents the permission to connect to and accept connections from {@code
066     * www.company.com} on ports in the range {@code 7000} to {@code 65535}.
067     * </p>
068     * 
069     * @since Android 1.0
070     */
071    public final class SocketPermission extends Permission implements Serializable {
072    
073        private static final long serialVersionUID = -7204263841984476862L;
074    
075        // Bit masks for each of the possible actions
076        static final int SP_CONNECT = 1;
077    
078        static final int SP_LISTEN = 2;
079    
080        static final int SP_ACCEPT = 4;
081    
082        static final int SP_RESOLVE = 8;
083    
084        // list of actions permitted for socket permission in order, indexed by mask
085        // value
086        @SuppressWarnings("nls")
087        private static final String[] actionNames = { "", "connect", "listen", "",
088                "accept", "", "", "", "resolve" };
089    
090        // If a wildcard is present store the information
091        private transient boolean isPartialWild;
092    
093        private transient boolean isWild;
094    
095        // The highest port number
096        private static final int HIGHEST_PORT = 65535;
097    
098        // The lowest port number
099        private static final int LOWEST_PORT = 0;
100    
101        transient String hostName; // Host name as returned by InetAddress
102    
103        transient String ipString; // IP address as returned by InetAddress
104    
105        transient boolean resolved; // IP address has been resolved
106    
107        // the port range;
108        transient int portMin = LOWEST_PORT;
109    
110        transient int portMax = HIGHEST_PORT;
111    
112        private String actions; // List of all actions allowed by this permission
113    
114        transient int actionsMask = SP_RESOLVE;

116        /**
117         * Constructs a new {@code SocketPermission} instance. The hostname can be a
118         * DNS name, an individual hostname, an IP address or the empty string which
119         * implies {@code localhost}. The port or port range is optional.
120         * <p>
121         * The action list is a comma-separated list which can consists of the
122         * possible operations {@code "connect"}, {@code "listen"}, {@code "accept"}
123         * , and {@code "resolve"}. They are case-insensitive and can be put
124         * together in any order. {@code "resolve"} is implied per default.
125         * </p>
126         * 
127         * @param host
128         *            the hostname this permission is valid for.
129         * @param action
130         *            the action string of this permission.
131         * @since Android 1.0
132         */
133        public SocketPermission(String host, String action) {
134            super(host.equals("") ? "localhost" : host); //$NON-NLS-1$ //$NON-NLS-2$
135            hostName = getHostString(host);
136            if (action == null) {
137                throw new NullPointerException();
138            }
139            if (action.equals("")) { //$NON-NLS-1$
140                throw new IllegalArgumentException();
141            }
142    
143            setActions(action);
144            actions = toCanonicalActionString(action);
145            // Use host since we are only checking for port presence
146            parsePort(host, hostName);
147        }
148    
149        /**
150         * Compares the argument {@code o} to this instance and returns {@code true}
151         * if they represent the same permission using a class specific comparison.
152         * 
153         * @param o
154         *            the object to compare with this {@code SocketPermission}
155         *            instance.
156         * @return {@code true} if they represent the same permission, {@code false}
157         *         otherwise.
158         * @see #hashCode
159         * @since Android 1.0
160         */
161        @Override
162        public boolean equals(Object o) {
163            if (this == o) {
164                return true;
165            }
166            if (o == null || this.getClass() != o.getClass()) {
167                return false;
168            }
169            SocketPermission sp = (SocketPermission) o;
170            if (!hostName.equals(sp.hostName)) {
171                if (getIPString() == null || !ipString.equals(sp.getIPString())) {
172                    return false;
173                }
174            }
175            if (this.actionsMask != SP_RESOLVE) {
176                if (this.portMin != sp.portMin) {
177                    return false;
178                }
179                if (this.portMax != sp.portMax) {
180                    return false;
181                }
182            }
183            return this.actionsMask == sp.actionsMask;
184        }
185    
186        /**
187         * Returns the hash value for this {@code SocketPermission} instance. Any
188         * two objects which returns {@code true} when passed to {@code equals()}
189         * must return the same value as a result of this method.
190         * 
191         * @return the hashcode value for this instance.
192         * @see #equals
193         * @since Android 1.0
194         */
195        @Override
196        public int hashCode() {
197            return hostName.hashCode() ^ actionsMask ^ portMin ^ portMax;
198        }

200        /**
201         * Gets a comma-separated list of all actions allowed by this permission. If
202         * more than one action is returned they follow this order: {@code connect},
203         * {@code listen}, {@code accept}, {@code resolve}.
204         * 
205         * @return the comma-separated action list.
206         * @since Android 1.0
207         */
208        @Override
209        public String getActions() {
210            return actions;
211        }

213        /**
214         * Stores the actions for this permission as a bit field.
215         * 
216         * @param actions
217         *            java.lang.String the action list
218         */
219        private void setActions(String actions) throws IllegalArgumentException {
220            if (actions.equals("")) { //$NON-NLS-1$
221                return;
222            }
223            boolean parsing = true;
224            String action;
225            StringBuffer sb = new StringBuffer();
226            int pos = 0, length = actions.length();
227            while (parsing) {
228                char c;
229                sb.setLength(0);
230                while (pos < length && (c = actions.charAt(pos++)) != ',') {
231                    sb.append(c);
232                }
233                if (pos == length) {
234                    parsing = false;
235                }
236                action = sb.toString().trim().toLowerCase();
237                if (action.equals(actionNames[SP_CONNECT])) {
238                    actionsMask |= SP_CONNECT;
239                } else if (action.equals(actionNames[SP_LISTEN])) {
240                    actionsMask |= SP_LISTEN;
241                } else if (action.equals(actionNames[SP_ACCEPT])) {
242                    actionsMask |= SP_ACCEPT;
243                } else if (action.equals(actionNames[SP_RESOLVE])) {
244                    // do nothing
245                } else {
246                    throw new IllegalArgumentException(Msg.getString("K0048", //$NON-NLS-1$
247                            action));
248                }
249            }
250        }
251    
252        /**
253         * Checks whether this {@code SocketPermission} instance allows all actions
254         * which are allowed by the given permission object {@code p}. All argument
255         * permission actions, hosts and ports must be implied by this permission
256         * instance in order to return {@code true}. This permission may imply
257         * additional actions not present in the argument permission.
258         * 
259         * @param p
260         *            the socket permission which has to be implied by this
261         *            instance.
262         * @return {@code true} if this permission instance implies all permissions
263         *         represented by {@code p}, {@code false} otherwise.
264         * @since Android 1.0
265         */
266        @Override
267        public boolean implies(Permission p) {
268            SocketPermission sp;
269            try {
270                sp = (SocketPermission) p;
271            } catch (ClassCastException e) {
272                return false;
273            }
274    
275            // tests if the action list of p is the subset of the one of the
276            // receiver
277            if (sp == null || (actionsMask & sp.actionsMask) != sp.actionsMask) {
278                return false;
279            }
280    
281            // only check the port range if the action string of the current object
282            // is not "resolve"
283            if (!p.getActions().equals("resolve")) { //$NON-NLS-1$
284                if ((sp.portMin < this.portMin) || (sp.portMax > this.portMax)) {
285                    return false;
286                }
287            }
288    
289            // Verify the host is valid
290            return checkHost(sp);
291        }

305        /**
306         * Parse the port, including the minPort, maxPort
307         * @param hostPort the host[:port] one
308         * @param host the host name we just get
309         * @throws IllegalArgumentException If the port is not a positive number or minPort
310         *                                  is not less than or equal maxPort
311         */
312        private void parsePort(String hostPort, String host) throws IllegalArgumentException {
313           String port = hostPort.substring(host.length());
314           String emptyString = ""; //$NON-NLS-1$
315    
316           if (emptyString.equals(port)) {
317               // Not specified
318               portMin = 80;
319               portMax = 80;
320               return;
321           }
322           
323           if (":*".equals(port)) {
324               // The port range should be 0-65535
325               portMin = 0;
326               portMax = 65535;
327               return;
328           }
329           
330           // Omit ':'
331           port = port.substring(1);
332           int negIdx = port.indexOf('-');
333           String strPortMin = emptyString;
334           String strPortMax = emptyString;
335           if (-1 == negIdx) {
336               // No neg mark, only one number
337               strPortMin = port;
338               strPortMax = port;
339           } else {
340               strPortMin = port.substring(0, negIdx);
341               strPortMax = port.substring(negIdx + 1);
342               if (emptyString.equals(strPortMin)) {
343                   strPortMin = "0";
344               }
345               if (emptyString.equals(strPortMax)) {
346                   strPortMax = "65535";
347               }
348           }
349           try {
350               portMin = Integer.valueOf(strPortMin).intValue();
351               portMax = Integer.valueOf(strPortMax).intValue();
352               
353               if (portMin > portMax) {
354                   throw new IllegalArgumentException(Msg.getString("K0049") + " " + port); //$NON-NLS-1$
355               }
356           } catch (NumberFormatException e) {
357               throw new IllegalArgumentException(Msg.getString("K004a") + " " + port); //$NON-NLS-1$
358           }
359        }

361        /**
362         * Creates a canonical action list.
363         * 
364         * @param action
365         *            java.lang.String
366         * 
367         * @return java.lang.String
368         */
369        private String toCanonicalActionString(String action) {
370            if (action == null || action.equals("") || actionsMask == SP_RESOLVE) { //$NON-NLS-1$
371                return actionNames[SP_RESOLVE]; // If none specified return the
372            }
373            // implied action resolve
374            StringBuffer sb = new StringBuffer();
375            if ((actionsMask & SP_CONNECT) == SP_CONNECT) {
376                sb.append(',');
377                sb.append(actionNames[SP_CONNECT]);
378            }
379            if ((actionsMask & SP_LISTEN) == SP_LISTEN) {
380                sb.append(',');
381                sb.append(actionNames[SP_LISTEN]);
382            }
383            if ((actionsMask & SP_ACCEPT) == SP_ACCEPT) {
384                sb.append(',');
385                sb.append(actionNames[SP_ACCEPT]);
386            }
387            sb.append(',');
388            sb.append(actionNames[SP_RESOLVE]);// Resolve is always implied
389            // Don't copy the first ','.
390            return actions = sb.substring(1, sb.length());
391        }

393        private String getIPString() {
394            if (!resolved) {
395                try {
396                    ipString = InetAddress.getHostNameInternal(hostName);
397                } catch (UnknownHostException e) {
398                    // ignore
399                }
400                resolved = true;
401            }
402            return ipString;
403        }

405        /**
406         * Get the host part from the host[:port] one.
407         * The host should be
408         *      host = (hostname | IPv4address | IPv6reference | IPv6 in full uncompressed form)
409         * The wildcard "*" may be included once in a DNS name host specification. If it is included, 
410         * it must be in the leftmost position
411         * 
412         * @param host
413         * @return
414         * @throws IllegalArgumentException   if the host is invalid.
415         */
416        private String getHostString(String host) throws IllegalArgumentException {
417            host = host.trim();
418            int idx = -1;
419            idx = host.indexOf(':');
420            isPartialWild = (host.length() > 0 && host.charAt(0) == '*');
421            if (isPartialWild) {
422                resolved = true;
423                isWild = (host.length() == 1);
424                if (isWild) {
425                    return host;
426                }
427                if (idx > -1) {
428                    host = host.substring(0, idx);
429                }
430                return host.toLowerCase();
431            }
432    
433            int lastIdx = host.lastIndexOf(':');
434            
435            if (idx == lastIdx) {
436                if (-1 != idx) {
437                    // only one colon, should be port
438                    host = host.substring(0, idx);
439                }
440                return host.toLowerCase();
441            }
442                // maybe ipv6
443            boolean isFirstBracket = (host.charAt(0) == '[');
444            if (!isFirstBracket) {
445                // No bracket, should be in full form
446                int colonNum = 0;
447                for (int i = 0; i < host.length(); ++i) {
448                    if (host.charAt(i) == ':') {
449                        colonNum++;
450                    }
451                }
452                // Get rid of the colon before port
453                if (8 == colonNum) {
454                    host = host.substring(0, lastIdx);
455                }
456                if (Inet6Util.isIP6AddressInFullForm(host)) {
457                    return host.toLowerCase();
458                }
459                throw new IllegalArgumentException(Msg.getString("K004a") + " "
460                        + host);
461            }
462            // forward bracket found
463            int bbracketIdx = host.indexOf(']');
464            if (-1 == bbracketIdx) {
465                // no back bracket found, wrong
466                throw new IllegalArgumentException(Msg.getString("K004a") + " "
467                        + host);
468            }
469            host = host.substring(0, bbracketIdx + 1);
470            if (Inet6Util.isValidIP6Address(host)) {
471                return host.toLowerCase();
472            }
473            throw new IllegalArgumentException(Msg.getString("K004a") + " " + host);
474        }

476        /**
477         * Determines whether or not this permission could refer to the same host as
478         * sp.
479         */
480        boolean checkHost(SocketPermission sp) {
481            if (isPartialWild) {
482                if (isWild) {
483                    return true; // Match on any host
484                }
485                int length = hostName.length() - 1;
486                return sp.hostName.regionMatches(sp.hostName.length() - length,
487                        hostName, 1, length);
488            }
489            // The ipString may not be the same, some hosts resolve to
490            // multiple ips
491            return (getIPString() != null && ipString.equals(sp.getIPString()))
492                    || hostName.equals(sp.hostName);
493        }

495        private void writeObject(ObjectOutputStream stream) throws IOException {
496            stream.defaultWriteObject();
497        }
498    
499        private void readObject(ObjectInputStream stream) throws IOException,
500                ClassNotFoundException {
501            stream.defaultReadObject();
502            // Initialize locals
503            isPartialWild = false;
504            isWild = false;
505            portMin = LOWEST_PORT;
506            portMax = HIGHEST_PORT;
507            actionsMask = SP_RESOLVE;
508            hostName = getHostString(getName());
509            parsePort(getName(), hostName);
510            setActions(actions);
511        }
