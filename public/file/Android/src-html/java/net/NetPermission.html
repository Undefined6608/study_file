018    package java.net;
019    
020    /**
021     * This class represents permissions to configure the access to network
022     * resources.
023     * <p>
024     * There are three valid target names:
025     * <dl>
026     * <dt>setDefaultAuthenticator</dt>
027     * <dd>Allows the default authenticator to be set.</dd>
028     * <dt>requestPasswordAuthentication</dt>
029     * <dd>Allows the default authenticator to be retrieved.</dd>
030     * <dt>specifyStreamHandler</dt>
031     * <dd>Allows a stream (protocol) handler to be set when constructing an URL
032     * object</dd>
033     * </dl>
034     * </p>
035     * 
036     * @see java.security.BasicPermission
037     * @see SecurityManager
038     * @since Android 1.0
039     */
040    public final class NetPermission extends java.security.BasicPermission {
041    
042        private static final long serialVersionUID = -8343910153355041693L;
043    
044        /**
045         * Creates an instance of this class with the given name.
046         * 
047         * @param name
048         *            the name of the new NetPermission instance.
049         * @since Android 1.0
050         */
051        public NetPermission(String name) {
052            super(name);
053        }
054    
055        /**
056         * Creates an instance of this class with the given name and an action list.
057         * The action list is ignored and should be {@code null}.
058         * 
059         * @param name
060         *            the name of the new {@code NetPermission} instance.
061         * @param actions
062         *            the ignored action string.
063         * @since Android 1.0
064         */
065        public NetPermission(String name, String actions) {
066            super(name, actions);
067        }
068    }
