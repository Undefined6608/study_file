016    package java.net;
017    
018    import java.security.Principal;
019    import java.security.cert.Certificate;
020    import java.util.List;
021    
022    import javax.net.ssl.SSLPeerUnverifiedException;
023    
024    /**
025     * A secure cache response represents data which is originally retrieved over a
026     * secure connection. Such a connection can be secured by using a cryptographic
027     * protocol like TLS or SSL.
028     * 
029     * @see ResponseCache
030     * @since Android 1.0
031     */
032    public abstract class SecureCacheResponse extends CacheResponse {
033    
034        /**
035         * Creates a new instance of this class.
036         * 
037         * @since Android 1.0
038         */
039        public SecureCacheResponse() {
040            super();
041        }
042    
043        /**
044         * Gets the cipher suite string on the connection which is originally used
045         * to retrieve the network resource.
046         * 
047         * @return the cipher suite string.
048         * @since Android 1.0
049         */
050        public abstract String getCipherSuite();
051    
052        /**
053         * Gets the local certificate chain. When the original connection retrieved
054         * the resource data, this certificate chain was sent to the server during
055         * handshaking process. This method only takes effect when certificate-based
056         * cipher suite is enabled.
057         * 
058         * @return the certificate chain that was sent to the server. If no
059         *         certificate chain was sent, the method returns {@code null}.
060         * @since Android 1.0
061         */
062        public abstract List<Certificate> getLocalCertificateChain();
063    
064        /**
065         * Gets the cached server's certificate chain. As part of defining the
066         * session, the certificate chain was established when the original
067         * connection retrieved network resource. This method can only be invoked
068         * when certificated-based cipher suite is enabled. Otherwise, it throws an
069         * {@code SSLPeerUnverifiedException}.
070         * 
071         * @return the server's certificate chain.
072         * @throws SSLPeerUnverifiedException
073         *             if the peer is unverified.
074         * @since Android 1.0
075         */
076        public abstract List<Certificate> getServerCertificateChain()
077                throws SSLPeerUnverifiedException;
078    
079        /**
080         * Gets the server's principle. When the original connection retrieved
081         * network resource, the principle was established when defining the
082         * session.
083         * 
084         * @return a principal object representing the server's principal.
085         * @throws SSLPeerUnverifiedException
086         *             if the peer is unverified.
087         * @since Android 1.0
088         */
089        public abstract Principal getPeerPrincipal()
090                throws SSLPeerUnverifiedException;
091    
092        /**
093         * Gets the local principle that the original connection sent to the server.
094         * When the original connection fetched the network resource, the principle
095         * was sent to the server during handshaking process.
096         * 
097         * @return the local principal object being sent to the server. Returns an
098         *         {@code X500Principal} object for X509-based cipher suites. If no
099         *         principal was sent, it returns {@code null}.
100         * @since Android 1.0
101         */
102        public abstract Principal getLocalPrincipal();
103    }
