017    package dalvik.system;
018    
019    import java.io.File;
020    import java.io.IOException;
021    import java.net.URL;
022    import java.util.zip.ZipFile;
023    import java.net.MalformedURLException;
024    
025    import dalvik.system.DexFile;
026    
027    /**
028     * Provides a simple {@link ClassLoader} implementation that operates on a
029     * list of jar/apk files with classes.dex entries.  The directory that
030     * holds the optimized form of the files is specified explicitly.  This
031     * can be used to execute code not installed as part of an application.
032     *
033     * The best place to put the optimized DEX files is in app-specific
034     * storage, so that removal of the app will automatically remove the
035     * optimized DEX files.  If other storage is used (e.g. /sdcard), the
036     * app may not have an opportunity to remove them.
037     */
038    public class DexClassLoader extends ClassLoader {
039    
040        private static final boolean VERBOSE_DEBUG = false;
041    
042        /* constructor args, held for init */
043        private final String mRawDexPath;
044        private final String mRawLibPath;
045        private final String mDexOutputPath;
046    
047        private boolean mInitialized;
048    
049        /*
050         * Parallel arrays for jar/apk files.
051         *
052         * (could stuff these into an object and have a single array;
053         * improves clarity but adds overhead)
054         */
055        private File[] mFiles;              // source file Files, for rsrc URLs
056        private ZipFile[] mZips;            // source zip files, with resources
057        private DexFile[] mDexs;            // opened, prepped DEX files
058    
059        /*
060         * Native library path.
061         */
062        private String[] mLibPaths;
063    
064    
065        /**
066         * Creates a {@code DexClassLoader} that finds interpreted and native
067         * code.  Interpreted classes are found in a set of DEX files contained
068         * in Jar or APK files.
069         *
070         * The path lists are separated using the character specified by
071         * the "path.separator" system property, which defaults to ":".
072         *
073         * @param dexPath
074         *  the list of jar/apk files containing classes and resources
075         * @param dexOutputDir
076         *  directory where optimized DEX files should be written
077         * @param libPath
078         *  the list of directories containing native libraries; may be null
079         * @param parent
080         *  the parent class loader
081         */
082        public DexClassLoader(String dexPath, String dexOutputDir, String libPath,
083            ClassLoader parent) {
084    
085            super(parent);
086    
087            if (dexPath == null || dexOutputDir == null)
088                throw new NullPointerException();
089    
090            mRawDexPath = dexPath;
091            mDexOutputPath = dexOutputDir;
092            mRawLibPath = libPath;
093        }
094    
095        /**
096         * Complete initialization on first use of the class loader.
097         */
098        private synchronized void ensureInit() {
099            if (mInitialized) {
100                return;
101            }
102    
103            String[] dexPathList;
104    
105            mInitialized = true;
106    
107            dexPathList = mRawDexPath.split(":");
108            int length = dexPathList.length;
109    
110            //System.out.println("DexClassLoader: " + dexPathList);
111            mFiles = new File[length];
112            mZips = new ZipFile[length];
113            mDexs = new DexFile[length];
114    
115            /* open all Zip and DEX files up front */
116            for (int i = 0; i < length; i++) {
117                System.out.println("My path is: " + dexPathList[i]);
118                File pathFile = new File(dexPathList[i]);
119                mFiles[i] = pathFile;
120    
121                if (pathFile.isFile()) {
122                    try {
123                        mZips[i] = new ZipFile(pathFile);
124                    } catch (IOException ioex) {
125                        // expecting IOException and ZipException
126                        System.out.println("Failed opening '" + pathFile
127                            + "': " + ioex);
128                        //ioex.printStackTrace();
129                    }
130    
131                    /* we need both DEX and Zip, because dex has no resources */
132                    try {
133                        String outputName =
134                            generateOutputName(dexPathList[i], mDexOutputPath);
135                        mDexs[i] = DexFile.loadDex(dexPathList[i], outputName, 0);
136                    } catch (IOException ioex) {
137                        // might be a resource-only zip
138                        System.out.println("Failed loadDex '" + pathFile
139                            + "': " + ioex);
140                    }
141                } else {
142                    if (VERBOSE_DEBUG)
143                        System.out.println("Not found: " + pathFile.getPath());
144                }
145            }
146    
147            /*
148             * Prep for native library loading.
149             */
150            String pathList = System.getProperty("java.library.path", ".");
151            String pathSep = System.getProperty("path.separator", ":");
152            String fileSep = System.getProperty("file.separator", "/");
153    
154            if (mRawLibPath != null) {
155                if (pathList.length() > 0) {
156                    pathList += pathSep + mRawLibPath;
157                }
158                else {
159                    pathList = mRawLibPath;
160                }
161            }
162    
163            mLibPaths = pathList.split(pathSep);
164            length = mLibPaths.length;
165    
166            // Add a '/' to the end so we don't have to do the property lookup
167            // and concatenation later.
168            for (int i = 0; i < length; i++) {
169                if (!mLibPaths[i].endsWith(fileSep))
170                    mLibPaths[i] += fileSep;
171                if (VERBOSE_DEBUG)
172                    System.out.println("Native lib path " +i+ ":  " + mLibPaths[i]);
173            }
174        }
175    
176        /**
177         * Convert a source path name and an output directory to an output
178         * file name.
179         */
180        private static String generateOutputName(String sourcePathName,
181            String outputDir) {
182    
183            StringBuilder newStr = new StringBuilder(80);
184    
185            /* start with the output directory */
186            newStr.append(outputDir);
187            if (!outputDir.endsWith("/"))
188                newStr.append("/");
189    
190            /* get the filename component of the path */
191            String sourceFileName;
192            int lastSlash = sourcePathName.lastIndexOf("/");
193            if (lastSlash < 0)
194                sourceFileName = sourcePathName;
195            else
196                sourceFileName = sourcePathName.substring(lastSlash+1);
197    
198            /*
199             * Replace ".jar", ".zip", whatever with ".dex".  We don't want to
200             * use ".odex", because the build system uses that for files that
201             * are paired with resource-only jar files.  If the VM can assume
202             * that there's no classes.dex in the matching jar, it doesn't need
203             * to open the jar to check for updated dependencies, providing a
204             * slight performance boost at startup.  The use of ".dex" here
205             * matches the use on files in /data/dalvik-cache.
206             */
207            int lastDot = sourceFileName.lastIndexOf(".");
208            if (lastDot < 0)
209                newStr.append(sourceFileName);
210            else
211                newStr.append(sourceFileName, 0, lastDot);
212            newStr.append(".dex");
213    
214            if (VERBOSE_DEBUG)
215                System.out.println("Output file will be " + newStr.toString());
216            return newStr.toString();
217        }

219        /**
220         * Finds a class. This method is called by {@code loadClass()} after the
221         * parent ClassLoader has failed to find a loaded class of the same name.
222         *
223         * @param name
224         *            The name of the class to search for, in a human-readable form
225         *            like "java.lang.String" or "java.net.URLClassLoader$3$1".
226         * @return the {@link Class} object representing the class
227         * @throws ClassNotFoundException
228         *             if the class cannot be found
229         */
230        @Override
231        protected Class<?> findClass(String name) throws ClassNotFoundException {
232            ensureInit();
233    
234            if (VERBOSE_DEBUG)
235                System.out.println("DexClassLoader " + this
236                    + ": findClass '" + name + "'");
237    
238            byte[] data = null;
239            int length = mFiles.length;
240    
241            for (int i = 0; i < length; i++) {
242                if (VERBOSE_DEBUG)
243                    System.out.println("  Now searching: " + mFiles[i].getPath());
244    
245                if (mDexs[i] != null) {
246                    String slashName = name.replace('.', '/');
247                    Class clazz = mDexs[i].loadClass(slashName, this);
248                    if (clazz != null) {
249                        if (VERBOSE_DEBUG)
250                            System.out.println("    found");
251                        return clazz;
252                    }
253                }
254            }
255    
256            throw new ClassNotFoundException(name + " in loader " + this);
257        }
258    
259        /**
260         * Finds a resource. This method is called by {@code getResource()} after
261         * the parent ClassLoader has failed to find a loaded resource of the same
262         * name.
263         *
264         * @param name
265         *            The name of the resource to find
266         * @return the location of the resource as a URL, or {@code null} if the
267         *         resource is not found.
268         */
269        @Override
270        protected URL findResource(String name) {
271            ensureInit();
272    
273            int length = mFiles.length;
274    
275            for (int i = 0; i < length; i++) {
276                File pathFile = mFiles[i];
277                ZipFile zip = mZips[i];
278    
279                if (zip.getEntry(name) != null) {
280                    if (VERBOSE_DEBUG)
281                        System.out.println("  found " + name + " in " + pathFile);
282                    try {
283                        // File.toURL() is compliant with RFC 1738 in always
284                        // creating absolute path names. If we construct the
285                        // URL by concatenating strings, we might end up with
286                        // illegal URLs for relative names.
287                        return new URL("jar:" + pathFile.toURL() + "!/" + name);
288                    } catch (MalformedURLException e) {
289                        throw new RuntimeException(e);
290                    }
291                }
292            }
293    
294            if (VERBOSE_DEBUG)
295                System.out.println("  resource " + name + " not found");
296    
297            return null;
298        }
299    
300        /**
301         * Finds a native library. This method is called after the parent
302         * ClassLoader has failed to find a native library of the same name.
303         *
304         * @param libname
305         *            The name of the library to find
306         * @return the complete path of the library, or {@code null} if the library
307         *         is not found.
308         */
309        @Override
310        protected String findLibrary(String libname) {
311            ensureInit();
312    
313            String fileName = System.mapLibraryName(libname);
314            for (int i = 0; i < mLibPaths.length; i++) {
315                String pathName = mLibPaths[i] + fileName;
316                File test = new File(pathName);
317    
318                if (test.exists()) {
319                    if (VERBOSE_DEBUG)
320                        System.out.println("  found " + libname);
321                    return pathName;
322                }
323            }
324    
325            if (VERBOSE_DEBUG)
326                System.out.println("  library " + libname + " not found");
327            return null;
328        }

330        /**
331         * Returns package information for the given package. Unfortunately, the
332         * DexClassLoader doesn't really have this information, and as a non-secure
333         * ClassLoader, it isn't even required to, according to the spec. Yet, we
334         * want to provide it, in order to make all those hopeful callers of
335         * <code>myClass.getPackage().getName()</code> happy. Thus we construct a
336         * Package object the first time it is being requested and fill most of the
337         * fields with dummy values. The Package object is then put into the
338         * ClassLoader's Package cache, so we see the same one next time. We don't
339         * create Package objects for null arguments or for the default package.
340         * <p>
341         * There a limited chance that we end up with multiple Package objects
342         * representing the same package: It can happen when when a package is
343         * scattered across different JAR files being loaded by different
344         * ClassLoaders. Rather unlikely, and given that this whole thing is more or
345         * less a workaround, probably not worth the effort.
346         *
347         * @param name
348         *            the name of the class
349         * @return the package information for the class, or {@code null} if there
350         *         is not package information available for it
351         */
352        @Override
353        protected Package getPackage(String name) {
354            if (name != null && !"".equals(name)) {
355                synchronized(this) {
356                    Package pack = super.getPackage(name);
357    
358                    if (pack == null) {
359                        pack = definePackage(name, "Unknown", "0.0", "Unknown",
360                                "Unknown", "0.0", "Unknown", null);
361                    }
362    
363                    return pack;
364                }
365            }
366    
367            return null;
368        }
