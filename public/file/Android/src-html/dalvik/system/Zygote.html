017    package dalvik.system;
018    
019    /**
020     * Provides access to the Dalvik "zygote" feature, which allows a VM instance to
021     * be partially initialized and then fork()'d from the partially initialized
022     * state.
023     * 
024     * @since Android 1.0
025     */
026    public class Zygote {
027        /*
028         * Bit values for "debugFlags" argument.  The definitions are duplicated
029         * in the native code.
030         */
031        /** enable debugging over JDWP */
032        public static final int DEBUG_ENABLE_DEBUGGER   = 1;
033        /** enable JNI checks */
034        public static final int DEBUG_ENABLE_CHECKJNI   = 1 << 1;
035        /** enable Java programming language "assert" statements */
036        public static final int DEBUG_ENABLE_ASSERT     = 1 << 2;
037    
038        private Zygote() {}
039    
040        /**
041         * Forks a new Zygote instance, but does not leave the zygote mode.
042         * The current VM must have been started with the -Xzygote flag. The
043         * new child is expected to eventually call forkAndSpecialize()
044         *
045         * @return 0 if this is the child, pid of the child
046         * if this is the parent, or -1 on error
047         */
048        native public static int fork();
049    
050        /**
051         * Forks a new VM instance.  The current VM must have been started
052         * with the -Xzygote flag. <b>NOTE: new instance keeps all
053         * root capabilities. The new process is expected to call capset()</b>.
054         *
055         * @param uid the UNIX uid that the new process should setuid() to after
056         * fork()ing and and before spawning any threads.
057         * @param gid the UNIX gid that the new process should setgid() to after
058         * fork()ing and and before spawning any threads.
059         * @param gids null-ok; a list of UNIX gids that the new process should
060         * setgroups() to after fork and before spawning any threads.
061         * @param debugFlags bit flags that enable debugging features.
062         * @param rlimits null-ok an array of rlimit tuples, with the second
063         * dimension having a length of 3 and representing
064         * (resource, rlim_cur, rlim_max). These are set via the posix
065         * setrlimit(2) call.
066         *
067         * @return 0 if this is the child, pid of the child
068         * if this is the parent, or -1 on error.
069         */
070        native public static int forkAndSpecialize(int uid, int gid, int[] gids,
071                int debugFlags, int[][] rlimits);
072    
073        /**
074         * Forks a new VM instance.
075         * @deprecated use {@link Zygote#forkAndSpecialize(int, int, int[], int, int[][])}
076         */
077        @Deprecated
078        public static int forkAndSpecialize(int uid, int gid, int[] gids,
079                boolean enableDebugger, int[][] rlimits) {
080            int debugFlags = enableDebugger ? DEBUG_ENABLE_DEBUGGER : 0;
081            return forkAndSpecialize(uid, gid, gids, debugFlags, rlimits);
082        }
083    
084        /**
085         * Special method to start the system server process. In addition to the
086         * common actions performed in forkAndSpecialize, the pid of the child
087         * process is recorded such that the death of the child process will cause 
088         * zygote to exit.
089         *
090         * @param uid the UNIX uid that the new process should setuid() to after
091         * fork()ing and and before spawning any threads.
092         * @param gid the UNIX gid that the new process should setgid() to after
093         * fork()ing and and before spawning any threads.
094         * @param gids null-ok; a list of UNIX gids that the new process should
095         * setgroups() to after fork and before spawning any threads.
096         * @param debugFlags bit flags that enable debugging features.
097         * @param rlimits null-ok an array of rlimit tuples, with the second
098         * dimension having a length of 3 and representing
099         * (resource, rlim_cur, rlim_max). These are set via the posix
100         * setrlimit(2) call.
101         *
102         * @return 0 if this is the child, pid of the child
103         * if this is the parent, or -1 on error.
104         */
105        native public static int forkSystemServer(int uid, int gid, 
106                int[] gids, int debugFlags, int[][] rlimits);
107    
108        /**
109         * Special method to start the system server process.
110         * @deprecated use {@link Zygote#forkSystemServer(int, int, int[], int, int[][])}
111         */
112        @Deprecated
113        public static int forkSystemServer(int uid, int gid, int[] gids,
114                boolean enableDebugger, int[][] rlimits) {
115            int debugFlags = enableDebugger ? DEBUG_ENABLE_DEBUGGER : 0;
116            return forkAndSpecialize(uid, gid, gids, debugFlags, rlimits);
117        }
118    }
