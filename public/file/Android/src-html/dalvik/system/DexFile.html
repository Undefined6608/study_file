017    package dalvik.system;
018    
019    import java.io.File;
020    import java.io.FileNotFoundException;
021    import java.io.IOException;
022    import java.security.ProtectionDomain;
023    import java.util.Enumeration;
024    
025    
026    /**
027     * Manipulates DEX files. The class is similar in principle to
028     * {@link java.util.zip.ZipFile}. It is used primarily by class loaders.
029     * <p>
030     * Note we don't directly open and read the DEX file here. They're memory-mapped
031     * read-only by the VM.
032     * 
033     * @since Android 1.0
034     */
035    public final class DexFile {
036        private final int mCookie;
037        private String mFileName;
038    
039        /**
040         * Opens a DEX file from a given File object. This will usually be a ZIP/JAR
041         * file with a "classes.dex" inside. The method should not be used for files
042         * inside the Dalvik cache.
043         * 
044         * @cts What will happen if we refer to the Dalvik cache? Should be either
045         *      specified or throw an exception...
046         * 
047         * @param file
048         *            the File object referencing the actual DEX file
049         * 
050         * @throws IOException
051         *             if an I/O error occurs, such as the file not being found or
052         *             access rights missing for opening it
053         */
054        public DexFile(File file) throws IOException {
055            this(file.getPath());
056        }
057    
058        /**
059         * Opens a DEX file from a given filename. This will usually be a ZIP/JAR
060         * file with a "classes.dex" inside. The method should not be used for files
061         * inside the Dalvik cache.
062         * 
063         * @cts What will happen if we refer to the Dalvik cache? Should be either
064         *      specified or throw an exception...
065         * 
066         * @param fileName
067         *            the filename of the DEX file
068         * 
069         * @throws IOException
070         *             if an I/O error occurs, such as the file not being found or
071         *             access rights missing for opening it
072         */
073        public DexFile(String fileName) throws IOException {
074            String wantDex = System.getProperty("android.vm.dexfile", "false");
075            if (!wantDex.equals("true"))
076                throw new UnsupportedOperationException("No dex in this VM");
077    
078            mCookie = openDexFile(fileName, null, 0);
079            mFileName = fileName;
080            //System.out.println("DEX FILE cookie is " + mCookie);
081        }
082    
083        /**
084         * Opens a DEX file from a given filename, using a specified file
085         * to hold the optimized data.
086         *
087         * @param sourceName
088         *  Jar or APK file with "classes.dex".
089         * @param outputName
090         *  File that will hold the optimized form of the DEX data.
091         * @param flags
092         *  Enable optional features.
093         */
094        private DexFile(String sourceName, String outputName, int flags)
095            throws IOException {
096    
097            String wantDex = System.getProperty("android.vm.dexfile", "false");
098            if (!wantDex.equals("true"))
099                throw new UnsupportedOperationException("No dex in this VM");
100    
101            mCookie = openDexFile(sourceName, outputName, flags);
102            mFileName = sourceName;
103            //System.out.println("DEX FILE cookie is " + mCookie);
104        }
105    
106        /**
107         * Open a DEX file, specifying the file in which the optimized DEX
108         * data should be written.  If the optimized form exists and appears
109         * to be current, it will be used; if not, the VM will attempt to
110         * regenerate it.
111         *
112         * This is intended for use by applications that wish to download
113         * and execute DEX files outside the usual application installation
114         * mechanism.  This function should not be called directly by an
115         * application; instead, use a class loader such as
116         * dalvik.system.DexClassLoader.
117         *
118         * @param sourcePathName
119         *  Jar or APK file with "classes.dex".  (May expand this to include
120         *  "raw DEX" in the future.)
121         * @param outputPathName
122         *  File that will hold the optimized form of the DEX data.
123         * @param flags
124         *  Enable optional features.  (Currently none defined.)
125         * @return
126         *  A new or previously-opened DexFile.
127         * @throws IOException
128         *  If unable to open the source or output file.
129         */
130        static public DexFile loadDex(String sourcePathName, String outputPathName,
131            int flags) throws IOException {
132    
133            /*
134             * TODO: we may want to cache previously-opened DexFile objects.
135             * The cache would be synchronized with close().  This would help
136             * us avoid mapping the same DEX more than once when an app
137             * decided to open it multiple times.  In practice this may not
138             * be a real issue.
139             */
140            return new DexFile(sourcePathName, outputPathName, flags);
141        }
142    
143        /**
144         * Gets the name of the (already opened) DEX file.
145         * 
146         * @return the file name
147         */
148        public String getName() {
149            return mFileName;
150        }
151    
152        /**
153         * Closes the DEX file.
154         * <p>
155         * This may not be able to release any resources. If classes have been
156         * loaded, the underlying storage can't be discarded.
157         * 
158         * @throws IOException
159         *             if an I/O error occurs during closing the file, which
160         *             normally should not happen
161         * 
162         * @cts Second sentence is a bit cryptic.
163         */
164        public void close() throws IOException {
165            closeDexFile(mCookie);
166        }
167    
168        /**
169         * Loads a class. Returns the class on success, or a {@code null} reference
170         * on failure.
171         * <p>
172         * If you are not calling this from a class loader, this is most likely not
173         * going to do what you want. Use {@link Class#forName(String)} instead.
174         * <p>
175         * The method does not throw {@link ClassNotFoundException} if the class
176         * isn't found because it isn't feasible to throw exceptions wildly every
177         * time a class is not found in the first DEX file we look at. It will
178         * throw exceptions for other failures, though.
179         * 
180         * @param name
181         *            the class name, which should look like "java/lang/String"
182         * 
183         * @param loader
184         *            the class loader that tries to load the class (in most cases
185         *            the caller of the method
186         * 
187         * @return the {@link Class} object representing the class, or {@code null}
188         *         if the class cannot be loaded
189         * 
190         * @cts Exception comment is a bit cryptic. What exception will be thrown?
191         */
192        public Class loadClass(String name, ClassLoader loader) {
193            return defineClass(name, loader, mCookie,
194                null);
195                //new ProtectionDomain(name) /*DEBUG ONLY*/);
196        }
197        
198        native private static Class defineClass(String name, ClassLoader loader,
199            int cookie, ProtectionDomain pd);
200    
201        /**
202         * Enumerate the names of the classes in this DEX file.
203         * 
204         * @return an enumeration of names of classes contained in the DEX file, in
205         *         the usual internal form (like "java/lang/String").
206         */
207        public Enumeration<String> entries() {
208            return new DFEnum(this);
209        }
210    
211        /*
212         * Helper class.
213         */
214        private class DFEnum implements Enumeration<String> {
215            private int mIndex;
216            private String[] mNameList;
217    
218            DFEnum(DexFile df) {
219                mIndex = 0;
220                mNameList = getClassNameList(mCookie);
221            }
222    
223            public boolean hasMoreElements() {
224                return (mIndex < mNameList.length);
225            }
226    
227            public String nextElement() {
228                return mNameList[mIndex++];
229            }
230        }
231    
232        /* return a String array with class names */
233        native private static String[] getClassNameList(int cookie);
234    
235        /** 
236         * Called when the class is finalized. Makes sure the DEX file is closed.
237         * 
238         * @throws IOException
239         *             if an I/O error occurs during closing the file, which
240         *             normally should not happen
241         */
242        protected void finalize() throws IOException {
243            close();
244        }
245    
246        /*
247         * Open a DEX file.  The value returned is a magic VM cookie.  On
248         * failure, an IOException is thrown.
249         */
250        native private static int openDexFile(String sourceName, String outputName,
251            int flags) throws IOException;
252    
253        /*
254         * Close DEX file.
255         */
256        native private static void closeDexFile(int cookie);
257    
258        /**
259         * Returns true if the VM believes that the apk/jar file is out of date
260         * and should be passed through "dexopt" again.
261         *
262         * @param fileName the absolute path to the apk/jar file to examine.
263         * @return true if dexopt should be called on the file, false otherwise.
264         * @throws java.io.FileNotFoundException if fileName is not readable,
265         *         not a file, or not present.
266         * @throws java.io.IOException if fileName is not a valid apk/jar file or
267         *         if problems occur while parsing it.
268         * @throws java.lang.NullPointerException if fileName is null.
269         * @throws dalvik.system.StaleDexCacheError if the optimized dex file
270         *         is stale but exists on a read-only partition.
271         */
272        native public static boolean isDexOptNeeded(String fileName)
273                throws FileNotFoundException, IOException;
274    }
