020    import java.io.ByteArrayInputStream;
021    import java.io.InputStream;
022    import java.lang.reflect.Constructor;
023    import java.math.BigInteger;
024    import java.security.AccessController;
025    import java.security.InvalidKeyException;
026    import java.security.NoSuchAlgorithmException;
027    import java.security.NoSuchProviderException;
028    import java.security.Principal;
029    import java.security.PublicKey;
030    import java.security.Security;
031    import java.security.SignatureException;
032    import java.security.cert.CertificateFactory;
033    import java.util.Date;
034    import javax.security.cert.Certificate;
035    import javax.security.cert.CertificateEncodingException;
036    import javax.security.cert.CertificateException;
037    import javax.security.cert.CertificateExpiredException;
038    import javax.security.cert.CertificateNotYetValidException;

040    import org.apache.harmony.security.internal.nls.Messages;

042    /**
043     * Abstract base class for X.509 certificates.
044     * <p>
045     * This represents a standard way for accessing the attributes of X.509 v1
046     * certificates.
047     * </p>
048     * <p>
049     * Note: This package is provided only for compatibility reasons.
050     * It contains a simplified version of the java.security.cert package that was
051     * previously used by JSSE (Java SSL package). All applications that do not have
052     * to be compatible with older versions of JSSE (that is before Java SDK 1.5)
053     * should only use java.security.cert.
054     * </p>
055     * @since Android 1.0
056     */
057    public abstract class X509Certificate extends Certificate {
058    
059        private static Constructor constructor;
060        
061        static {
062            try {
063                String classname = (String) AccessController.doPrivileged(
064                    new java.security.PrivilegedAction() {
065                        public Object run() {
066                            return Security.getProperty("cert.provider.x509v1"); //$NON-NLS-1$
067                        }
068                    }
069                );
070                Class cl = Class.forName(classname);
071                constructor =
072                    cl.getConstructor(new Class[] {InputStream.class});
073            } catch (Throwable e) {
074            }
075        }
076        
077        /**
078         * Creates a new {@code X509Certificate}.
079         * 
080         * @since Android 1.0
081         */
082        public X509Certificate() {
083            super();
084        }
085    
086        /**
087         * Creates a new {@code X509Certificate} and initializes it from the
088         * specified input stream.
089         * 
090         * @param inStream
091         *            input stream containing data to initialize the certificate.
092         * @return the certificate initialized from the specified input stream
093         * @throws CertificateException
094         *             if the certificate cannot be created or initialized.
095         * @since Android 1.0
096         */
097        public static final X509Certificate getInstance(InputStream inStream)
098                                                 throws CertificateException {
099            if (inStream == null) {
100                throw new CertificateException(Messages.getString("security.87")); //$NON-NLS-1$
101            }
102            if (constructor != null) {
103                try {
104                    return (X509Certificate) 
105                        constructor.newInstance(new Object[] {inStream});
106                } catch (Throwable e) {
107                    throw new CertificateException(e.getMessage());
108                }
109            }
110    
111            final java.security.cert.X509Certificate cert;
112            try {
113                CertificateFactory cf = CertificateFactory.getInstance("X.509"); //$NON-NLS-1$
114                cert = (java.security.cert.X509Certificate)
115                                                cf.generateCertificate(inStream);
116            } catch (java.security.cert.CertificateException e) {
117                throw new CertificateException(e.getMessage());
118            }
119    
120            return new X509Certificate() {
121    
122                public byte[] getEncoded() throws CertificateEncodingException {
123                    try {
124                        return cert.getEncoded();
125                    } catch (java.security.cert.CertificateEncodingException e) {
126                        throw new CertificateEncodingException(e.getMessage());
127                    }
128                }
129    
130                public void verify(PublicKey key) throws CertificateException,
131                                    NoSuchAlgorithmException, InvalidKeyException,
132                                    NoSuchProviderException, SignatureException {
133                    try {
134                        cert.verify(key);
135                    } catch (java.security.cert.CertificateException e) {
136                        throw new CertificateException(e.getMessage());
137                    }
138                }
139    
140                public void verify(PublicKey key, String sigProvider)
141                                throws CertificateException,
142                                    NoSuchAlgorithmException, InvalidKeyException,
143                                    NoSuchProviderException, SignatureException {
144                    try {
145                        cert.verify(key, sigProvider);
146                    } catch (java.security.cert.CertificateException e) {
147                        throw new CertificateException(e.getMessage());
148                    }
149                }
150    
151                public String toString() {
152                    return cert.toString();
153                }
154    
155                public PublicKey getPublicKey() {
156                    return cert.getPublicKey();
157                }
158    
159                public void checkValidity() throws CertificateExpiredException,
160                                       CertificateNotYetValidException {
161                    try {
162                        cert.checkValidity();
163                    } catch (java.security.cert.CertificateNotYetValidException e) {
164                        throw new CertificateNotYetValidException(e.getMessage());
165                    } catch (java.security.cert.CertificateExpiredException e) {
166                        throw new CertificateExpiredException(e.getMessage());
167                    }
168                }
169    
170                public void checkValidity(Date date) 
171                                throws CertificateExpiredException,
172                                       CertificateNotYetValidException {
173                    try {
174                        cert.checkValidity(date);
175                    } catch (java.security.cert.CertificateNotYetValidException e) {
176                        throw new CertificateNotYetValidException(e.getMessage());
177                    } catch (java.security.cert.CertificateExpiredException e) {
178                        throw new CertificateExpiredException(e.getMessage());
179                    }
180                }
181    
182                public int getVersion() {
183                    return 2;
184                }
185    
186                public BigInteger getSerialNumber() {
187                    return cert.getSerialNumber();
188                }
189    
190                public Principal getIssuerDN() {
191                    return cert.getIssuerDN();
192                }
193    
194                public Principal getSubjectDN() {
195                    return cert.getSubjectDN();
196                }
197    
198                public Date getNotBefore() {
199                    return cert.getNotBefore();
200                }
201    
202                public Date getNotAfter() {
203                    return cert.getNotAfter();
204                }
205    
206                public String getSigAlgName() {
207                    return cert.getSigAlgName();
208                }
209    
210                public String getSigAlgOID() {
211                    return cert.getSigAlgOID();
212                }
213    
214                public byte[] getSigAlgParams() {
215                    return cert.getSigAlgParams();
216                }
217            };
218        }
219    
220        /**
221         * Creates a new {@code X509Certificate} and initializes it from the
222         * specified byte array.
223         * 
224         * @param certData
225         *            byte array containing data to initialize the certificate.
226         * @return the certificate initialized from the specified byte array
227         * @throws CertificateException
228         *             if the certificate cannot be created or initialized.
229         * @since Android 1.0
230         */
231        public static final X509Certificate getInstance(byte[] certData)
232                                                 throws CertificateException {
233            if (certData == null) {
234                throw new CertificateException(Messages.getString("security.88")); //$NON-NLS-1$
235            }
236            ByteArrayInputStream bais = new ByteArrayInputStream(certData);
237            return getInstance(bais);
238        }
239    
240        /**
241         * Checks whether the certificate is currently valid.
242         * <p>
243         * The validity defined in ASN.1:
244         * 
245         * <pre>
246         * validity             Validity
247         * 
248         * Validity ::= SEQUENCE { 
249         *      notBefore       CertificateValidityDate, 
250         *      notAfter        CertificateValidityDate }
251         * 
252         * CertificateValidityDate ::= CHOICE { 
253         *      utcTime         UTCTime, 
254         *      generalTime     GeneralizedTime }
255         * </pre>
256         * 
257         * </p>
258         * 
259         * @throws CertificateExpiredException
260         *             if the certificate has expired.
261         * @throws CertificateNotYetValidException
262         *             if the certificate is not yet valid.
263         * @since Android 1.0
264         */
265        public abstract void checkValidity()
266                throws CertificateExpiredException, CertificateNotYetValidException;
267    
268    
269        /**
270         * Checks whether the certificate is valid at the specified date.
271         * 
272         * @param date
273         *            the date to check the validity against.
274         * @throws CertificateExpiredException
275         *             if the certificate has expired.
276         * @throws CertificateNotYetValidException
277         *             if the certificate is not yet valid.
278         * @see #checkValidity()
279         * @since Android 1.0
280         */
281        public abstract void checkValidity(Date date)
282                throws CertificateExpiredException, CertificateNotYetValidException;
283    
284        /**
285         * Returns the certificates {@code version} (version number).
286         * <p>
287         * The version defined is ASN.1:
288         * 
289         * <pre>
290         * Version ::=  INTEGER  {  v1(0), v2(1), v3(2)  }
291         * </pre>
292         * 
293         * </p>
294         * 
295         * @return the version number.
296         * @since Android 1.0
297         */
298        public abstract int getVersion();
299    
300        /**
301         * Returns the {@code serialNumber} of the certificate.
302         * <p>
303         * The ASN.1 definition of {@code serialNumber}:
304         * 
305         * <pre>
306         * CertificateSerialNumber  ::=  INTEGER
307         * </pre>
308         * 
309         * </p>
310         * 
311         * @return the serial number.
312         * @since Android 1.0
313         */
314        public abstract BigInteger getSerialNumber();
315    
316        /**
317         * Returns the {@code issuer} (issuer distinguished name) as an
318         * implementation specific {@code Principal} object.
319         * <p>
320         * The ASN.1 definition of {@code issuer}:
321         * 
322         * <pre>
323         *  issuer      Name
324         * 
325         *  Name ::= CHOICE {
326         *      RDNSequence }
327         * 
328         *    RDNSequence ::= SEQUENCE OF RelativeDistinguishedName
329         * 
330         *    RelativeDistinguishedName ::= SET OF AttributeTypeAndValue
331         * 
332         *    AttributeTypeAndValue ::= SEQUENCE {
333         *      type     AttributeType,
334         *      value    AttributeValue }
335         * 
336         *    AttributeType ::= OBJECT IDENTIFIER
337         * 
338         *    AttributeValue ::= ANY DEFINED BY AttributeType
339         * </pre>
340         * 
341         * </p>
342         * 
343         * @return the {@code issuer} as an implementation specific {@code
344         *         Principal}.
345         * @since Android 1.0
346         */
347        public abstract Principal getIssuerDN();
348    
349        /**
350         * Returns the {@code subject} (subject distinguished name) as an
351         * implementation specific {@code Principal} object.
352         * <p>
353         * The ASN.1 definition of {@code subject}:
354         * 
355         * <pre>
356         * subject      Name
357         * 
358         *  Name ::= CHOICE {
359         *      RDNSequence }
360         * 
361         *    RDNSequence ::= SEQUENCE OF RelativeDistinguishedName
362         * 
363         *    RelativeDistinguishedName ::= SET OF AttributeTypeAndValue
364         * 
365         *    AttributeTypeAndValue ::= SEQUENCE {
366         *      type     AttributeType,
367         *      value    AttributeValue }
368         * 
369         *    AttributeType ::= OBJECT IDENTIFIER
370         * 
371         *    AttributeValue ::= ANY DEFINED BY AttributeType
372         * </pre>
373         * 
374         * </p>
375         * 
376         * @return the {@code subject} (subject distinguished name).
377         * @since Android 1.0
378         */
379        public abstract Principal getSubjectDN();
380    
381        /**
382         * Returns the {@code notBefore} date from the validity period of the
383         * certificate.
384         * 
385         * @return the start of the validity period.
386         * @since Android 1.0
387         */
388        public abstract Date getNotBefore();
389    
390        /**
391         * Returns the {@code notAfter} date of the validity period of the
392         * certificate.
393         * 
394         * @return the end of the validity period.
395         * @since Android 1.0
396         */
397        public abstract Date getNotAfter();
398    
399        /**
400         * Returns the name of the algorithm for the certificate signature.
401         * 
402         * @return the signature algorithm name.
403         * @since Android 1.0
404         */
405        public abstract String getSigAlgName();
406    
407        /**
408         * Returns the OID of the signature algorithm from the certificate.
409         * 
410         * @return the OID of the signature algorithm.
411         * @since Android 1.0
412         */
413        public abstract String getSigAlgOID();
414    
415        /**
416         * Returns the parameters of the signature algorithm in DER-encoded format.
417         * 
418         * @return the parameters of the signature algorithm, or null if none are
419         *         used.
420         * @since Android 1.0
421         */
422        public abstract byte[] getSigAlgParams();
423    }
