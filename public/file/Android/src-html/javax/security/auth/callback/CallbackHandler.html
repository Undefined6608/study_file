018    package javax.security.auth.callback;
019    
020    import java.io.IOException;
021    
022    /**
023     * Needs to be implemented by classes that want to handle authentication
024     * {@link Callback}s. A single method {@link #handle(Callback[])} must be
025     * provided that checks the type of the incoming {@code Callback}s and reacts
026     * accordingly. {@code CallbackHandler}s can be installed per application. It is
027     * also possible to configure a system-default {@code CallbackHandler} by
028     * setting the {@code auth.login.defaultCallbackHandler} property in the
029     * standard {@code security.properties} file.
030     * 
031     * @since Android 1.0
032     */
033    public interface CallbackHandler {
034    
035        /**
036         * Handles the actual {@link Callback}. A {@code CallbackHandler} needs to
037         * implement this method. In the method, it is free to select which {@code
038         * Callback}s it actually wants to handle and in which way. For example, a
039         * console-based {@code CallbackHandler} might choose to sequentially ask
040         * the user for login and password, if it implements these {@code Callback}
041         * s, whereas a GUI-based one might open a single dialog window for both
042         * values. If a {@code CallbackHandler} is not able to handle a specific
043         * {@code Callback}, it needs to throw an
044         * {@link UnsupportedCallbackException}.
045         * 
046         * @param callbacks
047         *            the array of {@code Callback}s that need handling
048         * @throws IOException
049         *             if an I/O related error occurs
050         * @throws UnsupportedCallbackException
051         *             if the {@code CallbackHandler} is not able to handle a
052         *             specific {@code Callback}
053         */
054        void handle(Callback[] callbacks) throws IOException, UnsupportedCallbackException;
055    
056    }
