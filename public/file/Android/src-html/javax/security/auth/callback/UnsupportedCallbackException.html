018    package javax.security.auth.callback;
019    
020    /**
021     * Thrown when a {@link CallbackHandler} does not support a particular {@link
022     * Callback}.
023     * 
024     * @since Android 1.0
025     */
026    public class UnsupportedCallbackException extends Exception {
027    
028        private static final long serialVersionUID = -6873556327655666839L;
029    
030        private Callback callback;
031    
032        /**
033         * Creates a new exception instance and initializes it with just the
034         * unsupported {@code Callback}, but no error message.
035         * 
036         * @param callback
037         *            the {@code Callback}
038         */
039        public UnsupportedCallbackException(Callback callback) {
040            super();
041            this.callback = callback;
042        }
043    
044        /**
045         * Creates a new exception instance and initializes it with both the
046         * unsupported {@code Callback} and an error message.
047         * 
048         * @param callback
049         *            the {@code Callback}
050         * @param message
051         *            the error message
052         */
053        public UnsupportedCallbackException(Callback callback, String message) {
054            super(message);
055            this.callback = callback;
056        }
057    
058        /**
059         * Returns the unsupported {@code Callback} that triggered this exception.
060         * 
061         * @return the {@code Callback}
062         */
063        public Callback getCallback() {
064            return callback;
065        }
066    }
