018    package javax.security.auth;
019    
020    import java.security.DomainCombiner;
021    import java.security.Principal;
022    import java.security.ProtectionDomain;
023    import java.util.Set;
024    
025    /**
026     * Merges permissions based on code source and code signers with permissions
027     * granted to the specified {@link Subject}.
028     * 
029     * @since Android 1.0
030     */
031    public class SubjectDomainCombiner implements DomainCombiner {
032    
033        // subject to be associated
034        private Subject subject;
035    
036        // permission required to get a subject object
037        private static final AuthPermission _GET = new AuthPermission(
038                "getSubjectFromDomainCombiner"); //$NON-NLS-1$
039    
040        /**
041         * Creates a domain combiner for the entity provided in {@code subject}.
042         * 
043         * @param subject
044         *            the entity to which this domain combiner is associated.
045         */
046        public SubjectDomainCombiner(Subject subject) {
047            super();
048            if (subject == null) {
049                throw new NullPointerException();
050            }
051            this.subject = subject;
052        }
053    
054        /**
055         * Returns the entity to which this domain combiner is associated.
056         * 
057         * @return the entity to which this domain combiner is associated.
058         */
059        public Subject getSubject() {
060            SecurityManager sm = System.getSecurityManager();
061            if (sm != null) {
062                sm.checkPermission(_GET);
063            }
064    
065            return subject;
066        }
067    
068        /**
069         * Merges the {@code ProtectionDomain} with the {@code Principal}s
070         * associated with the subject of this {@code SubjectDomainCombiner}.
071         * 
072         * @param currentDomains
073         *            the {@code ProtectionDomain}s associated with the context of
074         *            the current thread. The domains must be sorted according to
075         *            the execution order, the most recent residing at the
076         *            beginning.
077         * @param assignedDomains
078         *            the {@code ProtectionDomain}s from the parent thread based on
079         *            code source and signers.
080         * @return a single {@code ProtectionDomain} array computed from the two
081         *         provided arrays, or {@code null}.
082         * @see ProtectionDomain
083         */
084        public ProtectionDomain[] combine(ProtectionDomain[] currentDomains,
085                ProtectionDomain[] assignedDomains) {
086            // get array length for combining protection domains
087            int len = 0;
088            if (currentDomains != null) {
089                len += currentDomains.length;
090            }
091            if (assignedDomains != null) {
092                len += assignedDomains.length;
093            }
094            if (len == 0) {
095                return null;
096            }
097    
098            ProtectionDomain[] pd = new ProtectionDomain[len];
099    
100            // for each current domain substitute set of principal with subject's
101            int cur = 0;
102            if (currentDomains != null) {
103    
104                Set<Principal> s = subject.getPrincipals();
105                Principal[] p = s.toArray(new Principal[s.size()]);
106    
107                for (cur = 0; cur < currentDomains.length; cur++) {
108                    ProtectionDomain newPD;
109                    newPD = new ProtectionDomain(currentDomains[cur].getCodeSource(),
110                            currentDomains[cur].getPermissions(), currentDomains[cur]
111                                    .getClassLoader(), p);
112                    pd[cur] = newPD;
113                }
114            }
115    
116            // copy assigned domains
117            if (assignedDomains != null) {
118                System.arraycopy(assignedDomains, 0, pd, cur, assignedDomains.length);
119            }
120    
121            return pd;
122        }
123    }
