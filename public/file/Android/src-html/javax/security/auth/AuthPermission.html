018    package javax.security.auth;
019    
020    import java.security.BasicPermission;
021    
022    import org.apache.harmony.auth.internal.nls.Messages;
023    
024    /**
025     * Governs the use of methods in this package and also its subpackages. A
026     * <i>target name</i> of the permission specifies which methods are allowed
027     * without specifying the concrete action lists. Possible target names and
028     * associated authentication permissions are:
029     * 
030     * <pre>
031     *    doAs                      invoke Subject.doAs methods.
032     *    doAsPrivileged            invoke the Subject.doAsPrivileged methods.
033     *    getSubject                invoke Subject.getSubject().
034     *    getSubjectFromDomainCombiner    invoke SubjectDomainCombiner.getSubject().
035     *    setReadOnly               invoke Subject.setReadonly().
036     *    modifyPrincipals          modify the set of principals 
037     *                              associated with a Subject.
038     *    modifyPublicCredentials   modify the set of public credentials
039     *                              associated with a Subject.
040     *    modifyPrivateCredentials  modify the set of private credentials
041     *                              associated with a Subject.
042     *    refreshCredential         invoke the refresh method on a credential of a 
043     *                              refreshable credential class.
044     *    destroyCredential         invoke the destroy method on a credential of a
045     *                              destroyable credential class.
046     *    createLoginContext.<i>name</i>   instantiate a LoginContext with the
047     *                              specified name. The wildcard name ('*')
048     *                              allows to a LoginContext of any name.
049     *    getLoginConfiguration     invoke the getConfiguration method of 
050     *                              javax.security.auth.login.Configuration.
051     *    refreshLoginConfiguration Invoke the refresh method of 
052     *                              javax.security.auth.login.Configuration.
053     * </pre>
054     * 
055     * @since Android 1.0
056     */
057    public final class AuthPermission extends BasicPermission {
058    
059        private static final long serialVersionUID = 5806031445061587174L;
060    
061        private static final String CREATE_LOGIN_CONTEXT = "createLoginContext"; //$NON-NLS-1$
062    
063        private static final String CREATE_LOGIN_CONTEXT_ANY = "createLoginContext.*"; //$NON-NLS-1$
064    
065        // inits permission name.
066        private static String init(String name) {
067    
068            if (name == null) {
069                throw new NullPointerException(Messages.getString("auth.13")); //$NON-NLS-1$
070            }
071    
072            if (CREATE_LOGIN_CONTEXT.equals(name)) {
073                return CREATE_LOGIN_CONTEXT_ANY;
074            }
075            return name;
076        }
077    
078        /**
079         * Creates an authentication permission with the specified target name.
080         * 
081         * @param name
082         *            the target name of this authentication permission.
083         */
084        public AuthPermission(String name) {
085            super(init(name));
086        }
087    
088        /**
089         * Creates an authentication permission with the specified target name.
090         * 
091         * @param name
092         *            the target name of this authentication permission.
093         * @param actions
094         *            this parameter is ignored and should be {@code null}.
095         */
096        public AuthPermission(String name, String actions) {
097            super(init(name), actions);
098        }
099    }
