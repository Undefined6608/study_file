017    package javax.xml;
018    
019    /**
020     * Defines several standard constants that are often used during XML processing.
021     *  
022     * @since Android 1.0
023     */
024    public class XMLConstants {
025    
026        /**
027         * The default namespace prefix. Defined to be the empty string. 
028         */
029        public static final String DEFAULT_NS_PREFIX = "";
030    
031        /**
032         * The SAX feature name for secure processing. Turning on this feature
033         * might result in a parser rejecting XML documents that are considered
034         * "insecure" (having a potential for DOS attacks, for example). The
035         * Android XML parsing implementation currently ignores this feature.  
036         */
037        public static final String FEATURE_SECURE_PROCESSING = 
038            "http://javax.xml.XMLConstants/feature/secure-processing";
039    
040        /**
041         * The namespace URI for the case that no namespace is being used at all.
042         * Defined to be the empty string.
043         */
044        public static final String NULL_NS_URI = "";
045    
046        /**
047         * The official Relax-NG namespace URI.
048         */
049        public static final String RELAXNG_NS_URI = 
050            "http://relaxng.org/ns/structure/1.0";
051    
052        /**
053         * The official XSchema instance namespace URI, as defined by W3C.
054         */
055        public static final String W3C_XML_SCHEMA_INSTANCE_NS_URI = 
056            "http://www.w3.org/2001/XMLSchema-instance";
057    
058        /**
059         * The official XSchema namespace URI, as defined by W3C.
060         */
061        public static final String W3C_XML_SCHEMA_NS_URI = 
062            "http://www.w3.org/2001/XMLSchema";
063    
064        /**
065         * The official XPath datatype namespace URI, as defined by W3C.
066         */
067        public static final String W3C_XPATH_DATATYPE_NS_URI = 
068            "http://www.w3.org/2003/11/xpath-datatypes";
069    
070        /**
071         * The official XML namespace attribute, as defined by W3C.
072         */
073        public static final String XMLNS_ATTRIBUTE = "xmlns";
074    
075        /**
076         * The official XML namespace attribute URI, as defined by W3C.
077         */
078        public static final String XMLNS_ATTRIBUTE_NS_URI = 
079            "http://www.w3.org/2000/xmlns/";
080    
081        /**
082         * The official XML DTD namespace URI, as defined by W3C.
083         */
084        public static final String XML_DTD_NS_URI = "http://www.w3.org/TR/REC-xml";
085    
086        /**
087         * The official XML namespace prefix, as defined by W3C. 
088         */
089        public static final String XML_NS_PREFIX = "xml";
090    
091        /**
092         * The official XML namespace URI, as defined by W3C. 
093         */
094        public static final String XML_NS_URI = 
095            "http://www.w3.org/XML/1998/namespace";
096    
097    }
