018    package javax.crypto;
019    
020    import java.security.Key;
021    import java.security.InvalidKeyException;
022    import java.security.InvalidAlgorithmParameterException;
023    import java.security.spec.AlgorithmParameterSpec;
024    import java.nio.ByteBuffer;
025    
026    /**
027     * The <i>Service-Provider Interface</i> (<b>SPI</b>) definition for the {@code
028     * Mac} class.
029     * 
030     * @see Mac
031     * @since Android 1.0
032     */
033    public abstract class MacSpi {
034        
035        /**
036         * Creates a new {@code MacSpi} instance.
037         * 
038         * @since Android 1.0
039         */
040        public MacSpi() {
041        }
042    
043        /**
044         * Returns the length of this MAC (in bytes).
045         * 
046         * @return the length of this MAC (in bytes).
047         * @since Android 1.0
048         */
049        protected abstract int engineGetMacLength();
050    
051        /**
052         * Initializes this {@code MacSpi} instance with the specified key and
053         * algorithm parameters.
054         * 
055         * @param key
056         *            the key to initialize this algorithm.
057         * @param params
058         *            the parameters for this algorithm.
059         * @throws InvalidKeyException
060         *             if the specified key cannot be used to initialize this
061         *             algorithm, or it is {@code null}.
062         * @throws InvalidAlgorithmParameterException
063         *             if the specified parameters cannot be used to initialize this
064         *             algorithm.
065         * @since Android 1.0
066         */
067        protected abstract void engineInit(Key key, AlgorithmParameterSpec params)
068                throws InvalidKeyException, InvalidAlgorithmParameterException;
069    
070        /**
071         * Updates this {@code MacSpi} instance with the specified byte.
072         * 
073         * @param input
074         *            the byte.
075         * @since Android 1.0
076         */
077        protected abstract void engineUpdate(byte input);
078    
079        /**
080         * Updates this {@code MacSpi} instance with the data from the specified
081         * buffer {@code input} from the specified {@code offset} and length {@code
082         * len}.
083         * 
084         * @param input
085         *            the buffer.
086         * @param offset
087         *            the offset in the buffer.
088         * @param len
089         *            the length of the data in the buffer.
090         * @since Android 1.0
091         */
092        protected abstract void engineUpdate(byte[] input, int offset, int len);
093    
094        /**
095         * Updates this {@code MacSpi} instance with the data from the specified
096         * buffer, starting at {@link ByteBuffer#position()}, including the next
097         * {@link ByteBuffer#remaining()} bytes.
098         * 
099         * @param input
100         *            the buffer.
101         * @since Android 1.0
102         */
103        protected void engineUpdate(ByteBuffer input) {
104            if (!input.hasRemaining()) {
105                return;
106            }
107            byte[] bInput;
108            if (input.hasArray()) {
109                bInput = input.array();
110                int offset = input.arrayOffset();
111                int position = input.position();
112                int limit = input.limit();
113                engineUpdate(bInput, offset + position, limit - position);
114                input.position(limit);
115            } else {
116                bInput = new byte[input.limit() - input.position()];
117                input.get(bInput);
118                engineUpdate(bInput, 0, bInput.length);
119            }
120        }
121    
122        /**
123         * Computes the digest of this MAC based on the data previously specified in
124         * {@link #engineUpdate} calls.
125         * <p>
126         * This {@code MacSpi} instance is reverted to its initial state and
127         * can be used to start the next MAC computation with the same parameters or
128         * initialized with different parameters.
129         * </p>
130         * 
131         * @return the generated digest.
132         * @since Android 1.0
133         */
134        protected abstract byte[] engineDoFinal();
135    
136        /**
137         * Resets this {@code MacSpi} instance to its initial state.
138         * <p>
139         * This {@code MacSpi} instance is reverted to its initial state and can be
140         * used to start the next MAC computation with the same parameters or
141         * initialized with different parameters.
142         * </p>
143         * 
144         * @since Android 1.0
145         */
146        protected abstract void engineReset();
147    
148        /**
149         * Clones this {@code MacSpi} instance.
150         * 
151         * @return the cloned instance.
152         * @throws CloneNotSupportedException
153         *             if cloning is not supported.
154         * @since Android 1.0
155         */
156        @Override
157        public Object clone() throws CloneNotSupportedException {
158            return super.clone();
159        }
160    }
