018    package javax.crypto;
019    
020    import java.io.ByteArrayInputStream;
021    import java.io.ByteArrayOutputStream;
022    import java.io.IOException;
023    import java.io.ObjectInputStream;
024    import java.io.ObjectOutputStream;
025    import java.io.Serializable;
026    import java.security.AlgorithmParameters;
027    import java.security.InvalidAlgorithmParameterException;
028    import java.security.InvalidKeyException;
029    import java.security.Key;
030    import java.security.NoSuchAlgorithmException;
031    import java.security.NoSuchProviderException;
032    import javax.crypto.IllegalBlockSizeException;
033    import javax.crypto.NoSuchPaddingException;
034    
035    import org.apache.harmony.crypto.internal.nls.Messages;
036    
037    /**
038     * A {@code SealedObject} is a wrapper around a {@code serializable} object
039     * instance and encrypts it using a cryptographic cipher.
040     * <p>
041     * Since a {@code SealedObject} instance is a serializable object itself it can
042     * either be stored or transmitted over an insecure channel.
043     * </p>
044     * The wrapped object can later be decrypted (unsealed) using the corresponding
045     * key and then be deserialized to retrieve the original object.The sealed
046     * object itself keeps track of the cipher and corresponding parameters.
047     * 
048     * @since Android 1.0
049     */
050    public class SealedObject implements Serializable {
051    
052        // the value of this field was derived by using serialver utility
053        /**
054         * @com.intel.drl.spec_ref
055         */
056        private static final long serialVersionUID = 4482838265551344752L;
057    
058        /**
059         * The {@link AlgorithmParameters} in encoded format.
060         */
061        protected byte[] encodedParams;
062        private byte[] encryptedContent;
063        private String sealAlg;
064        private String paramsAlg;
065    
066        private void readObject(ObjectInputStream s)
067                    throws IOException, ClassNotFoundException {
068            encodedParams = (byte []) s.readUnshared();
069            encryptedContent = (byte []) s.readUnshared();
070            sealAlg = (String) s.readUnshared();
071            paramsAlg = (String) s.readUnshared();
072        }
073    
074        /**
075         * Creates a new {@code SealedObject} instance wrapping the specified object
076         * and sealing it using the specified cipher.
077         * <p>
078         * The cipher must be fully initialized.
079         * </p>
080         * 
081         * @param object
082         *            the object to seal, can be {@code null}.
083         * @param c
084         *            the cipher to encrypt the object.
085         * @throws IOException
086         *             if the serialization fails.
087         * @throws IllegalBlockSizeException
088         *             if the specified cipher is a block cipher and the length of
089         *             the serialized data is not a multiple of the ciphers block
090         *             size.
091         * @throws NullPointerException
092         *             if the cipher is {@code null}.
093         * @since Android 1.0
094         */
095        public SealedObject(Serializable object, Cipher c)
096                    throws IOException, IllegalBlockSizeException {
097            if (c == null) {
098                throw new NullPointerException(Messages.getString("crypto.13")); //$NON-NLS-1$
099            }
100            try {
101                ByteArrayOutputStream bos = new ByteArrayOutputStream();
102                ObjectOutputStream oos = new ObjectOutputStream(bos);
103                oos.writeObject(object);
104                oos.flush();
105                AlgorithmParameters ap = c.getParameters();
106                this.encodedParams = (ap == null) ? null : ap.getEncoded();
107                this.paramsAlg = (ap == null) ? null : ap.getAlgorithm();
108                this.sealAlg = c.getAlgorithm();
109                this.encryptedContent = c.doFinal(bos.toByteArray());
110            } catch (BadPaddingException e) {
111                // should be never thrown because the cipher
112                // should be initialized for encryption
113                throw new IOException(e.toString());
114            }
115        }
116    
117        /**
118         * Creates a new {@code SealedObject} instance by copying the data from
119         * the specified object.
120         * 
121         * @param so
122         *            the object to copy.
123         * @since Android 1.0
124         */
125        protected SealedObject(SealedObject so) {
126            if (so == null) {
127                throw new NullPointerException(Messages.getString("crypto.14")); //$NON-NLS-1$
128            }
129            this.encryptedContent = so.encryptedContent;
130            this.encodedParams = so.encodedParams;
131            this.sealAlg = so.sealAlg;
132            this.paramsAlg = so.paramsAlg;
133        }
134    
135        /**
136         * Returns the algorithm this object was sealed with.
137         * 
138         * @return the algorithm this object was sealed with.
139         * @since Android 1.0
140         */
141        public final String getAlgorithm() {
142            return sealAlg;
143        }
144    
145        /**
146         * Returns the wrapped object, decrypting it using the specified key.
147         * 
148         * @param key
149         *            the key to decrypt the data with.
150         * @return the encapsulated object.
151         * @throws IOException
152         *             if deserialization fails.
153         * @throws ClassNotFoundException
154         *             if deserialization fails.
155         * @throws NoSuchAlgorithmException
156         *             if the algorithm to decrypt the data is not available.
157         * @throws InvalidKeyException
158         *             if the specified key cannot be used to decrypt the data.
159         * @since Android 1.0
160         */
161        public final Object getObject(Key key)
162                    throws IOException, ClassNotFoundException,
163                           NoSuchAlgorithmException, InvalidKeyException {
164            // BEGIN android-added
165            if (key == null) {
166                throw new InvalidKeyException(
167                        Messages.getString("crypto.05"));
168            }
169            // END android-added
170            try {
171                Cipher cipher = Cipher.getInstance(sealAlg);
172                if ((paramsAlg != null) && (paramsAlg.length() != 0)) {
173                    AlgorithmParameters params =
174                        AlgorithmParameters.getInstance(paramsAlg);
175                    params.init(encodedParams);
176                    cipher.init(Cipher.DECRYPT_MODE, key, params);
177                } else {
178                    cipher.init(Cipher.DECRYPT_MODE, key);
179                }
180                byte[] serialized = cipher.doFinal(encryptedContent);
181                ObjectInputStream ois =
182                        new ObjectInputStream(
183                                new ByteArrayInputStream(serialized));
184                return ois.readObject();
185            } catch (NoSuchPaddingException e)  {
186                // should not be thrown because cipher text was made
187                // with existing padding
188                throw new NoSuchAlgorithmException(e.toString());
189            } catch (InvalidAlgorithmParameterException e) {
190                // should not be thrown because cipher text was made
191                // with correct algorithm parameters
192                throw new NoSuchAlgorithmException(e.toString());
193            } catch (IllegalBlockSizeException e) {
194                // should not be thrown because the cipher text
195                // was correctly made
196                throw new NoSuchAlgorithmException(e.toString());
197            } catch (BadPaddingException e) {
198                // should not be thrown because the cipher text
199                // was correctly made
200                throw new NoSuchAlgorithmException(e.toString());
201            } catch (IllegalStateException  e) {
202                // should never be thrown because cipher is initialized
203                throw new NoSuchAlgorithmException(e.toString());
204            }
205        }
206    
207        /**
208         * Returns the wrapped object, decrypting it using the specified
209         * cipher.
210         * 
211         * @param c
212         *            the cipher to decrypt the data.
213         * @return the encapsulated object.
214         * @throws IOException
215         *             if deserialization fails.
216         * @throws ClassNotFoundException
217         *             if deserialization fails.
218         * @throws IllegalBlockSizeException
219         *             if the specified cipher is a block cipher and the length of
220         *             the serialized data is not a multiple of the ciphers block
221         *             size.
222         * @throws BadPaddingException
223         *             if the padding of the data does not match the padding scheme.
224         * @since Android 1.0
225         */
226        public final Object getObject(Cipher c)
227                    throws IOException, ClassNotFoundException,
228                           IllegalBlockSizeException, BadPaddingException {
229            if (c == null) {
230                throw new NullPointerException(Messages.getString("crypto.13")); //$NON-NLS-1$
231            }
232            byte[] serialized = c.doFinal(encryptedContent);
233            ObjectInputStream ois =
234                    new ObjectInputStream(
235                            new ByteArrayInputStream(serialized));
236            return ois.readObject();
237        }
238    
239        /**
240         * Returns the wrapped object, decrypting it using the specified key. The
241         * specified provider is used to retrieve the cipher algorithm.
242         * 
243         * @param key
244         *            the key to decrypt the data.
245         * @param provider
246         *            the name of the provider that provides the cipher algorithm.
247         * @return the encapsulated object.
248         * @throws IOException
249         *             if deserialization fails.
250         * @throws ClassNotFoundException
251         *             if deserialization fails.
252         * @throws NoSuchAlgorithmException
253         *             if the algorithm used to decrypt the data is not available.
254         * @throws NoSuchProviderException
255         *             if the specified provider is not available.
256         * @throws InvalidKeyException
257         *             if the specified key cannot be used to decrypt the data.
258         * @since Android 1.0
259         */
260        public final Object getObject(Key key, String provider)
261                    throws IOException, ClassNotFoundException,
262                           NoSuchAlgorithmException, NoSuchProviderException,
263                           InvalidKeyException {
264            if ((provider == null) || (provider.length() == 0)) {
265                throw new IllegalArgumentException(
266                        Messages.getString("crypto.15")); //$NON-NLS-1$
267            }
268            try {
269                Cipher cipher = Cipher.getInstance(sealAlg, provider);
270                if ((paramsAlg != null) && (paramsAlg.length() != 0)) {
271                    AlgorithmParameters params =
272                        AlgorithmParameters.getInstance(paramsAlg);
273                    params.init(encodedParams);
274                    cipher.init(Cipher.DECRYPT_MODE, key, params);
275                } else {
276                    cipher.init(Cipher.DECRYPT_MODE, key);
277                }
278                byte[] serialized = cipher.doFinal(encryptedContent);
279                ObjectInputStream ois =
280                        new ObjectInputStream(
281                                new ByteArrayInputStream(serialized));
282                return ois.readObject();
283            } catch (NoSuchPaddingException e)  {
284                // should not be thrown because cipher text was made
285                // with existing padding
286                throw new NoSuchAlgorithmException(e.toString());
287            } catch (InvalidAlgorithmParameterException e) {
288                // should not be thrown because cipher text was made
289                // with correct algorithm parameters
290                throw new NoSuchAlgorithmException(e.toString());
291            } catch (IllegalBlockSizeException e) {
292                // should not be thrown because the cipher text
293                // was correctly made
294                throw new NoSuchAlgorithmException(e.toString());
295            } catch (BadPaddingException e) {
296                // should not be thrown because the cipher text
297                // was correctly made
298                throw new NoSuchAlgorithmException(e.toString());
299            } catch (IllegalStateException  e) {
300                // should never be thrown because cipher is initialized
301                throw new NoSuchAlgorithmException(e.toString());
302            }
303        }
304    }
