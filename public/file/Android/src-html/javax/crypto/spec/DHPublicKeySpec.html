018    package javax.crypto.spec;
019    
020    import java.math.BigInteger;
021    import java.security.spec.KeySpec;
022    
023    /**
024     * The key specification for a Diffie-Hellman public key.
025     * 
026     * @since Android 1.0
027     */
028    public class DHPublicKeySpec implements KeySpec {
029    
030        private final BigInteger y;
031        private final BigInteger p;
032        private final BigInteger g;
033    
034        /**
035         * Creates a new <code>DHPublicKeySpec</code> instance with the specified
036         * <i>public value</i> <code>y</code>, the <i>prime modulus</i>
037         * <code>p</code> and the <i>base generator</i> <code>g</code>.
038         * 
039         * @param y
040         *            the public value.
041         * @param p
042         *            the prime modulus.
043         * @param g
044         *            the base generator.
045         */
046        public DHPublicKeySpec(BigInteger y, BigInteger p, BigInteger g) {
047            this.y = y;
048            this.p = p;
049            this.g = g;
050        }
051    
052        /**
053         * Returns the <i>public value</i> <code>y</code>.
054         * 
055         * @return the public value <code>y</code>.
056         */
057        public BigInteger getY() {
058            return y;
059        }
060    
061        /**
062         * Returns the <i>prime modulus</i> <code>p</code>.
063         * 
064         * @return the prime modulus <code>p</code>.
065         */
066        public BigInteger getP() {
067            return p;
068        }
069    
070        /**
071         * Returns the <i>base generator</i> <code>g</code>;
072         * 
073         * @return the base generator <code>g</code>;
074         */
075        public BigInteger getG() {
076            return g;
077        }
078    }
