022    package javax.crypto.spec;
023    
024    import java.security.spec.AlgorithmParameterSpec;
025    
026    import org.apache.harmony.crypto.internal.nls.Messages;
027    
028    /**
029     * The algorithm parameter specification for an <i>initialization vector</i>.
030     */
031    public class IvParameterSpec implements AlgorithmParameterSpec {
032    
033        private final byte[] iv;
034    
035        /**
036         * Creates a new <code>IvParameterSpec</code> instance with the bytes from
037         * the specified buffer <i>iv</i> used as <i>initialization vector</i>.
038         * 
039         * @param iv
040         *            the buffer used as initialization vector.
041         * @throws NullPointerException
042         *             if the specified buffer is null.
043         */
044        public IvParameterSpec(byte[] iv) {
045            if (iv == null) {
046                throw new NullPointerException(Messages.getString("crypto.38")); //$NON-NLS-1$
047            }
048            this.iv = new byte[iv.length];
049            System.arraycopy(iv, 0, this.iv, 0, iv.length);
050        }
051    
052        /**
053         * Creates a new <code>IvParameterSpec</code> instance with <code>len</code>
054         * bytes from the specified buffer <code>iv</code> starting at
055         * <code>offset</code>.
056         * 
057         * @param iv
058         *            the buffer used as initialization vector.
059         * @param offset
060         *            the offset to start in the buffer.
061         * @param len
062         *            the length of the data.
063         * @throws IllegalArgumentException
064         *             if the specified buffer is null or <code>offset</code> and
065         *             <code>len</code> do not specify a valid chunk in the
066         *             specified buffer.
067         * @throws ArrayIndexOutOfBoundsException
068         *             if <code>offset</code> or <code>len</code> are negative.
069         */
070        public IvParameterSpec(byte[] iv, int offset, int len) {
071            if ((iv == null) || (iv.length - offset < len)) {
072                throw new IllegalArgumentException(
073                        Messages.getString("crypto.39")); //$NON-NLS-1$
074            }
075            if (offset < 0 || len < 0) {
076                throw new ArrayIndexOutOfBoundsException(Messages.getString("crypto.3A")); //$NON-NLS-1$
077            }
078            this.iv = new byte[len];
079            System.arraycopy(iv, offset, this.iv, 0, len);
080        }
081    
082        /**
083         * Returns a copy of the <i>initialization vector</i> data.
084         * 
085         * @return a copy of the initialization vector data.
086         */
087        public byte[] getIV() {
088            byte[] res = new byte[iv.length];
089            System.arraycopy(iv, 0, res, 0, iv.length);
090            return res;
091        }
092    }
