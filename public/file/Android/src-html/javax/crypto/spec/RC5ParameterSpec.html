018    package javax.crypto.spec;
019    
020    import java.security.spec.AlgorithmParameterSpec;
021    import java.util.Arrays;
022    
023    import org.apache.harmony.crypto.internal.nls.Messages;
024    
025    /**
026     * The algorithm parameter specification for the <a
027     * href="http://www.ietf.org/rfc/rfc2040.txt">RC5</a> algorithm.
028     * 
029     * @since Android 1.0
030     */
031    public class RC5ParameterSpec implements AlgorithmParameterSpec {
032    
033        private final int version;
034        private final int rounds;
035        private final int wordSize;
036        private final byte[] iv;
037    
038        /**
039         * Creates a new <code>RC5ParameterSpec</code> instance with the specified
040         * version, round count an word size (in bits).
041         * 
042         * @param version
043         *            the version.
044         * @param rounds
045         *            the round count.
046         * @param wordSize
047         *            the word size (in bits).
048         */
049        public RC5ParameterSpec(int version, int rounds, int wordSize) {
050            this.version = version;
051            this.rounds = rounds;
052            this.wordSize = wordSize;
053            this.iv = null;
054        }
055    
056        /**
057         * Creates a new <code>RC5ParameterSpec</code> instance with the specified
058         * version, round count, word size (in bits) and an <i>initialization
059         * vector</i>.
060         * <p>
061         * The size of the <i>initialization vector</i> must be at least
062         * <code>2 * (wordSize / 8)</code> bytes which are copied to protect them
063         * against modification.
064         * 
065         * @param version
066         *            the version.
067         * @param rounds
068         *            the round count.
069         * @param wordSize
070         *            the word size (in bits).
071         * @param iv
072         *            the initialization vector.
073         * @throws IllegalArgumentException
074         *             if the initialization vector is null or shorter than <code>2
075         *             * (wordSize / 8)</code>.
076         */
077        public RC5ParameterSpec(int version, int rounds, int wordSize, byte[] iv) {
078            if (iv == null) {
079                throw new IllegalArgumentException(Messages.getString("crypto.31")); //$NON-NLS-1$
080            }
081            if (iv.length < 2 * (wordSize / 8)) {
082                throw new IllegalArgumentException(
083                        Messages.getString("crypto.32")); //$NON-NLS-1$
084            }
085            this.version = version;
086            this.rounds = rounds;
087            this.wordSize = wordSize;
088            this.iv = new byte[2*(wordSize/8)];
089            System.arraycopy(iv, 0, this.iv, 0, 2*(wordSize/8));
090        }
091    
092        /**
093         * Creates a new <code>RC5ParameterSpec</code> instance with the specified
094         * version, round count, wordSize (in bits), an <i>initialization vector</i>
095         * and an offset.
096         * <p>
097         * The size of the <i>initialization vector</i> must be at least
098         * <code>offset + (2 * (wordSize / 8))</code> bytes. The bytes starting at
099         * <code>offset</code> are copied to protect them against modification.
100         * 
101         * @param version
102         *            the version.
103         * @param rounds
104         *            the round count.
105         * @param wordSize
106         *            the word size (in bits).
107         * @param iv
108         *            the initialization vector.
109         * @param offset
110         *            the offset in the initialization vector.
111         * @throws IllegalArgumentException
112         *             if the initialization vector is null of shorter than
113         *             <code>offset + (2 * (wordSize / 8))</code>.
114         * @throws ArrayIndexOutOfBoundsException
115         *             if <code>offset</code> is negative.
116         */
117        public RC5ParameterSpec(int version, int rounds,
118                                    int wordSize, byte[] iv, int offset) {
119            if (iv == null) {
120                throw new IllegalArgumentException(Messages.getString("crypto.31")); //$NON-NLS-1$
121            }
122            if (offset < 0) {
123                throw new ArrayIndexOutOfBoundsException(Messages.getString("crypto.33")); //$NON-NLS-1$
124            }
125            if (iv.length - offset < 2 * (wordSize / 8)) {
126                throw new IllegalArgumentException(
127                        Messages.getString("crypto.34")); //$NON-NLS-1$
128            }
129            this.version = version;
130            this.rounds = rounds;
131            this.wordSize = wordSize;
132            this.iv = new byte[offset+2*(wordSize/8)];
133            System.arraycopy(iv, offset, this.iv, 0, 2*(wordSize/8));
134        }
135    
136        /**
137         * Returns the version.
138         * 
139         * @return the version.
140         */
141        public int getVersion() {
142            return version;
143        }
144    
145        /**
146         * Returns the round count.
147         * 
148         * @return the round count.
149         */
150        public int getRounds() {
151            return rounds;
152        }
153    
154        /**
155         * Returns the word size (in bits).
156         * 
157         * @return the word size (in bits).
158         */
159        public int getWordSize() {
160            return wordSize;
161        }
162    
163        /**
164         * Returns a copy of the initialization vector.
165         * 
166         * @return a copy of the initialization vector, or null if none specified.
167         */
168        public byte[] getIV() {
169            if (iv == null) {
170                return null;
171            }
172            byte[] result = new byte[iv.length];
173            System.arraycopy(iv, 0, result, 0, iv.length);
174            return result;
175        }
176    
177        /**
178         * Compares the specified object with this <code>RC5ParameterSpec</code>
179         * instance.
180         * 
181         * @param obj
182         *            the object to compare.
183         * @return true if version, round count, word size and initializaion vector
184         *         of both objects are equal, otherwise false.
185         */
186        @Override
187        public boolean equals(Object obj) {
188            if (obj == this) {
189                return true;
190            }
191            if (!(obj instanceof RC5ParameterSpec)) {
192                return false;
193            }
194            RC5ParameterSpec ps = (RC5ParameterSpec) obj;
195            return (version == ps.version)
196                && (rounds == ps.rounds)
197                && (wordSize == ps.wordSize)
198                && (Arrays.equals(iv, ps.iv));
199        }
200    
201        /**
202         * Returns the hash code of this <code>RC5ParameterSpec</code> instance.
203         * 
204         * @return the hash code.
205         */
206        @Override
207        public int hashCode() {
208            int result = version + rounds + wordSize;
209            if (iv == null) {
210                return result;
211            }
212            for (byte element : iv) {
213                result += element & 0xFF;
214            }
215            return result;
216        }
217    }
