018    package javax.crypto.spec;
019    
020    import java.security.InvalidKeyException;
021    import java.security.spec.KeySpec;
022    
023    import org.apache.harmony.crypto.internal.nls.Messages;

025    /**
026     * The key specification for a DES key.
027     * 
028     * @since Android 1.0
029     */
030    public class DESKeySpec implements KeySpec {
031    
032        /**
033         * The length of a DES key in bytes.
034         */
035        public static final int DES_KEY_LEN = 8;
036    
037        private final byte[] key;
038    
039        // DES weak and semi-weak keys
040        // Got from:
041        // FIP PUB 74
042        // FEDERAL INFORMATION PROCESSING STANDARDS PUBLICATION 1981
043        // GUIDELINES FOR IMPLEMENTING AND USING THE NBS DATA ENCRYPTION STANDARD 
044        // http://www.dice.ucl.ac.be/crypto/standards/fips/fip74/fip74-1.pdf
045        private static final byte[][] SEMIWEAKS = {
046                    {(byte) 0xE0, (byte) 0x01, (byte) 0xE0, (byte) 0x01,
047                     (byte) 0xF1, (byte) 0x01, (byte) 0xF1, (byte) 0x01},
048    
049                    {(byte) 0x01, (byte) 0xE0, (byte) 0x01, (byte) 0xE0,
050                     (byte) 0x01, (byte) 0xF1, (byte) 0x01, (byte) 0xF1},
051    
052                    {(byte) 0xFE, (byte) 0x1F, (byte) 0xFE, (byte) 0x1F,
053                     (byte) 0xFE, (byte) 0x0E, (byte) 0xFE, (byte) 0x0E},
054

056                     (byte) 0x0E, (byte) 0xFE, (byte) 0x0E, (byte) 0xFE},
057    
058                    {(byte) 0xE0, (byte) 0x1F, (byte) 0xE0, (byte) 0x1F,
059                     (byte) 0xF1, (byte) 0x0E, (byte) 0xF1, (byte) 0x0E},

093    
094                    };
095    
096        /**
097         * Creates a new <code>DESKeySpec</code> from the first 8 bytes of the
098         * specified key data.
099         * 
100         * @param key
101         *            the key data.
102         * @throws InvalidKeyException
103         *             if the length of the specified key data is less than 8.
104         */
105        public DESKeySpec(byte[] key) throws InvalidKeyException {
106            this(key, 0);
107        }
108    
109        /**
110         * Creates a new <code>DESKeySpec</code> from the first 8 bytes of the
111         * specified key data starting at <code>offset</code>.
112         * 
113         * @param key
114         *            the key data
115         * @param offset
116         *            the offset to start at.
117         * @throws InvalidKeyException
118         *             if the length of the specified key data starting at offset is
119         *             less than 8.
120         */
121        public DESKeySpec(byte[] key, int offset)
122                    throws InvalidKeyException {
123            if (key == null) {
124                throw new NullPointerException(Messages.getString("crypto.2F")); //$NON-NLS-1$
125            }
126            if (key.length - offset < DES_KEY_LEN) {
127                throw new InvalidKeyException(
128                        Messages.getString("crypto.40")); //$NON-NLS-1$
129            }
130            this.key = new byte[DES_KEY_LEN];
131            System.arraycopy(key, offset, this.key, 0, DES_KEY_LEN);
132        }
133    
134        /**
135         * Returns a copy of the key.
136         * 
137         * @return a copy of the key.
138         */
139        public byte[] getKey() {
140            byte[] result = new byte[DES_KEY_LEN];
141            System.arraycopy(this.key, 0, result, 0, DES_KEY_LEN);
142            return result;
143        }
144    
145        /**
146         * Returns whether the specified key data starting at <code>offset</code> is
147         * <i>parity-adjusted</i>.
148         * 
149         * @param key
150         *            the key data.
151         * @param offset
152         *            the offset to start checking at.
153         * @return {@code true} if the specified key data is parity-adjusted,
154         *            {@code false} otherwise.
155         * @throws InvalidKeyException
156         *             if the length of the key data starting at offset is less than
157         *             8, or the key is null.
158         */
159        public static boolean isParityAdjusted(byte[] key, int offset)
160                throws InvalidKeyException {
161            if (key == null) {
162                throw new InvalidKeyException(Messages.getString("crypto.2F")); //$NON-NLS-1$
163            }
164            if (key.length - offset < DES_KEY_LEN) {
165                throw new InvalidKeyException(
166                        Messages.getString("crypto.40")); //$NON-NLS-1$
167            }
168    
169            int byteKey = 0;
170    
171            for (int i = offset; i < DES_KEY_LEN; i++) {
172                byteKey = key[i];
173    
174                byteKey ^= byteKey >> 1;
175                byteKey ^= byteKey >> 2;
176                byteKey ^= byteKey >> 4;
177    
178                if ((byteKey & 1) == 0) {
179                    return false;
180                }
181            }
182            return true;
183        }
184    
185        /**
186         * Returns whether the specified key data starting at <code>offset</code> is
187         * weak or semi-weak.
188         * 
189         * @param key
190         *            the key data.
191         * @param offset
192         *            the offset to start checking at.
193         * @return {@code true} if the specified key data is weak or semi-weak.
194         * @throws InvalidKeyException
195         *             if the length of the key data starting at offset is less than
196         *             8, or it is null.
197         */
198        public static boolean isWeak(byte[] key, int offset)
199                  throws InvalidKeyException {
200            if (key == null) {
201                throw new InvalidKeyException(Messages.getString("crypto.2F")); //$NON-NLS-1$
202            }
203            if (key.length - offset < DES_KEY_LEN) {
204                throw new InvalidKeyException(
205                        Messages.getString("crypto.40")); //$NON-NLS-1$
206            }
207            I:
208            for (int i=0; i<SEMIWEAKS.length; i++) {
209                for (int j=0; j<DES_KEY_LEN; j++) {
210                    if (SEMIWEAKS[i][j] != key[offset+j]) {
211                        continue I;
212                    }
213                }
214                return true;
215            }
216            return false;
217        }
