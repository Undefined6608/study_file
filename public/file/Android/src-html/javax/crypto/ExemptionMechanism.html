019    
020    import java.security.AlgorithmParameters;
021    import java.security.InvalidAlgorithmParameterException;
022    import java.security.InvalidKeyException;
023    import java.security.Key;
024    import java.security.NoSuchAlgorithmException;
025    import java.security.NoSuchProviderException;
026    import java.security.Provider;
027    import java.security.Security;
028    import java.security.spec.AlgorithmParameterSpec;

031    import org.apache.harmony.crypto.internal.nls.Messages;

034    /**
035     * This class implements the functionality of an exemption mechanism such as
036     * <i>key recovery</i>, <i>key weakening</i>, or <i>key escrow</i>.
037     * 
038     * @since Android 1.0
039     */
040    public class ExemptionMechanism {
041    
042        // Used to access common engine functionality
043        private static final Engine engine = new Engine("ExemptionMechanism");
044    
045        // Store used provider
046        private final Provider provider;
047    
048        // Store used spi implementation
049        private final ExemptionMechanismSpi spiImpl;
050    
051        // Store mechanism name
052        private final String mechanism;
053    
054        // Store state (initialized or not)
055        private boolean isInit;
056    
057        // Store initKey value
058        private Key initKey;
059    
060        // Indicates if blob generated successfully
061        private boolean generated;
062    
063        /**
064         * Creates a {@code ExemptionMechanism} instance.
065         * 
066         * @param exmechSpi
067         *            the implementation delegate.
068         * @param provider
069         *            the associated provider.
070         * @param mechanism
071         *            the name of the mechanism.
072         * @since Android 1.0
073         */
074        protected ExemptionMechanism(ExemptionMechanismSpi exmechSpi,
075                Provider provider, String mechanism) {
076            this.mechanism = mechanism;
077            this.spiImpl = exmechSpi;
078            this.provider = provider;
079            isInit = false;
080        }
081    
082        /**
083         * Returns the name of this {@code ExemptionMechanism}.
084         * 
085         * @return the name of this {@code ExemptionMechanism}.
086         * @since Android 1.0
087         */
088        public final String getName() {
089            return mechanism;
090        }
091    
092        /**
093         * Returns a new {@code ExemptionMechanism} instance that provides the
094         * specified exemption mechanism algorithm.
095         * 
096         * @param algorithm
097         *            the name of the requested exemption mechanism.
098         * @return the new {@code ExemptionMechanism} instance.
099         * @throws NoSuchAlgorithmException
100         *             if the specified algorithm is not available by any provider.
101         * @throws NullPointerException
102         *             if the algorithm parameter is {@code null}.
103         * @since Android 1.0
104         */
105        public static final ExemptionMechanism getInstance(String algorithm)
106                throws NoSuchAlgorithmException {
107            if (algorithm == null) {
108                throw new NullPointerException(Messages.getString("crypto.02")); //$NON-NLS-1$
109            }
110            synchronized (engine) {
111                engine.getInstance(algorithm, null);
112                return new ExemptionMechanism((ExemptionMechanismSpi) engine.spi,
113                        engine.provider, algorithm);
114            }
115        }
116    
117        /**
118         * Returns a new {@code ExemptionMechansm} instance that provides the
119         * specified exemption mechanism algorithm from the specified provider.
120         * 
121         * @param algorithm
122         *            the name of the requested exemption mechanism.
123         * @param provider
124         *            the name of the provider that is providing the algorithm.
125         * @return the new {@code ExemptionMechanism} instance.
126         * @throws NoSuchAlgorithmException
127         *             if the specified algorithm is not provided by the specified
128         *             provider.
129         * @throws NoSuchProviderException
130         *             if the specified provider is not available.
131         * @throws NullPointerException
132         *             if the algorithm parameter is {@code null}.
133         * @throws IllegalArgumentException
134         *             if the provider parameter is {@code null}.
135         * @since Android 1.0
136         */
137        public static final ExemptionMechanism getInstance(String algorithm,
138                String provider) throws NoSuchAlgorithmException,
139                NoSuchProviderException {
140            if (provider == null) {
141                throw new IllegalArgumentException(Messages.getString("crypto.04")); //$NON-NLS-1$
142            }
143            Provider impProvider = Security.getProvider(provider);
144            if (impProvider == null) {
145                throw new NoSuchProviderException(provider);
146            }
147            if (algorithm == null) {
148                throw new NullPointerException(Messages.getString("crypto.02")); //$NON-NLS-1$
149            }
150            return getInstance(algorithm, impProvider);
151        }
152    
153        /**
154         * Returns a new {@code ExemptionMechanism} instance that provides the
155         * specified exemption mechanism algorithm from the specified provider.
156         * 
157         * @param algorithm
158         *            the name of the requested exemption mechanism.
159         * @param provider
160         *            the provider that is providing the algorithm.
161         * @return the new {@code ExemptionMechanism} instance.
162         * @throws NoSuchAlgorithmException
163         *             if the specified algorithm is not provided by the specified
164         *             provider.
165         * @throws NullPointerException
166         *             if the algorithm parameter is {@code null}.
167         * @throws IllegalArgumentException
168         *             if the provider parameter is {@code null}.
169         * @since Android 1.0
170         */
171        public static final ExemptionMechanism getInstance(String algorithm,
172                Provider provider) throws NoSuchAlgorithmException {
173            if (algorithm == null) {
174                throw new NullPointerException(Messages.getString("crypto.02")); //$NON-NLS-1$
175            }
176            if (provider == null) {
177                throw new IllegalArgumentException(Messages.getString("crypto.04")); //$NON-NLS-1$
178            }
179            synchronized (engine) {
180                engine.getInstance(algorithm, provider, null);
181                return new ExemptionMechanism((ExemptionMechanismSpi) engine.spi,
182                        provider, algorithm);
183            }
184        }
185    
186        /**
187         * Returns the provider of this {@code ExemptionMechanism} instance.
188         * 
189         * @return the provider of this {@code ExemptionMechanism} instance.
190         * @since Android 1.0
191         */
192        public final Provider getProvider() {
193            return provider;
194        }
195    
196        /**
197         * Returns whether the result blob for this {@code ExemptionMechanism}
198         * instance has been generated successfully and that the specified key is
199         * the same as the one that was used to initialize and generate.
200         * 
201         * @param key
202         *            the key to verify.
203         * @return whether the result blob for this {@code ExemptionMechanism}
204         *         instance has been generated successfully.
205         * @throws ExemptionMechanismException
206         *             if an error occurs while determining whether the result blob
207         *             has been generated successfully.
208         * @since Android 1.0
209         */
210        public final boolean isCryptoAllowed(Key key)
211                throws ExemptionMechanismException {
212    
213            if (generated
214                    && (initKey.equals(key) || Arrays.equals(initKey.getEncoded(),
215                            key.getEncoded()))) {
216                return true;
217            }
218            return false;
219        }
220    
221        /**
222         * Returns the size in bytes for the output buffer needed to hold the output
223         * of the next {@link #genExemptionBlob} call, given the specified {@code
224         * inputLen} (in bytes).
225         * 
226         * @param inputLen
227         *            the specified input length (in bytes).
228         * @return the size in bytes for the output buffer.
229         * @throws IllegalStateException
230         *             if this {@code ExemptionMechanism} instance is not
231         *             initialized.
232         * @since Android 1.0
233         */
234        public final int getOutputSize(int inputLen) throws IllegalStateException {
235            if (!isInit) {
236                throw new IllegalStateException(Messages.getString("crypto.2D"));
237            }
238            return spiImpl.engineGetOutputSize(inputLen);
239        }
240    
241        /**
242         * Initializes this {@code ExemptionMechanism} instance with the
243         * specified key.
244         * 
245         * @param key
246         *            the key to initialize this instance with.
247         * @throws InvalidKeyException
248         *             if the key cannot be used to initialize this mechanism.
249         * @throws ExemptionMechanismException
250         *             if error(s) occur during initialization.
251         * @since Android 1.0
252         */
253        public final void init(Key key) throws InvalidKeyException,
254                ExemptionMechanismException {
255            generated = false;
256            spiImpl.engineInit(key);
257            initKey = key;
258            isInit = true;
259        }
260    
261        /**
262         * Initializes this {@code ExemptionMechanism} instance with the
263         * specified key and algorithm parameters.
264         * 
265         * @param key
266         *            the key to initialize this instance with.
267         * @param param
268         *            the parameters for this exemption mechanism algorithm.
269         * @throws InvalidKeyException
270         *             if the key cannot be used to initialize this mechanism.
271         * @throws InvalidAlgorithmParameterException
272         *             if the parameters cannot be used to initialize this
273         *             mechanism.
274         * @throws ExemptionMechanismException
275         *             if error(s) occur during initialization.
276         * @since Android 1.0
277         */
278        public final void init(Key key, AlgorithmParameters param)
279                throws InvalidKeyException, InvalidAlgorithmParameterException,
280                ExemptionMechanismException {
281            generated = false;
282            spiImpl.engineInit(key, param);
283            initKey = key;
284            isInit = true;
285        }
286    
287        /**
288         * Initializes this {@code ExemptionMechanism} instance with the
289         * specified key and algorithm parameters.
290         * 
291         * @param key
292         *            the key to initialize this instance with.
293         * @param param
294         *            the parameters for this exemption mechanism algorithm.
295         * @throws InvalidKeyException
296         *             if the key cannot be used to initialize this mechanism.
297         * @throws InvalidAlgorithmParameterException
298         *             the the parameters cannot be used to initialize this
299         *             mechanism.
300         * @throws ExemptionMechanismException
301         *             if error(s) occur during initialization.
302         * @since Android 1.0
303         */
304        public final void init(Key key, AlgorithmParameterSpec param)
305                throws InvalidKeyException, InvalidAlgorithmParameterException,
306                ExemptionMechanismException {
307            generated = false;
308            spiImpl.engineInit(key, param);
309            initKey = key;
310            isInit = true;
311        }
312    
313        /**
314         * Generates the result key blob for this exemption mechanism.
315         * 
316         * @return the result key blob for this exemption mechanism.
317         * @throws IllegalStateException
318         *             if this {@code ExemptionMechanism} instance is not
319         *             initialized.
320         * @throws ExemptionMechanismException
321         *             if error(s) occur during generation.
322         * @since Android 1.0
323         */
324        public final byte[] genExemptionBlob() throws IllegalStateException,
325                ExemptionMechanismException {
326            if (!isInit) {
327                throw new IllegalStateException(Messages.getString("crypto.2D"));
328            }
329            generated = false;
330            byte[] result = spiImpl.engineGenExemptionBlob();
331            generated = true;
332            return result;
333        }
334    
335        /**
336         * Generates the result key blob for this exemption mechanism and stores it
337         * into the {@code output} buffer.
338         * 
339         * @param output
340         *            the output buffer for the result key blob.
341         * @return the number of bytes written to the {@code output} buffer.
342         * @throws IllegalStateException
343         *             if this {@code ExemptionMechanism} instance is not
344         *             initialized.
345         * @throws ShortBufferException
346         *             if the provided buffer is too small for the result key blob.
347         * @throws ExemptionMechanismException
348         *             if error(s) occur during generation.
349         * @since Android 1.0
350         */
351        public final int genExemptionBlob(byte[] output)
352                throws IllegalStateException, ShortBufferException,
353                ExemptionMechanismException {
354            return genExemptionBlob(output, 0);
355        }
356    
357        /**
358         * Generates the result key blob for this exemption mechanism and stores it
359         * into the {@code output} buffer at offset {@code outputOffset}.
360         * 
361         * @param output
362         *            the output buffer for the result key blob.
363         * @param outputOffset
364         *            the offset in the output buffer to start.
365         * @return the number of bytes written to the {@code output} buffer.
366         * @throws IllegalStateException
367         *             if this {@code ExemptionMechanism} instance is not
368         *             initialized.
369         * @throws ShortBufferException
370         *             if the provided buffer is too small for the result key blob.
371         * @throws ExemptionMechanismException
372         *             if error(s) occur during generation.
373         * @since Android 1.0
374         */
375        public final int genExemptionBlob(byte[] output, int outputOffset)
376                throws IllegalStateException, ShortBufferException,
377                ExemptionMechanismException {
378            if (!isInit) {
379                throw new IllegalStateException(Messages.getString("crypto.2D"));
380            }
381            generated = false;
382            int len = spiImpl.engineGenExemptionBlob(output, outputOffset);
383            generated = true;
384            return len;
385        }
386    
387        /**
388         * Frees the references to the key used to initialize this instance.
389         * 
390         * @since Android 1.0
391         */
392        @Override
393        protected void finalize() {
394            initKey = null;
395        }
396    }
