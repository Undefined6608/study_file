018    package javax.net.ssl;
019    
020    import org.apache.harmony.security.fortress.Engine;
021    
022    import java.security.AccessController;
023    import java.security.InvalidAlgorithmParameterException;
024    import java.security.KeyStore;
025    import java.security.KeyStoreException;
026    import java.security.NoSuchAlgorithmException;
027    import java.security.NoSuchProviderException;
028    import java.security.Provider;
029    import java.security.Security;
030    import java.security.UnrecoverableKeyException;
031    
032    
033    /**
034     * The public API for {@code KeyManagerFactory} implementations.
035     * 
036     * @since Android 1.0
037     */
038    public class KeyManagerFactory {
039        // Store KeyManagerFactory service name
040        private static final String SERVICE = "KeyManagerFactory";
041    
042        // Used to access common engine functionality
043        private static Engine engine = new Engine(SERVICE);
044    
045        // Store default property name
046        private static final String PROPERTY_NAME = "ssl.KeyManagerFactory.algorithm";
047    
048        // Store used provider
049        private final Provider provider;
050    
051        // Store used KeyManagerFactorySpi implementation
052        private final KeyManagerFactorySpi spiImpl;
053    
054        // Store used algorithm
055        private final String algorithm;
056    
057        /**
058         * Creates a new {@code KeyManagerFactory}.
059         * 
060         * @param factorySpi
061         *            the implementation delegate.
062         * @param provider
063         *            the provider.
064         * @param algorithm
065         *            the key management algorithm name.
066         * @since Android 1.0
067         */
068        protected KeyManagerFactory(KeyManagerFactorySpi factorySpi,
069                Provider provider, String algorithm) {
070            this.provider = provider;
071            this.algorithm = algorithm;
072            this.spiImpl = factorySpi;
073        }
074    
075        /**
076         * Returns the name of the key management algorithm.
077         * 
078         * @return the name of the key management algorithm.
079         * @since Android 1.0
080         */
081        public final String getAlgorithm() {
082            return algorithm;
083        }
084    
085        /**
086         * Creates a new {@code KeyManagerFactory} instance for the specified key
087         * management algorithm.
088         * 
089         * @param algorithm
090         *            the name of the requested key management algorithm.
091         * @return a key manager factory for the requested algorithm.
092         * @throws NoSuchAlgorithmException
093         *             if no installed provider can provide the requested algorithm.
094         * @throws NullPointerException
095         *             if {@code algorithm} is {@code null} (instead of
096         *             NoSuchAlgorithmException as in 1.4 release)
097         * @since Android 1.0
098         */
099        public static final KeyManagerFactory getInstance(String algorithm)
100                throws NoSuchAlgorithmException {
101            if (algorithm == null) {
102                throw new NullPointerException("algorith is null");
103            }
104            synchronized (engine) {
105                engine.getInstance(algorithm, null);
106                return new KeyManagerFactory((KeyManagerFactorySpi) engine.spi,
107                        engine.provider, algorithm);
108            }
109        }
110    
111        /**
112         * Creates a new {@code KeyManagerFactory} instance for the specified key
113         * management algorithm from the specified provider.
114         * 
115         * @param algorithm
116         *            the name of the requested key management algorithm name.
117         * @param provider
118         *            the name of the provider that provides the requested
119         *            algorithm.
120         * @return a key manager factory for the requested algorithm.
121         * @throws NoSuchAlgorithmException
122         *             if the specified provider cannot provide the requested
123         *             algorithm.
124         * @throws NoSuchProviderException
125         *             if the specified provider does not exist.
126         * @throws NullPointerException
127         *             if {@code algorithm} is {@code null} (instead of
128         *             NoSuchAlgorithmException as in 1.4 release)
129         * @since Android 1.0
130         */
131        public static final KeyManagerFactory getInstance(String algorithm,
132                String provider) throws NoSuchAlgorithmException,
133                NoSuchProviderException {
134            if ((provider == null) || (provider.length() == 0)) {
135                throw new IllegalArgumentException("Provider is null or empty");
136            }
137            Provider impProvider = Security.getProvider(provider);
138            if (impProvider == null) {
139                throw new NoSuchProviderException(provider);
140            }
141            return getInstance(algorithm, impProvider);
142        }
143    
144        /**
145         * Creates a new {@code KeyManagerFactory} instance for the specified key
146         * management algorithm from the specified provider.
147         * 
148         * @param algorithm
149         *            the name of the requested key management algorithm name.
150         * @param provider
151         *            the provider that provides the requested algorithm.
152         * @return a key manager factory for the requested algorithm.
153         * @throws NoSuchAlgorithmException
154         *             if the specified provider cannot provide the requested
155         *             algorithm.
156         * @throws NullPointerException
157         *             if {@code algorithm} is {@code null} (instead of
158         *             NoSuchAlgorithmException as in 1.4 release)
159         * @since Android 1.0
160         */
161        public static final KeyManagerFactory getInstance(String algorithm,
162                Provider provider) throws NoSuchAlgorithmException {
163            if (provider == null) {
164                throw new IllegalArgumentException("Provider is null");
165            }
166            if (algorithm == null) {
167                throw new NullPointerException("algorith is null");
168            }
169            synchronized (engine) {
170                engine.getInstance(algorithm, provider, null);
171                return new KeyManagerFactory((KeyManagerFactorySpi) engine.spi,
172                        provider, algorithm);
173            }
174        }
175    
176        /**
177         * Returns the provider for this {@code KeyManagerFactory} instance.
178         * 
179         * @return the provider for this {@code KeyManagerFactory} instance.
180         * @since Android 1.0
181         */
182        public final Provider getProvider() {
183            return provider;
184        }
185    
186        /**
187         * Initializes this instance with the specified key store and password.
188         * 
189         * @param ks
190         *            the key store or {@code null} to use the default key store.
191         * @param password
192         *            the password for the specified key store or {@code null} if no
193         *            key store is provided.
194         * @throws KeyStoreException
195         *             if initializing this key manager factory fails.
196         * @throws NoSuchAlgorithmException
197         *             if a required algorithm is not available.
198         * @throws UnrecoverableKeyException
199         *             if a key cannot be recovered.
200         * @since Android 1.0
201         */
202        public final void init(KeyStore ks, char[] password)
203                throws KeyStoreException, NoSuchAlgorithmException,
204                UnrecoverableKeyException {
205            spiImpl.engineInit(ks, password);
206        }
207    
208        /**
209         * Initializes this instance with the specified factory parameters.
210         * 
211         * @param spec
212         *            the factory parameters.
213         * @throws InvalidAlgorithmParameterException
214         *             if an error occurs.
215         * @since Android 1.0
216         */
217        public final void init(ManagerFactoryParameters spec)
218                throws InvalidAlgorithmParameterException {
219            spiImpl.engineInit(spec);
220        }
221    
222        /**
223         * Returns a list of key managers, one instance for each type of key in the
224         * key store.
225         * 
226         * @return a list of key managers.
227         * @since Android 1.0
228         */
229        public final KeyManager[] getKeyManagers() {
230            return spiImpl.engineGetKeyManagers();
231        }
232    
233        /**
234         * Returns the default key manager factory algorithm name.
235         * <p>
236         * The default algorithm name is specified by the security property:
237         *  {@code 'ssl.KeyManagerFactory.algorithm'}.
238         * </p>
239         * 
240         * @return the default algorithm name.
241         * @since Android 1.0
242         */
243        public static final String getDefaultAlgorithm() {
244            return AccessController
245                    .doPrivileged(new java.security.PrivilegedAction<String>() {
246                        public String run() {
247                            return Security.getProperty(PROPERTY_NAME);
248                        }
249                    });
250        }
251    }
