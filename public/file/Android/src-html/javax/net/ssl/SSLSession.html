019    package javax.net.ssl;
020    
021    import java.security.Principal;
022    import java.security.cert.Certificate;
023    import javax.security.cert.X509Certificate;
024    
025    /**
026     * The interface representing an SSL session.
027     * 
028     * @since Android 1.0
029     */
030    public interface SSLSession {
031    
032        /**
033         * Returns the maximum size that an application buffer can be for this
034         * session.
035         * 
036         * @return the maximum application buffer size.
037         * @since Android 1.0
038         */
039        public int getApplicationBufferSize();
040    
041        /**
042         * Returns the name of the cipher suite used in this session.
043         * 
044         * @return the name of the cipher suite used in this session.
045         * @since Android 1.0
046         */
047        public String getCipherSuite();
048    
049        /**
050         * Returns the time this session was created, in milliseconds since midnight
051         * January 1st 1970 UTC.
052         * 
053         * @return the time the session was created.
054         * @since Android 1.0
055         */
056        public long getCreationTime();
057    
058        /**
059         * Returns this sessions identifier.
060         * 
061         * @return this sessions identifier.
062         * @since Android 1.0
063         */
064        public byte[] getId();
065    
066        /**
067         * Returns the time this session was last accessed, in milliseconds since
068         * midnight January 1st 1970 UTC.
069         * 
070         * @return the time this session was last accessed.
071         * @since Android 1.0
072         */
073        public long getLastAccessedTime();
074    
075        /**
076         * Returns the list of certificates that were used to identify the local
077         * side to the peer during the handshake.
078         * 
079         * @return the list of certificates, ordered from local certificate to
080         *         CA's certificates.
081         * @since Android 1.0
082         */
083        public Certificate[] getLocalCertificates();
084    
085        /**
086         * Returns the principal used to identify the local side to the peer during
087         * the handshake.
088         * 
089         * @return the principal used to identify the local side.
090         * @since Android 1.0
091         */
092        public Principal getLocalPrincipal();
093    
094        /**
095         * Returns the maximum size that a network buffer can be for this session.
096         * 
097         * @return the maximum network buffer size.
098         * @since Android 1.0
099         */
100        public int getPacketBufferSize();
101    
102        /**
103         * Returns the list of certificates the peer used to identify itself during
104         * the handshake.
105         * <p>
106         * Note: this method exists for compatility reasons, use
107         * {@link #getPeerCertificates()} instead.
108         * </p>
109         * 
110         * @return the list of certificates, ordered from the identity certificate to
111         *         the CA's certificates
112         * @throws SSLPeerUnverifiedException
113         *             if the identity of the peer is not verified.
114         * @since Android 1.0
115         */
116        public X509Certificate[] getPeerCertificateChain()
117                throws SSLPeerUnverifiedException;
118    
119        /**
120         * Returns the list of certificates the peer used to identify itself during
121         * the handshake.
122         * 
123         * @return the list of certificates, ordered from the identity certificate to
124         *         the CA's certificates.
125         * @throws SSLPeerUnverifiedException
126         *             if the identity of the peer is not verified.
127         * @since Android 1.0
128         */
129        public Certificate[] getPeerCertificates()
130                throws SSLPeerUnverifiedException;
131    
132        /**
133         * Returns the host name of the peer of this session. The host name is not
134         * authenticated.
135         * 
136         * @return the host name of the peer of this session, or {@code null} if no
137         *         host name is available.
138         * @since Android 1.0
139         */
140        public String getPeerHost();
141    
142        /**
143         * Returns the port number of the peer of this session. The port number is
144         * not authenticated.
145         * 
146         * @return the port number of the peer, of {@code -1} is no port number is
147         *         available.
148         * @since Android 1.0
149         */
150        public int getPeerPort();
151    
152        /**
153         * Returns the principal identifying the peer during the handshake.
154         * 
155         * @return the principal identifying the peer.
156         * @throws SSLPeerUnverifiedException
157         *             if the identity of the peer has not been verified.
158         * @since Android 1.0
159         */
160        public Principal getPeerPrincipal() throws SSLPeerUnverifiedException;
161    
162        /**
163         * Returns the protocol name that is used for all connections in this
164         * session.
165         * 
166         * @return the protocol name that is used for all connections in this
167         *         session.
168         * @since Android 1.0
169         */
170        public String getProtocol();
171    
172        /**
173         * Returns the context of this session. If a context is available and a
174         * security manager is installed, the
175         * {@code SSLPermission("getSSLSessionContext"} is checked with the security
176         * manager.
177         * 
178         * @return the context of this session or {@code null} if no context is
179         *         available.
180         * @since Android 1.0
181         */
182        public SSLSessionContext getSessionContext();
183    
184        /**
185         * Returns the object bound to the specified name in this session's
186         * application layer data.
187         * 
188         * @param name
189         *            the name of the bound value.
190         * @return the value bound to the specified name, or {@code null} if the
191         *         specified name does not exist or is not accessible in the current
192         *         access control context.
193         * @throws IllegalArgumentException
194         *             if {@code name} is {@code null}.
195         * @since Android 1.0
196         */
197        public Object getValue(String name);
198    
199        /**
200         * Returns the list of the object names bound to this session's application
201         * layer data..
202         * <p>
203         * Depending on the current access control context, the list of object names
204         * may be different.
205         * </p>
206         * 
207         * @return the list of the object names bound to this session's application
208         *         layer data.
209         * @since Android 1.0
210         */
211        public String[] getValueNames();
212    
213        /**
214         * Invalidates this session.
215         * <p>
216         * No new connections can be created, but any existing connection remains
217         * valid until it is closed.
218         * </p>
219         * 
220         * @since Android 1.0
221         */
222        public void invalidate();
223    
224        /**
225         * Returns whether this session is valid.
226         * 
227         * @return {@code true} if this session is valid, otherwise {@code false}.
228         * @since Android 1.0
229         */
230        public boolean isValid();
231    
232        /**
233         * Binds the specified object under the specified name in this session's
234         * application layer data.
235         * <p>
236         * For bindings (new or existing) implementing the
237         * {@code SSLSessionBindingListener} interface the object will be notified.
238         * </p>
239         * 
240         * @param name
241         *            the name to bind the object to.
242         * @param value
243         *            the object to bind.
244         * @throws IllegalArgumentException
245         *             if either {@code name} or {@code value} is {@code null}.
246         * @since Android 1.0
247         */
248        public void putValue(String name, Object value);
249    
250        /**
251         * Removes the binding for the specified name in this session's application
252         * layer data. If the existing binding implements the
253         * {@code SSLSessionBindingListener} interface the object will be notified.
254         * 
255         * @param name
256         *            the binding to remove.
257         * @throws IllegalArgumentException
258         *             if {@code name} is {@code null}.
259         * @since Android 1.0
260         */
261        public void removeValue(String name);
262    }
