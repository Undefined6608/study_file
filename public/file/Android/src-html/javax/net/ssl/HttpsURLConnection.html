018    package javax.net.ssl;
019    
020    import java.net.HttpURLConnection;
021    import java.net.URL;
022    import java.security.Principal;
023    import java.security.cert.Certificate;
024    import java.security.cert.X509Certificate;
025    
026    /**
027     * This abstract subclass of {@code HttpURLConnection} defines methods for
028     * managing HTTPS connections according to the description given by RFC 2818.
029     * 
030     * @since Android 1.0
031     */
032    public abstract class HttpsURLConnection extends HttpURLConnection {
033    
034        private static HostnameVerifier defaultHostnameVerifier = new DefaultHostnameVerifier();
035    
036        private static SSLSocketFactory defaultSSLSocketFactory = (SSLSocketFactory) SSLSocketFactory
037                .getDefault();
038    
039        /**
040         * The host name verifier used by this connection. It is initialized from
041         * the default hostname verifier
042         * {@link #setDefaultHostnameVerifier(HostnameVerifier)} or
043         * {@link #getDefaultHostnameVerifier()}.
044         * 
045         * @since Android 1.0
046         */
047        protected HostnameVerifier hostnameVerifier;
048    
049        private static SSLSocketFactory socketFactory;
050    
051        /**
052         * Creates a new {@code HttpsURLConnection} with the specified {@code URL}.
053         * 
054         * @param url
055         *            the {@code URL} to connect to.
056         * @since Android 1.0
057         */
058        protected HttpsURLConnection(URL url) {
059            super(url);
060            hostnameVerifier = defaultHostnameVerifier;
061            socketFactory = defaultSSLSocketFactory;
062        }
063    
064        /**
065         * Returns the name of the cipher suite negotiated during the SSL handshake.
066         * 
067         * @return the name of the cipher suite negotiated during the SSL handshake.
068         * @throws IllegalStateException
069         *             if no connection has been established yet.
070         * @since Android 1.0
071         */
072        public abstract String getCipherSuite();
073    
074        /**
075         * Returns the list of local certificates used during the handshake. These
076         * certificates were sent to the peer.
077         * 
078         * @return Returns the list of certificates used during the handshake with
079         *         the local identity certificate followed by CAs, or {@code null}
080         *         if no certificates were used during the handshake.
081         * @throws IllegalStateException
082         *             if no connection has been established yet.
083         * @since Android 1.0
084         */    
085        public abstract Certificate[] getLocalCertificates();
086    
087        /**
088         * Return the list of certificates identifying the peer during the
089         * handshake.
090         * 
091         * @return the list of certificates identifying the peer with the peer's
092         *         identity certificate followed by CAs.
093         * @throws SSLPeerUnverifiedException
094         *             if the identity of the peer has not been verified..
095         * @throws IllegalStateException
096         *             if no connection has been established yet.
097         * @since Android 1.0
098         */    
099        public abstract Certificate[] getServerCertificates()
100                throws SSLPeerUnverifiedException;
101    
102        /**
103         * Returns the {@code Principal} identifying the peer.
104         * 
105         * @return the {@code Principal} identifying the peer.
106         * @throws SSLPeerUnverifiedException
107         *             if the identity of the peer has not been verified.
108         * @throws IllegalStateException
109         *             if no connection has been established yet.
110         * @since Android 1.0
111         */
112        public Principal getPeerPrincipal() throws SSLPeerUnverifiedException {
113            Certificate[] certs = getServerCertificates();
114            if (certs == null || certs.length == 0 || 
115                    (!(certs[0] instanceof X509Certificate))) {
116                throw new SSLPeerUnverifiedException(
117                        "No server's end-entity certificate");
118            }
119            return ((X509Certificate) certs[0]).getSubjectX500Principal();
120        }
121    
122        /**
123         * Returns the {@code Principal} used to identify the local host during the handshake.
124         * 
125         * @return the {@code Principal} used to identify the local host during the handshake, or
126         *         {@code null} if none was used.
127         * @throws IllegalStateException
128         *             if no connection has been established yet.
129         * @since Android 1.0
130         */    
131        public Principal getLocalPrincipal() {
132            Certificate[] certs = getLocalCertificates();
133            if (certs == null || certs.length == 0
134                    || (!(certs[0] instanceof X509Certificate))) {
135                return null;
136            }
137            return ((X509Certificate) certs[0]).getSubjectX500Principal();
138        }
139    
140        /**
141         * Sets the default hostname verifier to be used by new instances.
142         * 
143         * @param v
144         *            the new default hostname verifier
145         * @throws IllegalArgumentException
146         *             if the specified verifier is {@code null}.
147         * @since Android 1.0
148         */
149        public static void setDefaultHostnameVerifier(HostnameVerifier v) {
150            if (v == null) {
151                throw new IllegalArgumentException("HostnameVerifier is null");
152            }
153            defaultHostnameVerifier = v;
154        }
155    
156        /**
157         * Returns the default hostname verifier.
158         * 
159         * @return the default hostname verifier.
160         * @since Android 1.0
161         */
162        public static HostnameVerifier getDefaultHostnameVerifier() {
163            return defaultHostnameVerifier;
164        }
165    
166        /**
167         * Sets the hostname verifier for this instance.
168         * 
169         * @param v
170         *            the hostname verifier for this instance.
171         * @throws IllegalArgumentException
172         *             if the specified verifier is {@code null}.
173         * @since Android 1.0
174         */
175        public void setHostnameVerifier(HostnameVerifier v) {
176            if (v == null) {
177                throw new IllegalArgumentException("HostnameVerifier is null");
178            }
179            hostnameVerifier = v;
180        }
181    
182        /**
183         * Returns the hostname verifier used by this instance.
184         * 
185         * @return the hostname verifier used by this instance.
186         * @since Android 1.0
187         */
188        public HostnameVerifier getHostnameVerifier() {
189            return hostnameVerifier;
190        }
191    
192        /**
193         * Sets the default SSL socket factory to be used by new instances.
194         * 
195         * @param sf
196         *            the new default SSL socket factory.
197         * @throws IllegalArgumentException
198         *             if the specified socket factory is {@code null}.
199         * @since Android 1.0
200         */
201        public static void setDefaultSSLSocketFactory(SSLSocketFactory sf) {
202            if (sf == null) {
203                throw new IllegalArgumentException("SSLSocketFactory is null");
204            }
205            defaultSSLSocketFactory = sf;
206        }
207    
208        /**
209         * Returns the default SSL socket factory for new instances.
210         * 
211         * @return the default SSL socket factory for new instances.
212         * @since Android 1.0
213         */
214        public static SSLSocketFactory getDefaultSSLSocketFactory() {
215            return defaultSSLSocketFactory;
216        }
217    
218        /**
219         * Sets the SSL socket factory for this instance.
220         * 
221         * @param sf
222         *            the SSL socket factory to be used by this instance.
223         * @throws IllegalArgumentException
224         *             if the specified socket factory is {@code null}.
225         * @since Android 1.0
226         */
227        public void setSSLSocketFactory(SSLSocketFactory sf) {
228            if (sf == null) {
229                throw new IllegalArgumentException("SSLSocketFactory is null");
230            }
231            socketFactory = sf;
232        }
233    
234        /**
235         * Returns the SSL socket factory used by this instance.
236         * 
237         * @return the SSL socket factory used by this instance.
238         * @since Android 1.0
239         */
240        public SSLSocketFactory getSSLSocketFactory() {
241            return socketFactory;
242        }
243    
244    }
