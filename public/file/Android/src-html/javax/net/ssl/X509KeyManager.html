018    package javax.net.ssl;
019    
020    import java.net.Socket;
021    import java.security.Principal;
022    import java.security.PrivateKey;
023    import java.security.cert.X509Certificate;
024    
025    /**
026     * A Key Manager for X509 certificate-based key pairs.
027     * 
028     * @since Android 1.0
029     */
030    public interface X509KeyManager extends KeyManager {
031    
032        /**
033         * Chooses an alias for the client side of an SSL connection to authenticate
034         * it with the specified public key type and certificate issuers.
035         * 
036         * @param keyType
037         *            the list of public key algorithm names.
038         * @param issuers
039         *            the list of certificate issuers, or {@code null} if any issuer
040         *            will do.
041         * @param socket
042         *            the socket for the connection, or {@code null} if
043         *            the alias selected does not depend on a specific socket.
044         * @return the alias name of a matching key or {@code null} if there are no
045         *         matches.
046         * @since Android 1.0
047         */
048        public String chooseClientAlias(String[] keyType, Principal[] issuers,
049                Socket socket);
050    
051        /**
052         * Chooses an alias for the server side of an SSL connection to authenticate
053         * it with the specified public key type and certificate issuers.
054         * 
055         * @param keyType
056         *            the list of public key algorithm type names.
057         * @param issuers
058         *            the list of certificate issuers, or {@code null} if any issuer
059         *            will do.
060         * @param socket
061         *            the socket for the connection, or {@code null} if
062         *            the alias selected does not depend on a specific socket.
063         * @return the alias name of a matching key or {@code null} if there are no
064         *         matches.
065         * @since Android 1.0
066         */
067        public String chooseServerAlias(String keyType, Principal[] issuers,
068                Socket socket);
069    
070        /**
071         * Returns the certificate chain for the specified alias.
072         * 
073         * @param alias
074         *            the alias to get the certificate chain for.
075         * @return the certificate chain for the specified alias, or {@code null} if
076         *         the alias cannot be found.
077         * @since Android 1.0
078         */
079        public X509Certificate[] getCertificateChain(String alias);
080    
081        /**
082         * Returns the client aliases for the specified public key type and list of
083         * certificate issuers.
084         * 
085         * @param keyType
086         *            the public key algorithm type name.
087         * @param issuers
088         *            the list of certificate issuers, or {@code null} if any issuer
089         *            will do.
090         * @return the client aliases for the specified public key type, or
091         *         {@code null} if there are no matching aliases.
092         * @since Android 1.0
093         */
094        public String[] getClientAliases(String keyType, Principal[] issuers);
095    
096        /**
097         * Returns the server aliases for the specified public key type and list of
098         * certificate issuers.
099         * 
100         * @param keyType
101         *            the public key algorithm type name.
102         * @param issuers
103         *            the list of certificate issuers, or {@code null} if any issuer
104         *            will do.
105         * @return the client aliases for the specified public key type, or
106         *         {@code null} if there are no matching aliases.
107         * @since Android 1.0
108         */
109        public String[] getServerAliases(String keyType, Principal[] issuers);
110    
111        /**
112         * Returns the private key for the specified alias.
113         * 
114         * @param alias
115         *            the alias to get the private key for.
116         * @return the private key for the specified alias, or {@code null} if the
117         *         alias cannot be found.
118         * @since Android 1.0
119         */
120        public PrivateKey getPrivateKey(String alias);
121    }
