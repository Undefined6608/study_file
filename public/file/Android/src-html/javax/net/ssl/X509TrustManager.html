018    package javax.net.ssl;
019    
020    import java.security.cert.CertificateException;
021    import java.security.cert.X509Certificate;
022    
023    /**
024     * The trust manager for X509 certificates to be used to perform authentication
025     * for secure sockets. 
026     * 
027     * @since Android 1.0
028     */
029    public interface X509TrustManager extends TrustManager {
030    
031        /**
032         * Checks whether the specified certificate chain (partial or complete) can
033         * be validated and is trusted for client authentication for the specified
034         * authentication type.
035         * 
036         * @param chain
037         *            the certificate chain to validate.
038         * @param authType
039         *            the authentication type used.
040         * @throws CertificateException
041         *             if the certificate chain can't be validated or isn't trusted.
042         * @throws IllegalArgumentException
043         *             if the specified certificate chain is empty or {@code null},
044         *             or if the specified authentication type is {@code null} or an
045         *             empty string.
046         * @since Android 1.0
047         */
048        public void checkClientTrusted(X509Certificate[] chain, String authType)
049                throws CertificateException;
050    
051    
052        /**
053         * Checks whether the specified certificate chain (partial or complete) can
054         * be validated and is trusted for server authentication for the specified
055         * key exchange algorithm.
056         * 
057         * @param chain
058         *            the certificate chain to validate.
059         * @param authType
060         *            the key exchange algorithm name.
061         * @throws CertificateException
062         *             if the certificate chain can't be validated or isn't trusted.
063         * @throws IllegalArgumentException
064         *             if the specified certificate chain is empty or {@code null},
065         *             or if the specified authentication type is {@code null} or an
066         *             empty string.
067         * @since Android 1.0
068         */
069        public void checkServerTrusted(X509Certificate[] chain, String authType)
070                throws CertificateException;
071    
072        /**
073         * Returns the list of certificate issuer authorities which are trusted for
074         * authentication of peers.
075         * 
076         * @return the list of certificate issuer authorities which are trusted for
077         *         authentication of peers.
078         * @since Android 1.0
079         */
080        public X509Certificate[] getAcceptedIssuers();
081    }
