018    package javax.net.ssl;
019    
020    /**
021     * The result object describing the state of the {@code SSLEngine} produced
022     * by the {@code wrap()} and {@code unwrap()} operations.
023     * 
024     * @since Android 1.0
025     */
026    public class SSLEngineResult {
027        
028        // Store Status object
029        private final SSLEngineResult.Status status;
030    
031        // Store HandshakeStatus object
032        private final SSLEngineResult.HandshakeStatus handshakeStatus;
033    
034        // Store bytesConsumed
035        private final int bytesConsumed;
036    
037        // Store bytesProduced
038        private final int bytesProduced;
039    
040        /**
041         * Creates a new {@code SSLEngineResult} instance with the specified state
042         * values.
043         * 
044         * @param status
045         *            the return value of the {@code SSLEngine} operation.
046         * @param handshakeStatus
047         *            the status of the current handshake
048         * @param bytesConsumed
049         *            the number of bytes retrieved from the source buffer(s).
050         * @param bytesProduced
051         *            the number of bytes transferred to the destination buffer(s).
052         * @throws IllegalArgumentException
053         *             if {@code status} or {@code handshakeStatus} is {@code null},
054         *             or if {@code bytesConsumed} or {@code bytesProduces} are
055         *             negative.
056         * @since Android 1.0
057         */
058        public SSLEngineResult(SSLEngineResult.Status status,
059                SSLEngineResult.HandshakeStatus handshakeStatus, int bytesConsumed,
060                int bytesProduced) {
061            if (status == null) {
062                throw new IllegalArgumentException("status is null");
063            }
064            if (handshakeStatus == null) {
065                throw new IllegalArgumentException("handshakeStatus is null");
066            }
067            if (bytesConsumed < 0) {
068                throw new IllegalArgumentException("bytesConsumed is negative");
069            }
070            if (bytesProduced < 0) {
071                throw new IllegalArgumentException("bytesProduced is negative");
072            }
073            this.status = status;
074            this.handshakeStatus = handshakeStatus;
075            this.bytesConsumed = bytesConsumed;
076            this.bytesProduced = bytesProduced;
077        }
078    
079        /**
080         * Returns the return value of the {@code SSLEngine} operation.
081         * 
082         * @return the return value of the {@code SSLEngine} operation.
083         * @since Android 1.0
084         */
085        public final Status getStatus() {
086            return status;
087        }
088    
089        /**
090         * Returns the status of the current handshake.
091         * 
092         * @return the status of the current handshake.
093         * @since Android 1.0
094         */
095        public final HandshakeStatus getHandshakeStatus() {
096            return handshakeStatus;
097        }
098    
099        /**
100         * Returns the number of bytes retrieved from the source buffer(s).
101         * 
102         * @return the number of bytes retrieved from the source buffer(s).
103         * @since Android 1.0
104         */
105        public final int bytesConsumed() {
106            return bytesConsumed;
107        }
108    
109        /**
110         * Returns the number of bytes transferred to the destination buffer(s).
111         * 
112         * @return the number of bytes transferred to the destination buffer(s).
113         * @since Android 1.0
114         */
115        public final int bytesProduced() {
116            return bytesProduced;
117        }
118    
119        /**
120         * Returns a string representation of this instance.
121         * 
122         * @return a string representation of this instance.
123         * @since Android 1.0
124         */
125        public String toString() {
126            StringBuffer sb = new StringBuffer("SSLEngineReport: Status = ");
127            sb.append(status.toString());
128            sb.append("  HandshakeStatus = ");
129            sb.append(handshakeStatus.toString());
130            sb.append("\n                 bytesConsumed = ");
131            sb.append(Integer.toString(bytesConsumed));
132            sb.append(" bytesProduced = ");
133            sb.append(Integer.toString(bytesProduced));
134            return sb.toString();
135        }
136    
137        /**
138         * The {@code enum} describing the state of the current handshake.
139         * 
140         * @since Android 1.0
141         */
142        public enum HandshakeStatus {
143            /**
144             * No handshake in progress.
145             * 
146             * @since Android 1.0
147             */
148            NOT_HANDSHAKING,
149            /**
150             * The handshake is finished.
151             * 
152             * @since Android 1.0
153             */
154            FINISHED,
155            /**
156             * The results of one (or more) delegated tasks are needed to continue
157             * the handshake.
158             * 
159             * @since Android 1.0
160             */
161            NEED_TASK,
162            /**
163             * The engine must send data to the remote side to continue the
164             * handshake.
165             * 
166             * @since Android 1.0
167             */
168            NEED_WRAP,
169            /**
170             * The engine needs to receive data from the remote side to continue the
171             * handshake.
172             * 
173             * @since Android 1.0
174             */
175            NEED_UNWRAP
176        }
177    
178        /**
179         * The {@code enum} describing the result of the {@code SSLEngine}
180         * operation.
181         * 
182         * @since Android 1.0
183         */
184        public static enum Status {
185            /**
186             * The size of the destination buffer is too small to hold the result of
187             * the current operation.
188             * 
189             * @since Android 1.0
190             */
191            BUFFER_OVERFLOW,
192            /**
193             * There were not enough bytes available in the source buffer to
194             * complete the current operation.
195             * 
196             * @since Android 1.0
197             */
198            BUFFER_UNDERFLOW,
199            /**
200             * The operation closed this side of the communication or was already
201             * closed.
202             * 
203             * @since Android 1.0
204             */
205            CLOSED,
206            /**
207             * The operation completed successfully.
208             * 
209             * @since Android 1.0
210             */
211            OK
212        }
213    }
