018     package javax.net.ssl;
019    
020    import java.nio.ByteBuffer;
021    import java.nio.ReadOnlyBufferException;
022    
023    /**
024     * The abstract implementation of secure communications using SSL, TLS, or other
025     * protocols. It includes the setup, handshake, and encrypt/decrypt functionality
026     * needed to create a secure connection.
027     * 
028     * @since Android 1.0
029     */
030    public abstract class SSLEngine {
031        // Store host value
032        private final String host;
033    
034        // Store port value
035        private final int port;
036    
037        /**
038         * Creates a new {@code SSLEngine} instance.
039         * 
040         * @since Android 1.0
041         */
042        protected SSLEngine() {
043            host = null;
044            port = -1;
045        }
046    
047        /**
048         * Creates a new {@code SSLEngine} instance with the specified host and
049         * port.
050         * 
051         * @param host
052         *            the name of the host.
053         * @param port
054         *            the port of the host.
055         * @since Android 1.0
056         */
057        protected SSLEngine(String host, int port) {
058            this.host = host;
059            this.port = port;
060        }
061    
062        /**
063         * Initiates a handshake on this engine.
064         * <p>
065         * Calling this method is not needed for the initial handshake: it will be
066         * called by {@code wrap} or {@code unwrap} if the initial handshake has not
067         * been started yet.
068         * </p>
069         * 
070         * @throws SSLException
071         *             if starting the handshake fails.
072         * @throws IllegalStateException
073         *             if the engine does not have all the needed settings (e.g.
074         *             client/server mode not set).
075         * @since Android 1.0
076         */
077        public abstract void beginHandshake() throws SSLException;
078    
079        /**
080         * Notifies this engine instance that no more inbound network data will be
081         * sent to this engine.
082         * 
083         * @throws SSLException
084         *             if this engine did not receive a needed protocol specific
085         *             close notification message from the peer.
086         * @since Android 1.0
087         */
088        public abstract void closeInbound() throws SSLException;
089    
090        /**
091         * Notifies this engine instance that no more outbound application data will
092         * be sent to this engine.
093         * 
094         * @since Android 1.0
095         */
096        public abstract void closeOutbound();
097    
098        /**
099         * Returns a delegate task for this engine instance. Some engine operations
100         * may require the results of blocking or long running operations, and the
101         * {@code SSLEngineResult} instances returned by this engine may indicate
102         * that a delegated task result is needed. In this case the
103         * {@link Runnable#run() run} method of the returned {@code Runnable}
104         * delegated task must be called.
105         * 
106         * @return a delegate task, or {@code null} if none are available.
107         * @since Android 1.0
108         */
109        public abstract Runnable getDelegatedTask();
110    
111        /**
112         * Returns the SSL cipher suite names that are enabled in this engine
113         * instance.
114         * 
115         * @return the SSL cipher suite names that are enabled in this engine
116         *         instance.
117         * @since Android 1.0
118         */
119        public abstract String[] getEnabledCipherSuites();
120    
121        /**
122         * Returns the protocol version names that are enabled in this engine
123         * instance.
124         * 
125         * @return the protocol version names that are enabled in this engine
126         *         instance.
127         * @since Android 1.0
128         */
129        public abstract String[] getEnabledProtocols();
130    
131        /**
132         * Returns whether new SSL sessions may be established by this engine.
133         * 
134         * @return {@code true} if new session may be established, {@code false} if
135         *         existing sessions must be reused.
136         * @since Android 1.0
137         */
138        public abstract boolean getEnableSessionCreation();
139    
140        /**
141         * Returns the status of the handshake of this engine instance.
142         * 
143         * @return the status of the handshake of this engine instance.
144         * @since Android 1.0
145         */
146        public abstract SSLEngineResult.HandshakeStatus getHandshakeStatus();
147    
148        /**
149         * Returns whether this engine instance will require client authentication.
150         * 
151         * @return {@code true} if this engine will require client authentication,
152         *         {@code false} if no client authentication is needed.
153         * @since Android 1.0
154         */
155        public abstract boolean getNeedClientAuth();
156    
157        /**
158         * Returns the name of the peer host.
159         * 
160         * @return the name of the peer host, or {@code null} if none is available.
161         * @since Android 1.0
162         */
163        public String getPeerHost() {
164            return host;
165        }
166    
167        /**
168         * Returns the port number of the peer host.
169         * 
170         * @return the port number of the peer host, or {@code -1} is none is
171         *         available.
172         * @since Android 1.0
173         */
174        public int getPeerPort() {
175            return port;
176        }
177    
178        /**
179         * Returns the SSL session for this engine instance.
180         * 
181         * @return the SSL session for this engine instance.
182         * @since Android 1.0
183         */
184        public abstract SSLSession getSession();
185    
186        /**
187         * Returns the SSL cipher suite names that are supported by this engine.
188         * These cipher suites can be enabled using
189         * {@link #setEnabledCipherSuites(String[])}.
190         * 
191         * @return the SSL cipher suite names that are supported by this engine.
192         * @since Android 1.0
193         */
194        public abstract String[] getSupportedCipherSuites();
195    
196        /**
197         * Returns the protocol names that are supported by this engine. These
198         * protocols can be enables using {@link #setEnabledProtocols(String[])}.
199         * 
200         * @return the protocol names that are supported by this engine.
201         * @since Android 1.0
202         */
203        public abstract String[] getSupportedProtocols();
204    
205        /**
206         * Returns whether this engine is set to act in client mode when
207         * handshaking.
208         * 
209         * @return {@code true} if the engine is set to do handshaking in client
210         *         mode.
211         * @since Android 1.0
212         */
213        public abstract boolean getUseClientMode();
214    
215        /**
216         * Returns whether this engine will request client authentication.
217         * 
218         * @return {@code true} if client authentication will be requested,
219         *         {@code false} otherwise.
220         * @since Android 1.0
221         */
222        public abstract boolean getWantClientAuth();
223    
224        /**
225         * Returns whether no more inbound data will be accepted by this engine.
226         * 
227         * @return {@code true} if no more inbound data will be accepted by this
228         *         engine, {@code false} otherwise.
229         * @since Android 1.0
230         */
231        public abstract boolean isInboundDone();
232    
233        /**
234         * Returns whether no more outbound data will be produced by this engine.
235         * 
236         * @return {@code true} if no more outbound data will be producted by this
237         *         engine, {@code otherwise} false.
238         * @since Android 1.0
239         */
240        public abstract boolean isOutboundDone();
241    
242        /**
243         * Sets the SSL cipher suite names that should be enabled in this engine
244         * instance. Only cipher suites listed by {@code getSupportedCipherSuites()}
245         * are allowed.
246         * 
247         * @param suites
248         *            the SSL cipher suite names to be enabled.
249         * @throws IllegalArgumentException
250         *             if one of the specified cipher suites is not supported, or if
251         *             {@code suites} is {@code null}.
252         * @since Android 1.0
253         */
254        public abstract void setEnabledCipherSuites(String[] suites);
255    
256        /**
257         * Sets the protocol version names that should be enabled in this engine
258         * instance. Only protocols listed by {@code getSupportedProtocols()} are
259         * allowed.
260         * 
261         * @param protocols
262         *            the protocol version names to be enabled.
263         * @throws IllegalArgumentException
264         *             if one of the protocol version names is not supported, or if
265         *             {@code protocols} is {@code null}.
266         * @since Android 1.0
267         */
268        public abstract void setEnabledProtocols(String[] protocols);
269    
270        /**
271         * Sets whether new SSL sessions may be established by this engine instance.
272         * 
273         * @param flag
274         *            {@code true} if new SSL sessions may be established,
275         *            {@code false} if existing SSL sessions must be reused.
276         * @since Android 1.0
277         */
278        public abstract void setEnableSessionCreation(boolean flag);
279    
280        /**
281         * Sets whether this engine must require client authentication. The client
282         * authentication is one of:
283         * <ul>
284         * <li>authentication required</li>
285         * <li>authentication requested</li>
286         * <li>no authentication needed</li>
287         * </ul>
288         * This method overrides the setting of {@link #setWantClientAuth(boolean)}.
289         * 
290         * @param need
291         *            {@code true} if client authentication is required,
292         *            {@code false} if no authentication is needed.
293         * @since Android 1.0
294         */
295        public abstract void setNeedClientAuth(boolean need);
296    
297        /**
298         * Sets whether this engine should act in client (or server) mode when
299         * handshaking.
300         * 
301         * @param mode
302         *            {@code true} if this engine should act in client mode,
303         *            {@code false} if not.
304         * @throws IllegalArgumentException
305         *             if this method is called after starting the initial
306         *             handshake.
307         * @since Android 1.0
308         */
309        public abstract void setUseClientMode(boolean mode);
310    
311        /**
312         * Sets whether this engine should request client authentication. The client
313         * authentication is one of the following:
314         * <ul>
315         * <li>authentication required</li>
316         * <li>authentication requested</li>
317         * <li>no authentication needed</li>
318         * </ul>
319         * This method overrides the setting of {@link #setNeedClientAuth(boolean)}.
320         * 
321         * @param want
322         *            {@code true} if client authentication should be requested,
323         *            {@code false} if no authentication is needed.
324         * @since Android 1.0
325         */
326        public abstract void setWantClientAuth(boolean want);
327    
328        /**
329         * Decodes the incoming network data buffer into application data buffers.
330         * If a handshake has not been started yet, it will automatically be
331         * started.
332         * 
333         * @param src
334         *            the buffer with incoming network data
335         * @param dsts
336         *            the array of destination buffers for incoming application
337         *            data.
338         * @param offset
339         *            the offset in the array of destination buffers to which data
340         *            is to be transferred.
341         * @param length
342         *            the maximum number of destination buffers to be used. 
343         * @return the result object of this operation.
344         * @throws SSLException
345         *             if a problem occurred while processing the data.
346         * @throws IndexOutOfBoundsException
347         *             if {@code length} is greater than
348         *             {@code dsts.length - offset}.
349         * @throws ReadOnlyBufferException
350         *             if one of the destination buffers is read-only.
351         * @throws IllegalArgumentException
352         *             if {@code src}, {@code dsts}, or one of the entries in
353         *             {@code dsts} is {@code null}.
354         * @throws IllegalStateException
355         *             if the engine does not have all the needed settings (e.g.
356         *             client/server mode not set).
357         * @since Android 1.0
358         */
359        public abstract SSLEngineResult unwrap(ByteBuffer src, ByteBuffer[] dsts,
360                int offset, int length) throws SSLException;
361    
362        /**
363         * Encodes the outgoing application data buffers into the network data
364         * buffer. If a handshake has not been started yet, it will automatically be
365         * started.
366         * 
367         * @param srcs
368         *            the array of source buffers of outgoing application data.
369         * @param offset
370         *            the offset in the array of source buffers from which data is
371         *            to be retrieved.
372         * @param length
373         *            the maximum number of source buffers to be used.
374         * @param dst
375         *            the destination buffer for network data.
376         * @return the result object of this operation.
377         * @throws SSLException
378         *             if a problem occurred while processing the data.
379         * @throws IndexOutOfBoundsException
380         *             if {@code length} is greater than
381         *             {@code srcs.length - offset}.
382         * @throws ReadOnlyBufferException
383         *             if the destination buffer is readonly.
384         * @throws IllegalArgumentException
385         *             if {@code srcs}, {@code dst}, or one the entries in
386         *             {@code srcs} is {@code null}.
387         * @throws IllegalStateException
388         *             if the engine does not have all the needed settings (e.g.
389         *             client/server mode not set).
390         * @since Android 1.0
391         */
392        public abstract SSLEngineResult wrap(ByteBuffer[] srcs, int offset,
393                int length, ByteBuffer dst) throws SSLException;
394    
395        /**
396         * Decodes the incoming network data buffer into the application data
397         * buffer. If a handshake has not been started yet, it will automatically be
398         * started.
399         * 
400         * @param src
401         *            the buffer with incoming network data
402         * @param dst
403         *            the destination buffer for incoming application data.
404         * @return the result object of this operation.
405         * @throws SSLException
406         *             if a problem occurred while processing the data.
407         * @throws ReadOnlyBufferException
408         *             if one of the destination buffers is read-only.
409         * @throws IllegalArgumentException
410         *             if {@code src} or {@code dst} is {@code null}.
411         * @throws IllegalStateException
412         *             if the engine does not have all the needed settings (e.g.
413         *             client/server mode not set).
414         * @since Android 1.0
415         */
416        public SSLEngineResult unwrap(ByteBuffer src, ByteBuffer dst)
417                throws SSLException {
418    //        if (src == null) {
419    //            throw new IllegalArgumentException("Byte buffer src is null");
420    //        }
421    //        if (dst == null) {
422    //            throw new IllegalArgumentException("Byte buffer dst is null");
423    //        }
424    //        if (dst.isReadOnly()) {
425    //            throw new ReadOnlyBufferException();
426    //        }
427            return unwrap(src, new ByteBuffer[] { dst }, 0, 1);
428        }
429    
430        /**
431         * Decodes the incoming network data buffer into the application data
432         * buffers. If a handshake has not been started yet, it will automatically
433         * be started.

437         * @param dsts
438         *            the array of destination buffers for incoming application
439         *            data.
440         * @return the result object of this operation.
441         * @throws SSLException
442         *             if a problem occurred while processing the data.
443         * @throws ReadOnlyBufferException
444         *             if one of the destination buffers is read-only.
445         * @throws IllegalArgumentException
446         *             if {@code src} or {@code dsts} is {@code null}.
447         * @throws IllegalStateException
448         *             if the engine does not have all the needed settings (e.g.

453                throws SSLException {

456    //        }
457            if (dsts == null) {
458                throw new IllegalArgumentException("Byte buffer array dsts is null");
459            }
460    //        for (int i = 0; i < dsts.length; i++) {

471    
472        /**
473         * Encodes the outgoing application data buffers into the network data
474         * buffer. If a handshake has not been started yet, it will automatically be
475         * started.
476         * 
477         * @param srcs
478         *            the array of source buffers of outgoing application data.
479         * @param dst
480         *            the destination buffer for network data.
481         * @return the result object of this operation.
482         * @throws SSLException
483         *             if a problem occurred while processing the data.
484         * @throws ReadOnlyBufferException
485         *             if the destination buffer is readonly.
486         * @throws IllegalArgumentException
487         *             if {@code srcs} or {@code dst} is {@code null}.
488         * @throws IllegalStateException
489         *             if the engine does not have all the needed settings (e.g.
490         *             client/server mode not set).
491         * @since Android 1.0
492         */
493        public SSLEngineResult wrap(ByteBuffer[] srcs, ByteBuffer dst)
494                throws SSLException {
495            if (srcs == null) {
496                throw new IllegalArgumentException("Byte buffer array srcs is null");
497            }
498    //        for (int i = 0; i < srcs.length; i++) {
499    //            if (srcs[i] == null) {
500    //                throw new IllegalArgumentException("Byte buffer srcs[" + i
501    //                        + "]  is null");
502    //            }
503    //        }
504    //        if (dst == null) {
505    //            throw new IllegalArgumentException("Byte buffer array dst is null");
506    //        }
507    //        if (dst.isReadOnly()) {
508    //            throw new ReadOnlyBufferException();

510            return wrap(srcs, 0, srcs.length, dst);
511        }
512

514         * Encodes the outgoing application data buffer into the network data

519         *            the source buffers of outgoing application data.
520         * @param dst

524         *             if a problem occurred while processing the data.
