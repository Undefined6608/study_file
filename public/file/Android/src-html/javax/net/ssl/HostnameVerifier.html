018    package javax.net.ssl;
019    
020    /**
021     * The interface to be used to provide hostname verification functionality.
022     * <p>
023     * This is an extended verification option that implementers can provide. It is to be used
024     * during a handshake if the URL's hostname does not match the peer's
025     * identification hostname.
026     * </p>
027     * 
028     * @since Android 1.0
029     */
030    public interface HostnameVerifier {
031    
032        /**
033         * Verifies that the specified hostname is allowed within the specified SSL
034         * session.
035         * 
036         * @param hostname
037         *            the hostname.
038         * @param session
039         *            the SSL session of the connection.
040         * @return {@code true} if the specified hostname is allowed, otherwise
041         *         {@code false}.
042         * @since Android 1.0
043         */
044        public boolean verify(String hostname, SSLSession session);
045    }
