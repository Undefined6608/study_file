023    package javax.net;
024    
025    import java.io.IOException;
026    import java.net.InetAddress;
027    import java.net.ServerSocket;
028    import java.net.SocketException;
029    
030    /**
031     * This abstract class defines methods to create server sockets. It can be
032     * subclassed to create specific server socket types.
033     * 
034     * @since Android 1.0
035     */
036    public abstract class ServerSocketFactory {
037        static ServerSocketFactory defaultFactory;
038        
039        /**
040         * Creates a new {@code ServerSocketFactory} instance.
041         * 
042         * @since Android 1.0
043         */
044        protected ServerSocketFactory() {
045        }
046    
047        /**
048         * Gets the default server socket factory of the system which can be used to
049         * create new server sockets without creating a subclass of this factory.
050         * 
051         * @return the system default server socket factory.
052         * @since Android 1.0
053         */
054        public static synchronized ServerSocketFactory getDefault() {
055            if (defaultFactory == null) {
056                    defaultFactory = new DefaultServerSocketFactory();
057            }
058            return defaultFactory;
059        }
060    
061        /**
062         * Creates a new server socket which is not bound to any local address. This
063         * method has to be overridden by a subclass otherwise a {@code
064         * SocketException} is thrown.
065         * 
066         * @return the created unbound server socket.
067         * @throws IOException
068         *             if an error occurs while creating a new server socket.
069         * @since Android 1.0
070         */
071        public ServerSocket createServerSocket() throws IOException {
072            // follow RI's behavior 
073            throw new SocketException("Unbound server sockets not implemented");
074        }
075    
076        /**
077         * Creates a new server socket which is bound to the given port.
078         * 
079         * @param port
080         *            the port on which the created socket has to listen.
081         * @return the created bound server socket.
082         * @throws IOException
083         *             if an error occurs while creating a new server socket.
084         * @since Android 1.0
085         */
086        public abstract ServerSocket createServerSocket(int port)
087                throws IOException;
088    
089        /**
090         * Creates a new server socket which is bound to the given port and
091         * configures its maximum of queued connections.
092         * 
093         * @param port
094         *            the port on which the created socket has to listen.
095         * @param backlog
096         *            the maximum of queued connections.
097         * @return the created bound server socket.
098         * @throws IOException
099         *             if an error occurs while creating a new server socket.
100         * @since Android 1.0
101         */
102        public abstract ServerSocket createServerSocket(int port, int backlog)
103                throws IOException;
104    
105        /**
106         * Creates a new server socket which is bound to the given address on the
107         * specified port and configures its maximum of queued connections.
108         * 
109         * @param port
110         *            the port on which the created socket has to listen.
111         * @param backlog
112         *            the maximum of queued connections.
113         * @param iAddress
114         *            the address of the network interface which is used by the
115         *            created socket.
116         * @return the created bound server socket.
117         * @throws IOException
118         *             if an error occurs while creating a new server socket.
119         * @since Android 1.0
120         */
121        public abstract ServerSocket createServerSocket(int port, int backlog,
122                InetAddress iAddress) throws IOException;
123    
124    }
