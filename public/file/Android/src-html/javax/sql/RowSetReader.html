018    package javax.sql;
019    
020    import java.sql.SQLException;
021    
022    /**
023     * An interface which provides functionality for a disconnected {@code RowSet}
024     * to get data from a database into its rows. The {@code RowSet} calls the
025     * {@code RowSetReader} interface when the {@code RowSet}'s execute method is
026     * invoked - a {@code RowSetReader} must first be registered with the {@code
027     * RowSet} for this to work.
028     * 
029     * @see RowSet
030     * @since Android 1.0
031     */
032    public interface RowSetReader {
033    
034        /**
035         * Reads new data into the {@code RowSet}. The calling {@code RowSet} object
036         * must itself implement the {@code RowSetInternal} interface and the
037         * {@code RowSetReader} must be registered as a reader on the 
038         * {@code RowSet}.
039         * <p>
040         * This method adds rows into the calling {@code RowSet}. The reader may
041         * invoke any of the {@code RowSet}'s methods except for the {@code execute}
042         * method (calling {@code execute} will cause an {@code SQLException} to be
043         * thrown). However, when the reader calls the {@code RowSet}'s methods, no
044         * events are sent to listeners - any listeners are informed by the calling
045         * {@code RowSet}'s {@code execute} method once the reader returns from the
046         * {@code readData} method.
047         * </p>
048         * 
049         * @param theCaller
050         *            must be the calling {@code RowSet} object, which must have
051         *            implemented the {@code RowSetInternal} interface.
052         * @throws SQLException
053         *             if a problem occurs accessing the database or if the reader
054         *             calls the {@link RowSet#execute()} method.
055         * @see RowSetInternal
056         * @since Android 1.0
057         */
058        public void readData(RowSetInternal theCaller) throws SQLException;
059    }
