018    package javax.sql;
019    
020    import java.sql.SQLException;
021    
022    /**
023     * An interface which provides functionality for a disconnected {@code RowSet}
024     * to put data updates back to the data source from which the {@code RowSet} was
025     * originally populated. An object implementing this interface is called a
026     * writer.
027     * <p>
028     * The writer must establish a connection to the {@code RowSet}'s database
029     * before writing the data. The {@code RowSet} calling this interface must
030     * implement the {@code RowSetInternal} interface.
031     * </p>
032     * <p>
033     * The writer may encounter a situation where the updated data needs to be
034     * written back to the database, but has already been updated there in the mean
035     * time. How a conflict of this kind is handled is determined by the
036     * implementation of this writer.
037     * </p>
038     * 
039     * @see RowSetInternal
040     * @since Android 1.0
041     */
042    public interface RowSetWriter {
043    
044        /**
045         * Writes changes made in the {@code RowSet}, which is associated with this
046         * {@code RowSetWriter}, back to the database.
047         * 
048         * @param theRowSet
049         *            a row set that fulfills the following criteria:
050         *            <ul>
051         *            <li>it must implement the {@code RowSetInternal} interface,</li>
052         *            <li>have this {@code RowSetWriter} registered with it,</li>
053         *            <li>must call this method internally.</li>
054         *            </ul>
055         * @return {@code true} if the modified data was written, {@code false}
056         *         otherwise (which typically implies some form of conflict).
057         * @throws SQLException
058         *             if a problem occurs accessing the database.
059         * @since Android 1.0
060         */
061        public boolean writeData(RowSetInternal theRowSet) throws SQLException;
062    }
