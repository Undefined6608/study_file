018    package javax.sql;
019    
020    import java.util.EventListener;
021    
022    /**
023     * An interface used to receive events generated by a {@link PooledConnection}.
024     * <p>
025     * This interface would typically be implemented by a component which manages a
026     * connection pool (a connection pool manager). A connection triggers an event
027     * to a {@code ConnectionEventListener} either when the application closes a
028     * connection it has been using or when a significant error occurs while the
029     * connection is being used.
030     * </p>
031     * <p>
032     * The connection pool manager can return closed connections to the pool for
033     * later reuse. Connections experiencing an error should be discarded.
034     * </p>
035     * 
036     * @since Android 1.0
037     */
038    public interface ConnectionEventListener extends EventListener {
039    
040        /**
041         * Notifies the {@code ConnectionEventListener} that an application has
042         * called the {@code close} method on a pooled connection.
043         * 
044         * @param theEvent
045         *            a {@code ConnectionEvent} containing details about the source
046         *            of the event.
047         * @since Android 1.0
048         */
049        public void connectionClosed(ConnectionEvent theEvent);
050    
051        /**
052         * Notifies the {@code ConnectionEventListener} that an error has occurred
053         * on a {@code PooledConnection}. This notification is triggered <i>before</i> the
054         * {@code SQLException}, which is available through the event argument, is
055         * thrown.
056         * 
057         * @param theEvent
058         *            a {@code ConnectionEvent} containing details about the source
059         *            of the event and the {@code SQLException} that has occurred.
060         * @since Android 1.0
061         */
062        public void connectionErrorOccurred(ConnectionEvent theEvent);
063    }
