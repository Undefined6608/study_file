018    package javax.sql;
019    
020    import java.sql.SQLException;
021    import java.sql.Connection;
022    
023    /**
024     * An interface which provides facilities for handling connections to a database
025     * which are pooled.
026     * <p>
027     * Typically, a {@code PooledConnection} is recycled when it is no longer
028     * required by an application, rather than being closed and discarded. The
029     * reason for treating connections in this way is that it can be an expensive
030     * process both to establish a connection to a database and to destroy the
031     * connection. Reusing connections through a pool is a way of improving system
032     * performance and reducing overhead.
033     * </p>
034     * <p>
035     * It is not intended that an application uses the {@code PooledConnection}
036     * interface directly. The {@code PooledConnection} interface is intended for
037     * use by a component called a connection pool manager, typically part of the
038     * infrastructure that supports use of the database by applications.
039     * </p>
040     * <p>
041     * Applications obtain connections to the database by calling the
042     * {@link DataSource#getConnection} method. Behind the scenes, the connection
043     * pool manager will get a {@code PooledConnection} object from its connection
044     * pool and passes back a connection object that wraps or references the {@code
045     * PooledConnection} object. A new {@code PooledConnection} object will only be
046     * created if the pool is empty.
047     * </p>
048     * <p>
049     * When the application is finished using a {@code PooledConnection}, the
050     * application calls the {@link Connection#close} method. The connection pool
051     * manager is notified via a {@link ConnectionEvent} from the connection that
052     * this has happened (the pool manager registers itself with the connection
053     * before the connection is given to the application). The pool manager removes
054     * the underlying {@code PooledConnection} object from the connection and
055     * returns it to the pool for reuse - the {@code PooledConnection} is thus
056     * recycled rather than being destroyed.
057     * </p>
058     * <p>
059     * The connection to the database represented by the {@code PooledConnection} is
060     * kept open until the {@code PooledConnection} object itself is deactivated by
061     * the connection pool manager, which calls {@code PooledConnection.close()}.
062     * This is typically done if there are too many inactive connections in the
063     * pool, if the {@code PooledConnection} encounters a problem that makes it
064     * unusable or if the whole system is being shut down.
065     * </p>
066     * 
067     * @since Android 1.0
068     */
069    public interface PooledConnection {
070    
071        /**
072         * Registers the supplied {@code ConnectionEventListener} with this {@code
073         * PooledConnection}. Once registered, the {@code ConnectionEventListener}
074         * will receive {@link ConnectionEvent} events when they occur in the
075         * {@code PooledConnection}.
076         * 
077         * @param theListener
078         *            an object which implements the {@code ConnectionEventListener}
079         *            interface.
080         * @since Android 1.0
081         */
082        public void addConnectionEventListener(ConnectionEventListener theListener);
083    
084        /**
085         * Closes the connection to the database held by this {@code
086         * PooledConnection}. This method should not be called directly by
087         * application code - it is intended only for the connection pool manager
088         * component.
089         * 
090         * @throws SQLException
091         *             if there is a problem accessing the database.
092         * @since Android 1.0
093         */
094        public void close() throws SQLException;
095    
096        /**
097         * Creates a connection to the database. This method is typically called by
098         * the connection pool manager when an application invokes the method
099         * {@code DataSource.getConnection()} and there are no {@code
100         * PooledConnection} objects available in the connection pool.
101         * 
102         * @return a {@code Connection} object.
103         * @throws SQLException
104         *             if there is a problem accessing the database.
105         * @since Android 1.0
106         */
107        public Connection getConnection() throws SQLException;
108    
109        /**
110         * Unregisters the supplied {@code ConnectionEventListener} from this {@code
111         * PooledConnection}. Once unregistered, the {@code ConnectionEventListener}
112         * will no longer receive events occurring in the {@code PooledConnection}.
113         * 
114         * @param theListener
115         *            an object which implements the {@code ConnectionEventListener}
116         *            interface. This object should have previously been registered
117         *            with the {@code PooledConnection} using the {@code
118         *            addConnectionEventListener} method.
119         * @since Android 1.0
120         */
121        public void removeConnectionEventListener(
122                ConnectionEventListener theListener);
123    }
