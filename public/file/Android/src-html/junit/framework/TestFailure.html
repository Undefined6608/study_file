001    package junit.framework;
002    
003    import java.io.PrintWriter;
004    import java.io.StringWriter;
005    
006    
007    /**
008     * A <code>TestFailure</code> collects a failed test together with
009     * the caught exception.
010     * @see TestResult
011     */
012    public class TestFailure extends Object {
013        protected Test fFailedTest;
014        protected Throwable fThrownException;
015        
016    
017        /**
018         * Constructs a TestFailure with the given test and exception.
019         */
020        public TestFailure(Test failedTest, Throwable thrownException) {
021            fFailedTest= failedTest;
022            fThrownException= thrownException;
023        }
024        /**
025         * Gets the failed test.
026         */
027        public Test failedTest() {
028            return fFailedTest;
029        }
030        /**
031         * Gets the thrown exception.
032         */
033        public Throwable thrownException() {
034            return fThrownException;
035        }
036        /**
037         * Returns a short description of the failure.
038         */
039        public String toString() {
040            StringBuffer buffer= new StringBuffer();
041            buffer.append(fFailedTest+": "+fThrownException.getMessage());
042            return buffer.toString();
043        }
044        public String trace() {
045            StringWriter stringWriter= new StringWriter();
046            PrintWriter writer= new PrintWriter(stringWriter);
047            thrownException().printStackTrace(writer);
048            StringBuffer buffer= stringWriter.getBuffer();
049            return buffer.toString();
050        }
051        public String exceptionMessage() {
052            return thrownException().getMessage();
053        }
054        public boolean isFailure() {
055            return thrownException() instanceof AssertionFailedError;
056        }
057    }
