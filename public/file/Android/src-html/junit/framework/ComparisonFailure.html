001    package junit.framework;
002    
003    /**
004     * Thrown when an assert equals for Strings failed.
005     * 
006     * Inspired by a patch from Alex Chaffee mailto:alex@purpletech.com
007     */
008    public class ComparisonFailure extends AssertionFailedError {
009        private String fExpected;
010        private String fActual;
011    
012        /**
013         * Constructs a comparison failure.
014         * @param message the identifying message or null
015         * @param expected the expected string value
016         * @param actual the actual string value
017         */
018        public ComparisonFailure (String message, String expected, String actual) {
019            super (message);
020            fExpected= expected;
021            fActual= actual;
022        }
023        
024        /**
025         * Returns "..." in place of common prefix and "..." in
026         * place of common suffix between expected and actual.
027         * 
028         * @see java.lang.Throwable#getMessage()
029         */
030        public String getMessage() {
031            if (fExpected == null || fActual == null)
032                return Assert.format(super.getMessage(), fExpected, fActual);
033                
034            int end= Math.min(fExpected.length(), fActual.length());
035            
036            int i= 0;
037            for(; i < end; i++) {
038                if (fExpected.charAt(i) != fActual.charAt(i))
039                    break;
040            }
041            int j= fExpected.length()-1;
042            int k= fActual.length()-1;
043            for (; k >= i && j >= i; k--,j--) {
044                if (fExpected.charAt(j) != fActual.charAt(k))
045                    break;
046            }
047            String actual, expected;
048            
049            // equal strings
050            if (j < i && k < i) {
051                expected= fExpected;
052                actual= fActual;
053            } else {
054                expected= fExpected.substring(i, j+1);
055                actual= fActual.substring(i, k+1);
056                if (i <= end && i > 0) {
057                    expected= "..."+expected;
058                    actual= "..."+actual;
059                }
060                
061                if (j < fExpected.length()-1)
062                    expected= expected+"...";
063                if (k < fActual.length()-1)
064                    actual= actual+"...";
065            }    
066            return Assert.format(super.getMessage(), expected, actual);
067        }
068    }
