001    package junit.framework;
002    
003    import java.lang.reflect.*;
004    
005    /**
006     * A test case defines the fixture to run multiple tests. To define a test case<br>
007     * 1) implement a subclass of TestCase<br>
008     * 2) define instance variables that store the state of the fixture<br>
009     * 3) initialize the fixture state by overriding <code>setUp</code><br>
010     * 4) clean-up after a test by overriding <code>tearDown</code>.<br>
011     * Each test runs in its own fixture so there
012     * can be no side effects among test runs.
013     * Here is an example:
014     * <pre>
015     * public class MathTest extends TestCase {
016     *     protected double fValue1;
017     *     protected double fValue2;
018     *
019     *    protected void setUp() {
020     *         fValue1= 2.0;
021     *         fValue2= 3.0;
022     *     }
023     * }
024     * </pre>
025     *
026     * For each test implement a method which interacts
027     * with the fixture. Verify the expected results with assertions specified
028     * by calling <code>assertTrue</code> with a boolean.
029     * <pre>
030     *    public void testAdd() {
031     *        double result= fValue1 + fValue2;
032     *        assertTrue(result == 5.0);
033     *    }
034     * </pre>
035     * Once the methods are defined you can run them. The framework supports
036     * both a static type safe and more dynamic way to run a test.
037     * In the static way you override the runTest method and define the method to
038     * be invoked. A convenient way to do so is with an anonymous inner class.
039     * <pre>
040     * TestCase test= new MathTest("add") {
041     *        public void runTest() {
042     *            testAdd();
043     *        }
044     * };
045     * test.run();
046     * </pre>
047     * The dynamic way uses reflection to implement <code>runTest</code>. It dynamically finds
048     * and invokes a method.
049     * In this case the name of the test case has to correspond to the test method
050     * to be run.
051     * <pre>
052     * TestCase= new MathTest("testAdd");
053     * test.run();
054     * </pre>
055     * The tests to be run can be collected into a TestSuite. JUnit provides
056     * different <i>test runners</i> which can run a test suite and collect the results.
057     * A test runner either expects a static method <code>suite</code> as the entry
058     * point to get a test to run or it will extract the suite automatically.
059     * <pre>
060     * public static Test suite() {
061     *      suite.addTest(new MathTest("testAdd"));
062     *      suite.addTest(new MathTest("testDivideByZero"));
063     *      return suite;
064     *  }
065     * </pre>
066     * @see TestResult
067     * @see TestSuite
068     */
069    
070    public abstract class TestCase extends Assert implements Test {
071        /**
072         * the name of the test case
073         */
074        private String fName;
075    
076        /**
077         * No-arg constructor to enable serialization. This method
078         * is not intended to be used by mere mortals without calling setName().
079         */
080        public TestCase() {
081            fName= null;
082        }
083        /**
084         * Constructs a test case with the given name.
085         */
086        public TestCase(String name) {
087            fName= name;
088        }
089        /**
090         * Counts the number of test cases executed by run(TestResult result).
091         */
092        public int countTestCases() {
093            return 1;
094        }
095        /**
096         * Creates a default TestResult object
097         *
098         * @see TestResult
099         */
100        protected TestResult createResult() {
101            return new TestResult();
102        }
103        /**
104         * A convenience method to run this test, collecting the results with a
105         * default TestResult object.
106         *
107         * @see TestResult
108         */
109        public TestResult run() {
110            TestResult result= createResult();
111            run(result);
112            return result;
113        }
114        /**
115         * Runs the test case and collects the results in TestResult.
116         */
117        public void run(TestResult result) {
118            result.run(this);
119        }
120        /**
121         * Runs the bare test sequence.
122         * @exception Throwable if any exception is thrown
123         */
124        public void runBare() throws Throwable {
125            setUp();
126            try {
127                runTest();
128            }
129            finally {
130                tearDown();
131            }
132        }
133        /**
134         * Override to run the test and assert its state.
135         * @exception Throwable if any exception is thrown
136         */
137        protected void runTest() throws Throwable {
138            assertNotNull(fName);
139            Method runMethod= null;
140            try {
141                // use getMethod to get all public inherited
142                // methods. getDeclaredMethods returns all
143                // methods of this class but excludes the
144                // inherited ones.
145                runMethod= getClass().getMethod(fName, (Class[]) null);
146            } catch (NoSuchMethodException e) {
147                fail("Method \""+fName+"\" not found");
148            }
149            if (!Modifier.isPublic(runMethod.getModifiers())) {
150                fail("Method \""+fName+"\" should be public");
151            }
152    
153            try {
154                runMethod.invoke(this, (Object[]) null);
155            }
156            catch (InvocationTargetException e) {
157                e.fillInStackTrace();
158                throw e.getTargetException();
159            }
160            catch (IllegalAccessException e) {
161                e.fillInStackTrace();
162                throw e;
163            }
164        }
165        /**
166         * Sets up the fixture, for example, open a network connection.
167         * This method is called before a test is executed.
168         */
169        protected void setUp() throws Exception {
170        }
171        /**
172         * Tears down the fixture, for example, close a network connection.
173         * This method is called after a test is executed.
174         */
175        protected void tearDown() throws Exception {
176        }
177        /**
178         * Returns a string representation of the test case
179         */
180        public String toString() {
181            return getName() + "(" + getClass().getName() + ")";
182        }
183        /**
184         * Gets the name of a TestCase
185         * @return returns a String
186         */
187        public String getName() {
188            return fName;
189        }
190        /**
191         * Sets the name of a TestCase
192         * @param name The name to set
193         */
194        public void setName(String name) {
195            fName= name;
196        }
197    }
